16from pytket.circuit import Circuit # type: ignore 

17from pytket.passes import ContextSimp # type: ignore 

18from pytket.transform import separate_classical # type: ignore

21def prepare_circuit( 

22 circ: Circuit, allow_classical: bool = True, xcirc: Optional[Circuit] = None 

23) -> Tuple[Circuit, Circuit]: 

24 """ 

25 Prepare a circuit for processing by a backend device. 

26 

27 This method first makes all inputs into Create operations (assuming an initial all- 

28 zero state) and all outputs into Discard operations (so that the circuit can no 

29 longer be usefully extended or appended to another circuit). It then attempts to 

30 apply various simplifications that take advantage of the known initial state and the 

31 fact that any unmeasured state is discarded. Finally, it separates the circuit into 

32 two circuits, the first of which is to be run on the backend (after any further 

33 compilation has been applied), and the second of which is a pure-classical circuit 

34 (on the same bits) which encodes classical post-processing of the measurement 

35 results. This post-processing is applied automatically when you pass the classical 

36 circuit as the `ppcirc` argument to `BackendResult.get_counts()` or 

37 `BackendResult.get_shots()`. 

38 

39 The original circuit is not modified by this method. 

40 

41 :param circ: input circuit 

42 :param allow_classical: allow insertion of mid-circuit classical operations? 

43 :param xcirc: 1-qubit circuit implementing an X gate in the transformed circuit (if 

44 omitted, an X gate is used) 

45 :return: (c0, ppcirc) where c0 is the simplified circuit and ppcirc should be passed 

46 to `BackendResult.get_counts()` or `BackendResult.get_shots()` when retrieving 

47 the final results. 

48 """ 

49 c = circ.copy() 

50 c.qubit_create_all() 

51 c.qubit_discard_all() 

52 ContextSimp(allow_classical, xcirc).apply(c) 

53 return separate_classical(c) # type: ignore
