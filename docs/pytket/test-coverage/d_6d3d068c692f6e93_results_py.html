25class BitPermuter: 

26 """Class for permuting the bits in an integer 

27 

28 Enables inverse permuation and uses caching to speed up common uses. 

29 

30 """ 

31 

32 def __init__(self, permutation: Tuple[int, ...]): 

33 """Constructor 

34 

35 :param permutation: Map from current bit index (big-endian) to its new position, 

36 encoded as a list. 

37 :type permutation: Tuple[int, ...] 

38 :raises ValueError: Input permutation is not valid complete permutation 

39 of all bits 

40 """ 

41 if sorted(permutation) != list(range(len(permutation))): 

42 raise ValueError("Permutation is not a valid complete permutation.") 

43 self.perm = tuple(permutation) 

44 self.n_bits = len(self.perm) 

45 self.int_maps: Tuple[Dict[int, int], Dict[int, int]] = ({}, {}) 

46 

47 def permute(self, val: int, inverse: bool = False) -> int: 

48 """Return input with bit values permuted. 

49 

50 :param val: input integer 

51 :type val: int 

52 :param inverse: whether to use the inverse permutation, defaults to False 

53 :type inverse: bool, optional 

54 :return: permuted integer 

55 :rtype: int 

56 """ 

57 perm_map, other_map = self.int_maps[:: (-1) ** inverse] 

58 if val in perm_map: 58 ↛ 59line 58 didn't jump to line 59, because the condition on line 58 was never true

59 return perm_map[val] 

60 

61 res = 0 

62 for source_index, target_index in enumerate(self.perm): 

63 if inverse: 63 ↛ 64line 63 didn't jump to line 64, because the condition on line 63 was never true

64 target_index, source_index = source_index, target_index 

65 # if source bit set 

66 if val & (1 << (self.n_bits - 1 - source_index)): 

67 # set target bit 

68 res |= 1 << (self.n_bits - 1 - target_index) 

69 

70 perm_map[val] = res 

71 other_map[res] = val 

72 return res 

73 

74 def permute_all(self) -> List[int]: 

75 """Permute all integers within bit-width specified by permutation. 

76 

77 :return: List of permuted outputs. 

78 :rtype: List 

79 """ 

80 return list(map(self.permute, range(1 << self.n_bits)))

83def counts_from_shot_table(shot_table: np.ndarray) -> Dict[Tuple[int, ...], int]: 

84 """Summarises a shot table into a dictionary of counts for each observed outcome. 

85 

86 :param shot_table: Table of shots from a pytket backend. 

87 :type shot_table: np.ndarray 

88 :return: Dictionary mapping observed readouts to the number of times observed. 

89 :rtype: Dict[Tuple[int, ...], int] 

90 """ 

91 shot_values, counts = np.unique(shot_table, axis=0, return_counts=True) 

92 return {tuple(s): c for s, c in zip(shot_values, counts)}

95def probs_from_counts( 

96 counts: Dict[Tuple[int, ...], int] 

97) -> Dict[Tuple[int, ...], float]: 

98 """Converts raw counts of observed outcomes into the observed probability 

99 distribution. 

100 

101 :param counts: Dictionary mapping observed readouts to the number of times observed. 

102 :type counts: Dict[Tuple[int, ...], int] 

103 :return: Probability distribution over observed readouts. 

104 :rtype: Dict[Tuple[int, ...], float] 

105 """ 

106 total = sum(counts.values()) 

107 return {outcome: c / total for outcome, c in counts.items()}

110def _index_to_readout( 

111 index: int, width: int, basis: BasisOrder = BasisOrder.ilo 

112) -> Tuple[int, ...]: 

113 return tuple( 

114 (index >> i) & 1 for i in range(width)[:: (-1) ** (basis == BasisOrder.ilo)] 

115 )

118def _reverse_bits_of_index(index: int, width: int) -> int: 

119 """Reverse bits of a readout/statevector index to change :py:class:`BasisOrder`. 

120 Values in tket are ILO-BE (2 means [bit0, bit1] == [1, 0]). 

121 Values in qiskit are DLO-BE (2 means [bit1, bit0] == [1, 0]). 

122 Note: Since ILO-BE (DLO-BE) is indistinguishable from DLO-LE (ILO-LE), this can also 

123 be seen as changing the endianness of the value. 

124 

125 :param n: Value to reverse 

126 :type n: int 

127 :param width: Number of bits in bitstring 

128 :type width: int 

129 :return: Integer value of reverse bitstring 

130 :rtype: int 

131 """ 

132 permuter = BitPermuter(tuple(range(width - 1, -1, -1))) 

133 return permuter.permute(index)

136def _compute_probs_from_state(state: np.ndarray, min_p: float = 1e-10) -> np.ndarray: 

137 """ 

138 Converts statevector to a probability vector. 

139 Set probabilities lower than `min_p` to 0. 

140 

141 :param state: A statevector. 

142 :type state: np.ndarray 

143 :param min_p: Minimum probability to include in result 

144 :type min_p: float 

145 :return: Probability vector. 

146 :rtype: np.ndarray 

147 """ 

148 probs = state.real**2 + state.imag**2 

149 probs /= sum(probs) 

150 ignore = probs < min_p 

151 probs[ignore] = 0 

152 probs /= sum(probs) 

153 return probs

156def probs_from_state( 

157 state: np.ndarray, min_p: float = 1e-10 

158) -> Dict[Tuple[int, ...], float]: 

159 """ 

160 Converts statevector to the probability distribution over readouts in the 

161 computational basis. Ignores probabilities lower than `min_p`. 

162 

163 :param state: Full statevector with big-endian encoding. 

164 :type state: np.ndarray 

165 :param min_p: Minimum probability to include in result 

166 :type min_p: float 

167 :return: Probability distribution over readouts. 

168 :rtype: Dict[Tuple[int], float] 

169 """ 

170 width = get_n_qb_from_statevector(state) 

171 probs = _compute_probs_from_state(state, min_p) 

172 return {_index_to_readout(i, width): p for i, p in enumerate(probs) if p != 0}

191def get_n_qb_from_statevector(state: np.ndarray) -> int: 

192 """Given a statevector, returns the number of qubits described 

193 

194 :param state: Statevector to inspect 

195 :type state: np.ndarray 

196 :raises ValueError: If the dimension of the statevector is not a power of 2 

197 :return: `n` such that `len(state) == 2 ** n` 

198 :rtype: int 

199 """ 

200 n_qb = int(np.log2(state.shape[0])) 

201 if 2**n_qb != state.shape[0]: 

202 raise ValueError("Size is not a power of 2") 

203 return n_qb

206def _assert_compatible_state_permutation( 

207 state: np.ndarray, permutation: Tuple[int, ...] 

208) -> None: 

209 """Asserts that a statevector and a permutation list both refer to the same number 

210 of qubits 

211 

212 :param state: Statevector 

213 :type state: np.ndarray 

214 :param permutation: Permutation of qubit indices, encoded as a list. 

215 :type permutation: Tuple[int, ...] 

216 :raises ValueError: [description] 

217 """ 

218 n_qb = len(permutation) 

219 if 2**n_qb != state.shape[0]: 

220 raise ValueError("Invalid permutation: length does not match number of qubits")

223def permute_qubits_in_statevector( 

224 state: np.ndarray, permutation: Tuple[int, ...] 

225) -> np.ndarray: 

226 """Rearranges a statevector according to a permutation of the qubit indices. 

227 

228 >>> # A 3-qubit state: 

229 >>> state = np.array([0.0, 0.0625, 0.1875, 0.25, 0.375, 0.4375, 0.5, 0.5625]) 

230 >>> permutation = [1, 0, 2] # swap qubits 0 and 1 

231 >>> # Apply the permutation that swaps indices 2 (="010") and 4 (="100"), and swaps 

232 >>> # indices 3 (="011") and 5 (="101"): 

233 >>> permute_qubits_in_statevector(state, permutation) 

234 array([0. , 0.0625, 0.375 , 0.4375, 0.1875, 0.25 , 0.5 , 0.5625]) 

235 

236 :param state: Original statevector. 

237 :type state: np.ndarray 

238 :param permutation: Map from current qubit index (big-endian) to its new position, 

239 encoded as a list. 

240 :type permutation: Tuple[int, ...] 

241 :return: Updated statevector. 

242 :rtype: np.ndarray 

243 """ 

244 _assert_compatible_state_permutation(state, permutation) 

245 permuter = BitPermuter(permutation) 

246 return state[permuter.permute_all()]

249def permute_basis_indexing( 

250 matrix: np.ndarray, permutation: Tuple[int, ...] 

251) -> np.ndarray: 

252 """Rearranges the first dimensions of an array (statevector or unitary) 

253 according to a permutation of the bit indices in the binary representation 

254 of row indices. 

255 

256 :param matrix: Original unitary matrix 

257 :type matrix: np.ndarray 

258 :param permutation: Map from current qubit index (big-endian) to its new position, 

259 encoded as a list 

260 :type permutation: Tuple[int, ...] 

261 :return: Updated unitary matrix 

262 :rtype: np.ndarray 

263 """ 

264 _assert_compatible_state_permutation(matrix, permutation) 

265 permuter = BitPermuter(permutation) 

266 

267 result: np.ndarray = matrix[permuter.permute_all(), ...] 

268 return result

271def permute_rows_cols_in_unitary( 

272 matrix: np.ndarray, permutation: Tuple[int, ...] 

273) -> np.ndarray: 

274 """Rearranges the rows of a unitary matrix according to a permutation of the qubit 

275 indices. 

276 

277 :param matrix: Original unitary matrix 

278 :type matrix: np.ndarray 

279 :param permutation: Map from current qubit index (big-endian) to its new position, 

280 encoded as a list 

281 :type permutation: Tuple[int, ...] 

282 :return: Updated unitary matrix 

283 :rtype: np.ndarray 

284 """ 

285 _assert_compatible_state_permutation(matrix, permutation) 

286 permuter = BitPermuter(permutation) 

287 all_perms = permuter.permute_all() 

288 permat: np.ndarray = matrix[:, all_perms] 

289 permat = permat[all_perms, :] 

290 return permat

293def compare_statevectors(first: np.ndarray, second: np.ndarray) -> bool: 

294 """Check approximate equality up to global phase for statevectors. 

295 

296 :param first: First statevector. 

297 :type first: np.ndarray 

298 :param second: Second statevector. 

299 :type second: np.ndarray 

300 :return: Approximate equality. 

301 :rtype: bool 

302 """ 

303 return bool(np.isclose(np.abs(np.vdot(first, second)), 1))

306def compare_unitaries(first: np.ndarray, second: np.ndarray) -> bool: 

307 """Check approximate equality up to global phase for unitaries. 

308 

309 :param first: First unitary. 

310 :type first: np.ndarray 

311 :param second: Second unitary. 

312 :type second: np.ndarray 

313 :return: Approximate equality. 

314 :rtype: bool 

315 """ 

316 conjug_prod = first @ second.conjugate().transpose() 

317 identity = np.identity(conjug_prod.shape[0], dtype=complex) 

318 return bool(np.allclose(conjug_prod, identity * conjug_prod[0, 0]))
