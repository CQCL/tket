19std::pair<bool, unit_map_t> AASLabellingMethod::routing_method(
20 std::shared_ptr<MappingFrontier>& mapping_frontier,
21 const ArchitecturePtr& architecture) const {
22 bool found_unplaced_qubit = false;
23 bool found_unplaced_ppb = false;
24
25 // search for unplaced qubitto speed up the runtime
26 for (Qubit q : mapping_frontier->circuit_.all_qubits()) {
27 if (!architecture->node_exists(Node(q))) {
28 found_unplaced_qubit = true;
29 break;
30 }
31 }
32 if (found_unplaced_qubit) {
33 std::shared_ptr<unit_frontier_t> next_frontier =
34 frontier_convert_vertport_to_edge(
35 mapping_frontier->circuit_, mapping_frontier->linear_boundary);
36
37 CutFrontier next_cut = mapping_frontier->circuit_.next_cut(
38 next_frontier, std::make_shared<b_frontier_t>());
39
40 for (const Vertex& v : *next_cut.slice) {
41 if (mapping_frontier->circuit_.get_OpType_from_Vertex(v) ==
42 OpType::PhasePolyBox) {
43 TKET_ASSERT(mapping_frontier->circuit_.is_quantum_node(v));
44 Op_ptr op_ptr_ppb =
45 mapping_frontier->circuit_.get_Op_ptr_from_Vertex(v);
46
47 for (const Edge& e : mapping_frontier->circuit_.get_in_edges_of_type(
48 v, EdgeType::Quantum)) {
49 for (const std::pair<UnitID, Edge>& pair :
50 next_frontier->get<TagKey>()) {
51 if (pair.second == e) {
52 if (!architecture->node_exists(Node(pair.first))) {
53 found_unplaced_ppb = true;
54 }
55 }
56 }
57 }
58 }
59 }
60 }
61
62 if (!found_unplaced_ppb) {
63 return {false, {}};
64 } else {
65 qubit_vector_t q_vec = mapping_frontier->circuit_.all_qubits();
66 unit_map_t qubit_to_nodes_place;
67 node_set_t node_set_placed;
68
69 for (Qubit q : q_vec) {
70 if (architecture->node_exists(Node(q))) {
71 qubit_to_nodes_place.insert({q, Node(q)});
72 node_set_placed.insert(Node(q));
73 }
74 }
75
76 node_vector_t nodes_vec = architecture->get_all_nodes_vec();
77
78 // place all unplaced qubits
79
80 for (Qubit q : q_vec) {
81 if (!architecture->node_exists(Node(q))) {
82 // found unplaced qubit
83 // other checks could be added here to avoid placing unused qubits or
84 // qubits that are not in an ppb
85
86 unsigned index_to_use = 0;
87 while (node_set_placed.find(nodes_vec[index_to_use]) !=
88 node_set_placed.end()) {
89 ++index_to_use;
90 }
91 qubit_to_nodes_place.insert({q, nodes_vec[index_to_use]});
92 node_set_placed.insert(nodes_vec[index_to_use]);
93 mapping_frontier->update_bimaps(
94 mapping_frontier->get_qubit_from_circuit_uid(q),
95 nodes_vec[index_to_use]);
96 }
97 }
98
99 mapping_frontier->update_linear_boundary_uids(qubit_to_nodes_place);
100 mapping_frontier->circuit_.rename_units(qubit_to_nodes_place);
101
102 return {true, {}};
103 }
104}
