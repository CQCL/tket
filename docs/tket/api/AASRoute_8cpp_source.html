26std::pair<bool, unit_map_t> AASRouteRoutingMethod::routing_method(
27 std::shared_ptr<MappingFrontier>& mapping_frontier,
28 const ArchitecturePtr& architecture) const {
29 std::shared_ptr<unit_frontier_t> next_frontier =
30 frontier_convert_vertport_to_edge(
31 mapping_frontier->circuit_, mapping_frontier->linear_boundary);
32
33 CutFrontier next_cut = mapping_frontier->circuit_.next_cut(
34 next_frontier, std::make_shared<b_frontier_t>());
35
36 // search for ppb in slice
37 for (const Vertex& v : *next_cut.slice) {
38 if (mapping_frontier->circuit_.get_OpType_from_Vertex(v) ==
39 OpType::PhasePolyBox) {
40 TKET_ASSERT(mapping_frontier->circuit_.is_quantum_node(v));
41
42 unsigned number_of_qubits = mapping_frontier->circuit_.n_in_edges(v);
43 unit_vector_t qubit_vec(number_of_qubits);
44
45 // check all qubits of the ppb if they are placed
46 bool box_placed = true;
47 for (unsigned i = 0; i < number_of_qubits; ++i) {
48 const Edge& e = mapping_frontier->circuit_.get_nth_in_edge(v, i);
49 for (const std::pair<UnitID, Edge>& pair :
50 next_frontier->get<TagKey>()) {
51 if (pair.second == e) {
52 qubit_vec[i] = Qubit(pair.first);
53 if (!architecture->node_exists(Node(pair.first))) {
54 box_placed = false;
55 }
56 }
57 }
58 }
59
60 // check that the box we are working on is really placed and the check
61 // method has been executed
62 // this is important if the circuit contains more than one ppb and only
63 // one of them is placed
64
65 if (box_placed) {
66 // get ppb from op
67 Op_ptr op_ptr_ppb =
68 mapping_frontier->circuit_.get_Op_ptr_from_Vertex(v);
69 const PhasePolyBox& ppb = static_cast<const PhasePolyBox&>(*op_ptr_ppb);
70
71 // Circuit circuit_ppb_place(*ppb.to_circuit());
72
73 // generate aritecture to make sure that the result can be inserted into
74 // the given circuit by using flatten_registers
75 auto nodes_vec = architecture->get_all_nodes_vec();
76 auto edges_vec = architecture->get_all_edges_vec();
77
78 // create maps from qubits/node to int
79 std::map<UnitID, Node> orig_node_to_int_node;
80 std::map<UnitID, Node> orig_qubit_to_int_node;
81
82 unsigned id_node = 0;
83
84 unsigned n_nodes = architecture->n_nodes();
85
86 for (Node orig_node : nodes_vec) {
87 orig_node_to_int_node.insert({orig_node, Node(n_nodes)});
88 }
89
90 for (auto orig_qubit : qubit_vec) {
91 orig_node_to_int_node[orig_qubit] = Node(id_node);
92 ++id_node;
93 }
94
95 for (Node orig_node : nodes_vec) {
96 if (orig_node_to_int_node[orig_node] == Node(n_nodes)) {
97 orig_node_to_int_node[orig_node] = Node(id_node);
98 ++id_node;
99 }
100 }
101
102 // define new architecture with int nodes for ppb
103 std::vector<Architecture::Connection> new_con;
104 for (auto pair : architecture->get_all_edges_vec()) {
105 new_con.push_back(
106 {orig_node_to_int_node[UnitID(pair.first)],
107 orig_node_to_int_node[UnitID(pair.second)]});
108 }
109
110 Architecture new_int_arch = Architecture(new_con);
111
112 TKET_ASSERT(architecture->n_nodes() == new_int_arch.n_nodes());
113
114 Circuit result = aas::phase_poly_synthesis(
115 new_int_arch, ppb, aaslookahead_, cnotsynthtype_);
116
117 // make sure the circuit can be inserted
118 result.flatten_registers();
119
120 // substitute the ppb vertex in the initial circuit with the routed
121 // result
122 mapping_frontier->circuit_.substitute(result, v);
123 return {true, {}};
124 }
125 }
126 }
127 return {false, {}};
128}
