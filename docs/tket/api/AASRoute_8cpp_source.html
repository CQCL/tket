23 std::pair<bool, unit_map_t> AASRouteRoutingMethod::routing_method(
24  std::shared_ptr<MappingFrontier>& mapping_frontier,
25  const ArchitecturePtr& architecture) const {
26  std::shared_ptr<unit_frontier_t> next_frontier =
27  frontier_convert_vertport_to_edge(
28  mapping_frontier->circuit_, mapping_frontier->linear_boundary);
29 
30  CutFrontier next_cut = mapping_frontier->circuit_.next_cut(
31  next_frontier, std::make_shared<b_frontier_t>());
32 
33  // search for ppb in slice
34  for (const Vertex& v : *next_cut.slice) {
35  if (mapping_frontier->circuit_.get_OpType_from_Vertex(v) ==
36  OpType::PhasePolyBox) {
37  TKET_ASSERT(mapping_frontier->circuit_.is_quantum_node(v));
38 
39  unsigned number_of_qubits = mapping_frontier->circuit_.n_in_edges(v);
40  unit_vector_t qubit_vec(number_of_qubits);
41 
42  // check all qubits of the ppb if they are placed
43  bool box_placed = true;
44  for (unsigned i = 0; i < number_of_qubits; ++i) {
45  const Edge& e = mapping_frontier->circuit_.get_nth_in_edge(v, i);
46  for (const std::pair<UnitID, Edge>& pair :
47  next_frontier->get<TagKey>()) {
48  if (pair.second == e) {
49  qubit_vec[i] = Qubit(pair.first);
50  if (!architecture->node_exists(Node(pair.first))) {
51  box_placed = false;
52  }
53  }
54  }
55  }
56 
57  // check that the box we are working on is really placed and the check
58  // method has been executed
59  // this is imporant if the circuit contains more than one ppb and only one
60  // of them is placed
61 
62  if (box_placed) {
63  // get ppb from op
64  Op_ptr op_ptr_ppb =
65  mapping_frontier->circuit_.get_Op_ptr_from_Vertex(v);
66  const PhasePolyBox& ppb = static_cast<const PhasePolyBox&>(*op_ptr_ppb);
67 
68  // Circuit circuit_ppb_place(*ppb.to_circuit());
69 
70  // generate aritecture to make sure that the result can be inserted into
71  // the given circuit by using flatten_registers
72  auto nodes_vec = architecture->get_all_nodes_vec();
73  auto edges_vec = architecture->get_all_edges_vec();
74 
75  // create maps from qubits/node to int
76  std::map<UnitID, Node> orig_node_to_int_node;
77  std::map<UnitID, Node> orig_qubit_to_int_node;
78 
79  unsigned id_node = 0;
80 
81  unsigned n_nodes = architecture->n_nodes();
82 
83  for (Node orig_node : nodes_vec) {
84  orig_node_to_int_node.insert({orig_node, Node(n_nodes)});
85  }
86 
87  for (auto orig_qubit : qubit_vec) {
88  orig_node_to_int_node[orig_qubit] = Node(id_node);
89  ++id_node;
90  }
91 
92  for (Node orig_node : nodes_vec) {
93  if (orig_node_to_int_node[orig_node] == Node(n_nodes)) {
94  orig_node_to_int_node[orig_node] = Node(id_node);
95  ++id_node;
96  }
97  }
98 
99  // define new arcitecture with int nodes for ppb
100  std::vector<Architecture::Connection> new_con;
101  for (auto pair : architecture->get_all_edges_vec()) {
102  new_con.push_back(
103  {orig_node_to_int_node[UnitID(pair.first)],
104  orig_node_to_int_node[UnitID(pair.second)]});
105  }
106 
107  Architecture new_int_arch = Architecture(new_con);
108 
109  TKET_ASSERT(architecture->n_nodes() == new_int_arch.n_nodes());
110 
111  Circuit result = aas::phase_poly_synthesis(
112  new_int_arch, ppb, aaslookahead_, cnotsynthtype_);
113 
114  // make sure the circuit can be inserted
115  result.flatten_registers();
116 
117  // substitute the ppb vertex in the initial circuit with the routed
118  // result
119  mapping_frontier->circuit_.substitute(result, v);
120  return {true, {}};
121  }
122  }
123  }
124  return {false, {}};
125 }
