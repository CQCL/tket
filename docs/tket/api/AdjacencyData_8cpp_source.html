34 string AdjacencyData::to_string() const {
35  map<std::size_t, set<std::size_t>> data_to_display;
36  for (std::size_t i = 0; i < m_cleaned_data.size(); ++i) {
37  const auto& neighbours = m_cleaned_data[i];
38  auto& neighbours_to_display = data_to_display[i];
39  for (std::size_t v : neighbours) {
40  if (i <= v) {
41  neighbours_to_display.insert(v);
42  }
43  }
44  if (neighbours_to_display.empty()) {
45  // Don't display it after all
46  data_to_display.erase(i);
47  }
48  }
49 
50  stringstream ss;
51  ss << "\nThere are " << m_cleaned_data.size()
52  << " vertices in total.\nVertex neighbours:\n{";
53 
54  for (const auto& entry_to_display : data_to_display) {
55  ss << "\n { " << entry_to_display.first << ", { ";
56  for (auto v : entry_to_display.second) {
57  ss << v << ", ";
58  }
59  ss << "} },";
60  }
61  ss << "\n}\n";
62  return ss.str();
63 }

82 std::size_t AdjacencyData::get_number_of_edges() const {
83  // Each edge i-j is counted twice, i->j and j->i, except for loops.
84  std::size_t edges = 0;
85  std::size_t loops = 0;
86 
87  for (std::size_t ii = 0; ii < m_cleaned_data.size(); ++ii) {
88  edges += m_cleaned_data[ii].size();
89  loops += m_cleaned_data[ii].count(ii);
90  }
91  return loops + (edges - loops) / 2;
92 }

126 AdjacencyData::AdjacencyData(
127  const map<std::size_t, vector<std::size_t>>& raw_data,
128  std::size_t number_of_vertices) {
129  for (const auto& entry : raw_data) {
130  number_of_vertices = std::max(number_of_vertices, entry.first + 1);
131  for (std::size_t neighbour : entry.second) {
132  number_of_vertices = std::max(number_of_vertices, neighbour + 1);
133  }
134  }
135  m_cleaned_data.resize(number_of_vertices);
136  for (const auto& entry : raw_data) {
137  for (std::size_t neighbour : entry.second) {
138  add_edge(entry.first, neighbour);
139  }
140  }
141 }

143 AdjacencyData::AdjacencyData(
144  const vector<vector<std::size_t>>& raw_data, bool allow_loops) {
145  m_cleaned_data.resize(raw_data.size());
146 
147  for (std::size_t i = 0; i < m_cleaned_data.size(); ++i) {
148  for (std::size_t j : raw_data[i]) {
149  // GCOVR_EXCL_START
150  TKET_ASSERT(
151  i != j || allow_loops ||
152  AssertMessage() << "Vertex " << i << " out of "
153  << m_cleaned_data.size() << " has a loop.");
154  TKET_ASSERT(
155  j < m_cleaned_data.size() ||
156  AssertMessage() << "Vertex " << i << " has illegal neighbour vertex "
157  << j << ", the size is " << m_cleaned_data.size());
158  // GCOVR_EXCL_STOP
159  m_cleaned_data[i].insert(j);
160  m_cleaned_data[j].insert(i);
161  }
162  }
