21ArchitectureMapping::ArchitectureMapping(const Architecture& arch)
22 : m_arch(arch) {
23 const auto uids = arch.nodes();
24 m_vertex_to_node_mapping.reserve(uids.size());
25 for (const UnitID& uid : uids) {
26 m_vertex_to_node_mapping.emplace_back(Node(uid));
27 }
28
29 for (std::size_t ii = 0; ii < m_vertex_to_node_mapping.size(); ++ii) {
30 const auto& node = m_vertex_to_node_mapping[ii];
31 {
32 const auto citer = m_node_to_vertex_mapping.find(node);
33 // GCOVR_EXCL_START
34 TKET_ASSERT(
35 citer == m_node_to_vertex_mapping.cend() ||
36 AssertMessage() << "Duplicate node " << node.repr() << " at vertices "
37 << citer->second << ", " << ii);
38 // GCOVR_EXCL_STOP
39 }
40 m_node_to_vertex_mapping[node] = ii;
41 }
42}

44ArchitectureMapping::ArchitectureMapping(
45 const Architecture& arch,
46 const std::vector<std::pair<unsigned, unsigned>>& edges)
47 : m_arch(arch) {
48 auto& node_to_vertex_mapping = m_node_to_vertex_mapping;
49 auto& vertex_to_node_mapping = m_vertex_to_node_mapping;
50
51 const auto add_node = [&node_to_vertex_mapping,
52 &vertex_to_node_mapping](unsigned nn) {
53 const Node node(nn);
54 if (node_to_vertex_mapping.count(node) == 0) {
55 node_to_vertex_mapping[node] = vertex_to_node_mapping.size();
56 vertex_to_node_mapping.push_back(node);
57 }
58 };
59
60 // The nodes are labelled 0,1,2,... in order of appearance.
61 // Nothing special about this ordering, just for backwards compatibility.
62 for (const auto& entry : edges) {
63 add_node(entry.first);
64 add_node(entry.second);
65 }
66
67 // Check that the nodes agree with the architecture object.
68 const auto uids = arch.nodes();
69 // GCOVR_EXCL_START
70 TKET_ASSERT(
71 uids.size() == m_vertex_to_node_mapping.size() ||
72 AssertMessage() << "passed in " << edges.size() << " edges, giving "
73 << m_vertex_to_node_mapping.size()
74 << " vertices; but the architecture object has "
75 << uids.size() << " vertices");
76 // GCOVR_EXCL_STOP
77
78 for (const UnitID& uid : uids) {
79 const Node node(uid);
80 // GCOVR_EXCL_START
81 TKET_ASSERT(
82 m_node_to_vertex_mapping.count(node) != 0 ||
83 AssertMessage()
84 << "passed in " << edges.size() << " edges, giving "
85 << m_vertex_to_node_mapping.size()
86 << " vertices; but the architecture object has an unknown node "
87 << node.repr());
88 // GCOVR_EXCL_STOP
89 }
90}

96const Node& ArchitectureMapping::get_node(std::size_t vertex) const {
97 const auto num_vertices = number_of_vertices();
98 // GCOVR_EXCL_START
99 TKET_ASSERT(
100 vertex < num_vertices || AssertMessage() << "invalid vertex " << vertex
101 << " (architecture only has "
102 << num_vertices << " vertices)");
103 // GCOVR_EXCL_STOP
104
105 return m_vertex_to_node_mapping[vertex];
106}

108std::size_t ArchitectureMapping::get_vertex(const Node& node) const {
109 const auto citer = m_node_to_vertex_mapping.find(node);
110 // GCOVR_EXCL_START
111 TKET_ASSERT(
112 citer != m_node_to_vertex_mapping.cend() ||
113 AssertMessage() << "node " << node.repr() << " has no vertex number");
114 // GCOVR_EXCL_STOP
115 return citer->second;
116}

122std::vector<Swap> ArchitectureMapping::get_edges() const {
123 std::vector<Swap> edges;
124 for (auto [node1, node2] : m_arch.get_all_edges_vec()) {
125 edges.emplace_back(get_swap(get_vertex(node1), get_vertex(node2)));
