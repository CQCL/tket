23 ArchitectureMapping::ArchitectureMapping(const Architecture& arch)
24  : m_arch(arch) {
25  const auto uids = arch.nodes();
26  m_vertex_to_node_mapping.reserve(uids.size());
27  for (const UnitID& uid : uids) {
28  m_vertex_to_node_mapping.emplace_back(Node(uid));
29  }
30 
31  for (std::size_t ii = 0; ii < m_vertex_to_node_mapping.size(); ++ii) {
32  const auto& node = m_vertex_to_node_mapping[ii];
33  {
34  const auto citer = m_node_to_vertex_mapping.find(node);
35  // GCOVR_EXCL_START
36  TKET_ASSERT(
37  citer == m_node_to_vertex_mapping.cend() ||
38  AssertMessage() << "Duplicate node " << node.repr() << " at vertices "
39  << citer->second << ", " << ii);
40  // GCOVR_EXCL_STOP
41  }
42  m_node_to_vertex_mapping[node] = ii;
43  }
44 }

46 ArchitectureMapping::ArchitectureMapping(
47  const Architecture& arch,
48  const std::vector<std::pair<unsigned, unsigned>>& edges)
49  : m_arch(arch) {
50  auto& node_to_vertex_mapping = m_node_to_vertex_mapping;
51  auto& vertex_to_node_mapping = m_vertex_to_node_mapping;
52 
53  const auto add_node = [&node_to_vertex_mapping,
54  &vertex_to_node_mapping](unsigned nn) {
55  const Node node(nn);
56  if (node_to_vertex_mapping.count(node) == 0) {
57  node_to_vertex_mapping[node] = vertex_to_node_mapping.size();
58  vertex_to_node_mapping.push_back(node);
59  }
60  };
61 
62  // The nodes are labelled 0,1,2,... in order of appearance.
63  // Nothing special about this ordering, just for backwards compatibility.
64  for (const auto& entry : edges) {
65  add_node(entry.first);
66  add_node(entry.second);
67  }
68 
69  // Check that the nodes agree with the architecture object.
70  const auto uids = arch.nodes();
71  // GCOVR_EXCL_START
72  TKET_ASSERT(
73  uids.size() == m_vertex_to_node_mapping.size() ||
74  AssertMessage() << "passed in " << edges.size() << " edges, giving "
75  << m_vertex_to_node_mapping.size()
76  << " vertices; but the architecture object has "
77  << uids.size() << " vertices");
78  // GCOVR_EXCL_STOP
79 
80  for (const UnitID& uid : uids) {
81  const Node node(uid);
82  // GCOVR_EXCL_START
83  TKET_ASSERT(
84  m_node_to_vertex_mapping.count(node) != 0 ||
85  AssertMessage()
86  << "passed in " << edges.size() << " edges, giving "
87  << m_vertex_to_node_mapping.size()
88  << " vertices; but the architecture object has an unknown node "
89  << node.repr());
90  // GCOVR_EXCL_STOP
91  }
92 }
