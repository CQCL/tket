28 bool Architecture::valid_operation(const std::vector<Node>& uids) const {
29  for (Node n : uids) {
30  if (!this->node_exists(Node(n))) return false;
31  }
32  if (uids.size() == 1) {
33  return true;
34  } else if (uids.size() == 2) {
35  if (this->bidirectional_edge_exists(uids[0], uids[1])) {
36  return true;
37  }
38  }
39  return false;
40 }

53 // Given a vector of lengths of lines, returns a vector of lines of these sizes
54 // comprised of architecture nodes
55 std::vector<node_vector_t> Architecture::get_lines(
56  std::vector<unsigned> required_lengths) const {
57  // check total length doesn't exceed number of nodes
58  if (std::accumulate(required_lengths.begin(), required_lengths.end(), 0u) >
59  n_nodes()) {
60  throw ArchitectureInvalidity(
61  "Not enough nodes to satisfy required lengths.");
62  }
63  std::sort(
64  required_lengths.begin(), required_lengths.end(),
65  std::greater<unsigned>());
66 
67  UndirectedConnGraph curr_graph(get_undirected_connectivity());
68  std::vector<node_vector_t> found_lines;
69  for (unsigned length : required_lengths) {
70  std::vector<Vertex> longest(
71  graphs::longest_simple_path(curr_graph, length));
72  if (longest.size() >= length) {
73  longest.resize(length);
74  // convert Vertex to Node
75  node_vector_t to_add;
76  for (Vertex v : longest) {
77  to_add.push_back(curr_graph[v]);
78  }
79  found_lines.push_back(to_add);
80  for (Vertex v : longest) {
81  boost::clear_vertex(v, curr_graph);
82  }
83  }
84  }
85  return found_lines;
86 }

111 std::optional<Node> Architecture::find_worst_node(
112  const Architecture& original_arch) {
113  node_set_t ap = get_articulation_points();
114  node_set_t min_nodes = min_degree_nodes();
115 
116  std::set<Node> bad_nodes;
117  std::set_difference(
118  min_nodes.cbegin(), min_nodes.cend(), ap.cbegin(), ap.cend(),
119  std::inserter(bad_nodes, bad_nodes.begin()));
120 
121  if (bad_nodes.empty()) {
122  return std::nullopt;
123  }
124  auto make_lexicographic = [](std::vector<size_t> distances) {
125  unsigned max = distances.size();
126  std::vector<size_t> out(max + 1, size_t(0));
127  for (const auto& distance : distances) {
128  TKET_ASSERT(distance < max);
129  out[max - distance]++;
130  }
131  return out;
132  };
133  std::vector<size_t> worst_distances, temp_distances;
134  Node worst_node = *bad_nodes.begin();
135  worst_distances = make_lexicographic(get_distances(worst_node));
136  for (Node temp_node : bad_nodes) {
137  temp_distances = make_lexicographic(get_distances(temp_node));
138  int distance_comp =
139  tri_lexicographical_comparison(temp_distances, worst_distances);
140  if (distance_comp == 1) {
141  worst_node = temp_node;
142  worst_distances = temp_distances;
143  } else if (distance_comp == -1) {
144  std::vector<size_t> temp_distances_full =
145  make_lexicographic(original_arch.get_distances(temp_node));
146  std::vector<size_t> worst_distances_full =
147  make_lexicographic(original_arch.get_distances(worst_node));
148  if (lexicographical_comparison(
149  temp_distances_full, worst_distances_full)) {
150  worst_node = temp_node;
151  worst_distances = temp_distances;
152  }
153  }
154  }
155  return worst_node;
156 }

158 node_set_t Architecture::remove_worst_nodes(unsigned num) {
159  node_set_t out;
160  Architecture original_arch(*this);
161  for (unsigned k = 0; k < num; k++) {
162  std::optional<Node> v = find_worst_node(original_arch);
163  if (v.has_value()) {

165  out.insert(v.value());
166  }
167  }

171 int tri_lexicographical_comparison(
172  const std::vector<std::size_t>& dist1,
173  const std::vector<std::size_t>& dist2) {
174  // add assertion that these are the same size distance vectors
175  auto start_dist1 = dist1.cbegin();
176  auto start_dist2 = dist2.cbegin();
177 
178  while (start_dist1 != dist1.cend()) {
179  if (start_dist2 == dist2.cend() || *start_dist2 < *start_dist1) {
180  return 0;
181  } else if (*start_dist1 < *start_dist2) {
182  return 1;
183  }
184  ++start_dist1;
185  ++start_dist2;
186  }
187  return -1;
188 }

190 MatrixXb Architecture::get_connectivity() const {
191  unsigned n = n_nodes();
192  MatrixXb connectivity = MatrixXb(n, n);
193  for (unsigned i = 0; i != n; ++i) {
194  for (unsigned j = 0; j != n; ++j) {
195  connectivity(i, j) =
196  edge_exists(Node(i), Node(j)) || edge_exists(Node(j), Node(i));
197  }
198  }
199  return connectivity;
200 }

278 std::vector<Architecture::Connection> SquareGrid::get_edges(
279  const unsigned dim_r, const unsigned dim_c, const unsigned layers) {
280  std::vector<Connection> edges;
281  for (unsigned l = 0; l < layers; l++) {
282  for (unsigned ver = 0; ver < dim_r; ver++) {
283  for (unsigned hor = 0; hor < dim_c; hor++) {
284  Node n("gridNode", ver, hor, l);
285  if (hor != dim_c - 1) {
286  Node h_neighbour("gridNode", ver, hor + 1, l);
287  edges.push_back({n, h_neighbour});
288  }
289  if (ver != dim_r - 1) {
290  Node v_neighbour("gridNode", ver + 1, hor, l);
291  edges.push_back({n, v_neighbour});
292  }
293  if (l != layers - 1) {
294  Node l_neighbour("gridNode", ver, hor, l + 1);
295  edges.push_back({n, l_neighbour});
296  }
297  }
298  }
299  }
