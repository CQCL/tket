28 bool Architecture::valid_operation(const std::vector<Node>& uids) const {
29  for (Node n : uids) {
30  if (!this->node_exists(Node(n))) return false;
31  }
32  if (uids.size() == 1) {
33  return true;
34  } else if (uids.size() == 2) {
35  if (this->bidirectional_edge_exists(uids[0], uids[1])) {
36  return true;
37  }
38  }
39  return false;
40 }

53 // Given a vector of lengths of lines, returns a vector of lines of these sizes
54 // comprised of architecture nodes
55 std::vector<node_vector_t> Architecture::get_lines(
56  std::vector<unsigned> required_lengths) const {
57  // check total length doesn't exceed number of nodes
58  bool not_enough_nodes =
59  std::accumulate(required_lengths.begin(), required_lengths.end(), 0u) >
60  n_nodes();
61  TKET_ASSERT(!not_enough_nodes); // Not enough nodes for required lengths.
62 
63  std::sort(
64  required_lengths.begin(), required_lengths.end(),
65  std::greater<unsigned>());
66 
67  UndirectedConnGraph curr_graph(get_undirected_connectivity());
68  std::vector<node_vector_t> found_lines;
69  for (unsigned length : required_lengths) {
70  std::vector<Vertex> longest(
71  graphs::longest_simple_path(curr_graph, length));
72  if (longest.size() >= length) {
73  longest.resize(length);
74  // convert Vertex to Node
75  node_vector_t to_add;
76  for (Vertex v : longest) {
77  to_add.push_back(curr_graph[v]);
78  }
79  found_lines.push_back(to_add);
80  for (Vertex v : longest) {
81  boost::clear_vertex(v, curr_graph);
82  }
83  }
84  }
85  return found_lines;
86 }

112 std::optional<Node> Architecture::find_worst_node(
113  const Architecture& original_arch) {
114  node_set_t ap = get_articulation_points();
115  node_set_t min_nodes = min_degree_nodes();
116 
117  std::set<Node> bad_nodes;
118  std::set_difference(
119  min_nodes.cbegin(), min_nodes.cend(), ap.cbegin(), ap.cend(),
120  std::inserter(bad_nodes, bad_nodes.begin()));
121 
122  if (bad_nodes.empty()) {
123  return std::nullopt;
124  }
125  auto make_lexicographic = [](std::vector<std::size_t> distances) {
126  unsigned max = distances.size();
127  std::vector<std::size_t> out(max + 1, std::size_t(0));
128  for (const auto& distance : distances) {
129  TKET_ASSERT(distance < max);
130  out[max - distance]++;
131  }
132  return out;
133  };
134  std::vector<std::size_t> worst_distances, temp_distances;
135  Node worst_node = *bad_nodes.begin();
136  worst_distances = make_lexicographic(get_distances(worst_node));
137  for (Node temp_node : bad_nodes) {
138  temp_distances = make_lexicographic(get_distances(temp_node));
139  int distance_comp =
140  tri_lexicographical_comparison(temp_distances, worst_distances);
141  if (distance_comp == 1) {
142  worst_node = temp_node;
143  worst_distances = temp_distances;
144  } else if (distance_comp == -1) {
145  std::vector<std::size_t> temp_distances_full =
146  make_lexicographic(original_arch.get_distances(temp_node));
147  std::vector<std::size_t> worst_distances_full =
148  make_lexicographic(original_arch.get_distances(worst_node));
149  if (lexicographical_comparison(
150  temp_distances_full, worst_distances_full)) {
151  worst_node = temp_node;
152  worst_distances = temp_distances;
153  }
154  }
155  }
156  return worst_node;
157 }

159 node_set_t Architecture::remove_worst_nodes(unsigned num) {
160  node_set_t out;
161  Architecture original_arch(*this);
162  for (unsigned k = 0; k < num; k++) {
163  std::optional<Node> v = find_worst_node(original_arch);
164  if (v.has_value()) {

166  out.insert(v.value());
167  }
168  }

172 int tri_lexicographical_comparison(
173  const std::vector<std::size_t>& dist1,
174  const std::vector<std::size_t>& dist2) {
175  // add assertion that these are the same size distance vectors
176  auto start_dist1 = dist1.cbegin();
177  auto start_dist2 = dist2.cbegin();
178 
179  while (start_dist1 != dist1.cend()) {
180  if (start_dist2 == dist2.cend() || *start_dist2 < *start_dist1) {
181  return 0;
182  } else if (*start_dist1 < *start_dist2) {
183  return 1;
184  }
185  ++start_dist1;
186  ++start_dist2;
187  }
188  return -1;
189 }

191 MatrixXb Architecture::get_connectivity() const {
192  unsigned n = n_nodes();
193  MatrixXb connectivity = MatrixXb(n, n);
194  for (unsigned i = 0; i != n; ++i) {
195  for (unsigned j = 0; j != n; ++j) {
196  connectivity(i, j) =
197  edge_exists(Node(i), Node(j)) || edge_exists(Node(j), Node(i));
198  }
199  }
200  return connectivity;
201 }

280 std::vector<Architecture::Connection> SquareGrid::get_edges(
281  const unsigned dim_r, const unsigned dim_c, const unsigned layers,
282  const std::string& label) {
283  std::vector<Connection> edges;
284  for (unsigned l = 0; l < layers; l++) {
285  for (unsigned ver = 0; ver < dim_r; ver++) {
286  for (unsigned hor = 0; hor < dim_c; hor++) {
287  Node n(label, ver, hor, l);
288  if (hor != dim_c - 1) {
289  Node h_neighbour(label, ver, hor + 1, l);
290  edges.push_back({n, h_neighbour});
291  }
292  if (ver != dim_r - 1) {
293  Node v_neighbour(label, ver + 1, hor, l);
294  edges.push_back({n, v_neighbour});
295  }
296  if (l != layers - 1) {
297  Node l_neighbour(label, ver, hor, l + 1);
298  edges.push_back({n, l_neighbour});
299  }
300  }
301  }
302  }
303  return edges;
