26bool Architecture::valid_operation(
27 const std::vector<Node>& uids, bool bidirectional) const {
28 for (Node n : uids) {
29 if (!this->node_exists(Node(n))) return false;
30 }
31 if (uids.size() == 1) {
32 return true;
33 } else if (uids.size() == 2) {
34 if (bidirectional && this->bidirectional_edge_exists(uids[0], uids[1])) {
35 return true;
36 } else if (!bidirectional && this->edge_exists(uids[0], uids[1])) {
37 return false;
38 }
39 }
40 return false;
41}

54// Given a vector of lengths of lines, returns a vector of lines of these sizes
55// comprised of architecture nodes
56std::vector<node_vector_t> Architecture::get_lines(
57 std::vector<unsigned> required_lengths) const {
58 // check total length doesn't exceed number of nodes
59 bool not_enough_nodes =
60 std::accumulate(required_lengths.begin(), required_lengths.end(), 0u) >
61 n_nodes();
62 TKET_ASSERT(!not_enough_nodes); // Not enough nodes for required lengths.
63
64 std::sort(
65 required_lengths.begin(), required_lengths.end(),
66 std::greater<unsigned>());
67
68 UndirectedConnGraph curr_graph(get_undirected_connectivity());
69 std::vector<node_vector_t> found_lines;
70 for (unsigned length : required_lengths) {
71 std::vector<Vertex> longest(
72 graphs::longest_simple_path(curr_graph, length));
73 if (longest.size() >= length) {
74 longest.resize(length);
75 // convert Vertex to Node
76 node_vector_t to_add;
77 for (Vertex v : longest) {
78 to_add.push_back(curr_graph[v]);
79 }
80 found_lines.push_back(to_add);
81 for (Vertex v : longest) {
82 boost::clear_vertex(v, curr_graph);
83 }
84 }
85 }
86 return found_lines;
87}

106static bool lexicographical_comparison(
107 const std::vector<std::size_t>& dist1,
108 const std::vector<std::size_t>& dist2) {

113std::optional<Node> Architecture::find_worst_node(
114 const Architecture& original_arch) {
115 node_set_t ap = get_articulation_points();
116 node_set_t min_nodes = min_degree_nodes();
117
118 std::set<Node> bad_nodes;
119 std::set_difference(
120 min_nodes.cbegin(), min_nodes.cend(), ap.cbegin(), ap.cend(),
121 std::inserter(bad_nodes, bad_nodes.begin()));
122
123 if (bad_nodes.empty()) {
124 return std::nullopt;
125 }
126 auto make_lexicographic = [](std::vector<std::size_t> distances) {
127 unsigned max = distances.size();
128 std::vector<std::size_t> out(max + 1, std::size_t(0));
129 for (const auto& distance : distances) {
130 TKET_ASSERT(distance < max);
131 out[max - distance]++;
132 }
133 return out;
134 };
135 std::vector<std::size_t> worst_distances, temp_distances;
136 Node worst_node = *bad_nodes.begin();
137 worst_distances = make_lexicographic(get_distances(worst_node));
138 for (Node temp_node : bad_nodes) {
139 temp_distances = make_lexicographic(get_distances(temp_node));
140 int distance_comp =
141 tri_lexicographical_comparison(temp_distances, worst_distances);
142 if (distance_comp == 1) {
143 worst_node = temp_node;
144 worst_distances = temp_distances;
145 } else if (distance_comp == -1) {
146 std::vector<std::size_t> temp_distances_full =
147 make_lexicographic(original_arch.get_distances(temp_node));
148 std::vector<std::size_t> worst_distances_full =
149 make_lexicographic(original_arch.get_distances(worst_node));
150 if (lexicographical_comparison(
151 temp_distances_full, worst_distances_full)) {
152 worst_node = temp_node;
153 worst_distances = temp_distances;
154 }
155 }
156 }
157 return worst_node;
158}

160node_set_t Architecture::remove_worst_nodes(unsigned num) {
161 node_set_t out;
162 Architecture original_arch(*this);
163 for (unsigned k = 0; k < num; k++) {
164 std::optional<Node> v = find_worst_node(original_arch);
165 if (v.has_value()) {
166 remove_node(v.value());
167 out.insert(v.value());
168 }
169 }
170 return out;
171}

173int tri_lexicographical_comparison(
174 const std::vector<std::size_t>& dist1,
175 const std::vector<std::size_t>& dist2) {
176 // add assertion that these are the same size distance vectors
177 auto start_dist1 = dist1.cbegin();
178 auto start_dist2 = dist2.cbegin();
179
180 while (start_dist1 != dist1.cend()) {
181 if (start_dist2 == dist2.cend() || *start_dist2 < *start_dist1) {
182 return 0;
183 } else if (*start_dist1 < *start_dist2) {
184 return 1;
185 }
186 ++start_dist1;
187 ++start_dist2;
188 }
189 return -1;
190}

192MatrixXb Architecture::get_connectivity() const {
193 unsigned n = n_nodes();
194 MatrixXb connectivity = MatrixXb(n, n);
195 for (unsigned i = 0; i != n; ++i) {
196 for (unsigned j = 0; j != n; ++j) {
197 connectivity(i, j) =
198 edge_exists(Node(i), Node(j)) || edge_exists(Node(j), Node(i));
199 }
200 }
201 return connectivity;
202}

214void to_json(nlohmann::json& j, const Architecture& ar) {
215 // Preserve the internal order of ids since Placement depends on this
216 Architecture::ConnGraph g = ar.get_directed_connectivity();
217 std::vector<Node> nodes;
218 for (auto v : boost::make_iterator_range(boost::vertices(g)))
219 nodes.push_back(g[v]);
220 j["nodes"] = nodes;
221 nlohmann::json links;
222 for (const Architecture::Connection& con : ar.get_all_edges_vec()) {
223 nlohmann::json entry;
224 entry["link"] = con;
225 entry["weight"] = ar.get_connection_weight(con.first, con.second);
226 links.push_back(entry);
227 }
228 j["links"] = links;
229}

281std::vector<Architecture::Connection> SquareGrid::get_edges(
282 const unsigned dim_r, const unsigned dim_c, const unsigned layers,
283 const std::string& label) {
284 std::vector<Connection> edges;
285 for (unsigned l = 0; l < layers; l++) {
286 for (unsigned ver = 0; ver < dim_r; ver++) {
287 for (unsigned hor = 0; hor < dim_c; hor++) {
288 Node n(label, ver, hor, l);
289 if (hor != dim_c - 1) {
290 Node h_neighbour(label, ver, hor + 1, l);
291 edges.push_back({n, h_neighbour});
292 }
293 if (ver != dim_r - 1) {
294 Node v_neighbour(label, ver + 1, hor, l);
295 edges.push_back({n, v_neighbour});
296 }
297 if (l != layers - 1) {
298 Node l_neighbour(label, ver, hor, l + 1);
299 edges.push_back({n, l_neighbour});
300 }
301 }
302 }
303 }
304 return edges;
305}
