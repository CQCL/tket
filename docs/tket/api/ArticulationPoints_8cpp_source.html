32// Maps BGL edges descriptors to their (unique) biconnected component
33template <typename T>
34using EdgeToCompT = std::map<utils::edge<UndirectedConnGraph<T>>, unsigned>;

36template <typename T>
37BicomponentGraph<T>::BicomponentGraph(const UndirectedConnGraph<T>& graph)
38 : underlying_g(graph) {
39 // get map from APs to their bicomponents (as well as the number of bicomps)
40 compute_components_map();
41
42 // all that remains is to build the graph
43 build_graph();
44}

50template <typename T>
51void BicomponentGraph<T>::compute_components_map() {
52 // Map from edges to their (unique) biconnected component
53 // (this is filled out by call to boost::biconnected_components)
54 EdgeToCompT<T> edge_to_comp;
55
56 std::vector<unsigned> underlying_aps;
57 // call to BGL
58 auto bgl_res = boost::biconnected_components(
59 underlying_g, boost::make_assoc_property_map(edge_to_comp),
60 std::back_inserter(underlying_aps));
61 for (auto v_ap : underlying_aps) {
62 aps.push_back(underlying_g[v_ap]);
63 }
64 unsigned n_components = bgl_res.first;
65
66 // Now populate map from vertices to the set of biconnected components of v
67 for (auto v : boost::make_iterator_range(boost::vertices(underlying_g))) {
68 T node = underlying_g[v];
69 vertex_to_comps.insert({node, {}});
70 // we get the components `v` belongs to by looking at the components
71 // of its incident edges
72 if (boost::degree(v, underlying_g) > 0) {
73 for (auto e :
74 boost::make_iterator_range(boost::out_edges(v, underlying_g))) {
75 // it's a set, we do not need to worry about duplicates
76 vertex_to_comps[node].insert(edge_to_comp[e]);
77 }
78 } else {
79 // if vertex is disconnected, create new exclusive component
80 vertex_to_comps[node].insert(n_components++);
81 }
82 }
83 selected_comps = std::vector<bool>(n_components, false);
84}

86template <typename T>
87void BicomponentGraph<T>::build_graph() {
88 // initialise bicomponent graph with `n_components` vertices
89 unsigned n_components = selected_comps.size();
90 g = Graph(n_components);
91
92 // add edges according to vertex_to_comps
93 for (auto ap : aps) {
94 auto comps = vertex_to_comps[ap];
95 for (auto it = comps.cbegin(); it != comps.cend(); ++it) {
96 unsigned comp1 = *it;
97 for (auto it2 = it; ++it2 != comps.cend();) {
98 unsigned comp2 = *it2;
99 // add edge between any 2 components that `ap` links
100 boost::add_edge(comp1, comp2, {ap}, g);
101 }
102 }
103 }
104}

107template <typename Range>
108void BicomponentGraph<T>::select_comps(Range nodes) {
109 for (T node : nodes) {
110 for (unsigned comp : vertex_to_comps[node]) {
111 selected_comps[comp] = true;
112 }

134template <typename Graph>
135struct TrackUsedEdgesVisitor : public boost::default_dfs_visitor {
136 using edge =
137 std::pair<graphs::utils::vertex<Graph>, graphs::utils::vertex<Graph>>;
138
139 explicit TrackUsedEdgesVisitor(std::vector<bool>& s) : selected(s) {
140 tree_edges = std::make_shared<std::set<edge>>();
141 }
142 std::vector<bool>& selected;
143 std::shared_ptr<std::set<edge>> tree_edges;
144
145 // called as DFS leaves the edge and returns to parent
146 void finish_edge(graphs::utils::edge<Graph> e, const Graph& g) {
147 unsigned target = boost::target(e, g);
148 unsigned source = boost::source(e, g);
149 // only consider tree edges
150 if (!tree_edges->count({source, target})) {
151 return;
152 }
153 if (selected[target]) {
154 selected[source] = true;
155 }
156 }
157 // keep track of tree edges
158 void tree_edge(graphs::utils::edge<Graph> e, const Graph& g) {
159 unsigned target = boost::target(e, g);
160 unsigned source = boost::source(e, g);
161 tree_edges->insert({source, target});
162 }
163};

165template <typename T>
166void BicomponentGraph<T>::propagate_selected_comps() {
167 unsigned root = 0;
168 unsigned n_components = selected_comps.size();
169 while (root < n_components && !selected_comps[root]) {
170 ++root;
171 }
172
173 if (root == n_components) {
174 throw NoSelectedComponent(
175 "At least one component must be selected"
176 " to be able to propagate");
177 }
178
179 // using our custom visitor, DFS will propagate selected_comps as necessary
180 auto params = boost::visitor(TrackUsedEdgesVisitor<Graph>(selected_comps))
181 .root_vertex(root);
182 boost::depth_first_search(g, params);
183}

185template <typename T>
186std::set<T> BicomponentGraph<T>::get_inner_edges() {
187 std::set<T> out;
188 for (auto e : boost::make_iterator_range(boost::edges(g))) {
189 unsigned comp1 = boost::source(e, g);
190 unsigned comp2 = boost::target(e, g);
191 if (selected_comps[comp1] && selected_comps[comp2]) {
192 out.insert(g[e].ap);
193 }
194 }
195 return out;
196}

207template <typename T>
208std::set<T> get_subgraph_aps(
209 const UndirectedConnGraph<T>& graph,
210 const UndirectedConnGraph<T>& subgraph) {
211 detail::BicomponentGraph<T> bicomp_graph(graph);
212 using funcT = std::function<T(unsigned)>;
213 funcT to_node = [&subgraph](unsigned v) { return subgraph[v]; };
214 auto selected = boost::make_iterator_range(boost::vertices(subgraph)) |
215 boost::adaptors::transformed(to_node);
216 bicomp_graph.select_comps(selected);
217 bicomp_graph.propagate_selected_comps();
218 return bicomp_graph.get_inner_edges();
219}
