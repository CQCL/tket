51 template <typename T>
52 void BicomponentGraph<T>::compute_components_map() {
53  // Map from edges to their (unique) biconnected component
54  // (this is filled out by call to boost::biconnected_components)
55  EdgeToCompT<T> edge_to_comp;
56 
57  std::vector<unsigned> underlying_aps;
58  // call to BGL
59  auto bgl_res = boost::biconnected_components(
60  underlying_g, boost::make_assoc_property_map(edge_to_comp),
61  std::back_inserter(underlying_aps));
62  for (auto v_ap : underlying_aps) {
63  aps.push_back(underlying_g[v_ap]);
64  }
65  unsigned n_components = bgl_res.first;
66 
67  // Now populate map from vertices to the set of biconnected components of v
68  for (auto v : boost::make_iterator_range(boost::vertices(underlying_g))) {
69  T node = underlying_g[v];
70  vertex_to_comps.insert({node, {}});
71  // we get the components `v` belongs to by looking at the components
72  // of its incident edges
73  if (boost::degree(v, underlying_g) > 0) {
74  for (auto e :
75  boost::make_iterator_range(boost::out_edges(v, underlying_g))) {
76  // it's a set, we do not need to worry about duplicates
77  vertex_to_comps[node].insert(edge_to_comp[e]);
78  }
79  } else {
80  // if vertex is disconnected, create new exclusive component
81  vertex_to_comps[node].insert(n_components++);
82  }
83  }
84  selected_comps = std::vector<bool>(n_components, false);
85 }

87 template <typename T>
88 void BicomponentGraph<T>::build_graph() {
89  // initialise bicomponent graph with `n_components` vertices
90  unsigned n_components = selected_comps.size();
91  g = Graph(n_components);
92 
93  // add edges according to vertex_to_comps
94  for (auto ap : aps) {
95  auto comps = vertex_to_comps[ap];
96  for (auto it = comps.cbegin(); it != comps.cend(); ++it) {
97  unsigned comp1 = *it;
98  for (auto it2 = it; ++it2 != comps.cend();) {
99  unsigned comp2 = *it2;
100  // add edge between any 2 components that `ap` links
101  boost::add_edge(comp1, comp2, {ap}, g);
102  }
103  }
104  }
105 }

135 template <typename Graph>
136 struct TrackUsedEdgesVisitor : public boost::default_dfs_visitor {
137  using edge =
138  std::pair<graphs::utils::vertex<Graph>, graphs::utils::vertex<Graph>>;
139 
140  explicit TrackUsedEdgesVisitor(std::vector<bool>& s) : selected(s) {
141  tree_edges = std::make_shared<std::set<edge>>();
142  }
143  std::vector<bool>& selected;
144  std::shared_ptr<std::set<edge>> tree_edges;
145 
146  // called as DFS leaves the edge and returns to parent
147  void finish_edge(graphs::utils::edge<Graph> e, const Graph& g) {
148  unsigned target = boost::target(e, g);
149  unsigned source = boost::source(e, g);
150  // only consider tree edges
151  if (!tree_edges->count({source, target})) {
152  return;
153  }
154  if (selected[target]) {
155  selected[source] = true;
156  }
157  }
158  // keep track of tree edges
159  void tree_edge(graphs::utils::edge<Graph> e, const Graph& g) {
160  unsigned target = boost::target(e, g);
161  unsigned source = boost::source(e, g);
162  tree_edges->insert({source, target});
163  }
164 };

166 template <typename T>
167 void BicomponentGraph<T>::propagate_selected_comps() {
168  unsigned root = 0;
169  unsigned n_components = selected_comps.size();
170  while (root < n_components && !selected_comps[root]) {
171  ++root;
172  }
173 
174  if (root == n_components) {
175  throw NoSelectedComponent(
176  "At least one component must be selected"
177  " to be able to propagate");
178  }
179 
180  // using our custom visitor, DFS will propagate selected_comps as necessary
181  auto params = boost::visitor(TrackUsedEdgesVisitor<Graph>(selected_comps))
182  .root_vertex(root);
183  boost::depth_first_search(g, params);
184 }

186 template <typename T>
187 std::set<T> BicomponentGraph<T>::get_inner_edges() {
188  std::set<T> out;
189  for (auto e : boost::make_iterator_range(boost::edges(g))) {
190  unsigned comp1 = boost::source(e, g);
191  unsigned comp2 = boost::target(e, g);
192  if (selected_comps[comp1] && selected_comps[comp2]) {
193  out.insert(g[e].ap);
194  }
195  }
196  return out;
197 }

208 template <typename T>
209 std::set<T> get_subgraph_aps(
210  const UndirectedConnGraph<T>& graph,
211  const UndirectedConnGraph<T>& subgraph) {
212  detail::BicomponentGraph<T> bicomp_graph(graph);
213  using funcT = std::function<T(unsigned)>;
214  funcT to_node = [&subgraph](unsigned v) { return subgraph[v]; };
215  auto selected = boost::make_iterator_range(boost::vertices(subgraph)) |
216  boost::adaptors::transformed(to_node);
217  bicomp_graph.select_comps(selected);
218  bicomp_graph.propagate_selected_comps();
219  return bicomp_graph.get_inner_edges();
220 }
