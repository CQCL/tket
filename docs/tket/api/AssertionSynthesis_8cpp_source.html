47 static std::tuple<VectorXb, Eigen::MatrixXcd, int> projector_diagonalisation(
48  const Eigen::MatrixXcd &P) {
49  // Solve the eigen values
50  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXcd> eigen_solver(P);
51  // Make a permutation matrix to reorder D and U_dag
52  Eigen::PermutationMatrix<Eigen::Dynamic> perm(P.rows());
53  Eigen::MatrixXcd P1 = eigen_solver.eigenvectors() *
54  eigen_solver.eigenvalues().asDiagonal() *
55  eigen_solver.eigenvectors().adjoint();
56  TKET_ASSERT(P1.isApprox(P));
57  // Cast eigenvalues to booleans
58  VectorXb eigenvalues(eigen_solver.eigenvalues().rows());
59  for (unsigned i = 0; i < eigenvalues.rows(); i++) {
60  if (std::abs(eigen_solver.eigenvalues()[i]) < EPS) {
61  eigenvalues[i] = 0;
62  } else {
63  eigenvalues[i] = 1;
64  }
65  }
66  // Move 1 to the front
67  unsigned j = 0;
68  for (unsigned i = 0; i < P.rows(); i++) {
69  if (eigenvalues[i] == 1) {
70  perm.indices()[j] = i;
71  j++;
72  }
73  }
74  // Assign rank
75  unsigned rank = j;
76  // Pad 0 to the end
77  for (unsigned i = 0; i < P.rows(); i++) {
78  if (eigenvalues[i] == 0) {
79  perm.indices()[j] = i;
80  j++;
81  }
82  }
83  VectorXb D = perm.transpose() * eigenvalues;
84  Eigen::MatrixXcd U = eigen_solver.eigenvectors() * perm;
85  TKET_ASSERT((U * D.cast<double>().asDiagonal() * U.adjoint()).isApprox(P));
86  return {D, U, rank};
87 }

103 static std::vector<Vector2b> tensor_factorization(const VectorXb &D) {
104  unsigned dim = D.rows();
105  unsigned log_dim = (unsigned)log2(dim);
106  std::vector<Vector2b> factorisations;
107  for (unsigned i = 0; i < log_dim; i++) {
108  factorisations.push_back(Vector2b::Zero());
109  }
110  // The diagonal matrix D can be factorised into a tensorproduct of
111  // log_dim 2x2 diagonal boolean matrices in the set {\0><0|, |1><1|, I}
112  // The factorisation can be determined by the diagonals in D
113  // For example, if D[0,0] == 1, then all the 2x2 matices should have 1 in
114  // their top left entry.
115  for (unsigned i = 0; i < dim; i++) {
116  if (D(i) == 1) {
117  for (unsigned j = 0; j < log_dim; j++) {
118  if (get_bit(i, j)) {
119  factorisations[j](1) = 1;
120  } else {
121  factorisations[j](0) = 1;
122  }
123  }
124  }
125  }
126  return factorisations;
127 }

129 static void apply_unitary(Circuit *circ, const Eigen::MatrixXcd &U) {
130  if (U.rows() == 2) {
131  // Add a Unitary1qBox
132  Unitary1qBox ubox(U);
133  circ->add_op<unsigned>(std::make_shared<Unitary1qBox>(ubox), {0});
134  } else if (U.rows() == 4) {
135  // Add a Unitary2qBox
136  Unitary2qBox ubox(U);
137  circ->add_op<unsigned>(std::make_shared<Unitary2qBox>(ubox), {0, 1});
138  } else if (U.rows() == 8) {
139  // Add a Unitary3qBox
140  Unitary3qBox ubox(U);
141  circ->add_op<unsigned>(std::make_shared<Unitary3qBox>(ubox), {0, 1, 2});
142 
143  } else {
144  throw CircuitInvalidity("Only 2x2, 4x4, and 8x8 projectors are supported");
145  }
146 }

161 static unsigned apply_z_measurements(
162  Circuit &circ, const std::vector<Vector2b> z_projectors,
163  unsigned debug_bit_index, std::vector<bool> &expected_readouts) {
164  for (unsigned i = 0; i < z_projectors.size(); i++) {
165  if (z_projectors[i](0) == 1 && z_projectors[i](1) == 1) {
166  continue;
167  }
168  Qubit q(i);
169  Bit b(debug_bit_index++);
170  circ.add_bit(b);
171  circ.add_op<UnitID>(OpType::Measure, {q, b});
172  const bool is_1_projector =
173  z_projectors[i](0) == 0 && z_projectors[i](1) == 1;
174  expected_readouts.push_back(is_1_projector);
175  }
176  return debug_bit_index;
177 }

199 static std::tuple<VectorXb, Eigen::MatrixXcd, VectorXb, Eigen::MatrixXcd>
200 projector_split(
201  const VectorXb &D, const Eigen::MatrixXcd &U, const unsigned &rank) {
202  VectorXb D_new(D);
203  for (unsigned i = rank; i < D_new.rows() / 2; i++) {
204  D_new(i) = 1;
205  }
206  // U1
207  Eigen::MatrixXcd U1(U);
208  // U2
209  Eigen::PermutationMatrix<Eigen::Dynamic> perm(D_new.rows());
210  for (unsigned i = 0; i < D_new.rows(); i++) {
211  if (i >= rank && i < D_new.rows() / 2) {
212  perm.indices()[i] = i + D_new.rows() / 2 - rank;
213  } else if (i >= D_new.rows() / 2 && i < D_new.rows() - rank) {
214  perm.indices()[i] = i - (D_new.rows() / 2 - rank);
215  } else {
216  perm.indices()[i] = i;
217  }
218  }
219  Eigen::MatrixXcd U2 = U * perm;
220  return {D_new, U1, D_new, U2};
221 }

223 std::tuple<Circuit, std::vector<bool>> projector_assertion_synthesis(
224  const Eigen::MatrixXcd &P) {
225  // Diagonoalise the projector P
226  Eigen::MatrixXcd projector(P);
227  unsigned n_qubits = (unsigned)log2(projector.rows());
228  VectorXb D;
229  Eigen::MatrixXcd U;
230  unsigned rank;
231  std::tie(D, U, rank) = projector_diagonalisation(projector);
232  if (rank == 0) {
233  throw CircuitInvalidity("The projector must have none-zero rank");
234  }
235  std::vector<bool> expected_readouts;
236  Circuit circ(n_qubits);
237  if (rank > projector.rows() / 2) {
238  if (n_qubits >= 3) {
239  throw CircuitInvalidity(
240  "8x8 projector that requires an ancilla is not supported");
241  }
242  // Add an auxilary qubit to the end
243  Qubit ancilla(n_qubits);
244  circ.add_qubit(ancilla, false);
245  circ.add_op<Qubit>(OpType::Reset, {ancilla});
246  Eigen::MatrixXcd zero_projector = Eigen::MatrixXcd::Zero(2, 2);
247  zero_projector(0, 0) = 1;
248  Eigen::MatrixXcd new_projector =
249  Eigen::KroneckerProduct(projector, zero_projector);
250  std::tie(D, U, rank) = projector_diagonalisation(new_projector);
251  }
252 
253  // Initialise debug bit index
254  unsigned debug_bit_index = 0;
255  // Check if the rank is power of 2
256  // If a number is power of 2 then the count of binary 1s will be 1.
257  if (rank != 0 && (rank & (rank - 1)) == 0) {
258  // Implement the projection
259  // Apply the transformation before the measurements
260  std::vector<Vector2b> tensors = tensor_factorization(D);
261  apply_unitary(&circ, U.adjoint());
262  apply_z_measurements(circ, tensors, debug_bit_index, expected_readouts);
263  apply_unitary(&circ, U);
264  }
265  // Transform this into two projectors
266  else {
267  auto [D1, U1, D2, U2] = projector_split(D, U, rank);
268  std::vector<Vector2b> tensors1 = tensor_factorization(D1);
269  apply_unitary(&circ, U1.adjoint());
270  debug_bit_index = apply_z_measurements(
271  circ, tensors1, debug_bit_index, expected_readouts);
272  apply_unitary(&circ, U1);
273 
274  std::vector<Vector2b> tensors2 = tensor_factorization(D2);
275  apply_unitary(&circ, U2.adjoint());
276  apply_z_measurements(circ, tensors2, debug_bit_index, expected_readouts);
277  apply_unitary(&circ, U2);
278  }
279  return {circ, expected_readouts};
280 }

282 static unsigned get_n_qubits_from_stabilisers(
283  const PauliStabiliserVec &paulis) {
284  if (paulis.size() == 0) {
285  throw CircuitInvalidity("Stabilisers cannot be empty");
286  }
287  unsigned stabiliser_len = paulis[0].string.size();
288  for (unsigned i = 1; i < paulis.size(); i++) {
289  if (paulis[i].string.size() != stabiliser_len) {
290  throw CircuitInvalidity("Stabilisers have unequal lengths");
291  }
292  }
293  return stabiliser_len;
294 }

309 static unsigned add_assertion_operator(
310  Circuit &circ, const PauliStabiliser &pauli, unsigned debug_bit_index,
311  const Qubit &ancilla, std::vector<bool> &expected_readouts) {
312  circ.add_op<Qubit>(OpType::Reset, {ancilla});
313  circ.add_op<Qubit>(OpType::H, {ancilla});
314  bool identity = true;
315  for (unsigned i = 0; i < pauli.size(); i++) {
316  switch (pauli.string[i]) {
317  case Pauli::I:
318  break;
319  case Pauli::X:
320  circ.add_op<Qubit>(OpType::CX, {ancilla, Qubit(i)});
321  identity = false;
322  break;
323  case Pauli::Y:
324  circ.add_op<Qubit>(OpType::CY, {ancilla, Qubit(i)});
325  identity = false;
326  break;
327  case Pauli::Z:
328  circ.add_op<Qubit>(OpType::CZ, {ancilla, Qubit(i)});
329  identity = false;
330  break;
331  }
332  }
333  if (identity)
334  throw std::invalid_argument(
335  "StabiliserAssertionBox cannot assert an identity.");
336  circ.add_op<Qubit>(OpType::H, {ancilla});
337  Bit b(debug_bit_index++);
338  circ.add_bit(b);
339  circ.add_op<UnitID>(OpType::Measure, {ancilla, b});
340  expected_readouts.push_back(pauli.is_real_negative());
341  return debug_bit_index;
342 }

344 // Assume all Pauli stabilisers have equal lengths and no identity
345 std::tuple<Circuit, std::vector<bool>> stabiliser_assertion_synthesis(
346  const PauliStabiliserVec &paulis) {
347  unsigned n_qubits = get_n_qubits_from_stabilisers(paulis);
348  std::vector<bool> expected_readouts;
349  Circuit circ(n_qubits);
350 
351  // Initialise debug bit index
352  unsigned debug_bit_index = 0;
353  // Add an ancilla
354  Qubit ancilla(n_qubits);
355  circ.add_qubit(ancilla, false);
356 
357  for (auto &pauli : paulis) {
358  debug_bit_index = add_assertion_operator(
359  circ, pauli, debug_bit_index, ancilla, expected_readouts);
360  }
361 
362  return {circ, expected_readouts};
363 }
