41 // whether source and target commute
42 static bool ends_commute(const Circuit &circ, const Edge &e) {
43  const std::pair<port_t, port_t> ports = circ.get_ports(e);
44  const Vertex source = circ.source(e);
45  const Vertex target = circ.target(e);
46 
47  // We currently do not support commuting multi-qubit gates
48  // TODO: It would be useful to support commuting single-qubit gates with
49  // classical conditioning
50  if (circ.n_in_edges(source) > 1 && circ.n_in_edges(target) > 1) {
51  return false;
52  }
53 
54  auto colour = circ.commuting_basis(target, PortType::Target, ports.second);
55  return circ.commutes_with_basis(
56  source, colour, PortType::Source, ports.first);
57 }

59 // moves single qubit operations past multiqubit operations they commute with,
60 // towards front of circuit (hardcoded)
61 static bool commute_singles_to_front(Circuit &circ) {
62  bool success = false;
63  // follow each qubit path from output to input
64  for (const Qubit &q : circ.all_qubits()) {
65  Vertex prev_v = circ.get_out(q);
66  Edge current_e = circ.get_nth_in_edge(prev_v, 0);
67  Vertex current_v = circ.source(current_e);
68  while (!is_initial_q_type(circ.get_OpType_from_Vertex(current_v))) {
69  const Op_ptr curr_op = circ.get_Op_ptr_from_Vertex(current_v);
70  // if current vertex is a multiqubit gate
71  if (circ.n_in_edges_of_type(current_v, EdgeType::Quantum) > 1) {
72  while (circ.n_in_edges_of_type(prev_v, EdgeType::Quantum) == 1 &&
73  ends_commute(circ, current_e)) {
74  // subsequent op on qubit path is a single qubit gate
75  // and commutes with current multi qubit gate
76  success = true;
77  EdgeVec rewire_edges;
78  op_signature_t edge_types;
79  for (const Edge &e : circ.get_in_edges(prev_v)) {
80  EdgeType type = circ.get_edgetype(e);
81  Edge boundary_edge;
82  // Currently, only purely-quantum operations can be commuted
83  // through. This is guaranteed by `ends_commute`. It follows that
84  // any wire out of `prev_v` must be EdgeType::Quantum.
85  TKET_ASSERT(type == EdgeType::Quantum);
86  boundary_edge = circ.get_last_edge(current_v, current_e);
87  rewire_edges.push_back(boundary_edge);
88  edge_types.push_back(type);
89  }
90  const port_t backup_port = circ.get_source_port(current_e);
91  circ.remove_vertex(
92  prev_v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
93  circ.rewire(prev_v, rewire_edges, edge_types);
94  current_e = circ.get_nth_out_edge(current_v, backup_port);
95  prev_v = circ.target(current_e);
96  }
97  }
98  // move to next vertex (towards input)
99  prev_v = current_v;
100  std::tie(current_v, current_e) = circ.get_prev_pair(current_v, current_e);
101  }
102  }
103 
104  return success;
105 }

118 static bool replace_two_qubit_interaction(
119  Circuit &circ, Interaction &i, std::map<Qubit, Edge> &current_edges,
120  VertexList &bin, OpType target, double cx_fidelity, bool allow_swaps) {
121  EdgeVec in_edges = {i.e0, i.e1};
122  EdgeVec out_edges = {current_edges[i.q0], current_edges[i.q1]};
123  Edge next0, next1;
124  bool q0_is_out = is_final_q_type(
125  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q0])));
126  bool q1_is_out = is_final_q_type(
127  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q1])));
128  if (!q0_is_out) {
129  next0 = circ.get_next_edge(
130  circ.target(current_edges[i.q0]), current_edges[i.q0]);
131  }
132  if (!q1_is_out) {
133  next1 = circ.get_next_edge(
134  circ.target(current_edges[i.q1]), current_edges[i.q1]);
135  }
136  // Circuit to (potentially) substitute
137  Subcircuit sub = {in_edges, out_edges, i.vertices};
138  Circuit subc = circ.subcircuit(sub);
139 
140  // Try to simplify using KAK
141  Circuit replacement = subc;
142  decompose_multi_qubits_TK2().apply(replacement);
143  Eigen::Matrix4cd mat = get_matrix_from_2qb_circ(replacement);
144  replacement = two_qubit_canonical(mat);
145  TwoQbFidelities fid;
146  fid.CX_fidelity = cx_fidelity;
147  if (target != OpType::TK2) {
148  decompose_TK2(fid, allow_swaps).apply(replacement);
149  }
150 
151  // Whether to substitute old circuit with new
152  bool substitute = false;
153  for (Vertex v : subc.vertices_in_order()) {
154  unsigned n_ins = subc.n_in_edges_of_type(v, EdgeType::Quantum);
155  if (n_ins == 2 && subc.get_OpType_from_Vertex(v) != target) {
156  // Old circuit has non-target gates => we need to substitute
157  substitute = true;
158  break;
159  }
160  }
161  if (!substitute) {
162  if (target == OpType::CX) {
163  unsigned nb_2qb_old = subc.count_gates(target);
164  unsigned nb_2qb_new = replacement.count_gates(target);
165  substitute |= nb_2qb_new < nb_2qb_old;
166  } else if (target == OpType::TK2) {
167  unsigned cnt_2qb = 0;
168  for (Vertex v : subc.vertices_in_order()) {
169  unsigned n_ins = subc.n_in_edges_of_type(v, EdgeType::Quantum);
170  cnt_2qb += n_ins == 2;
171  }
172  substitute |= cnt_2qb >= 2;
173  }
174  }
175 
176  if (substitute) {
177  // Substitute interaction with new circuit
178  bin.insert(bin.end(), sub.verts.begin(), sub.verts.end());
179  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
180  if (!q0_is_out) {
181  current_edges[i.q0] = circ.get_last_edge(circ.source(next0), next0);
182  }
183  if (!q1_is_out) {
184  current_edges[i.q1] = circ.get_last_edge(circ.source(next1), next1);
185  }
186  return true;
187  } else {
188  // Leave circuit untouched
189  return false;
190  }
191 }

193 Transform commute_and_combine_HQS2() {
194  return Transform([](Circuit &circ) {
195  bool success = false;
196  VertexList bin;
197  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
198  EdgeVec outs = circ.get_all_out_edges(v);
199  if (circ.get_OpType_from_Vertex(v) == OpType::ZZMax && outs.size() == 2) {
200  Vertex next0 = boost::target(outs[0], circ.dag);
201  Vertex next1 = boost::target(outs[1], circ.dag);
202  if (next0 == next1 &&
203  circ.get_OpType_from_Vertex(next0) == OpType::ZZMax) {
204  success = true;
205  EdgeVec h_in = circ.get_in_edges(v);
206  EdgeVec h_out = circ.get_all_out_edges(next0);
207  if (circ.get_target_port(outs[0]) != 0) {
208  h_out = {h_out[1], h_out[0]};
209  }
210  bin.push_back(v);
211  bin.push_back(next0);
212  Subcircuit sub = {h_in, h_out};
213  circ.substitute(
214  CircPool::two_Rz1(), sub, Circuit::VertexDeletion::No);
215  circ.add_phase(0.5);
216  continue;
217  }
218  if (circ.get_OpType_from_Vertex(next0) == OpType::Rz) {
219  success = true;
220  circ.remove_vertex(
221  next0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
222  Edge in_0 = circ.get_nth_in_edge(v, 0);
223  circ.rewire(next0, {in_0}, {EdgeType::Quantum});
224  }
225  if (circ.get_OpType_from_Vertex(next1) == OpType::Rz) {
226  success = true;
227  circ.remove_vertex(
228  next1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
229  Edge in_1 = circ.get_nth_in_edge(v, 1);
230  circ.rewire(next1, {in_1}, {EdgeType::Quantum});
231  }
232  }
233  }
234  circ.remove_vertices(
235  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
236  return success;
237  });

244 Transform two_qubit_squash(
245  OpType target_2qb_gate, double cx_fidelity, bool allow_swaps) {
246  const std::set<OpType> accepted_ots{OpType::CX, OpType::TK2};
247  if (!accepted_ots.contains(target_2qb_gate)) {
248  throw BadOpType(
249  "KAKDecomposition currently supports CX and TK2. "
250  "Cannot decompose to",
251  target_2qb_gate);
252  }
253  if (cx_fidelity < 0 || cx_fidelity > 1) {
254  throw std::invalid_argument("The CX fidelity must be between 0 and 1.");
255  }
256 
257  return Transform([target_2qb_gate, cx_fidelity, allow_swaps](Circuit &circ) {
258  bool success = false;
259  VertexList bin;
260  // Get map from vertex/port to qubit number
261  std::map<VertPort, Qubit> v_to_qb;
262  std::map<Qubit, Edge> current_edge_on_qb;
263  std::vector<Interaction> i_vec;
264  std::map<Qubit, int> current_interaction;
265  for (const Qubit &qb : circ.all_qubits()) {
266  for (const VertPort &vp : circ.unit_path(qb)) {
267  v_to_qb.insert({vp, qb});
268  }
269  Vertex input = circ.get_in(qb);
270  Edge e = circ.get_nth_out_edge(input, 0);
271  current_edge_on_qb[qb] = e;
272  current_interaction[qb] = -1;
273  }
274  SliceVec slices = circ.get_slices();
275  slices.insert(slices.begin(), circ.q_inputs());
276  slices.push_back(circ.q_outputs());
277  for (SliceVec::iterator s = slices.begin(); s != slices.end(); ++s) {
278  for (Slice::iterator v = s->begin(); v != s->end(); ++v) {
279  const Op_ptr o = circ.get_Op_ptr_from_Vertex(*v);
280  OpType type = o->get_type();
281  unsigned n_ins = circ.n_in_edges_of_type(*v, EdgeType::Quantum);
282  // Ignore classical ops
283  if (is_classical_type(type)) {
284  continue;
285  } else if (
286  is_projective_type(type) || is_final_q_type(type) ||
287  type == OpType::Barrier || type == OpType::Conditional ||
288  n_ins > 2 || !o->free_symbols().empty()) {
289  // Measures, resets, outputs, barriers, symbolic gates, conditionals
290  // and many-qubit gates close interactions
291  EdgeVec q_edges = circ.get_in_edges_of_type(*v, EdgeType::Quantum);
292  std::vector<port_t> q_ports;
293  for (const Edge &e : q_edges) {
294  q_ports.push_back(circ.get_target_port(e));
295  }
296  for (const port_t &port : q_ports) {
297  Qubit q = v_to_qb.at({*v, port});
298  int i = current_interaction[q];
299  if (i != -1) {
300  if (i_vec[i].count >= 2) {
301  // Replace subcircuit
302  success |= replace_two_qubit_interaction(
303  circ, i_vec[i], current_edge_on_qb, bin, target_2qb_gate,
304  cx_fidelity, allow_swaps);
305  }
306  current_interaction[i_vec[i].q0] = -1;
307  current_interaction[i_vec[i].q1] = -1;
308  }
309  if (!is_final_q_type(type)) {
310  current_edge_on_qb[q] =
311  circ.get_next_edge(*v, current_edge_on_qb[q]);
312  }
313  }
314  } else if (circ.n_in_edges_of_type(*v, EdgeType::Quantum) == 2) {
315  // Check for 2qb gate
316  Qubit q0 = v_to_qb.at({*v, 0});
317  Qubit q1 = v_to_qb.at({*v, 1});
318  int i0 = current_interaction[q0];
319  int i1 = current_interaction[q1];
320  // If they are already interacting, extend it
321  if (i0 != -1 && i0 == i1) {
322  i_vec[i0].count++;
323  i_vec[i0].vertices.insert(*v);
324  current_edge_on_qb[q0] =
325  circ.get_next_edge(*v, current_edge_on_qb[q0]);
326  current_edge_on_qb[q1] =
327  circ.get_next_edge(*v, current_edge_on_qb[q1]);
328  } else {
329  // End any other interactions on q0
330  if (i0 != -1) {
331  if (i_vec[i0].count >= 2) {
332  // Replace subcircuit
333  success |= replace_two_qubit_interaction(
334  circ, i_vec[i0], current_edge_on_qb, bin, target_2qb_gate,
335  cx_fidelity, allow_swaps);
336  }
337  current_interaction[i_vec[i0].q0] = -1;
338  current_interaction[i_vec[i0].q1] = -1;
339  }
340  // End any other interactions on q1
341  if (i1 != -1) {
342  if (i_vec[i1].count >= 2) {
343  // Replace subcircuit
344 
345  success |= replace_two_qubit_interaction(
346  circ, i_vec[i1], current_edge_on_qb, bin, target_2qb_gate,
347  cx_fidelity, allow_swaps);
348  }
349  current_interaction[i_vec[i1].q0] = -1;
350  current_interaction[i_vec[i1].q1] = -1;
351  }
352  // Add new interaction
353  Interaction new_i(q0, q1);
354  new_i.e0 = current_edge_on_qb[q0];
355  new_i.e1 = current_edge_on_qb[q1];
356  new_i.count = 1;
357  new_i.vertices = {*v};
358  current_interaction[q0] = i_vec.size();
359  current_interaction[q1] = i_vec.size();
360  i_vec.push_back(new_i);
361  current_edge_on_qb[q0] =
362  circ.get_next_edge(*v, current_edge_on_qb[q0]);
363  current_edge_on_qb[q1] =
364  circ.get_next_edge(*v, current_edge_on_qb[q1]);
365  }
366  } else {
367  // We don't care about single-qubit vertices, so just update edges
368  // and add vertices if interactions exist
369  for (port_t i = 0; i < circ.n_in_edges(*v); i++) {
370  Qubit q = v_to_qb.at({*v, i});
371  current_edge_on_qb[q] =
372  circ.get_next_edge(*v, current_edge_on_qb[q]);
373  int inter = current_interaction[q];
374  if (inter != -1) {
375  i_vec[inter].vertices.insert(*v);
376  }
377  }
378  }
379  }
380  }
381  circ.remove_vertices(
382  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
383 
384  if (success) {
385  squash_1qb_to_tk1().apply(circ);
386  }
387  return success;
388  });
389 }

391 // Given a 'SWAP_chain', finds edge in chain (or qubit wire) with best fidelity
392 // and rewires the associated single qubit vertex in to it
393 static bool find_edge_rewire_vertex(
394  Circuit &circ,
395  std::pair<std::vector<std::pair<Edge, double>>, Vertex> &entry) {
396  std::vector<std::pair<Edge, double>> candidates = entry.first;
397  std::pair<Edge, double> best_pair = candidates[0];
398  bool do_rewire = false; // prevents rewiring if current edge is best edge
399  // (causes many issues...)
400  for (unsigned i = 1; i < candidates.size(); i++) { // find best edge
401  if (candidates[i].second > best_pair.second) {
402  best_pair = candidates[i];
403  do_rewire = true;
404  }
405  }
406  if (do_rewire) {
407  // rewire in to best edge
408  circ.remove_vertex(
409  entry.second, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
410  circ.rewire(entry.second, {best_pair.first}, {EdgeType::Quantum});
411  }
412  return do_rewire;
413 }

415 // Given a SWAP vertex, which has some predecessor SWAP vertex, find the
416 // 'SWAP_chain' this predecessor SWAP vertex is in and add to it
417 static void extend_SWAP_chain(
418  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
419  &swap_chains,
420  Edge entry_edge, Node entry_node, const Edge &match, const Circuit &circ,
421  const DeviceCharacterisation &characterisation) {
422  for (auto it = swap_chains.begin(); it != swap_chains.end(); ++it) {
423  if ((*it).first.back().first == match) {
424  // extend chain, adding a new pair of edge and double to the end
425  (*it).first.push_back(
426  {entry_edge,
427  1.0 - characterisation.get_error(
428  entry_node, circ.get_OpType_from_Vertex((*it).second))});
429  return;
430  }
431  }
432 }

434 // Finds sequences of adjacent SWAP gates, with a predecessor Single Qubit
435 // Vertex. The error rate of the required Single Qubit Vertex is stored for each
436 // of the PhysicalQubits the LogicalQubit passes through Once 'SWAP_chains' are
437 // found throughout the whole circuit, predecessor Single Qubit Vertices are
438 // rewired into the edge with best error rate
439 static bool find_rewire_sq(
440  Circuit &circ, const DeviceCharacterisation &characterisation) {
441  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
442  swap_chains;
443  for (auto it = circ.begin(); it != circ.end(); ++it) {
444  if (it->get_op_ptr()->get_type() == OpType::SWAP) {
445  // find SWAP, if either predecessor is a single qubit unitary
446  // find resulting swap chain...
447  Vertex swap_vert = it.get_vertex();
448  unit_vector_t qubits = it->get_args();
449  node_vector_t nodes = {qubits.begin(), qubits.end()};
450  VertexVec pred_verts = circ.get_predecessors(swap_vert);
451  EdgeVec pred_edges = circ.get_in_edges(swap_vert);
452  EdgeVec post_edges = circ.get_all_out_edges(swap_vert);
453  for (unsigned i = 0; i < pred_verts.size(); i++) {
454  OpType optype = circ.get_OpType_from_Vertex(
455  pred_verts[i]); // OPT IS PREDECESSOR OPTYPE
456  if (circ.detect_singleq_unitary_op(pred_verts[i])) {
457  // wire has single qubit unitary->add a new swap chain
458  std::vector<std::pair<Edge, double>> swap_chain = {
459  {pred_edges[i],
460  1.0 - characterisation.get_error(nodes[i], optype)},
461  {post_edges[1 - i],
462  1.0 - characterisation.get_error(nodes[1 - i], optype)}};
463  swap_chains.push_back(std::make_pair(swap_chain, pred_verts[i]));
464  } else if (optype == OpType::SWAP) { // wire has swap->assume this swap
465  // already in chain, find chain
466  Edge pre_edge = pred_edges[i];
467  extend_SWAP_chain(
468  swap_chains, post_edges[1 - i], nodes[1 - i], pre_edge, circ,
469  characterisation);
470  }
471  }
472  }
473  }
474  // having produced swap chains, now find best qubit for gate to act on and
475  // implement
476  bool success = false;
477  while (!swap_chains.empty()) {
478  if (find_edge_rewire_vertex(circ, (*swap_chains.begin())) == true) {
479  success = true;
480  }
481  swap_chains.erase(swap_chains.begin());
482  }
483  return success;
484 }

506 Transform absorb_Rz_NPhasedX() {
507  return Transform([](Circuit &circ) {
508  bool success = false;
509  VertexSet all_bins;
510 
511  // Start by squashing Rz gates
512  success |= squash_1qb_to_pqp(OpType::Rz, OpType::Rx).apply(circ);
513 
514  // Loop through all NPhasedX gates
515  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
516  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
517  if (op->get_type() == OpType::NPhasedX) {
518  // gather surrounding Rz gates
519  unsigned arity = op->n_qubits();
520  std::vector<Expr> in_rz(arity);
521  std::vector<Expr> out_rz(arity);
522  EdgeVec in_edges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
523  EdgeVec out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
524  TKET_ASSERT(in_edges.size() == arity);
525  TKET_ASSERT(out_edges.size() == arity);
526  for (unsigned i = 0; i < arity; ++i) {
527  Vertex in_v = circ.source(in_edges[i]);
528  Op_ptr in_op = circ.get_Op_ptr_from_Vertex(in_v);
529  Vertex out_v = circ.target(out_edges[i]);
530  Op_ptr out_op = circ.get_Op_ptr_from_Vertex(out_v);
531 
532  if (in_op->get_type() == OpType::Rz) {
533  in_rz[i] = -in_op->get_params().at(0);
534  } else {
535  in_rz[i] = 0.;
536  }
537  if (out_op->get_type() == OpType::Rz) {
538  out_rz[i] = out_op->get_params().at(0);
539  } else {
540  out_rz[i] = 0.;
541  }
542  }
543 
544  // Find out which Rz angle is most popular.
545  // Note that we only compare expr[i] with expr[j] when j < i. This means
546  // that only the largest i from a set of equivalent exprs will have the
547  // right occurence count, but that is good enough.
548  std::vector<Expr> all_rz = in_rz;
549  all_rz.insert(all_rz.end(), out_rz.begin(), out_rz.end());
550  std::vector<unsigned> occurences_count(2 * arity);
551  for (unsigned i = 0; i < 2 * arity; ++i) {
552  unsigned cnt = 0;
553  for (unsigned j = 0; j < i; ++j) {
554  if (equiv_expr(all_rz[i], all_rz[j], 4)) {
555  ++cnt;
556  }
557  }
558  occurences_count[i] = cnt;
559  }
560  unsigned max_i =
561  std::max_element(occurences_count.begin(), occurences_count.end()) -
562  occurences_count.begin();
563  Expr absorb_rz = all_rz[max_i];
564 
565  if (!equiv_0(absorb_rz, 4)) {
566  success = true;
567 
568  // Subtract absorb_rz in NPhasedX
569  std::vector<Expr> new_params = op->get_params();
570  TKET_ASSERT(new_params.size() == 2);
571  new_params[1] += absorb_rz;
572  circ.dag[v] = get_op_ptr(OpType::NPhasedX, new_params, arity);
573 
574  // Finally, adjust +-absorb_rz in Rz everywhere around
575  for (unsigned i = 0; i < arity; ++i) {
576  Vertex in_v = circ.source(in_edges[i]);
577  Op_ptr in_op = circ.get_Op_ptr_from_Vertex(in_v);
578  Vertex out_v = circ.target(out_edges[i]);
579  Op_ptr out_op = circ.get_Op_ptr_from_Vertex(out_v);
580 
581  Expr angle;
582  Edge in_e, out_e;
583  VertexSet bin;
584  if (in_op->get_type() == OpType::Rz) {
585  angle = in_op->get_params().at(0) + absorb_rz;
586  out_e = in_edges[i];
587  in_e = circ.get_last_edge(in_v, out_e);
588  bin = {in_v};
589  } else {
590  angle = absorb_rz;
591  out_e = in_edges[i];
592  in_e = out_e;
593  bin = {};
594  }
595  Subcircuit sub{{in_e}, {out_e}, bin};
596  Circuit c(1);
597  if (!equiv_0(angle, 4)) {
598  c.add_op<unsigned>(OpType::Rz, angle, {0});
599  }
600  circ.substitute(c, sub, Circuit::VertexDeletion::No);
601  all_bins.insert(bin.begin(), bin.end());
602 
603  if (out_op->get_type() == OpType::Rz) {
604  angle = out_op->get_params().at(0) - absorb_rz;
605  in_e = out_edges[i];
606  out_e = circ.get_next_edge(out_v, in_e);
607  bin = {out_v};
608  } else {
609  angle = -absorb_rz;
610  in_e = out_edges[i];
611  out_e = in_e;
612  bin = {};
613  }
614  sub = Subcircuit{{in_e}, {out_e}, bin};
615  c = Circuit(1);
616  if (!equiv_0(angle, 4)) {
617  c.add_op<unsigned>(OpType::Rz, angle, {0});
618  }
619  circ.substitute(c, sub, Circuit::VertexDeletion::No);
620  all_bins.insert(bin.begin(), bin.end());
621  }
622  }
623  }
624  }
625  circ.remove_vertices(
626  all_bins, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
627 
628  return success;
629  });
630 }

632 Transform ZZPhase_to_Rz() {
633  // basic optimisation, replace ZZPhase with two Rz(1)
634  return Transform([](Circuit &circ) {
635  bool success = false;
636  VertexSet bin;
637 
638  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
639  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);

641  auto params = op->get_params();

643  // evaluate

645  if (std::abs(param_value) == 1.0) {
646  success = true;
647  // basic optimisation, replace ZZPhase with two Rz(1)
648  Circuit replacement(2);

653  }
