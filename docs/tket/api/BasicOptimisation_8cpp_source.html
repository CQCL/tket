45 // whether source and target commute
46 static bool ends_commute(const Circuit &circ, const Edge &e) {
47  const std::pair<port_t, port_t> ports = circ.get_ports(e);
48  const Vertex source = circ.source(e);
49  const Vertex target = circ.target(e);
50 
51  // We currently do not support commuting multi-qubit gates
52  // TODO: It would be useful to support commuting single-qubit gates with
53  // classical conditioning
54  if (circ.n_in_edges(source) > 1 && circ.n_in_edges(target) > 1) {
55  return false;
56  }
57 
58  auto colour = circ.commuting_basis(target, PortType::Target, ports.second);
59  return circ.commutes_with_basis(
60  source, colour, PortType::Source, ports.first);
61 }

63 // moves single qubit operations past multiqubit operations they commute with,
64 // towards front of circuit (hardcoded)
65 static bool commute_singles_to_front(Circuit &circ) {
66  bool success = false;
67  // follow each qubit path from output to input
68  for (const Qubit &q : circ.all_qubits()) {
69  Vertex prev_v = circ.get_out(q);
70  Edge current_e = circ.get_nth_in_edge(prev_v, 0);
71  Vertex current_v = circ.source(current_e);
72  while (!is_initial_q_type(circ.get_OpType_from_Vertex(current_v))) {
73  const Op_ptr curr_op = circ.get_Op_ptr_from_Vertex(current_v);
74  // if current vertex is a multiqubit gate
75  if (circ.n_in_edges_of_type(current_v, EdgeType::Quantum) > 1) {
76  while (circ.n_in_edges_of_type(prev_v, EdgeType::Quantum) == 1 &&
77  ends_commute(circ, current_e)) {
78  // subsequent op on qubit path is a single qubit gate
79  // and commutes with current multi qubit gate
80  success = true;
81  EdgeVec rewire_edges;
82  op_signature_t edge_types;
83  for (const Edge &e : circ.get_in_edges(prev_v)) {
84  EdgeType type = circ.get_edgetype(e);
85  Edge boundary_edge;
86  // Currently, only purely-quantum operations can be commuted
87  // through. This is guaranteed by `ends_commute`. It follows that
88  // any wire out of `prev_v` must be EdgeType::Quantum.
89  TKET_ASSERT(type == EdgeType::Quantum);
90  boundary_edge = circ.get_last_edge(current_v, current_e);
91  rewire_edges.push_back(boundary_edge);
92  edge_types.push_back(type);
93  }
94  const port_t backup_port = circ.get_source_port(current_e);
95  circ.remove_vertex(
96  prev_v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
97  circ.rewire(prev_v, rewire_edges, edge_types);
98  current_e = circ.get_nth_out_edge(current_v, backup_port);
99  prev_v = circ.target(current_e);
100  }
101  }
102  // move to next vertex (towards input)
103  prev_v = current_v;
104  std::tie(current_v, current_e) = circ.get_prev_pair(current_v, current_e);
105  }
106  }
107 
108  return success;
109 }

122 static bool replace_two_qubit_interaction(
123  Circuit &circ, Interaction &i, std::map<Qubit, Edge> &current_edges,
124  VertexList &bin, OpType target, double cx_fidelity, bool allow_swaps) {
125  EdgeVec in_edges = {i.e0, i.e1};
126  EdgeVec out_edges = {current_edges[i.q0], current_edges[i.q1]};
127  Edge next0, next1;
128  bool q0_is_out = is_final_q_type(
129  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q0])));
130  bool q1_is_out = is_final_q_type(
131  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q1])));
132  if (!q0_is_out) {
133  next0 = circ.get_next_edge(
134  circ.target(current_edges[i.q0]), current_edges[i.q0]);
135  }
136  if (!q1_is_out) {
137  next1 = circ.get_next_edge(
138  circ.target(current_edges[i.q1]), current_edges[i.q1]);
139  }
140  // Circuit to (potentially) substitute
141  Subcircuit sub = {in_edges, out_edges, i.vertices};
142  Circuit subc = circ.subcircuit(sub);
143 
144  // Try to simplify using KAK
145  Circuit replacement = subc;
146  decompose_multi_qubits_TK2().apply(replacement);
147  Eigen::Matrix4cd mat = get_matrix_from_2qb_circ(replacement);
148  replacement = two_qubit_canonical(mat);
149  TwoQbFidelities fid;
150  fid.CX_fidelity = cx_fidelity;
151  if (target != OpType::TK2) {
152  decompose_TK2(fid, allow_swaps).apply(replacement);
153  }
154 
155  // Whether to substitute old circuit with new
156  bool substitute = false;
157  for (Vertex v : subc.vertices_in_order()) {
158  unsigned n_ins = subc.n_in_edges_of_type(v, EdgeType::Quantum);
159  if (n_ins == 2 && subc.get_OpType_from_Vertex(v) != target) {
160  // Old circuit has non-target gates => we need to substitute
161  substitute = true;
162  break;
163  }
164  }
165  if (!substitute) {
166  if (target == OpType::CX) {
167  unsigned nb_2qb_old = subc.count_gates(target);
168  unsigned nb_2qb_new = replacement.count_gates(target);
169  substitute |= nb_2qb_new < nb_2qb_old;
170  } else if (target == OpType::TK2) {
171  unsigned cnt_2qb = 0;
172  for (Vertex v : subc.vertices_in_order()) {
173  unsigned n_ins = subc.n_in_edges_of_type(v, EdgeType::Quantum);
174  cnt_2qb += n_ins == 2;
175  }
176  substitute |= cnt_2qb >= 2;
177  }
178  }
179 
180  if (substitute) {
181  // Substitute interaction with new circuit
182  bin.insert(bin.end(), sub.verts.begin(), sub.verts.end());
183  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
184  if (!q0_is_out) {
185  current_edges[i.q0] = circ.get_last_edge(circ.source(next0), next0);
186  }
187  if (!q1_is_out) {
188  current_edges[i.q1] = circ.get_last_edge(circ.source(next1), next1);
189  }
190  return true;
191  } else {
192  // Leave circuit untouched
193  return false;
194  }
195 }

197 Transform commute_and_combine_HQS2() {
198  return Transform([](Circuit &circ) {
199  bool success = false;
200  VertexList bin;
201  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
202  EdgeVec outs = circ.get_all_out_edges(v);
203  if (circ.get_OpType_from_Vertex(v) == OpType::ZZMax && outs.size() == 2) {
204  Vertex next0 = boost::target(outs[0], circ.dag);
205  Vertex next1 = boost::target(outs[1], circ.dag);
206  if (next0 == next1 &&
207  circ.get_OpType_from_Vertex(next0) == OpType::ZZMax) {
208  success = true;
209  EdgeVec h_in = circ.get_in_edges(v);
210  EdgeVec h_out = circ.get_all_out_edges(next0);
211  if (circ.get_target_port(outs[0]) != 0) {
212  h_out = {h_out[1], h_out[0]};
213  }
214  bin.push_back(v);
215  bin.push_back(next0);
216  Subcircuit sub = {h_in, h_out};
217  circ.substitute(
218  CircPool::two_Rz1(), sub, Circuit::VertexDeletion::No);
219  circ.add_phase(0.5);
220  continue;
221  }
222  if (circ.get_OpType_from_Vertex(next0) == OpType::Rz) {
223  success = true;
224  circ.remove_vertex(
225  next0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
226  Edge in_0 = circ.get_nth_in_edge(v, 0);
227  circ.rewire(next0, {in_0}, {EdgeType::Quantum});
228  }
229  if (circ.get_OpType_from_Vertex(next1) == OpType::Rz) {
230  success = true;
231  circ.remove_vertex(
232  next1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
233  Edge in_1 = circ.get_nth_in_edge(v, 1);
234  circ.rewire(next1, {in_1}, {EdgeType::Quantum});
235  }
236  }
237  }
238  circ.remove_vertices(
239  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
240  return success;
241  });

248 Transform two_qubit_squash(
249  OpType target_2qb_gate, double cx_fidelity, bool allow_swaps) {
250  const std::set<OpType> accepted_ots{OpType::CX, OpType::TK2};
251  if (!accepted_ots.contains(target_2qb_gate)) {
252  throw BadOpType(
253  "KAKDecomposition currently supports CX and TK2. "
254  "Cannot decompose to",
255  target_2qb_gate);
256  }
257  if (cx_fidelity < 0 || cx_fidelity > 1) {
258  throw std::invalid_argument("The CX fidelity must be between 0 and 1.");
259  }
260 
261  return Transform([target_2qb_gate, cx_fidelity, allow_swaps](Circuit &circ) {
262  bool success = false;
263  VertexList bin;
264  // Get map from vertex/port to qubit number
265  std::map<VertPort, Qubit> v_to_qb;
266  std::map<Qubit, Edge> current_edge_on_qb;
267  std::vector<Interaction> i_vec;
268  std::map<Qubit, int> current_interaction;
269  for (const Qubit &qb : circ.all_qubits()) {
270  for (const VertPort &vp : circ.unit_path(qb)) {
271  v_to_qb.insert({vp, qb});
272  }
273  Vertex input = circ.get_in(qb);
274  Edge e = circ.get_nth_out_edge(input, 0);
275  current_edge_on_qb[qb] = e;
276  current_interaction[qb] = -1;
277  }
278  SliceVec slices = circ.get_slices();
279  slices.insert(slices.begin(), circ.q_inputs());
280  slices.push_back(circ.q_outputs());
281  for (SliceVec::iterator s = slices.begin(); s != slices.end(); ++s) {
282  for (Slice::iterator v = s->begin(); v != s->end(); ++v) {
283  const Op_ptr o = circ.get_Op_ptr_from_Vertex(*v);
284  OpType type = o->get_type();
285  unsigned n_ins = circ.n_in_edges_of_type(*v, EdgeType::Quantum);
286  // Ignore classical ops
287  if (is_classical_type(type)) {
288  continue;
289  } else if (
290  is_projective_type(type) || is_final_q_type(type) ||
291  type == OpType::Barrier || type == OpType::Conditional ||
292  n_ins > 2 || !o->free_symbols().empty()) {
293  // Measures, resets, outputs, barriers, symbolic gates, conditionals
294  // and many-qubit gates close interactions
295  EdgeVec q_edges = circ.get_in_edges_of_type(*v, EdgeType::Quantum);
296  std::vector<port_t> q_ports;
297  for (const Edge &e : q_edges) {
298  q_ports.push_back(circ.get_target_port(e));
299  }
300  for (const port_t &port : q_ports) {
301  Qubit q = v_to_qb.at({*v, port});
302  int i = current_interaction[q];
303  if (i != -1) {
304  if (i_vec[i].count >= 2) {
305  // Replace subcircuit
306  success |= replace_two_qubit_interaction(
307  circ, i_vec[i], current_edge_on_qb, bin, target_2qb_gate,
308  cx_fidelity, allow_swaps);
309  }
310  current_interaction[i_vec[i].q0] = -1;
311  current_interaction[i_vec[i].q1] = -1;
312  }
313  if (!is_final_q_type(type)) {
314  current_edge_on_qb[q] =
315  circ.get_next_edge(*v, current_edge_on_qb[q]);
316  }
317  }
318  } else if (circ.n_in_edges_of_type(*v, EdgeType::Quantum) == 2) {
319  // Check for 2qb gate
320  Qubit q0 = v_to_qb.at({*v, 0});
321  Qubit q1 = v_to_qb.at({*v, 1});
322  int i0 = current_interaction[q0];
323  int i1 = current_interaction[q1];
324  // If they are already interacting, extend it
325  if (i0 != -1 && i0 == i1) {
326  i_vec[i0].count++;
327  i_vec[i0].vertices.insert(*v);
328  current_edge_on_qb[q0] =
329  circ.get_next_edge(*v, current_edge_on_qb[q0]);
330  current_edge_on_qb[q1] =
331  circ.get_next_edge(*v, current_edge_on_qb[q1]);
332  } else {
333  // End any other interactions on q0
334  if (i0 != -1) {
335  if (i_vec[i0].count >= 2) {
336  // Replace subcircuit
337  success |= replace_two_qubit_interaction(
338  circ, i_vec[i0], current_edge_on_qb, bin, target_2qb_gate,
339  cx_fidelity, allow_swaps);
340  }
341  current_interaction[i_vec[i0].q0] = -1;
342  current_interaction[i_vec[i0].q1] = -1;
343  }
344  // End any other interactions on q1
345  if (i1 != -1) {
346  if (i_vec[i1].count >= 2) {
347  // Replace subcircuit
348 
349  success |= replace_two_qubit_interaction(
350  circ, i_vec[i1], current_edge_on_qb, bin, target_2qb_gate,
351  cx_fidelity, allow_swaps);
352  }
353  current_interaction[i_vec[i1].q0] = -1;
354  current_interaction[i_vec[i1].q1] = -1;
355  }
356  // Add new interaction
357  Interaction new_i(q0, q1);
358  new_i.e0 = current_edge_on_qb[q0];
359  new_i.e1 = current_edge_on_qb[q1];
360  new_i.count = 1;
361  new_i.vertices = {*v};
362  current_interaction[q0] = i_vec.size();
363  current_interaction[q1] = i_vec.size();
364  i_vec.push_back(new_i);
365  current_edge_on_qb[q0] =
366  circ.get_next_edge(*v, current_edge_on_qb[q0]);
367  current_edge_on_qb[q1] =
368  circ.get_next_edge(*v, current_edge_on_qb[q1]);
369  }
370  } else {
371  // We don't care about single-qubit vertices, so just update edges
372  // and add vertices if interactions exist
373  for (port_t i = 0; i < circ.n_in_edges(*v); i++) {
374  Qubit q = v_to_qb.at({*v, i});
375  current_edge_on_qb[q] =
376  circ.get_next_edge(*v, current_edge_on_qb[q]);
377  int inter = current_interaction[q];
378  if (inter != -1) {
379  i_vec[inter].vertices.insert(*v);
380  }
381  }
382  }
383  }
384  }
385  circ.remove_vertices(
386  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
387 
388  if (success) {
389  squash_1qb_to_tk1().apply(circ);
390  }
391  return success;
392  });
393 }

395 // Given a 'SWAP_chain', finds edge in chain (or qubit wire) with best fidelity
396 // and rewires the associated single qubit vertex in to it
397 static bool find_edge_rewire_vertex(
398  Circuit &circ,
399  std::pair<std::vector<std::pair<Edge, double>>, Vertex> &entry) {
400  std::vector<std::pair<Edge, double>> candidates = entry.first;
401  std::pair<Edge, double> best_pair = candidates[0];
402  bool do_rewire = false; // prevents rewiring if current edge is best edge
403  // (causes many issues...)
404  for (unsigned i = 1; i < candidates.size(); i++) { // find best edge
405  if (candidates[i].second > best_pair.second) {
406  best_pair = candidates[i];
407  do_rewire = true;
408  }
409  }
410  if (do_rewire) {
411  // rewire in to best edge
412  circ.remove_vertex(
413  entry.second, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
414  circ.rewire(entry.second, {best_pair.first}, {EdgeType::Quantum});
415  }
416  return do_rewire;
417 }

419 // Given a SWAP vertex, which has some predecessor SWAP vertex, find the
420 // 'SWAP_chain' this predecessor SWAP vertex is in and add to it
421 static void extend_SWAP_chain(
422  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
423  &swap_chains,
424  Edge entry_edge, Node entry_node, const Edge &match, const Circuit &circ,
425  const DeviceCharacterisation &characterisation) {
426  for (auto it = swap_chains.begin(); it != swap_chains.end(); ++it) {
427  if ((*it).first.back().first == match) {
428  // extend chain, adding a new pair of edge and double to the end
429  (*it).first.push_back(
430  {entry_edge,
431  1.0 - characterisation.get_error(
432  entry_node, circ.get_OpType_from_Vertex((*it).second))});
433  return;
434  }
435  }
436 }

438 // Finds sequences of adjacent SWAP gates, with a predecessor Single Qubit
439 // Vertex. The error rate of the required Single Qubit Vertex is stored for each
440 // of the PhysicalQubits the LogicalQubit passes through Once 'SWAP_chains' are
441 // found throughout the whole circuit, predecessor Single Qubit Vertices are
442 // rewired into the edge with best error rate
443 static bool find_rewire_sq(
444  Circuit &circ, const DeviceCharacterisation &characterisation) {
445  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
446  swap_chains;
447  for (auto it = circ.begin(); it != circ.end(); ++it) {
448  if (it->get_op_ptr()->get_type() == OpType::SWAP) {
449  // find SWAP, if either predecessor is a single qubit unitary
450  // find resulting swap chain...
451  Vertex swap_vert = it.get_vertex();
452  unit_vector_t qubits = it->get_args();
453  node_vector_t nodes = {qubits.begin(), qubits.end()};
454  VertexVec pred_verts = circ.get_predecessors(swap_vert);
455  EdgeVec pred_edges = circ.get_in_edges(swap_vert);
456  EdgeVec post_edges = circ.get_all_out_edges(swap_vert);
457  for (unsigned i = 0; i < pred_verts.size(); i++) {
458  OpType optype = circ.get_OpType_from_Vertex(
459  pred_verts[i]); // OPT IS PREDECESSOR OPTYPE
460  if (circ.detect_singleq_unitary_op(pred_verts[i])) {
461  // wire has single qubit unitary->add a new swap chain
462  std::vector<std::pair<Edge, double>> swap_chain = {
463  {pred_edges[i],
464  1.0 - characterisation.get_error(nodes[i], optype)},
465  {post_edges[1 - i],
466  1.0 - characterisation.get_error(nodes[1 - i], optype)}};
467  swap_chains.push_back(std::make_pair(swap_chain, pred_verts[i]));
468  } else if (optype == OpType::SWAP) { // wire has swap->assume this swap
469  // already in chain, find chain
470  Edge pre_edge = pred_edges[i];
471  extend_SWAP_chain(
472  swap_chains, post_edges[1 - i], nodes[1 - i], pre_edge, circ,
473  characterisation);
474  }
475  }
476  }
477  }
478  // having produced swap chains, now find best qubit for gate to act on and
479  // implement
480  bool success = false;
481  while (!swap_chains.empty()) {
482  if (find_edge_rewire_vertex(circ, (*swap_chains.begin())) == true) {
483  success = true;
484  }
485  swap_chains.erase(swap_chains.begin());
486  }
487  return success;
488 }

510 Transform absorb_Rz_NPhasedX() {
511  return Transform([](Circuit &circ) {
512  bool success = false;
513  VertexSet all_bins;
514 
515  // Start by squashing Rz gates
516  success |= squash_1qb_to_pqp(OpType::Rz, OpType::Rx).apply(circ);
517 
518  // Loop through all NPhasedX gates
519  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
520  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
521  if (op->get_type() == OpType::NPhasedX) {
522  // gather surrounding Rz gates
523  unsigned arity = op->n_qubits();
524  std::vector<Expr> in_rz(arity);
525  std::vector<Expr> out_rz(arity);
526  EdgeVec in_edges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
527  EdgeVec out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
528  TKET_ASSERT(in_edges.size() == arity);
529  TKET_ASSERT(out_edges.size() == arity);
530  for (unsigned i = 0; i < arity; ++i) {
531  Vertex in_v = circ.source(in_edges[i]);
532  Op_ptr in_op = circ.get_Op_ptr_from_Vertex(in_v);
533  Vertex out_v = circ.target(out_edges[i]);
534  Op_ptr out_op = circ.get_Op_ptr_from_Vertex(out_v);
535 
536  if (in_op->get_type() == OpType::Rz) {
537  in_rz[i] = -in_op->get_params().at(0);
538  } else {
539  in_rz[i] = 0.;
540  }
541  if (out_op->get_type() == OpType::Rz) {
542  out_rz[i] = out_op->get_params().at(0);
543  } else {
544  out_rz[i] = 0.;
545  }
546  }
547 
548  // Find out which Rz angle is most popular.
549  // Note that we only compare expr[i] with expr[j] when j < i. This means
550  // that only the largest i from a set of equivalent exprs will have the
551  // right occurence count, but that is good enough.
552  std::vector<Expr> all_rz = in_rz;
553  all_rz.insert(all_rz.end(), out_rz.begin(), out_rz.end());
554  std::vector<unsigned> occurences_count(2 * arity);
555  for (unsigned i = 0; i < 2 * arity; ++i) {
556  unsigned cnt = 0;
557  for (unsigned j = 0; j < i; ++j) {
558  if (equiv_expr(all_rz[i], all_rz[j], 4)) {
559  ++cnt;
560  }
561  }
562  occurences_count[i] = cnt;
563  }
564  unsigned max_i =
565  std::max_element(occurences_count.begin(), occurences_count.end()) -
566  occurences_count.begin();
567  Expr absorb_rz = all_rz[max_i];
568 
569  if (!equiv_0(absorb_rz, 4)) {
570  success = true;
571 
572  // Subtract absorb_rz in NPhasedX
573  std::vector<Expr> new_params = op->get_params();
574  TKET_ASSERT(new_params.size() == 2);
575  new_params[1] += absorb_rz;
576  circ.dag[v] = get_op_ptr(OpType::NPhasedX, new_params, arity);
577 
578  // Finally, adjust +-absorb_rz in Rz everywhere around
579  for (unsigned i = 0; i < arity; ++i) {
580  Vertex in_v = circ.source(in_edges[i]);
581  Op_ptr in_op = circ.get_Op_ptr_from_Vertex(in_v);
582  Vertex out_v = circ.target(out_edges[i]);
583  Op_ptr out_op = circ.get_Op_ptr_from_Vertex(out_v);
584 
585  Expr angle;
586  Edge in_e, out_e;
587  VertexSet bin;
588  if (in_op->get_type() == OpType::Rz) {
589  angle = in_op->get_params().at(0) + absorb_rz;
590  out_e = in_edges[i];
591  in_e = circ.get_last_edge(in_v, out_e);
592  bin = {in_v};
593  } else {
594  angle = absorb_rz;
595  out_e = in_edges[i];
596  in_e = out_e;
597  bin = {};
598  }
599  Subcircuit sub{{in_e}, {out_e}, bin};
600  Circuit c(1);
601  if (!equiv_0(angle, 4)) {
602  c.add_op<unsigned>(OpType::Rz, angle, {0});
603  }
604  circ.substitute(c, sub, Circuit::VertexDeletion::No);
605  all_bins.insert(bin.begin(), bin.end());
606 
607  if (out_op->get_type() == OpType::Rz) {
608  angle = out_op->get_params().at(0) - absorb_rz;
609  in_e = out_edges[i];
610  out_e = circ.get_next_edge(out_v, in_e);
611  bin = {out_v};
612  } else {
613  angle = -absorb_rz;
614  in_e = out_edges[i];
615  out_e = in_e;
616  bin = {};
617  }
618  sub = Subcircuit{{in_e}, {out_e}, bin};
619  c = Circuit(1);
620  if (!equiv_0(angle, 4)) {
621  c.add_op<unsigned>(OpType::Rz, angle, {0});
622  }
623  circ.substitute(c, sub, Circuit::VertexDeletion::No);
624  all_bins.insert(bin.begin(), bin.end());
625  }
626  }
627  }
628  }
629  circ.remove_vertices(
630  all_bins, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
631 
632  return success;
633  });
634 }

706 Transform round_angles(unsigned n, bool only_zeros) {
707  if (n >= 32) {
708  throw std::invalid_argument("Precision parameter must be less than 32.");
709  }
710  return Transform([n, only_zeros](Circuit &circ) {
711  bool changed = false;
712  VertexSet bin;
713  const unsigned pow2n = 1 << n;
714  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
715  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
716  bool conditional = op->get_type() == OpType::Conditional;
717  if (conditional) {
718  const Conditional &cond = static_cast<const Conditional &>(*op);
719  op = cond.get_op();
720  }
721  if (op->get_desc().is_gate()) {
722  std::vector<Expr> params = op->get_params();
723  if (!params.empty()) {
724  std::vector<Expr> new_params;
725  for (const Expr &e : params) {
726  std::optional<double> eval = eval_expr(e);
727  if (eval) {
728  if (only_zeros) {
729  new_params.push_back(
730  pow2n * std::abs(eval.value()) < 0.5 ? 0. : e);
731  } else {
732  new_params.push_back(nearbyint(pow2n * eval.value()) / pow2n);
733  }
734  } else {
735  new_params.push_back(e);
736  }
737  }
738  if (params != new_params) {
739  unsigned n_qb = op->n_qubits();
740  Circuit replacement(n_qb);
741  if (std::any_of(
742  new_params.begin(), new_params.end(),
743  [](const Expr &e) { return (e != 0.); })) {
744  std::vector<Qubit> args;
745  for (unsigned i = 0; i < n_qb; i++) {
746  args.push_back(Qubit(i));
747  }
748  replacement.add_op(op->get_type(), new_params, args);
749  }
750  if (conditional) {
751  circ.substitute_conditional(
752  replacement, v, Circuit::VertexDeletion::No);
753  } else {
754  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
755  }
756  bin.insert(v);
757  changed = true;
758  }
759  }
760  }
761  }
762 
763  circ.remove_vertices(
764  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
765 
766  return changed;
767  });
768 }
