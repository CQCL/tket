39 std::vector<std::pair<Node, Node>> BestTsaWithArch::get_swaps(
40  const Architecture& architecture, const NodeMapping& node_mapping) {
41  std::vector<std::pair<Node, Node>> swaps;
42  // Before all the conversion and object construction,
43  // doesn't take long to check if it's actually trivial
44  bool trivial = true;
45  for (const auto& entry : node_mapping) {
46  if (entry.first != entry.second) {
47  trivial = false;
48  break;
49  }
50  }
51  if (trivial) {
52  return swaps;
53  }
54  // Now convert the Nodes into raw vertices for use in TSA objects.
55  const ArchitectureMapping arch_mapping(architecture);
56  VertexMapping vertex_mapping;
57  for (const auto& node_entry : node_mapping) {
58  vertex_mapping[arch_mapping.get_vertex(node_entry.first)] =
59  arch_mapping.get_vertex(node_entry.second);
60  }
61  TKET_ASSERT(vertex_mapping.size() == node_mapping.size());
62  check_mapping(vertex_mapping);
63 
64  SwapList raw_swap_list;
65  BestTsaWithArch::append_solution(raw_swap_list, vertex_mapping, arch_mapping);
66 
67  // Finally, convert the raw swaps back to nodes.
68  swaps.reserve(raw_swap_list.size());
69  for (auto id_opt = raw_swap_list.front_id(); id_opt;
70  id_opt = raw_swap_list.next(id_opt.value())) {
71  const auto& raw_swap = raw_swap_list.at(id_opt.value());
72  swaps.emplace_back(std::make_pair(
73  arch_mapping.get_node(raw_swap.first),
74  arch_mapping.get_node(raw_swap.second)));
75  }
76  return swaps;
77 }
