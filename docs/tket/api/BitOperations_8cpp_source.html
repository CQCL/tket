24 // We want to push back to ExpansionData, but check if it can be combined
25 // into an existing element.
26 static void push_back(
27  ExpansionData& result, SimUInt single_bit, unsigned left_shift_argument) {
28  if (!result.empty() && left_shift_argument == result.back().second) {
29  // Since the left shift arguments agree,
30  // they can be combined into a single operation,
31  // which is more efficient.
32  result.back().first |= single_bit;
33  return;
34  }
35  result.emplace_back(single_bit, left_shift_argument);
36 }

38 ExpansionData get_expansion_data(
39  SimUInt forbidden_bits, unsigned number_of_free_bits) {
40  ExpansionData result;
41  SimUInt next_bit = 1;
42 
43  for (unsigned bits_count = 0; bits_count < number_of_free_bits;
44  ++bits_count) {
45  auto test_bit = next_bit;
46  for (unsigned left_shift_arg = 0;; ++left_shift_arg) {
47  if ((test_bit & forbidden_bits) == 0) {
48  TKET_ASSERT(test_bit != 0);
49  // A free space has been found.
50  push_back(result, next_bit, left_shift_arg);
51  forbidden_bits |= test_bit;
52  break;
53  }
54  test_bit <<= 1;
55  }
56  next_bit <<= 1;
57  }
58  return result;
59 }

61 SimUInt get_expanded_bits(const ExpansionData& expansion_data, SimUInt bits) {
62  SimUInt result = 0;
63  for (const auto& entry : expansion_data) {
64  SimUInt block = bits & entry.first;
65  block <<= entry.second;
66  result |= block;
67  }
68  return result;
69 }
