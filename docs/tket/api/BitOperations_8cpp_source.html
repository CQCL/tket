23// We want to push back to ExpansionData, but check if it can be combined
24// into an existing element.
25static void push_back(
26 ExpansionData& result, SimUInt single_bit, unsigned left_shift_argument) {
27 if (!result.empty() && left_shift_argument == result.back().second) {
28 // Since the left shift arguments agree,
29 // they can be combined into a single operation,
30 // which is more efficient.
31 result.back().first |= single_bit;
32 return;
33 }
34 result.emplace_back(single_bit, left_shift_argument);
35}

37ExpansionData get_expansion_data(
38 SimUInt forbidden_bits, unsigned number_of_free_bits) {
39 ExpansionData result;
40 SimUInt next_bit = 1;
41
42 for (unsigned bits_count = 0; bits_count < number_of_free_bits;
43 ++bits_count) {
44 auto test_bit = next_bit;
45 for (unsigned left_shift_arg = 0;; ++left_shift_arg) {
46 if ((test_bit & forbidden_bits) == 0) {
47 TKET_ASSERT(test_bit != 0);
48 // A free space has been found.
49 push_back(result, next_bit, left_shift_arg);
50 forbidden_bits |= test_bit;
51 break;
52 }
53 test_bit <<= 1;
54 }
55 next_bit <<= 1;
56 }
57 return result;
58}

60SimUInt get_expanded_bits(const ExpansionData& expansion_data, SimUInt bits) {
61 SimUInt result = 0;
62 for (const auto& entry : expansion_data) {
63 SimUInt block = bits & entry.first;
64 block <<= entry.second;
65 result |= block;
66 }
67 return result;
68}
