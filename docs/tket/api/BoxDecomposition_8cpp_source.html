25bool BoxDecomposition::solve() {
26 // Box type vertices are later removed from DAG
27 VertexList bin;
28 bool modified = false;
29 std::shared_ptr<unit_frontier_t> frontier_edges =
30 frontier_convert_vertport_to_edge(
31 this->mapping_frontier_->circuit_,
32 this->mapping_frontier_->linear_boundary);
33 CutFrontier next_cut =
34 this->mapping_frontier_->circuit_.next_q_cut(frontier_edges);
35 for (Vertex &vert : *next_cut.slice) {
36 Op_ptr op = this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(vert);
37 while (op->get_type() == OpType::Conditional) {
38 op = static_cast<const Conditional &>(*op).get_op();
39 }
40 if (op->get_desc().is_box()) {
41 if (this->mapping_frontier_->circuit_.substitute_box_vertex(
42 vert, Circuit::VertexDeletion::No)) {
43 modified = true;
44 bin.push_back(vert);
45 }
46 }
47 }
48 if (!modified) {
49 return false;
50 }
51 // Delete vertices
52 this->mapping_frontier_->circuit_.remove_vertices(
53 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
54 return true;
55}
