25 bool BoxDecomposition::solve() {
26  // Box type vertices are later removed from DAG
27  VertexList bin;
28  bool modified = false;
29  std::shared_ptr<unit_frontier_t> frontier_edges =
30  frontier_convert_vertport_to_edge(
31  this->mapping_frontier_->circuit_,
32  this->mapping_frontier_->linear_boundary);
33  CutFrontier next_cut =
34  this->mapping_frontier_->circuit_.next_q_cut(frontier_edges);
35  for (Vertex &vert : *next_cut.slice) {
36  Op_ptr op = this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(vert);
37  if (op->get_desc().is_box() ||
38  (op->get_type() == OpType::Conditional &&
39  static_cast<const Conditional &>(*op).get_op()->get_desc().is_box())) {
40  if (this->mapping_frontier_->circuit_.substitute_box_vertex(
41  vert, Circuit::VertexDeletion::No)) {
42  modified = true;
43  bin.push_back(vert);
44  }
45  }
46  }
47  if (!modified) {
48  return false;
49  }
50  // Delete vertices
51  this->mapping_frontier_->circuit_.remove_vertices(
52  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
53  return true;
54 }
