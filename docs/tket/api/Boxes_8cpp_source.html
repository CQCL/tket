74CircBox::CircBox(const Circuit &circ) : Box(OpType::CircBox) {
75 try {
76 Circuit circ1 = circ;
77 circ1.flatten_registers();
78 } catch (const std::exception &e) {
79 std::stringstream ss;
80 ss << "Unable to construct CircBox: " << e.what();
81 throw std::runtime_error(ss.str());
82 }
83 signature_ = op_signature_t(circ.n_qubits(), EdgeType::Quantum);
84 op_signature_t bits(circ.n_bits(), EdgeType::Classical);
85 signature_.insert(signature_.end(), bits.begin(), bits.end());
86 circ_ = std::make_shared<Circuit>(circ);
87}

95bool CircBox::is_clifford() const {
96 BGL_FORALL_VERTICES(v, circ_->dag, DAG) {
97 Op_ptr op = circ_->get_Op_ptr_from_Vertex(v);
98 if (op->get_desc().is_meta()) continue;
99 if (!op->is_clifford()) return false;
100 }
101 return true;
102}

284Circuit CompositeGateDef::instance(const std::vector<Expr> &params) const {
285 Circuit circ = *def_;
286 symbol_map_t symbol_map;
287 for (unsigned i = 0; i < params.size(); i++) {
288 symbol_map.insert({args_.at(i), params.at(i)});
289 }
290 circ.symbol_substitution(symbol_map);
291 return circ;
292}

301bool CompositeGateDef::operator==(const CompositeGateDef &other) const {
302 if (this->get_name() != other.get_name()) return false;
303 std::vector<Expr> this_args = {this->args_.begin(), this->args_.end()};
304 std::vector<Expr> other_args = {other.args_.begin(), other.args_.end()};
305 if (this_args != other_args) return false;
306 return this->get_def()->circuit_equality(*other.get_def(), {}, false);
307}

393QControlBox::QControlBox(
394 const Op_ptr &op, unsigned n_controls,
395 const std::vector<bool> &control_state)
396 : Box(OpType::QControlBox),
397 op_(op),
398 n_controls_(n_controls),
399 control_state_(
400 control_state.empty() ? std::vector<bool>(n_controls, true)
401 : control_state) {
402 // Send warnings for inner ops that do not preserve global phase.
403 if (op_->get_type() == OpType::TermSequenceBox) {
404 const auto &inner_box = static_cast<const TermSequenceBox &>(*op_);
405 if (inner_box.get_synth_strategy() == Transforms::PauliSynthStrat::Greedy) {
406 tket_log()->error(
407 "Wrapping a TermSequenceBox with the Greedy synthesis strategy in a "
408 "QControlBox may result in an incorrect circuit, as the "
409 "TermSequenceBox decomposition "
410 "does not preserve global phase.");
411 }
412 }
413 if (n_controls != control_state_.size()) {
414 throw CircuitInvalidity(
415 "The size of control_state doesn't match the argument n_controls");
416 }
417 op_signature_t inner_sig = op_->get_signature();
418 n_inner_qubits_ = inner_sig.size();
419 if (std::count(inner_sig.begin(), inner_sig.end(), EdgeType::Quantum) !=
420 n_inner_qubits_) {
421 throw BadOpType(
422 "Quantum control of classical wires not supported", op_->get_type());
423 }
424 signature_ = op_signature_t(n_controls + n_inner_qubits_, EdgeType::Quantum);
425}

442std::string QControlBox::get_command_str(const unit_vector_t &args) const {
443 std::stringstream out;
444 out << "qif (";
445 if (n_controls_ > 0) {
446 out << args.at(0).repr() << " = " << control_state_.at(0);
447 for (unsigned i = 1; i < n_controls_; ++i) {
448 out << ", " << args.at(i).repr() << " = " << control_state_.at(i);
449 }
450 }
451 unit_vector_t inner_args(args.begin() + n_controls_, args.end());
452 out << ") " << op_->get_command_str(inner_args);
453 return out.str();
454}

456void QControlBox::generate_circuit() const {
457 Circuit c(n_inner_qubits_);
458 std::vector<unsigned> qbs(n_inner_qubits_);
459 std::iota(qbs.begin(), qbs.end(), 0);
460 c.add_op(op_, qbs);
461 // ConjugationBoxes will be handled by with_controls
462 c.decompose_boxes_recursively({OpType::ConjugationBox});
463 Circuit x_circ(n_controls_ + n_inner_qubits_);
464 for (unsigned i = 0; i < n_controls_; i++) {
465 if (!control_state_.at(i)) {
466 x_circ.add_op<unsigned>(OpType::X, {i});
467 }
468 }
469 c = with_controls(c, n_controls_);
470 circ_ = std::make_shared<Circuit>(x_circ >> c >> x_circ);
471}

485std::optional<Eigen::MatrixXcd> QControlBox::get_box_unitary() const {
486 const unsigned inner_sz = 1u << n_inner_qubits_;
487 const unsigned sz = inner_sz << n_controls_;
488 Eigen::MatrixXcd u = Eigen::MatrixXcd::Identity(sz, sz);
489 unsigned long long block_pos = bin_to_dec(control_state_) * inner_sz;
490 u.block(block_pos, block_pos, inner_sz, inner_sz) = op_->get_unitary();
491 return u;
492}
