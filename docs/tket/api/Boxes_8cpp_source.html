497 ToffoliBox::ToffoliBox(
498  unsigned _n_qubits,
499  std::map<std::vector<bool>, std::vector<bool>> _permutation)
500  : Box(OpType::ToffoliBox), n_qubits_(_n_qubits) {
501  // Convert passed permutation to cycles
502  while (!_permutation.empty()) {
503  auto it = _permutation.begin();
504  cycle_permutation_t cycle = {it->first};
505  if (it->first.size() != this->n_qubits_) {
506  throw std::invalid_argument(
507  "Size of bitstring does not match number of qubits.");
508  }
509  it = _permutation.find(it->second);
510  while (it->first != cycle[0]) {
511  if (it->first.size() != this->n_qubits_) {
512  throw std::invalid_argument(
513  "Size of bitstring does not match number of qubits.");
514  }
515  cycle.push_back(it->first);
516  it = _permutation.find(it->second);
517  if (it == _permutation.end()) {
518  throw std::invalid_argument("Permutation is not complete.");
519  }
520  }
521  if (cycle.size() > 1) {
522  this->cycles_.insert(cycle);
523  }
524  for (const std::vector<bool> &bitstring : cycle) {
525  _permutation.erase(bitstring);
526  }
527  }
528 }

537 unsigned get_hamming_distance(
538  const std::vector<bool> &a, const std::vector<bool> &b) {
539  if (a.size() != b.size()) {
540  throw std::invalid_argument("Bitstrings must have identical size.");
541  }
542  unsigned counter = 0;
543  for (unsigned i = 0; i < a.size(); i++) {
544  if (a[i] != b[i]) {
545  ++counter;
546  }
547  }
548  return counter;
549 }

551 ToffoliBox::cycle_transposition_t ToffoliBox::cycle_to_transposition(
552  cycle_permutation_t cycle) const {
565  cycle_transposition_t best_transposition;
566  unsigned best_hamming_distance = 0;
567 
568  for (unsigned i = 0; i < cycle.size(); i++) {
569  unsigned accumulated_hamming_distance = 0;
570  cycle_transposition_t transposition;
571  for (unsigned j = 1; j < cycle.size(); j++) {
572  transposition.push_back({cycle[0], cycle[j], cycle[0]});
573  accumulated_hamming_distance += get_hamming_distance(cycle[0], cycle[j]);
574  }
575  if (best_transposition.empty() ||
576  accumulated_hamming_distance < best_hamming_distance) {
577  best_transposition = transposition;
578  best_hamming_distance = accumulated_hamming_distance;
579  }
580  std::rotate(cycle.begin(), cycle.begin() + 1, cycle.end());
581  }
582  return best_transposition;
583 }

585 std::vector<ToffoliBox::cycle_transposition_t> ToffoliBox::get_transpositions()
586  const {
587  std::vector<ToffoliBox::cycle_transposition_t> transpositions;
588  for (const cycle_permutation_t &cycle : this->cycles_) {
589  // each cycle is costed via the Hamming distance to reduce the number of
590  // operations
591  transpositions.push_back(this->cycle_to_transposition(cycle));
592  }
593  return transpositions;
594 }

596 Circuit ToffoliBox::get_bitstring_circuit(
597  const std::vector<bool> &bitstring, const unsigned &target) const {
598  // flip qubits that need to be state 0
599  Circuit x_circuit(this->n_qubits_);
600  std::vector<unsigned> cnx_args;
601  for (unsigned i = 0; i < this->n_qubits_; i++) {
602  if (i != target) {
603  if (!bitstring[i]) {
604  x_circuit.add_op<unsigned>(OpType::X, {i});
605  }
606  cnx_args.push_back(i);
607  }
608  }
609  cnx_args.push_back(target);
610  TKET_ASSERT(cnx_args.size() == this->n_qubits_);
611 
612  Circuit return_circuit(this->n_qubits_);
613  return_circuit.append(x_circuit);

619 ToffoliBox::gray_code_t ToffoliBox::transposition_to_gray_code(
620  const ToffoliBox::transposition_t &transposition) const {
621  unsigned first_middle_hamming_distance =
622  get_hamming_distance(transposition.first, transposition.middle);
623  unsigned middle_last_hamming_distance =
624  get_hamming_distance(transposition.middle, transposition.last);
625  ToffoliBox::gray_code_t all_gray_code_entries;
626  // => that some optimisation is done to middle_last, so must go via
627  // transposition.last bitstring to allow proper cancellation
628  // At some point transposition.first == transposition.last
629  // If transposition.last != transposition.first & the hamming distance
630  // between transposition.last and transposition.middle is smaller than
631  // transposition.first and transposition.middle,
632  // then we need to make sure that the gray code between transposition.first
633  // and transposition.middle goes via transposition.last, such that the
634  // eventual gray code between transposition.middle and transposition.last
635  // uncomputes this we can assume that the remaining
636  // transposition.last->transposition.first gray code would be cancelled out if
637  // added
638  std::vector<bool> initial = transposition.first;
639  if (middle_last_hamming_distance < first_middle_hamming_distance) {
640  // get bitstrings for first -> last
641  for (unsigned i = 0; i < transposition.first.size(); i++) {
642  if (transposition.first[i] != transposition.last[i]) {
643  initial[i] = !initial[i];
644  all_gray_code_entries.push_back({initial, i});
645  }
646  }
647  }
648  // with the right middle bitstring now guaranteed, go from this bitstring to
649  // middle
650  std::vector<bool> bitstring = initial;
651  for (unsigned i = 0; i < transposition.first.size(); i++) {
652  if (initial[i] != transposition.middle[i]) {
653  bitstring[i] = !bitstring[i];
654  all_gray_code_entries.push_back({bitstring, i});
655  }
656  }
657  // now do the last->middle in reverse to guarantee right
658  // gray code path is taken
659  initial = transposition.last;
660  ToffoliBox::gray_code_t reverse_gray_code_entries;
661  // as before, implies some optimisation has been completed on
662  // first one
663  // thus make sure that gray code goes via transposition.first
664  // to make applied permutation right
665  if (first_middle_hamming_distance < middle_last_hamming_distance) {
666  // get bitstrings for first -> last
667  for (unsigned i = 0; i < transposition.first.size(); i++) {
668  if (initial[i] != transposition.first[i]) {
669  initial[i] = !initial[i];
670  reverse_gray_code_entries.push_back({initial, i});
671  }
672  }
673  }
674  // and then go from bitstring to middle
675  bitstring = initial;
676  for (unsigned i = 0; i < transposition.middle.size(); i++) {
677  if (transposition.middle[i] != initial[i]) {
678  bitstring[i] = !bitstring[i];
679  reverse_gray_code_entries.push_back({bitstring, i});
680  }
681  }
682  // don't want to add transformation for reaching final -> so pop_back
683  if (!reverse_gray_code_entries.empty()) {
684  reverse_gray_code_entries.pop_back();
685  }
686  all_gray_code_entries.insert(
687  all_gray_code_entries.end(), reverse_gray_code_entries.rbegin(),
688  reverse_gray_code_entries.rend());
689  return all_gray_code_entries;
690 }

692 ToffoliBox::cycle_transposition_t ToffoliBox::merge_cycles(
693  std::vector<ToffoliBox::cycle_transposition_t> &cycle_transpositions)
694  const {
695  ToffoliBox::cycle_transposition_t return_transposition;
696  for (unsigned k = 0; k < cycle_transpositions.size(); k++) {
697  ToffoliBox::cycle_transposition_t cycle = cycle_transpositions[k];
698  unsigned i = 0, j = 1;
699  while (j < cycle.size()) {
700  ToffoliBox::transposition_t transposition_i = cycle[i];
701  ToffoliBox::transposition_t transposition_j = cycle[j];
702 
703  std::vector<bool> transposition_j_first = transposition_j.first;
704  std::vector<bool> transposition_i_last = transposition_i.last;
705  std::vector<bool> transposition_i_first = transposition_i.first;
706 
707  TKET_ASSERT(transposition_i_last == transposition_j.first);
708  std::vector<bool> i_middle = transposition_i.middle;
709  std::vector<bool> j_middle = transposition_j.middle;
710 
711  TKET_ASSERT(i_middle.size() == transposition_i.last.size());
712  TKET_ASSERT(j_middle.size() == transposition_i.last.size());
713  // if a transposition has already been reduced, still need to make sure we
714  // uncompute it
715  if (transposition_i_first != transposition_i_last) {
716  unsigned middle_last_distance =
717  get_hamming_distance(i_middle, transposition_i_last);
718  unsigned middle_first_distance =
719  get_hamming_distance(i_middle, transposition_i_first);
720  // this => the reduced transposition is on a good gray code between the
721  // new "first" and target
722  if (middle_first_distance < middle_last_distance &&
723  middle_first_distance > 1) {
724  transposition_i_last = transposition_i_first;
725  std::vector<bool> starting_point = transposition_i_last;
726  for (unsigned k = 0; k < i_middle.size(); k++) {
727  if (i_middle[k] == j_middle[k] &&
728  get_hamming_distance(starting_point, i_middle) > 1) {
729  starting_point[k] = i_middle[k];
730  }
731  }
732 
733  cycle_transpositions[k][i].last = starting_point;
734  cycle_transpositions[k][j].first = starting_point;
735  }
736  } else { // else in this case just find any good transposition
737  std::vector<bool> starting_point = transposition_i_last;
738  for (unsigned k = 0; k < i_middle.size(); k++) {
739  if (i_middle[k] == j_middle[k] &&
740  get_hamming_distance(starting_point, i_middle) > 1) {
741  starting_point[k] = i_middle[k];
742  }
743  }
744 
745  cycle[i].last = starting_point;
746  cycle[j].first = starting_point;
747  }
748 
749  ++i;
750  ++j;
751  }
752 
753  return_transposition.insert(
754  return_transposition.end(), cycle.begin(), cycle.end());
755  }
756 
757  return return_transposition;
758 }

765 void ToffoliBox::generate_circuit() const {
766  // This decomposition is as described on page 191, section 4.5.2 "Single
767  // qubit and CNOT gates are universal" of Nielsen & Chuang
768  std::vector<ToffoliBox::cycle_transposition_t> cycle_transpositions =
769  this->get_transpositions();
770 
771  // optionally, order the transpositions and cycles to allow gate
772  // cancellation
773  cycle_transposition_t ordered_transpositions;
774  ordered_transpositions = merge_cycles(cycle_transpositions);
775 
776  if (ordered_transpositions.empty()) {
777  this->circ_ = std::make_shared<Circuit>(this->n_qubits_);
778  return;
779  }
780 
781  // Now we have ordered transpositions, produced front->middle and
782  // middle->back gray codes for each transposition and add to circuit
783  this->circ_ = std::make_shared<Circuit>(this->n_qubits_);
784  for (const transposition_t &transposition : ordered_transpositions) {
785  TKET_ASSERT(transposition.first.size() == this->n_qubits_);
786  TKET_ASSERT(transposition.middle.size() == this->n_qubits_);
787  TKET_ASSERT(transposition.last.size() == this->n_qubits_);
788  ToffoliBox::gray_code_t all_gray_code_entries =
789  transposition_to_gray_code(transposition);
790  for (const std::pair<std::vector<bool>, unsigned> &entry :
791  all_gray_code_entries) {
792  this->circ_->append(
793  this->get_bitstring_circuit(entry.first, entry.second));
794  }
795  }
796 }
