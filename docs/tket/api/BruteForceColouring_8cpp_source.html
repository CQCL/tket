31struct BruteForceColouring::Impl {
32 struct NodeColouringData {
33 vector<std::size_t> allowed_colours;
34
35 // The index in "allowed_colours", not the actual colour
36 std::size_t current_colour_index = 0;
37
38 bool is_valid_colour() const {
39 return current_colour_index < allowed_colours.size();
40 }
41
42 std::size_t get_colour() const {
43 return allowed_colours[current_colour_index];
44 }
45 };
46
47 // This exactly mirrors the nodes in the ColouringPriority object;
48 // it is just extra data specifically related to colouring.
49 vector<NodeColouringData> colouring_data;
50
51 // KEY: is the vertex, VALUE: the colour
52 map<std::size_t, std::size_t> colours;
53
54 // Fills in the colour possibilities,
55 // possibly increasing suggested_number_of_colours.
56 // Returns false if it failed.
57 bool initial_colouring_setup(
58 const ColouringPriority& priority,
59 std::size_t& suggested_number_of_colours) {
60 const auto& initial_clique = priority.get_initial_clique();
61
62 suggested_number_of_colours =
63 std::max(suggested_number_of_colours, initial_clique.size());
64
65 const auto& nodes = priority.get_nodes();
66
67 if (suggested_number_of_colours > nodes.size()) {
68 return false;
69 }
70 colouring_data.resize(nodes.size());
71
72 for (std::size_t i = 0; i < initial_clique.size(); ++i) {
73 colouring_data[i].allowed_colours.assign(1, i);
74 colours[nodes[i].vertex] = i;
75 }
76
77 // Now, recursively keep increasing the number of colours until we at least
78 // have nonempty colour possibility lists. (It may still be impossible, of
79 // course, due to the detailed graph structure).
80
81 set<std::size_t> forbidden_colours;
82
83 for (std::size_t i = initial_clique.size(); i < nodes.size(); ++i) {
84 forbidden_colours.clear();
85
86 for (std::size_t node_index : nodes[i].earlier_neighbour_node_indices) {
87 const auto& earlier_colours =
88 colouring_data[node_index].allowed_colours;
89
90 // It is only the initial CLIQUE vertices which have fixed colours;
91 // as the number of possible colours increases, EVERY other vertex
92 // has more colour possibilities; so we must NOT think that,
93 // just because CURRENTLY a vertex has only one colour,
94 // that it will ALWAYS be that way!
95 if (initial_clique.count(nodes[node_index].vertex) != 0) {
96 TKET_ASSERT(earlier_colours.size() == 1);
97 forbidden_colours.insert(earlier_colours[0]);
98 }
99 }
100 auto& possible_colours = colouring_data[i].allowed_colours;
101 possible_colours.clear();
102 for (std::size_t col = 0; col < suggested_number_of_colours; ++col) {
103 if (forbidden_colours.count(col) == 0) {
104 possible_colours.push_back(col);
105 }
106 }
107 if (possible_colours.empty()) {
108 ++suggested_number_of_colours;
109 return initial_colouring_setup(priority, suggested_number_of_colours);
110 }
111 }
112 return true;
113 }
114
115 void fill_colour_map(const ColouringPriority& priority) {
116 const auto& nodes = priority.get_nodes();
117 for (std::size_t i = 0; i < nodes.size(); ++i) {
118 colours[nodes[i].vertex] = colouring_data[i].get_colour();
119 }
120 }
121
122 bool attempt_brute_force_colouring(const ColouringPriority& priority) {
123 for (auto& data : colouring_data) {
124 data.current_colour_index = 0;
125 }
126 const auto& nodes = priority.get_nodes();
127 const std::size_t number_of_nodes = nodes.size();
128
129 for (std::size_t current_node_index = 0;;) {
130 const auto& current_node = nodes[current_node_index];
131 auto& current_colouring_node = colouring_data[current_node_index];
132
133 if (current_colouring_node.is_valid_colour()) {
134 // We have a candidate colour, now test it for consistency.
135 const auto current_col = current_colouring_node.get_colour();
136 bool colour_is_impossible = false;
137
138 // TODO: would a set of colours be faster here?
139 for (std::size_t earlier_node_index :
140 current_node.earlier_neighbour_node_indices) {
141 if (colouring_data[earlier_node_index].get_colour() == current_col) {
142 colour_is_impossible = true;
143 break;
144 }
145 }
146 if (!colour_is_impossible) {
147 // Advance to the next node to colour.
148 ++current_node_index;
149 if (current_node_index < number_of_nodes) {
150 colouring_data[current_node_index].current_colour_index = 0;
151 continue;
152 }
153 // We've hit the end! We are finished.
154 return true;
155 }
156 ++current_colouring_node.current_colour_index;
157 continue;
158 }
159
160 // We must backtrack.
161 if (current_node_index == 0) {
162 return false;
163 }
164 --current_node_index;
165
166 // Advance the colour.
167 ++colouring_data[current_node_index].current_colour_index;
168 }
169 // We cannot actually reach here, the outer loop only returns, never breaks.
170 }
171};

175BruteForceColouring::BruteForceColouring(
176 const ColouringPriority& priority, std::size_t suggested_number_of_colours)
177 : m_pimpl(std::make_unique<BruteForceColouring::Impl>()) {
178 const auto number_of_nodes = priority.get_nodes().size();
179 if (suggested_number_of_colours >= number_of_nodes) {
180 // We've been given permission to use many colours;
181 // so just use them all!
182 for (std::size_t i = 0; i < number_of_nodes; ++i) {
183 m_pimpl->colours[priority.get_nodes()[i].vertex] = i;
184 }
185 return;
186 }
187
188 const auto initial_suggested_number_of_colours = suggested_number_of_colours;
189
190 try {
191 if (!m_pimpl->initial_colouring_setup(
192 priority, suggested_number_of_colours)) {
193 throw std::runtime_error("initial_colouring_setup failed");
194 }
195
196 // From now on, every time we fail to colour with the specified number,
197 // we simply have to add the extra colour onto
198 // each list of possible colours;
199 // no need to redo "initial_colouring_setup".
200
201 for (; suggested_number_of_colours <= number_of_nodes;
202 ++suggested_number_of_colours) {
203 if (m_pimpl->attempt_brute_force_colouring(priority)) {
204 // We've succeeded!
205 m_pimpl->fill_colour_map(priority);
206 return;
207 }
208 // It's impossible with this number of colours,
209 // so try again with one more.
210 // If we were really fancy we might consider
211 // a binary search on the number of colours.
212 for (std::size_t i = priority.get_initial_clique().size();
213 i < number_of_nodes; ++i) {
214 m_pimpl->colouring_data[i].allowed_colours.push_back(
215 suggested_number_of_colours);
216 }
217 }
218 throw std::runtime_error("suggested_number_of_colours hit number_of_nodes");
219 } catch (const std::exception& e) {
220 // GCOVR_EXCL_START
221 TKET_ASSERT(
222 AssertMessage() << "initial_suggested_number_of_colours = "
223 << initial_suggested_number_of_colours
224 << ", reached suggested_number_of_colours = "
225 << suggested_number_of_colours << ", had "
226 << number_of_nodes << " nodes. Error: " << e.what()
227 << priority.print_raw_data());
228 // GCOVR_EXCL_STOP
229 }
230}
