32 struct BruteForceColouring::Impl {
33  struct NodeColouringData {
34  vector<size_t> allowed_colours;
35 
36  // The index in "allowed_colours", not the actual colour
37  size_t current_colour_index = 0;
38 
39  bool is_valid_colour() const {
40  return current_colour_index < allowed_colours.size();
41  }
42 
43  size_t get_colour() const { return allowed_colours[current_colour_index]; }
44  };
45 
46  // This exactly mirrors the nodes in the ColouringPriority object;
47  // it is just extra data specifically related to colouring.
48  vector<NodeColouringData> colouring_data;
49 
50  // KEY: is the vertex, VALUE: the colour
51  map<size_t, size_t> colours;
52 
53  // Fills in the colour possibilities,
54  // possibly increasing suggested_number_of_colours.
55  // Returns false if it failed.
56  bool initial_colouring_setup(
57  const ColouringPriority& priority, size_t& suggested_number_of_colours) {
58  const auto& initial_clique = priority.get_initial_clique();
59 
60  suggested_number_of_colours =
61  std::max(suggested_number_of_colours, initial_clique.size());
62 
63  const auto& nodes = priority.get_nodes();
64 
65  if (suggested_number_of_colours > nodes.size()) {
66  return false;
67  }
68  colouring_data.resize(nodes.size());
69 
70  for (size_t i = 0; i < initial_clique.size(); ++i) {
71  colouring_data[i].allowed_colours.assign(1, i);
72  colours[nodes[i].vertex] = i;
73  }
74 
75  // Now, recursively keep increasing the number of colours until we at least
76  // have nonempty colour possibility lists. (It may still be impossible, of
77  // course, due to the detailed graph structure).
78 
79  set<size_t> forbidden_colours;
80 
81  for (size_t i = initial_clique.size(); i < nodes.size(); ++i) {
82  forbidden_colours.clear();
83 
84  for (size_t node_index : nodes[i].earlier_neighbour_node_indices) {
85  const auto& earlier_colours =
86  colouring_data[node_index].allowed_colours;
87 
88  // It is only the initial CLIQUE vertices which have fixed colours;
89  // as the number of possible colours increases, EVERY other vertex
90  // has more colour possibilities; so we must NOT think that,
91  // just because CURRENTLY a vertex has only one colour,
92  // that it will ALWAYS be that way!
93  if (initial_clique.count(nodes[node_index].vertex) != 0) {
94  TKET_ASSERT(earlier_colours.size() == 1);
95  forbidden_colours.insert(earlier_colours[0]);
96  }
97  }
98  auto& possible_colours = colouring_data[i].allowed_colours;
99  possible_colours.clear();
100  for (size_t col = 0; col < suggested_number_of_colours; ++col) {
101  if (forbidden_colours.count(col) == 0) {
102  possible_colours.push_back(col);
103  }
104  }
105  if (possible_colours.empty()) {
106  ++suggested_number_of_colours;
107  return initial_colouring_setup(priority, suggested_number_of_colours);
108  }
109  }
110  return true;
111  }
112 
113  void fill_colour_map(const ColouringPriority& priority) {
114  const auto& nodes = priority.get_nodes();
115  for (size_t i = 0; i < nodes.size(); ++i) {
116  colours[nodes[i].vertex] = colouring_data[i].get_colour();
117  }
118  }
119 
120  bool attempt_brute_force_colouring(const ColouringPriority& priority) {
121  for (auto& data : colouring_data) {
122  data.current_colour_index = 0;
123  }
124  const auto& nodes = priority.get_nodes();
125  const size_t number_of_nodes = nodes.size();
126 
127  for (size_t current_node_index = 0;;) {
128  const auto& current_node = nodes[current_node_index];
129  auto& current_colouring_node = colouring_data[current_node_index];
130 
131  if (current_colouring_node.is_valid_colour()) {
132  // We have a candidate colour, now test it for consistency.
133  const auto current_col = current_colouring_node.get_colour();
134  bool colour_is_impossible = false;
135 
136  // TODO: would a set of colours be faster here?
137  for (size_t earlier_node_index :
138  current_node.earlier_neighbour_node_indices) {
139  if (colouring_data[earlier_node_index].get_colour() == current_col) {
140  colour_is_impossible = true;
141  break;
142  }
143  }
144  if (!colour_is_impossible) {
145  // Advance to the next node to colour.
146  ++current_node_index;
147  if (current_node_index < number_of_nodes) {
148  colouring_data[current_node_index].current_colour_index = 0;
149  continue;
150  }
151  // We've hit the end! We are finished.
152  return true;
153  }
154  ++current_colouring_node.current_colour_index;
155  continue;
156  }
157 
158  // We must backtrack.
159  if (current_node_index == 0) {
160  return false;
161  }
162  --current_node_index;
163 
164  // Advance the colour.
165  ++colouring_data[current_node_index].current_colour_index;
166  }
167  // We cannot actually reach here, the outer loop only returns, never breaks.
168  }
169 };

173 BruteForceColouring::BruteForceColouring(
174  const ColouringPriority& priority, size_t suggested_number_of_colours)
175  : m_pimpl(std::make_unique<BruteForceColouring::Impl>()) {
176  const auto number_of_nodes = priority.get_nodes().size();
177  if (suggested_number_of_colours >= number_of_nodes) {
178  // We've been given permission to use many colours;
179  // so just use them all!
180  for (size_t i = 0; i < number_of_nodes; ++i) {
181  m_pimpl->colours[priority.get_nodes()[i].vertex] = i;
182  }
183  return;
184  }
185 
186  const auto initial_suggested_number_of_colours = suggested_number_of_colours;
187 
188  try {
189  if (!m_pimpl->initial_colouring_setup(
190  priority, suggested_number_of_colours)) {
191  throw std::runtime_error("initial_colouring_setup failed");
192  }
193 
194  // From now on, every time we fail to colour with the specified number,
195  // we simply have to add the extra colour onto
196  // each list of possible colours;
197  // no need to redo "initial_colouring_setup".
198 
199  for (; suggested_number_of_colours <= number_of_nodes;
200  ++suggested_number_of_colours) {
201  if (m_pimpl->attempt_brute_force_colouring(priority)) {
202  // We've succeeded!
203  m_pimpl->fill_colour_map(priority);
204  return;
205  }
206  // It's impossible with this number of colours,
207  // so try again with one more.
208  // If we were really fancy we might consider
209  // a binary search on the number of colours.
210  for (size_t i = priority.get_initial_clique().size(); i < number_of_nodes;
211  ++i) {
212  m_pimpl->colouring_data[i].allowed_colours.push_back(
213  suggested_number_of_colours);
214  }
215  }
216  throw std::runtime_error("suggested_number_of_colours hit number_of_nodes");
217  } catch (const std::exception& e) {
218  // GCOVR_EXCL_START
219  TKET_ASSERT(
220  AssertMessage() << "initial_suggested_number_of_colours = "
221  << initial_suggested_number_of_colours
222  << ", reached suggested_number_of_colours = "
223  << suggested_number_of_colours << ", had "
224  << number_of_nodes << " nodes. Error: " << e.what()
225  << priority.print_raw_data());
226  // GCOVR_EXCL_STOP
227  }
228 }
