23 ChoiMixTableau circuit_to_cm_tableau(const Circuit& circ) {
24  ChoiMixTableau tab(circ.all_qubits());
25  for (const Qubit& q : circ.created_qubits()) {
26  tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
27  }
28  for (const Command& com : circ) {
29  auto args = com.get_args();
30  qubit_vector_t qbs = {args.begin(), args.end()};
31  tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
32  }
33  for (const Qubit& q : circ.discarded_qubits()) {
34  tab.discard_qubit(q);
35  }
36  return tab;
37 }

39 std::pair<Circuit, unit_map_t> cm_tableau_to_circuit(const ChoiMixTableau& t) {
78  // Operate on a copy of the tableau to track the remainder as we extract gates
79  ChoiMixTableau tab(t);
80 
81  // Canonicalise tableau and perform output-first gaussian elimination to
82  // isolate post-selected subspace in lower rows
83  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
84  tab.gaussian_form();
85 
86  // Set up circuits for extracting gates into (in_circ is set up after
87  // diagonalising the post-selected subspace)
88  qubit_vector_t input_qubits, output_qubits;
89  for (unsigned i = 0; i < tab.get_n_boundaries(); ++i) {
90  ChoiMixTableau::col_key_t key = tab.col_index_.right.at(i);
91  if (key.second == ChoiMixTableau::TableauSegment::Input)
92  input_qubits.push_back(key.first);
93  else
94  output_qubits.push_back(key.first);
95  }
96  Circuit out_circ_tp(output_qubits, {});
97  unit_map_t join_permutation;
98  std::set<Qubit> post_selected;
99  std::map<Qubit, std::optional<unsigned>> in_x_row;
100  std::map<Qubit, std::optional<unsigned>> in_z_row;
101  boost::bimap<Qubit, Qubit> matched_qubits;
102 
103  // Call diagonalisation methods to diagonalise post-selected subspace
104  std::list<SpSymPauliTensor> to_diag;
105  for (unsigned r = tab.get_n_rows(); r > 0;) {
106  --r;
107  ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
108  if (rten.second.size() != 0) {
109  // Reached the rows with non-empty output segment
110  break;
111  }
112  // Else, we add the row to the subspace
113  to_diag.push_back(rten.first);
114  }
115  unsigned post_selected_size = to_diag.size();
116  std::set<Qubit> diag_ins{input_qubits.begin(), input_qubits.end()};
117  Circuit in_circ = mutual_diagonalise(to_diag, diag_ins, CXConfigType::Tree);
118  // Extract the dagger of each gate in order from tab
119  for (const Command& com : in_circ) {
120  auto args = com.get_args();
121  qubit_vector_t qbs = {args.begin(), args.end()};
122  tab.apply_gate(
123  com.get_op_ptr()->dagger()->get_type(), qbs,
124  ChoiMixTableau::TableauSegment::Input);
125  }
126 
127  // Diagonalised rows should still be at the bottom of the tableau - reduce
128  // them to a minimal set of qubits for post-selection by first reducing to
129  // upper echelon form
130  std::vector<std::pair<unsigned, unsigned>> row_ops =
131  gaussian_elimination_row_ops(
132  tab.tab_.zmat_.bottomRows(post_selected_size));
133  for (const std::pair<unsigned, unsigned>& op : row_ops) {
134  tab.tab_.row_mult(op.first, op.second);
135  }
136  // Obtain CX instructions as column operations
137  std::vector<std::pair<unsigned, unsigned>> col_ops =
138  gaussian_elimination_col_ops(
139  tab.tab_.zmat_.bottomRows(post_selected_size));
140  // These gates will also swap qubits to isolate the post-selections on the
141  // first few qubits - this is fine as can be cleaned up later with peephole
142  // optimisations
143  for (const std::pair<unsigned, unsigned>& op : col_ops) {
144  tab.tab_.apply_CX(op.first, op.second);
145  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.first);
146  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.second);
147  in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
148  }
149 
150  // Post-select rows
151  // TODO Post-selection op is not yet available in tket - replace this once
152  // implemented; an implementation hint is provided below
153  if (post_selected_size != 0)
154  throw std::logic_error(
155  "Not yet implemented: post-selection required during ChoiMixTableau "
156  "synthesis");
157  // for (unsigned r = 0; r < post_selected_size; ++r) {
158  // ChoiMixTableau::row_tensor_t row = tab.get_row(tab.get_n_rows() - 1);
159  // if (row.second.string.map.size() != 0 || row.first.string.map.size() != 1
160  // || row.first.string.map.begin()->second != Pauli::Z) throw
161  // std::logic_error("Unexpected error during post-selection identification
162  // in ChoiMixTableau synthesis"); Qubit post_selected_qb =
163  // row.first.string.map.begin()->first; if (row.second.coeff == -1.)
164  // in_circ.add_op<Qubit>(OpType::X, {post_selected_qb});
165  // tab.remove_row(tab.get_n_rows() - 1);
166  // post_selected.insert(post_selected_qb);
167  // throw std::logic_error("Not yet implemented: post-selection required
168  // during ChoiMixTableau synthesis");
169  // }
170 
171  // Input-first gaussian elimination to solve input-sides of remaining rows
172  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
173  tab.gaussian_form();
174 
175  // Iterate through remaining inputs and reduce output portion to a single
176  // qubit
177  for (const Qubit& in_qb : input_qubits) {
178  // Skip post-selected qubits
179  if (post_selected.find(in_qb) != post_selected.end()) continue;
180 
181  unsigned col = tab.col_index_.left.at(ChoiMixTableau::col_key_t{
182  in_qb, ChoiMixTableau::TableauSegment::Input});
183  std::optional<Qubit> out_qb = std::nullopt;
184 
185  // Find the row with X_in_qb (if one exists)
186  std::optional<unsigned> x_row = std::nullopt;
187  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
188  if (tab.tab_.xmat_(r, col)) {
189  x_row = r;
190  break;
191  }
192  }
193 
194  if (x_row) {
195  // A possible optimisation could involve row multiplications to reduce the
196  // Hamming weight of the row before applying gates, but minimising this
197  // would solve minimum weight/distance of a binary linear code whose
198  // decision problem is NP-complete (shown by Vardy, "The intractability of
199  // computing the minimum distance of a code", 1997). Just settle on using
200  // the first row for now, reducing the input and output to a single qubit
201  ChoiMixTableau::row_tensor_t row_paulis = tab.get_row(*x_row);
202  for (const std::pair<const Qubit, Pauli>& p : row_paulis.second.string) {
203  if (matched_qubits.right.find(p.first) == matched_qubits.right.end()) {
204  out_qb = p.first;
205  matched_qubits.insert({in_qb, *out_qb});
206  break;
207  }
208  }
209 
210  // Reduce input string to just X_in_qb
211  if (row_paulis.first.get(in_qb) == Pauli::Y) {
212  // If it is a Y, extract an Sdg gate so the Pauli is exactly X
213  in_circ.add_op<Qubit>(OpType::Sdg, {in_qb});
214  tab.apply_S(in_qb, ChoiMixTableau::TableauSegment::Input);
215  }
216  for (const std::pair<const Qubit, Pauli>& qbp : row_paulis.first.string) {
217  if (qbp.first == in_qb) continue;
218  // Extract an entangling gate to eliminate the qubit
219  switch (qbp.second) {
220  case Pauli::X: {
221  in_circ.add_op<Qubit>(OpType::CX, {in_qb, qbp.first});
222  tab.apply_CX(
223  in_qb, qbp.first, ChoiMixTableau::TableauSegment::Input);
224  break;
225  }
226  case Pauli::Y: {
227  in_circ.add_op<Qubit>(OpType::CY, {in_qb, qbp.first});
228  tab.apply_gate(
229  OpType::CY, {in_qb, qbp.first},
230  ChoiMixTableau::TableauSegment::Input);
231  break;
232  }
233  case Pauli::Z: {
234  in_circ.add_op<Qubit>(OpType::CZ, {in_qb, qbp.first});
235  tab.apply_gate(
236  OpType::CZ, {in_qb, qbp.first},
237  ChoiMixTableau::TableauSegment::Input);
238  break;
239  }
240  default: {
241  break;
242  }
243  }
244  }
245 
246  // And then the same for X_out_qb
247  if (row_paulis.second.get(*out_qb) == Pauli::Y) {
248  // If it is a Y, extract an Sdg gate so the Pauli is exactly X
249  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
250  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
251  } else if (row_paulis.second.get(*out_qb) == Pauli::Z) {
252  // If it is a Z, extract an Vdg and Sdg gate so the Pauli is exactly X
253  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
254  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
255  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
256  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
257  }
258  for (const std::pair<const Qubit, Pauli>& qbp :
259  row_paulis.second.string) {
260  if (qbp.first == *out_qb) continue;
261  // Extract an entangling gate to eliminate the qubit
262  switch (qbp.second) {
263  case Pauli::X: {
264  out_circ_tp.add_op<Qubit>(OpType::CX, {*out_qb, qbp.first});
265  tab.apply_CX(
266  *out_qb, qbp.first, ChoiMixTableau::TableauSegment::Output);
267  break;
268  }
269  case Pauli::Y: {
270  // CY does not have a transpose OpType defined so decompose
271  out_circ_tp.add_op<Qubit>(OpType::S, {qbp.first});
272  out_circ_tp.add_op<Qubit>(OpType::CX, {*out_qb, qbp.first});
273  out_circ_tp.add_op<Qubit>(OpType::Sdg, {qbp.first});
274  tab.apply_gate(
275  OpType::CY, {*out_qb, qbp.first},
276  ChoiMixTableau::TableauSegment::Output);
277  break;
278  }
279  case Pauli::Z: {
280  out_circ_tp.add_op<Qubit>(OpType::CZ, {*out_qb, qbp.first});
281  tab.apply_gate(
282  OpType::CZ, {*out_qb, qbp.first},
283  ChoiMixTableau::TableauSegment::Output);
284  break;
285  }
286  default: {
287  break;
288  }
289  }
290  }
291  }
292 
293  // Find the row with Z_in_qb (if one exists)
294  std::optional<unsigned> z_row = std::nullopt;
295  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
296  if (tab.tab_.zmat_(r, col)) {
297  z_row = r;
298  break;
299  }
300  }
301  if (z_row) {
302  // If both an X and Z row exist, then out_qb should have a
303  // value and the rows should have anticommuting Paulis on out_qb to
304  // preserve commutativity of rows
305  ChoiMixTableau::row_tensor_t row_paulis = tab.get_row(*z_row);
306  if (!x_row) {
307  for (const std::pair<const Qubit, Pauli>& p :
308  row_paulis.second.string) {
309  if (matched_qubits.right.find(p.first) ==
310  matched_qubits.right.end()) {
311  out_qb = p.first;
312  matched_qubits.insert({in_qb, *out_qb});
313  break;
314  }
315  }
316  }
317 
318  // Reduce input string to just Z_in_qb.
319  // No need to consider different paulis on in_qb: if we had X or Y, this
320  // row would have been identified as x_row instead of Z row, and if
321  // another row was already chosen as x_row then canonical gaussian form
322  // would imply all other rows do not contain X_in_qb or Y_in_qb
323  for (const std::pair<const Qubit, Pauli>& qbp : row_paulis.first.string) {
324  if (qbp.first == in_qb) continue;
325  // Extract an entangling gate to eliminate the qubit
326  switch (qbp.second) {
327  case Pauli::X: {
328  in_circ.add_op<Qubit>(OpType::H, {qbp.first});
329  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
330  tab.apply_gate(
331  OpType::H, {qbp.first}, ChoiMixTableau::TableauSegment::Input);
332  tab.apply_CX(
333  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
334  break;
335  }
336  case Pauli::Y: {
337  in_circ.add_op<Qubit>(OpType::Vdg, {qbp.first});
338  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
339  tab.apply_V(qbp.first, ChoiMixTableau::TableauSegment::Input);
340  tab.apply_CX(
341  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
342  break;
343  }
344  case Pauli::Z: {
345  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
346  tab.apply_CX(
347  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
348  break;
349  }
350  default: {
351  break;
352  }
353  }
354  }
355 
356  // And then reduce output string to just Z_out_qb
357  if (row_paulis.second.get(*out_qb) == Pauli::Y) {
358  // If it is a Y, extract a Vdg gate so the Pauli is exactly Z
359  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
360  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
361  } else if (row_paulis.second.get(*out_qb) == Pauli::X) {
362  // If it is an X, extract an Sdg and Vdg gate so the Pauli is exactly Z
363  // We do not need to care about messing up the X row here since if we
364  // solved an X row then this row can't also have X by commutativity
365  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
366  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
367  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
368  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
369  }
370  for (const std::pair<const Qubit, Pauli>& qbp :
371  row_paulis.second.string) {
372  if (qbp.first == *out_qb) continue;
373  // Extract an entangling gate to eliminate the qubit
374  switch (qbp.second) {
375  case Pauli::X: {
376  out_circ_tp.add_op<Qubit>(OpType::H, {qbp.first});
377  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
378  tab.apply_gate(
379  OpType::H, {qbp.first}, ChoiMixTableau::TableauSegment::Output);
380  tab.apply_CX(
381  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
382  break;
383  }
384  case Pauli::Y: {
385  out_circ_tp.add_op<Qubit>(OpType::Vdg, {qbp.first});
386  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
387  tab.apply_V(qbp.first, ChoiMixTableau::TableauSegment::Output);
388  tab.apply_CX(
389  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
390  break;
391  }
392  case Pauli::Z: {
393  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
394  tab.apply_CX(
395  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
396  break;
397  }
398  default: {
399  break;
400  }
401  }
402  }
403  }
404 
405  in_x_row.insert({in_qb, x_row});
406  in_z_row.insert({in_qb, z_row});
407  }
408 
409  // X and Z row are guaranteed to be the unique rows with X_in_qb and Z_in_qb.
410  // If the Z row exists, then all rows must commute with Z_in_qb Z_out_qb, so
411  // if any row has X_out_qb it must also have X_in_qb. Hence if both exist then
412  // the X row is also the unique row with X_out_qb, and by similar argument the
413  // Z row is the unique row with Z_out_qb. However, if only one row exists,
414  // this uniqueness may not hold but can be forced by applying some unitary
415  // gates to eliminate e.g. Z_out_qb from all other rows. We use a gaussian
416  // elimination subroutine to identify a combination of gates that won't add
417  // Z_out_qb onto other qubits. Since we have already removed them from all
418  // other rows containing input components, we only need to consider the
419  // output-only rows, which all exist at the bottom of the tableau.
420  unsigned out_stabs = 0;
421  while (out_stabs < tab.get_n_rows()) {
422  ChoiMixTableau::row_tensor_t rten =
423  tab.get_row(tab.get_n_rows() - 1 - out_stabs);
424  if (rten.first.size() != 0) {
425  // Reached the rows with non-empty input segment
426  break;
427  }
428  ++out_stabs;
429  }
430  MatrixXb out_rows = MatrixXb::Zero(out_stabs, 2 * output_qubits.size());
431  out_rows(Eigen::all, Eigen::seq(0, Eigen::last, 2)) = tab.tab_.xmat_.block(
432  tab.get_n_rows() - out_stabs, input_qubits.size(), out_stabs,
433  output_qubits.size());
434  out_rows(Eigen::all, Eigen::seq(1, Eigen::last, 2)) = tab.tab_.zmat_.block(
435  tab.get_n_rows() - out_stabs, input_qubits.size(), out_stabs,
436  output_qubits.size());
437  // Identify other qubits to apply gates to for removing the extra matched
438  // output terms
439  MatrixXb unmatched_outs = MatrixXb::Zero(
440  out_stabs, 2 * (output_qubits.size() - matched_qubits.size()));
441  std::vector<std::pair<Qubit, Pauli>> col_lookup;
442  for (unsigned out_col = 0; out_col < output_qubits.size(); ++out_col) {
443  unsigned tab_col = input_qubits.size() + out_col;
444  Qubit out_qb = tab.col_index_.right.at(tab_col).first;
445  if (matched_qubits.right.find(out_qb) != matched_qubits.right.end())
446  continue;
447  unmatched_outs.col(col_lookup.size()) = out_rows.col(2 * out_col);
448  col_lookup.push_back({out_qb, Pauli::X});
449  unmatched_outs.col(col_lookup.size()) = out_rows.col(2 * out_col + 1);
450  col_lookup.push_back({out_qb, Pauli::Z});
451  }
452  row_ops = gaussian_elimination_row_ops(unmatched_outs);
453  for (const std::pair<unsigned, unsigned>& op : row_ops) {
454  for (unsigned c = 0; c < unmatched_outs.cols(); ++c) {
455  unmatched_outs(op.second, c) =
456  unmatched_outs(op.second, c) ^ unmatched_outs(op.first, c);
457  }
458  tab.tab_.row_mult(
459  tab.get_n_rows() - out_stabs + op.first,
460  tab.get_n_rows() - out_stabs + op.second);
461  }
462  // Go through the output-only rows and remove terms from matched qubits
463  for (unsigned r = 0; r < out_stabs; ++r) {
464  unsigned leading_col = 0;
465  for (unsigned c = 0; c < unmatched_outs.cols(); ++c) {
466  if (unmatched_outs(r, c)) {
467  leading_col = c;
468  break;
469  }
470  }
471  std::pair<Qubit, Pauli> alternate_qb = col_lookup.at(leading_col);
472 
473  if (alternate_qb.second != Pauli::X) {
474  // Make the alternate point of contact X so we only need one set of rule
475  // for eliminating Paulis
476  tab.apply_gate(
477  OpType::H, {alternate_qb.first},
478  ChoiMixTableau::TableauSegment::Output);
479  out_circ_tp.add_op<Qubit>(OpType::H, {alternate_qb.first});
480  }
481 
482  ChoiMixTableau::row_tensor_t row_paulis =
483  tab.get_row(tab.get_n_rows() - out_stabs + r);
484 
485  for (const std::pair<const Qubit, Pauli>& qbp : row_paulis.second.string) {
486  if (matched_qubits.right.find(qbp.first) == matched_qubits.right.end())
487  continue;
488  // Alternate point is guaranteed to be unmatched, so always needs an
489  // entangling gate
490  switch (qbp.second) {
491  case Pauli::X: {
492  out_circ_tp.add_op<Qubit>(
493  OpType::CX, {alternate_qb.first, qbp.first});
494  tab.apply_CX(
495  alternate_qb.first, qbp.first,
496  ChoiMixTableau::TableauSegment::Output);
497  break;
498  }
499  case Pauli::Z: {
500  out_circ_tp.add_op<Qubit>(
501  OpType::CZ, {alternate_qb.first, qbp.first});
502  tab.apply_gate(
503  OpType::CZ, {alternate_qb.first, qbp.first},
504  ChoiMixTableau::TableauSegment::Output);
505  break;
506  }
507  default: {
508  // Don't have to care about Y since any matched qubit has a row that
509  // is reduced to either X or Z and all other rows must commute with
510  // that
511  break;
512  }
513  }
514  }
515  }
516 
517  // Now that X_in_qb X_out_qb (or Zs) is the unique row for each of X_in_qb and
518  // X_out_qb, we can actually link up the qubit wires and remove the rows
519  for (const Qubit& in_qb : input_qubits) {
520  if (post_selected.find(in_qb) != post_selected.end()) continue;
521 
522  std::optional<unsigned> x_row = in_x_row.at(in_qb);
523  std::optional<unsigned> z_row = in_z_row.at(in_qb);
524  auto found = matched_qubits.left.find(in_qb);
525  std::optional<Qubit> out_qb = (found == matched_qubits.left.end())
526  ? std::nullopt
527  : std::optional<Qubit>{found->second};
528  // Handle phases and resolve qubit connections
529  if (x_row) {
530  if (z_row) {
531  // Hook up with an identity wire
532  if (tab.tab_.phase_(*z_row)) in_circ.add_op<Qubit>(OpType::X, {in_qb});
533  if (tab.tab_.phase_(*x_row)) in_circ.add_op<Qubit>(OpType::Z, {in_qb});
534  join_permutation.insert({*out_qb, in_qb});
535  } else {
536  // Just an X row, so must be connected to out_qb via a decoherence
537  if (tab.tab_.phase_(*x_row)) in_circ.add_op<Qubit>(OpType::Z, {in_qb});
538  in_circ.add_op<Qubit>(OpType::H, {in_qb});
539  in_circ.add_op<Qubit>(OpType::Collapse, {in_qb});
540  in_circ.add_op<Qubit>(OpType::H, {in_qb});
541  join_permutation.insert({*out_qb, in_qb});
542  }
543  } else {
544  if (z_row) {
545  // Just a Z row, so must be connected to out_qb via a decoherence
546  if (tab.tab_.phase_(*z_row)) in_circ.add_op<Qubit>(OpType::X, {in_qb});
547  in_circ.add_op<Qubit>(OpType::Collapse, {in_qb});
548  join_permutation.insert({*out_qb, in_qb});
549  } else {
550  // No rows involving this input, so it is discarded
551  in_circ.qubit_discard(in_qb);
552  }
553  }
554  }
555 
556  // Remove rows with inputs from the tableau
557  tab.tab_ = SymplecticTableau(
558  tab.tab_.xmat_.bottomRows(out_stabs),
559  tab.tab_.zmat_.bottomRows(out_stabs), tab.tab_.phase_.tail(out_stabs));
560 
561  // Can't use a template with multiple parameters within a macro since the
562  // comma will register as an argument delimiter for the macro
563  using match_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
564  BOOST_FOREACH (match_entry entry, matched_qubits.left) {
565  tab.discard_qubit(entry.first, ChoiMixTableau::TableauSegment::Input);
566  tab.discard_qubit(entry.second, ChoiMixTableau::TableauSegment::Output);
567  }
568  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
569 
570  // Only remaining rows must be completely over the outputs. Call
571  // diagonalisation methods to diagonalise coherent subspace
572  to_diag.clear();
573  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
574  ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
575  to_diag.push_back(rten.second);
576  }
577  std::set<Qubit> diag_outs;
578  for (const Qubit& out : output_qubits) {
579  if (matched_qubits.right.find(out) == matched_qubits.right.end())
580  diag_outs.insert(out);
581  }
582  Circuit out_diag_circ =
583  mutual_diagonalise(to_diag, diag_outs, CXConfigType::Tree);
584  // Extract the dagger of each gate in order from tab
585  for (const Command& com : out_diag_circ) {
586  auto args = com.get_args();
587  qubit_vector_t qbs = {args.begin(), args.end()};
588  tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
589  out_circ_tp.add_op<Qubit>(com.get_op_ptr()->transpose()->dagger(), qbs);
590  }
591 
592  // All rows are diagonalised so we can just focus on the Z matrix. Reduce them
593  // to a minimal set of qubits for initialisation by first reducing to upper
594  // echelon form
595  row_ops = gaussian_elimination_row_ops(tab.tab_.zmat_);
596  for (const std::pair<unsigned, unsigned>& op : row_ops) {
597  tab.tab_.row_mult(op.first, op.second);
598  }
599  // Obtain CX instructions as column operations
600  col_ops = gaussian_elimination_col_ops(tab.tab_.zmat_);
601  for (const std::pair<unsigned, unsigned>& op : col_ops) {
602  tab.tab_.apply_CX(op.second, op.first);
603  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.second);
604  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.first);
605  out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
606  }
607 
608  // Fix phases of zero_initialised qubits
609  std::set<Qubit> zero_initialised;
610  Circuit out_circ(output_qubits, {});
611  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
612  ChoiMixTableau::row_tensor_t row = tab.get_row(r);
613  if (row.first.size() != 0 || row.second.size() != 1 ||
614  row.second.string.begin()->second != Pauli::Z)
615  throw std::logic_error(
616  "Unexpected error during zero initialisation in ChoiMixTableau "
617  "synthesis");
618  Qubit initialised_qb = row.second.string.begin()->first;
619  out_circ.qubit_create(initialised_qb);
620  if (row.second.is_real_negative()) {
621  out_circ.add_op<Qubit>(OpType::X, {initialised_qb});
622  }
623  zero_initialised.insert(initialised_qb);
624  }
625 
626  // Remaining outputs that aren't zero initialised or matched need to be
627  // initialised in the maximally-mixed state.
628  // Also match up unmatched outputs to either unmatched inputs or reusable
629  // output names (ones that are already matched up to other input names),
630  // preferring the qubit of the same name
631  std::list<Qubit> reusable_names;
632  for (const Qubit& out_qb : output_qubits) {
633  if (matched_qubits.right.find(out_qb) != matched_qubits.right.end() &&
634  matched_qubits.left.find(out_qb) == matched_qubits.left.end())
635  reusable_names.push_back(out_qb);
636  }
637  for (const Qubit& out_qb : output_qubits) {
638  if (matched_qubits.right.find(out_qb) == matched_qubits.right.end()) {
639  if (zero_initialised.find(out_qb) == zero_initialised.end()) {
640  out_circ.qubit_create(out_qb);
641  out_circ.add_op<Qubit>(OpType::H, {out_qb});
642  out_circ.add_op<Qubit>(OpType::Collapse, {out_qb});
643  }
644  if (matched_qubits.left.find(out_qb) == matched_qubits.left.end()) {
645  matched_qubits.insert({out_qb, out_qb});
646  join_permutation.insert({out_qb, out_qb});
647  } else {
648  // Since the matching is a bijection, matched_qubits.left -
649  // matched_qubits.right (set difference) is the same size as
650  // matched_qubits.right - matched_qubits.left, so there will be exactly
651  // the right number of reusable names to pull from here
652  Qubit name = reusable_names.front();
653  reusable_names.pop_front();
654  matched_qubits.insert({name, out_qb});
655  join_permutation.insert({out_qb, name});
656  }
657  }
658  }
659 
660  // Initialise qubits with stabilizer rows and stitch subcircuits together
661  out_circ.append(out_circ_tp.transpose());
662  in_circ.append_with_map(out_circ, join_permutation);
663  return {in_circ, join_permutation};
664 }
