23ChoiMixTableau circuit_to_cm_tableau(const Circuit& circ) {
24 ChoiMixTableau tab(circ.all_qubits());
25 for (const Qubit& q : circ.created_qubits()) {
26 tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
27 }
28 for (const Command& com : circ) {
29 auto args = com.get_args();
30 qubit_vector_t qbs = {args.begin(), args.end()};
31 tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
32 }
33 tab.rename_qubits(
34 circ.implicit_qubit_permutation(),
35 ChoiMixTableau::TableauSegment::Output);
36 for (const Qubit& q : circ.discarded_qubits()) {
37 tab.discard_qubit(q);
38 }
39 tab.canonical_column_order();
40 tab.gaussian_form();
41 return tab;
42}

44struct ChoiMixBuilder {
74 Circuit in_circ;
75 std::set<Qubit> post_selected;
76 std::set<Qubit> discarded;
77 std::set<Qubit> collapsed;
78 ChoiMixTableau tab;
79 boost::bimap<Qubit, Qubit> in_out_permutation;
80 std::set<Qubit> zero_initialised;
81 std::set<Qubit> mix_initialised;
82 Circuit out_circ_tp;
83
84 // The CXConfigType preferred when invoking diagonalisation techniques
85 CXConfigType cx_config;
86
87 // Additional qubit names (distinct from qubits already on the respective
88 // segment of the tableau) than can be used for zero initialisations and
89 // post-selections when synthesising a unitary extension
90 qubit_vector_t unitary_init_names;
91 qubit_vector_t unitary_post_names;
92
93 // Initialises the builder with a tableau
94 explicit ChoiMixBuilder(const ChoiMixTableau& tab, CXConfigType cx_config);
95 // For synthesis of a unitary extension, initialises the builder with a
96 // tableau and some additional qubit names which the resulting circuit may
97 // optionally use, representing zero-initialised or post-selected qubits.
98 // These are the qubits on which we can freely add Zs to the rows of the given
99 // tableau to guarantee a unitary extension; none of these names should appear
100 // in tab
101 explicit ChoiMixBuilder(
102 const ChoiMixTableau& tab, CXConfigType cx_config,
103 const qubit_vector_t& init_names, const qubit_vector_t& post_names);
104
105 // Debug method: applies all staged operations back onto tab to provide the
106 // tableau that the synthesis result is currently aiming towards. In exact
107 // synthesis, this should remain invariant during synthesis. For synthesis of
108 // a unitary extension, this should at least span the rows of the original
109 // tableau up to additional Zs on spare input and output qubits.
110 ChoiMixTableau realised_tableau() const;
111
116 // Match up pairs of generators that anti-commute in the input segment but
117 // commute with all others; such pairs of rows reduce to an identity wire
118 // between a pair of qubits; we solve this by pairwise Pauli reduction methods
119 void solve_id_subspace();
120 // After removing the identity subspace, all remaining rows mutually commute
121 // within each tableau segment; diagonalise each segment individually
122 void diagonalise_segments();
123 // Solve the post-selected subspace which has already been diagonalised
124 void solve_postselected_subspace();
125 // Solve the zero-initialised subspace which has already been diagonalised
126 void solve_initialised_subspace();
127 // All remaining rows are in the collapsed subspace (each row is the unique
128 // stabilizer passing through some Collapse gate); solve it
129 void solve_collapsed_subspace();
130
131 // Simplifies the tableau by removing qubits on which all rows have I; such
132 // qubits are either discarded inputs or mixed-initialised outputs
133 void remove_unused_qubits();
134 // For synthesis of a unitary extension, match up qubits from
135 // post-selected/zero-initialised with unitary_post_names/unitary_init_names
136 // and add them to in_out_permutation
137 void assign_init_post_names();
138 // Fill out in_out_permutation to map all qubits; this typically takes the
139 // form of a standard qubit reuse pattern (e.g. discard and reinitialise)
140 void assign_remaining_names();
141 // Once tab has been completely reduced to no rows and no qubits, compose the
142 // staged operations to build the output circuit and return the renaming map
143 // from output names of original tableau to the qubits of the returned circuit
144 // they are mapped to
145 std::pair<Circuit, qubit_map_t> output_circuit();
146 std::pair<Circuit, qubit_map_t> unitary_output_circuit();
147};
148
149std::pair<Circuit, qubit_map_t> cm_tableau_to_exact_circuit(
150 const ChoiMixTableau& tab, CXConfigType cx_config) {
151 ChoiMixBuilder builder(tab, cx_config);
152 builder.remove_unused_qubits();
153 builder.solve_id_subspace();
154 builder.diagonalise_segments();
155 builder.solve_postselected_subspace();
156 builder.solve_initialised_subspace();
157 builder.solve_collapsed_subspace();
158 builder.remove_unused_qubits();
159 builder.assign_remaining_names();
160 return builder.output_circuit();
161}

163std::pair<Circuit, qubit_map_t> cm_tableau_to_unitary_extension_circuit(
164 const ChoiMixTableau& tab, const std::vector<Qubit>& init_names,
165 const std::vector<Qubit>& post_names, CXConfigType cx_config) {
166 ChoiMixBuilder builder(tab, cx_config, init_names, post_names);
167 builder.remove_unused_qubits();
168 builder.solve_id_subspace();
169 builder.diagonalise_segments();
170 builder.solve_postselected_subspace();
171 builder.solve_initialised_subspace();
172 builder.solve_collapsed_subspace();
173 builder.remove_unused_qubits();
174 builder.assign_init_post_names();
175 builder.assign_remaining_names();
176 return builder.unitary_output_circuit();
177}

182ChoiMixBuilder::ChoiMixBuilder(
183 const ChoiMixTableau& t, CXConfigType cx, const qubit_vector_t& inits,
184 const qubit_vector_t& posts)
185 : in_circ(),
186 post_selected(),
187 discarded(),
188 collapsed(),
189 tab(t),
190 in_out_permutation(),
191 zero_initialised(),
192 mix_initialised(),
193 out_circ_tp(),
194 cx_config(cx),
195 unitary_init_names(inits),
196 unitary_post_names(posts) {
197 for (unsigned i = 0; i < tab.get_n_boundaries(); ++i) {
198 ChoiMixTableau::col_key_t key = tab.col_index_.right.at(i);
199 if (key.second == ChoiMixTableau::TableauSegment::Input)
200 in_circ.add_qubit(key.first);
201 else
202 out_circ_tp.add_qubit(key.first);
203 }
204 for (const Qubit& init_q : unitary_init_names) {
205 if (tab.col_index_.left.find(
206 ChoiMixTableau::col_key_t{
207 init_q, ChoiMixTableau::TableauSegment::Input}) !=
208 tab.col_index_.left.end())
209 throw std::logic_error(
210 "Free qubit name for initialisation conflicts with existing live "
211 "input of ChoiMixTableau");
212 }
213 for (const Qubit& post_q : unitary_post_names) {
214 if (tab.col_index_.left.find(
215 ChoiMixTableau::col_key_t{
216 post_q, ChoiMixTableau::TableauSegment::Output}) !=
217 tab.col_index_.left.end())
218 throw std::logic_error(
219 "Free qubit name for post-selection conflicts with existing live "
220 "output of ChoiMixTableau");
221 }
222}

226ChoiMixTableau ChoiMixBuilder::realised_tableau() const {
227 ChoiMixTableau in_tab = circuit_to_cm_tableau(in_circ);
228 for (const Qubit& q : post_selected)
229 in_tab.post_select(q, ChoiMixTableau::TableauSegment::Output);
230 for (const Qubit& q : discarded)
231 in_tab.discard_qubit(q, ChoiMixTableau::TableauSegment::Output);
232 for (const Qubit& q : collapsed)
233 in_tab.collapse_qubit(q, ChoiMixTableau::TableauSegment::Output);
234 ChoiMixTableau out_tab = circuit_to_cm_tableau(out_circ_tp.transpose());
235 for (const Qubit& q : zero_initialised)
236 out_tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
237 for (const Qubit& q : mix_initialised)
238 out_tab.discard_qubit(q, ChoiMixTableau::TableauSegment::Input);
239 qubit_map_t out_in_permutation{};
240 using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
241 BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
242 out_in_permutation.insert({entry.second, entry.first});
243 }
244 out_tab.rename_qubits(
245 out_in_permutation, ChoiMixTableau::TableauSegment::Output);
246 return ChoiMixTableau::compose(ChoiMixTableau::compose(in_tab, tab), out_tab);
247}

250void ChoiMixBuilder::solve_id_subspace() {
251 // Input-first gaussian elimination to solve input-sides of remaining rows
252 tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
253 tab.gaussian_form();
254
255 std::set<unsigned> solved_rows;
256 std::set<Qubit> solved_ins, solved_outs;
257 for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
258 if (solved_rows.find(r) != solved_rows.end()) continue;
259
260 // Look for a row which anticommutes with row r over the inputs
261 std::list<unsigned> xcols, zcols;
262 for (unsigned c = 0; c < tab.get_n_inputs(); ++c) {
263 if (tab.tab_.xmat(r, c)) xcols.push_back(c);
264 if (tab.tab_.zmat(r, c)) zcols.push_back(c);
265 }
266 for (unsigned r2 = r + 1; r2 < tab.get_n_rows(); ++r2) {
267 if (solved_rows.find(r2) != solved_rows.end()) continue;
268 bool anti = false;
269 for (const unsigned c : xcols) anti ^= tab.tab_.zmat(r2, c);
270 for (const unsigned c : zcols) anti ^= tab.tab_.xmat(r2, c);
271 if (!anti) continue;
272
273 // Found a candidate pair of rows. Because of the Gaussian elimination, it
274 // is more likely that the first mismatching qubit is X for r and Z for
275 // r2, so favour reducing r2 to Z and r to X
276 ChoiMixTableau::row_tensor_t row_r = tab.get_row(r);
277 ChoiMixTableau::row_tensor_t row_r2 = tab.get_row(r2);
278 std::pair<Circuit, Qubit> in_diag_circ =
279 reduce_anticommuting_paulis_to_z_x(
280 row_r2.first, row_r.first, cx_config);
281 in_circ.append(in_diag_circ.first);
282 for (const Command& com : in_diag_circ.first) {
283 auto args = com.get_args();
284 qubit_vector_t qbs = {args.begin(), args.end()};
285 tab.apply_gate(
286 com.get_op_ptr()->dagger()->get_type(), qbs,
287 ChoiMixTableau::TableauSegment::Input);
288 }
289
290 // Since the full rows must commute but they anticommute over the inputs,
291 // they must also anticommute over the outputs; we similarly reduce these
292 // down to Z and X
293 std::pair<Circuit, Qubit> out_diag_circ_dag =
294 reduce_anticommuting_paulis_to_z_x(
295 row_r2.second, row_r.second, cx_config);
296 out_circ_tp.append(out_diag_circ_dag.first.dagger().transpose());
297 for (const Command& com : out_diag_circ_dag.first) {
298 auto args = com.get_args();
299 qubit_vector_t qbs = {args.begin(), args.end()};
300 tab.apply_gate(
301 com.get_op_ptr()->get_type(), qbs,
302 ChoiMixTableau::TableauSegment::Output);
303 }
304
305 // Check that rows have been successfully reduced
306 row_r = tab.get_row(r);
307 row_r2 = tab.get_row(r2);
308 if (row_r.first.size() != 1 ||
309 row_r.first.string.begin()->second != Pauli::X ||
310 row_r.second.size() != 1 ||
311 row_r.second.string.begin()->second != Pauli::X ||
312 row_r2.first.size() != 1 ||
313 row_r2.first.string.begin()->second != Pauli::Z ||
314 row_r2.second.size() != 1 ||
315 row_r2.second.string.begin()->second != Pauli::Z)
316 throw std::logic_error(
317 "Unexpected error during identity reduction in ChoiMixTableau "
318 "synthesis");
319 // Solve phases
320 if (row_r.second.is_real_negative()) {
321 in_circ.add_op<Qubit>(OpType::Z, {in_diag_circ.second});
322 tab.apply_gate(
323 OpType::Z, {in_diag_circ.second},
324 ChoiMixTableau::TableauSegment::Input);
325 }
326 if (row_r2.second.is_real_negative()) {
327 in_circ.add_op<Qubit>(OpType::X, {in_diag_circ.second});
328 tab.apply_gate(
329 OpType::X, {in_diag_circ.second},
330 ChoiMixTableau::TableauSegment::Input);
331 }
332 // Connect in permutation
333 in_out_permutation.insert(
334 {in_diag_circ.second, out_diag_circ_dag.second});
335 solved_rows.insert(r);
336 solved_rows.insert(r2);
337
338 // Remove these solved qubits from other rows; by commutation of rows, a
339 // row contains Z@in_diag_circ.second iff it contains
340 // Z@out_diag_circ_dag.second and similarly for X
341 unsigned in_c = tab.col_index_.left.at(
342 ChoiMixTableau::col_key_t{
343 in_diag_circ.second, ChoiMixTableau::TableauSegment::Input});
344 for (unsigned r3 = 0; r3 < tab.get_n_rows(); ++r3) {
345 if (r3 != r && tab.tab_.xmat(r3, in_c)) tab.tab_.row_mult(r, r3);
346 if (r3 != r2 && tab.tab_.zmat(r3, in_c)) tab.tab_.row_mult(r2, r3);
347 }
348 solved_ins.insert({in_diag_circ.second});
349 solved_outs.insert({out_diag_circ_dag.second});
350 break;
351 }
352 }
353
354 // Remove solved rows and qubits from tableau; since removing rows/columns
355 // replaces them with the row/column from the end, remove in reverse order
356 for (auto it = solved_rows.rbegin(); it != solved_rows.rend(); ++it)
357 tab.remove_row(*it);
358 for (auto it = solved_ins.rbegin(); it != solved_ins.rend(); ++it)
359 tab.discard_qubit(*it, ChoiMixTableau::TableauSegment::Input);
360 for (auto it = solved_outs.rbegin(); it != solved_outs.rend(); ++it)
361 tab.discard_qubit(*it, ChoiMixTableau::TableauSegment::Output);
362}

364// Given a matrix that is already in upper echelon form, use the fact that the
365// leading columns are already unique to give column operations that reduce it
366// down to identity over the leading columns, eliminating extra swap gates to
367// move to the first spaces
368static std::vector<std::pair<unsigned, unsigned>>
369leading_column_gaussian_col_ops(const MatrixXb& source) {
370 std::vector<unsigned> col_list;
371 std::set<unsigned> non_leads;
372 for (unsigned r = 0; r < source.rows(); ++r) {
373 bool leading_found = false;
374 for (unsigned c = 0; c < source.cols(); ++c) {
375 if (source(r, c)) {
376 if (leading_found)
377 non_leads.insert(c);
378 else {
379 leading_found = true;
380 col_list.push_back(c);
381 }
382 }
383 }
384 }
385 for (const unsigned& c : non_leads) col_list.push_back(c);
386 MatrixXb reordered = MatrixXb::Zero(source.rows(), col_list.size());
387 for (unsigned c = 0; c < col_list.size(); ++c)
388 reordered.col(c) = source.col(col_list.at(c));
389 std::vector<std::pair<unsigned, unsigned>> reordered_ops =
390 gaussian_elimination_col_ops(reordered);
391 std::vector<std::pair<unsigned, unsigned>> res;
392 for (const std::pair<unsigned, unsigned>& op : reordered_ops)
393 res.push_back({col_list.at(op.first), col_list.at(op.second)});
394 return res;
395}

397void ChoiMixBuilder::diagonalise_segments() {
398 // Canonicalise tableau
399 tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
400 tab.gaussian_form();
401
402 // Set up diagonalisation tasks
403 std::list<SpSymPauliTensor> to_diag_ins, to_diag_outs;
404 for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
405 ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
406 if (!rten.first.string.empty()) to_diag_ins.push_back(rten.first);
407 if (!rten.second.string.empty()) to_diag_outs.push_back(rten.second);
408 }
409 qubit_vector_t input_qubits = tab.input_qubits();
410 std::set<Qubit> diag_ins{input_qubits.begin(), input_qubits.end()};
411 Circuit in_diag_circ = mutual_diagonalise(to_diag_ins, diag_ins, cx_config);
412 for (const Command& com : in_diag_circ) {
413 auto args = com.get_args();
414 in_circ.add_op<UnitID>(com.get_op_ptr(), args);
415 qubit_vector_t qbs = {args.begin(), args.end()};
416 tab.apply_gate(
417 com.get_op_ptr()->dagger()->get_type(), qbs,
418 ChoiMixTableau::TableauSegment::Input);
419 }
420 qubit_vector_t output_qubits = tab.output_qubits();
421 std::set<Qubit> diag_outs{output_qubits.begin(), output_qubits.end()};
422 Circuit out_diag_circ =
423 mutual_diagonalise(to_diag_outs, diag_outs, cx_config);
424 for (const Command& com : out_diag_circ) {
425 auto args = com.get_args();
426 out_circ_tp.add_op<UnitID>(com.get_op_ptr()->dagger()->transpose(), args);
427 qubit_vector_t qbs = {args.begin(), args.end()};
428 tab.apply_gate(
429 com.get_op_ptr()->get_type(), qbs,
430 ChoiMixTableau::TableauSegment::Output);
431 }
432
433 // All rows are diagonalised, so we can just focus on the Z matrix
434 if (tab.tab_.xmat != MatrixXb::Zero(tab.get_n_rows(), tab.get_n_boundaries()))
435 throw std::logic_error(
436 "Diagonalisation in ChoiMixTableau synthesis failed");
437}

439void ChoiMixBuilder::solve_postselected_subspace() {
440 // As column order is currently output first, gaussian form will reveal the
441 // post-selected space at the bottom of the tableau and the submatrix of those
442 // rows will already be in upper echelon form
443 tab.gaussian_form();
444 // Reduce them to a minimal set of qubits using CX gates
445 unsigned n_postselected = 0;
446 for (; n_postselected < tab.get_n_rows(); ++n_postselected) {
447 if (!tab.get_row(tab.get_n_rows() - 1 - n_postselected)
448 .second.string.empty())
449 break;
450 }
451 unsigned n_ins = tab.get_n_inputs();
452 unsigned n_outs = tab.get_n_outputs();
453 MatrixXb subtableau = tab.tab_.zmat.bottomRightCorner(n_postselected, n_ins);
454 std::vector<std::pair<unsigned, unsigned>> col_ops =
455 leading_column_gaussian_col_ops(subtableau);
456 for (const std::pair<unsigned, unsigned>& op : col_ops) {
457 unsigned tab_ctrl_col = n_outs + op.second;
458 unsigned tab_trgt_col = n_outs + op.first;
459 tab.tab_.apply_CX(tab_ctrl_col, tab_trgt_col);
460 ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(tab_ctrl_col);
461 ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(tab_trgt_col);
462 in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
463 }
464 // Postselect rows
465 for (unsigned r = 0; r < n_postselected; ++r) {
466 unsigned final_row = tab.get_n_rows() - 1;
467 ChoiMixTableau::row_tensor_t row = tab.get_row(final_row);
468 if (row.second.size() != 0 || row.first.size() != 1 ||
469 row.first.string.begin()->second != Pauli::Z)
470 throw std::logic_error(
471 "Unexpected error during post-selection identification in "
472 "ChoiMixTableau synthesis");
473 Qubit post_selected_qb = row.first.string.begin()->first;
474 // Multiply other rows to remove Z_qb components
475 unsigned qb_col = tab.col_index_.left.at(
476 ChoiMixTableau::col_key_t{
477 post_selected_qb, ChoiMixTableau::TableauSegment::Input});
478 for (unsigned s = 0; s < final_row; ++s)
479 if (tab.tab_.zmat(s, qb_col)) tab.tab_.row_mult(final_row, s);
480 // Post-select on correct phase
481 if (row.second.is_real_negative())
482 in_circ.add_op<Qubit>(OpType::X, {post_selected_qb});
483 tab.remove_row(final_row);
484 post_selected.insert(post_selected_qb);
485 tab.discard_qubit(post_selected_qb, ChoiMixTableau::TableauSegment::Input);
486 }
487}

489void ChoiMixBuilder::solve_initialised_subspace() {
490 // Input-first gaussian elimination now sorts the remaining rows into the
491 // collapsed subspace followed by the zero-initialised subspace and the
492 // collapsed subspace rows are in upper echelon form over the inputs, giving
493 // unique leading columns and allowing us to solve them with CXs by
494 // column-wise gaussian elimination; same for zero-initialised rows over the
495 // outputs
496 tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
497 tab.gaussian_form();
498
499 // Reduce the zero-initialised space to a minimal set of qubits using CX gates
500 unsigned n_collapsed = 0;
501 for (; n_collapsed < tab.get_n_rows(); ++n_collapsed) {
502 if (tab.get_row(n_collapsed).first.string.empty()) break;
503 }
504 unsigned n_ins = tab.get_n_inputs();
505 unsigned n_outs = tab.get_n_outputs();
506 MatrixXb subtableau =
507 tab.tab_.zmat.bottomRightCorner(tab.get_n_rows() - n_collapsed, n_outs);
508 std::vector<std::pair<unsigned, unsigned>> col_ops =
509 leading_column_gaussian_col_ops(subtableau);
510 for (const std::pair<unsigned, unsigned>& op : col_ops) {
511 unsigned tab_ctrl_col = n_ins + op.second;
512 unsigned tab_trgt_col = n_ins + op.first;
513 tab.tab_.apply_CX(tab_ctrl_col, tab_trgt_col);
514 ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(tab_ctrl_col);
515 ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(tab_trgt_col);
516 out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
517 }
518 // Initialise rows
519 for (unsigned r = tab.get_n_rows(); r-- > n_collapsed;) {
520 // r always refers to the final row in the tableau
521 ChoiMixTableau::row_tensor_t row = tab.get_row(r);
522 if (row.first.size() != 0 || row.second.size() != 1 ||
523 row.second.string.begin()->second != Pauli::Z)
524 throw std::logic_error(
525 "Unexpected error during initialisation identification in "
526 "ChoiMixTableau synthesis");
527 Qubit initialised_qb = row.second.string.begin()->first;
528 // Multiply other rows to remove Z_qb components
529 unsigned qb_col = tab.col_index_.left.at(
530 ChoiMixTableau::col_key_t{
531 initialised_qb, ChoiMixTableau::TableauSegment::Output});
532 for (unsigned s = 0; s < r; ++s)
533 if (tab.tab_.zmat(s, qb_col)) tab.tab_.row_mult(r, s);
534 // Initialise with correct phase
535 if (row.second.is_real_negative())
536 out_circ_tp.add_op<Qubit>(OpType::X, {initialised_qb});
537 tab.remove_row(r);
538 zero_initialised.insert(initialised_qb);
539 tab.discard_qubit(initialised_qb, ChoiMixTableau::TableauSegment::Output);
540 }
541}

543void ChoiMixBuilder::solve_collapsed_subspace() {
544 // Solving the initialised subspace will have preserved the upper echelon form
545 // of the collapsed subspace; reduce the inputs of the collapsed space to a
546 // minimal set of qubits using CX gates
547 unsigned n_ins = tab.get_n_inputs();
548 unsigned n_outs = tab.get_n_outputs();
549 MatrixXb subtableau = tab.tab_.zmat.topLeftCorner(tab.get_n_rows(), n_ins);
550 std::vector<std::pair<unsigned, unsigned>> col_ops =
551 leading_column_gaussian_col_ops(subtableau);
552 for (const std::pair<unsigned, unsigned>& op : col_ops) {
553 tab.tab_.apply_CX(op.second, op.first);
554 ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.second);
555 ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.first);
556 in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
557 }
558 // Since row multiplications will unsolve the inputs, we cannot get the output
559 // segment into upper echelon form for the same CX-saving trick; instead we
560 // accept just removing any qubits that are now unused after solving the
561 // initialised subspace
562 remove_unused_qubits();
563 tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
564 // Solve the output segment using CX gates
565 n_ins = tab.get_n_inputs();
566 n_outs = tab.get_n_outputs();
567 col_ops = gaussian_elimination_col_ops(
568 tab.tab_.zmat.topRightCorner(tab.get_n_rows(), n_outs));
569 for (const std::pair<unsigned, unsigned>& op : col_ops) {
570 tab.tab_.apply_CX(n_ins + op.second, n_ins + op.first);
571 ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(n_ins + op.second);
572 ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(n_ins + op.first);
573 out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
574 }
575 // Connect up and remove rows and columns
576 for (unsigned r = tab.get_n_rows(); r-- > 0;) {
577 // r refers to the final row
578 // Check that row r has been successfully reduced
579 ChoiMixTableau::row_tensor_t row_r = tab.get_row(r);
580 if (row_r.first.size() != 1 ||
581 row_r.first.string.begin()->second != Pauli::Z ||
582 row_r.second.size() != 1 ||
583 row_r.second.string.begin()->second != Pauli::Z)
584 throw std::logic_error(
585 "Unexpected error during collapsed subspace reduction in "
586 "ChoiMixTableau synthesis");
587 Qubit in_q = row_r.first.string.begin()->first;
588 Qubit out_q = row_r.second.string.begin()->first;
589 // Solve phase
590 if (row_r.second.is_real_negative()) {
591 in_circ.add_op<Qubit>(OpType::X, {in_q});
592 tab.apply_gate(OpType::X, {in_q}, ChoiMixTableau::TableauSegment::Input);
593 }
594 // Connect in permutation
595 in_out_permutation.insert({in_q, out_q});
596 collapsed.insert(in_q);
597 tab.remove_row(r);
598 tab.discard_qubit(in_q, ChoiMixTableau::TableauSegment::Input);
599 tab.discard_qubit(out_q, ChoiMixTableau::TableauSegment::Output);
600 }
601}

603void ChoiMixBuilder::remove_unused_qubits() {
604 // Since removing a column replaces it with the last column, remove in reverse
605 // order to examine each column exactly once
606 for (unsigned c = tab.get_n_boundaries(); c-- > 0;) {
607 bool used = false;
608 for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
609 if (tab.tab_.zmat(r, c) || tab.tab_.xmat(r, c)) {
610 used = true;
611 break;
612 }
613 }
614 if (used) continue;
615 ChoiMixTableau::col_key_t col = tab.col_index_.right.at(c);
616 if (col.second == ChoiMixTableau::TableauSegment::Input)
617 discarded.insert(col.first);
618 else
619 mix_initialised.insert(col.first);
620 tab.discard_qubit(col.first, col.second);
621 }
622}

624void ChoiMixBuilder::assign_init_post_names() {
625 auto it = unitary_post_names.begin();
626 for (const Qubit& ps : post_selected) {
627 if (it == unitary_post_names.end())
628 throw std::logic_error(
629 "Not enough additional qubit names for unitary extension of "
630 "ChoiMixTableau to safely handle post-selected subspace");
631 in_out_permutation.insert({ps, *it});
632 ++it;
633 }
634 unitary_post_names = {it, unitary_post_names.end()};
635
636 it = unitary_init_names.begin();
637 for (const Qubit& zi : zero_initialised) {
638 if (it == unitary_init_names.end())
639 throw std::logic_error(
640 "Not enough additional qubit names for unitary extension of "
641 "ChoiMixTableau to safely handle initialised subspace");
642 in_out_permutation.insert({*it, zi});
643 ++it;
644 }
645 unitary_init_names = {it, unitary_init_names.end()};
646}

648void ChoiMixBuilder::assign_remaining_names() {
649 // Some post-selected or initialised qubits might have already been matched up
650 // for unitary synthesis, so we only need to match up the remainder
651 std::set<Qubit> unsolved_ins = discarded;
652 for (const Qubit& q : post_selected) {
653 if (in_out_permutation.left.find(q) == in_out_permutation.left.end())
654 unsolved_ins.insert(q);
655 }
656 std::set<Qubit> unsolved_outs = mix_initialised;
657 for (const Qubit& q : zero_initialised) {
658 if (in_out_permutation.right.find(q) == in_out_permutation.right.end())
659 unsolved_outs.insert(q);
660 }
661 // If there are more unsolved_ins than unsolved_outs, we want to pad out
662 // unsolved_outs with extra names that don't appear as output names of the
663 // original tableau; between unsolved_ins and the inputs already in
664 // in_out_permutation, there will be at least enough of these
665 if (unsolved_ins.size() > unsolved_outs.size()) {
666 for (const Qubit& q : unsolved_ins) {
667 if (in_out_permutation.right.find(q) == in_out_permutation.right.end()) {
668 unsolved_outs.insert(q);
669 if (unsolved_ins.size() == unsolved_outs.size()) break;
670 }
671 }
672 if (unsolved_ins.size() > unsolved_outs.size()) {
673 using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
674 BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
675 if (in_out_permutation.right.find(entry.first) ==
676 in_out_permutation.right.end()) {
677 unsolved_outs.insert(entry.first);
678 if (unsolved_ins.size() == unsolved_outs.size()) break;
679 }
680 }
681 }
682 } else if (unsolved_ins.size() < unsolved_outs.size()) {
683 for (const Qubit& q : unsolved_outs) {
684 if (in_out_permutation.left.find(q) == in_out_permutation.left.end()) {
685 unsolved_ins.insert(q);
686 if (unsolved_ins.size() == unsolved_outs.size()) break;
687 }
688 }
689 if (unsolved_ins.size() < unsolved_outs.size()) {
690 using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
691 BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
692 if (in_out_permutation.left.find(entry.second) ==
693 in_out_permutation.left.end()) {
694 unsolved_ins.insert(entry.second);
695 if (unsolved_ins.size() == unsolved_outs.size()) break;
696 }
697 }
698 }
699 }
700 // Prefer to connect qubits with the same names
701 for (auto in_it = unsolved_ins.begin(); in_it != unsolved_ins.end();) {
702 auto temp_it = in_it++;
703 auto out_it = unsolved_outs.find(*temp_it);
704 if (out_it != unsolved_outs.end()) {
705 in_out_permutation.insert({*temp_it, *temp_it});
706 unsolved_ins.erase(temp_it);
707 unsolved_outs.erase(out_it);
708 }
709 }
710 // Pair up remainders; by our earlier padding, they should have the exact same
711 // number of elements, so pair them up exactly
712 for (const Qubit& in : unsolved_ins) {
713 auto it = unsolved_outs.begin();
714 in_out_permutation.insert({in, *it});
715 unsolved_outs.erase(it);
716 }
717}

719std::pair<Circuit, qubit_map_t> ChoiMixBuilder::output_circuit() {
720 if (tab.get_n_rows() != 0 || tab.get_n_boundaries() != 0)
721 throw std::logic_error(
722 "Unexpected error during ChoiMixTableau synthesis, reached the end "
723 "with a non-empty tableau remaining");
724 if (!post_selected.empty()) {
725 throw std::logic_error(
726 "Not yet implemented: post-selection required during ChoiMixTableau "
727 "synthesis");
728 }
729 for (const Qubit& q : discarded) in_circ.qubit_discard(q);
730 for (const Qubit& q : collapsed) in_circ.add_op<Qubit>(OpType::Collapse, {q});
731 Circuit out_circ(out_circ_tp.all_qubits(), {});
732 for (const Qubit& q : zero_initialised) out_circ.qubit_create(q);
733 for (const Qubit& q : mix_initialised) {
734 out_circ.qubit_create(q);
735 out_circ.add_op<Qubit>(OpType::H, {q});
736 out_circ.add_op<Qubit>(OpType::Collapse, {q});
737 }
738 out_circ.append(out_circ_tp.transpose());
739 qubit_map_t return_perm;
740 unit_map_t append_perm;
741 using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
742 BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
743 return_perm.insert({entry.second, entry.first});
744 append_perm.insert({entry.second, entry.first});
745 }
746 in_circ.append_with_map(out_circ, append_perm);
747 return {in_circ, return_perm};
748}

750std::pair<Circuit, qubit_map_t> ChoiMixBuilder::unitary_output_circuit() {
751 if (tab.get_n_rows() != 0 || tab.get_n_boundaries() != 0)
752 throw std::logic_error(
753 "Unexpected error during ChoiMixTableau synthesis, reached the end "
754 "with a non-empty tableau remaining");

760 append_perm.insert({entry.second, entry.first});
761 }
