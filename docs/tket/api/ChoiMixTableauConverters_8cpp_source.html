22 ChoiMixTableau circuit_to_cm_tableau(const Circuit& circ) {
23  ChoiMixTableau tab(circ.all_qubits());
24  for (const Qubit& q : circ.created_qubits()) {
25  tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
26  }
27  for (const Command& com : circ) {
28  auto args = com.get_args();
29  qubit_vector_t qbs = {args.begin(), args.end()};
30  tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
31  }
32  tab.rename_qubits(
33  circ.implicit_qubit_permutation(),
34  ChoiMixTableau::TableauSegment::Output);
35  for (const Qubit& q : circ.discarded_qubits()) {
36  tab.discard_qubit(q);
37  }
38  tab.canonical_column_order();
39  tab.gaussian_form();
40  return tab;
41 }

43 struct ChoiMixBuilder {
73  Circuit in_circ;
74  std::set<Qubit> post_selected;
75  std::set<Qubit> discarded;
76  std::set<Qubit> collapsed;
77  ChoiMixTableau tab;
78  boost::bimap<Qubit, Qubit> in_out_permutation;
79  std::set<Qubit> zero_initialised;
80  std::set<Qubit> mix_initialised;
81  Circuit out_circ_tp;
82 
83  // The CXConfigType preferred when invoking diagonalisation techniques
84  CXConfigType cx_config;
85 
86  // Additional qubit names (distinct from qubits already on the respective
87  // segment of the tableau) than can be used for zero initialisations and
88  // post-selections when synthesising a unitary extension
89  qubit_vector_t unitary_init_names;
90  qubit_vector_t unitary_post_names;
91 
92  // Initialises the builder with a tableau
93  explicit ChoiMixBuilder(const ChoiMixTableau& tab, CXConfigType cx_config);
94  // For synthesis of a unitary extension, initialises the builder with a
95  // tableau and some additional qubit names which the resulting circuit may
96  // optionally use, representing zero-initialised or post-selected qubits.
97  // These are the qubits on which we can freely add Zs to the rows of the given
98  // tableau to guarantee a unitary extension; none of these names should appear
99  // in tab
100  explicit ChoiMixBuilder(
101  const ChoiMixTableau& tab, CXConfigType cx_config,
102  const qubit_vector_t& init_names, const qubit_vector_t& post_names);
103 
104  // Debug method: applies all staged operations back onto tab to provide the
105  // tableau that the synthesis result is currently aiming towards. In exact
106  // synthesis, this should remain invariant during synthesis. For synthesis of
107  // a unitary extension, this should at least span the rows of the original
108  // tableau up to additional Zs on spare input and output qubits.
109  ChoiMixTableau realised_tableau() const;
110 
115  // Match up pairs of generators that anti-commute in the input segment but
116  // commute with all others; such pairs of rows reduce to an identity wire
117  // between a pair of qubits; we solve this by pairwise Pauli reduction methods
118  void solve_id_subspace();
119  // After removing the identity subspace, all remaining rows mutually commute
120  // within each tableau segment; diagonalise each segment individually
121  void diagonalise_segments();
122  // Solve the post-selected subspace which has already been diagonalised
123  void solve_postselected_subspace();
124  // Solve the zero-initialised subspace which has already been diagonalised
125  void solve_initialised_subspace();
126  // All remaining rows are in the collapsed subspace (each row is the unique
127  // stabilizer passing through some Collapse gate); solve it
128  void solve_collapsed_subspace();
129 
130  // Simplifies the tableau by removing qubits on which all rows have I; such
131  // qubits are either discarded inputs or mixed-initialised outputs
132  void remove_unused_qubits();
133  // For synthesis of a unitary extension, match up qubits from
134  // post-selected/zero-initialised with unitary_post_names/unitary_init_names
135  // and add them to in_out_permutation
136  void assign_init_post_names();
137  // Fill out in_out_permutation to map all qubits; this typically takes the
138  // form of a standard qubit reuse pattern (e.g. discard and reinitialise)
139  void assign_remaining_names();
140  // Once tab has been completely reduced to no rows and no qubits, compose the
141  // staged operations to build the output circuit and return the renaming map
142  // from output names of original tableau to the qubits of the returned circuit
143  // they are mapped to
144  std::pair<Circuit, qubit_map_t> output_circuit();
145  std::pair<Circuit, qubit_map_t> unitary_output_circuit();
146 };

148 std::pair<Circuit, qubit_map_t> cm_tableau_to_exact_circuit(
149  const ChoiMixTableau& tab, CXConfigType cx_config) {
150  ChoiMixBuilder builder(tab, cx_config);
151  builder.remove_unused_qubits();
152  builder.solve_id_subspace();
153  builder.diagonalise_segments();
154  builder.solve_postselected_subspace();
155  builder.solve_initialised_subspace();
156  builder.solve_collapsed_subspace();
157  builder.remove_unused_qubits();
158  builder.assign_remaining_names();
159  return builder.output_circuit();
160 }

181 ChoiMixBuilder::ChoiMixBuilder(
182  const ChoiMixTableau& t, CXConfigType cx, const qubit_vector_t& inits,
183  const qubit_vector_t& posts)
184  : in_circ(),
185  post_selected(),
186  discarded(),
187  collapsed(),
188  tab(t),
189  in_out_permutation(),
190  zero_initialised(),
191  mix_initialised(),
192  out_circ_tp(),
193  cx_config(cx),
194  unitary_init_names(inits),
195  unitary_post_names(posts) {
196  for (unsigned i = 0; i < tab.get_n_boundaries(); ++i) {
197  ChoiMixTableau::col_key_t key = tab.col_index_.right.at(i);
198  if (key.second == ChoiMixTableau::TableauSegment::Input)
199  in_circ.add_qubit(key.first);
200  else
201  out_circ_tp.add_qubit(key.first);
202  }
203  for (const Qubit& init_q : unitary_init_names) {
204  if (tab.col_index_.left.find(ChoiMixTableau::col_key_t{
205  init_q, ChoiMixTableau::TableauSegment::Input}) !=
206  tab.col_index_.left.end())
207  throw std::logic_error(
208  "Free qubit name for initialisation conflicts with existing live "
209  "input of ChoiMixTableau");
210  }
211  for (const Qubit& post_q : unitary_post_names) {
212  if (tab.col_index_.left.find(ChoiMixTableau::col_key_t{
213  post_q, ChoiMixTableau::TableauSegment::Output}) !=
214  tab.col_index_.left.end())
215  throw std::logic_error(
216  "Free qubit name for post-selection conflicts with existing live "
217  "output of ChoiMixTableau");
218  }
219 }

221 ChoiMixTableau ChoiMixBuilder::realised_tableau() const {
222  ChoiMixTableau in_tab = circuit_to_cm_tableau(in_circ);
223  for (const Qubit& q : post_selected)
224  in_tab.post_select(q, ChoiMixTableau::TableauSegment::Output);
225  for (const Qubit& q : discarded)
226  in_tab.discard_qubit(q, ChoiMixTableau::TableauSegment::Output);
227  for (const Qubit& q : collapsed)
228  in_tab.collapse_qubit(q, ChoiMixTableau::TableauSegment::Output);
229  ChoiMixTableau out_tab = circuit_to_cm_tableau(out_circ_tp.transpose());
230  for (const Qubit& q : zero_initialised)
231  out_tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
232  for (const Qubit& q : mix_initialised)
233  out_tab.discard_qubit(q, ChoiMixTableau::TableauSegment::Input);
234  qubit_map_t out_in_permutation{};
235  using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
236  BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
237  out_in_permutation.insert({entry.second, entry.first});
238  }
239  out_tab.rename_qubits(
240  out_in_permutation, ChoiMixTableau::TableauSegment::Output);
241  return ChoiMixTableau::compose(ChoiMixTableau::compose(in_tab, tab), out_tab);
242 }

244 void ChoiMixBuilder::solve_id_subspace() {
245  // Input-first gaussian elimination to solve input-sides of remaining rows
246  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
247  tab.gaussian_form();
248 
249  std::set<unsigned> solved_rows;
250  std::set<Qubit> solved_ins, solved_outs;
251  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
252  if (solved_rows.find(r) != solved_rows.end()) continue;
253 
254  // Look for a row which anticommutes with row r over the inputs
255  std::list<unsigned> xcols, zcols;
256  for (unsigned c = 0; c < tab.get_n_inputs(); ++c) {
257  if (tab.tab_.xmat(r, c)) xcols.push_back(c);
258  if (tab.tab_.zmat(r, c)) zcols.push_back(c);
259  }
260  for (unsigned r2 = r + 1; r2 < tab.get_n_rows(); ++r2) {
261  if (solved_rows.find(r2) != solved_rows.end()) continue;
262  bool anti = false;
263  for (const unsigned c : xcols) anti ^= tab.tab_.zmat(r2, c);
264  for (const unsigned c : zcols) anti ^= tab.tab_.xmat(r2, c);
265  if (!anti) continue;
266 
267  // Found a candidate pair of rows. Because of the Gaussian elimination, it
268  // is more likely that the first mismatching qubit is X for r and Z for
269  // r2, so favour reducing r2 to Z and r to X
270  ChoiMixTableau::row_tensor_t row_r = tab.get_row(r);
271  ChoiMixTableau::row_tensor_t row_r2 = tab.get_row(r2);
272  std::pair<Circuit, Qubit> in_diag_circ =
273  reduce_anticommuting_paulis_to_z_x(
274  row_r2.first, row_r.first, cx_config);
275  in_circ.append(in_diag_circ.first);
276  for (const Command& com : in_diag_circ.first) {
277  auto args = com.get_args();
278  qubit_vector_t qbs = {args.begin(), args.end()};
279  tab.apply_gate(
280  com.get_op_ptr()->dagger()->get_type(), qbs,
281  ChoiMixTableau::TableauSegment::Input);
282  }
283 
284  // Since the full rows must commute but they anticommute over the inputs,
285  // they must also anticommute over the outputs; we similarly reduce these
286  // down to Z and X
287  std::pair<Circuit, Qubit> out_diag_circ_dag =
288  reduce_anticommuting_paulis_to_z_x(
289  row_r2.second, row_r.second, cx_config);
290  out_circ_tp.append(out_diag_circ_dag.first.dagger().transpose());
291  for (const Command& com : out_diag_circ_dag.first) {
292  auto args = com.get_args();
293  qubit_vector_t qbs = {args.begin(), args.end()};
294  tab.apply_gate(
295  com.get_op_ptr()->get_type(), qbs,
296  ChoiMixTableau::TableauSegment::Output);
297  }
298 
299  // Check that rows have been successfully reduced
300  row_r = tab.get_row(r);
301  row_r2 = tab.get_row(r2);
302  if (row_r.first.size() != 1 ||
303  row_r.first.string.begin()->second != Pauli::X ||
304  row_r.second.size() != 1 ||
305  row_r.second.string.begin()->second != Pauli::X ||
306  row_r2.first.size() != 1 ||
307  row_r2.first.string.begin()->second != Pauli::Z ||
308  row_r2.second.size() != 1 ||
309  row_r2.second.string.begin()->second != Pauli::Z)
310  throw std::logic_error(
311  "Unexpected error during identity reduction in ChoiMixTableau "
312  "synthesis");
313  // Solve phases
314  if (row_r.second.is_real_negative()) {
315  in_circ.add_op<Qubit>(OpType::Z, {in_diag_circ.second});
316  tab.apply_gate(
317  OpType::Z, {in_diag_circ.second},
318  ChoiMixTableau::TableauSegment::Input);
319  }
320  if (row_r2.second.is_real_negative()) {
321  in_circ.add_op<Qubit>(OpType::X, {in_diag_circ.second});
322  tab.apply_gate(
323  OpType::X, {in_diag_circ.second},
324  ChoiMixTableau::TableauSegment::Input);
325  }
326  // Connect in permutation
327  in_out_permutation.insert(
328  {in_diag_circ.second, out_diag_circ_dag.second});
329  solved_rows.insert(r);
330  solved_rows.insert(r2);
331 
332  // Remove these solved qubits from other rows; by commutation of rows, a
333  // row contains Z@in_diag_circ.second iff it contains
334  // Z@out_diag_circ_dag.second and similarly for X
335  unsigned in_c = tab.col_index_.left.at(ChoiMixTableau::col_key_t{
336  in_diag_circ.second, ChoiMixTableau::TableauSegment::Input});
337  for (unsigned r3 = 0; r3 < tab.get_n_rows(); ++r3) {
338  if (r3 != r && tab.tab_.xmat(r3, in_c)) tab.tab_.row_mult(r, r3);
339  if (r3 != r2 && tab.tab_.zmat(r3, in_c)) tab.tab_.row_mult(r2, r3);
340  }
341  solved_ins.insert({in_diag_circ.second});
342  solved_outs.insert({out_diag_circ_dag.second});
343  break;
344  }
345  }
346 
347  // Remove solved rows and qubits from tableau; since removing rows/columns
348  // replaces them with the row/column from the end, remove in reverse order
349  for (auto it = solved_rows.rbegin(); it != solved_rows.rend(); ++it)
350  tab.remove_row(*it);
351  for (auto it = solved_ins.rbegin(); it != solved_ins.rend(); ++it)
352  tab.discard_qubit(*it, ChoiMixTableau::TableauSegment::Input);
353  for (auto it = solved_outs.rbegin(); it != solved_outs.rend(); ++it)
354  tab.discard_qubit(*it, ChoiMixTableau::TableauSegment::Output);
355 }

357 // Given a matrix that is already in upper echelon form, use the fact that the
358 // leading columns are already unique to give column operations that reduce it
359 // down to identity over the leading columns, eliminating extra swap gates to
360 // move to the first spaces
361 static std::vector<std::pair<unsigned, unsigned>>
362 leading_column_gaussian_col_ops(const MatrixXb& source) {
363  std::vector<unsigned> col_list;
364  std::set<unsigned> non_leads;
365  for (unsigned r = 0; r < source.rows(); ++r) {
366  bool leading_found = false;
367  for (unsigned c = 0; c < source.cols(); ++c) {
368  if (source(r, c)) {
369  if (leading_found)
370  non_leads.insert(c);
371  else {
372  leading_found = true;
373  col_list.push_back(c);
374  }
375  }
376  }
377  }
378  for (const unsigned& c : non_leads) col_list.push_back(c);
379  MatrixXb reordered = MatrixXb::Zero(source.rows(), col_list.size());
380  for (unsigned c = 0; c < col_list.size(); ++c)
381  reordered.col(c) = source.col(col_list.at(c));
382  std::vector<std::pair<unsigned, unsigned>> reordered_ops =
383  gaussian_elimination_col_ops(reordered);
384  std::vector<std::pair<unsigned, unsigned>> res;
385  for (const std::pair<unsigned, unsigned>& op : reordered_ops)
386  res.push_back({col_list.at(op.first), col_list.at(op.second)});
387  return res;
388 }

390 void ChoiMixBuilder::diagonalise_segments() {
391  // Canonicalise tableau
392  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
393  tab.gaussian_form();
394 
395  // Set up diagonalisation tasks
396  std::list<SpSymPauliTensor> to_diag_ins, to_diag_outs;
397  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
398  ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
399  if (!rten.first.string.empty()) to_diag_ins.push_back(rten.first);
400  if (!rten.second.string.empty()) to_diag_outs.push_back(rten.second);
401  }
402  qubit_vector_t input_qubits = tab.input_qubits();
403  std::set<Qubit> diag_ins{input_qubits.begin(), input_qubits.end()};
404  Circuit in_diag_circ = mutual_diagonalise(to_diag_ins, diag_ins, cx_config);
405  for (const Command& com : in_diag_circ) {
406  auto args = com.get_args();
407  in_circ.add_op<UnitID>(com.get_op_ptr(), args);
408  qubit_vector_t qbs = {args.begin(), args.end()};
409  tab.apply_gate(
410  com.get_op_ptr()->dagger()->get_type(), qbs,
411  ChoiMixTableau::TableauSegment::Input);
412  }
413  qubit_vector_t output_qubits = tab.output_qubits();
414  std::set<Qubit> diag_outs{output_qubits.begin(), output_qubits.end()};
415  Circuit out_diag_circ =
416  mutual_diagonalise(to_diag_outs, diag_outs, cx_config);
417  for (const Command& com : out_diag_circ) {
418  auto args = com.get_args();
419  out_circ_tp.add_op<UnitID>(com.get_op_ptr()->dagger()->transpose(), args);
420  qubit_vector_t qbs = {args.begin(), args.end()};
421  tab.apply_gate(
422  com.get_op_ptr()->get_type(), qbs,
423  ChoiMixTableau::TableauSegment::Output);
424  }
425 
426  // All rows are diagonalised, so we can just focus on the Z matrix
427  if (tab.tab_.xmat != MatrixXb::Zero(tab.get_n_rows(), tab.get_n_boundaries()))
428  throw std::logic_error(
429  "Diagonalisation in ChoiMixTableau synthesis failed");
430 }

432 void ChoiMixBuilder::solve_postselected_subspace() {
433  // As column order is currently output first, gaussian form will reveal the
434  // post-selected space at the bottom of the tableau and the submatrix of those
435  // rows will already be in upper echelon form
436  tab.gaussian_form();
437  // Reduce them to a minimal set of qubits using CX gates
438  unsigned n_postselected = 0;
439  for (; n_postselected < tab.get_n_rows(); ++n_postselected) {
440  if (!tab.get_row(tab.get_n_rows() - 1 - n_postselected)
441  .second.string.empty())
442  break;
443  }
444  unsigned n_ins = tab.get_n_inputs();
445  unsigned n_outs = tab.get_n_outputs();
446  MatrixXb subtableau = tab.tab_.zmat.bottomRightCorner(n_postselected, n_ins);
447  std::vector<std::pair<unsigned, unsigned>> col_ops =
448  leading_column_gaussian_col_ops(subtableau);
449  for (const std::pair<unsigned, unsigned>& op : col_ops) {
450  unsigned tab_ctrl_col = n_outs + op.second;
451  unsigned tab_trgt_col = n_outs + op.first;
452  tab.tab_.apply_CX(tab_ctrl_col, tab_trgt_col);
453  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(tab_ctrl_col);
454  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(tab_trgt_col);
455  in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
456  }
457  // Postselect rows
458  for (unsigned r = 0; r < n_postselected; ++r) {
459  unsigned final_row = tab.get_n_rows() - 1;
460  ChoiMixTableau::row_tensor_t row = tab.get_row(final_row);
461  if (row.second.size() != 0 || row.first.size() != 1 ||
462  row.first.string.begin()->second != Pauli::Z)
463  throw std::logic_error(
464  "Unexpected error during post-selection identification in "
465  "ChoiMixTableau synthesis");
466  Qubit post_selected_qb = row.first.string.begin()->first;
467  // Multiply other rows to remove Z_qb components
468  unsigned qb_col = tab.col_index_.left.at(ChoiMixTableau::col_key_t{
469  post_selected_qb, ChoiMixTableau::TableauSegment::Input});
470  for (unsigned s = 0; s < final_row; ++s)
471  if (tab.tab_.zmat(s, qb_col)) tab.tab_.row_mult(final_row, s);
472  // Post-select on correct phase
473  if (row.second.is_real_negative())
474  in_circ.add_op<Qubit>(OpType::X, {post_selected_qb});
475  tab.remove_row(final_row);
476  post_selected.insert(post_selected_qb);
477  tab.discard_qubit(post_selected_qb, ChoiMixTableau::TableauSegment::Input);
478  }
479 }

481 void ChoiMixBuilder::solve_initialised_subspace() {
482  // Input-first gaussian elimination now sorts the remaining rows into the
483  // collapsed subspace followed by the zero-initialised subspace and the
484  // collapsed subspace rows are in upper echelon form over the inputs, giving
485  // unique leading columns and allowing us to solve them with CXs by
486  // column-wise gaussian elimination; same for zero-initialised rows over the
487  // outputs
488  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
489  tab.gaussian_form();
490 
491  // Reduce the zero-initialised space to a minimal set of qubits using CX gates
492  unsigned n_collapsed = 0;
493  for (; n_collapsed < tab.get_n_rows(); ++n_collapsed) {
494  if (tab.get_row(n_collapsed).first.string.empty()) break;
495  }
496  unsigned n_ins = tab.get_n_inputs();
497  unsigned n_outs = tab.get_n_outputs();
498  MatrixXb subtableau =
499  tab.tab_.zmat.bottomRightCorner(tab.get_n_rows() - n_collapsed, n_outs);
500  std::vector<std::pair<unsigned, unsigned>> col_ops =
501  leading_column_gaussian_col_ops(subtableau);
502  for (const std::pair<unsigned, unsigned>& op : col_ops) {
503  unsigned tab_ctrl_col = n_ins + op.second;
504  unsigned tab_trgt_col = n_ins + op.first;
505  tab.tab_.apply_CX(tab_ctrl_col, tab_trgt_col);
506  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(tab_ctrl_col);
507  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(tab_trgt_col);
508  out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
509  }
510  // Initialise rows
511  for (unsigned r = tab.get_n_rows(); r-- > n_collapsed;) {
512  // r always refers to the final row in the tableau
513  ChoiMixTableau::row_tensor_t row = tab.get_row(r);
514  if (row.first.size() != 0 || row.second.size() != 1 ||
515  row.second.string.begin()->second != Pauli::Z)
516  throw std::logic_error(
517  "Unexpected error during initialisation identification in "
518  "ChoiMixTableau synthesis");
519  Qubit initialised_qb = row.second.string.begin()->first;
520  // Multiply other rows to remove Z_qb components
521  unsigned qb_col = tab.col_index_.left.at(ChoiMixTableau::col_key_t{
522  initialised_qb, ChoiMixTableau::TableauSegment::Output});
523  for (unsigned s = 0; s < r; ++s)
524  if (tab.tab_.zmat(s, qb_col)) tab.tab_.row_mult(r, s);
525  // Initialise with correct phase
526  if (row.second.is_real_negative())
527  out_circ_tp.add_op<Qubit>(OpType::X, {initialised_qb});
528  tab.remove_row(r);
529  zero_initialised.insert(initialised_qb);
530  tab.discard_qubit(initialised_qb, ChoiMixTableau::TableauSegment::Output);
531  }

534 void ChoiMixBuilder::solve_collapsed_subspace() {
535  // Solving the initialised subspace will have preserved the upper echelon form
536  // of the collapsed subspace; reduce the inputs of the collapsed space to a
537  // minimal set of qubits using CX gates
538  unsigned n_ins = tab.get_n_inputs();
539  unsigned n_outs = tab.get_n_outputs();
540  MatrixXb subtableau = tab.tab_.zmat.topLeftCorner(tab.get_n_rows(), n_ins);
541  std::vector<std::pair<unsigned, unsigned>> col_ops =
542  leading_column_gaussian_col_ops(subtableau);
543  for (const std::pair<unsigned, unsigned>& op : col_ops) {
544  tab.tab_.apply_CX(op.second, op.first);
545  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.second);
546  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.first);
547  in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
548  }
549  // Since row multiplications will unsolve the inputs, we cannot get the output
550  // segment into upper echelon form for the same CX-saving trick; instead we
551  // accept just removing any qubits that are now unused after solving the
552  // initialised subspace
553  remove_unused_qubits();
554  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
555  // Solve the output segment using CX gates
556  n_ins = tab.get_n_inputs();
557  n_outs = tab.get_n_outputs();
558  col_ops = gaussian_elimination_col_ops(
559  tab.tab_.zmat.topRightCorner(tab.get_n_rows(), n_outs));
560  for (const std::pair<unsigned, unsigned>& op : col_ops) {
561  tab.tab_.apply_CX(n_ins + op.second, n_ins + op.first);
562  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(n_ins + op.second);
563  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(n_ins + op.first);
564  out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
565  }
566  // Connect up and remove rows and columns
567  for (unsigned r = tab.get_n_rows(); r-- > 0;) {
568  // r refers to the final row
569  // Check that row r has been successfully reduced
570  ChoiMixTableau::row_tensor_t row_r = tab.get_row(r);
571  if (row_r.first.size() != 1 ||
572  row_r.first.string.begin()->second != Pauli::Z ||
573  row_r.second.size() != 1 ||
574  row_r.second.string.begin()->second != Pauli::Z)
575  throw std::logic_error(
576  "Unexpected error during collapsed subspace reduction in "
577  "ChoiMixTableau synthesis");
578  Qubit in_q = row_r.first.string.begin()->first;
579  Qubit out_q = row_r.second.string.begin()->first;
580  // Solve phase
581  if (row_r.second.is_real_negative()) {
582  in_circ.add_op<Qubit>(OpType::X, {in_q});
583  tab.apply_gate(OpType::X, {in_q}, ChoiMixTableau::TableauSegment::Input);
584  }
585  // Connect in permutation
586  in_out_permutation.insert({in_q, out_q});
587  collapsed.insert(in_q);
588  tab.remove_row(r);
589  tab.discard_qubit(in_q, ChoiMixTableau::TableauSegment::Input);
590  tab.discard_qubit(out_q, ChoiMixTableau::TableauSegment::Output);
591  }
592 }

594 void ChoiMixBuilder::remove_unused_qubits() {
595  // Since removing a column replaces it with the last column, remove in reverse
596  // order to examine each column exactly once
597  for (unsigned c = tab.get_n_boundaries(); c-- > 0;) {
598  bool used = false;
599  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
600  if (tab.tab_.zmat(r, c) || tab.tab_.xmat(r, c)) {
601  used = true;
602  break;
603  }
604  }
605  if (used) continue;
606  ChoiMixTableau::col_key_t col = tab.col_index_.right.at(c);
607  if (col.second == ChoiMixTableau::TableauSegment::Input)
608  discarded.insert(col.first);
609  else
610  mix_initialised.insert(col.first);
611  tab.discard_qubit(col.first, col.second);
612  }
613 }

616  auto it = unitary_post_names.begin();
617  for (const Qubit& ps : post_selected) {
618  if (it == unitary_post_names.end())
619  throw std::logic_error(
620  "Not enough additional qubit names for unitary extension of "
621  "ChoiMixTableau to safely handle post-selected subspace");
622  in_out_permutation.insert({ps, *it});
623  ++it;

626 
627  it = unitary_init_names.begin();
628  for (const Qubit& zi : zero_initialised) {
629  if (it == unitary_init_names.end())
630  throw std::logic_error(
631  "Not enough additional qubit names for unitary extension of "
632  "ChoiMixTableau to safely handle initialised subspace");
633  in_out_permutation.insert({*it, zi});
634  ++it;
635  }
636  unitary_init_names = {it, unitary_init_names.end()};

639 void ChoiMixBuilder::assign_remaining_names() {
640  // Some post-selected or initialised qubits might have already been matched up
641  // for unitary synthesis, so we only need to match up the remainder
642  std::set<Qubit> unsolved_ins = discarded;
643  for (const Qubit& q : post_selected) {
644  if (in_out_permutation.left.find(q) == in_out_permutation.left.end())
645  unsolved_ins.insert(q);
646  }
647  std::set<Qubit> unsolved_outs = mix_initialised;
648  for (const Qubit& q : zero_initialised) {
649  if (in_out_permutation.right.find(q) == in_out_permutation.right.end())
650  unsolved_outs.insert(q);
651  }
652  // If there are more unsolved_ins than unsolved_outs, we want to pad out
653  // unsolved_outs with extra names that don't appear as output names of the
654  // original tableau; between unsolved_ins and the inputs already in
655  // in_out_permutation, there will be at least enough of these
656  if (unsolved_ins.size() > unsolved_outs.size()) {
657  for (const Qubit& q : unsolved_ins) {
658  if (in_out_permutation.right.find(q) == in_out_permutation.right.end()) {
659  unsolved_outs.insert(q);
660  if (unsolved_ins.size() == unsolved_outs.size()) break;
661  }
662  }
663  if (unsolved_ins.size() > unsolved_outs.size()) {
664  using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
665  BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
666  if (in_out_permutation.right.find(entry.first) ==
667  in_out_permutation.right.end()) {
668  unsolved_outs.insert(entry.first);
669  if (unsolved_ins.size() == unsolved_outs.size()) break;
670  }
671  }
672  }
673  } else if (unsolved_ins.size() < unsolved_outs.size()) {
674  for (const Qubit& q : unsolved_outs) {
675  if (in_out_permutation.left.find(q) == in_out_permutation.left.end()) {
676  unsolved_ins.insert(q);
677  if (unsolved_ins.size() == unsolved_outs.size()) break;
678  }
679  }
680  if (unsolved_ins.size() < unsolved_outs.size()) {
681  using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
682  BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
683  if (in_out_permutation.left.find(entry.second) ==
684  in_out_permutation.left.end()) {
685  unsolved_ins.insert(entry.second);
686  if (unsolved_ins.size() == unsolved_outs.size()) break;
687  }
688  }
689  }
690  }
691  // Prefer to connect qubits with the same names
692  for (auto in_it = unsolved_ins.begin(); in_it != unsolved_ins.end();) {
693  auto temp_it = in_it++;
694  auto out_it = unsolved_outs.find(*temp_it);
695  if (out_it != unsolved_outs.end()) {
696  in_out_permutation.insert({*temp_it, *temp_it});
697  unsolved_ins.erase(temp_it);
698  unsolved_outs.erase(out_it);
699  }
700  }
701  // Pair up remainders; by our earlier padding, they should have the exact same
702  // number of elements, so pair them up exactly
703  for (const Qubit& in : unsolved_ins) {
704  auto it = unsolved_outs.begin();
705  in_out_permutation.insert({in, *it});
706  unsolved_outs.erase(it);
707  }
708 }

710 std::pair<Circuit, qubit_map_t> ChoiMixBuilder::output_circuit() {
711  if (tab.get_n_rows() != 0 || tab.get_n_boundaries() != 0)
712  throw std::logic_error(
713  "Unexpected error during ChoiMixTableau synthesis, reached the end "
714  "with a non-empty tableau remaining");
715  if (!post_selected.empty()) {
716  throw std::logic_error(
717  "Not yet implemented: post-selection required during ChoiMixTableau "
718  "synthesis");
719  }
720  for (const Qubit& q : discarded) in_circ.qubit_discard(q);
721  for (const Qubit& q : collapsed) in_circ.add_op<Qubit>(OpType::Collapse, {q});
722  Circuit out_circ(out_circ_tp.all_qubits(), {});
723  for (const Qubit& q : zero_initialised) out_circ.qubit_create(q);
724  for (const Qubit& q : mix_initialised) {
725  out_circ.qubit_create(q);
726  out_circ.add_op<Qubit>(OpType::H, {q});
727  out_circ.add_op<Qubit>(OpType::Collapse, {q});
728  }
729  out_circ.append(out_circ_tp.transpose());
730  qubit_map_t return_perm;
731  unit_map_t append_perm;
732  using perm_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
733  BOOST_FOREACH (perm_entry entry, in_out_permutation.left) {
734  return_perm.insert({entry.second, entry.first});
735  append_perm.insert({entry.second, entry.first});
736  }
737  in_circ.append_with_map(out_circ, append_perm);
738  return {in_circ, return_perm};
739 }
