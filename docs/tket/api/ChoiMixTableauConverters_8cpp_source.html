23 ChoiMixTableau circuit_to_cm_tableau(const Circuit& circ) {
24  ChoiMixTableau tab(circ.all_qubits());
25  for (const Qubit& q : circ.created_qubits()) {
26  tab.post_select(q, ChoiMixTableau::TableauSegment::Input);
27  }
28  for (const Command& com : circ) {
29  auto args = com.get_args();
30  qubit_vector_t qbs = {args.begin(), args.end()};
31  tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
32  }
33  for (const Qubit& q : circ.discarded_qubits()) {
34  tab.discard_qubit(q);
35  }
36  return tab;
37 }

39 std::pair<Circuit, unit_map_t> cm_tableau_to_circuit(const ChoiMixTableau& t) {
78  // Operate on a copy of the tableau to track the remainder as we extract gates
79  ChoiMixTableau tab(t);
80 
81  // Canonicalise tableau and perform output-first gaussian elimination to
82  // isolate post-selected subspace in lower rows
83  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
84  tab.gaussian_form();
85 
86  // Set up circuits for extracting gates into (in_circ is set up after
87  // diagonalising the post-selected subspace)
88  qubit_vector_t input_qubits, output_qubits;
89  for (unsigned i = 0; i < tab.get_n_boundaries(); ++i) {
90  ChoiMixTableau::col_key_t key = tab.col_index_.right.at(i);
91  if (key.second == ChoiMixTableau::TableauSegment::Input)
92  input_qubits.push_back(key.first);
93  else
94  output_qubits.push_back(key.first);
95  }
96  Circuit out_circ_tp(output_qubits, {});
97  unit_map_t join_permutation;
98  std::set<Qubit> post_selected;
99  std::map<Qubit, std::optional<unsigned>> in_x_row;
100  std::map<Qubit, std::optional<unsigned>> in_z_row;
101  boost::bimap<Qubit, Qubit> matched_qubits;
102 
103  // Call diagonalisation methods to diagonalise post-selected subspace
104  std::list<std::pair<QubitPauliTensor, Expr>> to_diag;
105  for (unsigned r = tab.get_n_rows(); r > 0;) {
106  --r;
107  ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
108  if (rten.second.string.map.size() != 0) {
109  // Reached the rows with non-empty output segment
110  break;
111  }
112  // Else, we add the row to the subspace
113  to_diag.push_back({rten.first, 1.});
114  }
115  unsigned post_selected_size = to_diag.size();
116  std::set<Qubit> diag_ins{input_qubits.begin(), input_qubits.end()};
117  Circuit in_circ = mutual_diagonalise(to_diag, diag_ins, CXConfigType::Tree);
118  // Extract the dagger of each gate in order from tab
119  for (const Command& com : in_circ) {
120  auto args = com.get_args();
121  qubit_vector_t qbs = {args.begin(), args.end()};
122  tab.apply_gate(
123  com.get_op_ptr()->dagger()->get_type(), qbs,
124  ChoiMixTableau::TableauSegment::Input);
125  }
126 
127  // Diagonalised rows should still be at the bottom of the tableau - reduce
128  // them to a minimal set of qubits for post-selection by first reducing to
129  // upper echelon form
130  std::vector<std::pair<unsigned, unsigned>> row_ops =
131  gaussian_elimination_row_ops(
132  tab.tab_.zmat_.bottomRows(post_selected_size));
133  for (const std::pair<unsigned, unsigned>& op : row_ops) {
134  tab.tab_.row_mult(op.first, op.second);
135  }
136  // Obtain CX instructions as column operations
137  std::vector<std::pair<unsigned, unsigned>> col_ops =
138  gaussian_elimination_col_ops(
139  tab.tab_.zmat_.bottomRows(post_selected_size));
140  // These gates will also swap qubits to isolate the post-selections on the
141  // first few qubits - this is fine as can be cleaned up later with peephole
142  // optimisations
143  for (const std::pair<unsigned, unsigned>& op : col_ops) {
144  tab.tab_.apply_CX(op.first, op.second);
145  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.first);
146  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.second);
147  in_circ.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
148  }
149 
150  // Post-select rows
151  // TODO Post-selection op is not yet available in tket - replace this once
152  // implemented; an implementation hint is provided below
153  if (post_selected_size != 0)
154  throw std::logic_error(
155  "Not yet implemented: post-selection required during ChoiMixTableau "
156  "synthesis");
157  // for (unsigned r = 0; r < post_selected_size; ++r) {
158  // ChoiMixTableau::row_tensor_t row = tab.get_row(tab.get_n_rows() - 1);
159  // if (row.second.string.map.size() != 0 || row.first.string.map.size() != 1
160  // || row.first.string.map.begin()->second != Pauli::Z) throw
161  // std::logic_error("Unexpected error during post-selection identification
162  // in ChoiMixTableau synthesis"); Qubit post_selected_qb =
163  // row.first.string.map.begin()->first; if (row.second.coeff == -1.)
164  // in_circ.add_op<Qubit>(OpType::X, {post_selected_qb});
165  // tab.remove_row(tab.get_n_rows() - 1);
166  // post_selected.insert(post_selected_qb);
167  // throw std::logic_error("Not yet implemented: post-selection required
168  // during ChoiMixTableau synthesis");
169  // }
170 
171  // Input-first gaussian elimination to solve input-sides of remaining rows
172  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Input);
173  tab.gaussian_form();
174 
175  // Iterate through remaining inputs and reduce output portion to a single
176  // qubit
177  for (const Qubit& in_qb : input_qubits) {
178  // Skip post-selected qubits
179  if (post_selected.find(in_qb) != post_selected.end()) continue;
180 
181  unsigned col = tab.col_index_.left.at(ChoiMixTableau::col_key_t{
182  in_qb, ChoiMixTableau::TableauSegment::Input});
183  std::optional<Qubit> out_qb = std::nullopt;
184 
185  // Find the row with X_in_qb (if one exists)
186  std::optional<unsigned> x_row = std::nullopt;
187  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
188  if (tab.tab_.xmat_(r, col)) {
189  x_row = r;
190  break;
191  }
192  }
193 
194  if (x_row) {
195  // A possible optimisation could involve row multiplications to reduce the
196  // Hamming weight of the row before applying gates, but minimising this
197  // would solve minimum weight/distance of a binary linear code whose
198  // decision problem is NP-complete (shown by Vardy, "The intractability of
199  // computing the minimum distance of a code", 1997). Just settle on using
200  // the first row for now, reducing the input and output to a single qubit
201  ChoiMixTableau::row_tensor_t row_paulis = tab.get_row(*x_row);
202  for (const std::pair<const Qubit, Pauli>& p :
203  row_paulis.second.string.map) {
204  if (matched_qubits.right.find(p.first) == matched_qubits.right.end()) {
205  out_qb = p.first;
206  matched_qubits.insert({in_qb, *out_qb});
207  break;
208  }
209  }
210 
211  // Reduce input string to just X_in_qb
212  if (row_paulis.first.string.get(in_qb) == Pauli::Y) {
213  // If it is a Y, extract an Sdg gate so the Pauli is exactly X
214  in_circ.add_op<Qubit>(OpType::Sdg, {in_qb});
215  tab.apply_S(in_qb, ChoiMixTableau::TableauSegment::Input);
216  }
217  for (const std::pair<const Qubit, Pauli>& qbp :
218  row_paulis.first.string.map) {
219  if (qbp.first == in_qb) continue;
220  // Extract an entangling gate to eliminate the qubit
221  switch (qbp.second) {
222  case Pauli::X: {
223  in_circ.add_op<Qubit>(OpType::CX, {in_qb, qbp.first});
224  tab.apply_CX(
225  in_qb, qbp.first, ChoiMixTableau::TableauSegment::Input);
226  break;
227  }
228  case Pauli::Y: {
229  in_circ.add_op<Qubit>(OpType::CY, {in_qb, qbp.first});
230  tab.apply_gate(
231  OpType::CY, {in_qb, qbp.first},
232  ChoiMixTableau::TableauSegment::Input);
233  break;
234  }
235  case Pauli::Z: {
236  in_circ.add_op<Qubit>(OpType::CZ, {in_qb, qbp.first});
237  tab.apply_gate(
238  OpType::CZ, {in_qb, qbp.first},
239  ChoiMixTableau::TableauSegment::Input);
240  break;
241  }
242  default: {
243  break;
244  }
245  }
246  }
247 
248  // And then the same for X_out_qb
249  if (row_paulis.second.string.get(*out_qb) == Pauli::Y) {
250  // If it is a Y, extract an Sdg gate so the Pauli is exactly X
251  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
252  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
253  } else if (row_paulis.second.string.get(*out_qb) == Pauli::Z) {
254  // If it is a Z, extract an Vdg and Sdg gate so the Pauli is exactly X
255  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
256  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
257  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
258  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
259  }
260  for (const std::pair<const Qubit, Pauli>& qbp :
261  row_paulis.second.string.map) {
262  if (qbp.first == *out_qb) continue;
263  // Extract an entangling gate to eliminate the qubit
264  switch (qbp.second) {
265  case Pauli::X: {
266  out_circ_tp.add_op<Qubit>(OpType::CX, {*out_qb, qbp.first});
267  tab.apply_CX(
268  *out_qb, qbp.first, ChoiMixTableau::TableauSegment::Output);
269  break;
270  }
271  case Pauli::Y: {
272  // CY does not have a transpose OpType defined so decompose
273  out_circ_tp.add_op<Qubit>(OpType::S, {qbp.first});
274  out_circ_tp.add_op<Qubit>(OpType::CX, {*out_qb, qbp.first});
275  out_circ_tp.add_op<Qubit>(OpType::Sdg, {qbp.first});
276  tab.apply_gate(
277  OpType::CY, {*out_qb, qbp.first},
278  ChoiMixTableau::TableauSegment::Output);
279  break;
280  }
281  case Pauli::Z: {
282  out_circ_tp.add_op<Qubit>(OpType::CZ, {*out_qb, qbp.first});
283  tab.apply_gate(
284  OpType::CZ, {*out_qb, qbp.first},
285  ChoiMixTableau::TableauSegment::Output);
286  break;
287  }
288  default: {
289  break;
290  }
291  }
292  }
293  }
294 
295  // Find the row with Z_in_qb (if one exists)
296  std::optional<unsigned> z_row = std::nullopt;
297  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
298  if (tab.tab_.zmat_(r, col)) {
299  z_row = r;
300  break;
301  }
302  }
303  if (z_row) {
304  // If both an X and Z row exist, then out_qb should have a
305  // value and the rows should have anticommuting Paulis on out_qb to
306  // preserve commutativity of rows
307  ChoiMixTableau::row_tensor_t row_paulis = tab.get_row(*z_row);
308  if (!x_row) {
309  for (const std::pair<const Qubit, Pauli>& p :
310  row_paulis.second.string.map) {
311  if (matched_qubits.right.find(p.first) ==
312  matched_qubits.right.end()) {
313  out_qb = p.first;
314  matched_qubits.insert({in_qb, *out_qb});
315  break;
316  }
317  }
318  }
319 
320  // Reduce input string to just Z_in_qb.
321  // No need to consider different paulis on in_qb: if we had X or Y, this
322  // row would have been identified as x_row instead of Z row, and if
323  // another row was already chosen as x_row then canonical gaussian form
324  // would imply all other rows do not contain X_in_qb or Y_in_qb
325  for (const std::pair<const Qubit, Pauli>& qbp :
326  row_paulis.first.string.map) {
327  if (qbp.first == in_qb) continue;
328  // Extract an entangling gate to eliminate the qubit
329  switch (qbp.second) {
330  case Pauli::X: {
331  in_circ.add_op<Qubit>(OpType::H, {qbp.first});
332  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
333  tab.apply_gate(
334  OpType::H, {qbp.first}, ChoiMixTableau::TableauSegment::Input);
335  tab.apply_CX(
336  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
337  break;
338  }
339  case Pauli::Y: {
340  in_circ.add_op<Qubit>(OpType::Vdg, {qbp.first});
341  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
342  tab.apply_V(qbp.first, ChoiMixTableau::TableauSegment::Input);
343  tab.apply_CX(
344  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
345  break;
346  }
347  case Pauli::Z: {
348  in_circ.add_op<Qubit>(OpType::CX, {qbp.first, in_qb});
349  tab.apply_CX(
350  qbp.first, in_qb, ChoiMixTableau::TableauSegment::Input);
351  break;
352  }
353  default: {
354  break;
355  }
356  }
357  }
358 
359  // And then reduce output string to just Z_out_qb
360  if (row_paulis.second.string.get(*out_qb) == Pauli::Y) {
361  // If it is a Y, extract a Vdg gate so the Pauli is exactly Z
362  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
363  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
364  } else if (row_paulis.second.string.get(*out_qb) == Pauli::X) {
365  // If it is an X, extract an Sdg and Vdg gate so the Pauli is exactly Z
366  // We do not need to care about messing up the X row here since if we
367  // solved an X row then this row can't also have X by commutativity
368  out_circ_tp.add_op<Qubit>(OpType::Sdg, {*out_qb});
369  out_circ_tp.add_op<Qubit>(OpType::Vdg, {*out_qb});
370  tab.apply_S(*out_qb, ChoiMixTableau::TableauSegment::Output);
371  tab.apply_V(*out_qb, ChoiMixTableau::TableauSegment::Output);
372  }
373  for (const std::pair<const Qubit, Pauli>& qbp :
374  row_paulis.second.string.map) {
375  if (qbp.first == *out_qb) continue;
376  // Extract an entangling gate to eliminate the qubit
377  switch (qbp.second) {
378  case Pauli::X: {
379  out_circ_tp.add_op<Qubit>(OpType::H, {qbp.first});
380  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
381  tab.apply_gate(
382  OpType::H, {qbp.first}, ChoiMixTableau::TableauSegment::Output);
383  tab.apply_CX(
384  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
385  break;
386  }
387  case Pauli::Y: {
388  out_circ_tp.add_op<Qubit>(OpType::Vdg, {qbp.first});
389  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
390  tab.apply_V(qbp.first, ChoiMixTableau::TableauSegment::Output);
391  tab.apply_CX(
392  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
393  break;
394  }
395  case Pauli::Z: {
396  out_circ_tp.add_op<Qubit>(OpType::CX, {qbp.first, *out_qb});
397  tab.apply_CX(
398  qbp.first, *out_qb, ChoiMixTableau::TableauSegment::Output);
399  break;
400  }
401  default: {
402  break;
403  }
404  }
405  }
406  }
407 
408  in_x_row.insert({in_qb, x_row});
409  in_z_row.insert({in_qb, z_row});
410  }
411 
412  // X and Z row are guaranteed to be the unique rows with X_in_qb and Z_in_qb.
413  // If the Z row exists, then all rows must commute with Z_in_qb Z_out_qb, so
414  // if any row has X_out_qb it must also have X_in_qb. Hence if both exist then
415  // the X row is also the unique row with X_out_qb, and by similar argument the
416  // Z row is the unique row with Z_out_qb. However, if only one row exists,
417  // this uniqueness may not hold but can be forced by applying some unitary
418  // gates to eliminate e.g. Z_out_qb from all other rows. We use a gaussian
419  // elimination subroutine to identify a combination of gates that won't add
420  // Z_out_qb onto other qubits. Since we have already removed them from all
421  // other rows containing input components, we only need to consider the
422  // output-only rows, which all exist at the bottom of the tableau.
423  unsigned out_stabs = 0;
424  while (out_stabs < tab.get_n_rows()) {
425  ChoiMixTableau::row_tensor_t rten =
426  tab.get_row(tab.get_n_rows() - 1 - out_stabs);
427  if (rten.first.string.map.size() != 0) {
428  // Reached the rows with non-empty input segment
429  break;
430  }
431  ++out_stabs;
432  }
433  MatrixXb out_rows = MatrixXb::Zero(out_stabs, 2 * output_qubits.size());
434  out_rows(Eigen::all, Eigen::seq(0, Eigen::last, 2)) = tab.tab_.xmat_.block(
435  tab.get_n_rows() - out_stabs, input_qubits.size(), out_stabs,
436  output_qubits.size());
437  out_rows(Eigen::all, Eigen::seq(1, Eigen::last, 2)) = tab.tab_.zmat_.block(
438  tab.get_n_rows() - out_stabs, input_qubits.size(), out_stabs,
439  output_qubits.size());
440  // Identify other qubits to apply gates to for removing the extra matched
441  // output terms
442  MatrixXb unmatched_outs = MatrixXb::Zero(
443  out_stabs, 2 * (output_qubits.size() - matched_qubits.size()));
444  std::vector<std::pair<Qubit, Pauli>> col_lookup;
445  for (unsigned out_col = 0; out_col < output_qubits.size(); ++out_col) {
446  unsigned tab_col = input_qubits.size() + out_col;
447  Qubit out_qb = tab.col_index_.right.at(tab_col).first;
448  if (matched_qubits.right.find(out_qb) != matched_qubits.right.end())
449  continue;
450  unmatched_outs.col(col_lookup.size()) = out_rows.col(2 * out_col);
451  col_lookup.push_back({out_qb, Pauli::X});
452  unmatched_outs.col(col_lookup.size()) = out_rows.col(2 * out_col + 1);
453  col_lookup.push_back({out_qb, Pauli::Z});
454  }
455  row_ops = gaussian_elimination_row_ops(unmatched_outs);
456  for (const std::pair<unsigned, unsigned>& op : row_ops) {
457  for (unsigned c = 0; c < unmatched_outs.cols(); ++c) {
458  unmatched_outs(op.second, c) =
459  unmatched_outs(op.second, c) ^ unmatched_outs(op.first, c);
460  }
461  tab.tab_.row_mult(
462  tab.get_n_rows() - out_stabs + op.first,
463  tab.get_n_rows() - out_stabs + op.second);
464  }
465  // Go through the output-only rows and remove terms from matched qubits
466  for (unsigned r = 0; r < out_stabs; ++r) {
467  unsigned leading_col = 0;
468  for (unsigned c = 0; c < unmatched_outs.cols(); ++c) {
469  if (unmatched_outs(r, c)) {
470  leading_col = c;
471  break;
472  }
473  }
474  std::pair<Qubit, Pauli> alternate_qb = col_lookup.at(leading_col);
475 
476  if (alternate_qb.second != Pauli::X) {
477  // Make the alternate point of contact X so we only need one set of rule
478  // for eliminating Paulis
479  tab.apply_gate(
480  OpType::H, {alternate_qb.first},
481  ChoiMixTableau::TableauSegment::Output);
482  out_circ_tp.add_op<Qubit>(OpType::H, {alternate_qb.first});
483  }
484 
485  ChoiMixTableau::row_tensor_t row_paulis =
486  tab.get_row(tab.get_n_rows() - out_stabs + r);
487 
488  for (const std::pair<const Qubit, Pauli>& qbp :
489  row_paulis.second.string.map) {
490  if (matched_qubits.right.find(qbp.first) == matched_qubits.right.end())
491  continue;
492  // Alternate point is guaranteed to be unmatched, so always needs an
493  // entangling gate
494  switch (qbp.second) {
495  case Pauli::X: {
496  out_circ_tp.add_op<Qubit>(
497  OpType::CX, {alternate_qb.first, qbp.first});
498  tab.apply_CX(
499  alternate_qb.first, qbp.first,
500  ChoiMixTableau::TableauSegment::Output);
501  break;
502  }
503  case Pauli::Z: {
504  out_circ_tp.add_op<Qubit>(
505  OpType::CZ, {alternate_qb.first, qbp.first});
506  tab.apply_gate(
507  OpType::CZ, {alternate_qb.first, qbp.first},
508  ChoiMixTableau::TableauSegment::Output);
509  break;
510  }
511  default: {
512  // Don't have to care about Y since any matched qubit has a row that
513  // is reduced to either X or Z and all other rows must commute with
514  // that
515  break;
516  }
517  }
518  }
519  }
520 
521  // Now that X_in_qb X_out_qb (or Zs) is the unique row for each of X_in_qb and
522  // X_out_qb, we can actually link up the qubit wires and remove the rows
523  for (const Qubit& in_qb : input_qubits) {
524  if (post_selected.find(in_qb) != post_selected.end()) continue;
525 
526  std::optional<unsigned> x_row = in_x_row.at(in_qb);
527  std::optional<unsigned> z_row = in_z_row.at(in_qb);
528  auto found = matched_qubits.left.find(in_qb);
529  std::optional<Qubit> out_qb = (found == matched_qubits.left.end())
530  ? std::nullopt
531  : std::optional<Qubit>{found->second};
532  // Handle phases and resolve qubit connections
533  if (x_row) {
534  if (z_row) {
535  // Hook up with an identity wire
536  if (tab.tab_.phase_(*z_row)) in_circ.add_op<Qubit>(OpType::X, {in_qb});
537  if (tab.tab_.phase_(*x_row)) in_circ.add_op<Qubit>(OpType::Z, {in_qb});
538  join_permutation.insert({*out_qb, in_qb});
539  } else {
540  // Just an X row, so must be connected to out_qb via a decoherence
541  if (tab.tab_.phase_(*x_row)) in_circ.add_op<Qubit>(OpType::Z, {in_qb});
542  in_circ.add_op<Qubit>(OpType::H, {in_qb});
543  in_circ.add_op<Qubit>(OpType::Collapse, {in_qb});
544  in_circ.add_op<Qubit>(OpType::H, {in_qb});
545  join_permutation.insert({*out_qb, in_qb});
546  }
547  } else {
548  if (z_row) {
549  // Just a Z row, so must be connected to out_qb via a decoherence
550  if (tab.tab_.phase_(*z_row)) in_circ.add_op<Qubit>(OpType::X, {in_qb});
551  in_circ.add_op<Qubit>(OpType::Collapse, {in_qb});
552  join_permutation.insert({*out_qb, in_qb});
553  } else {
554  // No rows involving this input, so it is discarded
555  in_circ.qubit_discard(in_qb);
556  }
557  }
558  }
559 
560  // Remove rows with inputs from the tableau
561  tab.tab_ = SymplecticTableau(
562  tab.tab_.xmat_.bottomRows(out_stabs),
563  tab.tab_.zmat_.bottomRows(out_stabs), tab.tab_.phase_.tail(out_stabs));
564 
565  // Can't use a template with multiple parameters within a macro since the
566  // comma will register as an argument delimiter for the macro
567  using match_entry = boost::bimap<Qubit, Qubit>::left_const_reference;
568  BOOST_FOREACH (match_entry entry, matched_qubits.left) {
569  tab.discard_qubit(entry.first, ChoiMixTableau::TableauSegment::Input);
570  tab.discard_qubit(entry.second, ChoiMixTableau::TableauSegment::Output);
571  }
572  tab.canonical_column_order(ChoiMixTableau::TableauSegment::Output);
573 
574  // Only remaining rows must be completely over the outputs. Call
575  // diagonalisation methods to diagonalise coherent subspace
576  to_diag.clear();
577  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
578  ChoiMixTableau::row_tensor_t rten = tab.get_row(r);
579  to_diag.push_back({rten.second, 1.});
580  }
581  std::set<Qubit> diag_outs;
582  for (const Qubit& out : output_qubits) {
583  if (matched_qubits.right.find(out) == matched_qubits.right.end())
584  diag_outs.insert(out);
585  }
586  Circuit out_diag_circ =
587  mutual_diagonalise(to_diag, diag_outs, CXConfigType::Tree);
588  // Extract the dagger of each gate in order from tab
589  for (const Command& com : out_diag_circ) {
590  auto args = com.get_args();
591  qubit_vector_t qbs = {args.begin(), args.end()};
592  tab.apply_gate(com.get_op_ptr()->get_type(), qbs);
593  out_circ_tp.add_op<Qubit>(com.get_op_ptr()->transpose()->dagger(), qbs);
594  }
595 
596  // All rows are diagonalised so we can just focus on the Z matrix. Reduce them
597  // to a minimal set of qubits for initialisation by first reducing to upper
598  // echelon form
599  row_ops = gaussian_elimination_row_ops(tab.tab_.zmat_);
600  for (const std::pair<unsigned, unsigned>& op : row_ops) {
601  tab.tab_.row_mult(op.first, op.second);
602  }
603  // Obtain CX instructions as column operations
604  col_ops = gaussian_elimination_col_ops(tab.tab_.zmat_);
605  for (const std::pair<unsigned, unsigned>& op : col_ops) {
606  tab.tab_.apply_CX(op.second, op.first);
607  ChoiMixTableau::col_key_t ctrl = tab.col_index_.right.at(op.second);
608  ChoiMixTableau::col_key_t trgt = tab.col_index_.right.at(op.first);
609  out_circ_tp.add_op<Qubit>(OpType::CX, {ctrl.first, trgt.first});
610  }
611 
612  // Fix phases of zero_initialised qubits
613  std::set<Qubit> zero_initialised;
614  Circuit out_circ(output_qubits, {});
615  for (unsigned r = 0; r < tab.get_n_rows(); ++r) {
616  ChoiMixTableau::row_tensor_t row = tab.get_row(r);
617  if (row.first.string.map.size() != 0 || row.second.string.map.size() != 1 ||
618  row.second.string.map.begin()->second != Pauli::Z)
619  throw std::logic_error(
620  "Unexpected error during zero initialisation in ChoiMixTableau "
621  "synthesis");
622  Qubit initialised_qb = row.second.string.map.begin()->first;
623  out_circ.qubit_create(initialised_qb);
624  if (row.second.coeff == -1.) {
625  out_circ.add_op<Qubit>(OpType::X, {initialised_qb});
626  }
627  zero_initialised.insert(initialised_qb);
628  }
629 
630  // Remaining outputs that aren't zero initialised or matched need to be
631  // initialised in the maximally-mixed state.
632  // Also match up unmatched outputs to either unmatched inputs or reusable
633  // output names (ones that are already matched up to other input names),
634  // preferring the qubit of the same name
635  std::list<Qubit> reusable_names;
636  for (const Qubit& out_qb : output_qubits) {
637  if (matched_qubits.right.find(out_qb) != matched_qubits.right.end() &&
638  matched_qubits.left.find(out_qb) == matched_qubits.left.end())
639  reusable_names.push_back(out_qb);
640  }
641  for (const Qubit& out_qb : output_qubits) {
642  if (matched_qubits.right.find(out_qb) == matched_qubits.right.end()) {
643  if (zero_initialised.find(out_qb) == zero_initialised.end()) {
644  out_circ.qubit_create(out_qb);
645  out_circ.add_op<Qubit>(OpType::H, {out_qb});
646  out_circ.add_op<Qubit>(OpType::Collapse, {out_qb});
647  }
648  if (matched_qubits.left.find(out_qb) == matched_qubits.left.end()) {
649  matched_qubits.insert({out_qb, out_qb});
650  join_permutation.insert({out_qb, out_qb});
651  } else {
652  // Since the matching is a bijection, matched_qubits.left -
653  // matched_qubits.right (set difference) is the same size as
654  // matched_qubits.right - matched_qubits.left, so there will be exactly
655  // the right number of reusable names to pull from here
656  Qubit name = reusable_names.front();
657  reusable_names.pop_front();
658  matched_qubits.insert({name, out_qb});
659  join_permutation.insert({out_qb, name});
660  }
661  }
662  }
663 
664  // Initialise qubits with stabilizer rows and stitch subcircuits together
665  out_circ.append(out_circ_tp.transpose());
666  in_circ.append_with_map(out_circ, join_permutation);
667  return {in_circ, join_permutation};
668 }
