23static SymplecticTableau id_tab(unsigned n) {
24 MatrixXb xmat(2 * n, 2 * n);
25 xmat << MatrixXb::Identity(n, n), MatrixXb::Identity(n, n),
26 MatrixXb::Zero(n, 2 * n);
27 MatrixXb zmat(2 * n, 2 * n);
28 zmat << MatrixXb::Zero(n, 2 * n), MatrixXb::Identity(n, n),
29 MatrixXb::Identity(n, n);
30 return SymplecticTableau(xmat, zmat, VectorXb::Zero(2 * n));
31}

33ChoiMixTableau::ChoiMixTableau(unsigned n) : tab_(id_tab(n)), col_index_() {
34 for (unsigned i = 0; i < n; ++i) {
35 col_index_.insert({{Qubit(i), TableauSegment::Input}, i});
36 col_index_.insert({{Qubit(i), TableauSegment::Output}, n + i});
37 }

40ChoiMixTableau::ChoiMixTableau(const qubit_vector_t& qbs)
41 : tab_(id_tab(qbs.size())), col_index_() {

43 unsigned i = 0;

51ChoiMixTableau::ChoiMixTableau(
52 const MatrixXb& xmat, const MatrixXb& zmat, const VectorXb& phase,
53 unsigned n_ins)
54 : tab_({}), col_index_() {
55 unsigned n_rows = xmat.rows();
56 unsigned n_bounds = xmat.cols();
57 if (n_ins > n_bounds)
58 throw std::invalid_argument(
59 "Number of inputs of a Choi tableau cannot be larger than the "
60 "number of qubits");
61 if ((zmat.cols() != n_bounds) || (zmat.rows() != n_rows) ||
62 (phase.size() != n_rows))
63 throw std::invalid_argument(
64 "Choi tableau requires equally-sized components");
65 tab_ = SymplecticTableau(xmat, zmat, phase);
66 if (tab_.anticommuting_rows() != MatrixXb::Zero(n_rows, n_rows))
67 throw std::invalid_argument("Rows of Choi tableau do not commute");
68 if (tab_.rank() != n_rows)
69 throw std::invalid_argument("Rows of Choi tableau are not independent");
70 for (unsigned i = 0; i < n_ins; ++i) {
71 col_index_.insert({{Qubit(i), TableauSegment::Input}, i});
72 }
73 for (unsigned i = 0; i < n_bounds - n_ins; ++i) {
74 col_index_.insert({{Qubit(i), TableauSegment::Output}, i});
75 }
76}

78ChoiMixTableau::ChoiMixTableau(const std::list<row_tensor_t>& rows)
79 : tab_({}), col_index_() {
80 std::set<Qubit> in_qubits;
81 std::set<Qubit> out_qubits;
82 for (const row_tensor_t& row : rows) {
83 for (const std::pair<const Qubit, Pauli>& qb : row.first.string) {
84 in_qubits.insert(qb.first);
85 }
86 for (const std::pair<const Qubit, Pauli>& qb : row.second.string) {
87 out_qubits.insert(qb.first);
88 }
89 }
90 unsigned n_rows = rows.size();
91 unsigned n_qbs = in_qubits.size() + out_qubits.size();
92 unsigned i = 0;
93 for (const Qubit& qb : in_qubits) {
94 col_index_.insert({{qb, TableauSegment::Input}, i});
95 ++i;
96 }
97 for (const Qubit& qb : out_qubits) {
98 col_index_.insert({{qb, TableauSegment::Output}, i});
99 ++i;
100 }
101 MatrixXb xmat = MatrixXb::Zero(n_rows, n_qbs);
102 MatrixXb zmat = MatrixXb::Zero(n_rows, n_qbs);
103 VectorXb phase = VectorXb::Zero(n_rows);
104 unsigned r = 0;
105 for (const row_tensor_t& row : rows) {
106 unsigned n_ys = 0;
107 for (const std::pair<const Qubit, Pauli>& qb : row.first.string) {
108 unsigned c =
109 col_index_.left.at(col_key_t{qb.first, TableauSegment::Input});
110 if (qb.second == Pauli::X || qb.second == Pauli::Y) xmat(r, c) = true;
111 if (qb.second == Pauli::Z || qb.second == Pauli::Y) zmat(r, c) = true;
112 if (qb.second == Pauli::Y) ++n_ys;
113 }
114 for (const std::pair<const Qubit, Pauli>& qb : row.second.string) {
115 unsigned c =
116 col_index_.left.at(col_key_t{qb.first, TableauSegment::Output});
117 if (qb.second == Pauli::X || qb.second == Pauli::Y) xmat(r, c) = true;
118 if (qb.second == Pauli::Z || qb.second == Pauli::Y) zmat(r, c) = true;
119 }
120 phase(r) = row.first.is_real_negative() ^ row.second.is_real_negative() ^
121 (n_ys % 2 == 1);
122 ++r;
123 }
124 tab_ = SymplecticTableau(xmat, zmat, phase);
125}

157ChoiMixTableau::row_tensor_t ChoiMixTableau::stab_to_row_tensor(
158 const PauliStabiliser& stab) const {
159 QubitPauliMap in_qpm, out_qpm;
160 for (unsigned i = 0; i < stab.string.size(); ++i) {
161 Pauli p = stab.string.at(i);
162 col_key_t col = col_index_.right.at(i);
163 if (p != Pauli::I) {
164 if (col.second == TableauSegment::Input)
165 in_qpm.insert({col.first, p});
166 else
167 out_qpm.insert({col.first, p});
168 }
169 }
170 return {SpPauliStabiliser(in_qpm), SpPauliStabiliser(out_qpm, stab.coeff)};
171}

173PauliStabiliser ChoiMixTableau::row_tensor_to_stab(
174 const row_tensor_t& ten) const {
175 std::vector<Pauli> ps;
176 for (unsigned i = 0; i < col_index_.size(); ++i) {
177 col_key_t qb = col_index_.right.at(i);
178 if (qb.second == TableauSegment::Input)
179 ps.push_back(ten.first.get(qb.first));
180 else
181 ps.push_back(ten.second.get(qb.first));
182 }
183 return PauliStabiliser(ps, (ten.first.coeff + ten.second.coeff) % 4);
184}

186ChoiMixTableau::row_tensor_t ChoiMixTableau::get_row(unsigned i) const {
187 ChoiMixTableau::row_tensor_t res = stab_to_row_tensor(tab_.get_pauli(i));
188 res.first.transpose();
189 res.second.coeff = (res.first.coeff + res.second.coeff) % 4;
190 res.first.coeff = 0;
191 return res;
192}

194ChoiMixTableau::row_tensor_t ChoiMixTableau::get_row_product(
195 const std::vector<unsigned>& rows) const {
196 row_tensor_t result = {{}, {}};
197 for (unsigned i : rows) {
198 row_tensor_t row_i = stab_to_row_tensor(tab_.get_pauli(i));
199 result.first = result.first * row_i.first;
200 result.second = result.second * row_i.second;
201 }
202 result.first.transpose();
203 result.second.coeff = (result.first.coeff + result.second.coeff) % 4;
204 result.first.coeff = 0;
205 return result;
206}

240void ChoiMixTableau::apply_gate(
241 OpType type, const qubit_vector_t& qbs, TableauSegment seg) {
242 switch (type) {
243 case OpType::Z: {
244 apply_Z(qbs.at(0), seg);
245 break;
246 }
247 case OpType::X: {
248 apply_X(qbs.at(0), seg);
249 break;
250 }
251 case OpType::Y: {
252 apply_Z(qbs.at(0), seg);
253 apply_X(qbs.at(0), seg);
254 break;
255 }
256 case OpType::S: {
257 apply_S(qbs.at(0), seg);
258 break;
259 }
260 case OpType::Sdg: {
261 apply_S(qbs.at(0), seg);
262 apply_Z(qbs.at(0), seg);
263 break;
264 }
265 case OpType::SX:
266 case OpType::V: {
267 apply_V(qbs.at(0), seg);
268 break;
269 }
270 case OpType::SXdg:
271 case OpType::Vdg: {
272 apply_V(qbs.at(0), seg);
273 apply_X(qbs.at(0), seg);
274 break;
275 }
276 case OpType::H: {
277 apply_H(qbs.at(0), seg);
278 break;
279 }
280 case OpType::CX: {
281 apply_CX(qbs.at(0), qbs.at(1), seg);
282 break;
283 }
284 case OpType::CY: {
285 if (seg == TableauSegment::Input) {
286 apply_S(qbs.at(1), seg);
287 apply_CX(qbs.at(0), qbs.at(1), seg);
288 apply_S(qbs.at(1), seg);
289 apply_Z(qbs.at(1), seg);
290 } else {
291 apply_S(qbs.at(1), seg);
292 apply_Z(qbs.at(1), seg);
293 apply_CX(qbs.at(0), qbs.at(1), seg);
294 apply_S(qbs.at(1), seg);
295 }
296 break;
297 }
298 case OpType::CZ: {
299 apply_H(qbs.at(1), seg);
300 apply_CX(qbs.at(0), qbs.at(1), seg);
301 apply_H(qbs.at(1), seg);
302 break;
303 }
304 case OpType::ZZMax: {
305 apply_CX(qbs.at(0), qbs.at(1), seg);
306 apply_S(qbs.at(1), seg);
307 apply_CX(qbs.at(0), qbs.at(1), seg);
308 break;
309 }
310 case OpType::ECR: {
311 if (seg == TableauSegment::Input) {
312 apply_X(qbs.at(0), seg);
313 apply_S(qbs.at(0), seg);
314 apply_V(qbs.at(1), seg);
315 apply_CX(qbs.at(0), qbs.at(1), seg);
316 } else {
317 apply_CX(qbs.at(0), qbs.at(1), seg);
318 apply_S(qbs.at(0), seg);
319 apply_X(qbs.at(0), seg);
320 apply_V(qbs.at(1), seg);
321 }
322 break;
323 }
324 case OpType::ISWAPMax: {
325 apply_V(qbs.at(0), seg);
326 apply_V(qbs.at(1), seg);
327 apply_CX(qbs.at(0), qbs.at(1), seg);
328 apply_V(qbs.at(0), seg);
329 apply_S(qbs.at(1), seg);
330 apply_Z(qbs.at(1), seg);
331 apply_CX(qbs.at(0), qbs.at(1), seg);
332 apply_V(qbs.at(0), seg);
333 apply_V(qbs.at(1), seg);
334 break;
335 }
336 case OpType::SWAP: {
337 apply_CX(qbs.at(0), qbs.at(1), seg);
338 apply_CX(qbs.at(1), qbs.at(0), seg);
339 apply_CX(qbs.at(0), qbs.at(1), seg);
340 break;
341 }
342 case OpType::BRIDGE: {
343 apply_CX(qbs.at(0), qbs.at(2), seg);
344 break;
345 }
346 case OpType::Phase:
347 case OpType::noop: {
348 break;
349 }
350 case OpType::Reset: {
351 if (seg == TableauSegment::Input) {
352 post_select(qbs.at(0), TableauSegment::Input);
353 unsigned col = get_n_boundaries();
354 unsigned rows = get_n_rows();
355 // reinsert qubit initialised to maximally mixed state (no coherent
356 // stabilizers)
357 col_index_.insert({{qbs.at(0), TableauSegment::Input}, col});
358 tab_.xmat.conservativeResize(rows, col + 1);
359 tab_.xmat.col(col) = MatrixXb::Zero(rows, 1);
360 tab_.zmat.conservativeResize(rows, col + 1);
361 tab_.zmat.col(col) = MatrixXb::Zero(rows, 1);
362 } else {
363 discard_qubit(qbs.at(0), TableauSegment::Output);
364 unsigned col = get_n_boundaries();
365 unsigned rows = get_n_rows();
366 // reinsert qubit initialised to |0> (add a Z stabilizer)
367 col_index_.insert({{qbs.at(0), TableauSegment::Output}, col});
368 tab_.xmat.conservativeResize(rows + 1, col + 1);
369 tab_.xmat.col(col) = MatrixXb::Zero(rows + 1, 1);
370 tab_.xmat.row(rows) = MatrixXb::Zero(1, col + 1);
371 tab_.zmat.conservativeResize(rows + 1, col + 1);
372 tab_.zmat.col(col) = MatrixXb::Zero(rows + 1, 1);
373 tab_.zmat.row(rows) = MatrixXb::Zero(1, col + 1);
374 tab_.zmat(rows, col) = true;
375 tab_.phase.conservativeResize(rows + 1);
376 tab_.phase(rows) = false;
377 }
378 break;
379 }
380 case OpType::Collapse: {
381 collapse_qubit(qbs.at(0), seg);
382 break;
383 }
384 default: {
385 throw BadOpType(
386 "Cannot be applied to a ChoiMixTableau: not a unitary Clifford gate",
387 type);
388 }
389 }
390}

405void ChoiMixTableau::post_select(const Qubit& qb, TableauSegment seg) {
406 tab_.gaussian_form();
407 // If +Z or -Z is a stabilizer, it will appear as the only row containing Z
408 // after gaussian elimination Check for the deterministic cases
409 unsigned n_rows = get_n_rows();
410 unsigned n_cols = get_n_boundaries();
411 unsigned col = col_index_.left.at(col_key_t{qb, seg});
412 for (unsigned r = 0; r < n_rows; ++r) {
413 if (tab_.zmat(r, col)) {
414 bool only_z = true;
415 for (unsigned c = 0; c < n_cols; ++c) {
416 if ((tab_.xmat(r, c) || tab_.zmat(r, c)) && (c != col)) {
417 only_z = false;
418 break;
419 }
420 }
421 if (!only_z) break; // Not deterministic
422 // From here, we know we are in a deterministic case
423 // If deterministically fail, throw an exception
424 if (tab_.phase(r))
425 throw std::logic_error(
426 "Post-selecting a tableau fails deterministically");
427 // Otherwise, we succeed and remove the stabilizer
428 remove_row(r);
429 remove_col(col);
430 return;
431 }
432 }
433 // For the remainder of the method we handle the non-deterministic case
434 // Isolate a single row with an X (if one exists)
435 std::optional<unsigned> x_row = std::nullopt;
436 for (unsigned r = 0; r < n_rows; ++r) {
437 if (tab_.xmat(r, col)) {
438 if (x_row) {
439 // Already found another row with an X, so combine them
440 tab_.row_mult(*x_row, r);
441 } else {
442 // This is the first row with an X
443 // Continue searching the rest to make it unique
444 x_row = r;
445 }
446 }
447 }
448 if (x_row) {
449 // Remove the anti-commuting stabilizer
450 remove_row(*x_row);
451 }
452 remove_col(col);
453}

455void ChoiMixTableau::discard_qubit(const Qubit& qb, TableauSegment seg) {
456 unsigned col = col_index_.left.at(col_key_t{qb, seg});
457 // Isolate a single row with an X (if one exists)
458 std::optional<unsigned> x_row = std::nullopt;
459 for (unsigned r = 0; r < get_n_rows(); ++r) {

518void ChoiMixTableau::remove_row(unsigned row) {
519 if (row >= get_n_rows())
520 throw std::invalid_argument(
521 "Cannot remove row " + std::to_string(row) + " from tableau with " +
522 std::to_string(get_n_rows()) + " rows");
523 unsigned n_rows = get_n_rows();
524 unsigned n_cols = get_n_boundaries();
525 if (row < n_rows - 1) {
526 tab_.xmat.row(row) = tab_.xmat.row(n_rows - 1);
527 tab_.zmat.row(row) = tab_.zmat.row(n_rows - 1);
528 tab_.phase(row) = tab_.phase(n_rows - 1);
529 }
530 tab_.xmat.conservativeResize(n_rows - 1, n_cols);
531 tab_.zmat.conservativeResize(n_rows - 1, n_cols);
532 tab_.phase.conservativeResize(n_rows - 1);
533}

535void ChoiMixTableau::remove_col(unsigned col) {
536 if (col >= get_n_boundaries())
537 throw std::invalid_argument(
538 "Cannot remove column " + std::to_string(col) + " from tableau with " +
539 std::to_string(get_n_boundaries()) + " columns");
540 unsigned n_rows = get_n_rows();
541 unsigned n_cols = get_n_boundaries();
542 if (col < n_cols - 1) {
543 tab_.xmat.col(col) = tab_.xmat.col(n_cols - 1);
544 tab_.zmat.col(col) = tab_.zmat.col(n_cols - 1);
545 }
546 tab_.xmat.conservativeResize(n_rows, n_cols - 1);
547 tab_.zmat.conservativeResize(n_rows, n_cols - 1);
548 col_index_.right.erase(col);
549 if (col < n_cols - 1) {
550 tableau_col_index_t::right_iterator it = col_index_.right.find(n_cols - 1);
551 col_key_t last = it->second;
552 col_index_.right.erase(it);
553 col_index_.insert({last, col});
554 }
555}

557void ChoiMixTableau::canonical_column_order(TableauSegment first) {
558 std::set<Qubit> ins;
559 std::set<Qubit> outs;
560 BOOST_FOREACH (
561 tableau_col_index_t::left_const_reference entry, col_index_.left) {
562 if (entry.first.second == TableauSegment::Input)
563 ins.insert(entry.first.first);
564 else
565 outs.insert(entry.first.first);
566 }
567 tableau_col_index_t new_index;
568 unsigned i = 0;
569 if (first == TableauSegment::Input) {
570 for (const Qubit& q : ins) {
571 new_index.insert({{q, TableauSegment::Input}, i});
572 ++i;
573 }
574 }
575 for (const Qubit& q : outs) {
576 new_index.insert({{q, TableauSegment::Output}, i});
577 ++i;
578 }
579 if (first == TableauSegment::Output) {
580 for (const Qubit& q : ins) {
581 new_index.insert({{q, TableauSegment::Input}, i});
582 ++i;
583 }
584 }
585 unsigned n_rows = get_n_rows();
586 MatrixXb xmat = MatrixXb::Zero(n_rows, i);
587 MatrixXb zmat = MatrixXb::Zero(n_rows, i);
588 for (unsigned j = 0; j < i; ++j) {
589 col_key_t key = new_index.right.at(j);
590 unsigned c = col_index_.left.at(key);
591 xmat.col(j) = tab_.xmat.col(c);
592 zmat.col(j) = tab_.zmat.col(c);
593 }
594 tab_ = SymplecticTableau(xmat, zmat, tab_.phase);
595 col_index_ = new_index;
596}

600void ChoiMixTableau::rename_qubits(
601 const qubit_map_t& qmap, TableauSegment seg) {
602 tableau_col_index_t new_index;
603 BOOST_FOREACH (
604 tableau_col_index_t::left_const_reference entry, col_index_.left) {
605 auto found = qmap.find(entry.first.first);
606 if (entry.first.second == seg && found != qmap.end())
607 new_index.insert({{found->second, seg}, entry.second});
608 else
609 new_index.insert({entry.first, entry.second});
610 }
611 col_index_ = new_index;
612}

614ChoiMixTableau ChoiMixTableau::compose(
615 const ChoiMixTableau& first, const ChoiMixTableau& second) {
616 // Merge tableau into a single one with only output qubits with default
617 // indexing
618 tableau_col_index_t first_qubits_to_names;
619 tableau_col_index_t second_qubits_to_names;
620 unsigned f_rows = first.get_n_rows();
621 unsigned f_cols = first.get_n_boundaries();
622 unsigned s_rows = second.get_n_rows();
623 unsigned s_cols = second.get_n_boundaries();
624 for (unsigned i = 0; i < f_cols; ++i) {
625 first_qubits_to_names.insert({first.col_index_.right.at(i), i});
626 }
627 for (unsigned i = 0; i < s_cols; ++i) {
628 second_qubits_to_names.insert({second.col_index_.right.at(i), i + f_cols});
629 }
630 MatrixXb fullx(f_rows + s_rows, f_cols + s_cols),
631 fullz(f_rows + s_rows, f_cols + s_cols);
632 fullx << first.tab_.xmat, MatrixXb::Zero(f_rows, s_cols),
633 MatrixXb::Zero(s_rows, f_cols), second.tab_.xmat;
634 fullz << first.tab_.zmat, MatrixXb::Zero(f_rows, s_cols),
635 MatrixXb::Zero(s_rows, f_cols), second.tab_.zmat;
636 VectorXb fullph(f_rows + s_rows);
637 fullph << first.tab_.phase, second.tab_.phase;
638 ChoiMixTableau combined(fullx, fullz, fullph, 0);
639 // For each connecting pair of qubits, compose via a Bell post-selection
640 for (unsigned i = 0; i < f_cols; ++i) {
641 col_key_t ind = first_qubits_to_names.right.at(i);
642 if (ind.second == TableauSegment::Output) {
643 auto found = second_qubits_to_names.left.find(
644 col_key_t{ind.first, TableauSegment::Input});
645 if (found != second_qubits_to_names.left.end()) {
646 // Found a matching pair
647 Qubit f_qb(i), s_qb(found->second);
648 combined.apply_CX(f_qb, s_qb);
649 combined.apply_gate(OpType::H, {f_qb});
650 combined.post_select(f_qb);
651 combined.post_select(s_qb);
652 }
653 }
654 }
655 // Rename qubits to original names
656 tableau_col_index_t new_index;
657 BOOST_FOREACH (
658 tableau_col_index_t::left_const_reference col, combined.col_index_.left) {
659 bool success = false;
660 unsigned qb_num = col.first.first.index().at(0);
661 auto found = first_qubits_to_names.right.find(qb_num);
662 if (found != first_qubits_to_names.right.end()) {
663 success = new_index.insert({found->second, col.second}).second;
664 } else {
665 success =
666 new_index
667 .insert({second_qubits_to_names.right.at(qb_num), col.second})
668 .second;
669 }
670 if (!success)
671 throw std::logic_error(
672 "Qubits aliasing after composing two ChoiMixTableau objects");
673 }
674 combined.col_index_ = new_index;
675 return combined;
676}
