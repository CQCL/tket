23 static SymplecticTableau id_tab(unsigned n) {
24  MatrixXb xmat(2 * n, 2 * n);
25  xmat << MatrixXb::Identity(n, n), MatrixXb::Identity(n, n),
26  MatrixXb::Zero(n, 2 * n);
27  MatrixXb zmat(2 * n, 2 * n);
28  zmat << MatrixXb::Zero(n, 2 * n), MatrixXb::Identity(n, n),
29  MatrixXb::Identity(n, n);
30  return SymplecticTableau(xmat, zmat, VectorXb::Zero(2 * n));
31 }

51 ChoiMixTableau::ChoiMixTableau(
52  const MatrixXb& xmat, const MatrixXb& zmat, const VectorXb& phase,
53  unsigned n_ins)
54  : tab_({}), col_index_() {
55  unsigned n_rows = xmat.rows();
56  unsigned n_bounds = xmat.cols();
57  if (n_ins > n_bounds)
58  throw std::invalid_argument(
59  "Number of inputs of a Choi tableau cannot be larger than the "
60  "number of qubits");
61  if ((zmat.cols() != n_bounds) || (zmat.rows() != n_rows) ||
62  (phase.size() != n_rows))
63  throw std::invalid_argument(
64  "Choi tableau requires equally-sized components");
65  tab_ = SymplecticTableau(xmat, zmat, phase);
66  if (tab_.anticommuting_rows() != MatrixXb::Zero(n_rows, n_rows))
67  throw std::invalid_argument("Rows of Choi tableau do not commute");
68  if (tab_.rank() != n_rows)
69  throw std::invalid_argument("Rows of Choi tableau are not independent");
70  for (unsigned i = 0; i < n_ins; ++i) {
71  col_index_.insert({{Qubit(i), TableauSegment::Input}, i});
72  }
73  for (unsigned i = 0; i < n_bounds - n_ins; ++i) {
74  col_index_.insert({{Qubit(i), TableauSegment::Output}, i});
75  }
76 }

78 ChoiMixTableau::ChoiMixTableau(const std::list<row_tensor_t>& rows)
79  : tab_({}), col_index_() {
80  std::set<Qubit> in_qubits;
81  std::set<Qubit> out_qubits;
82  for (const row_tensor_t& row : rows) {
83  for (const std::pair<const Qubit, Pauli>& qb : row.first.string.map) {
84  in_qubits.insert(qb.first);
85  }
86  for (const std::pair<const Qubit, Pauli>& qb : row.second.string.map) {
87  out_qubits.insert(qb.first);
88  }
89  }
90  unsigned n_rows = rows.size();
91  unsigned n_qbs = in_qubits.size() + out_qubits.size();
92  unsigned i = 0;
93  for (const Qubit& qb : in_qubits) {
94  col_index_.insert({{qb, TableauSegment::Input}, i});
95  ++i;
96  }
97  for (const Qubit& qb : out_qubits) {
98  col_index_.insert({{qb, TableauSegment::Output}, i});
99  ++i;
100  }
101  MatrixXb xmat = MatrixXb::Zero(n_rows, n_qbs);
102  MatrixXb zmat = MatrixXb::Zero(n_rows, n_qbs);
103  VectorXb phase = VectorXb::Zero(n_rows);
104  unsigned r = 0;
105  for (const row_tensor_t& row : rows) {
106  for (const std::pair<const Qubit, Pauli>& qb : row.first.string.map) {
107  unsigned c =
108  col_index_.left.at(col_key_t{qb.first, TableauSegment::Input});
109  if (qb.second == Pauli::X || qb.second == Pauli::Y) xmat(r, c) = true;
110  if (qb.second == Pauli::Z || qb.second == Pauli::Y) zmat(r, c) = true;
111  }
112  for (const std::pair<const Qubit, Pauli>& qb : row.second.string.map) {
113  unsigned c =
114  col_index_.left.at(col_key_t{qb.first, TableauSegment::Output});
115  if (qb.second == Pauli::X || qb.second == Pauli::Y) xmat(r, c) = true;
116  if (qb.second == Pauli::Z || qb.second == Pauli::Y) zmat(r, c) = true;
117  }
118  Complex ph = row.first.coeff * row.second.coeff;
119  if (std::abs(ph - 1.) < EPS)
120  phase(r) = false;
121  else if (std::abs(ph + 1.) < EPS)
122  phase(r) = true;
123  else
124  throw std::invalid_argument(
125  "Phase coefficient of a Choi tableau row must be +-1");
126  ++r;
127  }
128  tab_ = SymplecticTableau(xmat, zmat, phase);
129 }

153 ChoiMixTableau::row_tensor_t ChoiMixTableau::stab_to_row_tensor(
154  const PauliStabiliser& stab) const {
155  QubitPauliMap in_qpm, out_qpm;
156  for (unsigned i = 0; i < stab.string.size(); ++i) {
157  Pauli p = stab.string.at(i);
158  col_key_t col = col_index_.right.at(i);
159  if (p != Pauli::I) {
160  if (col.second == TableauSegment::Input)
161  in_qpm.insert({col.first, p});
162  else
163  out_qpm.insert({col.first, p});
164  }
165  }
166  return {
167  QubitPauliTensor(in_qpm),
168  QubitPauliTensor(out_qpm, stab.coeff ? 1. : -1.)};
169 }

171 PauliStabiliser ChoiMixTableau::row_tensor_to_stab(
172  const row_tensor_t& ten) const {
173  std::vector<Pauli> ps;
174  for (unsigned i = 0; i < col_index_.size(); ++i) {
175  col_key_t qb = col_index_.right.at(i);
176  if (qb.second == TableauSegment::Input)
177  ps.push_back(ten.first.string.get(qb.first));
178  else
179  ps.push_back(ten.second.string.get(qb.first));
180  }
181  return PauliStabiliser(
182  ps, std::abs(ten.first.coeff * ten.second.coeff - 1.) < EPS);
183 }

189 ChoiMixTableau::row_tensor_t ChoiMixTableau::get_row_product(
190  const std::vector<unsigned>& rows) const {
191  row_tensor_t result = {{}, {}};
192  for (unsigned i : rows) {
193  row_tensor_t row_i = get_row(i);
194  result.first = result.first * row_i.first;
195  result.second = result.second * row_i.second;
196  }
197  result.second.coeff *= result.first.coeff;
198  result.first.coeff = 1.;
199  return result;
200 }

219 void ChoiMixTableau::apply_gate(
220  OpType type, const qubit_vector_t& qbs, TableauSegment seg) {
221  switch (type) {
222  case OpType::Z: {
223  apply_S(qbs.at(0), seg);
224  apply_S(qbs.at(0), seg);
225  break;
226  }
227  case OpType::X: {
228  apply_V(qbs.at(0), seg);
229  apply_V(qbs.at(0), seg);
230  break;
231  }
232  case OpType::Y: {
233  apply_S(qbs.at(0), seg);
234  apply_S(qbs.at(0), seg);
235  apply_V(qbs.at(0), seg);
236  apply_V(qbs.at(0), seg);
237  break;
238  }
239  case OpType::S: {
240  apply_S(qbs.at(0), seg);
241  break;
242  }
243  case OpType::Sdg: {
244  apply_S(qbs.at(0), seg);
245  apply_S(qbs.at(0), seg);
246  apply_S(qbs.at(0), seg);
247  break;
248  }
249  case OpType::SX:
250  case OpType::V: {
251  apply_V(qbs.at(0), seg);
252  break;
253  }
254  case OpType::SXdg:
255  case OpType::Vdg: {
256  apply_V(qbs.at(0), seg);
257  apply_V(qbs.at(0), seg);
258  apply_V(qbs.at(0), seg);
259  break;
260  }
261  case OpType::H: {
262  apply_S(qbs.at(0), seg);
263  apply_V(qbs.at(0), seg);
264  apply_S(qbs.at(0), seg);
265  break;
266  }
267  case OpType::CX: {
268  apply_CX(qbs.at(0), qbs.at(1), seg);
269  break;
270  }
271  case OpType::CY: {
272  if (seg == TableauSegment::Input) {
273  apply_S(qbs.at(1), seg);
274  apply_CX(qbs.at(0), qbs.at(1), seg);
275  apply_S(qbs.at(1), seg);
276  apply_S(qbs.at(1), seg);
277  apply_S(qbs.at(1), seg);
278  } else {
279  apply_S(qbs.at(1), seg);
280  apply_S(qbs.at(1), seg);
281  apply_S(qbs.at(1), seg);
282  apply_CX(qbs.at(0), qbs.at(1), seg);
283  apply_S(qbs.at(1), seg);
284  }
285  break;
286  }
287  case OpType::CZ: {
288  apply_S(qbs.at(1), seg);
289  apply_V(qbs.at(1), seg);
290  apply_S(qbs.at(1), seg);
291  apply_CX(qbs.at(0), qbs.at(1), seg);
292  apply_S(qbs.at(1), seg);
293  apply_V(qbs.at(1), seg);
294  apply_S(qbs.at(1), seg);
295  break;
296  }
297  case OpType::ZZMax: {
298  apply_CX(qbs.at(0), qbs.at(1), seg);
299  apply_S(qbs.at(1), seg);
300  apply_CX(qbs.at(0), qbs.at(1), seg);
301  break;
302  }
303  case OpType::ECR: {
304  if (seg == TableauSegment::Input) {
305  apply_V(qbs.at(0), seg);
306  apply_V(qbs.at(0), seg);
307  apply_S(qbs.at(0), seg);
308  apply_V(qbs.at(1), seg);
309  apply_CX(qbs.at(0), qbs.at(1), seg);
310  } else {
311  apply_CX(qbs.at(0), qbs.at(1), seg);
312  apply_S(qbs.at(0), seg);
313  apply_V(qbs.at(0), seg);
314  apply_V(qbs.at(0), seg);
315  apply_V(qbs.at(1), seg);
316  }
317  break;
318  }
319  case OpType::ISWAPMax: {
320  apply_V(qbs.at(0), seg);
321  apply_V(qbs.at(1), seg);
322  apply_CX(qbs.at(0), qbs.at(1), seg);
323  apply_V(qbs.at(0), seg);
324  apply_S(qbs.at(1), seg);
325  apply_S(qbs.at(1), seg);
326  apply_S(qbs.at(1), seg);
327  apply_CX(qbs.at(0), qbs.at(1), seg);
328  apply_V(qbs.at(0), seg);
329  apply_V(qbs.at(1), seg);
330  break;
331  }
332  case OpType::SWAP: {
333  apply_CX(qbs.at(0), qbs.at(1), seg);
334  apply_CX(qbs.at(1), qbs.at(0), seg);
335  apply_CX(qbs.at(0), qbs.at(1), seg);
336  break;
337  }
338  case OpType::BRIDGE: {
339  apply_CX(qbs.at(0), qbs.at(2), seg);
340  break;
341  }
342  case OpType::Phase:
343  case OpType::noop: {
344  break;
345  }
346  case OpType::Reset: {
347  if (seg == TableauSegment::Input) {
348  post_select(qbs.at(0), TableauSegment::Input);
349  unsigned col = get_n_boundaries();
350  unsigned rows = get_n_rows();
351  // reinsert qubit initialised to maximally mixed state (no coherent
352  // stabilizers)
353  col_index_.insert({{qbs.at(0), TableauSegment::Input}, col});
354  tab_.xmat_.conservativeResize(rows, col + 1);
355  tab_.xmat_.col(col) = MatrixXb::Zero(rows, 1);
356  tab_.zmat_.conservativeResize(rows, col + 1);
357  tab_.zmat_.col(col) = MatrixXb::Zero(rows, 1);
358  ++tab_.n_qubits_;
359  } else {
360  discard_qubit(qbs.at(0), TableauSegment::Output);
361  unsigned col = get_n_boundaries();
362  unsigned rows = get_n_rows();
363  // reinsert qubit initialised to |0> (add a Z stabilizer)
364  col_index_.insert({{qbs.at(0), TableauSegment::Output}, col});
365  tab_.xmat_.conservativeResize(rows + 1, col + 1);
366  tab_.xmat_.col(col) = MatrixXb::Zero(rows + 1, 1);
367  tab_.xmat_.row(rows) = MatrixXb::Zero(1, col + 1);
368  tab_.zmat_.conservativeResize(rows + 1, col + 1);
369  tab_.zmat_.col(col) = MatrixXb::Zero(rows + 1, 1);
370  tab_.zmat_.row(rows) = MatrixXb::Zero(1, col + 1);
371  tab_.zmat_(rows, col) = true;
372  tab_.phase_.conservativeResize(rows + 1);
373  tab_.phase_(rows) = false;
374  ++tab_.n_rows_;
375  ++tab_.n_qubits_;
376  }
377  break;
378  }
379  case OpType::Collapse: {
380  collapse_qubit(qbs.at(0), seg);
381  break;
382  }
383  default: {
384  throw BadOpType(
385  "Cannot be applied to a ChoiMixTableau: not a unitary Clifford gate",
386  type);
387  }
388  }
389 }

391 void ChoiMixTableau::apply_pauli(
392  const QubitPauliTensor& pauli, unsigned half_pis, TableauSegment seg) {
393  if (std::abs(pauli.coeff - 1.) > EPS && std::abs(pauli.coeff + 1.) > EPS)
394  throw std::invalid_argument(
395  "In ChoiMixTableau::apply_pauli, can only rotate about a "
396  "QubitPauliTensor with coeff +-1");
397  PauliStabiliser ps;
398  if (seg == TableauSegment::Input) {
399  QubitPauliTensor tr = pauli;
400  tr.transpose();
401  ps = row_tensor_to_stab({tr, {}});
402  } else {
403  ps = row_tensor_to_stab({{}, pauli});
404  }
405  tab_.apply_pauli_gadget(ps, half_pis);
406 }

408 void ChoiMixTableau::post_select(const Qubit& qb, TableauSegment seg) {
409  tab_.gaussian_form();
410  // If +Z or -Z is a stabilizer, it will appear as the only row containing Z
411  // after gaussian elimination Check for the deterministic cases
412  unsigned n_rows = get_n_rows();
413  unsigned n_cols = get_n_boundaries();
414  unsigned col = col_index_.left.at(col_key_t{qb, seg});
415  for (unsigned r = 0; r < n_rows; ++r) {
416  if (tab_.zmat_(r, col)) {
417  bool only_z = true;
418  for (unsigned c = 0; c < n_cols; ++c) {
419  if ((tab_.xmat_(r, c) || tab_.zmat_(r, c)) && (c != col)) {
420  only_z = false;
421  break;
422  }
423  }
424  if (!only_z) break; // Not deterministic
425  // From here, we know we are in a deterministic case
426  // If deterministically fail, throw an exception
427  if (tab_.phase_(r))
428  throw std::logic_error(
429  "Post-selecting a tableau fails deterministically");
430  // Otherwise, we succeed and remove the stabilizer
431  remove_row(r);
432  remove_col(col);
433  return;
434  }
435  }
436  // For the remainder of the method we handle the non-deterministic case
437  // Isolate a single row with an X (if one exists)
438  std::optional<unsigned> x_row = std::nullopt;
439  for (unsigned r = 0; r < n_rows; ++r) {
440  if (tab_.xmat_(r, col)) {
441  if (x_row) {
442  // Already found another row with an X, so combine them
443  tab_.row_mult(*x_row, r);
444  } else {
445  // This is the first row with an X
446  // Continue searching the rest to make it unique
447  x_row = r;
448  }
449  }
450  }
451  if (x_row) {
452  // Remove the anti-commuting stabilizer
453  remove_row(*x_row);
454  }
455  remove_col(col);
456 }

459  unsigned col = col_index_.left.at(col_key_t{qb, seg});

521 void ChoiMixTableau::remove_row(unsigned row) {
522  if (row >= get_n_rows())
523  throw std::invalid_argument(
524  "Cannot remove row " + std::to_string(row) + " from tableau with " +
525  std::to_string(get_n_rows()) + " rows");
526  unsigned n_rows = get_n_rows();
527  unsigned n_cols = get_n_boundaries();
528  if (row < n_rows - 1) {
529  tab_.xmat_.row(row) = tab_.xmat_.row(n_rows - 1);
530  tab_.zmat_.row(row) = tab_.zmat_.row(n_rows - 1);
531  tab_.phase_(row) = tab_.phase_(n_rows - 1);
532  }
533  tab_.xmat_.conservativeResize(n_rows - 1, n_cols);
534  tab_.zmat_.conservativeResize(n_rows - 1, n_cols);
535  tab_.phase_.conservativeResize(n_rows - 1);
536  --tab_.n_rows_;
537 }

539 void ChoiMixTableau::remove_col(unsigned col) {
540  if (col >= get_n_boundaries())
541  throw std::invalid_argument(
542  "Cannot remove column " + std::to_string(col) + " from tableau with " +
543  std::to_string(get_n_boundaries()) + " columns");
544  unsigned n_rows = get_n_rows();
545  unsigned n_cols = get_n_boundaries();
546  if (col < n_cols - 1) {
547  tab_.xmat_.col(col) = tab_.xmat_.col(n_cols - 1);
548  tab_.zmat_.col(col) = tab_.zmat_.col(n_cols - 1);
549  }
550  tab_.xmat_.conservativeResize(n_rows, n_cols - 1);
551  tab_.zmat_.conservativeResize(n_rows, n_cols - 1);
552  col_index_.right.erase(col);
553  if (col < n_cols - 1) {
554  tableau_col_index_t::right_iterator it = col_index_.right.find(n_cols - 1);
555  col_key_t last = it->second;
556  col_index_.right.erase(it);
557  col_index_.insert({last, col});
558  }
559  --tab_.n_qubits_;
560 }

562 void ChoiMixTableau::canonical_column_order(TableauSegment first) {
563  std::set<Qubit> ins;
564  std::set<Qubit> outs;
565  BOOST_FOREACH (
566  tableau_col_index_t::left_const_reference entry, col_index_.left) {
567  if (entry.first.second == TableauSegment::Input)
568  ins.insert(entry.first.first);
569  else
570  outs.insert(entry.first.first);
571  }
572  tableau_col_index_t new_index;
573  unsigned i = 0;
574  if (first == TableauSegment::Input) {
575  for (const Qubit& q : ins) {
576  new_index.insert({{q, TableauSegment::Input}, i});
577  ++i;
578  }
579  }
580  for (const Qubit& q : outs) {
581  new_index.insert({{q, TableauSegment::Output}, i});
582  ++i;
583  }
584  if (first == TableauSegment::Output) {
585  for (const Qubit& q : ins) {
586  new_index.insert({{q, TableauSegment::Input}, i});
587  ++i;
588  }
589  }
590  unsigned n_rows = get_n_rows();
591  MatrixXb xmat = MatrixXb::Zero(n_rows, i);
592  MatrixXb zmat = MatrixXb::Zero(n_rows, i);
593  for (unsigned j = 0; j < i; ++j) {
594  col_key_t key = new_index.right.at(j);
595  unsigned c = col_index_.left.at(key);
596  xmat.col(j) = tab_.xmat_.col(c);
597  zmat.col(j) = tab_.zmat_.col(c);
598  }
599  tab_ = SymplecticTableau(xmat, zmat, tab_.phase_);
600  col_index_ = new_index;
601 }

619 ChoiMixTableau ChoiMixTableau::compose(
620  const ChoiMixTableau& first, const ChoiMixTableau& second) {
621  // Merge tableau into a single one with only output qubits with default
622  // indexing
623  tableau_col_index_t first_qubits_to_names;
624  tableau_col_index_t second_qubits_to_names;
625  unsigned f_rows = first.get_n_rows();
626  unsigned f_cols = first.get_n_boundaries();
627  unsigned s_rows = second.get_n_rows();
628  unsigned s_cols = second.get_n_boundaries();
629  for (unsigned i = 0; i < f_cols; ++i) {
630  first_qubits_to_names.insert({first.col_index_.right.at(i), i});
631  }
632  for (unsigned i = 0; i < s_cols; ++i) {
633  second_qubits_to_names.insert({second.col_index_.right.at(i), i + f_cols});
634  }
635  MatrixXb fullx(f_rows + s_rows, f_cols + s_cols),
636  fullz(f_rows + s_rows, f_cols + s_cols);
637  fullx << first.tab_.xmat_, MatrixXb::Zero(f_rows, s_cols),
638  MatrixXb::Zero(s_rows, f_cols), second.tab_.xmat_;
639  fullz << first.tab_.zmat_, MatrixXb::Zero(f_rows, s_cols),
640  MatrixXb::Zero(s_rows, f_cols), second.tab_.zmat_;
641  VectorXb fullph(f_rows + s_rows);
642  fullph << first.tab_.phase_, second.tab_.phase_;
643  ChoiMixTableau combined(fullx, fullz, fullph, 0);
644  // For each connecting pair of qubits, compose via a Bell post-selection
645  for (unsigned i = 0; i < f_cols; ++i) {
646  col_key_t ind = first_qubits_to_names.right.at(i);
647  if (ind.second == TableauSegment::Output) {
648  auto found = second_qubits_to_names.left.find(
649  col_key_t{ind.first, TableauSegment::Input});
650  if (found != second_qubits_to_names.left.end()) {
651  // Found a matching pair
652  Qubit f_qb(i), s_qb(found->second);
653  combined.apply_CX(f_qb, s_qb);
654  combined.apply_gate(OpType::H, {f_qb});
655  combined.post_select(f_qb);
656  combined.post_select(s_qb);
657  }
658  }
659  }
660  // Rename qubits to original names
661  tableau_col_index_t new_index;
662  BOOST_FOREACH (
663  tableau_col_index_t::left_const_reference col, combined.col_index_.left) {
664  bool success = false;
665  unsigned qb_num = col.first.first.index().at(0);
666  auto found = first_qubits_to_names.right.find(qb_num);
667  if (found != first_qubits_to_names.right.end()) {
668  success = new_index.insert({found->second, col.second}).second;
669  } else {
670  success =
671  new_index
672  .insert({second_qubits_to_names.right.at(qb_num), col.second})
673  .second;
674  }
675  if (!success)
676  throw std::logic_error(
677  "Qubits aliasing after composing two ChoiMixTableau objects");
678  }
679  combined.col_index_ = new_index;
680  return combined;
681 }
