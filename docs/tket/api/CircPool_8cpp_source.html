1002Circuit TK2_using_CX(const Expr &alpha, const Expr &beta, const Expr &gamma) {
1003 Circuit c = TK2_using_normalised_TK2(alpha, beta, gamma);
1004 // Find the TK2 vertex and replace it.
1005 BGL_FORALL_VERTICES(v, c.dag, DAG) {
1006 Op_ptr op = c.get_Op_ptr_from_Vertex(v);
1007 if (op->get_type() == OpType::TK2) {
1008 std::vector<Expr> params = op->get_params();
1009 TKET_ASSERT(params.size() == 3);
1010 Circuit rep = normalised_TK2_using_CX(params[0], params[1], params[2]);
1011 c.substitute(rep, v, Circuit::VertexDeletion::Yes);
1012 break;
1013 }
1014 }
1015 return c;
1016}

1028static Circuit normalised_TK2_using_CX_and_swap(
1029 const Expr &alpha, const Expr &beta, const Expr &gamma) {
1030 // first construct parameters for producing TK2 with wire swap
1031 std::tuple<Circuit, std::array<Expr, 3>, Circuit> pre_angles_post =
1032 normalise_TK2_angles(alpha + 0.5, beta + 0.5, gamma + 0.5);
1033 std::array<Expr, 3> swap_angles = std::get<1>(pre_angles_post);
1034
1035 // generate two circuits for both sets of angles
1036 Circuit without_swap = normalised_TK2_using_CX(alpha, beta, gamma);
1037 Circuit with_swap =
1038 normalised_TK2_using_CX(swap_angles[0], swap_angles[1], swap_angles[2]);
1039
1040 // if without_swap has same or fewer number of CX gates, return it, else build
1041 // with_swap circuit
1042 if (without_swap.count_gates(OpType::CX) <=
1043 with_swap.count_gates(OpType::CX)) {
1044 return without_swap;
1045 }
1046 Circuit swap(2);
1047 swap.add_op<unsigned>(OpType::SWAP, {0, 1});
1048 with_swap = std::get<0>(pre_angles_post) >> with_swap >>
1049 std::get<2>(pre_angles_post) >> swap;
1050 with_swap.add_phase(0.25);
1051 with_swap.replace_SWAPs();
1052
1053 // This decomposition can leave many extraneous single qubits gates: squash
1054 // them into TK1 that can be resynthesised
1055 Transforms::squash_1qb_to_tk1().apply(with_swap);
1056 return with_swap;
1057}

1059Circuit TK2_using_CX_and_swap(
1060 const Expr &alpha, const Expr &beta, const Expr &gamma) {
1061 Circuit c = TK2_using_normalised_TK2(alpha, beta, gamma);
1062 // Find the TK2 vertex and replace it.
1063 BGL_FORALL_VERTICES(v, c.dag, DAG) {
1064 Op_ptr op = c.get_Op_ptr_from_Vertex(v);
1065 if (op->get_type() == OpType::TK2) {
1066 std::vector<Expr> params = op->get_params();
1067 TKET_ASSERT(params.size() == 3);
1068 Circuit rep =
1069 normalised_TK2_using_CX_and_swap(params[0], params[1], params[2]);
1070 c.substitute(rep, v, Circuit::VertexDeletion::Yes);
1071 break;
1072 }
1073 }
1074 return c;

1373static Circuit _tk1_to_rzsx(
1374 const Expr &alpha, const Expr &beta, const Expr &gamma, bool allow_x) {
1375 Circuit c(1);
1376 Expr correction_phase = 0;
1377 if (equiv_0(beta)) {
1378 // b = 2k, if k is odd, then Rx(b) = -I
1379 c.add_op<unsigned>(OpType::Rz, alpha + gamma, {0});
1380 correction_phase = int_half(beta);
1381 } else if (equiv_0(beta + 1)) {
1382 // Use Rx(2k-1) = i(-1)^{k}SxSx
1383 correction_phase = -0.5 + int_half(beta - 1);
1384 if (equiv_0(alpha - gamma)) {
1385 // a - c = 2m
1386 // overall operation is (-1)^{m}Rx(2k -1)
1387 if (allow_x) {
1388 c.add_op<unsigned>(OpType::X, {0});
1389 } else {
1390 c.add_op<unsigned>(OpType::SX, {0});
1391 c.add_op<unsigned>(OpType::SX, {0});
1392 }
1393 correction_phase += int_half(alpha - gamma);
1394 } else {
1395 c.add_op<unsigned>(OpType::Rz, gamma, {0});
1396 if (allow_x) {
1397 c.add_op<unsigned>(OpType::X, {0});
1398 } else {
1399 c.add_op<unsigned>(OpType::SX, {0});
1400 c.add_op<unsigned>(OpType::SX, {0});
1401 }
1402 c.add_op<unsigned>(OpType::Rz, alpha, {0});
1403 }
1404 } else if (equiv_0(beta - 0.5) && equiv_0(alpha) && equiv_0(gamma)) {
1405 // a = 2k, b = 2m+0.5, c = 2n
1406 // Rz(2k)Rx(2m + 0.5)Rz(2n) = (-1)^{k+m+n}e^{-i \pi /4} SX
1407 c.add_op<unsigned>(OpType::SX, {0});
1408 correction_phase =
1409 int_half(beta - 0.5) + int_half(alpha) + int_half(gamma) - 0.25;
1410 } else if (equiv_0(beta - 0.5)) {
1411 // SX.Rz(2m-0.5).SX = (-1)^{m}e^{i \pi /4} Rz(-0.5).SX.Rz(-0.5)
1412 c.add_op<unsigned>(OpType::Rz, gamma, {0});
1413 c.add_op<unsigned>(OpType::SX, {0});
1414 c.add_op<unsigned>(OpType::Rz, alpha, {0});
1415 correction_phase = int_half(beta - 0.5) - 0.25;
1416 } else if (equiv_0(beta + 0.5)) {
1417 // SX.Rz(2m+0.5).SX = (-1)^{m}e^{i \pi /4} Rz(0.5).SX.Rz(0.5)
1418 c.add_op<unsigned>(OpType::Rz, gamma + 1, {0});
1419 c.add_op<unsigned>(OpType::SX, {0});
1420 c.add_op<unsigned>(OpType::Rz, alpha + 1, {0});
1421 correction_phase = int_half(beta - 1.5) - 0.25;
1422 } else if (equiv_0(alpha - 0.5) && equiv_0(gamma - 0.5)) {
1423 // Rz(2k + 0.5)Rx(b)Rz(2m + 0.5) = -i(-1)^{k+m}SX.Rz(1-b).SX
1424 c.add_op<unsigned>(OpType::SX, {0});
1425 c.add_op<unsigned>(OpType::Rz, 1 - beta, {0});
1426 c.add_op<unsigned>(OpType::SX, {0});
1427 correction_phase = int_half(alpha - 0.5) + int_half(gamma - 0.5) - 0.5;
1428 } else {
1429 c.add_op<unsigned>(OpType::Rz, gamma + 0.5, {0});
1430 c.add_op<unsigned>(OpType::SX, {0});
1431 c.add_op<unsigned>(OpType::Rz, beta - 1, {0});
1432 c.add_op<unsigned>(OpType::SX, {0});
1433 c.add_op<unsigned>(OpType::Rz, alpha + 0.5, {0});
1434 correction_phase = -0.5;
1435 }
1436 c.add_phase(correction_phase);
1437 c.remove_noops();
1438 return c;
1439}
