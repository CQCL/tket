1101 Circuit tk1_to_rzsx(const Expr &alpha, const Expr &beta, const Expr &gamma) {
1102  Circuit c(1);
1103  Expr correction_phase = 0;
1104  if (equiv_0(beta)) {
1105  // b = 2k, if k is odd, then Rx(b) = -I
1106  c.add_op<unsigned>(OpType::Rz, alpha + gamma, {0});
1107  correction_phase = int_half(beta);
1108  } else if (equiv_0(beta + 1)) {
1109  // Use Rx(2k-1) = i(-1)^{k}SxSx
1110  correction_phase = -0.5 + int_half(beta - 1);
1111  if (equiv_0(alpha - gamma)) {
1112  // a - c = 2m
1113  // overall operation is (-1)^{m}Rx(2k -1)
1114  c.add_op<unsigned>(OpType::SX, {0});
1115  c.add_op<unsigned>(OpType::SX, {0});
1116  correction_phase += int_half(alpha - gamma);
1117  } else {
1118  c.add_op<unsigned>(OpType::Rz, gamma, {0});
1119  c.add_op<unsigned>(OpType::SX, {0});
1120  c.add_op<unsigned>(OpType::SX, {0});
1121  c.add_op<unsigned>(OpType::Rz, alpha, {0});
1122  }
1123  } else if (equiv_0(beta - 0.5) && equiv_0(alpha) && equiv_0(gamma)) {
1124  // a = 2k, b = 2m+0.5, c = 2n
1125  // Rz(2k)Rx(2m + 0.5)Rz(2n) = (-1)^{k+m+n}e^{-i \pi /4} SX
1126  c.add_op<unsigned>(OpType::SX, {0});
1127  correction_phase =
1128  int_half(beta - 0.5) + int_half(alpha) + int_half(gamma) - 0.25;
1129  } else if (equiv_0(beta - 0.5)) {
1130  // SX.Rz(2m-0.5).SX = (-1)^{m}e^{i \pi /4} Rz(-0.5).SX.Rz(-0.5)
1131  c.add_op<unsigned>(OpType::Rz, gamma, {0});
1132  c.add_op<unsigned>(OpType::SX, {0});
1133  c.add_op<unsigned>(OpType::Rz, alpha, {0});
1134  correction_phase = int_half(beta - 0.5) - 0.25;
1135  } else if (equiv_0(beta + 0.5)) {
1136  // SX.Rz(2m+0.5).SX = (-1)^{m}e^{i \pi /4} Rz(0.5).SX.Rz(0.5)
1137  c.add_op<unsigned>(OpType::Rz, gamma + 1, {0});
1138  c.add_op<unsigned>(OpType::SX, {0});
1139  c.add_op<unsigned>(OpType::Rz, alpha + 1, {0});
1140  correction_phase = int_half(beta - 1.5) - 0.25;
1141  } else if (equiv_0(alpha - 0.5) && equiv_0(gamma - 0.5)) {
1142  // Rz(2k + 0.5)Rx(b)Rz(2m + 0.5) = -i(-1)^{k+m}SX.Rz(1-b).SX
1143  c.add_op<unsigned>(OpType::SX, {0});
1144  c.add_op<unsigned>(OpType::Rz, 1 - beta, {0});
1145  c.add_op<unsigned>(OpType::SX, {0});
1146  correction_phase = int_half(alpha - 0.5) + int_half(gamma - 0.5) - 0.5;
1147  } else {
1148  c.add_op<unsigned>(OpType::Rz, gamma + 0.5, {0});
1149  c.add_op<unsigned>(OpType::SX, {0});
1150  c.add_op<unsigned>(OpType::Rz, beta - 1, {0});
1151  c.add_op<unsigned>(OpType::SX, {0});
1152  c.add_op<unsigned>(OpType::Rz, alpha + 0.5, {0});
1153  correction_phase = -0.5;
1154  }
1155  c.add_phase(correction_phase);
1156  c.remove_noops();
1157  return c;
1158 }
