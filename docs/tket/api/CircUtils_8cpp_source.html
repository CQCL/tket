47 Eigen::Matrix2cd get_matrix_from_circ(const Circuit &circ) {
48  if (circ.n_qubits() != 1)
49  throw CircuitInvalidity(
50  "Getting Matrix: expected 1 qubit circuit, found " +
51  std::to_string(circ.n_qubits()));
52  Complex factor = std::exp(i_ * PI * eval_expr(circ.get_phase()).value());
53  VertexVec qpath = circ.qubit_path_vertices(circ.all_qubits()[0]);
54  unsigned N = qpath.size();
55  if (N == 2) return factor * Eigen::Matrix2cd::Identity();
56  Eigen::Matrix2cd m = get_matrix(circ, qpath[N - 2]);
57  for (unsigned x = N - 3; x >= 1; --x) {
58  m = m * get_matrix(circ, qpath[x]);
59  }
60  return factor * m;
61 }

63 Eigen::Matrix4cd get_matrix_from_2qb_circ(const Circuit &circ) {
64  std::vector<QPathDetailed> all_paths = circ.all_qubit_paths();
65  std::map<Vertex, Eigen::Matrix4cd> v_to_op;
66  Eigen::Matrix4cd cnot, tonc, swap;
67  // clang-format off
68  cnot << 1, 0, 0, 0,
69  0, 1, 0, 0,
70  0, 0, 0, 1,
71  0, 0, 1, 0;
72  tonc << 1, 0, 0, 0,
73  0, 0, 0, 1,
74  0, 0, 1, 0,
75  0, 1, 0, 0;
76  swap << 1, 0, 0, 0,
77  0, 0, 1, 0,
78  0, 1, 0, 0,
79  0, 0, 0, 1;
80  // clang-format on
81 
82  for (unsigned uqb = 0; uqb < 2; uqb++) {
83  for (QPathDetailed::iterator it = all_paths[uqb].begin();
84  it != all_paths[uqb].end(); ++it) {
85  const Op_ptr o = circ.get_Op_ptr_from_Vertex(it->first);
86  switch (o->get_type()) {
87  case OpType::Input:
88  case OpType::Create:
89  case OpType::Output:
90  case OpType::Discard: {
91  v_to_op[it->first] = Eigen::Matrix4cd::Identity();
92  break;
93  }
94  case OpType::SWAP: {
95  if (uqb == 0) v_to_op[it->first] = swap;
96  break;
97  }
98  case OpType::CX: {
99  if (uqb == 0) {
100  if (it->second == 0) {
101  v_to_op[it->first] = cnot;
102  } else {
103  v_to_op[it->first] = tonc;
104  }
105  }
106  break;
107  }
108  case OpType::TK2: {
109  auto params = o->get_params();
110  TKET_ASSERT(params.size() == 3);
111  v_to_op[it->first] =
112  get_matrix_from_2qb_circ(CircPool::normalised_TK2_using_CX(
113  params[0], params[1], params[2]));
114  break;
115  }
116  default: {
117  if (o->get_desc().is_gate() && circ.n_in_edges(it->first) == 1 &&
118  circ.n_out_edges(it->first) == 1) {
119  const Op_ptr g = o;
120  std::vector<Expr> ps = as_gate_ptr(g)->get_tk1_angles();
121  Eigen::Matrix2cd mat = get_matrix_from_tk1_angles(ps);
122  if (uqb == 0) {
123  v_to_op[it->first] =
124  Eigen::kroneckerProduct(mat, Eigen::Matrix2cd::Identity());
125  } else {
126  v_to_op[it->first] =
127  Eigen::kroneckerProduct(Eigen::Matrix2cd::Identity(), mat);
128  }
129  } else
130  throw BadOpType("Cannot obtain matrix from op", o->get_type());
131  }
132  }
133  }
134  }
135  Eigen::Matrix4cd m = Eigen::Matrix4cd::Identity();
136  SliceVec slices = circ.get_slices();
137  for (const Slice &s : slices) {
138  for (const Vertex &v : s) {
139  m = v_to_op[v] * m;
140  }
141  }
142  return std::exp(i_ * PI * eval_expr(circ.get_phase()).value()) * m;
143 }

145 Circuit two_qubit_canonical(const Eigen::Matrix4cd &U, OpType target_2qb_gate) {
146  if (!is_unitary(U)) {
147  throw std::invalid_argument(
148  "Non-unitary matrix passed to two_qubit_canonical");
149  }
150 
151  auto [K1, A, K2] = get_information_content(U);
152 
153  K1 /= pow(K1.determinant(), 0.25);
154  K2 /= pow(K2.determinant(), 0.25);
155  auto [a, b, c] = A;
156 
157  // Decompose single qubits
158  auto [K1a, K1b] = kronecker_decomposition(K1);
159  auto [K2a, K2b] = kronecker_decomposition(K2);
160 
161  Circuit result(2);
162 
163  std::vector<double> angles_q0 = tk1_angles_from_unitary(K2a);
164  std::vector<double> angles_q1 = tk1_angles_from_unitary(K2b);
165  result.add_op<unsigned>(
166  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
167  result.add_op<unsigned>(
168  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
169 
170  switch (target_2qb_gate) {
171  case OpType::TK2:
172  result.append(CircPool::TK2_using_normalised_TK2(a, b, c));
173  break;
174  case OpType::CX:
175  result.append(CircPool::TK2_using_CX(a, b, c));
176  break;
177  default:
178  throw std::invalid_argument("target_2qb_gate must be CX or TK2.");
179  }
180 
181  angles_q0 = tk1_angles_from_unitary(K1a);
182  angles_q1 = tk1_angles_from_unitary(K1b);
183  result.add_op<unsigned>(
184  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
185  result.add_op<unsigned>(
186  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
187 
188  // this fixes phase if decomposition is exact
189  Eigen::Matrix4cd reminder = get_matrix_from_2qb_circ(result).adjoint() * U;
190  const Complex phase = reminder(0, 0); // reminder = phase * I
191  result.add_phase(arg(phase) / PI);
192  return result;
193 }

195 // Factorize U as VD where V corresponds to a 2-CX circuit and
196 // D = diag(z, z*, z*, z). Return V and z.
197 static std::pair<Eigen::Matrix4cd, Complex> decompose_VD(
198  const Eigen::Matrix4cd &U) {
199  if (!is_unitary(U)) {
200  throw std::invalid_argument("Non-unitary matrix passed to decompose_VD");
201  }
202 
203  // The calculations below are derived from the proof of Proposition V.2 in
204  // https://arxiv.org/abs/quant-ph/0308033.
205 
206  Eigen::Matrix4cd u = U / pow(U.determinant(), 0.25);
207  Complex a = u(3, 0) * u(0, 3) - u(2, 0) * u(1, 3) - u(1, 0) * u(2, 3) +
208  u(0, 0) * u(3, 3);
209  Complex b = u(3, 1) * u(0, 2) - u(2, 1) * u(1, 2) - u(1, 1) * u(2, 2) +
210  u(0, 1) * u(3, 2);
211  // Now we want to find z such that |z|=1 and (az* - bz) is real.
212  // The numerical stability of this function is a concern when a is close to
213  // -b*. This problem can be demonstrated in artificially constructed
214  // examples (passing unitaries very close to, but not quite, the identity to
215  // the functions below). In these cases the product VD (or DV) may not
216  // approximate U to within Eigen's default tolerance. Is there a way to
217  // dodge this issue?
218  Complex w = a + std::conj(b);
219  double d = std::abs(w);
220  // If w = 0 then we can set z = 1.
221  Complex z = (d < EPS) ? 1. : w / d;
222  Complex z0 = sqrt(z);
223  Complex z1 = std::conj(z0);
224  Eigen::Matrix4cd V = U;
225  V.col(0) *= z1;
226  V.col(1) *= z0;
227  V.col(2) *= z0;
228  V.col(3) *= z1;
229  return {V, z0};
230 }

232 static void replace_TK2_2CX(Circuit &circ) {
233  VertexList bin;
234  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
235  if (circ.get_OpType_from_Vertex(v) != OpType::TK2) continue;
236  auto params = circ.get_Op_ptr_from_Vertex(v)->get_params();
237  TKET_ASSERT(params.size() == 3);
238  // Rounding errors can accumulate here; warn if so:
239  if (!equiv_0(params[2], 4, 1e-6)) {
240  std::stringstream ss;
241  ss << "Rounding errors in CX decomposition: ZZPhase parameter = "
242  << params[2] << " when it should be 0 (mod 4). Ignoring.";
243  tket_log()->warn(ss.str());
244  }
245  Circuit sub = CircPool::approx_TK2_using_2xCX(params[0], params[1]);
246  bin.push_back(v);
247  circ.substitute(sub, v, Circuit::VertexDeletion::No);
248  }
249  TKET_ASSERT(bin.size() == 1);
250  circ.remove_vertices(
251  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
252 }

261 std::pair<Circuit, Complex> decompose_2cx_DV(const Eigen::Matrix4cd &U) {
262  auto [V, z0] = decompose_VD(U.adjoint());
263  V.adjointInPlace();
264  Circuit circ = two_qubit_canonical(V);
265  replace_TK2_2CX(circ);
266  return {circ, std::conj(z0)};

269 Circuit phase_gadget(unsigned n_qubits, const Expr &t, CXConfigType cx_config) {
270  // Handle n_qubits==0 as a special case, or the calculations below
271  // go badly wrong.
272  Circuit new_circ(n_qubits);
273  if (n_qubits == 0) {
274  new_circ.add_phase(-t / 2);
275  return new_circ;
276  }
277  switch (cx_config) {
278  case CXConfigType::Snake: {
279  for (unsigned i = n_qubits - 1; i != 0; --i) {
280  unsigned j = i - 1;
281  new_circ.add_op<unsigned>(OpType::CX, {i, j});
282  }
283  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
284  for (unsigned i = 0; i != n_qubits - 1; ++i) {
285  unsigned j = i + 1;
286  new_circ.add_op<unsigned>(OpType::CX, {j, i});
287  }
288  break;
289  }
290  case CXConfigType::Star: {
291  for (unsigned i = n_qubits - 1; i != 0; --i) {
292  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
293  }
294  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
295  for (unsigned i = 1; i != n_qubits; ++i) {
296  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
297  }
298  break;
299  }
300  case CXConfigType::Tree: {
301  unsigned complete_layers = floor(log2(n_qubits));
302  unsigned dense_end = pow(2, complete_layers);
303  for (unsigned i = 0; i < n_qubits - dense_end; i++)
304  new_circ.add_op<unsigned>(
305  OpType::CX, {dense_end + i, dense_end - 1 - i});
306  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
307  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
308  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
309  }
310  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
311  for (unsigned step_size = dense_end / 2; step_size >= 1; step_size /= 2) {
312  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
313  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
314  }
315  for (unsigned i = 0; i < n_qubits - dense_end; i++)
316  new_circ.add_op<unsigned>(
317  OpType::CX, {dense_end + i, dense_end - 1 - i});
318  break;
319  }
320  case CXConfigType::MultiQGate: {
321  std::vector<std::vector<unsigned>> conjugations;
322  int sign_correction = 1;
323  for (int q = n_qubits - 1; q > 0; q -= 2) {
324  if (q - 1 > 0) {
325  unsigned i = q, j = q - 1;
326  // this is only equal to the CX decompositions above
327  // up to phase, but phase differences are cancelled out by
328  // its dagger XXPhase(-1/2) below.
329  new_circ.add_op<unsigned>(OpType::H, {i});
330  new_circ.add_op<unsigned>(OpType::H, {j});
331  new_circ.add_op<unsigned>(OpType::XXPhase3, 0.5, {i, j, 0});
332  sign_correction *= -1;
333  conjugations.push_back({i, j, 0});
334  } else {
335  unsigned i = q;
336  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
337  conjugations.push_back({i, 0});
338  }
339  }
340  new_circ.add_op<unsigned>(OpType::Rz, sign_correction * t, {0});
341  for (const auto &conj : conjugations) {
342  if (conj.size() == 2) {
343  new_circ.add_op<unsigned>(OpType::CX, conj);
344  } else {
345  TKET_ASSERT(conj.size() == 3);
346  new_circ.add_op<unsigned>(OpType::XXPhase3, -0.5, conj);
347  new_circ.add_op<unsigned>(OpType::H, {conj[0]});
348  new_circ.add_op<unsigned>(OpType::H, {conj[1]});
349  }
350  }
351  break;
352  }
353  }
354  return new_circ;
355 }

357 Circuit pauli_gadget(
358  const std::vector<Pauli> &paulis, const Expr &t, CXConfigType cx_config) {
359  unsigned n = paulis.size();
360  Circuit circ(n);
361  std::vector<unsigned> qubits;
362  for (unsigned i = 0; i < n; i++) {
363  switch (paulis[i]) {
364  case Pauli::I:
365  break;
366  case Pauli::X:
367  circ.add_op<unsigned>(OpType::H, {i});
368  qubits.push_back(i);
369  break;
370  case Pauli::Y:
371  circ.add_op<unsigned>(OpType::V, {i});
372  qubits.push_back(i);
373  break;
374  case Pauli::Z:
375  qubits.push_back(i);
376  break;
377  }
378  }
379  if (qubits.empty()) {
380  circ.add_phase(-t / 2);
381  } else {
382  Vertex v = circ.add_op<unsigned>(OpType::PhaseGadget, t, qubits);
383  Circuit cx_gadget = phase_gadget(circ.n_in_edges(v), t, cx_config);
384  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
385  circ.substitute(cx_gadget, sub, Circuit::VertexDeletion::Yes);
386  for (unsigned i = 0; i < n; i++) {
387  switch (paulis[i]) {
388  case Pauli::I:
389  break;
390  case Pauli::X:
391  circ.add_op<unsigned>(OpType::H, {i});
392  break;
393  case Pauli::Y:
394  circ.add_op<unsigned>(OpType::Vdg, {i});
395  break;
396  case Pauli::Z:
397  break;
398  }
399  }
400  }
401  return circ;
402 }

409 Circuit with_TK2(Gate_ptr op) {
410  std::vector<Expr> params = op->get_params();
411  unsigned n = op->n_qubits();
412  if (n == 0) {
413  Circuit c(0);
414  if (op->get_type() == OpType::Phase) {
415  c.add_phase(op->get_params()[0]);
416  }
417  return c;
418  } else if (n == 1) {
419  Circuit c(1);
420  c.add_op(op, std::vector<unsigned>{0});
421  return c;
422  } else if (n == 2 && op->free_symbols().empty()) {
423  Eigen::Matrix4cd U = op->get_unitary();
424  auto [K1, A, K2] = get_information_content(U);
425  // Decompose single qubits
426  auto [K1a, K1b] = kronecker_decomposition(K1);
427  auto [K2a, K2b] = kronecker_decomposition(K2);
428  Circuit c(2);
429  std::vector<double> angles_K1a = tk1_angles_from_unitary(K1a);
430  std::vector<double> angles_K1b = tk1_angles_from_unitary(K1b);
431  std::vector<double> angles_K2a = tk1_angles_from_unitary(K2a);
432  std::vector<double> angles_K2b = tk1_angles_from_unitary(K2b);
433  c.add_op<unsigned>(
434  OpType::TK1, {angles_K2a.begin(), angles_K2a.end() - 1}, {0});
435  c.add_op<unsigned>(
436  OpType::TK1, {angles_K2b.begin(), angles_K2b.end() - 1}, {1});
437  double alpha = std::get<0>(A);
438  double beta = std::get<1>(A);
439  double gamma = std::get<2>(A);
440 
441  c.append(CircPool::TK2_using_normalised_TK2(alpha, beta, gamma));
442 
443  c.add_op<unsigned>(
444  OpType::TK1, {angles_K1a.begin(), angles_K1a.end() - 1}, {0});
445  c.add_op<unsigned>(
446  OpType::TK1, {angles_K1b.begin(), angles_K1b.end() - 1}, {1});
447 
448  // Correct phase by computing the unitary and comparing with U:
449  Eigen::Matrix4cd V_K1 = Eigen::KroneckerProduct(
450  get_matrix_from_tk1_angles(
451  {angles_K1a[0], angles_K1a[1], angles_K1a[2], 0}),
452  get_matrix_from_tk1_angles(
453  {angles_K1b[0], angles_K1b[1], angles_K1b[2], 0}));
454  Eigen::Matrix4cd V_A =
455  internal::GateUnitaryMatrixImplementations::TK2(alpha, beta, gamma);
456  Eigen::Matrix4cd V_K2 = Eigen::KroneckerProduct(
457  get_matrix_from_tk1_angles(
458  {angles_K2a[0], angles_K2a[1], angles_K2a[2], 0}),
459  get_matrix_from_tk1_angles(
460  {angles_K2b[0], angles_K2b[1], angles_K2b[2], 0}));
461  Eigen::Matrix4cd V = V_K1 * V_A * V_K2;
462  Eigen::Matrix4cd R = V.adjoint() * U;
463  const Complex phase = R(0, 0); // R = phase * I
464  c.add_phase(arg(phase) / PI);
465 
466  return c;
467  }
468  // Now the non-trivial cases.
469  switch (op->get_type()) {
470  case OpType::ISWAP:
471  return CircPool::ISWAP_using_TK2(params[0]);
472  case OpType::PhasedISWAP:
473  return CircPool::PhasedISWAP_using_TK2(params[0], params[1]);
474  case OpType::XXPhase:
475  return CircPool::XXPhase_using_TK2(params[0]);
476  case OpType::YYPhase:
477  return CircPool::YYPhase_using_TK2(params[0]);
478  case OpType::ZZPhase:
479  return CircPool::ZZPhase_using_TK2(params[0]);
480  case OpType::NPhasedX:
481  return CircPool::NPhasedX_using_PhasedX(n, params[0], params[1]);
482  case OpType::ESWAP:
483  return CircPool::ESWAP_using_TK2(params[0]);
484  case OpType::FSim:
485  return CircPool::FSim_using_TK2(params[0], params[1]);
486  case OpType::CRx:
487  return CircPool::CRx_using_TK2(params[0]);
488  case OpType::CRy:
489  return CircPool::CRy_using_TK2(params[0]);
490  case OpType::CRz:
491  return CircPool::CRz_using_TK2(params[0]);
492  case OpType::CU1:
493  return CircPool::CU1_using_TK2(params[0]);
494  case OpType::XXPhase3:
495  return CircPool::XXPhase3_using_TK2(params[0]);
496  case OpType::CCX:
497  case OpType::CSWAP:
498  case OpType::BRIDGE:
499  case OpType::CU3:
500  case OpType::PhaseGadget: {
501  // As a first, inefficient, solution, decompose these into CX and then
502  // replace each CX with a TK2 (and some single-qubit gates).
503  // TODO Find more efficient decompositions for these gates.
504  Circuit c = with_CX(op);
505  replace_CX_with_TK2(c);
506  return c;
507  }
508  default:
509  throw CircuitInvalidity("Cannot decompose " + op->get_name());
510  }
511 }

513 Circuit with_CX(Gate_ptr op) {
514  OpType optype = op->get_type();
515  std::vector<Expr> params = op->get_params();
516  unsigned n = op->n_qubits();
517  if (n == 0) {
518  Circuit c(0);
519  if (op->get_type() == OpType::Phase) {
520  c.add_phase(op->get_params()[0]);
521  }
522  return c;
523  } else if (n == 1) {
524  Circuit c(1);
525  c.add_op(op, std::vector<unsigned>{0});
526  return c;
527  }
528  switch (optype) {
529  case OpType::CX: {
530  Circuit c(2);
531  c.add_op(op, std::vector<unsigned>{0, 1});
532  return c;
533  }
534  case OpType::CCX:
535  return CircPool::CCX_normal_decomp();
536  case OpType::CY:
537  return CircPool::CY_using_CX();
538  case OpType::CZ:
539  return CircPool::CZ_using_CX();
540  case OpType::CH:
541  return CircPool::CH_using_CX();
542  case OpType::CV:
543  return CircPool::CV_using_CX();
544  case OpType::CVdg:
545  return CircPool::CVdg_using_CX();
546  case OpType::CSX:
547  return CircPool::CSX_using_CX();
548  case OpType::CSXdg:

621 static Circuit with_controls_symbolic(const Circuit &c, unsigned n_controls) {
622  if (c.n_bits() != 0 || !c.is_simple()) {
623  throw CircuitInvalidity("Only default qubit register allowed");
624  }
625 
626  Circuit c1(c);
627  // Replace wire swaps with SWAP gates
628  c1.replace_all_implicit_wire_swaps();
629 
630  // Dispose of the trivial case
631  if (n_controls == 0) {
632  return c1;
633  }
634 
635  static const OpTypeSet multiq_gate_set = {
636  OpType::CX, OpType::CCX, OpType::CnX, OpType::CRy, OpType::CnRy,
637  OpType::CZ, OpType::CnZ, OpType::CY, OpType::CnY};
638 
639  unsigned c_n_qubits = c1.n_qubits();
640 
641  // 1. Rebase to {CX, CCX, CnX, CnRy} and single-qubit gates
642  VertexList bin;
643  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
644  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
645  OpType optype = op->get_type();
646  if (is_gate_type(optype)) {
647  if (is_projective_type(optype)) {
648  throw CircuitInvalidity("Projective operations present");
649  }
650  if (is_box_type(optype)) {
651  throw CircuitInvalidity("Undecomposed boxes present");
652  }
653  if (is_single_qubit_type(optype)) {
654  continue;
655  }
656  if (multiq_gate_set.find(optype) != multiq_gate_set.end()) {
657  continue;
658  }
659  Circuit replacement = with_CX(as_gate_ptr(op));
660  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
661  bin.push_back(v);
662  }
663  }
664  c1.remove_vertices(
665  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
666 
667  // Capture the phase. We may adjust this during replacements below.
668  Expr a = c1.get_phase();
669 
670  // 2. Replace all gates with controlled versions
671  Circuit c2(n_controls + c_n_qubits);
672  for (Circuit::CommandIterator cit = c1.begin(); cit != c1.end(); ++cit) {
673  Op_ptr op = cit->get_op_ptr();
674  OpType optype = op->get_type();
675  unit_vector_t args = cit->get_args();
676  unsigned n_args = args.size();
677  if (optype == OpType::Barrier) {
678  qubit_vector_t barrier_args(n_args);
679  for (unsigned i = 0; i < n_args; i++) {
680  barrier_args[i] = Qubit(n_controls + args[i].index()[0]);
681  }
682  c2.add_op(op, barrier_args);
683  continue;
684  }
685  unsigned n_new_args = n_controls + n_args;
686  qubit_vector_t new_args(n_new_args);
687  for (unsigned i = 0; i < n_controls; i++) {
688  new_args[i] = Qubit(i);
689  }
690  for (unsigned i = 0; i < n_args; i++) {
691  new_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
692  }
693  std::vector<Expr> params = op->get_params();
694  switch (optype) {
695  case OpType::noop:
696  break;
697  case OpType::X:
698  case OpType::CX:
699  case OpType::CCX:
700  case OpType::CnX:
701  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
702  break;
703  case OpType::Ry:
704  case OpType::CRy:
705  case OpType::CnRy:
706  c2.add_op<Qubit>(CNRYTYPE(n_new_args), params, new_args);
707  break;
708  case OpType::Z:
709  case OpType::CZ:
710  case OpType::CnZ:
711  c2.add_op<Qubit>(CNZTYPE(n_new_args), new_args);
712  break;
713  case OpType::Y:
714  case OpType::CY:
715  case OpType::CnY:
716  c2.add_op<Qubit>(CNYTYPE(n_new_args), new_args);
717  break;
718  default: {
719  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
720  Expr theta = tk1_angles[1];
721  Expr phi = tk1_angles[0] - 0.5;
722  Expr lambda = tk1_angles[2] + 0.5;
723  Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
724  // Operation is U3(theta, phi, lambda) + phase t.
725  // First absorb t in the overall phase.
726  a += t;
727  // Construct a multi-controlled U3, by extending the standard
728  // CU3-to-CX decomposition.
729  Qubit target = new_args[n_controls];
730  Circuit cnu1 = CnU1(n_controls - 1, 0.5 * (lambda + phi));
731  c2.append(cnu1);
732  c2.add_op<Qubit>(OpType::U1, 0.5 * (lambda - phi), {target});
733  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
734  c2.add_op<Qubit>(
735  OpType::U3, {-0.5 * theta, 0, -0.5 * (lambda + phi)}, {target});
736  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
737  c2.add_op<Qubit>(OpType::U3, {0.5 * theta, phi, 0}, {target});
738  } break;
739  }
740  }
741 
742  // 3. Account for phase by appending a CnU1 to the control qubits.
743  if (!equiv_0(a)) {
744  Circuit cnu1 = CnU1(n_controls - 1, a);
745  c2.append(cnu1);
746  }
747 
748  c2.remove_noops();
749  return c2;
750 }

797 // A gate block containing Cn* gates that can be merged as a single CnU gate
798 // a block can also contain a single Barrier, which will be left in place
799 struct CnGateBlock {
800  enum class MergeMode { append, prepend };
801  CnGateBlock(const Command &command) {
802  // Assumes the color of the target is not identity
803  Op_ptr op = command.get_op_ptr();
804  ops.push_back(op);
805  unit_vector_t args = command.get_args();
806  TKET_ASSERT(!args.empty());
807  for (unsigned i = 0; i < args.size() - 1; i++) {
808  control_qubits.insert(args[i].index()[0]);
809  }
810  target_qubit = args.back().index()[0];
811  is_barrier = (op->get_type() == OpType::Barrier);
812  is_symmetric =
813  (op->get_type() == OpType::CZ || op->get_type() == OpType::CnZ ||
814  op->get_type() == OpType::CU1);
815  color = is_barrier ? std::nullopt
816  : as_gate_ptr(op)->commuting_basis(args.size() - 1);
817  if (color == Pauli::I) {
818  throw std::invalid_argument(
819  "CnGateBlock doesn't accept multi-controlled identity gate.");
820  }
821  }
822 
823  // Check whether commute with another CnGateBlock
824  bool commutes_with(const CnGateBlock &other) {
825  if (is_barrier || other.is_barrier) {
826  // they commute only if they have no args in common
827  std::set<unsigned> common_args;
828  std::set<unsigned> args = control_qubits;
829  args.insert(target_qubit);
830  std::set<unsigned> other_args = other.control_qubits;
831  other_args.insert(other.target_qubit);
832  std::set_intersection(
833  args.begin(), args.end(), other_args.begin(), other_args.end(),
834  std::inserter(common_args, common_args.begin()));
835  return common_args.empty();
836  }
837  if (target_qubit == other.target_qubit) {
838  return (color == other.color && color != std::nullopt);
839  }
840  if (control_qubits.contains(other.target_qubit) &&
841  other.color != Pauli::Z) {
842  return false;
843  }
844  if (other.control_qubits.contains(target_qubit) && color != Pauli::Z) {
845  return false;
846  }
847  return true;
848  }
849 
850  // Check whether can be merged with another CnGateBlock
851  bool is_mergeable_with(const CnGateBlock &other) {
852  if (is_barrier || other.is_barrier) {
853  return false;
854  }
855  // check if sizes match
856  if (control_qubits.size() != other.control_qubits.size()) {
857  return false;
858  }
859  // check if they act on the same set of qubits
860  std::set<unsigned> args_a = control_qubits;
861  args_a.insert(target_qubit);
862  std::set<unsigned> args_b = other.control_qubits;
863  args_b.insert(other.target_qubit);
864  if (args_a != args_b) {
865  return false;
866  }
867  // false if target don't match and none of them is symmetric
868  if (target_qubit != other.target_qubit && !is_symmetric &&
869  !other.is_symmetric) {
870  return false;
871  }
872  return true;
873  }
874 
875  // Merge with another CnGateBlock
876  void merge(CnGateBlock &other, const MergeMode &mode) {
877  if (mode == MergeMode::append) {
878  ops.insert(ops.end(), other.ops.begin(), other.ops.end());
879  } else {
880  ops.insert(ops.begin(), other.ops.begin(), other.ops.end());
881  }
882  color = (color != other.color) ? std::nullopt : color;
883  if (is_symmetric && !other.is_symmetric) {
884  control_qubits = other.control_qubits;
885  target_qubit = other.target_qubit;
886  is_symmetric = false;
887  }
888  // empty the other CnGateBlock
889  other.ops.clear();
890  }
891 
892  Eigen::Matrix2cd get_target_unitary() const {
893  Eigen::Matrix2cd m = Eigen::Matrix2cd::Identity();
894  for (const Op_ptr &op : ops) {
895  m = get_target_op_matrix(op) * m;
896  }
897  return m;
898  }
899 
900  // ops in the block
901  std::vector<Op_ptr> ops;
902  // target qubit index
903  unsigned target_qubit;
904  // control indices
905  std::set<unsigned> control_qubits;
906  // whether the block is used as a barrier
907  bool is_barrier;
908  // whether the target can act on any of its qubits
909  bool is_symmetric;
910  // color of the target qubit
911  std::optional<Pauli> color;
912 };

914 // Construct a controlled version of a given circuit
915 // with the assumption that the circuit does not have symbols.
916 static Circuit with_controls_numerical(const Circuit &c, unsigned n_controls) {
917  if (c.n_bits() != 0 || !c.is_simple()) {
918  throw CircuitInvalidity("Only default qubit register allowed");
919  }
920 
921  Circuit c1(c);
922  // Replace wire swaps with SWAP gates
923  c1.replace_all_implicit_wire_swaps();
924 
925  // Dispose of the trivial case
926  if (n_controls == 0) {
927  return c1;
928  }
929  // 1. Rebase to Cn* gates (n=0 for single qubit gates)
930  VertexList bin;
931  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
932  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
933  OpType optype = op->get_type();
934  if (is_gate_type(optype)) {
935  if (is_projective_type(optype)) {
936  throw CircuitInvalidity("Projective operations present");
937  }
938  if (is_box_type(optype)) {
939  throw CircuitInvalidity("Undecomposed boxes present");
940  }
941  if (is_single_qubit_type(optype) || is_controlled_gate_type(optype)) {
942  continue;
943  }
944  Circuit replacement = with_CX(as_gate_ptr(op));
945  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
946  bin.push_back(v);
947  } else if (
948  optype != OpType::Input && optype != OpType::Output &&
949  optype != OpType::Barrier) {
950  throw CircuitInvalidity(
951  "Cannot construct the controlled version of " + op->get_name());
952  }
953  }
954  c1.remove_vertices(
955  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
956 
957  // 2. try to partition the circuit into blocks of Cn* gates such that
958  // the gates in each block can be merged into a single CnU gate
959  std::vector<Command> commands = c1.get_commands();
960  std::vector<CnGateBlock> blocks;
961 
962  Expr controlled_phase = c1.get_phase();
963 
964  for (const Command &cmd : commands) {
965  // if the gate is an identity up to a phase, add it as a controlled phase
966  std::optional<double> phase = std::nullopt;
967  if (cmd.get_op_ptr()->get_type() != OpType::Barrier) {
968  phase = cmd.get_op_ptr()->is_identity();
969  }
970  if (phase != std::nullopt) {
971  controlled_phase += phase.value();
972  } else {
973  blocks.push_back(CnGateBlock(cmd));
974  }
975  }
976 
977  // iterate the blocks from left to right
978  for (unsigned i = 0; i + 1 < blocks.size(); i++) {
979  CnGateBlock &b = blocks[i];
980  if (b.ops.empty()) {
981  continue;
982  }
983  // try to merge b to a block on the right
984  for (unsigned j = i + 1; j < blocks.size(); j++) {
985  CnGateBlock &candidate = blocks[j];
986  if (candidate.ops.empty()) {
987  continue;
988  }
989  if (b.is_mergeable_with(candidate)) {
990  candidate.merge(b, CnGateBlock::MergeMode::prepend);
991  break;
992  }
993  if (!b.commutes_with(candidate)) {
994  break;
995  }
996  }
997  }
998 
999  // iterate the blocks from right to left
1000  for (unsigned i = blocks.size(); i-- > 1;) {
1001  CnGateBlock &b = blocks[i];
1002  if (b.ops.empty()) {
1003  continue;
1004  }
1005  // try to merge b to a block on the left
1006  // iterate from i-1 to 0
1007  for (unsigned j = i; j-- > 0;) {
1008  CnGateBlock &candidate = blocks[j];
1009  if (candidate.ops.empty()) {
1010  continue;
1011  }
1012  if (b.is_mergeable_with(candidate)) {
1013  candidate.merge(b, CnGateBlock::MergeMode::append);
1014  break;
1015  }
1016  if (!b.commutes_with(candidate)) {
1017  break;
1018  }
1019  }
1020  }
1021  // 3. Add each block to c2 either as a CnX, CnZ, CnY, CnU decomposition
1022  // or an in-place Barrier
1023  Circuit c2(n_controls + c1.n_qubits());
1024  const static Eigen::Matrix2cd X = Gate(OpType::X, {}, 1).get_unitary();
1025  const static Eigen::Matrix2cd Y = Gate(OpType::Y, {}, 1).get_unitary();
1026  const static Eigen::Matrix2cd Z = Gate(OpType::Z, {}, 1).get_unitary();
1027 
1028  for (const CnGateBlock &b : blocks) {
1029  if (b.ops.empty()) {
1030  continue;
1031  }
1032  // Barriers are left in place
1033  if (b.is_barrier) {
1034  qubit_vector_t new_args;
1035  for (const unsigned i : b.control_qubits) {
1036  new_args.push_back(Qubit(i + n_controls));
1037  }
1038  new_args.push_back(Qubit(b.target_qubit + n_controls));
1039  TKET_ASSERT(b.ops.size() == 1);
1040  c2.add_op(b.ops[0], new_args);
1041  continue;
1042  }
1043  // Computes the target unitary
1044  Eigen::Matrix2cd m = b.get_target_unitary();
1045  if (m.isApprox(Eigen::Matrix2cd::Identity(), EPS)) {
1046  continue;
1047  }
1048  std::function<qubit_vector_t()> get_args = [&]() {
1049  qubit_vector_t new_args;
1050  for (unsigned i = 0; i < n_controls; i++) {
1051  new_args.push_back(Qubit(i));
1052  }
1053  for (const unsigned i : b.control_qubits) {
1054  new_args.push_back(Qubit(i + n_controls));
1055  }
1056  new_args.push_back(Qubit(b.target_qubit + n_controls));
1057  return new_args;
1058  };
1059  if (m.isApprox(X, EPS)) {
1060  qubit_vector_t new_args = get_args();
1061  c2.add_op<Qubit>(CNXTYPE(new_args.size()), new_args);
1062  continue;
1063  }
1064  if (m.isApprox(Y, EPS)) {
1065  qubit_vector_t new_args = get_args();
1066  c2.add_op<Qubit>(CNYTYPE(new_args.size()), new_args);
1067  continue;
1068  }
1069  if (m.isApprox(Z, EPS)) {
1070  qubit_vector_t new_args = get_args();
1071  c2.add_op<Qubit>(CNZTYPE(new_args.size()), new_args);
1072  continue;
1073  }
1074  unit_map_t unit_map;
1075  for (unsigned i = 0; i < n_controls; i++) {
1076  unit_map.insert({Qubit(i), Qubit(i)});
1077  }
1078  unsigned control_index = n_controls;
1079  for (const unsigned i : b.control_qubits) {
1080  unit_map.insert({Qubit(control_index++), Qubit(i + n_controls)});
1081  }
1082  unit_map.insert({Qubit(control_index), Qubit(b.target_qubit + n_controls)});
1083 
1084  unsigned total_controls = b.control_qubits.size() + n_controls;
1085 
1086  Circuit replacement;
1087 
1088  // Check if the matrix is SU(2)
1089  if (std::abs(m.determinant() - 1.) < EPS) {
1090  // We have three functions that can decompose a multi-controlled SU(2).
1091  // The choice is based on the average number of CXs they produce for a
1092  // random n-controlled SU(2) gate.
1093  if (total_controls > 2 && total_controls < 5) {
1094  replacement = CircPool::CnU_gray_code_decomp(total_controls, m);
1095  } else if (total_controls >= 5 && total_controls < 9) {
1096  replacement = CircPool::CnU_linear_depth_decomp(total_controls, m);
1097  } else {
1098  // Compute the SU(2) angles from the TK1 angles
1099  std::vector<double> angles = tk1_angles_from_unitary(m);
1100  if (equiv_val(angles[3], 1., 2)) {
1101  // if the phase is odd, it can be absorbed into the first Rz rotation
1102  angles[0] = angles[0] + 2;
1103  } else {
1104  // because it's SU(2), the phase must be integers
1105  TKET_ASSERT(equiv_0(angles[3], 2));

1146  Expr a, Expr b, Expr c) {
1147  std::optional<double> a_eval = eval_expr_mod(a, 4);
1148  std::optional<double> b_eval = eval_expr_mod(b, 4);
1149  std::optional<double> c_eval = eval_expr_mod(c, 4);
1150 
1151  Circuit pre(2), post(2);
1152 
1153  // Add ot.dagger() at beggining and ot at end.
1154  auto conj = [&pre, &post](OpType ot) {
1155  Op_ptr op = get_op_ptr(ot);
1156  Op_ptr opdg = op->dagger();
1157  pre.add_op<unsigned>(opdg, {0});
1158  pre.add_op<unsigned>(opdg, {1});
1159  // These get undaggered at the end
1160  post.add_op<unsigned>(opdg, {0});
1161  post.add_op<unsigned>(opdg, {1});
1162  };
1163 
1164  // Step 1: For non-symbolic: a, b, c [0, 1] [3, 4].
1165  if (a_eval && *a_eval > 1. && *a_eval <= 3.) {
1166  a -= 2;
1167  *a_eval -= 2;
1168  pre.add_phase(1);
1169  *a_eval = fmodn(*a_eval, 4);
1170  }
1171  if (b_eval && *b_eval > 1. && *b_eval <= 3.) {
1172  b -= 2;
1173  *b_eval -= 2;
1174  pre.add_phase(1);
1175  *b_eval = fmodn(*b_eval, 4);
1176  }
1177  if (c_eval && *c_eval > 1. && *c_eval <= 3.) {
1178  c -= 2;
1179  *c_eval -= 2;
1180  pre.add_phase(1);
1181  *c_eval = fmodn(*c_eval, 4);
1182  }
1183 
1184  // Step 2: Make sure that symbolic expressions come before non-symbolics.
1185  if (a_eval && !b_eval) {
1186  // Swap XX and YY.
1187  conj(OpType::S);
1188  std::swap(a, b);
1189  std::swap(a_eval, b_eval);
1190  } else if (a_eval && !c_eval) {
1191  // Swap XX and ZZ.
1192  conj(OpType::H);
1193  std::swap(a, c);
1194  std::swap(a_eval, c_eval);
1195  }
1196  if (b_eval && !c_eval) {
1197  // Swap YY and ZZ.
1198  conj(OpType::V);
1199  std::swap(b, c);
1200  std::swap(b_eval, c_eval);
1201  }
1202 
1203  // Step 3: Order non-symbolic expressions in decreasing order.
1204  auto val_in_weyl = [](double r) {
1205  // Value of r once projected into Weyl chamber.
1206  return std::min(fmodn(r, 1), 1 - fmodn(r, 1));
1207  };
1208  if (a_eval && b_eval && val_in_weyl(*a_eval) < val_in_weyl(*b_eval)) {
1209  // Swap XX and YY.
1210  conj(OpType::S);
1211  std::swap(a, b);
1212  std::swap(a_eval, b_eval);
1213  }
1214  if (b_eval && c_eval && val_in_weyl(*b_eval) < val_in_weyl(*c_eval)) {
1215  // Swap YY and ZZ.
1216  conj(OpType::V);
1217  std::swap(b, c);
1218  std::swap(b_eval, c_eval);
1219  }
1220  if (a_eval && b_eval && val_in_weyl(*a_eval) < val_in_weyl(*b_eval)) {
1221  // Swap XX and YY.
1222  conj(OpType::S);
1223  std::swap(a, b);
1224  std::swap(a_eval, b_eval);
1225  }
1226 
1227  // Step 4: Project into Weyl chamber.
1228  if (a_eval && *a_eval > 1.) {
1229  a -= 3.;
1230  *a_eval -= 3;
1231  post.add_op<unsigned>(OpType::X, {0});
1232  post.add_op<unsigned>(OpType::X, {1});
1233  pre.add_phase(0.5);
1234  }
1235  if (b_eval && *b_eval > 1.) {
1236  b -= 3.;
1237  *b_eval -= 3;
1238  post.add_op<unsigned>(OpType::Y, {0});
1239  post.add_op<unsigned>(OpType::Y, {1});
1240  pre.add_phase(0.5);
1241  }
1242  if (c_eval && *c_eval > 1.) {
1243  c -= 3.;
1244  *c_eval -= 3;
1245  post.add_op<unsigned>(OpType::Z, {0});
1246  post.add_op<unsigned>(OpType::Z, {1});
1247  pre.add_phase(0.5);
1248  }
1249  if (a_eval && *a_eval > .5) {
1250  a = 1. - a;
1251  *a_eval = 1. - *a_eval;
1252  b = 1. - b;
1253  *b_eval = 1. - *b_eval;
1254  pre.add_op<unsigned>(OpType::Z, {0});
1255  post.add_op<unsigned>(OpType::Z, {1});
1256  }
1257  if (b_eval && (*b_eval > .5)) {
1258  b = 1 - b;
1259  *b_eval = 1. - *b_eval;
1260  c = 1 - c;
1261  *c_eval = 1. - *c_eval;
1262  pre.add_op<unsigned>(OpType::X, {0});
1263  post.add_op<unsigned>(OpType::X, {1});
1264  }
1265  if (c_eval && *c_eval > .5) {
1266  c -= 1;
1267  *c_eval -= 1.;
1268  post.add_op<unsigned>(OpType::Z, {0});
1269  post.add_op<unsigned>(OpType::Z, {1});
1270  pre.add_phase(-0.5);
1271  }
1272  // Cheeky way of reversing order of ops.
1273  post = post.dagger();
1274 
1275  return {pre, {a, b, c}, post};
1276 }
