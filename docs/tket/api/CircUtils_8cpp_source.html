47 Eigen::Matrix2cd get_matrix_from_circ(const Circuit &circ) {
48  if (circ.n_qubits() != 1)
49  throw CircuitInvalidity(
50  "Getting Matrix: expected 1 qubit circuit, found " +
51  std::to_string(circ.n_qubits()));
52  Complex factor = std::exp(i_ * PI * eval_expr(circ.get_phase()).value());
53  VertexVec qpath = circ.qubit_path_vertices(circ.all_qubits()[0]);
54  unsigned N = qpath.size();
55  if (N == 2) return factor * Eigen::Matrix2cd::Identity();
56  Eigen::Matrix2cd m = get_matrix(circ, qpath[N - 2]);
57  for (unsigned x = N - 3; x >= 1; --x) {
58  m = m * get_matrix(circ, qpath[x]);
59  }
60  return factor * m;
61 }

63 Eigen::Matrix4cd get_matrix_from_2qb_circ(const Circuit &circ) {
64  std::vector<QPathDetailed> all_paths = circ.all_qubit_paths();
65  std::map<Vertex, Eigen::Matrix4cd> v_to_op;
66  Eigen::Matrix4cd cnot, tonc, swap;
67  // clang-format off
68  cnot << 1, 0, 0, 0,
69  0, 1, 0, 0,
70  0, 0, 0, 1,
71  0, 0, 1, 0;
72  tonc << 1, 0, 0, 0,
73  0, 0, 0, 1,
74  0, 0, 1, 0,
75  0, 1, 0, 0;
76  swap << 1, 0, 0, 0,
77  0, 0, 1, 0,
78  0, 1, 0, 0,
79  0, 0, 0, 1;
80  // clang-format on
81 
82  for (unsigned uqb = 0; uqb < 2; uqb++) {
83  for (QPathDetailed::iterator it = all_paths[uqb].begin();
84  it != all_paths[uqb].end(); ++it) {
85  const Op_ptr o = circ.get_Op_ptr_from_Vertex(it->first);
86  switch (o->get_type()) {
87  case OpType::Input:
88  case OpType::Create:
89  case OpType::Output:
90  case OpType::Discard: {
91  v_to_op[it->first] = Eigen::Matrix4cd::Identity();
92  break;
93  }
94  case OpType::SWAP: {
95  if (uqb == 0) v_to_op[it->first] = swap;
96  break;
97  }
98  case OpType::CX: {
99  if (uqb == 0) {
100  if (it->second == 0) {
101  v_to_op[it->first] = cnot;
102  } else {
103  v_to_op[it->first] = tonc;
104  }
105  }
106  break;
107  }
108  case OpType::TK2: {
109  auto params = o->get_params();
110  TKET_ASSERT(params.size() == 3);
111  v_to_op[it->first] =
112  get_matrix_from_2qb_circ(CircPool::normalised_TK2_using_CX(
113  params[0], params[1], params[2]));
114  break;
115  }
116  default: {
117  if (o->get_desc().is_gate() && circ.n_in_edges(it->first) == 1 &&
118  circ.n_out_edges(it->first) == 1) {
119  const Op_ptr g = o;
120  std::vector<Expr> ps = as_gate_ptr(g)->get_tk1_angles();
121  Eigen::Matrix2cd mat = get_matrix_from_tk1_angles(ps);
122  if (uqb == 0) {
123  v_to_op[it->first] =
124  Eigen::kroneckerProduct(mat, Eigen::Matrix2cd::Identity());
125  } else {
126  v_to_op[it->first] =
127  Eigen::kroneckerProduct(Eigen::Matrix2cd::Identity(), mat);
128  }
129  } else
130  throw BadOpType("Cannot obtain matrix from op", o->get_type());
131  }
132  }
133  }
134  }
135  Eigen::Matrix4cd m = Eigen::Matrix4cd::Identity();
136  SliceVec slices = circ.get_slices();
137  for (const Slice &s : slices) {
138  for (const Vertex &v : s) {
139  m = v_to_op[v] * m;
140  }
141  }
142  return std::exp(i_ * PI * eval_expr(circ.get_phase()).value()) * m;
143 }

145 Circuit two_qubit_canonical(const Eigen::Matrix4cd &U, OpType target_2qb_gate) {
146  if (!is_unitary(U)) {
147  throw std::invalid_argument(
148  "Non-unitary matrix passed to two_qubit_canonical");
149  }
150 
151  auto [K1, A, K2] = get_information_content(U);
152 
153  K1 /= pow(K1.determinant(), 0.25);
154  K2 /= pow(K2.determinant(), 0.25);
155  auto [a, b, c] = A;
156 
157  // Decompose single qubits
158  auto [K1a, K1b] = kronecker_decomposition(K1);
159  auto [K2a, K2b] = kronecker_decomposition(K2);
160 
161  Circuit result(2);
162 
163  std::vector<double> angles_q0 = tk1_angles_from_unitary(K2a);
164  std::vector<double> angles_q1 = tk1_angles_from_unitary(K2b);
165  result.add_op<unsigned>(
166  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
167  result.add_op<unsigned>(
168  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
169 
170  switch (target_2qb_gate) {
171  case OpType::TK2:
172  result.append(CircPool::TK2_using_normalised_TK2(a, b, c));
173  break;
174  case OpType::CX:
175  result.append(CircPool::TK2_using_CX(a, b, c));
176  break;
177  default:
178  throw std::invalid_argument("target_2qb_gate must be CX or TK2.");
179  }
180 
181  angles_q0 = tk1_angles_from_unitary(K1a);
182  angles_q1 = tk1_angles_from_unitary(K1b);
183  result.add_op<unsigned>(
184  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
185  result.add_op<unsigned>(
186  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
187 
188  // this fixes phase if decomposition is exact
189  Eigen::Matrix4cd reminder = get_matrix_from_2qb_circ(result).adjoint() * U;
190  const Complex phase = reminder(0, 0); // reminder = phase * I
191  result.add_phase(arg(phase) / PI);
192  return result;
193 }

195 // Factorize U as VD where V corresponds to a 2-CX circuit and
196 // D = diag(z, z*, z*, z). Return V and z.
197 static std::pair<Eigen::Matrix4cd, Complex> decompose_VD(
198  const Eigen::Matrix4cd &U) {
199  if (!is_unitary(U)) {
200  throw std::invalid_argument("Non-unitary matrix passed to decompose_VD");
201  }
202 
203  // The calculations below are derived from the proof of Proposition V.2 in
204  // https://arxiv.org/abs/quant-ph/0308033.
205 
206  Eigen::Matrix4cd u = U / pow(U.determinant(), 0.25);
207  Complex a = u(3, 0) * u(0, 3) - u(2, 0) * u(1, 3) - u(1, 0) * u(2, 3) +
208  u(0, 0) * u(3, 3);
209  Complex b = u(3, 1) * u(0, 2) - u(2, 1) * u(1, 2) - u(1, 1) * u(2, 2) +
210  u(0, 1) * u(3, 2);
211  // Now we want to find z such that |z|=1 and (az* - bz) is real.
212  // The numerical stability of this function is a concern when a is close to
213  // -b*. This problem can be demonstrated in artificially constructed
214  // examples (passing unitaries very close to, but not quite, the identity to
215  // the functions below). In these cases the product VD (or DV) may not
216  // approximate U to within Eigen's default tolerance. Is there a way to
217  // dodge this issue?
218  Complex w = a + std::conj(b);
219  double d = std::abs(w);
220  // If w = 0 then we can set z = 1.
221  Complex z = (d < EPS) ? 1. : w / d;
222  Complex z0 = sqrt(z);
223  Complex z1 = std::conj(z0);
224  Eigen::Matrix4cd V = U;
225  V.col(0) *= z1;
226  V.col(1) *= z0;
227  V.col(2) *= z0;
228  V.col(3) *= z1;
229  return {V, z0};
230 }

232 static void replace_TK2_2CX(Circuit &circ) {
233  VertexList bin;
234  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
235  if (circ.get_OpType_from_Vertex(v) != OpType::TK2) continue;
236  auto params = circ.get_Op_ptr_from_Vertex(v)->get_params();
237  TKET_ASSERT(params.size() == 3);
238  // Rounding errors can accumulate here; warn if so:
239  if (!equiv_0(params[2], 4, 1e-6)) {
240  std::stringstream ss;
241  ss << "Rounding errors in CX decomposition: ZZPhase parameter = "
242  << params[2] << " when it should be 0 (mod 4). Ignoring.";
243  tket_log()->warn(ss.str());
244  }
245  Circuit sub = CircPool::approx_TK2_using_2xCX(params[0], params[1]);
246  bin.push_back(v);
247  circ.substitute(sub, v, Circuit::VertexDeletion::No);
248  }
249  TKET_ASSERT(bin.size() == 1);
250  circ.remove_vertices(
251  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
252 }

255  auto [V, z0] = decompose_VD(U);
256  Circuit circ = two_qubit_canonical(V);
257  replace_TK2_2CX(circ);
258  return {circ, z0};

261 std::pair<Circuit, Complex> decompose_2cx_DV(const Eigen::Matrix4cd &U) {
262  auto [V, z0] = decompose_VD(U.adjoint());
263  V.adjointInPlace();
264  Circuit circ = two_qubit_canonical(V);
265  replace_TK2_2CX(circ);
266  return {circ, std::conj(z0)};

269 Circuit phase_gadget(unsigned n_qubits, const Expr &t, CXConfigType cx_config) {
270  // Handle n_qubits==0 as a special case, or the calculations below
271  // go badly wrong.
272  Circuit new_circ(n_qubits);
273  if (n_qubits == 0) {
274  new_circ.add_phase(-t / 2);
275  return new_circ;
276  }
277  switch (cx_config) {
278  case CXConfigType::Snake: {
279  for (unsigned i = n_qubits - 1; i != 0; --i) {
280  unsigned j = i - 1;
281  new_circ.add_op<unsigned>(OpType::CX, {i, j});
282  }
283  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
284  for (unsigned i = 0; i != n_qubits - 1; ++i) {
285  unsigned j = i + 1;
286  new_circ.add_op<unsigned>(OpType::CX, {j, i});
287  }
288  break;
289  }
290  case CXConfigType::Star: {
291  for (unsigned i = n_qubits - 1; i != 0; --i) {
292  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
293  }
294  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
295  for (unsigned i = 1; i != n_qubits; ++i) {
296  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
297  }
298  break;
299  }
300  case CXConfigType::Tree: {
301  unsigned complete_layers = floor(log2(n_qubits));
302  unsigned dense_end = pow(2, complete_layers);
303  for (unsigned i = 0; i < n_qubits - dense_end; i++)
304  new_circ.add_op<unsigned>(
305  OpType::CX, {dense_end + i, dense_end - 1 - i});
306  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
307  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
308  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
309  }
310  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
311  for (unsigned step_size = dense_end / 2; step_size >= 1; step_size /= 2) {
312  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
313  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
314  }
315  for (unsigned i = 0; i < n_qubits - dense_end; i++)
316  new_circ.add_op<unsigned>(
317  OpType::CX, {dense_end + i, dense_end - 1 - i});
318  break;
319  }
320  case CXConfigType::MultiQGate: {
321  std::vector<std::vector<unsigned>> conjugations;
322  int sign_correction = 1;
323  for (int q = n_qubits - 1; q > 0; q -= 2) {
324  if (q - 1 > 0) {
325  unsigned i = q, j = q - 1;
326  // this is only equal to the CX decompositions above
327  // up to phase, but phase differences are cancelled out by
328  // its dagger XXPhase(-1/2) below.
329  new_circ.add_op<unsigned>(OpType::H, {i});
330  new_circ.add_op<unsigned>(OpType::H, {j});
331  new_circ.add_op<unsigned>(OpType::XXPhase3, 0.5, {i, j, 0});
332  sign_correction *= -1;
333  conjugations.push_back({i, j, 0});
334  } else {
335  unsigned i = q;
336  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
337  conjugations.push_back({i, 0});
338  }
339  }
340  new_circ.add_op<unsigned>(OpType::Rz, sign_correction * t, {0});
341  for (const auto &conj : conjugations) {
342  if (conj.size() == 2) {
343  new_circ.add_op<unsigned>(OpType::CX, conj);
344  } else {
345  TKET_ASSERT(conj.size() == 3);
346  new_circ.add_op<unsigned>(OpType::XXPhase3, -0.5, conj);
347  new_circ.add_op<unsigned>(OpType::H, {conj[0]});
348  new_circ.add_op<unsigned>(OpType::H, {conj[1]});
349  }
350  }
351  break;
352  }
353  }
354  return new_circ;
355 }

357 Circuit pauli_gadget(
358  const std::vector<Pauli> &paulis, const Expr &t, CXConfigType cx_config) {
359  unsigned n = paulis.size();
360  Circuit circ(n);
361  std::vector<unsigned> qubits;
362  for (unsigned i = 0; i < n; i++) {
363  switch (paulis[i]) {
364  case Pauli::I:
365  break;
366  case Pauli::X:
367  circ.add_op<unsigned>(OpType::H, {i});
368  qubits.push_back(i);
369  break;
370  case Pauli::Y:
371  circ.add_op<unsigned>(OpType::V, {i});
372  qubits.push_back(i);
373  break;
374  case Pauli::Z:
375  qubits.push_back(i);
376  break;
377  }
378  }
379  if (qubits.empty()) {
380  circ.add_phase(-t / 2);
381  } else {
382  Vertex v = circ.add_op<unsigned>(OpType::PhaseGadget, t, qubits);
383  Circuit cx_gadget = phase_gadget(circ.n_in_edges(v), t, cx_config);
384  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
385  circ.substitute(cx_gadget, sub, Circuit::VertexDeletion::Yes);
386  for (unsigned i = 0; i < n; i++) {
387  switch (paulis[i]) {
388  case Pauli::I:
389  break;
390  case Pauli::X:
391  circ.add_op<unsigned>(OpType::H, {i});
392  break;
393  case Pauli::Y:
394  circ.add_op<unsigned>(OpType::Vdg, {i});
395  break;
396  case Pauli::Z:
397  break;
398  }
399  }
400  }
401  return circ;
402 }

409 Circuit with_TK2(Gate_ptr op) {
410  std::vector<Expr> params = op->get_params();
411  unsigned n = op->n_qubits();
412  if (n == 0) {
413  Circuit c(0);
414  if (op->get_type() == OpType::Phase) {
415  c.add_phase(op->get_params()[0]);
416  }
417  return c;
418  } else if (n == 1) {
419  Circuit c(1);
420  c.add_op(op, std::vector<unsigned>{0});
421  return c;
422  } else if (n == 2 && op->free_symbols().empty()) {
423  Eigen::Matrix4cd U = op->get_unitary();
424  auto [K1, A, K2] = get_information_content(U);
425  // Decompose single qubits
426  auto [K1a, K1b] = kronecker_decomposition(K1);
427  auto [K2a, K2b] = kronecker_decomposition(K2);
428  Circuit c(2);
429  std::vector<double> angles_K1a = tk1_angles_from_unitary(K1a);
430  std::vector<double> angles_K1b = tk1_angles_from_unitary(K1b);
431  std::vector<double> angles_K2a = tk1_angles_from_unitary(K2a);
432  std::vector<double> angles_K2b = tk1_angles_from_unitary(K2b);
433  c.add_op<unsigned>(
434  OpType::TK1, {angles_K2a.begin(), angles_K2a.end() - 1}, {0});
435  c.add_op<unsigned>(
436  OpType::TK1, {angles_K2b.begin(), angles_K2b.end() - 1}, {1});
437  double alpha = std::get<0>(A);
438  double beta = std::get<1>(A);
439  double gamma = std::get<2>(A);
440 
441  c.append(CircPool::TK2_using_normalised_TK2(alpha, beta, gamma));
442 
443  c.add_op<unsigned>(
444  OpType::TK1, {angles_K1a.begin(), angles_K1a.end() - 1}, {0});
445  c.add_op<unsigned>(
446  OpType::TK1, {angles_K1b.begin(), angles_K1b.end() - 1}, {1});
447 
448  // Correct phase by computing the unitary and comparing with U:
449  Eigen::Matrix4cd V_K1 = Eigen::KroneckerProduct(
450  get_matrix_from_tk1_angles(
451  {angles_K1a[0], angles_K1a[1], angles_K1a[2], 0}),
452  get_matrix_from_tk1_angles(
453  {angles_K1b[0], angles_K1b[1], angles_K1b[2], 0}));
454  Eigen::Matrix4cd V_A =
455  internal::GateUnitaryMatrixImplementations::TK2(alpha, beta, gamma);
456  Eigen::Matrix4cd V_K2 = Eigen::KroneckerProduct(
457  get_matrix_from_tk1_angles(
458  {angles_K2a[0], angles_K2a[1], angles_K2a[2], 0}),
459  get_matrix_from_tk1_angles(
460  {angles_K2b[0], angles_K2b[1], angles_K2b[2], 0}));
461  Eigen::Matrix4cd V = V_K1 * V_A * V_K2;
462  Eigen::Matrix4cd R = V.adjoint() * U;
463  const Complex phase = R(0, 0); // R = phase * I
464  c.add_phase(arg(phase) / PI);
465 
466  return c;
467  }
468  // Now the non-trivial cases.
469  switch (op->get_type()) {
470  case OpType::ISWAP:
471  return CircPool::ISWAP_using_TK2(params[0]);
472  case OpType::PhasedISWAP:
473  return CircPool::PhasedISWAP_using_TK2(params[0], params[1]);
474  case OpType::XXPhase:
475  return CircPool::XXPhase_using_TK2(params[0]);
476  case OpType::YYPhase:
477  return CircPool::YYPhase_using_TK2(params[0]);
478  case OpType::ZZPhase:
479  return CircPool::ZZPhase_using_TK2(params[0]);
480  case OpType::NPhasedX:
481  return CircPool::NPhasedX_using_PhasedX(n, params[0], params[1]);
482  case OpType::ESWAP:
483  return CircPool::ESWAP_using_TK2(params[0]);
484  case OpType::FSim:
485  return CircPool::FSim_using_TK2(params[0], params[1]);
486  case OpType::CRx:
487  return CircPool::CRx_using_TK2(params[0]);
488  case OpType::CRy:
489  return CircPool::CRy_using_TK2(params[0]);
490  case OpType::CRz:
491  return CircPool::CRz_using_TK2(params[0]);
492  case OpType::CU1:
493  return CircPool::CU1_using_TK2(params[0]);
494  case OpType::XXPhase3:
495  return CircPool::XXPhase3_using_TK2(params[0]);
496  case OpType::CCX:
497  case OpType::CSWAP:
498  case OpType::BRIDGE:
499  case OpType::CU3:
500  case OpType::PhaseGadget: {
501  // As a first, inefficient, solution, decompose these into CX and then
502  // replace each CX with a TK2 (and some single-qubit gates).
503  // TODO Find more efficient decompositions for these gates.
504  Circuit c = with_CX(op);
505  replace_CX_with_TK2(c);
506  return c;
507  }
508  default:
509  throw CircuitInvalidity("Cannot decompose " + op->get_name());
510  }
511 }

513 Circuit with_CX(Gate_ptr op) {
514  OpType optype = op->get_type();
515  std::vector<Expr> params = op->get_params();
516  unsigned n = op->n_qubits();
517  if (n == 0) {
518  Circuit c(0);
519  if (op->get_type() == OpType::Phase) {
520  c.add_phase(op->get_params()[0]);
521  }
522  return c;
523  } else if (n == 1) {
524  Circuit c(1);
525  c.add_op(op, std::vector<unsigned>{0});
526  return c;
527  }
528  switch (optype) {
529  case OpType::CX: {
530  Circuit c(2);
531  c.add_op(op, std::vector<unsigned>{0, 1});
532  return c;
533  }
534  case OpType::CCX:
535  return CircPool::CCX_normal_decomp();
536  case OpType::CY:
537  return CircPool::CY_using_CX();
538  case OpType::CZ:
539  return CircPool::CZ_using_CX();
540  case OpType::CH:
541  return CircPool::CH_using_CX();
542  case OpType::CV:
543  return CircPool::CV_using_CX();
544  case OpType::CVdg:
545  return CircPool::CVdg_using_CX();
546  case OpType::CSX:
547  return CircPool::CSX_using_CX();
548  case OpType::CSXdg:
549  return CircPool::CSXdg_using_CX();
550  case OpType::CRz:

621 static Circuit with_controls_symbolic(const Circuit &c, unsigned n_controls) {
622  if (c.n_bits() != 0 || !c.is_simple()) {
623  throw CircuitInvalidity("Only default qubit register allowed");
624  }
625 
626  Circuit c1(c);
627  // Replace wire swaps with SWAP gates
628  c1.replace_all_implicit_wire_swaps();
629 
630  // Dispose of the trivial case
631  if (n_controls == 0) {
632  return c1;

635  static const OpTypeSet multiq_gate_set = {
636  OpType::CX, OpType::CCX, OpType::CnX, OpType::CRy, OpType::CnRy,
637  OpType::CZ, OpType::CnZ, OpType::CY, OpType::CnY};
638 
639  unsigned c_n_qubits = c1.n_qubits();
640 
641  // 1. Rebase to {CX, CCX, CnX, CnRy} and single-qubit gates
642  VertexList bin;
643  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
644  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
645  OpType optype = op->get_type();
646  if (is_gate_type(optype)) {
647  if (is_projective_type(optype)) {
648  throw CircuitInvalidity("Projective operations present");
649  }
650  if (is_box_type(optype)) {
651  throw CircuitInvalidity("Undecomposed boxes present");
652  }
653  if (is_single_qubit_type(optype)) {
654  continue;
655  }
656  if (multiq_gate_set.find(optype) != multiq_gate_set.end()) {
657  continue;
658  }
659  Circuit replacement = with_CX(as_gate_ptr(op));
660  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
661  bin.push_back(v);
662  }
663  }
664  c1.remove_vertices(
665  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
666 
667  // Capture the phase. We may adjust this during replacements below.
668  Expr a = c1.get_phase();
669 
670  // 2. Replace all gates with controlled versions
671  Circuit c2(n_controls + c_n_qubits);
672  for (Circuit::CommandIterator cit = c1.begin(); cit != c1.end(); ++cit) {
673  Op_ptr op = cit->get_op_ptr();
674  unit_vector_t args = cit->get_args();
675  unsigned n_args = args.size();
676  unsigned n_new_args = n_controls + n_args;
677  qubit_vector_t new_args(n_new_args);
678  for (unsigned i = 0; i < n_controls; i++) {
679  new_args[i] = Qubit(i);
680  }
681  for (unsigned i = 0; i < n_args; i++) {
682  new_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
683  }
684  OpType optype = op->get_type();
685  std::vector<Expr> params = op->get_params();
686  switch (optype) {
687  case OpType::noop:
688  break;
689  case OpType::X:
690  case OpType::CX:
691  case OpType::CCX:
692  case OpType::CnX:
693  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
694  break;
695  case OpType::Ry:
696  case OpType::CRy:
697  case OpType::CnRy:
698  c2.add_op<Qubit>(CNRYTYPE(n_new_args), params, new_args);
699  break;
700  case OpType::Z:
701  case OpType::CZ:
702  case OpType::CnZ:
703  c2.add_op<Qubit>(CNZTYPE(n_new_args), new_args);
704  break;
705  case OpType::Y:
706  case OpType::CY:
707  case OpType::CnY:
708  c2.add_op<Qubit>(CNYTYPE(n_new_args), new_args);
709  break;
710  default: {
711  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
712  Expr theta = tk1_angles[1];
713  Expr phi = tk1_angles[0] - 0.5;
714  Expr lambda = tk1_angles[2] + 0.5;
715  Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
716  // Operation is U3(theta, phi, lambda) + phase t.
717  // First absorb t in the overall phase.
718  a += t;
719  // Construct a multi-controlled U3, by extending the standard
720  // CU3-to-CX decomposition.
721  Qubit target = new_args[n_controls];
722  Circuit cnu1 = CnU1(n_controls - 1, 0.5 * (lambda + phi));
723  c2.append(cnu1);
724  c2.add_op<Qubit>(OpType::U1, 0.5 * (lambda - phi), {target});
725  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
726  c2.add_op<Qubit>(
727  OpType::U3, {-0.5 * theta, 0, -0.5 * (lambda + phi)}, {target});
728  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
729  c2.add_op<Qubit>(OpType::U3, {0.5 * theta, phi, 0}, {target});
730  } break;
731  }
732  }
733 
734  // 3. Account for phase by appending a CnU1 to the control qubits.
735  if (!equiv_0(a)) {
736  Circuit cnu1 = CnU1(n_controls - 1, a);
737  c2.append(cnu1);
738  }
739 
740  c2.remove_noops();
741  return c2;
742 }

789 // A gate block containing Cn* gates that can be merged as a single CnU gate
790 struct CnGateBlock {
791  enum class MergeMode { append, prepend };
792  CnGateBlock(const Command &command) {
793  // Assumes the color of the target is not identity
794  Op_ptr op = command.get_op_ptr();
795  ops.push_back(op);
796  unit_vector_t args = command.get_args();
797  TKET_ASSERT(!args.empty());
798  for (unsigned i = 0; i < args.size() - 1; i++) {
799  control_qubits.insert(args[i].index()[0]);
800  }
801  target_qubit = args.back().index()[0];
802  is_symmetric =
803  (op->get_type() == OpType::CZ || op->get_type() == OpType::CnZ ||
804  op->get_type() == OpType::CU1);
805  color = as_gate_ptr(op)->commuting_basis(args.size() - 1);
806  if (color == Pauli::I) {
807  throw std::invalid_argument(
808  "CnGateBlock doesn't accept multi-controlled identity gate.");
809  }
810  }
811 
812  // Check whether commute with another CnGateBlock
813  bool commutes_with(const CnGateBlock &other) {
814  if (target_qubit == other.target_qubit) {
815  return (color == other.color && color != std::nullopt);
816  }
817  if (control_qubits.contains(other.target_qubit) &&
818  other.color != Pauli::Z) {
819  return false;
820  }
821  if (other.control_qubits.contains(target_qubit) && color != Pauli::Z) {
822  return false;
823  }
824  return true;
825  }
826 
827  // Check whether can be merged with another CnGateBlock
828  bool is_mergeable_with(const CnGateBlock &other) {
829  // check if sizes match
830  if (control_qubits.size() != other.control_qubits.size()) {
831  return false;
832  }
833  // check if they act on the same set of qubits
834  std::set<unsigned> args_a = control_qubits;
835  args_a.insert(target_qubit);
836  std::set<unsigned> args_b = other.control_qubits;
837  args_b.insert(other.target_qubit);
838  if (args_a != args_b) {
839  return false;
840  }
841  // false if target don't match and none of them is symmetric
842  if (target_qubit != other.target_qubit && !is_symmetric &&
843  !other.is_symmetric) {
844  return false;
845  }
846  return true;
847  }
848 
849  // Merge with another CnGateBlock
850  void merge(CnGateBlock &other, const MergeMode &mode) {
851  if (mode == MergeMode::append) {
852  ops.insert(ops.end(), other.ops.begin(), other.ops.end());
853  } else {
854  ops.insert(ops.begin(), other.ops.begin(), other.ops.end());
855  }
856  color = (color != other.color) ? std::nullopt : color;
857  if (is_symmetric && !other.is_symmetric) {
858  control_qubits = other.control_qubits;
859  target_qubit = other.target_qubit;
860  is_symmetric = false;
861  }
862  // empty the other CnGateBlock
863  other.ops.clear();
864  }
865 
866  Eigen::Matrix2cd get_target_unitary() const {
867  Eigen::Matrix2cd m = Eigen::Matrix2cd::Identity();
868  for (const Op_ptr &op : ops) {
869  m = get_target_op_matrix(op) * m;
870  }
871  return m;
872  }
873 
874  // ops in the block
875  std::vector<Op_ptr> ops;
876  // target qubit index
877  unsigned target_qubit;
878  // control indices
879  std::set<unsigned> control_qubits;
880  // whether the target can act on any of its qubits
881  bool is_symmetric;
882  // color of the target qubit
883  std::optional<Pauli> color;
884 };

886 // Construct a controlled version of a given circuit
887 // with the assumption that the circuit does not have symbols.
888 static Circuit with_controls_numerical(const Circuit &c, unsigned n_controls) {
889  if (c.n_bits() != 0 || !c.is_simple()) {
890  throw CircuitInvalidity("Only default qubit register allowed");
891  }
892 
893  Circuit c1(c);
894  // Replace wire swaps with SWAP gates
895  c1.replace_all_implicit_wire_swaps();
896 
897  // Dispose of the trivial case
898  if (n_controls == 0) {
899  return c1;
900  }
901  // 1. Rebase to Cn* gates (n=0 for single qubit gates)
902  VertexList bin;
903  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
904  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
905  OpType optype = op->get_type();
906  if (is_gate_type(optype)) {
907  if (is_projective_type(optype)) {
908  throw CircuitInvalidity("Projective operations present");
909  }
910  if (is_box_type(optype)) {
911  throw CircuitInvalidity("Undecomposed boxes present");
912  }
913  if (is_single_qubit_type(optype) || is_controlled_gate_type(optype)) {
914  continue;
915  }
916  Circuit replacement = with_CX(as_gate_ptr(op));
917  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
918  bin.push_back(v);
919  } else if (optype != OpType::Input && optype != OpType::Output) {
920  throw CircuitInvalidity(
921  "Cannot construct the controlled version of " + op->get_name());
922  }
923  }
924  c1.remove_vertices(
925  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
926 
927  // 2. try to partition the circuit into blocks of Cn* gates such that
928  // the gates in each block can be merged into a single CnU gate
929  std::vector<Command> commands = c1.get_commands();
930  std::vector<CnGateBlock> blocks;
931 
932  Expr controlled_phase = c1.get_phase();
933 
934  for (const Command &cmd : commands) {
935  // if the gate is an identity up to a phase, add it as a controlled phase
936  std::optional<double> phase = cmd.get_op_ptr()->is_identity();
937  if (phase != std::nullopt) {
938  controlled_phase += phase.value();
939  } else {
940  blocks.push_back(CnGateBlock(cmd));
941  }
942  }
943 
944  // iterate the blocks from left to right
945  for (unsigned i = 0; i + 1 < blocks.size(); i++) {
946  CnGateBlock &b = blocks[i];
947  if (b.ops.empty()) {
948  continue;
949  }
950  // try to merge b to a block on the right
951  for (unsigned j = i + 1; j < blocks.size(); j++) {
952  CnGateBlock &candidate = blocks[j];
953  if (candidate.ops.empty()) {
954  continue;
955  }
956  if (b.is_mergeable_with(candidate)) {
957  candidate.merge(b, CnGateBlock::MergeMode::prepend);
958  break;
959  }
960  if (!b.commutes_with(candidate)) {
961  break;
962  }
963  }
964  }
965 
966  // iterate the blocks from right to left
967  for (unsigned i = blocks.size(); i-- > 1;) {
968  CnGateBlock &b = blocks[i];
969  if (b.ops.empty()) {
970  continue;
971  }
972  // try to merge b to a block on the left
973  // iterate from i-1 to 0
974  for (unsigned j = i; j-- > 0;) {
975  CnGateBlock &candidate = blocks[j];
976  if (candidate.ops.empty()) {
977  continue;
978  }
979  if (b.is_mergeable_with(candidate)) {
980  candidate.merge(b, CnGateBlock::MergeMode::append);
981  break;
982  }
983  if (!b.commutes_with(candidate)) {
984  break;
985  }
986  }
987  }
988  // 3. Add each block to c2 either as a CnX, CnZ, CnY or a CnU decomposition
989  Circuit c2(n_controls + c1.n_qubits());
990  const static Eigen::Matrix2cd X = Gate(OpType::X, {}, 1).get_unitary();
991  const static Eigen::Matrix2cd Y = Gate(OpType::Y, {}, 1).get_unitary();
992  const static Eigen::Matrix2cd Z = Gate(OpType::Z, {}, 1).get_unitary();
993 
994  for (const CnGateBlock &b : blocks) {
995  if (b.ops.empty()) {
996  continue;
997  }
998  // Computes the target unitary
999  Eigen::Matrix2cd m = b.get_target_unitary();
1000  if (m.isApprox(Eigen::Matrix2cd::Identity(), EPS)) {
1001  continue;
1002  }
1003  std::function<qubit_vector_t()> get_args = [&]() {
1004  qubit_vector_t new_args;
1005  for (unsigned i = 0; i < n_controls; i++) {
1006  new_args.push_back(Qubit(i));
1007  }
1008  for (const unsigned i : b.control_qubits) {
1009  new_args.push_back(Qubit(i + n_controls));
1010  }
1011  new_args.push_back(Qubit(b.target_qubit + n_controls));
1012  return new_args;
1013  };
1014  if (m.isApprox(X, EPS)) {
1015  qubit_vector_t new_args = get_args();
1016  c2.add_op<Qubit>(CNXTYPE(new_args.size()), new_args);
1017  continue;
1018  }
1019  if (m.isApprox(Y, EPS)) {
1020  qubit_vector_t new_args = get_args();
1021  c2.add_op<Qubit>(CNYTYPE(new_args.size()), new_args);
1022  continue;
1023  }
1024  if (m.isApprox(Z, EPS)) {
1025  qubit_vector_t new_args = get_args();
1026  c2.add_op<Qubit>(CNZTYPE(new_args.size()), new_args);
1027  continue;
1028  }
1029  unit_map_t unit_map;
1030  for (unsigned i = 0; i < n_controls; i++) {
1031  unit_map.insert({Qubit(i), Qubit(i)});
1032  }
1033  unsigned control_index = n_controls;
1034  for (const unsigned i : b.control_qubits) {
1035  unit_map.insert({Qubit(control_index++), Qubit(i + n_controls)});
1036  }
1037  unit_map.insert({Qubit(control_index), Qubit(b.target_qubit + n_controls)});
1038 
1039  unsigned total_controls = b.control_qubits.size() + n_controls;
1040 
1041  Circuit replacement;
1042 
1043  // Check if the matrix is SU(2)
1044  if (m.determinant() == 1.) {
1045  // We have three functions that can decompose a multi-controlled SU(2).
1046  // The choice is based on the average number of CXs they produce for a
1047  // random n-controlled SU(2) gate.
1048  if (total_controls > 2 && total_controls < 5) {
1049  replacement = CircPool::CnU_gray_code_decomp(total_controls, m);
1050  } else if (total_controls >= 5 && total_controls < 9) {
1051  replacement = CircPool::CnU_linear_depth_decomp(total_controls, m);
1052  } else {
1053  // Compute the SU(2) angles from the TK1 angles
1054  std::vector<double> angles = tk1_angles_from_unitary(m);
1055  if (equiv_val(angles[3], 1., 2)) {
1056  // if the phase is odd, it can be absorbed into the first Rz rotation
1057  angles[0] = angles[0] + 2;
1058  } else {
1059  // because it's SU(2), the phase must be integers
1060  TKET_ASSERT(equiv_0(angles[3], 2));
1061  }
1062  // convert tk1 angles to zyz angles
1063  std::vector<double> zyz_angles = {
1064  angles[0] - 0.5, angles[1], angles[2] + 0.5};
1065  replacement = CircPool::CnSU2_linear_decomp(
1066  total_controls, zyz_angles[0], zyz_angles[1], zyz_angles[2]);
1067  }
1068  } else {
1069  // The gray code method produces less CXs when total_controls is 3 or 4
1070  if (total_controls == 3 || total_controls == 4) {
1071  replacement = CircPool::CnU_gray_code_decomp(total_controls, m);
1072  } else {
1073  replacement = CircPool::CnU_linear_depth_decomp(total_controls, m);
1074  }
1075  }
1076  c2.append_with_map(replacement, unit_map);
1077  }
1078 
1079  // 4. implement the controlled phase as a CnU1 gate
1080  if (!equiv_0(controlled_phase)) {
1081  Circuit cnu1_circ = CnU1(n_controls - 1, controlled_phase);
1082  c2.append(cnu1_circ);
1083  }
1084  return c2;
1085 }

1100 std::tuple<Circuit, std::array<Expr, 3>, Circuit> normalise_TK2_angles(
1101  Expr a, Expr b, Expr c) {
1102  std::optional<double> a_eval = eval_expr_mod(a, 4);
1103  std::optional<double> b_eval = eval_expr_mod(b, 4);
1104  std::optional<double> c_eval = eval_expr_mod(c, 4);
1105 
1106  Circuit pre(2), post(2);
1107 
1108  // Add ot.dagger() at beggining and ot at end.
1109  auto conj = [&pre, &post](OpType ot) {
1110  Op_ptr op = get_op_ptr(ot);
1111  Op_ptr opdg = op->dagger();

1162  };

1164  // Swap XX and YY.
1165  conj(OpType::S);
1166  std::swap(a, b);
1167  std::swap(a_eval, b_eval);
1168  }

1171  conj(OpType::V);

1173  std::swap(b_eval, c_eval);
1174  }

1187  post.add_op<unsigned>(OpType::X, {1});
1188  pre.add_phase(0.5);

1194  post.add_op<unsigned>(OpType::Y, {1});

1196  }

1200  post.add_op<unsigned>(OpType::Z, {0});
1201  post.add_op<unsigned>(OpType::Z, {1});
1202  pre.add_phase(0.5);
1203  }

1205  a = 1. - a;
1206  *a_eval = 1. - *a_eval;
1207  b = 1. - b;
1208  *b_eval = 1. - *b_eval;

1210  post.add_op<unsigned>(OpType::Z, {1});
1211  }

1224  post.add_op<unsigned>(OpType::Z, {1});
1225  pre.add_phase(-0.5);
1226  }

1228  post = post.dagger();
1229
