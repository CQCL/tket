37Eigen::Matrix2cd get_matrix(const Circuit &circ, const Vertex &vert) {
38 const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
39 if (op->get_type() != OpType::TK1) {
40 throw BadOpType("Cannot compute matrix from gate", op->get_type());
41 }
42 std::vector<Expr> ps = op->get_params();
43 ps.push_back(0);
44 return get_matrix_from_tk1_angles(ps);
45}

47Eigen::Matrix2cd get_matrix_from_circ(const Circuit &circ) {
48 if (circ.n_qubits() != 1)
49 throw CircuitInvalidity(
50 "Getting Matrix: expected 1 qubit circuit, found " +
51 std::to_string(circ.n_qubits()));
52 Complex factor = std::exp(i_ * PI * eval_expr(circ.get_phase()).value());
53 VertexVec qpath = circ.qubit_path_vertices(circ.all_qubits()[0]);
54 unsigned N = qpath.size();
55 if (N == 2) return factor * Eigen::Matrix2cd::Identity();
56 Eigen::Matrix2cd m = get_matrix(circ, qpath[N - 2]);
57 for (unsigned x = N - 3; x >= 1; --x) {
58 m = unitary_product2(m, get_matrix(circ, qpath[x]));
59 }
60 return factor * m;
61}

63Eigen::Matrix4cd get_matrix_from_2qb_circ(const Circuit &circ) {
64 std::vector<QPathDetailed> all_paths = circ.all_qubit_paths();
65 std::map<Vertex, Eigen::Matrix4cd> v_to_op;
66 Eigen::Matrix4cd cnot, tonc, swap;
67 // clang-format off
68 cnot << 1, 0, 0, 0,
69 0, 1, 0, 0,
70 0, 0, 0, 1,
71 0, 0, 1, 0;
72 tonc << 1, 0, 0, 0,
73 0, 0, 0, 1,
74 0, 0, 1, 0,
75 0, 1, 0, 0;
76 swap << 1, 0, 0, 0,
77 0, 0, 1, 0,
78 0, 1, 0, 0,
79 0, 0, 0, 1;
80 // clang-format on
81
82 for (unsigned uqb = 0; uqb < 2; uqb++) {
83 for (QPathDetailed::iterator it = all_paths[uqb].begin();
84 it != all_paths[uqb].end(); ++it) {
85 const Op_ptr o = circ.get_Op_ptr_from_Vertex(it->first);
86 switch (o->get_type()) {
87 case OpType::Input:
88 case OpType::Create:
89 case OpType::Output:
90 case OpType::Discard: {
91 v_to_op[it->first] = Eigen::Matrix4cd::Identity();
92 break;
93 }
94 case OpType::SWAP: {
95 if (uqb == 0) v_to_op[it->first] = swap;
96 break;
97 }
98 case OpType::CX: {
99 if (uqb == 0) {
100 if (it->second == 0) {
101 v_to_op[it->first] = cnot;
102 } else {
103 v_to_op[it->first] = tonc;
104 }
105 }
106 break;
107 }
108 case OpType::TK2: {
109 auto params = o->get_params();
110 TKET_ASSERT(params.size() == 3);
111 v_to_op[it->first] = get_matrix_from_2qb_circ(
112 CircPool::normalised_TK2_using_CX(
113 params[0], params[1], params[2]));
114 break;
115 }
116 default: {
117 if (o->get_desc().is_gate() && circ.n_in_edges(it->first) == 1 &&
118 circ.n_out_edges(it->first) == 1) {
119 const Op_ptr g = o;
120 std::vector<Expr> ps = as_gate_ptr(g)->get_tk1_angles();
121 Eigen::Matrix2cd mat = get_matrix_from_tk1_angles(ps);
122 if (uqb == 0) {
123 v_to_op[it->first] =
124 Eigen::kroneckerProduct(mat, Eigen::Matrix2cd::Identity());
125 } else {
126 v_to_op[it->first] =
127 Eigen::kroneckerProduct(Eigen::Matrix2cd::Identity(), mat);
128 }
129 } else
130 throw BadOpType("Cannot obtain matrix from op", o->get_type());
131 }
132 }
133 }
134 }
135 Eigen::Matrix4cd m = Eigen::Matrix4cd::Identity();
136 SliceVec slices = circ.get_slices();
137 for (const Slice &s : slices) {
138 for (const Vertex &v : s) {
139 m = unitary_product2(v_to_op[v], m);
140 }
141 }
142 return std::exp(i_ * PI * eval_expr(circ.get_phase()).value()) * m;
143}

145Circuit two_qubit_canonical(const Eigen::Matrix4cd &U, OpType target_2qb_gate) {
146 if (!is_unitary(U)) {
147 throw std::invalid_argument(
148 "Non-unitary matrix passed to two_qubit_canonical");
149 }
150
151 auto [K1, A, K2] = get_information_content(U);
152
153 K1 /= pow(K1.determinant(), 0.25);
154 K2 /= pow(K2.determinant(), 0.25);
155 auto [a, b, c] = A;
156
157 // Decompose single qubits
158 auto [K1a, K1b] = kronecker_decomposition(K1);
159 auto [K2a, K2b] = kronecker_decomposition(K2);
160
161 Circuit result(2);
162
163 std::vector<double> angles_q0 = tk1_angles_from_unitary(K2a);
164 std::vector<double> angles_q1 = tk1_angles_from_unitary(K2b);
165 result.add_op<unsigned>(
166 OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
167 result.add_op<unsigned>(
168 OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
169
170 switch (target_2qb_gate) {
171 case OpType::TK2:
172 result.append(CircPool::TK2_using_normalised_TK2(a, b, c));
173 break;
174 case OpType::CX:
175 result.append(CircPool::TK2_using_CX(a, b, c));
176 break;
177 default:
178 throw std::invalid_argument("target_2qb_gate must be CX or TK2.");
179 }
180
181 angles_q0 = tk1_angles_from_unitary(K1a);
182 angles_q1 = tk1_angles_from_unitary(K1b);
183 result.add_op<unsigned>(
184 OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
185 result.add_op<unsigned>(
186 OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
187
188 // this fixes phase if decomposition is exact
189 Eigen::Matrix4cd V = get_matrix_from_2qb_circ(result).adjoint();
190 Eigen::Matrix4cd reminder = unitary_product2(V, U);
191 const Complex phase = reminder(0, 0); // reminder = phase * I
192 result.add_phase(arg(phase) / PI);
193 return result;
194}

196// Factorize U as VD where V corresponds to a 2-CX circuit and
197// D = diag(z, z*, z*, z). Return V and z.
198static std::pair<Eigen::Matrix4cd, Complex> decompose_VD(
199 const Eigen::Matrix4cd &U) {
200 if (!is_unitary(U)) {
201 throw std::invalid_argument("Non-unitary matrix passed to decompose_VD");
202 }
203
204 // The calculations below are derived from the proof of Proposition V.2 in
205 // https://arxiv.org/abs/quant-ph/0308033.
206
207 Eigen::Matrix4cd u = U / pow(U.determinant(), 0.25);
208 Complex a = u(3, 0) * u(0, 3) - u(2, 0) * u(1, 3) - u(1, 0) * u(2, 3) +
209 u(0, 0) * u(3, 3);
210 Complex b = u(3, 1) * u(0, 2) - u(2, 1) * u(1, 2) - u(1, 1) * u(2, 2) +
211 u(0, 1) * u(3, 2);
212 // Now we want to find z such that |z|=1 and (az* - bz) is real.
213 // The numerical stability of this function is a concern when a is close to
214 // -b*. This problem can be demonstrated in artificially constructed
215 // examples (passing unitaries very close to, but not quite, the identity to
216 // the functions below). In these cases the product VD (or DV) may not
217 // approximate U to within Eigen's default tolerance. Is there a way to
218 // dodge this issue?
219 Complex w = a + std::conj(b);
220 double d = std::abs(w);
221 // If w = 0 then we can set z = 1.
222 Complex z = (d < EPS) ? 1. : w / d;
223 Complex z0 = sqrt(z);
224 Complex z1 = std::conj(z0);
225 Eigen::Matrix4cd V = U;
226 V.col(0) *= z1;
227 V.col(1) *= z0;
228 V.col(2) *= z0;
229 V.col(3) *= z1;
230 return {V, z0};
231}

233static void replace_TK2_2CX(Circuit &circ) {
234 VertexList bin;
235 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
236 if (circ.get_OpType_from_Vertex(v) != OpType::TK2) continue;
237 auto params = circ.get_Op_ptr_from_Vertex(v)->get_params();
238 TKET_ASSERT(params.size() == 3);
239 // Rounding errors can accumulate here; warn if so:
240 if (!equiv_0(params[2], 4, 1e-6)) {
241 std::stringstream ss;
242 ss << "Rounding errors in CX decomposition: ZZPhase parameter = "
243 << params[2] << " when it should be 0 (mod 4). Ignoring.";
244 tket_log()->warn(ss.str());
245 }
246 Circuit sub = CircPool::approx_TK2_using_2xCX(params[0], params[1]);
247 bin.push_back(v);
248 circ.substitute(sub, v, Circuit::VertexDeletion::No);
249 }
250 TKET_ASSERT(bin.size() == 1);
251 circ.remove_vertices(
252 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
253}

255std::pair<Circuit, Complex> decompose_2cx_VD(const Eigen::Matrix4cd &U) {
256 auto [V, z0] = decompose_VD(U);
257 Circuit circ = two_qubit_canonical(V);
258 replace_TK2_2CX(circ);
259 return {circ, z0};
260}

262std::pair<Circuit, Complex> decompose_2cx_DV(const Eigen::Matrix4cd &U) {
263 auto [V, z0] = decompose_VD(U.adjoint());
264 V.adjointInPlace();
265 Circuit circ = two_qubit_canonical(V);
266 replace_TK2_2CX(circ);
267 return {circ, std::conj(z0)};
268}

275Circuit pauli_gadget(SpSymPauliTensor paulis, CXConfigType cx_config) {
276 Circuit circ;
277 for (const std::pair<const Qubit, Pauli> &qp : paulis.string)
278 circ.add_qubit(qp.first);
279 if (SpPauliString(paulis.string) == SpPauliString()) {
280 circ.add_phase(-paulis.coeff / 2);
281 return circ;
282 }
283 Circuit compute(circ);
284 Circuit action(circ);
285 qubit_vector_t qubits;
286 for (const std::pair<const Qubit, Pauli> &qp : paulis.string) {
287 switch (qp.second) {
288 case Pauli::I:
289 break;
290 case Pauli::X:
291 compute.add_op<Qubit>(OpType::H, {qp.first});
292 qubits.push_back(qp.first);
293 break;
294 case Pauli::Y:
295 compute.add_op<Qubit>(OpType::V, {qp.first});
296 qubits.push_back(qp.first);
297 break;
298 case Pauli::Z:
299 qubits.push_back(qp.first);
300 break;
301 }
302 }
303 unsigned n_qubits = qubits.size();
304 switch (cx_config) {
305 case CXConfigType::Snake: {
306 for (unsigned i = n_qubits - 1; i != 0; --i) {
307 unsigned j = i - 1;
308 compute.add_op<Qubit>(OpType::CX, {qubits[i], qubits[j]});
309 }
310 action.add_op<Qubit>(OpType::Rz, paulis.coeff, {qubits[0]});
311 break;
312 }
313 case CXConfigType::Star: {
314 for (unsigned i = n_qubits - 1; i != 0; --i) {
315 compute.add_op<Qubit>(OpType::CX, {qubits[i], qubits[0]});
316 }
317 action.add_op<Qubit>(OpType::Rz, paulis.coeff, {qubits[0]});
318 break;
319 }
320 case CXConfigType::Tree: {
321 unsigned complete_layers = floor(log2(n_qubits));
322 unsigned dense_end = pow(2, complete_layers);
323 for (unsigned i = 0; i < n_qubits - dense_end; i++)
324 compute.add_op<Qubit>(
325 OpType::CX, {qubits[dense_end + i], qubits[dense_end - 1 - i]});
326 for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
327 for (unsigned i = 0; i < dense_end; i += 2 * step_size)
328 compute.add_op<Qubit>(OpType::CX, {qubits[i + step_size], qubits[i]});
329 }
330 action.add_op<Qubit>(OpType::Rz, paulis.coeff, {qubits[0]});
331 break;
332 }
333 case CXConfigType::MultiQGate: {
334 int sign_correction = 1;
335 for (int q = n_qubits - 1; q > 0; q -= 2) {
336 if (q - 1 > 0) {
337 unsigned i = q, j = q - 1;
338 // this is only equal to the CX decompositions above
339 // up to phase, but phase differences are cancelled out by
340 // its dagger XXPhase(-1/2) below.
341 compute.add_op<Qubit>(OpType::H, {qubits[i]});
342 compute.add_op<Qubit>(OpType::H, {qubits[j]});
343 compute.add_op<Qubit>(
344 OpType::XXPhase3, 0.5, {qubits[i], qubits[j], qubits[0]});
345 sign_correction *= -1;
346 } else {
347 unsigned i = q;
348 compute.add_op<Qubit>(OpType::CX, {qubits[i], qubits[0]});
349 }
350 }
351 action.add_op<Qubit>(
352 OpType::Rz, sign_correction * paulis.coeff, {qubits[0]});
353 break;
354 }
355 }
356 // ConjugationBox components must be in the default register
357 compute.flatten_registers();
358 action.flatten_registers();
359 ConjugationBox box(
360 std::make_shared<CircBox>(compute), std::make_shared<CircBox>(action));
361 circ.add_box(box, circ.all_qubits());
362 return circ;
363}

365Circuit pauli_gadget_pair(
366 SpSymPauliTensor paulis0, SpSymPauliTensor paulis1,
367 CXConfigType cx_config) {
368 Circuit circ;
369 for (const std::pair<const Qubit, Pauli> &qp : paulis0.string)
370 circ.add_qubit(qp.first);
371 for (const std::pair<const Qubit, Pauli> &qp : paulis1.string)
372 circ.add_qubit(qp.first, false);
373 if (SpPauliString(paulis0.string) == SpPauliString{}) {
374 circ.append(pauli_gadget(paulis1, cx_config));
375 circ.add_phase(-paulis0.coeff / 2);
376 return circ;
377 } else if (SpPauliString(paulis1.string) == SpPauliString{}) {
378 circ.append(pauli_gadget(paulis0, cx_config));
379 circ.add_phase(-paulis1.coeff / 2);
380 return circ;
381 }
382 paulis0.compress();
383 paulis1.compress();
384 Circuit u(circ);
385 Circuit v(circ);
386
387 // Reduce the overlap down to at most 1 qubit, which may be matching or
388 // mismatching; allow both gadgets to build on that qubit
389 SpPauliStabiliser p0stab(paulis0.string);
390 SpPauliStabiliser p1stab(paulis1.string);
391 u.append(reduce_overlap_of_paulis(p0stab, p1stab, cx_config, true).first);
392 paulis0 = SpSymPauliTensor(p0stab) * SpSymPauliTensor({}, paulis0.coeff);
393 paulis1 = SpSymPauliTensor(p1stab) * SpSymPauliTensor({}, paulis1.coeff);
394
395 /*
396 * Combine circuits to give final result
397 */
398 v.append(pauli_gadget(paulis0));
399 v.append(pauli_gadget(paulis1));
400 // ConjugationBox components must be in the default register
401 qubit_vector_t all_qubits = u.all_qubits();
402 u.flatten_registers();
403 v.flatten_registers();
404 ConjugationBox cjbox(
405 std::make_shared<CircBox>(u), std::make_shared<CircBox>(v));
406 circ.add_box(cjbox, all_qubits);
407 return circ;
408}

415Circuit with_TK2(Gate_ptr op) {
416 std::vector<Expr> params = op->get_params();
417 unsigned n = op->n_qubits();
418 if (n == 0) {
419 Circuit c(0);
420 if (op->get_type() == OpType::Phase) {
421 c.add_phase(op->get_params()[0]);
422 }
423 return c;
424 } else if (n == 1) {
425 Circuit c(1);
426 c.add_op(op, std::vector<unsigned>{0});
427 return c;
428 } else if (n == 2 && op->free_symbols().empty()) {
429 Eigen::Matrix4cd U = op->get_unitary();
430 auto [K1, A, K2] = get_information_content(U);
431 // Decompose single qubits
432 auto [K1a, K1b] = kronecker_decomposition(K1);
433 auto [K2a, K2b] = kronecker_decomposition(K2);
434 Circuit c(2);
435 std::vector<double> angles_K1a = tk1_angles_from_unitary(K1a);
436 std::vector<double> angles_K1b = tk1_angles_from_unitary(K1b);
437 std::vector<double> angles_K2a = tk1_angles_from_unitary(K2a);
438 std::vector<double> angles_K2b = tk1_angles_from_unitary(K2b);
439 c.add_op<unsigned>(
440 OpType::TK1, {angles_K2a.begin(), angles_K2a.end() - 1}, {0});
441 c.add_op<unsigned>(
442 OpType::TK1, {angles_K2b.begin(), angles_K2b.end() - 1}, {1});
443 double alpha = std::get<0>(A);
444 double beta = std::get<1>(A);
445 double gamma = std::get<2>(A);
446
447 c.append(CircPool::TK2_using_normalised_TK2(alpha, beta, gamma));
448
449 c.add_op<unsigned>(
450 OpType::TK1, {angles_K1a.begin(), angles_K1a.end() - 1}, {0});
451 c.add_op<unsigned>(
452 OpType::TK1, {angles_K1b.begin(), angles_K1b.end() - 1}, {1});
453
454 // Correct phase by computing the unitary and comparing with U:
455 Eigen::Matrix4cd V_K1 = Eigen::KroneckerProduct(
456 get_matrix_from_tk1_angles(
457 {angles_K1a[0], angles_K1a[1], angles_K1a[2], 0}),
458 get_matrix_from_tk1_angles(
459 {angles_K1b[0], angles_K1b[1], angles_K1b[2], 0}));
460 Eigen::Matrix4cd V_A =
461 internal::GateUnitaryMatrixImplementations::TK2(alpha, beta, gamma);
462 Eigen::Matrix4cd V_K2 = Eigen::KroneckerProduct(
463 get_matrix_from_tk1_angles(
464 {angles_K2a[0], angles_K2a[1], angles_K2a[2], 0}),
465 get_matrix_from_tk1_angles(
466 {angles_K2b[0], angles_K2b[1], angles_K2b[2], 0}));
467 Eigen::Matrix4cd V_adj = unitary_product3(V_K1, V_A, V_K2).adjoint();
468 Eigen::Matrix4cd R = unitary_product2(V_adj, U);
469 const Complex phase = R(0, 0); // R = phase * I
470 c.add_phase(arg(phase) / PI);
471
472 return c;
473 }
474 // Now the non-trivial cases.
475 switch (op->get_type()) {
476 case OpType::ISWAP:
477 return CircPool::ISWAP_using_TK2(params[0]);
478 case OpType::ISWAPMax:
479 return CircPool::ISWAPMax_using_TK2();
480 case OpType::PhasedISWAP:
481 return CircPool::PhasedISWAP_using_TK2(params[0], params[1]);
482 case OpType::XXPhase:
483 return CircPool::XXPhase_using_TK2(params[0]);
484 case OpType::YYPhase:
485 return CircPool::YYPhase_using_TK2(params[0]);
486 case OpType::ZZPhase:
487 return CircPool::ZZPhase_using_TK2(params[0]);
488 case OpType::NPhasedX:
489 return CircPool::NPhasedX_using_PhasedX(n, params[0], params[1]);
490 case OpType::ESWAP:
491 return CircPool::ESWAP_using_TK2(params[0]);
492 case OpType::FSim:
493 return CircPool::FSim_using_TK2(params[0], params[1]);
494 case OpType::CRx:
495 return CircPool::CRx_using_TK2(params[0]);
496 case OpType::CRy:
497 return CircPool::CRy_using_TK2(params[0]);
498 case OpType::CRz:
499 return CircPool::CRz_using_TK2(params[0]);
500 case OpType::CU1:
501 return CircPool::CU1_using_TK2(params[0]);
502 case OpType::XXPhase3:
503 return CircPool::XXPhase3_using_TK2(params[0]);
504 case OpType::AAMS:
505 return CircPool::AAMS_using_TK2(params[0], params[1], params[2]);
506 case OpType::CCX:
507 case OpType::CSWAP:
508 case OpType::BRIDGE:
509 case OpType::CU3:
510 case OpType::PhaseGadget: {
511 // As a first, inefficient, solution, decompose these into CX and then
512 // replace each CX with a TK2 (and some single-qubit gates).
513 // TODO Find more efficient decompositions for these gates.
514 Circuit c = with_CX(op);
515 replace_CX_with_TK2(c);
516 return c;
517 }
518 default:
519 throw CircuitInvalidity("Cannot decompose " + op->get_name());
520 }
521}

523Circuit with_CX(Gate_ptr op) {
524 OpType optype = op->get_type();
525 std::vector<Expr> params = op->get_params();
526 unsigned n = op->n_qubits();
527 if (n == 0) {
528 Circuit c(0);
529 if (op->get_type() == OpType::Phase) {
530 c.add_phase(op->get_params()[0]);
531 }
532 return c;
533 } else if (n == 1) {
534 Circuit c(1);
535 c.add_op(op, std::vector<unsigned>{0});
536 return c;
537 }
538 switch (optype) {
539 case OpType::CX: {
540 Circuit c(2);
541 c.add_op(op, std::vector<unsigned>{0, 1});
542 return c;
543 }
544 case OpType::CCX:
545 return CircPool::CCX_normal_decomp();
546 case OpType::CY:
547 return CircPool::CY_using_CX();
548 case OpType::CZ:
549 return CircPool::CZ_using_CX();
550 case OpType::CH:
551 return CircPool::CH_using_CX();
552 case OpType::CV:
553 return CircPool::CV_using_CX();
554 case OpType::CVdg:
555 return CircPool::CVdg_using_CX();
556 case OpType::CSX:
557 return CircPool::CSX_using_CX();
558 case OpType::CSXdg:
559 return CircPool::CSXdg_using_CX();
560 case OpType::CS:
561 return CircPool::CS_using_CX();
562 case OpType::CSdg:
563 return CircPool::CSdg_using_CX();
564 case OpType::CRz:
565 return CircPool::CRz_using_CX(params[0]);
566 case OpType::CRx:
567 return CircPool::CRx_using_CX(params[0]);
568 case OpType::CRy:
569 return CircPool::CRy_using_CX(params[0]);
570 case OpType::CU1:
571 return CircPool::CU1_using_CX(params[0]);
572 case OpType::CU3:

578 }
579 case OpType::SWAP:

581 case OpType::CSWAP:
582 return CircPool::CSWAP_using_CX();
583 case OpType::BRIDGE:
584 return CircPool::BRIDGE_using_CX_0();
585 case OpType::ECR:
586 return CircPool::ECR_using_CX();
587 case OpType::ISWAP:
588 return CircPool::ISWAP_using_CX(params[0]);
589 case OpType::ZZMax:
590 return CircPool::ZZMax_using_CX();
591 case OpType::XXPhase:
592 return CircPool::XXPhase_using_CX(params[0]);
593 case OpType::YYPhase:
594 return CircPool::YYPhase_using_CX(params[0]);
595 case OpType::ZZPhase:
596 return CircPool::ZZPhase_using_CX(params[0]);
597 case OpType::TK2:
598 return CircPool::TK2_using_CX(params[0], params[1], params[2]);
599 case OpType::XXPhase3:
600 return CircPool::XXPhase3_using_CX(params[0]);
601 case OpType::ESWAP:
602 return CircPool::ESWAP_using_CX(params[0]);
603 case OpType::FSim:
604 return CircPool::FSim_using_CX(params[0], params[1]);
605 case OpType::Sycamore:
606 return CircPool::FSim_using_CX(1. / 2., 1. / 6.);
607 case OpType::ISWAPMax:
608 return CircPool::ISWAPMax_using_CX();
609 case OpType::PhasedISWAP:
610 return CircPool::PhasedISWAP_using_CX(params[0], params[1]);
611 case OpType::NPhasedX:

621 (((n) == 2) ? OpType::CX : ((n) == 3) ? OpType::CCX : OpType::CnX)

630static Circuit CnU1(unsigned n_controls, Expr lambda) {
631 Gate_ptr u1_gate = as_gate_ptr(get_op_ptr(OpType::U1, lambda));
632 // Use the gray code method if lambda contains symbols
633 // The gray code decomp also produces less CXs when n_controls is 3 or 4
634 if (eval_expr(lambda) == std::nullopt || n_controls == 3 || n_controls == 4) {
635 return CircPool::CnU_gray_code_decomp(n_controls, u1_gate);
636 } else {
637 return CircPool::CnU_linear_depth_decomp(
638 n_controls, u1_gate->get_unitary());
639 }

651static Circuit controlled_conjugation_box(
652 const Op_ptr &op, unsigned n_controls, const unit_vector_t &args) {
653 const ConjugationBox &conj_box = static_cast<const ConjugationBox &>(*op);
654 unsigned n_targets = args.size();
655 Op_ptr compute = conj_box.get_compute();
656 Op_ptr action = conj_box.get_action();
657 std::optional<Op_ptr> uncompute_opt = conj_box.get_uncompute();
658 Op_ptr uncompute = uncompute_opt ? uncompute_opt.value() : compute->dagger();
659 qubit_vector_t all_args(n_controls + n_targets);
660 qubit_vector_t target_args(n_targets);
661 for (unsigned i = 0; i < n_controls; i++) {
662 all_args[i] = Qubit(i);
663 }
664 for (unsigned i = 0; i < n_targets; i++) {
665 TKET_ASSERT(
666 args[i].reg_name() == q_default_reg() && args[i].reg_dim() == 1);
667 all_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
668 target_args[i] = Qubit(n_controls + args[i].index()[0]);
669 }
670 Circuit circ;
671 for (const Qubit &q : all_args) {
672 circ.add_qubit(q);
673 }
674 circ.add_op(compute, target_args);
675 QControlBox controlled_action(action, n_controls);
676 circ.add_box(controlled_action, all_args);
677 circ.add_op(uncompute, target_args);
678 circ.decompose_boxes_recursively();
679 return circ;
680}

682static Circuit with_controls_symbolic(const Circuit &c, unsigned n_controls) {
683 if (c.n_bits() != 0 || !c.is_simple()) {
684 throw CircuitInvalidity("Only default qubit register allowed");
685 }
686
687 Circuit c1(c);
688 // Replace wire swaps with SWAP gates
689 c1.replace_all_implicit_wire_swaps();
690
691 // Dispose of the trivial case
692 if (n_controls == 0) {
693 return c1;
694 }
695
696 static const OpTypeSet multiq_gate_set = {
697 OpType::CX, OpType::CCX, OpType::CnX, OpType::CRy,
698 OpType::CnRy, OpType::CZ, OpType::CnZ, OpType::CY,
699 OpType::CnY, OpType::CnRx, OpType::CnRz};
700
701 unsigned c_n_qubits = c1.n_qubits();
702
703 // 1. Rebase to {CX, CCX, CnX, CnRy, CnRx, CnRz} and single-qubit gates
704 VertexList bin;
705 BGL_FORALL_VERTICES(v, c1.dag, DAG) {
706 Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
707 OpType optype = op->get_type();
708 if (is_gate_type(optype)) {
709 if (is_projective_type(optype)) {
710 throw CircuitInvalidity("Projective operations present");
711 }
712 if (is_single_qubit_type(optype)) {
713 continue;
714 }
715 if (multiq_gate_set.find(optype) != multiq_gate_set.end()) {
716 continue;
717 }
718 Circuit replacement;
719 if (optype == OpType::PhaseGadget) {
720 replacement = phase_gadget(
721 op->n_qubits(), op->get_params()[0], CXConfigType::Snake);
722 if (replacement.n_gates() > 0) {
723 TKET_ASSERT(
724 replacement.n_gates() == 1 &&
725 replacement.count_gates(OpType::ConjugationBox) == 1);
726 }
727 } else {
728 replacement = with_CX(as_gate_ptr(op));
729 }
730 c1.substitute(replacement, v, Circuit::VertexDeletion::No);
731 bin.push_back(v);
732 } else if (is_box_type(optype) && optype != OpType::ConjugationBox) {
733 throw CircuitInvalidity("Undecomposed boxes present");
734 }
735 }
736 c1.remove_vertices(
737 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
738
739 // Capture the phase. We may adjust this during replacements below.
740 Expr a = c1.get_phase();
741
742 // 2. Replace all gates with controlled versions
743 Circuit c2(n_controls + c_n_qubits);
744 for (Circuit::CommandIterator cit = c1.begin(); cit != c1.end(); ++cit) {
745 Op_ptr op = cit->get_op_ptr();
746 OpType optype = op->get_type();
747 unit_vector_t args = cit->get_args();
748 unsigned n_args = args.size();
749 if (optype == OpType::Barrier) {
750 qubit_vector_t barrier_args(n_args);
751 for (unsigned i = 0; i < n_args; i++) {
752 barrier_args[i] = Qubit(n_controls + args[i].index()[0]);
753 }
754 c2.add_op(op, barrier_args);
755 continue;
756 }
757 if (optype == OpType::ConjugationBox) {
758 c2.append(controlled_conjugation_box(op, n_controls, args));
759 continue;
760 }
761 unsigned n_new_args = n_controls + n_args;
762 qubit_vector_t new_args(n_new_args);
763 for (unsigned i = 0; i < n_controls; i++) {
764 new_args[i] = Qubit(i);
765 }
766 for (unsigned i = 0; i < n_args; i++) {
767 new_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
768 }
769 std::vector<Expr> params = op->get_params();
770 switch (optype) {
771 case OpType::noop:
772 break;
773 case OpType::X:
774 case OpType::CX:
775 case OpType::CCX:
776 case OpType::CnX:
777 c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
778 break;
779 case OpType::Ry:
780 case OpType::CRy:
781 case OpType::CnRy:
782 c2.add_op<Qubit>(CNRYTYPE(n_new_args), params, new_args);
783 break;
784 case OpType::Rx:
785 case OpType::CRx:
786 case OpType::CnRx:
787 c2.add_op<Qubit>(CNRXTYPE(n_new_args), params, new_args);
788 break;
789 case OpType::Rz:
790 case OpType::CRz:
791 case OpType::CnRz:
792 c2.add_op<Qubit>(CNRZTYPE(n_new_args), params, new_args);
793 break;
794 case OpType::Z:
795 case OpType::CZ:
796 case OpType::CnZ:
797 c2.add_op<Qubit>(CNZTYPE(n_new_args), new_args);
798 break;
799 case OpType::Y:
800 case OpType::CY:
801 case OpType::CnY:
802 c2.add_op<Qubit>(CNYTYPE(n_new_args), new_args);
803 break;
804 default: {
805 std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
806 Expr theta = tk1_angles[1];
807 Expr phi = tk1_angles[0] - 0.5;
808 Expr lambda = tk1_angles[2] + 0.5;
809 Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
810 // Operation is U3(theta, phi, lambda) + phase t.
811 // First absorb t in the overall phase.
812 a += t;
813 // Construct a multi-controlled U3, by extending the standard
814 // CU3-to-CX decomposition.
815 Qubit target = new_args[n_controls];
816 Circuit cnu1 = CnU1(n_controls - 1, 0.5 * (lambda + phi));
817 c2.append(cnu1);
818 c2.add_op<Qubit>(OpType::U1, 0.5 * (lambda - phi), {target});
819 c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
820 c2.add_op<Qubit>(
821 OpType::U3, {-0.5 * theta, 0, -0.5 * (lambda + phi)}, {target});
822 c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
823 c2.add_op<Qubit>(OpType::U3, {0.5 * theta, phi, 0}, {target});
824 } break;
825 }
826 }
827
828 // 3. Account for phase by appending a CnU1 to the control qubits.
829 if (!equiv_0(a)) {
830 Circuit cnu1 = CnU1(n_controls - 1, a);
831 c2.append(cnu1);
832 }
833
834 c2.remove_noops();
835 return c2;
836}

889// A gate block containing Cn* gates that can be merged as a single CnU gate
890// a block can also contain a single Barrier, which will be left in place
891// TODO: conjugation boxs are accepted as well; however they don't fit the
892// semantics. control_qubits and target_qubit don't mean anything for a
893// conjugation box.
894struct CnGateBlock {
895 enum class MergeMode { append, prepend };
896 CnGateBlock(const Command &command) {
897 // Assumes the color of the target is not identity
898 Op_ptr op = command.get_op_ptr();
899 ops.push_back(op);
900 unit_vector_t args = command.get_args();
901 TKET_ASSERT(!args.empty());
902 for (unsigned i = 0; i < args.size() - 1; i++) {
903 control_qubits.insert(args[i].index()[0]);
904 }
905 target_qubit = args.back().index()[0];
906 is_barrier = (op->get_type() == OpType::Barrier);
907 is_conjugation_box = (op->get_type() == OpType::ConjugationBox);
908 is_symmetric =
909 (op->get_type() == OpType::CZ || op->get_type() == OpType::CnZ ||
910 op->get_type() == OpType::CU1);
911 color = (is_barrier || is_conjugation_box)
912 ? std::nullopt
913 : as_gate_ptr(op)->commuting_basis(args.size() - 1);
914 if (color == Pauli::I) {
915 throw std::invalid_argument(
916 "CnGateBlock doesn't accept multi-controlled identity gate.");
917 }
918 }
919
920 // Check whether commute with another CnGateBlock
921 bool commutes_with(const CnGateBlock &other) {
922 if (is_barrier || other.is_barrier || is_conjugation_box ||
923 other.is_conjugation_box) {
924 // they commute only if they have no args in common
925 std::set<unsigned> common_args;
926 std::set<unsigned> args = control_qubits;
927 args.insert(target_qubit);
928 std::set<unsigned> other_args = other.control_qubits;
929 other_args.insert(other.target_qubit);
930 std::set_intersection(
931 args.begin(), args.end(), other_args.begin(), other_args.end(),
932 std::inserter(common_args, common_args.begin()));
933 return common_args.empty();
934 }
935 if (target_qubit == other.target_qubit) {
936 return (color == other.color && color != std::nullopt);
937 }
938 if (control_qubits.contains(other.target_qubit) &&
939 other.color != Pauli::Z) {
940 return false;
941 }
942 if (other.control_qubits.contains(target_qubit) && color != Pauli::Z) {
943 return false;
944 }
945 return true;
946 }
947
948 // Check whether can be merged with another CnGateBlock
949 bool is_mergeable_with(const CnGateBlock &other) {
950 if (is_barrier || other.is_barrier || is_conjugation_box ||
951 other.is_conjugation_box) {
952 return false;
953 }
954 // check if sizes match
955 if (control_qubits.size() != other.control_qubits.size()) {
956 return false;
957 }
958 // check if they act on the same set of qubits
959 std::set<unsigned> args_a = control_qubits;
960 args_a.insert(target_qubit);
961 std::set<unsigned> args_b = other.control_qubits;
962 args_b.insert(other.target_qubit);
963 if (args_a != args_b) {
964 return false;
965 }
966 // false if target don't match and none of them is symmetric
967 if (target_qubit != other.target_qubit && !is_symmetric &&
968 !other.is_symmetric) {
969 return false;
970 }
971 return true;
972 }
973
974 // Merge with another CnGateBlock
975 void merge(CnGateBlock &other, const MergeMode &mode) {
976 if (mode == MergeMode::append) {
977 ops.insert(ops.end(), other.ops.begin(), other.ops.end());
978 } else {
979 ops.insert(ops.begin(), other.ops.begin(), other.ops.end());
980 }
981 color = (color != other.color) ? std::nullopt : color;
982 if (is_symmetric && !other.is_symmetric) {
983 control_qubits = other.control_qubits;
984 target_qubit = other.target_qubit;
985 is_symmetric = false;
986 }
987 // empty the other CnGateBlock
988 other.ops.clear();
989 }
990
991 Eigen::Matrix2cd get_target_unitary() const {
992 Eigen::Matrix2cd m = Eigen::Matrix2cd::Identity();
993 for (const Op_ptr &op : ops) {
994 m = unitary_product2(get_target_op_matrix(op), m);
995 }
996 return m;
997 }
998
999 // ops in the block
1000 std::vector<Op_ptr> ops;
1001 // target qubit index
1002 unsigned target_qubit;
1003 // control indices
1004 std::set<unsigned> control_qubits;
1005 // whether the block is used as a barrier
1006 bool is_barrier;
1007 // whether the block contains a single ConjugationBox
1008 bool is_conjugation_box;
1009 // whether the target can act on any of its qubits
1010 bool is_symmetric;
1011 // color of the target qubit
1012 std::optional<Pauli> color;
1013};

1015// Construct a controlled version of a given circuit
1016// with the assumption that the circuit does not have symbols.
1017static Circuit with_controls_numerical(const Circuit &c, unsigned n_controls) {
1018 if (c.n_bits() != 0 || !c.is_simple()) {
1019 throw CircuitInvalidity("Only default qubit register allowed");
1020 }
1021
1022 Circuit c1(c);
1023 // Replace wire swaps with SWAP gates
1024 c1.replace_all_implicit_wire_swaps();
1025
1026 // Dispose of the trivial case
1027 if (n_controls == 0) {
1028 return c1;
1029 }
1030 // 1. Rebase to Cn* gates (n=0 for single qubit gates)
1031 VertexList bin;
1032 BGL_FORALL_VERTICES(v, c1.dag, DAG) {
1033 Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
1034 OpType optype = op->get_type();
1035 if (is_gate_type(optype)) {
1036 if (is_projective_type(optype)) {
1037 throw CircuitInvalidity("Projective operations present");
1038 }
1039 if (is_single_qubit_type(optype) || is_controlled_gate_type(optype)) {
1040 continue;
1041 }
1042 Circuit replacement;
1043 if (optype == OpType::PhaseGadget) {
1044 replacement = phase_gadget(
1045 op->n_qubits(), op->get_params()[0], CXConfigType::Snake);
1046 if (replacement.n_gates() > 0) {
1047 TKET_ASSERT(
1048 replacement.n_gates() == 1 &&
1049 replacement.count_gates(OpType::ConjugationBox) == 1);
1050 }
1051 } else {
1052 replacement = with_CX(as_gate_ptr(op));
1053 }
1054 c1.substitute(replacement, v, Circuit::VertexDeletion::No);
1055 bin.push_back(v);
1056 } else if (is_box_type(optype) && optype != OpType::ConjugationBox) {
1057 throw CircuitInvalidity("Undecomposed boxes present");
1058 } else if (
1059 optype != OpType::Input && optype != OpType::Output &&
1060 optype != OpType::Barrier && optype != OpType::ConjugationBox) {
1061 throw CircuitInvalidity(
1062 "Cannot construct the controlled version of " + op->get_name());
1063 }
1064 }
1065 c1.remove_vertices(
1066 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
1067
1068 // 2. try to partition the circuit into blocks of Cn* gates such that
1069 // the gates in each block can be merged into a single CnU gate
1070 std::vector<Command> commands = c1.get_commands();
1071 std::vector<CnGateBlock> blocks;
1072
1073 Expr controlled_phase = c1.get_phase();
1074
1075 for (const Command &cmd : commands) {
1076 // if the gate is an identity up to a phase, add it as a controlled phase
1077 std::optional<double> phase = std::nullopt;
1078 OpType optype = cmd.get_op_ptr()->get_type();
1079 if (optype != OpType::Barrier && optype != OpType::ConjugationBox) {
1080 phase = cmd.get_op_ptr()->is_identity();
1081 }
1082 if (phase != std::nullopt) {
1083 controlled_phase += phase.value();
1084 } else {
1085 blocks.push_back(CnGateBlock(cmd));
1086 }
1087 }
1088
1089 // iterate the blocks from left to right
1090 for (unsigned i = 0; i + 1 < blocks.size(); i++) {
1091 CnGateBlock &b = blocks[i];
1092 if (b.ops.empty()) {
1093 continue;
1094 }
1095 // try to merge b to a block on the right
1096 for (unsigned j = i + 1; j < blocks.size(); j++) {
1097 CnGateBlock &candidate = blocks[j];
1098 if (candidate.ops.empty()) {
1099 continue;
1100 }
1101 if (b.is_mergeable_with(candidate)) {
1102 candidate.merge(b, CnGateBlock::MergeMode::prepend);
1103 break;
1104 }
1105 if (!b.commutes_with(candidate)) {
1106 break;
1107 }
1108 }
1109 }
1110
1111 // iterate the blocks from right to left
1112 for (unsigned i = blocks.size(); i-- > 1;) {
1113 CnGateBlock &b = blocks[i];
1114 if (b.ops.empty()) {
1115 continue;
1116 }
1117 // try to merge b to a block on the left
1118 // iterate from i-1 to 0
1119 for (unsigned j = i; j-- > 0;) {
1120 CnGateBlock &candidate = blocks[j];
1121 if (candidate.ops.empty()) {
1122 continue;
1123 }
1124 if (b.is_mergeable_with(candidate)) {
1125 candidate.merge(b, CnGateBlock::MergeMode::append);
1126 break;
1127 }
1128 if (!b.commutes_with(candidate)) {
1129 break;
1130 }
1131 }
1132 }
1133 // 3. Add each block to c2 either as a CnX, CnZ, CnY, CnU decomposition
1134 // or an in-place Barrier
1135 Circuit c2(n_controls + c1.n_qubits());
1136 const static Eigen::Matrix2cd X = Gate(OpType::X, {}, 1).get_unitary();
1137 const static Eigen::Matrix2cd Y = Gate(OpType::Y, {}, 1).get_unitary();
1138 const static Eigen::Matrix2cd Z = Gate(OpType::Z, {}, 1).get_unitary();
1139
1140 for (const CnGateBlock &b : blocks) {
1141 if (b.ops.empty()) {
1142 continue;
1143 }
1144 // Barriers are left in place
1145 if (b.is_barrier) {
1146 qubit_vector_t new_args;
1147 for (const unsigned i : b.control_qubits) {
1148 new_args.push_back(Qubit(i + n_controls));
1149 }
1150 new_args.push_back(Qubit(b.target_qubit + n_controls));
1151 TKET_ASSERT(b.ops.size() == 1);
1152 c2.add_op(b.ops[0], new_args);
1153 continue;
1154 }
1155 if (b.is_conjugation_box) {
1156 unit_vector_t args;
1157 for (const unsigned i : b.control_qubits) {
1158 args.push_back(Qubit(i));
1159 }
1160 args.push_back(Qubit(b.target_qubit));
1161 TKET_ASSERT(b.ops.size() == 1);
1162 c2.append(controlled_conjugation_box(b.ops[0], n_controls, args));
1163 continue;
1164 }
1165 // Computes the target unitary
1166 Eigen::Matrix2cd m = b.get_target_unitary();

1168 continue;

1171 qubit_vector_t new_args;
1172 for (unsigned i = 0; i < n_controls; i++) {
1173 new_args.push_back(Qubit(i));
1174 }
1175 for (const unsigned i : b.control_qubits) {
1176 new_args.push_back(Qubit(i + n_controls));
1177 }
1178 new_args.push_back(Qubit(b.target_qubit + n_controls));

1180 };
1181 if (m.isApprox(X, EPS)) {
1182 qubit_vector_t new_args = get_args();
1183 c2.add_op<Qubit>(CNXTYPE(new_args.size()), new_args);
1184 continue;
1185 }
1186 if (m.isApprox(Y, EPS)) {

1188 c2.add_op<Qubit>(CNYTYPE(new_args.size()), new_args);
1189 continue;
1190 }

1192 qubit_vector_t new_args = get_args();
1193 c2.add_op<Qubit>(CNZTYPE(new_args.size()), new_args);
1194 continue;
1195 }
1196 unit_map_t unit_map;
1197 for (unsigned i = 0; i < n_controls; i++) {
1198 unit_map.insert({Qubit(i), Qubit(i)});
1199 }

1201 for (const unsigned i : b.control_qubits) {
1202 unit_map.insert({Qubit(control_index++), Qubit(i + n_controls)});
1203 }
1204 unit_map.insert({Qubit(control_index), Qubit(b.target_qubit + n_controls)});
1205
1206 unsigned total_controls = b.control_qubits.size() + n_controls;
1207
1208 Circuit replacement;
1209
1210 // Check if the matrix is SU(2)
1211 if (std::abs(m.determinant() - 1.) < EPS) {
1212 // We have three functions that can decompose a multi-controlled SU(2).
1213 // The choice is based on the average number of CXs they produce for a
1214 // random n-controlled SU(2) gate.
1215 if (total_controls > 2 && total_controls < 5) {
1216 replacement = CircPool::CnU_gray_code_decomp(total_controls, m);
1217 } else if (total_controls >= 5 && total_controls < 9) {
1218 replacement = CircPool::CnU_linear_depth_decomp(total_controls, m);
1219 } else {
1220 // Compute the SU(2) angles from the TK1 angles
1221 std::vector<double> angles = tk1_angles_from_unitary(m);
1222 if (equiv_val(angles[3], 1., 2)) {
1223 // if the phase is odd, it can be absorbed into the first Rz rotation
1224 angles[0] = angles[0] + 2;
1225 } else {
1226 // because it's SU(2), the phase must be integers
1227 TKET_ASSERT(equiv_0(angles[3], 2));
1228 }
1229 // convert tk1 angles to zyz angles
1230 std::vector<double> zyz_angles = {
1231 angles[0] - 0.5, angles[1], angles[2] + 0.5};
1232 replacement = CircPool::CnSU2_linear_decomp(
1233 total_controls, zyz_angles[0], zyz_angles[1], zyz_angles[2]);
1234 }
1235 } else {
1236 // The gray code method produces less CXs when total_controls is 3 or 4
1237 if (total_controls == 3 || total_controls == 4) {
1238 replacement = CircPool::CnU_gray_code_decomp(total_controls, m);
1239 } else {
1240 replacement = CircPool::CnU_linear_depth_decomp(total_controls, m);
1241 }
1242 }
1243 c2.append_with_map(replacement, unit_map);
1244 }
1245
1246 // 4. implement the controlled phase as a CnU1 gate
1247 if (!equiv_0(controlled_phase)) {
1248 Circuit cnu1_circ = CnU1(n_controls - 1, controlled_phase);
1249 c2.append(cnu1_circ);
1250 }
1251 return c2;
1252}

1269std::tuple<Circuit, std::array<Expr, 3>, Circuit> normalise_TK2_angles(
1270 Expr a, Expr b, Expr c) {
1271 std::optional<double> a_eval = eval_expr_mod(a, 4);
1272 std::optional<double> b_eval = eval_expr_mod(b, 4);
1273 std::optional<double> c_eval = eval_expr_mod(c, 4);
1274
1275 Circuit pre(2), post(2);
1276
1277 // Add ot.dagger() at beggining and ot at end.
1278 auto conj = [&pre, &post](OpType ot) {
1279 Op_ptr op = get_op_ptr(ot);
1280 Op_ptr opdg = op->dagger();
1281 pre.add_op<unsigned>(opdg, {0});
1282 pre.add_op<unsigned>(opdg, {1});
1283 // These get undaggered at the end
1284 post.add_op<unsigned>(opdg, {0});
1285 post.add_op<unsigned>(opdg, {1});
1286 };
1287
1288 // Step 1: For non-symbolic: a, b, c [0, 1] [3, 4].
1289 if (a_eval && *a_eval > 1. && *a_eval <= 3.) {
1290 a -= 2;
1291 *a_eval -= 2;
1292 pre.add_phase(1);
1293 *a_eval = fmodn(*a_eval, 4);
1294 }
1295 if (b_eval && *b_eval > 1. && *b_eval <= 3.) {
1296 b -= 2;
1297 *b_eval -= 2;
1298 pre.add_phase(1);
1299 *b_eval = fmodn(*b_eval, 4);
1300 }
1301 if (c_eval && *c_eval > 1. && *c_eval <= 3.) {
1302 c -= 2;
1303 *c_eval -= 2;
1304 pre.add_phase(1);
1305 *c_eval = fmodn(*c_eval, 4);
1306 }
1307
1308 // Step 2: Make sure that symbolic expressions come before non-symbolics.
1309 if (a_eval && !b_eval) {
1310 // Swap XX and YY.
1311 conj(OpType::S);
1312 std::swap(a, b);
1313 std::swap(a_eval, b_eval);
1314 } else if (a_eval && !c_eval) {
1315 // Swap XX and ZZ.
1316 conj(OpType::H);
1317 std::swap(a, c);
1318 std::swap(a_eval, c_eval);
1319 }
1320 if (b_eval && !c_eval) {
1321 // Swap YY and ZZ.
1322 conj(OpType::V);
1323 std::swap(b, c);
1324 std::swap(b_eval, c_eval);
1325 }
1326
1327 // Step 3: Order non-symbolic expressions in decreasing order.
1328 auto val_in_weyl = [](double r) {
1329 // Value of r once projected into Weyl chamber.
1330 return std::min(fmodn(r, 1), 1 - fmodn(r, 1));
1331 };
1332 if (a_eval && b_eval && val_in_weyl(*a_eval) < val_in_weyl(*b_eval)) {
1333 // Swap XX and YY.
1334 conj(OpType::S);
1335 std::swap(a, b);
1336 std::swap(a_eval, b_eval);
1337 }
1338 if (b_eval && c_eval && val_in_weyl(*b_eval) < val_in_weyl(*c_eval)) {
1339 // Swap YY and ZZ.
1340 conj(OpType::V);
1341 std::swap(b, c);
1342 std::swap(b_eval, c_eval);
1343 }
1344 if (a_eval && b_eval && val_in_weyl(*a_eval) < val_in_weyl(*b_eval)) {
1345 // Swap XX and YY.
1346 conj(OpType::S);
1347 std::swap(a, b);
1348 std::swap(a_eval, b_eval);
1349 }
1350
1351 // Step 4: Project into Weyl chamber.
1352 if (a_eval && *a_eval > 1.) {
1353 a -= 3.;
1354 *a_eval -= 3;
1355 post.add_op<unsigned>(OpType::X, {0});
1356 post.add_op<unsigned>(OpType::X, {1});
1357 pre.add_phase(0.5);
1358 }
1359 if (b_eval && *b_eval > 1.) {
1360 b -= 3.;
1361 *b_eval -= 3;
1362 post.add_op<unsigned>(OpType::Y, {0});
1363 post.add_op<unsigned>(OpType::Y, {1});
1364 pre.add_phase(0.5);
1365 }
1366 if (c_eval && *c_eval > 1.) {
1367 c -= 3.;
1368 *c_eval -= 3;
1369 post.add_op<unsigned>(OpType::Z, {0});
1370 post.add_op<unsigned>(OpType::Z, {1});
1371 pre.add_phase(0.5);
1372 }
1373 if (a_eval && *a_eval > .5) {
1374 a = 1. - a;
1375 *a_eval = 1. - *a_eval;
1376 b = 1. - b;
1377 *b_eval = 1. - *b_eval;
1378 pre.add_op<unsigned>(OpType::Z, {0});
1379 post.add_op<unsigned>(OpType::Z, {1});
1380 }
1381 if (b_eval && (*b_eval > .5)) {
1382 b = 1 - b;
1383 *b_eval = 1. - *b_eval;
1384 c = 1 - c;
1385 *c_eval = 1. - *c_eval;
1386 pre.add_op<unsigned>(OpType::X, {0});
1387 post.add_op<unsigned>(OpType::X, {1});
1388 }
1389 if (c_eval && *c_eval > .5) {
1390 c -= 1;
1391 *c_eval -= 1.;
1392 post.add_op<unsigned>(OpType::Z, {0});
1393 post.add_op<unsigned>(OpType::Z, {1});
1394 pre.add_phase(-0.5);
1395 }
1396 // Calling post.dagger() is a cheeky way of reversing order of ops.
1397 return {pre, {a, b, c}, post.dagger()};
1398}
