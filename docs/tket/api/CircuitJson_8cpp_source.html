20 void to_json(nlohmann::json& j, const Circuit& circ) {
21  const auto name = circ.get_name();
22  if (name) {
23  j["name"] = name.value();
24  }
25  j["phase"] = circ.get_phase();
26  j["qubits"] = circ.all_qubits();
27  j["bits"] = circ.all_bits();
28  if (circ._number_of_wasm_wires > 0) {
29  j["number_of_ws"] = circ._number_of_wasm_wires;
30  }
31 
32  const auto impl = circ.implicit_qubit_permutation();
33  // empty maps are mapped to null instead of empty array
34  if (impl.empty()) {
35  j["implicit_permutation"] = nlohmann::json::array();
36  } else {
37  j["implicit_permutation"] = impl;
38  }
39  j["commands"] = nlohmann::json::array();
40  for (const Command& com : circ) {
41  j["commands"].push_back(com);
42  }
43  j["created_qubits"] = circ.created_qubits();
44  j["discarded_qubits"] = circ.discarded_qubits();
45 }

47 void from_json(const nlohmann::json& j, Circuit& circ) {
48  circ = Circuit();
49 
50  if (j.contains("name")) {
51  circ.set_name(j["name"].get<std::string>());
52  }
53  circ.add_phase(j.at("phase").get<Expr>());
54  const auto& qubits = j.at("qubits").get<qubit_vector_t>();
55  for (const auto& qb : qubits) {
56  circ.add_qubit(qb);
57  }
58  const auto& bits = j.at("bits").get<bit_vector_t>();
59  for (const auto& b : bits) {
60  circ.add_bit(b);
61  }
62  if (j.contains("number_of_ws")) {
63  circ.add_wasm_register(j.at("number_of_ws").get<unsigned>());
64  }
65 
66  for (const auto& j_com : j.at("commands")) {
67  const auto& com = j_com.get<Command>();
68  circ.add_op(com.get_op_ptr(), com.get_args(), com.get_opgroup());
69  }
70  const auto& imp_perm = j.at("implicit_permutation").get<qubit_map_t>();
71  circ.permute_boundary_output(imp_perm);
72  // Check if key exists to work with circuits serialised using older tket
73  // versions
74  if (j.contains("created_qubits")) {
75  for (const auto& j_q : j.at("created_qubits")) {
76  const auto& q = j_q.get<Qubit>();
77  circ.qubit_create(q);
78  }
79  }
80  if (j.contains("discarded_qubits")) {
81  for (const auto& j_q : j.at("discarded_qubits")) {
82  const auto& q = j_q.get<Qubit>();
83  circ.qubit_discard(q);
84  }
85  }
86 }
