19void to_json(nlohmann::json& j, const Circuit& circ) {
20 const auto name = circ.get_name();
21 if (name) {
22 j["name"] = name.value();
23 }
24 j["phase"] = circ.get_phase();
25 j["qubits"] = circ.all_qubits();
26 j["bits"] = circ.all_bits();
27 if (circ._number_of_wasm_wires > 0) {
28 j["number_of_ws"] = circ._number_of_wasm_wires;
29 }
30
31 const auto impl = circ.implicit_qubit_permutation();
32 // empty maps are mapped to null instead of empty array
33 if (impl.empty()) {
34 j["implicit_permutation"] = nlohmann::json::array();
35 } else {
36 j["implicit_permutation"] = impl;
37 }
38 j["commands"] = nlohmann::json::array();
39 for (const Command& com : circ) {
40 j["commands"].push_back(com);
41 }
42 j["created_qubits"] = circ.created_qubits();
43 j["discarded_qubits"] = circ.discarded_qubits();
44}

46void from_json(const nlohmann::json& j, Circuit& circ) {
47 circ = Circuit();
48
49 if (j.contains("name")) {
50 circ.set_name(j["name"].get<std::string>());
51 }
52 circ.add_phase(j.at("phase").get<Expr>());
53 const auto& qubits = j.at("qubits").get<qubit_vector_t>();
54 for (const auto& qb : qubits) {
55 circ.add_qubit(qb);
56 }
57 const auto& bits = j.at("bits").get<bit_vector_t>();
58 for (const auto& b : bits) {
59 circ.add_bit(b);
60 }
61 if (j.contains("number_of_ws")) {
62 circ.add_wasm_register(j.at("number_of_ws").get<unsigned>());
63 }
64
65 for (const auto& j_com : j.at("commands")) {
66 const auto& com = j_com.get<Command>();
67 circ.add_op(com.get_op_ptr(), com.get_args(), com.get_opgroup());
68 }
69 const auto& imp_perm = j.at("implicit_permutation").get<qubit_map_t>();
70 circ.permute_boundary_output(imp_perm);
71 // Check if key exists to work with circuits serialised using older tket
72 // versions
73 if (j.contains("created_qubits")) {
74 for (const auto& j_q : j.at("created_qubits")) {
75 const auto& q = j_q.get<Qubit>();
76 circ.qubit_create(q);
77 }
78 }
79 if (j.contains("discarded_qubits")) {
80 for (const auto& j_q : j.at("discarded_qubits")) {
81 const auto& q = j_q.get<Qubit>();
82 circ.qubit_discard(q);
83 }
84 }
85}
