36 static void apply_unitary_may_throw(
37  const Circuit& circ, Eigen::MatrixXcd& matr, double abs_epsilon,
38  unsigned max_number_of_qubits) {
39  if (circ.n_qubits() > max_number_of_qubits) {
40  throw GateUnitaryMatrixError(
41  "Circuit to simulate has too many qubits",
42  GateUnitaryMatrixError::Cause::TOO_MANY_QUBITS);
43  }
44  if (matr.cols() <= 0) {
45  throw GateUnitaryMatrixError(
46  "M has no columns", GateUnitaryMatrixError::Cause::INPUT_ERROR);
47  }
48  const auto full_matr_size = get_matrix_size(circ.n_qubits());
49  if (matr.rows() != full_matr_size) {
50  throw GateUnitaryMatrixError(
51  "M has wrong number of rows",
52  GateUnitaryMatrixError::Cause::INPUT_ERROR);
53  }
54  internal::GateNodesBuffer buffer(matr, abs_epsilon);
55  internal::decompose_circuit(circ, buffer, abs_epsilon);
56  matr = apply_qubit_permutation(matr, circ.implicit_qubit_permutation());
57 }

59 void apply_unitary(
60  const Circuit& circ, Eigen::MatrixXcd& matr, double abs_epsilon,
61  unsigned max_number_of_qubits) {
62  try {
63  apply_unitary_may_throw(circ, matr, abs_epsilon, max_number_of_qubits);
64  } catch (const GateUnitaryMatrixError& e) {
65  const auto full_matr_size = get_matrix_size(circ.n_qubits());
66  std::stringstream ss;
67  ss << "Error trying to simulate circuit " << circ << " with "
68  << circ.n_qubits() << " qubits, " << circ.get_commands().size()
69  << " commands; U is size " << full_matr_size << "x" << full_matr_size
70  << ", premultiplying M with " << matr.rows() << " rows, " << matr.cols()
71  << " cols: " << e.what();
72  if (e.cause == GateUnitaryMatrixError::Cause::GATE_NOT_IMPLEMENTED) {
73  throw Unsupported(ss.str());
74  } else if (e.cause == GateUnitaryMatrixError::Cause::SYMBOLIC_PARAMETERS) {
75  throw SymbolsNotSupported(ss.str());
76  } else {
77  throw e;
78  }
79  }
80 }
