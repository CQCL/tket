35 // Out stream of graphviz code describing the circuit from Top-Bottom,
36 // information apart from qubit path and slices can be seen easily.
37 // Very useful for debugging and eyeball comparison of circuits.
38 // out << "\nrankdir=\"LR\"" <--- put this in for Left-Right circuit
39 void Circuit::to_graphviz(std::ostream &out) const {
40  IndexMap im = index_map();
41 
42  out << "digraph G {\n";
43  out << "{ rank = same\n";
44  for (const Vertex &v : all_inputs()) {
45  out << im[v] << " ";
46  }
47  out << "}\n";
48  out << "{ rank = same\n";
49  for (const Vertex &v : all_outputs()) {
50  out << im[v] << " ";
51  }
52  out << "}\n";
53 
54  BGL_FORALL_VERTICES(v, dag, DAG) {
55  out << im[v] << " [label = \"" << get_Op_ptr_from_Vertex(v)->get_name()
56  << ", " << im[v] << "\"];\n";
57  }
58  BGL_FORALL_EDGES(e, dag, DAG) {
59  Vertex v_so = source(e);
60  Vertex v_ta = target(e);
61  unsigned v_s = im[v_so];
62  unsigned v_t = im[v_ta];
63  out << v_s << " -> " << v_t << " [label = \"" << get_source_port(e) << ", "
64  << get_target_port(e) << "\"];\n";
65  }
66  out << "}";
67 }

80 void Circuit::extract_slice_segment(unsigned slice_one, unsigned slice_two) {
81  SliceVec slices = get_slices();
82  VertexList bin;
83  for (unsigned i = 0; i < slice_one - 1; ++i) {
84  for (Vertex it : slices[i]) {
85  bin.push_back(it);
86  remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
87  }
88  }
89  for (unsigned i = slice_two; i < slices.size(); ++i) {
90  for (Vertex it : slices[i]) {
91  bin.push_back(it);
92  remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
93  }
94  }
95  remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
96 }

143 void Circuit::symbol_substitution(const symbol_map_t &symbol_map) {
144  SymEngine::map_basic_basic sub_map;
145  for (const std::pair<const Sym, Expr> &p : symbol_map) {
146  ExprPtr s = p.first;
147  ExprPtr e = p.second;
148  // This is a workaround for a symengine issue: symengine currently has poor
149  // handling of symbolic evaluations for atan2. However, this may not catch
150  // every such issue, so we should revisit it.
151  if (approx_0(e)) {
152  sub_map[s] = SymEngine::zero;
153  } else {
154  sub_map[s] = e;
155  }
156  }
157  symbol_substitution(sub_map);
158 }

192 // check aspects of circuit for equality, and optionally throw exceptions when
193 // not met
194 bool Circuit::circuit_equality(
195  const Circuit &other, const std::set<Check> &except,
196  bool throw_error) const {
197  bool check = true;
198  check &= check_iterators_equality(*this, other);
199  if (throw_error && !check) {
200  throw CircuitInequality(std::string("Circuit operations do not match."));
201  }
202 
203  if (except.count(Check::Phase) == 0) {
204  const Expr thisphase = this->get_phase();
205  const Expr othephase = other.get_phase();
206  check &= equiv_expr(thisphase, othephase);
207  if (throw_error && !check) {
208  throw CircuitInequality(
209  std::string("Circuit phases do not match: ") +
210  ExprPtr(thisphase)->__str__() +
211  " != " + ExprPtr(othephase)->__str__());
212  }
213  }
214  if (except.count(Check::Units) == 0) {
215  check &= (this->all_qubits() == other.all_qubits());
216  if (throw_error && !check) {
217  throw CircuitInequality(std::string("Circuit qubits do not match."));
218  }
219 
220  check &= (this->all_bits() == other.all_bits());
221  if (throw_error && !check) {
222  throw CircuitInequality(std::string("Circuit bits do not match."));
223  }
224  check &= (this->created_qubits() == other.created_qubits());
225  if (throw_error && !check) {
226  throw CircuitInequality(
227  std::string("Circuit created qubits do not match."));
228  }
229  check &= (this->discarded_qubits() == other.discarded_qubits());
230  if (throw_error && !check) {
231  throw CircuitInequality(
232  std::string("Circuit discarded qubits do not match."));
233  }
234  }
235 
236  if (except.count(Check::ImplicitPermutation) == 0) {
237  check &=
238  (this->implicit_qubit_permutation() ==
239  other.implicit_qubit_permutation());
240  if (throw_error && !check) {
241  throw CircuitInequality(
242  std::string("Circuit implicit permutations do not match."));
243  }
244  }
245  if (except.count(Check::Name) == 0) {
246  check &= (this->get_name() == other.get_name());
247  if (throw_error && !check) {
248  const std::optional<std::string> thisname = this->get_name();
249  const std::optional<std::string> othename = other.get_name();
250  std::string errormsg = "Circuit names do not match: ";
251  errormsg += (thisname ? thisname.value() : "None");
252  errormsg += " != ";
253  errormsg += (othename ? othename.value() : "None");
254 
255  throw CircuitInequality(errormsg);
256  }
257  }
258  return check;
259 }

261 // Performs a traversal from the given vertex forwards through the dag, looking
262 // for something on the target qubit We can prune a path if it reaches the depth
263 // of the target forward = true returns true if target is in causal future of
264 // from forward = false checks for causal past (v_to_depth should give reverse
265 // depth)
266 // TODO:: rewrite to work with classical boxes
267 bool Circuit::in_causal_order(
268  const Vertex &target, const Vertex &from, bool forward,
269  const std::map<Vertex, unsigned> &v_to_depth,
270  const std::map<Vertex, unit_set_t> &v_to_units, bool strict) const {
271  unsigned target_depth = v_to_depth.at(target);
272  if (!strict && from == target) return true;
273  if (v_to_depth.at(from) >= target_depth) return false;
274  typedef std::function<bool(Vertex, Vertex)> Comp;
275  Comp c = [&](Vertex a, Vertex b) {
276  unsigned deptha = v_to_depth.at(a);
277  unsigned depthb = v_to_depth.at(b);
278  if (deptha == depthb) {
279  unit_set_t unitsa = v_to_units.at(a);
280  unit_set_t unitsb = v_to_units.at(b);
281  return unitsa < unitsb;
282  }
283  return deptha < depthb;
284  };
285  std::set<Vertex, Comp> to_search(c);
286  if (forward) {
287  VertexVec succs = get_successors(from);
288  for (const Vertex &s : succs) {
289  if (v_to_depth.find(s) != v_to_depth.end()) {
290  to_search.insert(s);
291  }
292  }
293  } else {
294  VertexVec preds = get_predecessors(from);
295  to_search.insert(preds.begin(), preds.end());
296  }
297  unit_set_t lookup_units = v_to_units.at(target);
298  while (!to_search.empty()) {
299  Vertex v = *to_search.begin();
300  to_search.erase(to_search.begin());
301  if (v_to_depth.at(v) > target_depth) continue;
302  unit_set_t v_units = v_to_units.at(v);
303  for (const UnitID &u : lookup_units) {
304  if (v_units.find(u) != v_units.end()) {
305  return true;
306  }
307  }
308  if (forward) {
309  VertexVec succs = get_successors(v);
310  for (const Vertex &s : succs) {
311  if (v_to_depth.find(s) != v_to_depth.end()) {
312  to_search.insert(s);
313  }
314  }
315  } else {
316  VertexVec preds = get_predecessors(v);
317  to_search.insert(preds.begin(), preds.end());
318  }
319  }
320  return false;
321 }
