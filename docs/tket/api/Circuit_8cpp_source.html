44 // information apart from qubit path and slices can be seen easily.

47 void Circuit::to_graphviz(std::ostream& out) const {
48  IndexMap im = index_map();
49 
50  out << "digraph G {\n";
51  out << "{ rank = same\n";
52  for (const Vertex& v : all_inputs()) {
53  out << im[v] << " ";
54  }
55  out << "}\n";
56  out << "{ rank = same\n";
57  for (const Vertex& v : all_outputs()) {
58  out << im[v] << " ";
59  }
60  out << "}\n";
61 
62  BGL_FORALL_VERTICES(v, dag, DAG) {
63  out << im[v] << " [label = \"" << get_Op_ptr_from_Vertex(v)->get_name()
64  << ", " << im[v] << "\"];\n";
65  }
66  BGL_FORALL_EDGES(e, dag, DAG) {
67  Vertex v_so = source(e);
68  Vertex v_ta = target(e);
69  unsigned v_s = im[v_so];
70  unsigned v_t = im[v_ta];
71  out << v_s << " -> " << v_t << " [label = \"" << get_source_port(e) << ", "
72  << get_target_port(e) << "\"];\n";
73  }
74  out << "}";
75 }

88 void Circuit::extract_slice_segment(unsigned slice_one, unsigned slice_two) {
89  SliceVec slices = get_slices();
90  VertexList bin;
91  for (unsigned i = 0; i < slice_one - 1; ++i) {
92  for (Vertex it : slices[i]) {
93  bin.push_back(it);
94  remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
95  }
96  }
97  for (unsigned i = slice_two; i < slices.size(); ++i) {
98  for (Vertex it : slices[i]) {
99  bin.push_back(it);
100  remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
101  }
102  }
103  remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
104 }

151 void Circuit::symbol_substitution(const symbol_map_t& symbol_map) {
152  SymEngine::map_basic_basic sub_map;
153  for (const std::pair<const Sym, Expr>& p : symbol_map) {
154  ExprPtr s = p.first;
155  ExprPtr e = p.second;
156  // This is a workaround for a symengine issue: symengine currently has poor
157  // handling of symbolic evaluations for atan2. However, this may not catch
158  // every such issue, so we should revisit it.
159  if (approx_0(e)) {
160  sub_map[s] = SymEngine::zero;
161  } else {
162  sub_map[s] = e;
163  }
164  }
165  symbol_substitution(sub_map);
166 }

200 // check aspects of circuit for equality, and optionally throw exceptions when
201 // not met
202 bool Circuit::circuit_equality(
203  const Circuit& other, const std::set<Check>& except,
204  bool throw_error) const {
205  bool check = true;
206  check &= check_iterators_equality(*this, other);
207  if (throw_error && !check) {
208  throw CircuitInequality(std::string("Circuit operations do not match."));
209  }
210 
211  if (except.count(Check::Phase) == 0) {
212  const Expr thisphase = this->get_phase();
213  const Expr othephase = other.get_phase();
214  check &= equiv_expr(thisphase, othephase);
215  if (throw_error && !check) {
216  throw CircuitInequality(
217  std::string("Circuit phases do not match: ") +
218  ExprPtr(thisphase)->__str__() +
219  " != " + ExprPtr(othephase)->__str__());
220  }
221  }
222  if (except.count(Check::Units) == 0) {
223  check &= (this->all_qubits() == other.all_qubits());
224  if (throw_error && !check) {
225  throw CircuitInequality(std::string("Circuit qubits do not match."));
226  }
227 
228  check &= (this->all_bits() == other.all_bits());
229  if (throw_error && !check) {
230  throw CircuitInequality(std::string("Circuit bits do not match."));
231  }
232  check &= (this->created_qubits() == other.created_qubits());
233  if (throw_error && !check) {
234  throw CircuitInequality(
235  std::string("Circuit created qubits do not match."));
236  }
237  check &= (this->discarded_qubits() == other.discarded_qubits());
238  if (throw_error && !check) {
239  throw CircuitInequality(
240  std::string("Circuit discarded qubits do not match."));
241  }
242  }
243 
244  if (except.count(Check::ImplicitPermutation) == 0) {
245  check &=
246  (this->implicit_qubit_permutation() ==
247  other.implicit_qubit_permutation());
248  if (throw_error && !check) {
249  throw CircuitInequality(
250  std::string("Circuit implicit permutations do not match."));
251  }
252  }
253  if (except.count(Check::Name) == 0) {
254  check &= (this->get_name() == other.get_name());
255  if (throw_error && !check) {
256  const std::optional<std::string> thisname = this->get_name();
257  const std::optional<std::string> othename = other.get_name();
258  std::string errormsg = "Circuit names do not match: ";
259  errormsg += (thisname ? thisname.value() : "None");
260  errormsg += " != ";
261  errormsg += (othename ? othename.value() : "None");
262 
263  throw CircuitInequality(errormsg);
264  }
265  }
266  return check;
267 }

269 // Performs a traversal from the given vertex forwards through the dag, looking
270 // for something on the target qubit We can prune a path if it reaches the depth
271 // of the target forward = true returns true if target is in causal future of
272 // from forward = false checks for causal past (v_to_depth should give reverse
273 // depth)
274 // TODO:: rewrite to work with classical boxes
275 bool Circuit::in_causal_order(
276  const Vertex& target, const Vertex& from, bool forward,
277  const std::map<Vertex, unsigned>& v_to_depth,
278  const std::map<Vertex, unit_set_t>& v_to_units, bool strict) const {
279  unsigned target_depth = v_to_depth.at(target);
280  if (!strict && from == target) return true;
281  if (v_to_depth.at(from) >= target_depth) return false;
282  typedef std::function<bool(Vertex, Vertex)> Comp;
283  Comp c = [&](Vertex a, Vertex b) {
284  unsigned deptha = v_to_depth.at(a);
285  unsigned depthb = v_to_depth.at(b);
286  if (deptha == depthb) {
287  unit_set_t unitsa = v_to_units.at(a);
288  unit_set_t unitsb = v_to_units.at(b);
289  return unitsa < unitsb;
290  }
291  return deptha < depthb;
292  };
293  std::set<Vertex, Comp> to_search(c);
294  if (forward) {
295  VertexVec succs = get_successors(from);
296  for (const Vertex& s : succs) {
297  if (v_to_depth.find(s) != v_to_depth.end()) {
298  to_search.insert(s);
299  }
300  }
301  } else {
302  VertexVec preds = get_predecessors(from);
303  to_search.insert(preds.begin(), preds.end());
304  }
305  unit_set_t lookup_units = v_to_units.at(target);
306  while (!to_search.empty()) {
307  Vertex v = *to_search.begin();
308  to_search.erase(to_search.begin());
309  if (v_to_depth.at(v) > target_depth) continue;
310  unit_set_t v_units = v_to_units.at(v);
311  for (const UnitID& u : lookup_units) {
312  if (v_units.find(u) != v_units.end()) {
313  return true;
314  }
315  }
316  if (forward) {
317  VertexVec succs = get_successors(v);
318  for (const Vertex& s : succs) {
319  if (v_to_depth.find(s) != v_to_depth.end()) {
320  to_search.insert(s);
321  }
322  }
323  } else {
324  VertexVec preds = get_predecessors(v);
325  to_search.insert(preds.begin(), preds.end());
326  }
327  }
328  return false;
329 }

331 // Update depth fields of `data` for a vertex with data for the vertex's
332 // predecessors `preds`, which is assumed to be already stored in `datamap`.
333 static void update_from_predecessors(
334  ResourceData& data, const VertexVec& preds,
335  const std::map<Vertex, ResourceData>& datamap) {
336  std::vector<ResourceData> pre_data;
337  for (const Vertex& pre_v : preds) {
338  pre_data.push_back(datamap.at(pre_v));
339  }
340  // 1. GateDepth
341  data.GateDepth.min += std::max_element(
342  pre_data.begin(), pre_data.end(),
343  [](const ResourceData& a, const ResourceData& b) {
344  return a.GateDepth.min < b.GateDepth.min;
345  })
346  ->GateDepth.min;
347  data.GateDepth.max += std::max_element(
348  pre_data.begin(), pre_data.end(),
349  [](const ResourceData& a, const ResourceData& b) {
350  return a.GateDepth.max < b.GateDepth.max;
351  })
352  ->GateDepth.max;
353  // 2. OpTypeDepth
354  std::map<OpType, unsigned> min_depths;
355  std::map<OpType, unsigned> max_depths;
356  for (const ResourceData& pre_v_data : pre_data) {
357  for (const auto& pair : pre_v_data.OpTypeDepth) {
358  if (pair.second.min > min_depths[pair.first]) {
359  min_depths[pair.first] = pair.second.min;
360  }
361  if (pair.second.max > max_depths[pair.first]) {
362  max_depths[pair.first] = pair.second.max;
363  }
364  }
365  }
366  for (const auto& pair : min_depths) {
367  data.OpTypeDepth[pair.first].min += min_depths[pair.first];
368  }
369  for (const auto& pair : max_depths) {
370  data.OpTypeDepth[pair.first].max += max_depths[pair.first];
371  }
372  // 3. TwoQubitGateDepth
373  data.TwoQubitGateDepth.min +=
374  std::max_element(
375  pre_data.begin(), pre_data.end(),
376  [](const ResourceData& a, const ResourceData& b) {
377  return a.TwoQubitGateDepth.min < b.TwoQubitGateDepth.min;
378  })
379  ->TwoQubitGateDepth.min;
380  data.TwoQubitGateDepth.max +=
381  std::max_element(
382  pre_data.begin(), pre_data.end(),
383  [](const ResourceData& a, const ResourceData& b) {
384  return a.TwoQubitGateDepth.max < b.TwoQubitGateDepth.max;
385  })
386  ->TwoQubitGateDepth.max;
387 }

389 ResourceData Circuit::get_resources() /*const*/ {
390  // Traverse the DAG in topological order. At each vertex compute a new
391  // ResourceData based on the ResourceData already computed for its immediate
392  // predecessors. Compute final ResourceData based on terminal nodes.
393  const VertexVec vertices = vertices_in_order();
394  std::map<Vertex, ResourceData> datamap;
395  std::map<OpType, ResourceBounds<unsigned>> op_type_count;
396  for (const Vertex& v : vertices) {
397  ResourceData data;
398  OpType optype = get_OpType_from_Vertex(v);
399  if (!is_initial_type(optype)) {
400  if (!is_final_type(optype)) {
401  if (optype == OpType::DummyBox) {
402  const DummyBox& dbox =
403  static_cast<const DummyBox&>(*get_Op_ptr_from_Vertex(v));
404  data = dbox.get_resource_data();
405  for (const auto& pair : data.OpTypeCount) {
406  op_type_count[pair.first].min += pair.second.min;
407  op_type_count[pair.first].max += pair.second.max;
408  }
409  } else {
410  data.GateDepth = {1};
411  data.OpTypeDepth[optype] = {1};
412  if (OpDesc(optype).is_gate() &&
413  get_Op_ptr_from_Vertex(v)->n_qubits() == 2) {
414  data.TwoQubitGateDepth = {1};
415  }
416  op_type_count[optype].min += 1;
417  op_type_count[optype].max += 1;
418  }
419  }
420  // Aggregate with predecessors
421  update_from_predecessors(data, get_predecessors(v), datamap);
422  }
423  datamap[v] = data;
424  }
425  // Finally aggregate outputs
426  ResourceData final_data;
427  update_from_predecessors(final_data, all_outputs(), datamap);
428  final_data.OpTypeCount = op_type_count;
429  return final_data;
430 }
