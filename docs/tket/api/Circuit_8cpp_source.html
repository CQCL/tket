42// Out stream of graphviz code describing the circuit from Top-Bottom,
43// information apart from qubit path and slices can be seen easily.
44// Very useful for debugging and eyeball comparison of circuits.
45// out << "\nrankdir=\"LR\"" <--- put this in for Left-Right circuit
46void Circuit::to_graphviz(std::ostream& out) const {
47 IndexMap im = index_map();
48
49 out << "digraph G {\n";
50 out << "{ rank = same\n";
51 for (const Vertex& v : all_inputs()) {
52 out << im[v] << " ";
53 }
54 out << "}\n";
55 out << "{ rank = same\n";
56 for (const Vertex& v : all_outputs()) {
57 out << im[v] << " ";
58 }
59 out << "}\n";
60
61 BGL_FORALL_VERTICES(v, dag, DAG) {
62 out << im[v] << " [label = \"" << get_Op_ptr_from_Vertex(v)->get_name()
63 << ", " << im[v] << "\"];\n";
64 }
65 BGL_FORALL_EDGES(e, dag, DAG) {
66 Vertex v_so = source(e);
67 Vertex v_ta = target(e);
68 unsigned v_s = im[v_so];
69 unsigned v_t = im[v_ta];
70 out << v_s << " -> " << v_t << " [label = \"" << get_source_port(e) << ", "
71 << get_target_port(e) << "\"];\n";
72 }
73 out << "}";
74}

87void Circuit::extract_slice_segment(unsigned slice_one, unsigned slice_two) {
88 SliceVec slices = get_slices();
89 VertexList bin;
90 for (unsigned i = 0; i < slice_one - 1; ++i) {
91 for (Vertex it : slices[i]) {
92 bin.push_back(it);
93 remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
94 }
95 }
96 for (unsigned i = slice_two; i < slices.size(); ++i) {
97 for (Vertex it : slices[i]) {
98 bin.push_back(it);
99 remove_vertex(it, GraphRewiring::Yes, VertexDeletion::No);
100 }
101 }
102 remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
103}

150void Circuit::symbol_substitution(const symbol_map_t& symbol_map) {
151 SymEngine::map_basic_basic sub_map;
152 for (const std::pair<const Sym, Expr>& p : symbol_map) {
153 ExprPtr s = p.first;
154 ExprPtr e = p.second;
155 // This is a workaround for a symengine issue: symengine currently has poor
156 // handling of symbolic evaluations for atan2. However, this may not catch
157 // every such issue, so we should revisit it.
158 if (approx_0(e)) {
159 sub_map[s] = SymEngine::zero;
160 } else {
161 sub_map[s] = e;
162 }
163 }
164 symbol_substitution(sub_map);
165}

199// check aspects of circuit for equality, and optionally throw exceptions when
200// not met
201bool Circuit::circuit_equality(
202 const Circuit& other, const std::set<Check>& except,
203 bool throw_error) const {
204 bool check = true;
205 check &= check_iterators_equality(*this, other);
206 if (throw_error && !check) {
207 throw CircuitInequality(std::string("Circuit operations do not match."));
208 }
209
210 if (except.count(Check::Phase) == 0) {
211 const Expr thisphase = this->get_phase();
212 const Expr othephase = other.get_phase();
213 check &= equiv_expr(thisphase, othephase);
214 if (throw_error && !check) {
215 throw CircuitInequality(
216 std::string("Circuit phases do not match: ") +
217 ExprPtr(thisphase)->__str__() +
218 " != " + ExprPtr(othephase)->__str__());
219 }
220 }
221 if (except.count(Check::Units) == 0) {
222 check &= (this->all_qubits() == other.all_qubits());
223 if (throw_error && !check) {
224 throw CircuitInequality(std::string("Circuit qubits do not match."));
225 }
226
227 check &= (this->all_bits() == other.all_bits());
228 if (throw_error && !check) {
229 throw CircuitInequality(std::string("Circuit bits do not match."));
230 }
231 check &= (this->created_qubits() == other.created_qubits());
232 if (throw_error && !check) {
233 throw CircuitInequality(
234 std::string("Circuit created qubits do not match."));
235 }
236 check &= (this->discarded_qubits() == other.discarded_qubits());
237 if (throw_error && !check) {
238 throw CircuitInequality(
239 std::string("Circuit discarded qubits do not match."));
240 }
241 }
242
243 if (except.count(Check::ImplicitPermutation) == 0) {
244 check &=
245 (this->implicit_qubit_permutation() ==
246 other.implicit_qubit_permutation());
247 if (throw_error && !check) {
248 throw CircuitInequality(
249 std::string("Circuit implicit permutations do not match."));
250 }
251 }
252 if (except.count(Check::Name) == 0) {
253 check &= (this->get_name() == other.get_name());
254 if (throw_error && !check) {
255 const std::optional<std::string> thisname = this->get_name();
256 const std::optional<std::string> othename = other.get_name();
257 std::string errormsg = "Circuit names do not match: ";
258 errormsg += (thisname ? thisname.value() : "None");
259 errormsg += " != ";
260 errormsg += (othename ? othename.value() : "None");
261
262 throw CircuitInequality(errormsg);
263 }
264 }
265 return check;
266}

268// Performs a traversal from the given vertex forwards through the dag, looking
269// for something on the target qubit We can prune a path if it reaches the depth
270// of the target forward = true returns true if target is in causal future of
271// from forward = false checks for causal past (v_to_depth should give reverse
272// depth)
273// TODO:: rewrite to work with classical boxes
274bool Circuit::in_causal_order(
275 const Vertex& target, const Vertex& from, bool forward,
276 const std::map<Vertex, unsigned>& v_to_depth,
277 const std::map<Vertex, unit_set_t>& v_to_units, bool strict) const {
278 unsigned target_depth = v_to_depth.at(target);
279 if (!strict && from == target) return true;
280 if (v_to_depth.at(from) >= target_depth) return false;
281 typedef std::function<bool(Vertex, Vertex)> Comp;
282 Comp c = [&](Vertex a, Vertex b) {
283 unsigned deptha = v_to_depth.at(a);
284 unsigned depthb = v_to_depth.at(b);
285 if (deptha == depthb) {
286 unit_set_t unitsa = v_to_units.at(a);
287 unit_set_t unitsb = v_to_units.at(b);
288 return unitsa < unitsb;
289 }
290 return deptha < depthb;
291 };
292 std::set<Vertex, Comp> to_search(c);
293 if (forward) {
294 VertexVec succs = get_successors(from);
295 for (const Vertex& s : succs) {
296 if (v_to_depth.find(s) != v_to_depth.end()) {
297 to_search.insert(s);
298 }
299 }
300 } else {
301 VertexVec preds = get_predecessors(from);
302 to_search.insert(preds.begin(), preds.end());
303 }
304 unit_set_t lookup_units = v_to_units.at(target);
305 while (!to_search.empty()) {
306 Vertex v = *to_search.begin();
307 to_search.erase(to_search.begin());
308 if (v_to_depth.at(v) > target_depth) continue;
309 unit_set_t v_units = v_to_units.at(v);
310 for (const UnitID& u : lookup_units) {
311 if (v_units.find(u) != v_units.end()) {
312 return true;
313 }
314 }
315 if (forward) {
316 VertexVec succs = get_successors(v);
317 for (const Vertex& s : succs) {
318 if (v_to_depth.find(s) != v_to_depth.end()) {
319 to_search.insert(s);
320 }
321 }
322 } else {
323 VertexVec preds = get_predecessors(v);
324 to_search.insert(preds.begin(), preds.end());
325 }
326 }
327 return false;
328}

330// Update depth fields of `data` for a vertex with data for the vertex's
331// predecessors `preds`, which is assumed to be already stored in `datamap`.
332static void update_from_predecessors(
333 ResourceData& data, const VertexVec& preds,
334 const std::map<Vertex, ResourceData>& datamap) {
335 std::vector<ResourceData> pre_data;
336 for (const Vertex& pre_v : preds) {
337 pre_data.push_back(datamap.at(pre_v));
338 }
339 // 1. GateDepth
340 data.GateDepth.min += std::max_element(
341 pre_data.begin(), pre_data.end(),
342 [](const ResourceData& a, const ResourceData& b) {
343 return a.GateDepth.min < b.GateDepth.min;
344 })
345 ->GateDepth.min;
346 data.GateDepth.max += std::max_element(
347 pre_data.begin(), pre_data.end(),
348 [](const ResourceData& a, const ResourceData& b) {
349 return a.GateDepth.max < b.GateDepth.max;
350 })
351 ->GateDepth.max;
352 // 2. OpTypeDepth
353 std::map<OpType, unsigned> min_depths;
354 std::map<OpType, unsigned> max_depths;
355 for (const ResourceData& pre_v_data : pre_data) {
356 for (const auto& pair : pre_v_data.OpTypeDepth) {
357 if (pair.second.min > min_depths[pair.first]) {
358 min_depths[pair.first] = pair.second.min;
359 }
360 if (pair.second.max > max_depths[pair.first]) {
361 max_depths[pair.first] = pair.second.max;
362 }
363 }
364 }
365 for (const auto& pair : min_depths) {
366 data.OpTypeDepth[pair.first].min += min_depths[pair.first];
367 }
368 for (const auto& pair : max_depths) {
369 data.OpTypeDepth[pair.first].max += max_depths[pair.first];
370 }
371 // 3. TwoQubitGateDepth
372 data.TwoQubitGateDepth.min +=
373 std::max_element(
374 pre_data.begin(), pre_data.end(),
375 [](const ResourceData& a, const ResourceData& b) {
376 return a.TwoQubitGateDepth.min < b.TwoQubitGateDepth.min;
377 })
378 ->TwoQubitGateDepth.min;
379 data.TwoQubitGateDepth.max +=
380 std::max_element(
381 pre_data.begin(), pre_data.end(),
382 [](const ResourceData& a, const ResourceData& b) {
383 return a.TwoQubitGateDepth.max < b.TwoQubitGateDepth.max;
384 })
385 ->TwoQubitGateDepth.max;
386}

388ResourceData Circuit::get_resources() /*const*/ {
389 // Traverse the DAG in topological order. At each vertex compute a new
390 // ResourceData based on the ResourceData already computed for its immediate
391 // predecessors. Compute final ResourceData based on terminal nodes.
392 const VertexVec vertices = vertices_in_order();
393 std::map<Vertex, ResourceData> datamap;
394 std::map<OpType, ResourceBounds<unsigned>> op_type_count;
395 for (const Vertex& v : vertices) {
396 ResourceData data;
397 OpType optype = get_OpType_from_Vertex(v);
398 if (!is_initial_type(optype)) {
399 if (!is_final_type(optype)) {
400 if (optype == OpType::DummyBox) {
401 const DummyBox& dbox =
402 static_cast<const DummyBox&>(*get_Op_ptr_from_Vertex(v));
403 data = dbox.get_resource_data();
404 for (const auto& pair : data.OpTypeCount) {
405 op_type_count[pair.first].min += pair.second.min;
406 op_type_count[pair.first].max += pair.second.max;
407 }
408 } else {
409 data.GateDepth = {1};
410 data.OpTypeDepth[optype] = {1};
411 if (OpDesc(optype).is_gate() &&
412 get_Op_ptr_from_Vertex(v)->n_qubits() == 2) {
413 data.TwoQubitGateDepth = {1};
414 }
415 op_type_count[optype].min += 1;
416 op_type_count[optype].max += 1;
417 }
418 }
419 // Aggregate with predecessors
420 update_from_predecessors(data, get_predecessors(v), datamap);
421 }
422 datamap[v] = data;
423 }
424 // Finally aggregate outputs
425 ResourceData final_data;
426 update_from_predecessors(final_data, all_outputs(), datamap);
427 final_data.OpTypeCount = op_type_count;
428 return final_data;
429}
