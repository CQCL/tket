1668 bool Circuit::rename_units(const std::map<UnitA, UnitB> &qm) {
1669  // Can only work for Unit classes
1670  static_assert(std::is_base_of<UnitID, UnitA>::value);
1671  static_assert(std::is_base_of<UnitID, UnitB>::value);
1672  // Unit types must be related, so cannot rename e.g. Bits to Qubits
1673  static_assert(
1674  std::is_base_of<UnitA, UnitB>::value ||
1675  std::is_base_of<UnitB, UnitA>::value);
1676  std::map<UnitID, BoundaryElement> new_elems;
1677  bool modified = false;
1678  std::map<Bit, Bit> bm;
1679  for (const std::pair<const UnitA, UnitB> &pair : qm) {
1680  boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1681  if (found == boundary.get<TagID>().end()) {
1682  std::stringstream ss;
1683  ss << "unit " << pair.first.repr() << " not found in circuit";
1684  tket_log()->warn(ss.str());
1685  continue;
1686  }
1687 
1688  opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1689  if (target_reg_info) {
1690  if (target_reg_info.value().first != found->type())
1691  throw CircuitInvalidity(
1692  "Incompatible registers: " + pair.first.reg_name() + " and " +
1693  pair.second.reg_name());
1694  if (target_reg_info.value().second != pair.second.reg_dim())
1695  throw CircuitInvalidity(
1696  "Existing register " + pair.second.reg_name() +
1697  " cannot support id: " + pair.second.repr());
1698  }
1699  std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1700  new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1701  if (!inserted.second)
1702  throw CircuitInvalidity(
1703  "Mapping two units to the same id: " + pair.second.repr());
1704  modified = true;
1705  boundary.erase(found);
1706  if (pair.first.type() == UnitType::Bit) {
1707  bm.insert({Bit(pair.first), Bit(pair.second)});
1708  }
1709  }
1710  for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1711  std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1712  if (!added.second)
1713  throw CircuitInvalidity(
1714  "Unit already exists in circuit: " + pair.first.repr());
1715  TKET_ASSERT(modified);
1716  }
1717 
1718  // For every ClassicalExpBox, update its logic expressions
1719  if (!bm.empty()) {
1720  BGL_FORALL_VERTICES(v, dag, DAG) {
1721  Op_ptr op = get_Op_ptr_from_Vertex(v);
1722  if (op->get_type() == OpType::ClassicalExpBox) {
1723  const ClassicalExpBoxBase &cbox =
1724  static_cast<const ClassicalExpBoxBase &>(*op);
1725  // rename_units is marked as const to get around the Op_ptr
1726  // cast, but it can still mutate a python object
1727  modified |= cbox.rename_units(bm);
1728  }
1729  }
1730  }
1731 
1732  return modified;
1733 }
