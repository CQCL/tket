814 template <class ID>
815 Vertex add_conditional_gate(
816 OpType type, const std::vector<Expr> &params, const std::vector<ID> &args,
817 const std::vector<ID> &bits, unsigned value,
818 std::optional<std::string> opgroup = std::nullopt) {
819 if (is_metaop_type(type)) {
820 throw CircuitInvalidity("Cannot add a conditional metaop.");
821 }
822 if (is_barrier_type(type)) {
823 throw CircuitInvalidity(
824 "Please use 'add_conditional_barrier' to add a conditional barrier "
825 "gate.");
826 }
827 Op_ptr cond = std::make_shared<Conditional>(
828 get_op_ptr(type, params, (unsigned)args.size()), (unsigned)bits.size(),
829 value);
830 std::vector<ID> new_args = bits;
831 new_args.insert(new_args.end(), args.begin(), args.end());
832 return add_op(cond, new_args, opgroup);
833 }

1650template <typename UnitA, typename UnitB>
1651bool Circuit::rename_units(const std::map<UnitA, UnitB> &qm) {
1652 // Can only work for Unit classes
1653 static_assert(std::is_base_of<UnitID, UnitA>::value);
1654 static_assert(std::is_base_of<UnitID, UnitB>::value);
1655 // Unit types must be related, so cannot rename e.g. Bits to Qubits
1656 static_assert(
1657 std::is_base_of<UnitA, UnitB>::value ||
1658 std::is_base_of<UnitB, UnitA>::value);
1659 std::map<UnitID, BoundaryElement> new_elems;
1660 bool modified = false;
1661 std::map<Bit, Bit> bm;
1662 for (const std::pair<const UnitA, UnitB> &pair : qm) {
1663 boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1664 if (found == boundary.get<TagID>().end()) {
1665 std::stringstream ss;
1666 ss << "unit " << pair.first.repr() << " not found in circuit";
1667 tket_log()->warn(ss.str());
1668 continue;
1669 }
1670
1671 opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1672 if (target_reg_info) {
1673 if (target_reg_info.value().first != found->type())
1674 throw CircuitInvalidity(
1675 "Incompatible registers: " + pair.first.reg_name() + " and " +
1676 pair.second.reg_name());
1677 if (target_reg_info.value().second != pair.second.reg_dim())
1678 throw CircuitInvalidity(
1679 "Existing register " + pair.second.reg_name() +
1680 " cannot support id: " + pair.second.repr());
1681 }
1682 std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1683 new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1684 if (!inserted.second)
1685 throw CircuitInvalidity(
1686 "Mapping two units to the same id: " + pair.second.repr());
1687 modified = true;
1688 boundary.erase(found);
1689 if (pair.first.type() == UnitType::Bit) {
1690 bm.insert({Bit(pair.first), Bit(pair.second)});
1691 }
1692 }
1693 for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1694 std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1695 if (!added.second)
1696 throw CircuitInvalidity(
1697 "Unit already exists in circuit: " + pair.first.repr());
1698 TKET_ASSERT(modified);
1699 }
1700 return modified;
1701}

1709Vertex Circuit::add_op(
1710 const Op_ptr &op, const std::vector<ID> &args,
1711 std::optional<std::string> opgroup) {
1712 static_assert(std::is_base_of<UnitID, ID>::value);
1713 op_signature_t sig = op->get_signature();
1714
1715 // check if there are wasm or rng wires in the signature
1716 unsigned count_wasm_sig = 0;
1717 unsigned count_rng_sig = 0;
1718 for (EdgeType e : sig) {
1719 if (e == EdgeType::WASM) {
1720 ++count_wasm_sig;
1721 } else if (e == EdgeType::RNG) {
1722 ++count_rng_sig;
1723 }
1724 }
1725
1726 if (sig.size() != args.size()) {
1727 throw CircuitInvalidity(
1728 std::to_string(args.size()) + " args provided, but " + op->get_name() +
1729 " requires " + std::to_string(sig.size()));
1730 }
1731
1732 if (opgroup) {
1733 auto opgroupsig = opgroupsigs.find(opgroup.value());
1734 if (opgroupsig != opgroupsigs.end()) {
1735 if (opgroupsig->second != sig) {
1736 throw CircuitInvalidity("Mismatched signature for operation group");
1737 }
1738 } else {
1739 opgroupsigs[opgroup.value()] = sig;
1740 }
1741 }
1742
1743 add_wasm_register(count_wasm_sig);
1744 add_rng_register(count_rng_sig);
1745
1746 unit_set_t write_arg_set;
1747 EdgeVec preds;
1748 for (unsigned i = 0; i < args.size(); ++i) {
1749 const UnitID &arg = args[i];
1750 if (sig[i] != EdgeType::Boolean) {
1751 if (write_arg_set.find(arg) != write_arg_set.end())
1752 throw CircuitInvalidity(
1753 "Multiple operation arguments reference " + arg.repr());
1754 write_arg_set.insert(arg);
1755 }
1756 Vertex out_vert = get_out(arg);
1757 Edge pred_out_e = get_nth_in_edge(out_vert, 0);
1758 preds.push_back(pred_out_e);
1759 }
1760
1761 Vertex new_v;
1762 OpType optype = op->get_type();
1763 if (optype == OpType::CnRy && args.size() == 1) {
1764 new_v = add_vertex(
1765 get_op_ptr(OpType::Ry, as_gate_ptr(op)->get_params()), opgroup);
1766 } else if (optype == OpType::CnRx && args.size() == 1) {
1767 new_v = add_vertex(
1768 get_op_ptr(OpType::Rx, as_gate_ptr(op)->get_params()), opgroup);
1769 } else if (optype == OpType::CnRz && args.size() == 1) {
1770 new_v = add_vertex(
1771 get_op_ptr(OpType::Rz, as_gate_ptr(op)->get_params()), opgroup);
1772 } else if (optype == OpType::CnX && args.size() == 1) {
1773 new_v = add_vertex(get_op_ptr(OpType::X), opgroup);
1774 } else if (optype == OpType::CnZ && args.size() == 1) {
1775 new_v = add_vertex(get_op_ptr(OpType::Z), opgroup);
1776 } else if (optype == OpType::CnY && args.size() == 1) {
1777 new_v = add_vertex(get_op_ptr(OpType::Y), opgroup);
1778 } else {
1779 new_v = add_vertex(op, opgroup);
1780 }
1781 rewire(new_v, preds, sig);
1782 return new_v;
1783}
