843 template <class ID>
844 Vertex add_conditional_gate(
845 OpType type, const std::vector<Expr> &params, const std::vector<ID> &args,
846 const std::vector<ID> &bits, unsigned value,
847 std::optional<std::string> opgroup = std::nullopt) {
848 if (is_metaop_type(type)) {
849 throw CircuitInvalidity("Cannot add a conditional metaop.");
850 }
851 if (is_barrier_type(type)) {
852 throw CircuitInvalidity(
853 "Please use 'add_conditional_barrier' to add a conditional barrier "
854 "gate.");
855 }
856 Op_ptr cond = std::make_shared<Conditional>(
857 get_op_ptr(type, params, (unsigned)args.size()), (unsigned)bits.size(),
858 value);
859 std::vector<ID> new_args = bits;
860 new_args.insert(new_args.end(), args.begin(), args.end());
861 return add_op(cond, new_args, opgroup);
862 }

1662template <typename UnitA, typename UnitB>
1663bool Circuit::rename_units(
1664 const std::map<UnitA, UnitB> &qm, bool relabel_classicalexpbox) {
1665 // Can only work for Unit classes
1666 static_assert(std::is_base_of<UnitID, UnitA>::value);
1667 static_assert(std::is_base_of<UnitID, UnitB>::value);
1668 // Unit types must be related, so cannot rename e.g. Bits to Qubits
1669 static_assert(
1670 std::is_base_of<UnitA, UnitB>::value ||
1671 std::is_base_of<UnitB, UnitA>::value);
1672 std::map<UnitID, BoundaryElement> new_elems;
1673 bool modified = false;
1674 std::map<Bit, Bit> bm;
1675 for (const std::pair<const UnitA, UnitB> &pair : qm) {
1676 boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1677 if (found == boundary.get<TagID>().end()) {
1678 std::stringstream ss;
1679 ss << "unit " << pair.first.repr() << " not found in circuit";
1680 tket_log()->warn(ss.str());
1681 continue;
1682 }
1683
1684 opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1685 if (target_reg_info) {
1686 if (target_reg_info.value().first != found->type())
1687 throw CircuitInvalidity(
1688 "Incompatible registers: " + pair.first.reg_name() + " and " +
1689 pair.second.reg_name());
1690 if (target_reg_info.value().second != pair.second.reg_dim())
1691 throw CircuitInvalidity(
1692 "Existing register " + pair.second.reg_name() +
1693 " cannot support id: " + pair.second.repr());
1694 }
1695 std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1696 new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1697 if (!inserted.second)
1698 throw CircuitInvalidity(
1699 "Mapping two units to the same id: " + pair.second.repr());
1700 modified = true;
1701 boundary.erase(found);
1702 if (pair.first.type() == UnitType::Bit) {
1703 bm.insert({Bit(pair.first), Bit(pair.second)});
1704 }
1705 }
1706 for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1707 std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1708 if (!added.second)
1709 throw CircuitInvalidity(
1710 "Unit already exists in circuit: " + pair.first.repr());
1711 TKET_ASSERT(modified);
1712 }
1713 // For every ClassicalExpBox, update its logic expressions
1714 if (!bm.empty() && relabel_classicalexpbox) {
1715 BGL_FORALL_VERTICES(v, dag, DAG) {
1716 Op_ptr op = get_Op_ptr_from_Vertex(v);
1717 if (op->get_type() == OpType::ClassicalExpBox) {
1718 const ClassicalExpBoxBase &cbox =
1719 static_cast<const ClassicalExpBoxBase &>(*op);
1720 // rename_units is marked as const to get around the Op_ptr
1721 modified |= cbox.rename_units(bm);
1722 }
1723 }
1724 }
1725 return modified;
1726}

1755 if (opgroupsig != opgroupsigs.end()) {
1756 if (opgroupsig->second != sig) {
1757 throw CircuitInvalidity("Mismatched signature for operation group");
1758 }
1759 } else {
1760 opgroupsigs[opgroup.value()] = sig;
1761 }
1762 }
1763
1764 add_wasm_register(count_wasm_sig);
1765
1766 unit_set_t write_arg_set;
1767 EdgeVec preds;
1768 for (unsigned i = 0; i < args.size(); ++i) {
1769 const UnitID &arg = args[i];
1770 if (sig[i] != EdgeType::Boolean) {
1771 if (write_arg_set.find(arg) != write_arg_set.end())
1772 throw CircuitInvalidity(
1773 "Multiple operation arguments reference " + arg.repr());

1778 preds.push_back(pred_out_e);
