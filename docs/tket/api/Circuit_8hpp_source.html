811 template <class ID>
812 Vertex add_conditional_gate(
813 OpType type, const std::vector<Expr> &params, const std::vector<ID> &args,
814 const std::vector<ID> &bits, unsigned value,
815 std::optional<std::string> opgroup = std::nullopt) {
816 if (is_metaop_type(type)) {

829 return add_op(cond, new_args, opgroup);

1642template <typename UnitA, typename UnitB>
1643bool Circuit::rename_units(const std::map<UnitA, UnitB> &qm) {
1644 // Can only work for Unit classes
1645 static_assert(std::is_base_of<UnitID, UnitA>::value);
1646 static_assert(std::is_base_of<UnitID, UnitB>::value);
1647 // Unit types must be related, so cannot rename e.g. Bits to Qubits
1648 static_assert(
1649 std::is_base_of<UnitA, UnitB>::value ||
1650 std::is_base_of<UnitB, UnitA>::value);
1651 std::map<UnitID, BoundaryElement> new_elems;
1652 bool modified = false;
1653 std::map<Bit, Bit> bm;
1654 for (const std::pair<const UnitA, UnitB> &pair : qm) {
1655 boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1656 if (found == boundary.get<TagID>().end()) {
1657 std::stringstream ss;
1658 ss << "unit " << pair.first.repr() << " not found in circuit";
1659 tket_log()->warn(ss.str());
1660 continue;
1661 }
1662
1663 opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1664 if (target_reg_info) {
1665 if (target_reg_info.value().first != found->type())
1666 throw CircuitInvalidity(
1667 "Incompatible registers: " + pair.first.reg_name() + " and " +
1668 pair.second.reg_name());
1669 if (target_reg_info.value().second != pair.second.reg_dim())
1670 throw CircuitInvalidity(
1671 "Existing register " + pair.second.reg_name() +
1672 " cannot support id: " + pair.second.repr());
1673 }
1674 std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1675 new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1676 if (!inserted.second)
1677 throw CircuitInvalidity(
1678 "Mapping two units to the same id: " + pair.second.repr());
1679 modified = true;
1680 boundary.erase(found);
1681 if (pair.first.type() == UnitType::Bit) {
1682 bm.insert({Bit(pair.first), Bit(pair.second)});
1683 }
1684 }
1685 for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1686 std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1687 if (!added.second)
1688 throw CircuitInvalidity(
1689 "Unit already exists in circuit: " + pair.first.repr());
1690 TKET_ASSERT(modified);
1691 }
1692 return modified;
1693}

1700template <class ID>
1701Vertex Circuit::add_op(
1702 const Op_ptr &op, const std::vector<ID> &args,
1703 std::optional<std::string> opgroup) {
1704 static_assert(std::is_base_of<UnitID, ID>::value);
1705 op_signature_t sig = op->get_signature();
1706
1707 // check if there is wasm in the signature
1708 unsigned count_wasm_sig = 0;
1709 for (EdgeType e : sig) {
1710 if (e == EdgeType::WASM) {
1711 ++count_wasm_sig;
1712 }
1713 }
1714 if (sig.size() != args.size()) {
1715 throw CircuitInvalidity(
1716 std::to_string(args.size()) + " args provided, but " + op->get_name() +
1717 " requires " + std::to_string(sig.size()));
1718 }
1719
1720 if (opgroup) {
1721 auto opgroupsig = opgroupsigs.find(opgroup.value());
1722 if (opgroupsig != opgroupsigs.end()) {
1723 if (opgroupsig->second != sig) {
1724 throw CircuitInvalidity("Mismatched signature for operation group");
1725 }
1726 } else {
1727 opgroupsigs[opgroup.value()] = sig;
1728 }
1729 }
1730
1731 add_wasm_register(count_wasm_sig);
1732
1733 unit_set_t write_arg_set;
1734 EdgeVec preds;
1735 for (unsigned i = 0; i < args.size(); ++i) {
1736 const UnitID &arg = args[i];
1737 if (sig[i] != EdgeType::Boolean) {
1738 if (write_arg_set.find(arg) != write_arg_set.end())
1739 throw CircuitInvalidity(
1740 "Multiple operation arguments reference " + arg.repr());
1741 write_arg_set.insert(arg);
1742 }
1743 Vertex out_vert = get_out(arg);
1744 Edge pred_out_e = get_nth_in_edge(out_vert, 0);
1745 preds.push_back(pred_out_e);
1746 }
1747
1748 Vertex new_v;
1749 OpType optype = op->get_type();
1750 if (optype == OpType::CnRy && args.size() == 1) {
1751 new_v = add_vertex(
1752 get_op_ptr(OpType::Ry, as_gate_ptr(op)->get_params()), opgroup);
1753 } else if (optype == OpType::CnRx && args.size() == 1) {
1754 new_v = add_vertex(
1755 get_op_ptr(OpType::Rx, as_gate_ptr(op)->get_params()), opgroup);
1756 } else if (optype == OpType::CnRz && args.size() == 1) {
1757 new_v = add_vertex(
1758 get_op_ptr(OpType::Rz, as_gate_ptr(op)->get_params()), opgroup);
1759 } else if (optype == OpType::CnX && args.size() == 1) {
1760 new_v = add_vertex(get_op_ptr(OpType::X), opgroup);
1761 } else if (optype == OpType::CnZ && args.size() == 1) {
1762 new_v = add_vertex(get_op_ptr(OpType::Z), opgroup);
1763 } else if (optype == OpType::CnY && args.size() == 1) {
1764 new_v = add_vertex(get_op_ptr(OpType::Y), opgroup);
1765 } else {
1766 new_v = add_vertex(op, opgroup);
1767 }
1768 rewire(new_v, preds, sig);
1769 return new_v;
1770}
