1636 bool Circuit::rename_units(const std::map<UnitA, UnitB> &qm) {
1637  // Can only work for Unit classes
1638  static_assert(std::is_base_of<UnitID, UnitA>::value);
1639  static_assert(std::is_base_of<UnitID, UnitB>::value);
1640  // Unit types must be related, so cannot rename e.g. Bits to Qubits
1641  static_assert(
1642  std::is_base_of<UnitA, UnitB>::value ||
1643  std::is_base_of<UnitB, UnitA>::value);
1644  std::map<UnitID, BoundaryElement> new_elems;
1645  bool modified = false;
1646  std::map<Bit, Bit> bm;
1647  for (const std::pair<const UnitA, UnitB> &pair : qm) {
1648  boundary_t::iterator found = boundary.get<TagID>().find(pair.first);
1649  if (found == boundary.get<TagID>().end()) {
1650  std::stringstream ss;
1651  ss << "unit " << pair.first.repr() << " not found in circuit";
1652  tket_log()->warn(ss.str());
1653  continue;
1654  }
1655 
1656  opt_reg_info_t target_reg_info = get_reg_info(pair.second.reg_name());
1657  if (target_reg_info) {
1658  if (target_reg_info.value().first != found->type())
1659  throw CircuitInvalidity(
1660  "Incompatible registers: " + pair.first.reg_name() + " and " +
1661  pair.second.reg_name());
1662  if (target_reg_info.value().second != pair.second.reg_dim())
1663  throw CircuitInvalidity(
1664  "Existing register " + pair.second.reg_name() +
1665  " cannot support id: " + pair.second.repr());
1666  }
1667  std::pair<std::map<UnitID, BoundaryElement>::iterator, bool> inserted =
1668  new_elems.insert({pair.second, {pair.second, found->in_, found->out_}});
1669  if (!inserted.second)
1670  throw CircuitInvalidity(
1671  "Mapping two units to the same id: " + pair.second.repr());
1672  modified = true;
1673  boundary.erase(found);
1674  if (pair.first.type() == UnitType::Bit) {
1675  bm.insert({Bit(pair.first), Bit(pair.second)});
1676  }
1677  }
1678  for (const std::pair<const UnitID, BoundaryElement> &pair : new_elems) {
1679  std::pair<boundary_t::iterator, bool> added = boundary.insert(pair.second);
1680  if (!added.second)
1681  throw CircuitInvalidity(
1682  "Unit already exists in circuit: " + pair.first.repr());
1683  TKET_ASSERT(modified);
1684  }
1685 
1686  // For every ClassicalExpBox, update its logic expressions
1687  if (!bm.empty()) {
1688  BGL_FORALL_VERTICES(v, dag, DAG) {
1689  Op_ptr op = get_Op_ptr_from_Vertex(v);
1690  if (op->get_type() == OpType::ClassicalExpBox) {
1691  const ClassicalExpBoxBase &cbox =
1692  static_cast<const ClassicalExpBoxBase &>(*op);
1693  // rename_units is marked as const to get around the Op_ptr
1694  // cast, but it can still mutate a python object
1695  modified |= cbox.rename_units(bm);
1696  }
1697  }
1698  }
1699 
1700  return modified;
1701 }
