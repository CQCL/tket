202ClExpr::ClExpr(ClOp op, std::vector<ClExprArg> args)
203 : op(op), args(args), all_bit_vars(), all_reg_vars() {
204 for (const ClExprArg& input : args) {
205 if (std::holds_alternative<ClExprTerm>(input)) {
206 ClExprTerm basic_input = std::get<ClExprTerm>(input);
207 if (std::holds_alternative<ClExprVar>(basic_input)) {
208 ClExprVar var = std::get<ClExprVar>(basic_input);
209 if (std::holds_alternative<ClBitVar>(var)) {
210 ClBitVar bit_var = std::get<ClBitVar>(var);
211 all_bit_vars.insert(bit_var.index);
212 } else {
213 ClRegVar reg_var = std::get<ClRegVar>(var);
214 all_reg_vars.insert(reg_var.index);
215 }
216 }
217 } else {
218 ClExpr expr = std::get<ClExpr>(input);
219 std::set<unsigned> expr_bit_vars = expr.all_bit_variables();
220 std::set<unsigned> expr_reg_vars = expr.all_reg_variables();
221 all_bit_vars.insert(expr_bit_vars.begin(), expr_bit_vars.end());
222 all_reg_vars.insert(expr_reg_vars.begin(), expr_reg_vars.end());
223 }
224 }

268WiredClExpr::WiredClExpr(
269 const ClExpr& expr, const std::map<unsigned, unsigned>& bit_posn,
270 const std::map<unsigned, std::vector<unsigned>>& reg_posn,
271 const std::vector<unsigned> output_posn)
272 : expr(expr),
273 bit_posn(bit_posn),
274 reg_posn(reg_posn),
275 output_posn(output_posn) {
276 std::set<unsigned> b;
277 std::set<unsigned> r;
278 std::set<unsigned> posns;
279 for (const auto& pair : bit_posn) {
280 b.insert(pair.first);
281 unsigned bit_pos = pair.second;
282 if (posns.contains(bit_pos)) {
283 throw ClExprWiringError("Invalid maps constructing WiredClExpr");
284 }
285 posns.insert(bit_pos);
286 all_bit_posns.insert(bit_pos);
287 }
288 for (const auto& pair : reg_posn) {
289 r.insert(pair.first);
290 for (unsigned bit_pos : pair.second) {
291 if (posns.contains(bit_pos)) {
292 throw ClExprWiringError("Invalid maps constructing WiredClExpr");
293 }
294 posns.insert(bit_pos);
295 }
296 all_reg_posns.insert(pair.second);
297 }
298 total_n_bits = posns.size();
299 for (const unsigned& posn : output_posn) {
300 if (!posns.contains(posn)) {
301 total_n_bits++;
302 }
303 }
304 if (output_posn.size() == 1) {
305 // It mustn't be one of an input register of size > 1
306 unsigned i = output_posn[0];
307 for (const std::vector<unsigned>& reg : all_reg_posns) {
308 if (reg.size() > 1 && std::any_of(
309 reg.begin(), reg.end(),
310 [&i](const unsigned& j) { return i == j; })) {
311 throw ClExprWiringError(
312 "Output bit contained in a larger input register");
313 }
314 }
315 } else {
316 // It must either be disjoint from everything or match one of the registers
317 if (std::any_of(
318 output_posn.begin(), output_posn.end(),
319 [&posns](const unsigned& j) { return posns.contains(j); })) {
320 if (!std::any_of(
321 all_reg_posns.begin(), all_reg_posns.end(),
322 [&output_posn](const std::vector<unsigned>& reg) {
323 return output_posn == reg;
324 })) {
325 throw ClExprWiringError("Output register inconsistent with inputs");
326 }
327 }
328 }
329 if (b != expr.all_bit_variables()) {
330 throw ClExprWiringError(
331 "Mismatch of bit variables constructing WiredClExpr");
332 }
333 if (r != expr.all_reg_variables()) {
334 throw ClExprWiringError(
335 "Mismatch of register variables constructing WiredClExpr");
336 }
337}

344std::ostream& operator<<(std::ostream& os, const WiredClExpr& expr) {
345 os << expr.expr << " [";
346 unsigned n_vars = expr.bit_posn.size() + expr.reg_posn.size();
347 unsigned i = 0;
348 for (const std::pair<unsigned, unsigned> pair : expr.bit_posn) {
349 os << "b" << pair.first << ":" << pair.second;
350 i++;
351 if (i < n_vars) {
352 os << ", ";
353 }
354 }
355 for (const std::pair<unsigned, std::vector<unsigned>> pair : expr.reg_posn) {
356 os << "r" << pair.first << ":(";
357 unsigned reg_size = pair.second.size();
358 for (unsigned j = 0; j < reg_size; j++) {
359 os << pair.second[j];
360 if (j + 1 < reg_size) {
361 os << ",";
362 }
363 }
364 os << ")";
365 i++;
366 if (i < n_vars) {
367 os << ", ";
368 }
369 }
370 os << " --> (";
371 unsigned n_outs = expr.output_posn.size();
372 for (unsigned i = 0; i < n_outs; i++) {
373 os << expr.output_posn[i];
374 if (i + 1 < n_outs) {
375 os << ",";
376 }
377 }
378 os << ")]";
379 return os;
