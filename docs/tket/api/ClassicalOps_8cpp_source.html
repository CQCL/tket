25static uint64_t u64_from_boolvec(const std::vector<bool> &x) {
26 unsigned n = x.size();
27 if (n > 64) {
28 throw std::domain_error("Vector of bool exceeds maximum size (64)");
29 }
30 uint64_t X = 0;
31 for (unsigned i = 0; i < n; i++) {
32 if (x[i]) X |= (1u << i);
33 }
34 return X;
35}

37static nlohmann::json classical_to_json(const Op_ptr &op, const OpType &type) {
38 nlohmann::json j_class = nlohmann::json::object();
39 switch (type) {
40 case OpType::MultiBit: {
41 const auto &multibit = static_cast<const MultiBitOp &>(*op);
42 j_class["op"] = multibit.get_op();
43 j_class["n"] = multibit.get_n();
44 return j_class;
45 }
46 case OpType::RangePredicate: {
47 const auto &rangop = static_cast<const RangePredicateOp &>(*op);
48 j_class["lower"] = rangop.lower();
49 j_class["upper"] = rangop.upper();
50 j_class["n_i"] = rangop.get_n_i();
51 return j_class;
52 }
53 case OpType::ExplicitModifier: {
54 const auto &expmodop = static_cast<const ExplicitModifierOp &>(*op);
55 j_class["n_i"] = expmodop.get_n_i();
56 j_class["values"] = expmodop.get_values();
57 j_class["name"] = expmodop.get_name();
58 return j_class;
59 }
60 case OpType::ExplicitPredicate: {
61 const auto &exppredop = static_cast<const ExplicitPredicateOp &>(*op);
62 j_class["n_i"] = exppredop.get_n_i();
63 j_class["values"] = exppredop.get_values();
64 j_class["name"] = exppredop.get_name();
65 return j_class;
66 }
67 case OpType::ClassicalTransform: {
68 const auto &classtop = static_cast<const ClassicalTransformOp &>(*op);
69 j_class["n_io"] = classtop.get_n_io();
70 j_class["values"] = classtop.get_values();
71 j_class["name"] = classtop.get_name();
72 return j_class;
73 }
74 case OpType::SetBits: {
75 const auto &setop = static_cast<const SetBitsOp &>(*op);
76 j_class["values"] = setop.get_values();
77 return j_class;
78 }
79 case OpType::CopyBits: {
80 const auto &cop = static_cast<const CopyBitsOp &>(*op);
81 j_class["n_i"] = cop.get_n_i();
82 return j_class;
83 }
84 default:
85 throw JsonError(
86 "Classical op with type " + optypeinfo().at(type).name +
87 " cannot be serialized.");
88 }
89}

91static std::shared_ptr<ClassicalEvalOp> classical_from_json(
92 const nlohmann::json &j_class, const OpType &type) {
93 switch (type) {
94 case OpType::MultiBit:
95 return std::make_shared<MultiBitOp>(
96 classical_from_json(
97 j_class.at("op").at("classical"),
98 j_class.at("op").at("type").get<OpType>()),
99 j_class.at("n").get<unsigned>());
100 case OpType::RangePredicate:
101 return std::make_shared<RangePredicateOp>(
102 j_class.at("n_i").get<unsigned>(),
103 j_class.at("lower").get<uint64_t>(),
104 j_class.at("upper").get<uint64_t>());
105 case OpType::CopyBits:
106 return std::make_shared<CopyBitsOp>(j_class.at("n_i").get<unsigned>());
107 case OpType::SetBits:
108 return std::make_shared<SetBitsOp>(
109 j_class.at("values").get<std::vector<bool>>());
110 case OpType::ExplicitModifier:
111 return std::make_shared<ExplicitModifierOp>(
112 j_class.at("n_i").get<unsigned>(),
113 j_class.at("values").get<std::vector<bool>>(),
114 j_class.at("name").get<std::string>());
115 case OpType::ExplicitPredicate:
116 return std::make_shared<ExplicitPredicateOp>(
117 j_class.at("n_i").get<unsigned>(),
118 j_class.at("values").get<std::vector<bool>>(),
119 j_class.at("name").get<std::string>());
120 case OpType::ClassicalTransform:
121 return std::make_shared<ClassicalTransformOp>(
122 j_class.at("n_io").get<unsigned>(),
123 j_class.at("values").get<std::vector<uint64_t>>(),
124 j_class.at("name").get<std::string>());
125 default:
126 throw JsonError(
127 "Classical op with name " + j_class.at("name").get<std::string>() +
128 " cannot be deserialized.");
129 }
130}

258bool ClassicalEvalOp::is_equal(const Op &op_other) const {
259 const ClassicalEvalOp &other =
260 dynamic_cast<const ClassicalEvalOp &>(op_other);
261
262 if (n_i_ != other.get_n_i()) return false;
263 if (n_io_ != other.get_n_io()) return false;
264 if (n_o_ != other.get_n_o()) return false;
265 unsigned N = n_i_ + n_io_;
266 uint64_t xlim = 1u << N;
267 std::vector<bool> v(N);
268 for (uint64_t x = 0; x < xlim; x++) {
269 for (unsigned i = 0; i < N; i++) {
270 v[i] = (x >> i) & 1;
271 }
272 if (eval(v) != other.eval(v)) return false;
273 }
274 return true;
275}

286std::vector<bool> ClassicalTransformOp::eval(const std::vector<bool> &x) const {
287 if (x.size() != n_io_) {
288 throw std::domain_error("Incorrect input size");
289 }
290 uint64_t val = values_[u64_from_boolvec(x)];
291 std::vector<bool> y(n_io_);
292 for (unsigned j = 0; j < n_io_; j++) {
293 y[j] = (val >> j) & 1;
294 }
295 return y;
296}

298WASMOp::WASMOp(
299 unsigned _n, unsigned _ww_n, std::vector<unsigned> _width_i_parameter,
300 std::vector<unsigned> _width_o_parameter, const std::string &_func_name,
301 const std::string &_wasm_file_uid)
302 : ClassicalOp(
303 OpType::WASM,
304 std::accumulate(
305 _width_i_parameter.begin(), _width_i_parameter.end(), 0),
306 0,
307 std::accumulate(
308 _width_o_parameter.begin(), _width_o_parameter.end(), 0),
309 "WASM"),
310 n_(_n),
311 ww_n_(_ww_n),
312 n_i32_(_width_i_parameter.size() + _width_o_parameter.size()),
313 width_i_parameter_(_width_i_parameter),
314 width_o_parameter_(_width_o_parameter),
315 func_name_(_func_name),
316 wasm_file_uid_(_wasm_file_uid) {
317 unsigned sum_of_i32 =
318 std::accumulate(width_i_parameter_.begin(), width_i_parameter_.end(), 0) +
319 std::accumulate(width_o_parameter_.begin(), width_o_parameter_.end(), 0);
320
321 if (sum_of_i32 != n_) {
322 throw std::domain_error("number of given args unexpected");
323 }
324
325 // add wasm edge to signature of this op.
326 for (unsigned i = 0; i < ww_n_; ++i) {
327 sig_.push_back(EdgeType::WASM);
328 }
329}

331bool WASMOp::is_equal(const Op &other) const {
332 if (other.get_type() != OpType::WASM) {
333 return false;
334 }
335
336 const WASMOp &other_wasm = dynamic_cast<const WASMOp &>(other);
337 if (other_wasm.get_n() != n_) return false;
338 if (other_wasm.get_n_i32() != n_i32_) return false;
339 if (other_wasm.get_width_i_parameter() != width_i_parameter_) return false;
340 if (other_wasm.get_width_o_parameter() != width_o_parameter_) return false;
341 if (other_wasm.get_func_name() != func_name_) return false;
342 if (other_wasm.get_wasm_file_uid() != wasm_file_uid_) return false;
343 return true;
344}

428MultiBitOp::MultiBitOp(std::shared_ptr<const ClassicalEvalOp> op, unsigned n)
429 : ClassicalEvalOp(
430 OpType::MultiBit, n * op->get_n_i(), n * op->get_n_io(),
431 n * op->get_n_o(), op->get_name()),
432 op_(op),
433 n_(n) {
434 op_signature_t op_sig = op->get_signature();
435 const std::size_t op_sig_size = op_sig.size();
436 sig_.clear();
437 sig_.reserve(n_ * op_sig.size());
438 for (unsigned i = 0; i < n_; i++) {
439 std::copy_n(op_sig.begin(), op_sig_size, std::back_inserter(sig_));
440 }
441}

449std::vector<bool> MultiBitOp::eval(const std::vector<bool> &x) const {
450 if (x.size() != n_i_ + n_io_) {
451 throw std::domain_error("Incorrect input size");
452 }
453 unsigned n_op_inputs = op_->get_n_i() + op_->get_n_io();
454 unsigned n_op_outputs = op_->get_n_io() + op_->get_n_o();
455 std::vector<bool> y(n_io_ + n_o_);
456 for (unsigned i = 0; i < n_; i++) {
457 std::vector<bool> x_i(n_op_inputs);
458 for (unsigned j = 0; j < n_op_inputs; j++) {
459 x_i[j] = x[n_op_inputs * i + j];
460 }
461 std::vector<bool> y_i = op_->eval(x_i);
462 for (unsigned j = 0; j < n_op_outputs; j++) {
463 y[n_op_outputs * i + j] = y_i[j];
464 }
465 }
466 return y;
467}
