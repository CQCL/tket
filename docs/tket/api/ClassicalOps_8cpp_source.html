24 static uint32_t u32_from_boolvec(const std::vector<bool> &x) {
25  unsigned n = x.size();
26  if (n > 32) {
27  throw std::domain_error("Vector of bool exceeds maximum size (32)");
28  }
29  uint32_t X = 0;
30  for (unsigned i = 0; i < n; i++) {
31  if (x[i]) X |= (1u << i);
32  }
33  return X;
34 }

36 static nlohmann::json classical_to_json(const Op_ptr &op, const OpType &type) {
37  nlohmann::json j_class;
38  switch (type) {
39  case OpType::MultiBit: {
40  const auto &multibit = static_cast<const MultiBitOp &>(*op);
41  j_class["op"] = multibit.get_op();
42  j_class["n"] = multibit.get_n();
43  return j_class;
44  }
45  case OpType::RangePredicate: {
46  const auto &rangop = static_cast<const RangePredicateOp &>(*op);
47  j_class["lower"] = rangop.lower();
48  j_class["upper"] = rangop.upper();
49  j_class["n_i"] = rangop.get_n_i();
50  return j_class;
51  }
52  case OpType::ExplicitModifier: {
53  const auto &expmodop = static_cast<const ExplicitModifierOp &>(*op);
54  j_class["n_i"] = expmodop.get_n_i();
55  j_class["values"] = expmodop.get_values();
56  j_class["name"] = expmodop.get_name();
57  return j_class;
58  }
59  case OpType::ExplicitPredicate: {
60  const auto &exppredop = static_cast<const ExplicitPredicateOp &>(*op);
61  j_class["n_i"] = exppredop.get_n_i();
62  j_class["values"] = exppredop.get_values();
63  j_class["name"] = exppredop.get_name();
64  return j_class;
65  }
66  case OpType::ClassicalTransform: {
67  const auto &classtop = static_cast<const ClassicalTransformOp &>(*op);
68  j_class["n_io"] = classtop.get_n_io();
69  j_class["values"] = classtop.get_values();
70  j_class["name"] = classtop.get_name();
71  return j_class;
72  }
73  case OpType::SetBits: {
74  const auto &setop = static_cast<const SetBitsOp &>(*op);
75  j_class["values"] = setop.get_values();
76  return j_class;
77  }
78  case OpType::CopyBits: {
79  const auto &cop = static_cast<const CopyBitsOp &>(*op);
80  j_class["n_i"] = cop.get_n_i();
81  return j_class;
82  }
83  default:
84  throw JsonError(
85  "Classical op with type " + optypeinfo().at(type).name +
86  " cannot be serialized.");
87  }
88 }

90 static std::shared_ptr<ClassicalEvalOp> classical_from_json(
91  const nlohmann::json &j_class, const OpType &type) {
92  switch (type) {
93  case OpType::MultiBit:
94  return std::make_shared<MultiBitOp>(
95  classical_from_json(
96  j_class.at("op").at("classical"),
97  j_class.at("op").at("type").get<OpType>()),
98  j_class.at("n").get<unsigned>());
99  case OpType::RangePredicate:
100  return std::make_shared<RangePredicateOp>(
101  j_class.at("n_i").get<unsigned>(),
102  j_class.at("lower").get<unsigned>(),
103  j_class.at("upper").get<unsigned>());
104  case OpType::CopyBits:
105  return std::make_shared<CopyBitsOp>(j_class.at("n_i").get<unsigned>());
106  case OpType::SetBits:
107  return std::make_shared<SetBitsOp>(
108  j_class.at("values").get<std::vector<bool>>());
109  case OpType::ExplicitModifier:
110  return std::make_shared<ExplicitModifierOp>(
111  j_class.at("n_i").get<unsigned>(),
112  j_class.at("values").get<std::vector<bool>>(),
113  j_class.at("name").get<std::string>());
114  case OpType::ExplicitPredicate:
115  return std::make_shared<ExplicitPredicateOp>(
116  j_class.at("n_i").get<unsigned>(),
117  j_class.at("values").get<std::vector<bool>>(),
118  j_class.at("name").get<std::string>());
119  case OpType::ClassicalTransform:
120  return std::make_shared<ClassicalTransformOp>(
121  j_class.at("n_io").get<unsigned>(),
122  j_class.at("values").get<std::vector<uint32_t>>(),
123  j_class.at("name").get<std::string>());
124  default:
125  throw JsonError(
126  "Classical op with name " + j_class.at("name").get<std::string>() +
127  " cannot be deserialized.");
128  }
129 }

202 bool ClassicalEvalOp::is_equal(const Op &op_other) const {
203  const ClassicalEvalOp &other =
204  dynamic_cast<const ClassicalEvalOp &>(op_other);
205 
206  if (n_i_ != other.get_n_i()) return false;
207  if (n_io_ != other.get_n_io()) return false;
208  if (n_o_ != other.get_n_o()) return false;
209  unsigned N = n_i_ + n_io_;
210  uint32_t xlim = 1u << N;
211  std::vector<bool> v(N);
212  for (uint32_t x = 0; x < xlim; x++) {
213  for (unsigned i = 0; i < N; i++) {
214  v[i] = (x >> i) & 1;
215  }
216  if (eval(v) != other.eval(v)) return false;
217  }
218  return true;
219 }

230 std::vector<bool> ClassicalTransformOp::eval(const std::vector<bool> &x) const {
231  if (x.size() != n_io_) {
232  throw std::domain_error("Incorrect input size");
233  }
234  uint32_t val = values_[u32_from_boolvec(x)];
235  std::vector<bool> y(n_io_);
236  for (unsigned j = 0; j < n_io_; j++) {
237  y[j] = (val >> j) & 1;
238  }
239  return y;
240 }

242 WASMOp::WASMOp(
243  unsigned _n, unsigned _ww_n, std::vector<unsigned> _width_i_parameter,
244  std::vector<unsigned> _width_o_parameter, const std::string &_func_name,
245  const std::string &_wasm_file_uid)
246  : ClassicalOp(
247  OpType::WASM,
248  std::accumulate(
249  _width_i_parameter.begin(), _width_i_parameter.end(), 0),
250  0,
251  std::accumulate(
252  _width_o_parameter.begin(), _width_o_parameter.end(), 0),
253  "WASM"),
254  n_(_n),
255  ww_n_(_ww_n),
256  n_i32_(_width_i_parameter.size() + _width_o_parameter.size()),
257  width_i_parameter_(_width_i_parameter),
258  width_o_parameter_(_width_o_parameter),
259  func_name_(_func_name),
260  wasm_file_uid_(_wasm_file_uid) {
261  unsigned sum_of_i32 =
262  std::accumulate(width_i_parameter_.begin(), width_i_parameter_.end(), 0) +
263  std::accumulate(width_o_parameter_.begin(), width_o_parameter_.end(), 0);
264 
265  TKET_ASSERT(sum_of_i32 == n_);
266 
267  // add wasm edge to signature of this op.
268  for (unsigned i = 0; i < ww_n_; ++i) {
269  sig_.push_back(EdgeType::WASM);
270  }
271 }

391 std::vector<bool> MultiBitOp::eval(const std::vector<bool> &x) const {
392  if (x.size() != n_i_ + n_io_) {
393  throw std::domain_error("Incorrect input size");
394  }
395  unsigned n_op_inputs = op_->get_n_i() + op_->get_n_io();
396  unsigned n_op_outputs = op_->get_n_io() + op_->get_n_o();
397  std::vector<bool> y(n_io_ + n_o_);
398  for (unsigned i = 0; i < n_; i++) {
399  std::vector<bool> x_i(n_op_inputs);
400  for (unsigned j = 0; j < n_op_inputs; j++) {
401  x_i[j] = x[n_op_inputs * i + j];
402  }
403  std::vector<bool> y_i = op_->eval(x_i);
404  for (unsigned j = 0; j < n_op_outputs; j++) {
405  y[n_op_outputs * i + j] = y_i[j];
406  }
407  }
408  return y;
409 }
