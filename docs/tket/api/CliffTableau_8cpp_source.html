76 static QubitPauliTensor get_pauli(
77  const Qubit &qb, const boost::bimap<Qubit, unsigned> &qubits_,
78  const MatrixXb &pauli_x, const MatrixXb &pauli_z,
79  const VectorXb &pauli_phase) {
80  unsigned uqb = qubits_.left.at(qb);
81  Complex phase = 1.;
82  if (pauli_phase(uqb)) phase = -1.;
83  QubitPauliTensor res(phase);
84  for (boost::bimap<Qubit, unsigned>::const_iterator iter = qubits_.begin(),
85  iend = qubits_.end();
86  iter != iend; ++iter) {
87  unsigned origin = iter->right;
88  if (pauli_x(uqb, origin)) {
89  if (pauli_z(uqb, origin)) {
90  res = res * QubitPauliTensor(iter->left, Pauli::Y);
91  } else {
92  res = res * QubitPauliTensor(iter->left, Pauli::X);
93  }
94  } else if (pauli_z(uqb, origin)) {
95  res = res * QubitPauliTensor(iter->left, Pauli::Z);
96  }
97  }
98  return res;
99 }

119 void CliffTableau::row_mult(
120  const MatrixXb::RowXpr &xa, const MatrixXb::RowXpr &za, const bool &pa,
121  const MatrixXb::RowXpr &xb, const MatrixXb::RowXpr &zb, const bool &pb,
122  Complex phase, MatrixXb::RowXpr &xw, MatrixXb::RowXpr &zw, bool &pw) {
123  if (pa) phase *= -1;
124  if (pb) phase *= -1;
125  for (unsigned i = 0; i < size_; i++) {
126  std::pair<BoolPauli, Complex> res =
127  mult_lut.at({{xa(i), za(i)}, {xb(i), zb(i)}});
128  xw(i) = res.first.x;
129  zw(i) = res.first.z;
130  phase *= res.second;
131  }
132  pw = (phase == -1.);
133 }

135 void CliffTableau::col_mult(
136  const MatrixXb::ColXpr &a, const MatrixXb::ColXpr &b, bool flip,
137  MatrixXb::ColXpr &w, VectorXb &pw) {
138  for (unsigned i = 0; i < size_; i++) {
139  pw(i) = pw(i) ^ (a(i) && (b(i) ^ flip));
140  w(i) = a(i) ^ b(i);

182 void CliffTableau::apply_CX_at_front(unsigned control, unsigned target) {
183  for (unsigned i = 0; i < size_; i++) {
184  xpauli_phase(i) =
185  xpauli_phase(i) ^ (xpauli_x(i, control) && xpauli_z(i, target) &&
186  !(xpauli_x(i, target) ^ xpauli_z(i, control)));
187  xpauli_x(i, target) = xpauli_x(i, control) ^ xpauli_x(i, target);
188  xpauli_z(i, control) = xpauli_z(i, control) ^ xpauli_z(i, target);
189  zpauli_phase(i) =
190  zpauli_phase(i) ^ (zpauli_x(i, control) && zpauli_z(i, target) &&
191  !(zpauli_x(i, target) ^ zpauli_z(i, control)));
192  zpauli_x(i, target) = zpauli_x(i, control) ^ zpauli_x(i, target);
193  zpauli_z(i, control) = zpauli_z(i, control) ^ zpauli_z(i, target);
194  }
195 }

197 void CliffTableau::apply_CX_at_end(unsigned control, unsigned target) {
198  MatrixXb::RowXpr xxc = xpauli_x.row(control);
199  MatrixXb::RowXpr xzc = xpauli_z.row(control);
200  MatrixXb::RowXpr zxt = zpauli_x.row(target);
201  MatrixXb::RowXpr zzt = zpauli_z.row(target);
202  row_mult(
203  xxc, xzc, xpauli_phase(control), xpauli_x.row(target),
204  xpauli_z.row(target), xpauli_phase(target), 1., xxc, xzc,
205  xpauli_phase(control));
206  row_mult(
207  zpauli_x.row(control), zpauli_z.row(control), zpauli_phase(control), zxt,
208  zzt, zpauli_phase(target), 1., zxt, zzt, zpauli_phase(target));
209 }

211 void CliffTableau::apply_gate_at_front(
212  OpType type, const std::vector<unsigned> &qbs) {
213  switch (type) {
214  case OpType::Z: {
215  apply_S_at_front(qbs.at(0));
216  apply_S_at_front(qbs.at(0));
217  break;
218  }
219  case OpType::X: {
220  apply_V_at_front(qbs.at(0));
221  apply_V_at_front(qbs.at(0));
222  break;
223  }
224  case OpType::Y: {
225  apply_S_at_front(qbs.at(0));
226  apply_S_at_front(qbs.at(0));
227  apply_V_at_front(qbs.at(0));
228  apply_V_at_front(qbs.at(0));
229  break;
230  }
231  case OpType::S: {
232  apply_S_at_front(qbs.at(0));
233  break;
234  }
235  case OpType::Sdg: {
236  apply_S_at_front(qbs.at(0));
237  apply_S_at_front(qbs.at(0));
238  apply_S_at_front(qbs.at(0));
239  break;
240  }
241  case OpType::V: {
242  apply_V_at_front(qbs.at(0));
243  break;
244  }
245  case OpType::Vdg: {
246  apply_V_at_front(qbs.at(0));
247  apply_V_at_front(qbs.at(0));
248  apply_V_at_front(qbs.at(0));
249  break;
250  }
251  case OpType::H: {
252  apply_S_at_front(qbs.at(0));
253  apply_V_at_front(qbs.at(0));
254  apply_S_at_front(qbs.at(0));
255  break;
256  }
257  case OpType::CX: {
258  apply_CX_at_front(qbs.at(0), qbs.at(1));
259  break;
260  }
261  case OpType::CY: {
262  apply_V_at_front(qbs.at(1));
263  apply_V_at_front(qbs.at(1));
264  apply_V_at_front(qbs.at(1));
265  apply_CX_at_front(qbs.at(0), qbs.at(1));
266  apply_V_at_front(qbs.at(1));
267  break;
268  }
269  case OpType::CZ: {
270  apply_S_at_front(qbs.at(1));
271  apply_V_at_front(qbs.at(1));
272  apply_S_at_front(qbs.at(1));
273  apply_CX_at_front(qbs.at(0), qbs.at(1));
274  apply_S_at_front(qbs.at(1));
275  apply_V_at_front(qbs.at(1));
276  apply_S_at_front(qbs.at(1));
277  break;
278  }
279  case OpType::SWAP: {
280  apply_CX_at_front(qbs.at(0), qbs.at(1));
281  apply_CX_at_front(qbs.at(1), qbs.at(0));
282  apply_CX_at_front(qbs.at(0), qbs.at(1));
283  break;
284  }
285  case OpType::BRIDGE: {
286  apply_CX_at_front(qbs.at(0), qbs.at(2));
287  break;
288  }
289  case OpType::noop:
290  case OpType::Phase: {
291  break;
292  }
293  default: {
294  throw BadOpType("Not a Clifford gate", type);
295  }
296  }

403 void CliffTableau::apply_pauli_at_front(
404  const QubitPauliTensor &pauli, unsigned half_pis) {
405  CliffTableau pauli_tab(size_);
406  pauli_tab.qubits_ = this->qubits_;
407  pauli_tab.apply_pauli_at_end(pauli, half_pis);
408  CliffTableau product = CliffTableau::compose(pauli_tab, *this);
409  this->xpauli_x = product.xpauli_x;
410  this->xpauli_z = product.xpauli_z;
411  this->xpauli_phase = product.xpauli_phase;
412  this->zpauli_x = product.zpauli_x;
413  this->zpauli_z = product.zpauli_z;
414  this->zpauli_phase = product.zpauli_phase;
415 }

417 void CliffTableau::apply_pauli_at_end(
418  const QubitPauliTensor &pauli, unsigned half_pis) {
419  half_pis = half_pis % 4;
420  if (half_pis == 0) return; // Identity
421  if (half_pis == 2) { // Degenerates to product of PI rotations
422  for (const std::pair<const Qubit, Pauli> &term : pauli.string.map) {
423  switch (term.second) {
424  case Pauli::I: {
425  break;
426  }
427  case Pauli::X: {
428  apply_gate_at_end(OpType::X, {term.first});
429  break;
430  }
431  case Pauli::Y: {
432  apply_gate_at_end(OpType::Y, {term.first});
433  break;
434  }
435  case Pauli::Z: {
436  apply_gate_at_end(OpType::Z, {term.first});
437  break;
438  }
439  }
440  }
441  return;
442  }
443 
444  // From here, half_pis == 1 or 3
445  // They act the same except for a phase flip on the product term
446  MatrixXb product_x = MatrixXb::Zero(1, size_);
447  MatrixXb product_z = MatrixXb::Zero(1, size_);
448  MatrixXb::RowXpr px = product_x.row(0);
449  MatrixXb::RowXpr pz = product_z.row(0);
450  if (pauli.coeff != 1. && pauli.coeff != -1.)
451  throw std::invalid_argument(
452  "Can only apply Paulis with real unit coefficients to "
453  "CliffTableaus");
454  bool phase = (pauli.coeff == -1.) ^ (half_pis == 3);
455 
456  // Collect the product term
457  for (const std::pair<const Qubit, Pauli> &term : pauli.string.map) {
458  unsigned uqb = qubits_.left.at(term.first);
459  switch (term.second) {
460  case Pauli::I: {
461  break;
462  }
463  case Pauli::X: {
464  row_mult(
465  xpauli_x.row(uqb), xpauli_z.row(uqb), xpauli_phase(uqb), px, pz,
466  phase, 1., px, pz, phase);
467  break;
468  }
469  case Pauli::Y: {
470  row_mult(
471  zpauli_x.row(uqb), zpauli_z.row(uqb), zpauli_phase(uqb), px, pz,
472  phase, 1., px, pz, phase);
473  row_mult(
474  xpauli_x.row(uqb), xpauli_z.row(uqb), xpauli_phase(uqb), px, pz,
475  phase, i_, px, pz, phase);
476  break;
477  }
478  case Pauli::Z: {
479  row_mult(
480  zpauli_x.row(uqb), zpauli_z.row(uqb), zpauli_phase(uqb), px, pz,
481  phase, 1., px, pz, phase);
482  break;
483  }
484  }
485  }
486 
487  // Apply the product term on the anti-commuting rows
488  for (const std::pair<const Qubit, Pauli> &term : pauli.string.map) {
489  unsigned uqb = qubits_.left.at(term.first);
490  MatrixXb::RowXpr xx = xpauli_x.row(uqb);
491  MatrixXb::RowXpr xz = xpauli_z.row(uqb);
492  MatrixXb::RowXpr zx = zpauli_x.row(uqb);
493  MatrixXb::RowXpr zz = zpauli_z.row(uqb);
494  switch (term.second) {
495  case Pauli::I: {
496  break;
497  }
498  case Pauli::X: {
499  row_mult(
500  px, pz, phase, zx, zz, zpauli_phase(uqb), i_, zx, zz,
501  zpauli_phase(uqb));
502  break;
503  }
504  case Pauli::Y: {
505  row_mult(
506  px, pz, phase, zx, zz, zpauli_phase(uqb), i_, zx, zz,
507  zpauli_phase(uqb));
508  row_mult(
509  px, pz, phase, xx, xz, xpauli_phase(uqb), i_, xx, xz,
510  xpauli_phase(uqb));
511  break;
512  }
513  case Pauli::Z: {
514  row_mult(
515  px, pz, phase, xx, xz, xpauli_phase(uqb), i_, xx, xz,
516  xpauli_phase(uqb));
517  break;
518  }
519  }
520  }
521 }

544  const MatrixXb &pauli_x, const MatrixXb &pauli_z,
545  const VectorXb &pauli_phase) {
546  std::vector<Complex> cphase;
547  const unsigned n = pauli_phase.size();
548  for (unsigned i = 0; i < n; i++) {
549  Complex cp = pauli_phase(i) ? -1. : 1.;
550  for (unsigned j = 0; j < n; j++) {
551  if (pauli_x(i, j) && pauli_z(i, j)) cp *= i_;
552  }
553  cphase.push_back(cp);
554  }
555  return cphase;
556 }

558 CliffTableau CliffTableau::compose(
559  const CliffTableau &first, const CliffTableau &second) {
560  if (first.qubits_ != second.qubits_)
561  throw std::logic_error(
562  "Cannot compose Clifford Tableaus with different qubit maps");
563  const unsigned n = first.size_;
564  std::vector<Complex> first_x_cphase =
565  cphase_from_tableau(first.xpauli_x, first.xpauli_z, first.xpauli_phase);
566  std::vector<Complex> first_z_cphase =
567  cphase_from_tableau(first.zpauli_x, first.zpauli_z, first.zpauli_phase);
568  std::vector<Complex> second_x_cphase = cphase_from_tableau(
569  second.xpauli_x, second.xpauli_z, second.xpauli_phase);
570  std::vector<Complex> second_z_cphase = cphase_from_tableau(
571  second.zpauli_x, second.zpauli_z, second.zpauli_phase);
572 
573  CliffTableau result(n);
574  result.xpauli_x.setZero();
575  result.xpauli_z.setZero();
576  result.zpauli_x.setZero();
577  result.zpauli_z.setZero();
578  result.qubits_ = first.qubits_;
579 
580  // Target cphases
581  std::vector<Complex> xpauli_cphase(n);
582  std::vector<Complex> zpauli_cphase(n);
583 
584  // Calculate xpauli rows
585  for (unsigned i = 0; i < n; i++) { // row in second
586  // Sum rows from first according to second.xpauli
587  xpauli_cphase[i] = second_x_cphase[i];
588  bool phase_flip = false;
589  for (unsigned j = 0; j < n; j++) { // col in second
590  if (second.xpauli_x(i, j)) {
591  // add first.xpauli to xpauli
592  for (unsigned k = 0; k < n; k++) {
593  phase_flip ^= result.xpauli_z(i, k) && first.xpauli_x(j, k);
594  result.xpauli_x(i, k) = result.xpauli_x(i, k) != first.xpauli_x(j, k);
595  result.xpauli_z(i, k) = result.xpauli_z(i, k) != first.xpauli_z(j, k);
596  }
597  xpauli_cphase[i] *= first_x_cphase[j];
598  }
599  if (second.xpauli_z(i, j)) {
600  // add first.zpauli to xpauli
601  for (unsigned k = 0; k < n; k++) {
602  phase_flip ^= result.xpauli_z(i, k) && first.zpauli_x(j, k);
603  result.xpauli_x(i, k) = result.xpauli_x(i, k) != first.zpauli_x(j, k);
604  result.xpauli_z(i, k) = result.xpauli_z(i, k) != first.zpauli_z(j, k);
605  }
606  xpauli_cphase[i] *= first_z_cphase[j];
607  }
608  }
609  if (phase_flip) xpauli_cphase[i] *= -1.;
610  }
611 
612  // Calculate zpauli rows
613  for (unsigned i = 0; i < n; i++) { // row in second
614  // Sum rows from first according to second.zpauli
615  zpauli_cphase[i] = second_z_cphase[i];
616  bool phase_flip = false;
617  for (unsigned j = 0; j < n; j++) { // col in second
618  if (second.zpauli_x(i, j)) {
619  // add first.xpauli to zpauli
620  for (unsigned k = 0; k < n; k++) {
621  phase_flip ^= result.zpauli_z(i, k) && first.xpauli_x(j, k);
622  result.zpauli_x(i, k) = result.zpauli_x(i, k) != first.xpauli_x(j, k);
623  result.zpauli_z(i, k) = result.zpauli_z(i, k) != first.xpauli_z(j, k);
624  }
625  zpauli_cphase[i] *= first_x_cphase[j];
626  }
627  if (second.zpauli_z(i, j)) {
628  // add first.zpauli to zpauli
629  for (unsigned k = 0; k < n; k++) {
630  phase_flip ^= result.zpauli_z(i, k) && first.zpauli_x(j, k);
631  result.zpauli_x(i, k) = result.zpauli_x(i, k) != first.zpauli_x(j, k);
632  result.zpauli_z(i, k) = result.zpauli_z(i, k) != first.zpauli_z(j, k);
633  }
634  zpauli_cphase[i] *= first_z_cphase[j];
635  }
636  }
637  if (phase_flip) zpauli_cphase[i] *= -1.;
638  }
639 
640  std::vector<Complex> current_x_cphase = cphase_from_tableau(
641  result.xpauli_x, result.xpauli_z, result.xpauli_phase);
642  std::vector<Complex> current_z_cphase = cphase_from_tableau(
643  result.zpauli_x, result.zpauli_z, result.zpauli_phase);
644 
645  for (unsigned i = 0; i < n; i++) {
646  TKET_ASSERT((current_x_cphase[i] * xpauli_cphase[i]).imag() == 0);
647  result.xpauli_phase(i) = (current_x_cphase[i] == -xpauli_cphase[i]);
648  TKET_ASSERT((current_z_cphase[i] * zpauli_cphase[i]).imag() == 0);
649  result.zpauli_phase(i) = (current_z_cphase[i] == -zpauli_cphase[i]);
650  }
651 
652  return result;
653 }
