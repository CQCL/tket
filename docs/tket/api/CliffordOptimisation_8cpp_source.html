45static bool multiq_clifford_match(Circuit &circ, bool allow_swaps) {
46 bool success = false;
47 // map from vertex/port to qubit number
48 std::map<Vertex, unit_set_t> v_to_qb = circ.vertex_unit_map();
49 // map from vertex to its depth from slices/reverse_slices
50 std::map<Vertex, unsigned> v_to_depth = circ.vertex_depth_map();
51 std::map<Vertex, unsigned> v_to_rev_depth = circ.vertex_rev_depth_map();
52 // Analysis complete, we can now go through and actually do the transformation
53 VertexList bin;
54 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
55 if (circ.get_OpType_from_Vertex(v) == OpType::CX &&
56 circ.n_out_edges(v) == 2) {
57 // Walk to next CX on the same pair
58 unit_set_t cx_units = v_to_qb.at(v);
59 qubit_vector_t qbs(cx_units.begin(), cx_units.end());
60 QPathDetailed path0;
61 QPathDetailed path1;
62 bool found_cnot = false;
63 Edge e0 = circ.get_nth_out_edge(v, 0);
64 Vertex v0 = circ.target(e0);
65 port_t port = circ.get_target_port(e0);
66 while (true) {
67 OpType v_type = circ.get_OpType_from_Vertex(v0);
68 if (is_final_q_type(v_type)) {
69 break;
70 } else if (v_type == OpType::CX && v_to_qb.at(v0) == cx_units) {
71 found_cnot = true;
72 break;
73 }
74 path0.push_back({v0, port});
75 e0 = circ.get_next_edge(v0, e0);
76 v0 = circ.target(e0);
77 port = circ.get_target_port(e0);
78 }
79 if (!found_cnot) {
80 continue;
81 }
82 Edge e1 = circ.get_nth_out_edge(v, 1);
83 Vertex v1 = circ.target(e1);
84 while (v1 != v0) {
85 path1.push_back({v1, circ.get_target_port(e1)});
86 e1 = circ.get_next_edge(v1, e1);
87 v1 = circ.target(e1);
88 }
89 unsigned path0_length = path0.size();
90 unsigned path1_length = path1.size();
91 unsigned front0_index = 0;
92 unsigned front1_index = 0;
93 while (front0_index < path0_length &&
94 circ.commutes_with_basis(
95 path0[front0_index].first, Pauli::Z,
96 path0[front0_index].second)) {
97 front0_index++;
98 }
99 while (front1_index < path1_length &&
100 circ.commutes_with_basis(
101 path1[front1_index].first, Pauli::X,
102 path1[front1_index].second)) {
103 front1_index++;
104 }
105 if (port == 0) {
106 // CXs have same orientation
107 // We walk back from the end of the path
108 // unsigned is used as this is semantically an index and the check for
109 // -1 functions the same as if it were signed
110 unsigned back0_index = path0_length - 1;
111 while (back0_index != front0_index && back0_index != (unsigned)-1 &&
112 circ.commutes_with_basis(
113 path0[back0_index].first, Pauli::Z,
114 path0[back0_index].second)) {
115 back0_index--;
116 }
117 unsigned back1_index = path1_length - 1;
118 while (back1_index != front1_index && back1_index != (unsigned)-1 &&
119 circ.commutes_with_basis(
120 path1[back1_index].first, Pauli::X,
121 path1[back1_index].second)) {
122 back1_index--;
123 }
124 // Check for valid causal ordering
125 Vertex front0pre =
126 (front0_index == 0) ? v : path0[front0_index - 1].first;
127 Vertex front0post =
128 (front0_index == path0_length) ? v0 : path0[front0_index].first;
129 Vertex front1pre =
130 (front1_index == 0) ? v : path1[front1_index - 1].first;
131 Vertex front1post =
132 (front1_index == path1_length) ? v0 : path1[front1_index].first;
133 Vertex back0pre =
134 (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
135 Vertex back0post = (back0_index == path0_length - 1)
136 ? v0
137 : path0[back0_index + 1].first;
138 Vertex back1pre =
139 (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
140 Vertex back1post = (back1_index == path1_length - 1)
141 ? v0
142 : path1[back1_index + 1].first;
143 if (circ.in_causal_order(
144 front1pre, front0post, true, v_to_depth, v_to_qb))
145 continue;
146 if (circ.in_causal_order(
147 front0pre, front1post, true, v_to_depth, v_to_qb))
148 continue;
149 if (circ.in_causal_order(
150 back1post, back0pre, false, v_to_rev_depth, v_to_qb))
151 continue;
152 if (circ.in_causal_order(
153 back0post, back1pre, false, v_to_rev_depth, v_to_qb))
154 continue;
155 // Identify which pattern to replace
156 bool v_on_q0 = false;
157 if (front0_index != path0_length) {
158 if (circ.get_OpType_from_Vertex(path0[front0_index].first) !=
159 OpType::V) {
160 // The only non-empty pattern we can recognise is from a V
161 continue;
162 }
163 v_on_q0 = true;
164 if (back0_index != front0_index) {
165 // There are some ops after the V that we cannot commute past the CX
166 continue;
167 }
168 }
169 bool s_on_q1 = false;
170 if (front1_index != path1_length) {
171 if (circ.get_OpType_from_Vertex(path1[front1_index].first) !=
172 OpType::S) {
173 // The only non-empty pattern we can recognise is from a S
174 continue;
175 }
176 s_on_q1 = true;
177 if (back1_index == front1_index + 2 &&
178 circ.get_OpType_from_Vertex(path1[front1_index + 1].first) ==
179 OpType::V &&
180 circ.get_OpType_from_Vertex(path1[front1_index + 2].first) ==
181 OpType::S) {
182 /* --C-- ... --C-- --C-- ... --C-- */
183 /* | | => | | */
184 /* --X--S--V--S--X-- --X--V--S--V--X-- */
185 circ.dag[path1[front1_index].first] = {get_op_ptr(OpType::V)};
186 circ.dag[path1[front1_index + 1].first] = {get_op_ptr(OpType::S)};
187 circ.dag[path1[front1_index + 2].first] = {get_op_ptr(OpType::V)};
188 circ.add_phase(0.25);
189 front1_index++;
190 back1_index--;
191 } else if (back1_index != front1_index) {
192 // There are some ops after the S that we cannot commute past the CX
193 continue;
194 }
195 }
196 if (!allow_swaps && v_on_q0 && s_on_q1) {
197 continue;
198 }
199 success = true;
200 // Always remove the CX pair, saving the new edges from the second
201 Edge next_e0 = circ.get_nth_out_edge(v0, 0);
202 Edge next_e1 = circ.get_nth_out_edge(v0, 1);
203 Vertex next_v0 = circ.target(next_e0);
204 Vertex next_v1 = circ.target(next_e1);
205 port_t next_p0 = circ.get_target_port(next_e0);
206 port_t next_p1 = circ.get_target_port(next_e1);
207 bin.push_back(v);
208 bin.push_back(v0);
209 circ.remove_vertex(
210 v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
211 circ.remove_vertex(
212 v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
213 Edge default_h0 = circ.get_nth_in_edge(next_v0, next_p0);
214 Edge default_h1 = circ.get_nth_in_edge(next_v1, next_p1);
215
216 Subcircuit sub;
217 const Circuit *replacement;
218 Qubit q0, q1;
219 if (v_on_q0) {
220 Vertex v_gate = path0[front0_index].first;
221 q0 = Qubit(*v_to_qb.at(v_gate).begin());
222 q1 = qbs.at(0);
223 if (q0 == q1) q1 = qbs.at(1);
224 if (s_on_q1) {
225 /* --C--V--C-- --Z--S--V--X--S--\ /-- */
226 /* | | => | \ */
227 /* --X--S--X-- --X--V--S--C--V--/ \-- */
228 Vertex s_gate = path1[front1_index].first;
229 bin.push_back(v_gate);
230 bin.push_back(s_gate);
231 sub = {
232 {circ.get_nth_in_edge(v_gate, 0),
233 circ.get_nth_in_edge(s_gate, 0)},
234 {circ.get_nth_out_edge(v_gate, 0),
235 circ.get_nth_out_edge(s_gate, 0)},
236 {},
237 {v_gate, s_gate}};
238 replacement = &CircPool::CX_VS_CX_reduced();
239 } else {
240 /* --C--V--C-- --X--V--S--V--C--V--S-- */
241 /* | | => | */
242 /* --X-----X-- --V-----------X-------- */
243 bin.push_back(v_gate);
244 sub = {
245 {circ.get_nth_in_edge(v_gate, 0), default_h1},
246 {circ.get_nth_out_edge(v_gate, 0), default_h1},
247 {},
248 {v_gate}};
249 replacement = &CircPool::CX_V_CX_reduced();
250 }
251 } else if (s_on_q1) {
252 /* --C-----C-- --S-----------C-------- */
253 /* | | => | */
254 /* --X--S--X-- --Z--S--V--S--X--S--V-- */
255 Vertex s_gate = path1[front1_index].first;
256 q1 = Qubit(*v_to_qb.at(s_gate).begin());
257 q0 = qbs.at(0);
258 if (q0 == q1) q0 = qbs.at(1);
259 bin.push_back(s_gate);
260 sub = {
261 {default_h0, circ.get_nth_in_edge(s_gate, 0)},
262 {default_h0, circ.get_nth_out_edge(s_gate, 0)},
263 {},
264 {s_gate}};
265 replacement = &CircPool::CX_S_CX_reduced();
266 } else {
267 /* --C--C-- ---- */
268 /* | | => */
269 /* --X--X-- ---- */
270 continue;
271 }
272 Vertex b0 = circ.source(sub.in_hole[0]);
273 port_t p0 = circ.get_source_port(sub.in_hole[0]);
274 Vertex b1 = circ.source(sub.in_hole[1]);
275 port_t p1 = circ.get_source_port(sub.in_hole[1]);
276 Vertex a0 = circ.target(*sub.out_hole[0]);
277 Vertex a1 = circ.target(*sub.out_hole[1]);
278 circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
279 // Scan through hole and add vertices to v_to_qb
280 // and give estimates of depth and rev_depth
281 unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
282 unsigned new_rev_depth =
283 std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
284 Edge ei0 = circ.get_nth_out_edge(b0, p0);
285 Vertex vi0 = circ.target(ei0);
286 while (vi0 != a0) {
287 v_to_qb.insert({vi0, {q0}});
288 v_to_depth[vi0] = new_depth;
289 v_to_rev_depth[vi0] = new_rev_depth;
290 ei0 = circ.get_next_edge(vi0, ei0);
291 vi0 = circ.target(ei0);
292 }
293 Edge ei1 = circ.get_nth_out_edge(b1, p1);
294 Vertex vi1 = circ.target(ei1);
295 while (vi1 != a1) {
296 if (v_to_qb.find(vi1) == v_to_qb.end())
297 v_to_qb.insert({vi1, {q1}});
298 else
299 v_to_qb.at(vi1).insert(q1);
300 v_to_depth[vi1] = new_depth;
301 v_to_rev_depth[vi1] = new_rev_depth;
302 ei1 = circ.get_next_edge(vi1, ei1);
303 vi1 = circ.target(ei1);
304 }
305 } else {
306 // CXs have different orientation
307 unsigned back0_index = path0_length - 1;
308 while (back0_index >= front0_index && back0_index != (unsigned)-1 &&
309 circ.commutes_with_basis(
310 path0[back0_index].first, Pauli::X,
311 path0[back0_index].second)) {
312 back0_index--;
313 }
314 unsigned back1_index = path1_length - 1;
315 while (back1_index >= front1_index && back1_index != (unsigned)-1 &&
316 circ.commutes_with_basis(
317 path1[back1_index].first, Pauli::Z,
318 path1[back1_index].second)) {
319 back1_index--;
320 }
321 Vertex front0pre =
322 (front0_index == 0) ? v : path0[front0_index - 1].first;
323 Vertex front0post =
324 (front0_index == path0_length) ? v0 : path0[front0_index].first;
325 Vertex front1pre =
326 (front1_index == 0) ? v : path1[front1_index - 1].first;
327 Vertex front1post =
328 (front1_index == path1_length) ? v0 : path1[front1_index].first;
329 Vertex back0pre =
330 (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
331 Vertex back0post = (back0_index == path0_length - 1)
332 ? v0
333 : path0[back0_index + 1].first;
334 Vertex back1pre =
335 (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
336 Vertex back1post = (back1_index == path1_length - 1)
337 ? v0
338 : path1[back1_index + 1].first;
339 if (circ.in_causal_order(
340 front1pre, front0post, true, v_to_depth, v_to_qb))
341 continue;
342 if (circ.in_causal_order(
343 front0pre, front1post, true, v_to_depth, v_to_qb))
344 continue;
345 if (circ.in_causal_order(
346 back1post, back0pre, false, v_to_rev_depth, v_to_qb))
347 continue;
348 if (circ.in_causal_order(
349 back0post, back1pre, false, v_to_rev_depth, v_to_qb))
350 continue;
351 // Identify which pattern
352 bool vs_on_q0 = false;
353 if (back0_index >= front0_index && back0_index != (unsigned)-1) {
354 if (!(back0_index == front0_index + 1 &&
355 circ.get_OpType_from_Vertex(path0[front0_index].first) ==
356 OpType::V &&
357 circ.get_OpType_from_Vertex(path0[back0_index].first) ==
358 OpType::S)) {
359 // The only non-empty pattern we can recognise is from VS
360 continue;
361 }
362 vs_on_q0 = true;
363 }
364 bool sv_on_q1 = false;
365 if (back1_index >= front1_index && back1_index != (unsigned)-1) {
366 if (!(back1_index == front1_index + 1 &&
367 circ.get_OpType_from_Vertex(path1[front1_index].first) ==
368 OpType::S &&
369 circ.get_OpType_from_Vertex(path1[back1_index].first) ==
370 OpType::V)) {
371 // The only non-empty pattern we can recognise is from SV
372 continue;
373 }
374 sv_on_q1 = true;
375 }
376 if (!allow_swaps && !vs_on_q0 && !sv_on_q1) {
377 continue;
378 }
379 success = true;
380 // Always remove the CX pair, saving the edges to insert the replacement
381 // circuit
382 Vertex before0, before1, after0, after1;
383 port_t pb0, pb1, pa0, pa1;
384 if (front0_index > 0) {
385 before0 = path0[front0_index - 1].first;
386 pb0 = path0[front0_index - 1].second;
387 } else {
388 Edge pre0 = circ.get_nth_in_edge(v, 0);
389 before0 = circ.source(pre0);
390 pb0 = circ.get_source_port(pre0);
391 }
392 if (front1_index > 0) {
393 before1 = path1[front1_index - 1].first;
394 pb1 = path1[front1_index - 1].second;
395 } else {
396 Edge pre1 = circ.get_nth_in_edge(v, 1);
397 before1 = circ.source(pre1);
398 pb1 = circ.get_source_port(pre1);
399 }
400 if (back0_index + 1 < path0_length) {
401 after0 = path0[back0_index + 1].first;
402 pa0 = path0[back0_index + 1].second;
403 } else {
404 Edge post0 = circ.get_nth_out_edge(v0, 1);
405 after0 = circ.target(post0);
406 pa0 = circ.get_target_port(post0);
407 }
408 if (back1_index + 1 < path1_length) {
409 after1 = path1[back1_index + 1].first;
410 pa1 = path1[back1_index + 1].second;
411 } else {
412 Edge post1 = circ.get_nth_out_edge(v0, 0);
413 after1 = circ.target(post1);
414 pa1 = circ.get_target_port(post1);
415 }
416 bin.push_back(v);
417 bin.push_back(v0);
418 circ.remove_vertex(
419 v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
420 circ.remove_vertex(
421 v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
422 VertexSet to_remove;
423 const Circuit *replacement;
424 Qubit q0, q1;
425 if (vs_on_q0) {
426 Vertex v_gate = path0[front0_index].first;
427 q0 = Qubit(*v_to_qb.at(v_gate).begin());
428 q1 = qbs.at(0);
429 if (q0 == q1) q1 = qbs.at(1);
430 if (sv_on_q1) {
431 /* --C--V--S--X-- --V--S-- */
432 /* | | => */
433 /* --X--S--V--C-- --S--V-- */
434 continue;
435 } else {
436 /* --C--V--S--X-- --S-----C--V--S-- */
437 /* | | => | */
438 /* --X--------C-- --Z--S--X--S----- */
439 replacement = &CircPool::CX_V_S_XC_reduced();
440 }
441 bin.push_back(path0[front0_index].first);
442 bin.push_back(path0[back0_index].first);
443 to_remove.insert(path0[front0_index].first);
444 to_remove.insert(path0[back0_index].first);
445 } else if (sv_on_q1) {
446 /* --C--------X-- --X--V--C--V----- */
447 /* | | => | */
448 /* --X--S--V--C-- --V-----X--S--V-- */
449 Vertex s_gate = path1[front1_index].first;
450 q1 = Qubit(*v_to_qb.at(s_gate).begin());
451 q0 = qbs.at(0);
452 if (q0 == q1) q0 = qbs.at(1);
453 replacement = &CircPool::CX_S_V_XC_reduced();
454 bin.push_back(path1[front1_index].first);
455 bin.push_back(path1[back1_index].first);
456 to_remove.insert(path1[front1_index].first);
457 to_remove.insert(path1[back1_index].first);
458 } else {
459 /* --C--X-- --X--\ /-- */
460 /* | | => | \ */
461 /* --X--C-- --C--/ \-- */
462 q0 = qbs.at(0);
463 q1 = qbs.at(1);
464 replacement = &CircPool::CX_XC_reduced();
465 }
466 Subcircuit sub = {
467 {circ.get_nth_out_edge(before0, pb0),
468 circ.get_nth_out_edge(before1, pb1)},
469 {circ.get_nth_in_edge(after0, pa0),
470 circ.get_nth_in_edge(after1, pa1)},
471 {},
472 to_remove};
473 Vertex b0 = circ.source(sub.in_hole[0]);
474 port_t p0 = circ.get_source_port(sub.in_hole[0]);
475 Vertex b1 = circ.source(sub.in_hole[1]);
476 port_t p1 = circ.get_source_port(sub.in_hole[1]);
477 Vertex a0 = circ.target(*sub.out_hole[0]);
478 Vertex a1 = circ.target(*sub.out_hole[1]);
479 circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
480 // Scan through hole and add vertices to v_to_qb
481 unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
482 unsigned new_rev_depth =
483 std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
484 Edge ei0 = circ.get_nth_out_edge(b0, p0);
485 Vertex vi0 = circ.target(ei0);
486 while (vi0 != a0) {
487 v_to_qb.insert({vi0, {q0}});
488 v_to_depth[vi0] = new_depth;
489 v_to_rev_depth[vi0] = new_rev_depth;
490 ei0 = circ.get_next_edge(vi0, ei0);
491 vi0 = circ.target(ei0);
492 }
493 Edge ei1 = circ.get_nth_out_edge(b1, p1);
494 Vertex vi1 = circ.target(ei1);
495 while (vi1 != a1) {
496 if (v_to_qb.find(vi1) == v_to_qb.end())
497 v_to_qb.insert({vi1, {q1}});
498 else
499 v_to_qb.at(vi1).insert(q1);
500 v_to_depth[vi1] = new_depth;
501 v_to_rev_depth[vi1] = new_rev_depth;
502 ei1 = circ.get_next_edge(vi1, ei1);
503 vi1 = circ.target(ei1);
504 }
505 }
506 }
507 }
508 if (allow_swaps) {
509 circ.replace_SWAPs();
510 }
511 circ.remove_vertices(
512 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
513 return success;
514}

518// TODO:: Copy classical-controls and any controls from CX
519static bool copy_pi_through_CX_method(Circuit &circ) {
520 bool success = false;
521 VertexList bin;
522 BGL_FORALL_VERTICES(vert, circ.dag, DAG) {
523 if (circ.get_OpType_from_Vertex(vert) == OpType::CX &&
524 circ.n_out_edges(vert) == 2) {
525 Edge e0 = circ.get_nth_out_edge(vert, 0);
526 Vertex v0 = circ.target(e0);
527 if (circ.get_OpType_from_Vertex(v0) == OpType::X) {
528 success = true;
529 Edge afterX = circ.get_next_edge(v0, e0);
530 Edge after1 = circ.get_nth_out_edge(vert, 1);
531 Vertex v_after1 = circ.target(after1);
532 port_t pa1 = circ.get_target_port(after1);
533 bin.push_back(vert);
534 circ.remove_vertex(
535 vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
536 after1 = circ.get_nth_in_edge(v_after1, pa1);
537 Subcircuit sub = {{afterX, after1}, {afterX, after1}, {}, {}};
538 circ.substitute(CircPool::X1_CX(), sub, Circuit::VertexDeletion::No);
539 continue;
540 }
541 Edge e1 = circ.get_nth_out_edge(vert, 1);
542 Vertex v1 = circ.target(e1);
543 if (circ.get_OpType_from_Vertex(v1) == OpType::Z) {
544 success = true;
545 Edge afterZ = circ.get_next_edge(v1, e1);
546 Edge after0 = circ.get_nth_out_edge(vert, 0);
547 Vertex v_after0 = circ.target(after0);
548 port_t pa0 = circ.get_target_port(after0);
549 bin.push_back(vert);
550 circ.remove_vertex(
551 vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
552 after0 = circ.get_nth_in_edge(v_after0, pa0);
553 Subcircuit sub = {{after0, afterZ}, {after0, afterZ}, {}, {}};
554 circ.substitute(CircPool::Z0_CX(), sub, Circuit::VertexDeletion::No);
555 }
556 }
557 }
558 circ.remove_vertices(
559 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
560 return success;

563// TODO:: Detect classical controls and stop simplification
564static bool singleq_clifford_from_edge(
565 Circuit &circ, Edge &e, VertexList &bin) {
566 VertexSet single_vs;
567 Edge ei = e;
568 Vertex vi = circ.target(ei);
569 // stuff to check if it is already in standard form
570 // 6 = not consumed anything, 5=Z, 4=X, 3=first S, 2=V, 1=second S, 0=other
571 unsigned cliff_last = 6;
572 while (circ.detect_singleq_unitary_op(vi)) {
573 // i.e. we have a single qubit unitary
574 single_vs.insert(vi);
575
576 // keep track of clifford standard form
577 switch (circ.get_OpType_from_Vertex(vi)) {
578 case OpType::Z: {
579 cliff_last = (cliff_last > 5) ? 5 : 0;
580 break;
581 }
582 case OpType::X: {
583 cliff_last = (cliff_last > 4) ? 4 : 0;
584 break;
585 }
586 case OpType::S: {
587 if (cliff_last > 3)
588 cliff_last = 3;
589 else if (cliff_last == 2)
590 cliff_last = 1;
591 else
592 cliff_last = 0;
593 break;
594 }
595 case OpType::V: {
596 cliff_last = (cliff_last > 2) ? 2 : 0;
597 break;
598 }
599 default: {
600 cliff_last = 0;
601 break;
602 }
603 }
604
605 ei = circ.get_next_edge(vi, ei);
606 vi = circ.target(ei);
607 }
608 // if the sequence is not in the standard form, replace it
609 if (cliff_last == 0) {
610 Subcircuit s = {{e}, {ei}, {}, single_vs};
611 Circuit sub = circ.subcircuit(s);
612 bool reduced = (decompose_single_qubits_TK1() >> squash_1qb_to_tk1() >>
613 decompose_cliffords_std())
614 .apply(sub);
615 if (reduced) {
616 circ.substitute(sub, s, Circuit::VertexDeletion::No);
617 bin.insert(bin.end(), single_vs.begin(), single_vs.end());
618 return true;
619 }
620 }
621 return false;
622}

624Transform singleq_clifford_sweep() {
625 return Transform([](Circuit &circ) {
626 bool success = false;
627 VertexList bin;
628 std::vector<Vertex> vertices = circ.vertices_in_order();
629 for (auto it = vertices.crbegin(); it != vertices.crend(); ++it) {
630 const Vertex &v = *it;
631 OpType v_type = circ.get_OpType_from_Vertex(v);
632 if (v_type != OpType::Conditional &&
633 circ.n_out_edges_of_type(v, EdgeType::Quantum) == 2) {
634 for (port_t p = 0; p < 2; p++) {
635 // put both subsequent single qubit sequences into normal form
636 // stuff for obtaining next single qubit subcircuit
637 Edge e = circ.get_nth_out_edge(v, p);
638 success = singleq_clifford_from_edge(circ, e, bin) || success;
639 }
640 if (v_type == OpType::CX) {
641 // commute Z, X and whatever else we can through
642 Edge e0 = circ.get_nth_out_edge(v, 0);
643 Vertex v0 = circ.target(e0);
644
645 if (circ.get_OpType_from_Vertex(v0) == OpType::Z) {
646 circ.remove_vertex(
647 v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
648 Edge in_e = circ.get_nth_in_edge(v, 0);
649 circ.rewire(v0, {in_e}, {EdgeType::Quantum});
650 success = true;
651 e0 = circ.get_nth_out_edge(v, 0);
652 v0 = circ.target(e0);
653 }
654 if (circ.get_OpType_from_Vertex(v0) == OpType::X) {
655 circ.remove_vertex(
656 v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
657 Edge in_e = circ.get_nth_in_edge(v, 0);
658 circ.rewire(v0, {in_e}, {EdgeType::Quantum});
659 Vertex new_v = circ.add_vertex(OpType::X);
660 in_e = circ.get_nth_in_edge(v, 1);
661 circ.rewire(new_v, {in_e}, {EdgeType::Quantum});
662 success = true;
663 e0 = circ.get_nth_out_edge(v, 0);
664 v0 = circ.target(e0);
665 }
666 if (circ.get_OpType_from_Vertex(v0) == OpType::S) {
667 circ.remove_vertex(
668 v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
669 Edge in_e = circ.get_nth_in_edge(v, 0);
670 circ.rewire(v0, {in_e}, {EdgeType::Quantum});
671 success = true;
672 }
673 Edge e1 = circ.get_nth_out_edge(v, 1);
674 Vertex v1 = circ.target(e1);
675 if (circ.get_OpType_from_Vertex(v1) == OpType::Z) {
676 circ.remove_vertex(
677 v1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
678 Edge in_e = circ.get_nth_in_edge(v, 1);
679 circ.rewire(v1, {in_e}, {EdgeType::Quantum});
680 Vertex new_v = circ.add_vertex(OpType::Z);
681 in_e = circ.get_nth_in_edge(v, 0);
682 circ.rewire(new_v, {in_e}, {EdgeType::Quantum});
683 success = true;
684 e1 = circ.get_nth_out_edge(v, 1);
685 v1 = circ.target(e1);
686 }
687 if (circ.get_OpType_from_Vertex(v1) == OpType::X) {
688 circ.remove_vertex(
689 v1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
690 Edge in_e = circ.get_nth_in_edge(v, 1);
691 circ.rewire(v1, {in_e}, {EdgeType::Quantum});
692 success = true;
693 e1 = circ.get_nth_out_edge(v, 1);
694 v1 = circ.target(e1);
695 }
696 if (circ.get_OpType_from_Vertex(v1) == OpType::V) {
697 circ.remove_vertex(
698 v1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
699 Edge in_e = circ.get_nth_in_edge(v, 1);
700 circ.rewire(v1, {in_e}, {EdgeType::Quantum});
701 success = true;
702 }
703 }
704 }
705 }
706 // clean up the start of the circuit
707 for (const Vertex &i : circ.q_inputs()) {
708 Edge e = circ.get_nth_out_edge(i, 0);
709 success = singleq_clifford_from_edge(circ, e, bin) || success;
710 }
711 circ.remove_vertices(
712 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
713 return success;
714 });

717struct MeasureVertices {
718 Qubit qubit;
719 Bit bit;
720 Vertex measure;
721 MeasureVertices(const Qubit &q, const Bit &b, const Vertex &v)
722 : qubit(q), bit(b), measure(v) {}
723};

725std::pair<VertexSet, std::vector<MeasureVertices>> get_end_of_circuit_clifford(
726 const Circuit &circ) {
727 // Initialize vector to store MeasureVertices for end-of-circuit Measures
728 std::vector<MeasureVertices> end_of_circuit_measures;
729
730 // Iterate over Qubit boundaries to find Measure gates
731 for (auto [it, end] =
732 circ.boundary.get<TagType>().equal_range(UnitType::Qubit);
733 it != end; it++) {
734 Vertex q_out = it->out_;
735 Edge last_gate_out_edge = circ.get_nth_in_edge(q_out, 0);
736 Vertex last_gate = circ.source(last_gate_out_edge);
737
738 // Check if last gate is a Measure gate
739 if (circ.get_OpType_from_Vertex(last_gate) == OpType::Measure) {
740 Edge possible_c_out_in_edge = circ.get_nth_out_edge(last_gate, 1);
741 Vertex possible_c_out = circ.target(possible_c_out_in_edge);
742
743 // If the Measure is followed by a ClOutput, store the MeasureVertices
744 if (circ.get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
745 Bit b(circ.get_id_from_out(possible_c_out));
746 end_of_circuit_measures.push_back(
747 MeasureVertices(Qubit(it->id_), b, last_gate));
748 }
749 }
750 }
751
752 // Initialize variables for constructing Clifford circuit
753 VertexSet clifford_vertices;
754 std::map<Edge, Qubit> frontier;
755 VertexSet previous;
756
757 // Populate frontier map with Measure edges and associated qubits
758 for (const auto &mv : end_of_circuit_measures) {
759 frontier.insert({circ.get_nth_in_edge(mv.measure, 0), mv.qubit});
760 }
761
762 // Start adding Clifford vertices to set of vertices
763 std::vector<std::pair<OpType, std::vector<Qubit>>> clifford_commands;
764 std::vector<Edge> to_erase;
765 while (!frontier.empty()) {
766 VertexSet current;
767 // Iterate over frontier edges to identify Clifford gates
768 for (const auto &e : frontier) {
769 Vertex source = circ.source(e.first);
770 OpDesc desc = circ.get_OpDesc_from_Vertex(source);
771 // Check if gate is a Clifford gate without classical or boolean inputs
772 if (desc.is_clifford_gate() && desc.n_boolean() == 0 &&
773 desc.n_classical() == 0) {
774 current.insert(source);
775 } else {
776 // Remove non-Clifford gates from frontier
777 to_erase.push_back(e.first);
778 }
779 }
780 for (const Edge &e : to_erase) {
781 frontier.erase(e);
782 }
783 // Check if identical to previous vertices
784 if (current == previous) {
785 break;
786 }
787 // Update previous set with current set
788 previous = current;
789
790 // Process Clifford gates and update frontier
791 for (const Vertex &v : current) {
792 EdgeVec out_edges = circ.get_all_out_edges(v);
793 std::vector<Edge> edges;
794 std::vector<Qubit> qubits;
795
796 // Check which edges connect to existing qubits in the frontier
797 for (const Edge &edge : out_edges) {
798 auto it = frontier.find(edge);
799 if (it != frontier.end()) {
800 edges.push_back(it->first);
801 qubits.push_back(it->second);
802 }
803 }
804
805 // If all outgoing edges connect to frontier qubits, add gate to Clifford
806 // circuit
807 if (qubits.size() == out_edges.size()) {
808 clifford_vertices.insert(v);
809 EdgeVec in_edges = circ.get_in_edges(v);
810
811 // Update frontier with incoming edges to processed gate
812 for (unsigned i = 0; i < edges.size(); i++) {
813 Edge e = edges[i];
814 port_t source = circ.get_source_port(e);
815 frontier.insert({in_edges[source], qubits[i]});
816 frontier.erase(e);
817 }
818 }
819 }
820 }
821 return std::make_pair(clifford_vertices, end_of_circuit_measures);
822}

824Transform push_cliffords_through_measures() {
825 return Transform([](Circuit &circ) {
826 // Extract Clifford and Measure vertices information from provided circuit
827 std::pair<VertexSet, std::vector<MeasureVertices>> clifford_info =
828 get_end_of_circuit_clifford(circ);
829 VertexSet clifford_vertices = clifford_info.first;
830 std::vector<MeasureVertices> measure_vertices = clifford_info.second;
831
832 // Initialize vectors to track various information
833 std::vector<Qubit> qubits;
834 std::vector<Bit> bits;
835 QubitPauliMap base;
836 VertexSet final_measure_vertices;
837
838 // Populate qubits, bits, basic measurement operators and final measurement
839 // vertices
840 for (const MeasureVertices &mv : measure_vertices) {
841 qubits.push_back(mv.qubit);
842 bits.push_back(mv.bit);
843 base.insert({mv.qubit, Pauli::I});
844 final_measure_vertices.insert(mv.measure);
845 }
846
847 // Construct a circuit with Clifford gates
848 Circuit clifford_circuit =
849 circ.subcircuit(circ.make_subcircuit(clifford_vertices));
850 UnitaryRevTableau cliff_tab =
851 circuit_to_unitary_rev_tableau(clifford_circuit);
852
853 // Generate updated measurement operator for each end of circuit measurement
854 std::list<SpSymPauliTensor> measurement_operators;
855 for (const Qubit &q : qubits) {
856 QubitPauliMap copy = base;
857 copy[q] = Pauli::Z;
858 SpPauliStabiliser sps(copy, 2);
859 measurement_operators.push_back(cliff_tab.get_row_product(sps));
860 }
861
862 // Mutually diagonalize updated measuement operators
863 Circuit mutual_c = mutual_diagonalise(
864 measurement_operators, std::set<Qubit>(qubits.begin(), qubits.end()),
865 CXConfigType::Snake);
866
867 // If mutual diagonalisation circuit doesn't improve 2-qubit gate
868 // count then keep circuit as is
869 if (mutual_c.count_n_qubit_gates(2) >=
870 clifford_circuit.count_n_qubit_gates(2)) {
871 return false;
872 }
873
874 // Remove Clifford and Measure vertices from the original circuit
875 // before adding mutual diagonalisation circuit
876 circ.remove_vertices(
877 clifford_vertices, Circuit::GraphRewiring::Yes,
878 Circuit::VertexDeletion::Yes);
879 circ.remove_vertices(
880 final_measure_vertices, Circuit::GraphRewiring::Yes,
881 Circuit::VertexDeletion::Yes);
882 // add mutual diagonalisation circuit
883 circ.append(mutual_c);
884
885 // add back in end of circuit measurements
886 TKET_ASSERT(qubits.size() == bits.size());
887 for (unsigned i = 0; i < qubits.size(); i++) {
888 circ.add_measure(qubits[i], bits[i]);
889 }
890
891 // Add classical logic to permute output measurements to correct result
892 std::string reg_name =
893 circ.get_next_c_reg_name(c_permutation_scratch_name());
894 register_t scratch_r = circ.add_c_register(reg_name, bits.size() + 1);
895 // Convert Bit to vector for ease of indexing and assigning
896 std::vector<Bit> scratch_v;
897 for (const auto &b : scratch_r) {
898 Bit scratch_b = Bit(b.second);
899 scratch_v.push_back(scratch_b);
900 }
901
902 // We need to collect ClassicalX due to phase correction
903 // and permutation due to Z terms in operator
904 std::vector<Bit> phase_correction;
905
906 TKET_ASSERT(measurement_operators.size() == bits.size());
907 auto it = measurement_operators.begin();
908 for (unsigned i = 0; i < measurement_operators.size(); ++i, ++it) {
909 // For each measurement operator, we collect the qubits in the string
910 // which have Z terms
911 auto string = it->string;
912 std::vector<Bit> parity_bits;
913 for (unsigned j = 0; j < qubits.size(); j++) {
914 Qubit q = qubits[j];
915 auto jt = string.find(q);
916 if (jt != string.end()) {
917 // If the qubit has a Z term, then it's measurement result
918 // needs to be Xored with the original target qubit of the
919 // measurement
920 if (jt->second == Pauli::Z) {
921 parity_bits.push_back(bits[j]);
922 } else {
923 TKET_ASSERT(jt->second == Pauli::I);
924 }
925 }
926 }
927 // The measurement operator should never be empty, meaning
928 // that parity_bits should always have at least one entry
929 TKET_ASSERT(!parity_bits.empty());
930 // we now add the Xor corresponding to the correction
931 for (const Bit &pb : parity_bits) {
932 std::vector<Bit> arg = {pb, Bit(scratch_v[i])};
933 circ.add_op(XorWithOp(), arg);
934 }
935
936 // Finally, we check the phase, and store the required Bit
937 // to flip as appropriate
938 if (it->coeff == 1) {
939 phase_correction.push_back(scratch_v[i]);
940 } else {
941 TKET_ASSERT(it->coeff == -1);
942 }
943 }
944
945 // Apply the constant change to bitstring due to phase
946 circ.add_op(
947 std::make_shared<SetBitsOp>(std::vector<bool>({true})),
948 std::vector<Bit>({scratch_v.back()}));
949 for (const auto &p_bit : phase_correction) {
950 circ.add_op(XorWithOp(), std::vector<Bit>({scratch_v.back(), p_bit}));
951 }
952
953 // Copy scratch results over to original Bit
954 TKET_ASSERT(bits.size() + 1 == scratch_v.size());
955 scratch_v.pop_back();
956 scratch_v.insert(scratch_v.end(), bits.begin(), bits.end());
957 circ.add_op(std::make_shared<CopyBitsOp>(unsigned(bits.size())), scratch_v);
958 return true;
959 });
960}
