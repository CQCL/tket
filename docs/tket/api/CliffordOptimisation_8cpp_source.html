38 static bool multiq_clifford_match(Circuit &circ, bool allow_swaps) {
39  bool success = false;
40  // map from vertex/port to qubit number
41  std::map<Vertex, unit_set_t> v_to_qb = circ.vertex_unit_map();
42  // map from vertex to its depth from slices/reverse_slices
43  std::map<Vertex, unsigned> v_to_depth = circ.vertex_depth_map();
44  std::map<Vertex, unsigned> v_to_rev_depth = circ.vertex_rev_depth_map();
45  // Analysis complete, we can now go through and actually do the transformation
46  VertexList bin;
47  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
48  if (circ.get_OpType_from_Vertex(v) == OpType::CX &&
49  circ.n_out_edges(v) == 2) {
50  // Walk to next CX on the same pair
51  unit_set_t cx_units = v_to_qb.at(v);
52  qubit_vector_t qbs(cx_units.begin(), cx_units.end());
53  QPathDetailed path0;
54  QPathDetailed path1;
55  bool found_cnot = false;
56  Edge e0 = circ.get_nth_out_edge(v, 0);
57  Vertex v0 = circ.target(e0);
58  port_t port = circ.get_target_port(e0);
59  while (true) {
60  OpType v_type = circ.get_OpType_from_Vertex(v0);
61  if (is_final_q_type(v_type)) {
62  break;
63  } else if (v_type == OpType::CX && v_to_qb.at(v0) == cx_units) {
64  found_cnot = true;
65  break;
66  }
67  path0.push_back({v0, port});
68  e0 = circ.get_next_edge(v0, e0);
69  v0 = circ.target(e0);
70  port = circ.get_target_port(e0);
71  }
72  if (!found_cnot) {
73  continue;
74  }
75  Edge e1 = circ.get_nth_out_edge(v, 1);
76  Vertex v1 = circ.target(e1);
77  while (v1 != v0) {
78  path1.push_back({v1, circ.get_target_port(e1)});
79  e1 = circ.get_next_edge(v1, e1);
80  v1 = circ.target(e1);
81  }
82  unsigned path0_length = path0.size();
83  unsigned path1_length = path1.size();
84  unsigned front0_index = 0;
85  unsigned front1_index = 0;
86  while (front0_index < path0_length &&
87  circ.commutes_with_basis(
88  path0[front0_index].first, Pauli::Z, PortType::Target,
89  path0[front0_index].second)) {
90  front0_index++;
91  }
92  while (front1_index < path1_length &&
93  circ.commutes_with_basis(
94  path1[front1_index].first, Pauli::X, PortType::Target,
95  path1[front1_index].second)) {
96  front1_index++;
97  }
98  if (port == 0) {
99  // CXs have same orientation
100  // We walk back from the end of the path
101  // unsigned is used as this is semantically an index and the check for
102  // -1 functions the same as if it were signed
103  unsigned back0_index = path0_length - 1;
104  while (back0_index != front0_index && back0_index != (unsigned)-1 &&
105  circ.commutes_with_basis(
106  path0[back0_index].first, Pauli::Z, PortType::Target,
107  path0[back0_index].second)) {
108  back0_index--;
109  }
110  unsigned back1_index = path1_length - 1;
111  while (back1_index != front1_index && back1_index != (unsigned)-1 &&
112  circ.commutes_with_basis(
113  path1[back1_index].first, Pauli::X, PortType::Target,
114  path1[back1_index].second)) {
115  back1_index--;
116  }
117  // Check for valid causal ordering
118  Vertex front0pre =
119  (front0_index == 0) ? v : path0[front0_index - 1].first;
120  Vertex front0post =
121  (front0_index == path0_length) ? v0 : path0[front0_index].first;
122  Vertex front1pre =
123  (front1_index == 0) ? v : path1[front1_index - 1].first;
124  Vertex front1post =
125  (front1_index == path1_length) ? v0 : path1[front1_index].first;
126  Vertex back0pre =
127  (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
128  Vertex back0post = (back0_index == path0_length - 1)
129  ? v0
130  : path0[back0_index + 1].first;
131  Vertex back1pre =
132  (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
133  Vertex back1post = (back1_index == path1_length - 1)
134  ? v0
135  : path1[back1_index + 1].first;
136  if (circ.in_causal_order(
137  front1pre, front0post, true, v_to_depth, v_to_qb))
138  continue;
139  if (circ.in_causal_order(
140  front0pre, front1post, true, v_to_depth, v_to_qb))
141  continue;
142  if (circ.in_causal_order(
143  back1post, back0pre, false, v_to_rev_depth, v_to_qb))
144  continue;
145  if (circ.in_causal_order(
146  back0post, back1pre, false, v_to_rev_depth, v_to_qb))
147  continue;
148  // Identify which pattern to replace
149  bool v_on_q0 = false;
150  if (front0_index != path0_length) {
151  if (circ.get_OpType_from_Vertex(path0[front0_index].first) !=
152  OpType::V) {
153  // The only non-empty pattern we can recognise is from a V
154  continue;
155  }
156  v_on_q0 = true;
157  if (back0_index != front0_index) {
158  // There are some ops after the V that we cannot commute past the CX
159  continue;
160  }
161  }
162  bool s_on_q1 = false;
163  if (front1_index != path1_length) {
164  if (circ.get_OpType_from_Vertex(path1[front1_index].first) !=
165  OpType::S) {
166  // The only non-empty pattern we can recognise is from a S
167  continue;
168  }
169  s_on_q1 = true;
170  if (back1_index == front1_index + 2 &&
171  circ.get_OpType_from_Vertex(path1[front1_index + 1].first) ==
172  OpType::V &&
173  circ.get_OpType_from_Vertex(path1[front1_index + 2].first) ==
174  OpType::S) {
175  /* --C-- ... --C-- --C-- ... --C-- */
176  /* | | => | | */
177  /* --X--S--V--S--X-- --X--V--S--V--X-- */
178  circ.dag[path1[front1_index].first] = {get_op_ptr(OpType::V)};
179  circ.dag[path1[front1_index + 1].first] = {get_op_ptr(OpType::S)};
180  circ.dag[path1[front1_index + 2].first] = {get_op_ptr(OpType::V)};
181  circ.add_phase(0.25);
182  front1_index++;
183  back1_index--;
184  } else if (back1_index != front1_index) {
185  // There are some ops after the S that we cannot commute past the CX
186  continue;
187  }
188  }
189  if (!allow_swaps && v_on_q0 && s_on_q1) {
190  continue;
191  }
192  success = true;
193  // Always remove the CX pair, saving the new edges from the second
194  Edge next_e0 = circ.get_nth_out_edge(v0, 0);
195  Edge next_e1 = circ.get_nth_out_edge(v0, 1);
196  Vertex next_v0 = circ.target(next_e0);
197  Vertex next_v1 = circ.target(next_e1);
198  port_t next_p0 = circ.get_target_port(next_e0);
199  port_t next_p1 = circ.get_target_port(next_e1);
200  bin.push_back(v);
201  bin.push_back(v0);
202  circ.remove_vertex(
203  v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
204  circ.remove_vertex(
205  v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
206  Edge default_h0 = circ.get_nth_in_edge(next_v0, next_p0);
207  Edge default_h1 = circ.get_nth_in_edge(next_v1, next_p1);
208 
209  Subcircuit sub;
210  const Circuit *replacement;
211  Qubit q0, q1;
212  if (v_on_q0) {
213  Vertex v_gate = path0[front0_index].first;
214  q0 = Qubit(*v_to_qb.at(v_gate).begin());
215  q1 = qbs.at(0);
216  if (q0 == q1) q1 = qbs.at(1);
217  if (s_on_q1) {
218  /* --C--V--C-- --Z--S--V--X--S--\ /-- */
219  /* | | => | \ */
220  /* --X--S--X-- --X--V--S--C--V--/ \-- */
221  Vertex s_gate = path1[front1_index].first;
222  bin.push_back(v_gate);
223  bin.push_back(s_gate);
224  sub = {
225  {circ.get_nth_in_edge(v_gate, 0),
226  circ.get_nth_in_edge(s_gate, 0)},
227  {circ.get_nth_out_edge(v_gate, 0),
228  circ.get_nth_out_edge(s_gate, 0)},
229  {v_gate, s_gate}};
230  replacement = &CircPool::CX_VS_CX_reduced();
231  } else {
232  /* --C--V--C-- --X--V--S--V--C--V--S-- */
233  /* | | => | */
234  /* --X-----X-- --V-----------X-------- */
235  bin.push_back(v_gate);
236  sub = {
237  {circ.get_nth_in_edge(v_gate, 0), default_h1},
238  {circ.get_nth_out_edge(v_gate, 0), default_h1},
239  {v_gate}};
240  replacement = &CircPool::CX_V_CX_reduced();
241  }
242  } else if (s_on_q1) {
243  /* --C-----C-- --S-----------C-------- */
244  /* | | => | */
245  /* --X--S--X-- --Z--S--V--S--X--S--V-- */
246  Vertex s_gate = path1[front1_index].first;
247  q1 = Qubit(*v_to_qb.at(s_gate).begin());
248  q0 = qbs.at(0);
249  if (q0 == q1) q0 = qbs.at(1);
250  bin.push_back(s_gate);
251  sub = {
252  {default_h0, circ.get_nth_in_edge(s_gate, 0)},
253  {default_h0, circ.get_nth_out_edge(s_gate, 0)},
254  {s_gate}};
255  replacement = &CircPool::CX_S_CX_reduced();
256  } else {
257  /* --C--C-- ---- */
258  /* | | => */
259  /* --X--X-- ---- */
260  continue;
261  }
262  Vertex b0 = circ.source(sub.q_in_hole[0]);
263  port_t p0 = circ.get_source_port(sub.q_in_hole[0]);
264  Vertex b1 = circ.source(sub.q_in_hole[1]);
265  port_t p1 = circ.get_source_port(sub.q_in_hole[1]);
266  Vertex a0 = circ.target(sub.q_out_hole[0]);
267  Vertex a1 = circ.target(sub.q_out_hole[1]);
268  circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
269  // Scan through hole and add vertices to v_to_qb
270  // and give estimates of depth and rev_depth
271  unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
272  unsigned new_rev_depth =
273  std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
274  Edge ei0 = circ.get_nth_out_edge(b0, p0);
275  Vertex vi0 = circ.target(ei0);
276  while (vi0 != a0) {
277  v_to_qb.insert({vi0, {q0}});
278  v_to_depth[vi0] = new_depth;
279  v_to_rev_depth[vi0] = new_rev_depth;
280  ei0 = circ.get_next_edge(vi0, ei0);
281  vi0 = circ.target(ei0);
282  }
283  Edge ei1 = circ.get_nth_out_edge(b1, p1);
284  Vertex vi1 = circ.target(ei1);
285  while (vi1 != a1) {
286  if (v_to_qb.find(vi1) == v_to_qb.end())
287  v_to_qb.insert({vi1, {q1}});
288  else
289  v_to_qb.at(vi1).insert(q1);
290  v_to_depth[vi1] = new_depth;
291  v_to_rev_depth[vi1] = new_rev_depth;
292  ei1 = circ.get_next_edge(vi1, ei1);
293  vi1 = circ.target(ei1);
294  }
295  } else {
296  // CXs have different orientation
297  unsigned back0_index = path0_length - 1;
298  while (back0_index >= front0_index && back0_index != (unsigned)-1 &&
299  circ.commutes_with_basis(
300  path0[back0_index].first, Pauli::X, PortType::Target,
301  path0[back0_index].second)) {
302  back0_index--;
303  }
304  unsigned back1_index = path1_length - 1;
305  while (back1_index >= front1_index && back1_index != (unsigned)-1 &&
306  circ.commutes_with_basis(
307  path1[back1_index].first, Pauli::Z, PortType::Target,
308  path1[back1_index].second)) {
309  back1_index--;
310  }
311  Vertex front0pre =
312  (front0_index == 0) ? v : path0[front0_index - 1].first;
313  Vertex front0post =
314  (front0_index == path0_length) ? v0 : path0[front0_index].first;
315  Vertex front1pre =
316  (front1_index == 0) ? v : path1[front1_index - 1].first;
317  Vertex front1post =
318  (front1_index == path1_length) ? v0 : path1[front1_index].first;
319  Vertex back0pre =
320  (back0_index == (unsigned)-1) ? v : path0[back0_index].first;
321  Vertex back0post = (back0_index == path0_length - 1)
322  ? v0
323  : path0[back0_index + 1].first;
324  Vertex back1pre =
325  (back1_index == (unsigned)-1) ? v : path1[back1_index].first;
326  Vertex back1post = (back1_index == path1_length - 1)
327  ? v0
328  : path1[back1_index + 1].first;
329  if (circ.in_causal_order(
330  front1pre, front0post, true, v_to_depth, v_to_qb))
331  continue;
332  if (circ.in_causal_order(
333  front0pre, front1post, true, v_to_depth, v_to_qb))
334  continue;
335  if (circ.in_causal_order(
336  back1post, back0pre, false, v_to_rev_depth, v_to_qb))
337  continue;
338  if (circ.in_causal_order(
339  back0post, back1pre, false, v_to_rev_depth, v_to_qb))
340  continue;
341  // Identify which pattern
342  bool vs_on_q0 = false;
343  if (back0_index >= front0_index && back0_index != (unsigned)-1) {
344  if (!(back0_index == front0_index + 1 &&
345  circ.get_OpType_from_Vertex(path0[front0_index].first) ==
346  OpType::V &&
347  circ.get_OpType_from_Vertex(path0[back0_index].first) ==
348  OpType::S)) {
349  // The only non-empty pattern we can recognise is from VS
350  continue;
351  }
352  vs_on_q0 = true;
353  }
354  bool sv_on_q1 = false;
355  if (back1_index >= front1_index && back1_index != (unsigned)-1) {
356  if (!(back1_index == front1_index + 1 &&
357  circ.get_OpType_from_Vertex(path1[front1_index].first) ==
358  OpType::S &&
359  circ.get_OpType_from_Vertex(path1[back1_index].first) ==
360  OpType::V)) {
361  // The only non-empty pattern we can recognise is from SV
362  continue;
363  }
364  sv_on_q1 = true;
365  }
366  if (!allow_swaps && !vs_on_q0 && !sv_on_q1) {
367  continue;
368  }
369  success = true;
370  // Always remove the CX pair, saving the edges to insert the replacement
371  // circuit
372  Vertex before0, before1, after0, after1;
373  port_t pb0, pb1, pa0, pa1;
374  if (front0_index > 0) {
375  before0 = path0[front0_index - 1].first;
376  pb0 = path0[front0_index - 1].second;
377  } else {
378  Edge pre0 = circ.get_nth_in_edge(v, 0);
379  before0 = circ.source(pre0);
380  pb0 = circ.get_source_port(pre0);
381  }
382  if (front1_index > 0) {
383  before1 = path1[front1_index - 1].first;
384  pb1 = path1[front1_index - 1].second;
385  } else {
386  Edge pre1 = circ.get_nth_in_edge(v, 1);
387  before1 = circ.source(pre1);
388  pb1 = circ.get_source_port(pre1);
389  }
390  if (back0_index + 1 < path0_length) {
391  after0 = path0[back0_index + 1].first;
392  pa0 = path0[back0_index + 1].second;
393  } else {
394  Edge post0 = circ.get_nth_out_edge(v0, 1);
395  after0 = circ.target(post0);
396  pa0 = circ.get_target_port(post0);
397  }
398  if (back1_index + 1 < path1_length) {
399  after1 = path1[back1_index + 1].first;
400  pa1 = path1[back1_index + 1].second;
401  } else {
402  Edge post1 = circ.get_nth_out_edge(v0, 0);
403  after1 = circ.target(post1);
404  pa1 = circ.get_target_port(post1);
405  }
406  bin.push_back(v);
407  bin.push_back(v0);
408  circ.remove_vertex(
409  v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
410  circ.remove_vertex(
411  v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
412  VertexSet to_remove;
413  const Circuit *replacement;
414  Qubit q0, q1;
415  if (vs_on_q0) {
416  Vertex v_gate = path0[front0_index].first;
417  q0 = Qubit(*v_to_qb.at(v_gate).begin());
418  q1 = qbs.at(0);
419  if (q0 == q1) q1 = qbs.at(1);
420  if (sv_on_q1) {
421  /* --C--V--S--X-- --V--S-- */
422  /* | | => */
423  /* --X--S--V--C-- --S--V-- */
424  continue;
425  } else {
426  /* --C--V--S--X-- --S-----C--V--S-- */
427  /* | | => | */
428  /* --X--------C-- --Z--S--X--S----- */
429  replacement = &CircPool::CX_V_S_XC_reduced();
430  }
431  bin.push_back(path0[front0_index].first);
432  bin.push_back(path0[back0_index].first);
433  to_remove.insert(path0[front0_index].first);
434  to_remove.insert(path0[back0_index].first);
435  } else if (sv_on_q1) {
436  /* --C--------X-- --X--V--C--V----- */
437  /* | | => | */
438  /* --X--S--V--C-- --V-----X--S--V-- */
439  Vertex s_gate = path1[front1_index].first;
440  q1 = Qubit(*v_to_qb.at(s_gate).begin());
441  q0 = qbs.at(0);
442  if (q0 == q1) q0 = qbs.at(1);
443  replacement = &CircPool::CX_S_V_XC_reduced();
444  bin.push_back(path1[front1_index].first);
445  bin.push_back(path1[back1_index].first);
446  to_remove.insert(path1[front1_index].first);
447  to_remove.insert(path1[back1_index].first);
448  } else {
449  /* --C--X-- --X--\ /-- */
450  /* | | => | \ */
451  /* --X--C-- --C--/ \-- */
452  q0 = qbs.at(0);
453  q1 = qbs.at(1);
454  replacement = &CircPool::CX_XC_reduced();
455  }
456  Subcircuit sub = {
457  {circ.get_nth_out_edge(before0, pb0),
458  circ.get_nth_out_edge(before1, pb1)},
459  {circ.get_nth_in_edge(after0, pa0),
460  circ.get_nth_in_edge(after1, pa1)},
461  to_remove};
462  Vertex b0 = circ.source(sub.q_in_hole[0]);
463  port_t p0 = circ.get_source_port(sub.q_in_hole[0]);
464  Vertex b1 = circ.source(sub.q_in_hole[1]);
465  port_t p1 = circ.get_source_port(sub.q_in_hole[1]);
466  Vertex a0 = circ.target(sub.q_out_hole[0]);
467  Vertex a1 = circ.target(sub.q_out_hole[1]);
468  circ.substitute(*replacement, sub, Circuit::VertexDeletion::No);
469  // Scan through hole and add vertices to v_to_qb
470  unsigned new_depth = std::min(v_to_depth[a0], v_to_depth[a1]);
471  unsigned new_rev_depth =
472  std::min(v_to_rev_depth[b0], v_to_rev_depth[b1]);
473  Edge ei0 = circ.get_nth_out_edge(b0, p0);
474  Vertex vi0 = circ.target(ei0);
475  while (vi0 != a0) {
476  v_to_qb.insert({vi0, {q0}});
477  v_to_depth[vi0] = new_depth;
478  v_to_rev_depth[vi0] = new_rev_depth;
479  ei0 = circ.get_next_edge(vi0, ei0);
480  vi0 = circ.target(ei0);
481  }
482  Edge ei1 = circ.get_nth_out_edge(b1, p1);
483  Vertex vi1 = circ.target(ei1);
484  while (vi1 != a1) {
485  if (v_to_qb.find(vi1) == v_to_qb.end())
486  v_to_qb.insert({vi1, {q1}});
487  else
488  v_to_qb.at(vi1).insert(q1);
489  v_to_depth[vi1] = new_depth;
490  v_to_rev_depth[vi1] = new_rev_depth;
491  ei1 = circ.get_next_edge(vi1, ei1);
492  vi1 = circ.target(ei1);
493  }
494  }
495  }
496  }
497  if (allow_swaps) {
498  circ.replace_SWAPs();
499  }
500  circ.remove_vertices(
501  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
502  return success;
503 }

507 // TODO:: Copy classical-controls and any controls from CX
508 static bool copy_pi_through_CX_method(Circuit &circ) {
509  bool success = false;
510  VertexList bin;
511  BGL_FORALL_VERTICES(vert, circ.dag, DAG) {
512  if (circ.get_OpType_from_Vertex(vert) == OpType::CX &&

552 // TODO:: Detect classical controls and stop simplification
553 static bool singleq_clifford_from_edge(
554  Circuit &circ, Edge &e, VertexList &bin) {
555  VertexSet single_vs;
556  Edge ei = e;
557  Vertex vi = circ.target(ei);
558  // stuff to check if it is already in standard form
559  // 6 = not consumed anything, 5=Z, 4=X, 3=first S, 2=V, 1=second S, 0=other
560  unsigned cliff_last = 6;
561  while (circ.detect_singleq_unitary_op(vi)) {
562  // i.e. we have a single qubit unitary
563  single_vs.insert(vi);
564 
565  // keep track of clifford standard form
566  switch (circ.get_OpType_from_Vertex(vi)) {
567  case OpType::Z: {
568  cliff_last = (cliff_last > 5) ? 5 : 0;
569  break;
570  }
571  case OpType::X: {
572  cliff_last = (cliff_last > 4) ? 4 : 0;
573  break;
574  }
575  case OpType::S: {
576  if (cliff_last > 3)
577  cliff_last = 3;
578  else if (cliff_last == 2)
579  cliff_last = 1;
580  else
581  cliff_last = 0;
582  break;
583  }
584  case OpType::V: {
585  cliff_last = (cliff_last > 2) ? 2 : 0;
586  break;
587  }
588  default: {
589  cliff_last = 0;
590  break;
591  }
592  }
593 
594  ei = circ.get_next_edge(vi, ei);
595  vi = circ.target(ei);
596  }
597  // if the sequence is not in the standard form, replace it
598  if (cliff_last == 0) {
599  Subcircuit s = {{e}, {ei}, single_vs};
600  Circuit sub = circ.subcircuit(s);
601  bool reduced = (decompose_single_qubits_TK1() >> squash_1qb_to_tk1() >>
602  decompose_cliffords_std())
603  .apply(sub);
604  if (reduced) {
605  circ.substitute(sub, s, Circuit::VertexDeletion::No);
606  bin.insert(bin.end(), single_vs.begin(), single_vs.end());
607  return true;
608  }
609  }
610  return false;
611 }
