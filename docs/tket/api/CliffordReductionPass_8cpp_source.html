27static const std::map<std::pair<Pauli, Pauli>, std::list<OpType>>
28 mapping_to_zz_or_zy_lut{
29 {{Pauli::X, Pauli::X}, {OpType::H}},
30 {{Pauli::X, Pauli::Y}, {OpType::H, OpType::Z}},
31 {{Pauli::X, Pauli::Z}, {OpType::H, OpType::S}},
32 {{Pauli::Y, Pauli::X}, {OpType::V, OpType::S}},
33 {{Pauli::Y, Pauli::Y}, {OpType::V}},
34 {{Pauli::Y, Pauli::Z}, {OpType::V, OpType::Z}},
35 {{Pauli::Z, Pauli::X}, {OpType::S}},
36 {{Pauli::Z, Pauli::Y}, {}},
37 {{Pauli::Z, Pauli::Z}, {}},
38 };

40static const std::map<Pauli, OpType> pauli_to_pauli_gate_lut{
41 {Pauli::X, OpType::X},
42 {Pauli::Y, OpType::Y},
43 {Pauli::Z, OpType::Z},
44};

51static Circuit interaction_replacement(const InteractionMatch &match) {
52 const Pauli &p0 = match.point0.p;
53 const Pauli &p1 = match.point1.p;
54 const Pauli &q0 = match.rev0.p;
55 const Pauli &q1 = match.rev1.p;
56 Circuit replacement(2);
57 if (match.point0.phase ^ match.point1.phase) {
58 replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(p0), {0});
59 replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(p1), {1});
60 replacement.add_phase(0.5);
61 }
62 if (p0 == q0) {
63 if (p1 == q1) {
64 // R[p0, p1](1) = R[p0, I](1); R[I, p1](1)
65 OpType op0 = pauli_to_pauli_gate_lut.at(p0);
66 OpType op1 = pauli_to_pauli_gate_lut.at(p1);
67 replacement.add_op<unsigned>(op0, {0});
68 replacement.add_op<unsigned>(op1, {1});
69 replacement.add_phase(-0.5);
70 } else {
71 // Map to R[Z, Z](0.5); R[Z, Y](0.5)
72 Circuit basis_change(2);
73 std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
74 std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
75 for (OpType op : ops0) {
76 basis_change.add_op<unsigned>(op, {0});
77 }
78 for (OpType op : ops1) {
79 basis_change.add_op<unsigned>(op, {1});
80 }
81 replacement.append(basis_change);
82 replacement.add_op<unsigned>(OpType::V, {1});
83 replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
84 replacement.append(basis_change.dagger());
85 }
86 } else {
87 if (p1 == q1) {
88 // Map to R[Z, Z](0.5); R[Y, Z](0.5)
89 Circuit basis_change(2);
90 std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
91 std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
92 for (OpType op : ops0) {
93 basis_change.add_op<unsigned>(op, {0});
94 }
95 for (OpType op : ops1) {
96 basis_change.add_op<unsigned>(op, {1});
97 }
98 replacement.append(basis_change);
99 replacement.add_op<unsigned>(OpType::V, {0});
100 replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
101 replacement.append(basis_change.dagger());
102 } else {
103 // Map to R[Z, Z](0.5); R[Y, Y](0.5)
104 Circuit basis_change(2);
105 std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
106 std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
107 for (OpType op : ops0) {
108 basis_change.add_op<unsigned>(op, {0});
109 }
110 for (OpType op : ops1) {
111 basis_change.add_op<unsigned>(op, {1});
112 }
113 replacement.append(basis_change);
114 replacement.add_op<unsigned>(OpType::H, {0});
115 replacement.add_op<unsigned>(OpType::H, {1});
116 replacement.add_op<unsigned>(OpType::Z, {0});
117 replacement.add_op<unsigned>(OpType::Z, {1});
118 replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
119 replacement.add_op<unsigned>(OpType::H, {0});
120 replacement.add_op<unsigned>(OpType::H, {1});
121 replacement.add_op<unsigned>(OpType::SWAP, {0, 1});
122 replacement.add_phase(0.25);
123 replacement.append(basis_change.dagger());
124 }
125 }
126 if (match.rev0.phase ^ match.rev1.phase) {
127 replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(q0), {0});
128 replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(q1), {1});
129 replacement.add_phase(0.5);
130 }
131 return replacement;
132}

138static Circuit local_cliffords(OpType op) {
139 Circuit locals(2);
140 switch (op) {
141 case OpType::CX: {
142 locals.add_op<unsigned>(OpType::Sdg, {0});
143 locals.add_op<unsigned>(OpType::Vdg, {1});
144 break;
145 }
146 case OpType::CZ: {
147 locals.add_op<unsigned>(OpType::Sdg, {0});
148 locals.add_op<unsigned>(OpType::Sdg, {1});
149 locals.add_phase(0.25);
150 break;
151 }
152 case OpType::CY: {
153 locals.add_op<unsigned>(OpType::Sdg, {0});
154 locals.add_op<unsigned>(OpType::V, {1});
155 locals.add_op<unsigned>(OpType::Sdg, {1});
156 locals.add_op<unsigned>(OpType::Vdg, {1});
157 locals.add_phase(0.25);
158 break;
159 }
160 case OpType::ZZMax: {
161 break;
162 }
163 default: {
164 throw CircuitInvalidity(
165 "Attempting to replace non-Clifford gate with Clifford "
166 "optimisation");
167 break;
168 }
169 }
170 return locals;
171}

173void CliffordReductionPass::insert_interaction_point(InteractionPoint ip) {
174 itable.insert(ip);
175 Vertex next = circ.target(ip.e);
176 port_t next_p = circ.get_target_port(ip.e);
177 bool commute = true;
178 while (commute) {
179 if (v_to_depth.find(next) == v_to_depth.end()) {
180 commute = false;
181 continue;
182 }
183 Op_ptr op = circ.get_Op_ptr_from_Vertex(next);
184 if (!op->get_desc().is_gate()) {
185 commute = false;
186 continue;
187 }
188 OpType type = op->get_type();
189 switch (type) {
190 case OpType::H:
191 case OpType::S:
192 case OpType::Sdg:
193 case OpType::V:
194 case OpType::Vdg:
195 case OpType::X:
196 case OpType::Y:
197 case OpType::Z: {
198 std::pair<Pauli, bool> new_basis = conjugate_Pauli(type, ip.p, true);
199 ip.p = new_basis.first;
200 ip.phase ^= new_basis.second;
201 break;
202 }
203 case OpType::SWAP: {
204 next_p = 1 - next_p;
205 break;
206 }
207 default: {
208 if (!circ.commutes_with_basis(next, ip.p, next_p)) {
209 commute = false;
210 continue;
211 }
212 break;
213 }
214 }
215 ip.e = circ.get_nth_out_edge(next, next_p);
216 auto inserted = itable.insert(ip);
217 commute = inserted.second;
218 if (!commute) {
219 // Now `inserted.first` points to the element of the table that
220 // blocked insertion, i.e. had the same source/edge combination.
221 // Check that its `p` and `phase` are correct: if not, something has
222 // gone wrong.
223 auto blocker = inserted.first;
224 TKET_ASSERT(blocker->p == ip.p && blocker->phase == ip.phase);
225 }
226 next = circ.target(ip.e);
227 next_p = circ.get_target_port(ip.e);
228 }
229}

231std::optional<InteractionMatch> CliffordReductionPass::search_back_for_match(
232 const RevInteractionPoint &rip0, const RevInteractionPoint &rip1) const {
233 RevInteractionPoint point[2];
234 point[0] = rip0;
235 point[1] = rip1;
236 std::map<Edge, RevInteractionPoint> point_lookup;
237
238 // interactions met when commuting back; point lists are in causal order of
239 // circuit:
240 std::map<IVertex, std::list<InteractionPoint>> candidates[2];
241
242 for (unsigned i = 0; i < 2; ++i) {
243 // Commute edge i back as far as possible
244 bool commute = true;
245 while (commute) {
246 point_lookup.insert({point[i].e, point[i]});
247 auto r = itable.get<TagEdge>().equal_range(point[i].e);
248 for (auto it = r.first; it != r.second; ++it) {
249 Vertex v = it->source;
250 candidates[i][{im.at(v), v}].push_front(*it);
251 }
252 Vertex pred = circ.source(point[i].e);
253 port_t pred_port = circ.get_source_port(point[i].e);
254 Op_ptr pred_op = circ.get_Op_ptr_from_Vertex(pred);
255 if (!pred_op->get_desc().is_gate()) {
256 commute = false;
257 continue;
258 }
259 OpType type = pred_op->get_type();
260 switch (type) {
261 case OpType::H:
262 case OpType::S:
263 case OpType::Sdg:
264 case OpType::V:
265 case OpType::Vdg:
266 case OpType::X:
267 case OpType::Y:
268 case OpType::Z: {
269 std::pair<Pauli, bool> new_basis = conjugate_Pauli(type, point[i].p);
270 point[i].p = new_basis.first;
271 point[i].phase ^= new_basis.second;
272 break;
273 }
274 case OpType::SWAP: {
275 pred_port = 1 - pred_port;
276 break;
277 }
278 default: {
279 commute = circ.commutes_with_basis(pred, point[i].p, pred_port);
280 break;
281 }
282 }
283 point[i].e = circ.get_nth_in_edge(pred, pred_port);
284 }
285 }
286 // Check for matching interactions
287 for (const std::pair<const IVertex, std::list<InteractionPoint>> &pair :
288 candidates[0]) {
289 auto found = candidates[1].find(pair.first);
290 if (found != candidates[1].end()) {
291 std::optional<std::pair<InteractionPoint, InteractionPoint>>
292 insert_point = valid_insertion_point(pair.second, found->second);
293 if (insert_point) {
294 InteractionMatch match = {
295 insert_point->first, insert_point->second,
296 point_lookup.at(insert_point->first.e),
297 point_lookup.at(insert_point->second.e)};
298 if (!allow_swaps) {
299 if (match.point0.p != match.rev0.p && match.point1.p != match.rev1.p)
300 continue;
301 }
302 return match;
303 }
304 }
305 }
306 return std::nullopt;
307}

309void CliffordReductionPass::process_new_interaction(const Vertex &inter) {
310 // Process the vertex as well as any new 2qb Cliffords that get inserted
311 // while doing so (and so on until there are none left to process).
312 std::list<Vertex> to_process = {inter};
313 while (!to_process.empty()) {
314 Vertex v = to_process.front();
315 to_process.pop_front();
316 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
317 Pauli basis0 = *op->commuting_basis(0);
318 Pauli basis1 = *op->commuting_basis(1);
319 EdgeVec ins = circ.get_in_edges(v);
320 RevInteractionPoint rip0 = {ins.at(0), basis0, false};
321 RevInteractionPoint rip1 = {ins.at(1), basis1, false};
322 std::optional<InteractionMatch> match = search_back_for_match(rip0, rip1);
323 if (match) {
324 Circuit replacement = interaction_replacement(*match);
325 Subcircuit site;
326 site.in_hole = {match->point0.e, match->point1.e};
327 site.out_hole = {match->point0.e, match->point1.e};
328 Subcircuit inserted = substitute(replacement, site);
329 const Vertex &source = match->point0.source;
330 Circuit source_locals =
331 local_cliffords(circ.get_OpType_from_Vertex(source));
332 Subcircuit source_site = circ.singleton_subcircuit(source);
333 substitute(source_locals, source_site);
334 Circuit v_locals = local_cliffords(op->get_type());
335 Subcircuit v_site = circ.singleton_subcircuit(v);
336 substitute(v_locals, v_site);
337 for (const Vertex &new_v : inserted.verts) {
338 if (circ.n_in_edges(new_v) == 2 &&
339 circ.get_OpType_from_Vertex(new_v) != OpType::SWAP) {
340 to_process.push_back(new_v);
341 break;
342 }
343 }
344 im = circ.index_map();
345 success = true;
346 } else {
347 std::vector<std::optional<Edge>> outs = circ.get_linear_out_edges(v);
348 InteractionPoint ip0 = {*outs.at(0), v, basis0, false};
349 insert_interaction_point(ip0);
350 InteractionPoint ip1 = {*outs.at(1), v, basis1, false};
351 insert_interaction_point(ip1);
352 }
353 }

356Subcircuit CliffordReductionPass::substitute(
357 const Circuit &to_insert, const Subcircuit &to_replace) {
358 unsigned q_width = to_replace.in_hole.size();
359 TKET_ASSERT(q_width == 2);
360
361 // Construct tables of predecessors, successors, units, in-edges, out-edges.
362 // Only quantum circuit replacments here so don't care about classical stuff
363 std::vector<VertPort> preds(q_width);
364 std::vector<VertPort> succs(q_width);
365 std::vector<UnitID> units(q_width);
366 std::vector<Edge> in_edges(q_width);
367 std::vector<Edge> out_edges(q_width);
368 for (unsigned qi = 0; qi < q_width; ++qi) {
369 const Edge &in = to_replace.in_hole.at(qi);
370 const Edge &out = *to_replace.out_hole.at(qi);
371 preds[qi] = {circ.source(in), circ.get_source_port(in)};
372 succs[qi] = {circ.target(out), circ.get_target_port(out)};
373 units[qi] = e_to_unit.at(in);
374 in_edges[qi] = in;
375 out_edges[qi] = out;
376 TKET_ASSERT(in == out || circ.target(in) == circ.source(out));
377 }
378
379 // List of points that will be invalidated by the substitution.
380 std::list<InteractionPoint> invalidated_points;
381
382 // Lists of points having the same "in"/"out" edge as the replacement:
383 // These are all invalidated (though the "in" ones can be replaced later
384 // with the new edge).
385 std::vector<std::list<InteractionPoint>> points_with_in(q_width);
386 std::vector<std::list<InteractionPoint>> points_with_out(q_width);
387 for (unsigned qi = 0; qi < q_width; ++qi) {
388 auto r = itable.get<TagEdge>().equal_range(in_edges[qi]);
389 for (auto it = r.first; it != r.second; it++) {
390 points_with_in[qi].push_back(*it);
391 invalidated_points.push_back(*it);
392 }
393 r = itable.get<TagEdge>().equal_range(out_edges[qi]);
394 for (auto it = r.first; it != r.second; it++) {
395 points_with_out[qi].push_back(*it);
396 invalidated_points.push_back(*it);
397 }
398 }
399
400 // For any (e0, v0) in points_with_out, any point (e1, v0) where e1 is in
401 // the causal future of e0 is also invalidated. Calculate all the future
402 // edges.
403 EdgeList future_edges;
404 VertexSet v_frontier;
405 // keep track of visited vertices
406 VertexSet visited;
407 for (unsigned qi = 0; qi < q_width; ++qi) {
408 Vertex target = circ.target(out_edges[qi]);
409 visited.insert(target);
410 v_frontier.insert(target);
411 }
412 while (!v_frontier.empty()) {
413 EdgeSet out_edges;
414 for (auto v : v_frontier) {
415 if (v_to_depth.find(v) != v_to_depth.end()) {
416 EdgeVec v_out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
417 out_edges.insert(v_out_edges.begin(), v_out_edges.end());
418 }
419 }
420 future_edges.insert(future_edges.end(), out_edges.begin(), out_edges.end());
421 VertexSet new_v_frontier;
422 for (auto e : out_edges) {
423 Vertex target = circ.target(e);
424 auto it = visited.insert(target);
425 if (it.second) {
426 // add vertex to the next frontier if the vertex has not been visited
427 new_v_frontier.insert(target);
428 }
429 }
430 v_frontier = std::move(new_v_frontier);
431 }
432
433 // Invalidate the (e1, v0) as above.
434 VertexSet invalid_sources;
435 for (unsigned qi = 0; qi < q_width; ++qi) {
436 for (auto ip : points_with_out[qi]) {
437 TKET_ASSERT(ip.e == out_edges[qi]);
438 invalid_sources.insert(ip.source);
439 }
440 }
441 for (auto e1 : future_edges) {
442 auto r = itable.get<TagEdge>().equal_range(e1);
443 for (auto it = r.first; it != r.second; ++it) {
444 if (invalid_sources.find(it->source) != invalid_sources.end()) {
445 invalidated_points.push_back(*it);
446 }
447 }
448 }
449
450 // Erase the invalidated points from the table.
451 for (auto ip : invalidated_points) {
452 itable.erase(ip.key());
453 }
454
455 // Erase edges from e_to_unit
456 for (unsigned qi = 0; qi < q_width; ++qi) {
457 e_to_unit.erase(in_edges[qi]);
458 e_to_unit.erase(out_edges[qi]);
459 // Depth of to_replace is at most 1, so there are no other edges
460 }
461
462 // Remove replaced vertices from depth and units maps and erase all points
463 // from the itable that have a replaced vertex as source.
464 for (const Vertex &v : to_replace.verts) {
465 v_to_depth.erase(v);
466 v_to_units.erase(v);
467 auto r = itable.get<TagSource>().equal_range(v);
468 for (auto next = r.first; next != r.second; r.first = next) {
469 ++next;
470 itable.erase(itable.project<TagKey>(r.first));
471 }
472 }
473
474 circ.substitute(to_insert, to_replace);
475
476 // Update the tables of in and out edges, and amend the stored points
477 for (unsigned qi = 0; qi < q_width; ++qi) {
478 in_edges[qi] = circ.get_nth_out_edge(preds[qi].first, preds[qi].second);
479 out_edges[qi] = circ.get_nth_in_edge(succs[qi].first, succs[qi].second);
480 for (InteractionPoint &ip : points_with_in[qi]) {
481 ip.e = in_edges[qi];
482 }
483 }
484
485 // Construct inserted Subcircuit, update e_to_unit and v_to_units, and
486 // add new vertices to v_to_depth, with (temporary) value 0.
487 Subcircuit inserted;
488 for (unsigned qi = 0; qi < q_width; ++qi) {
489 Edge in = in_edges[qi];
490 inserted.in_hole.push_back(in);
491 Edge out = out_edges[qi];
492 inserted.out_hole.push_back(out);
493 while (in != out) {
494 Vertex next = circ.target(in);
495 inserted.verts.insert(next);
496 e_to_unit.insert({in, units[qi]});
497 v_to_depth.insert({next, 0});
498 v_to_units[next].insert(units[qi]);
499 in = circ.get_nth_out_edge(next, circ.get_target_port(in));
500 }
501 e_to_unit.insert({in, units[qi]});
502 }
503
504 // Now `v_to_depth` is 0 at all `inserted.verts`. Fix this and propagate
505 // updates to the depth map into the future cone, ensuring that the depths
506 // are strictly increasing along wires. Stop when we reach a vertex that
507 // isn't already in v_to_depth (because we haven't processed it yet).
508 for (unsigned qi = 0; qi < q_width; ++qi) {
509 Edge in = in_edges[qi];
510 Vertex next = circ.target(in);
511 if (next != succs[qi].first) {
512 if (v_to_depth.at(preds[qi].first) >= v_to_depth[next]) {
513 // We may have already set v_to_depth[next] to a higher value
514 // when tracing another qubit, so the check above is necessary.
515 v_to_depth[next] = v_to_depth.at(preds[qi].first) + 1;
516 }
517 std::function<bool(const Vertex &, const Vertex &)> c =
518 [&](const Vertex &a, const Vertex &b) {
519 unsigned deptha = v_to_depth.at(a);
520 unsigned depthb = v_to_depth.at(b);
521 if (deptha == depthb) {
522 unit_set_t unitsa = v_to_units.at(a);
523 unit_set_t unitsb = v_to_units.at(b);
524 return unitsa < unitsb;
525 }
526 return deptha < depthb;
527 };
528 std::set<Vertex> to_search;
529 to_search.insert(next);
530 while (!to_search.empty()) {
531 Vertex v = *std::min_element(to_search.begin(), to_search.end(), c);
532 to_search.erase(v);
533 unsigned v_depth = v_to_depth.at(v);
534 EdgeVec outs = circ.get_all_out_edges(v);
535 for (const Edge &e : outs) {
536 Vertex succ = circ.target(e);
537 std::map<Vertex, unsigned>::iterator succ_it = v_to_depth.find(succ);
538 if (succ_it != v_to_depth.end()) {
539 if (succ_it->second <= v_depth) {
540 succ_it->second = v_depth + 1;
541 if (v_depth >= current_depth) {
542 current_depth = v_depth + 1;
543 }
544 to_search.insert(succ);
545 }
546 }
547 }
548 }
549 }
550 }
551
552 // Re-insert all the interaction points we erased above.
553 for (unsigned qi = 0; qi < q_width; ++qi) {
554 for (const InteractionPoint &ip : points_with_in[qi]) {
555 insert_interaction_point(ip);
556 }
557 }
558
559 return inserted;
560}

562std::optional<Edge> CliffordReductionPass::find_earliest_successor(
563 const Edge &source, const EdgeSet &candidates) const {
564 typedef std::function<bool(Vertex, Vertex)> Comp;
565 Comp c = [&](Vertex a, Vertex b) {
566 unsigned deptha = v_to_depth.at(a);
567 unsigned depthb = v_to_depth.at(b);
568 if (deptha == depthb) {
569 unit_set_t unitsa = v_to_units.at(a);
570 unit_set_t unitsb = v_to_units.at(b);
571 return unitsa < unitsb;
572 }
573 return deptha < depthb;
574 };
575 std::set<Vertex, Comp> to_search(c);
576 to_search.insert(circ.target(source));
577 while (!to_search.empty()) {
578 Vertex v = *to_search.begin();
579 to_search.erase(to_search.begin());
580 EdgeVec outs = circ.get_all_out_edges(v);
581 for (const Edge &e : outs) {
582 if (candidates.find(e) != candidates.end()) return e;
583 Vertex succ = circ.target(e);
584 if (v_to_depth.find(succ) != v_to_depth.end()) to_search.insert(succ);
585 }
586 }
587 return std::nullopt;
588}

590std::optional<std::pair<InteractionPoint, InteractionPoint>>
591CliffordReductionPass::valid_insertion_point(
592 const std::list<InteractionPoint> &seq0,
593 const std::list<InteractionPoint> &seq1) const {
594 // seq0 is chain of edges (in temporal order) from the first qubit
595 // likewise seq1 for the other qubit
596 InteractionPoint seq0max = seq0.back();
597 InteractionPoint seq1max = seq1.back();
598 if (circ.in_causal_order(
599 circ.source(seq1max.e), circ.target(seq0max.e), true, v_to_depth,
600 v_to_units, false)) {
601 // Search for any points in seq1 from future of seq0max
602 EdgeSet candidates;
603 std::map<Edge, InteractionPoint> lookup;
604 for (const InteractionPoint &ip : seq1) {
605 candidates.insert(ip.e);
606 lookup.insert({ip.e, ip});
607 }
608 std::optional<Edge> successor =
609 find_earliest_successor(seq0max.e, candidates);
610 if (!successor || successor == seq1.front().e) return std::nullopt;
611 Vertex v = circ.source(*successor);
612 port_t p = circ.get_source_port(*successor);
613 if (circ.get_OpType_from_Vertex(v) == OpType::SWAP) p = 1 - p;
614 return {{seq0max, lookup.at(circ.get_nth_in_edge(v, p))}};
615 } else if (circ.in_causal_order(
616 circ.source(seq0max.e), circ.target(seq1max.e), true,
617 v_to_depth, v_to_units, false)) {
618 // Search for any points in seq0 from future of seq1max
619 EdgeSet candidates;
620 std::map<Edge, InteractionPoint> lookup;
621 for (const InteractionPoint &ip : seq0) {
622 candidates.insert(ip.e);
623 lookup.insert({ip.e, ip});
624 }
625 std::optional<Edge> successor =
626 find_earliest_successor(seq1max.e, candidates);
627 if (!successor || successor == seq0.front().e) return std::nullopt;
628 Vertex v = circ.source(*successor);
629 port_t p = circ.get_source_port(*successor);
630 if (circ.get_OpType_from_Vertex(v) == OpType::SWAP) p = 1 - p;
631 return {{lookup.at(circ.get_nth_in_edge(v, p)), seq1max}};
632 } else {
633 // seq0max and seq1max are space-like separated
634 return {{seq0max, seq1max}};
635 }
636}

650bool CliffordReductionPass::reduce_circuit(Circuit &circ, bool allow_swaps) {
651 CliffordReductionPass context(circ, allow_swaps);
652
653 SliceVec slices = circ.get_slices();
654
655 for (const Vertex &in : circ.all_inputs()) {
656 context.v_to_depth.insert({in, 0});
657 }
658
659 // Process all 2qb Clifford vertices.
660 for (const Slice &sl : slices) {
661 for (const Vertex &v : sl) {
662 context.v_to_depth.insert({v, context.current_depth});
663 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
664 if (!op->get_desc().is_gate()) continue;
665 EdgeVec ins = circ.get_in_edges(v);
666 std::vector<std::optional<Edge>> outs = circ.get_linear_out_edges(v);
667 OpType type = op->get_type();
668 std::list<InteractionPoint> new_points;
669 switch (type) {
670 case OpType::H:
671 case OpType::S:
672 case OpType::Sdg:
673 case OpType::V:
674 case OpType::Vdg:
675 case OpType::X:
676 case OpType::Y:
677 case OpType::Z: {
678 auto r = context.itable.get<TagEdge>().equal_range(ins[0]);
679 for (auto it = r.first; it != r.second; ++it) {
680 InteractionPoint ip = *it;
681 std::pair<Pauli, bool> new_basis =
682 conjugate_Pauli(type, ip.p, true);
683 ip.p = new_basis.first;
684 ip.phase ^= new_basis.second;
685 ip.e = *outs[0];
686 new_points.push_back(ip);
687 }
688 break;
689 }
690 case OpType::SWAP: {
691 auto r0 = context.itable.get<TagEdge>().equal_range(ins[0]);
692 for (auto it = r0.first; it != r0.second; ++it) {
693 InteractionPoint ip = *it;
694 ip.e = *outs[1];
695 new_points.push_back(ip);
696 }
697 auto r1 = context.itable.get<TagEdge>().equal_range(ins[1]);
698 for (auto it = r1.first; it != r1.second; ++it) {
699 InteractionPoint ip = *it;
700 ip.e = *outs[0];
701 new_points.push_back(ip);
702 }
703 break;
704 }
705 default: {
706 for (unsigned i = 0; i < ins.size(); ++i) {
707 auto r = context.itable.get<TagEdge>().equal_range(ins[i]);
708 for (auto it = r.first; it != r.second; ++it) {
709 InteractionPoint ip = *it;
710 if (circ.commutes_with_basis(v, ip.p, i)) {
711 ip.e = *outs[i];
712 new_points.push_back(ip);
713 }
714 }
715 }
716 break;
717 }
718 }
719 for (const InteractionPoint &ip : new_points) {
720 context.itable.insert(ip);
721 }
722 switch (type) {
723 case OpType::CX:
724 case OpType::CY:
725 case OpType::CZ:
726 case OpType::ZZMax: {
727 context.process_new_interaction(v);
728 break;
729 }
730 default: {
731 break;
732 }
733 }
734 }
735 ++context.current_depth;
736 }
737
738 if (allow_swaps) {
739 circ.replace_SWAPs();
740 }
741
742 return context.success;
743}
