42  {Pauli::Y, OpType::Y},

51 static Circuit interaction_replacement(const InteractionMatch &match) {
52  const Pauli &p0 = match.point0.p;
53  const Pauli &p1 = match.point1.p;
54  const Pauli &q0 = match.rev0.p;
55  const Pauli &q1 = match.rev1.p;
56  Circuit replacement(2);
57  if (match.point0.phase ^ match.point1.phase) {
58  replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(p0), {0});
59  replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(p1), {1});
60  replacement.add_phase(0.5);
61  }
62  if (p0 == q0) {
63  if (p1 == q1) {
64  // R[p0, p1](1) = R[p0, I](1); R[I, p1](1)
65  OpType op0 = pauli_to_pauli_gate_lut.at(p0);
66  OpType op1 = pauli_to_pauli_gate_lut.at(p1);
67  replacement.add_op<unsigned>(op0, {0});
68  replacement.add_op<unsigned>(op1, {1});
69  replacement.add_phase(-0.5);
70  } else {
71  // Map to R[Z, Z](0.5); R[Z, Y](0.5)
72  Circuit basis_change(2);
73  std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
74  std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
75  for (OpType op : ops0) {
76  basis_change.add_op<unsigned>(op, {0});
77  }
78  for (OpType op : ops1) {
79  basis_change.add_op<unsigned>(op, {1});
80  }
81  replacement.append(basis_change);
82  replacement.add_op<unsigned>(OpType::V, {1});
83  replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
84  replacement.append(basis_change.dagger());
85  }
86  } else {
87  if (p1 == q1) {
88  // Map to R[Z, Z](0.5); R[Y, Z](0.5)
89  Circuit basis_change(2);
90  std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
91  std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
92  for (OpType op : ops0) {
93  basis_change.add_op<unsigned>(op, {0});
94  }
95  for (OpType op : ops1) {
96  basis_change.add_op<unsigned>(op, {1});
97  }
98  replacement.append(basis_change);
99  replacement.add_op<unsigned>(OpType::V, {0});
100  replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
101  replacement.append(basis_change.dagger());
102  } else {
103  // Map to R[Z, Z](0.5); R[Y, Y](0.5)
104  Circuit basis_change(2);
105  std::list<OpType> ops0 = mapping_to_zz_or_zy_lut.at({p0, q0});
106  std::list<OpType> ops1 = mapping_to_zz_or_zy_lut.at({p1, q1});
107  for (OpType op : ops0) {
108  basis_change.add_op<unsigned>(op, {0});
109  }
110  for (OpType op : ops1) {
111  basis_change.add_op<unsigned>(op, {1});
112  }
113  replacement.append(basis_change);
114  replacement.add_op<unsigned>(OpType::H, {0});
115  replacement.add_op<unsigned>(OpType::H, {1});
116  replacement.add_op<unsigned>(OpType::Z, {0});
117  replacement.add_op<unsigned>(OpType::Z, {1});
118  replacement.add_op<unsigned>(OpType::ZZMax, {0, 1});
119  replacement.add_op<unsigned>(OpType::H, {0});
120  replacement.add_op<unsigned>(OpType::H, {1});
121  replacement.add_op<unsigned>(OpType::SWAP, {0, 1});
122  replacement.add_phase(0.25);
123  replacement.append(basis_change.dagger());
124  }
125  }
126  if (match.rev0.phase ^ match.rev1.phase) {
127  replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(q0), {0});
128  replacement.add_op<unsigned>(pauli_to_pauli_gate_lut.at(q1), {1});
129  replacement.add_phase(0.5);
130  }
131  return replacement;
132 }

138 static Circuit local_cliffords(OpType op) {
139  Circuit locals(2);
140  switch (op) {
141  case OpType::CX: {
142  locals.add_op<unsigned>(OpType::Sdg, {0});
143  locals.add_op<unsigned>(OpType::Vdg, {1});
144  break;
145  }
146  case OpType::CZ: {
147  locals.add_op<unsigned>(OpType::Sdg, {0});
148  locals.add_op<unsigned>(OpType::Sdg, {1});
149  locals.add_phase(0.25);
150  break;
151  }
152  case OpType::CY: {
153  locals.add_op<unsigned>(OpType::Sdg, {0});
154  locals.add_op<unsigned>(OpType::V, {1});
155  locals.add_op<unsigned>(OpType::Sdg, {1});
156  locals.add_op<unsigned>(OpType::Vdg, {1});
157  locals.add_phase(0.25);
158  break;
159  }
160  case OpType::ZZMax: {
161  break;
162  }
163  default: {
164  throw CircuitInvalidity(
165  "Attempting to replace non-Clifford gate with Clifford "
166  "optimisation");
167  break;
168  }
169  }
170  return locals;
171 }

173 void CliffordReductionPass::insert_interaction_point(InteractionPoint ip) {
174  itable.insert(ip);
175  Vertex next = circ.target(ip.e);
176  port_t next_p = circ.get_target_port(ip.e);
177  bool commute = true;
178  while (commute) {
179  if (v_to_depth.find(next) == v_to_depth.end()) {
180  commute = false;
181  continue;
182  }
183  Op_ptr op = circ.get_Op_ptr_from_Vertex(next);
184  if (!op->get_desc().is_gate()) {
185  commute = false;
186  continue;
187  }
188  OpType type = op->get_type();
189  switch (type) {
190  case OpType::H:
191  case OpType::S:
192  case OpType::Sdg:
193  case OpType::V:
194  case OpType::Vdg:
195  case OpType::X:
196  case OpType::Y:
197  case OpType::Z: {
198  std::pair<Pauli, bool> new_basis = conjugate_Pauli(type, ip.p, true);
199  ip.p = new_basis.first;
200  ip.phase ^= new_basis.second;
201  break;
202  }
203  case OpType::SWAP: {
204  next_p = 1 - next_p;
205  break;
206  }
207  default: {
208  if (!circ.commutes_with_basis(next, ip.p, PortType::Target, next_p)) {
209  commute = false;
210  continue;
211  }
212  break;
213  }
214  }
215  ip.e = circ.get_nth_out_edge(next, next_p);
216  auto inserted = itable.insert(ip);
217  commute = inserted.second;
218  if (!commute) {
219  // Now `inserted.first` points to the element of the table that
220  // blocked insertion, i.e. had the same source/edge combination.
221  // Check that its `p` and `phase` are correct: if not, something has
222  // gone wrong.
223  auto blocker = inserted.first;
224  TKET_ASSERT(blocker->p == ip.p && blocker->phase == ip.phase);
225  }
226  next = circ.target(ip.e);
227  next_p = circ.get_target_port(ip.e);
228  }
229 }

231 std::optional<InteractionMatch> CliffordReductionPass::search_back_for_match(
232  const RevInteractionPoint &rip0, const RevInteractionPoint &rip1) const {
233  RevInteractionPoint point[2];
234  point[0] = rip0;
235  point[1] = rip1;
236  std::map<Edge, RevInteractionPoint> point_lookup;
237  IndexMap im = circ.index_map();
238 
239  // interactions met when commuting back; point lists are in causal order of
240  // circuit:
241  std::map<IVertex, std::list<InteractionPoint>> candidates[2];
242 
243  for (unsigned i = 0; i < 2; ++i) {
244  // Commute edge i back as far as possible
245  bool commute = true;
246  while (commute) {
247  point_lookup.insert({point[i].e, point[i]});
248  auto r = itable.get<TagEdge>().equal_range(point[i].e);
249  for (auto it = r.first; it != r.second; ++it) {
250  Vertex v = it->source;
251  candidates[i][{im.at(v), v}].push_front(*it);
252  }
253  Vertex pred = circ.source(point[i].e);
254  port_t pred_port = circ.get_source_port(point[i].e);
255  Op_ptr pred_op = circ.get_Op_ptr_from_Vertex(pred);
256  if (!pred_op->get_desc().is_gate()) {
257  commute = false;
258  continue;
259  }
260  OpType type = pred_op->get_type();
261  switch (type) {
262  case OpType::H:
263  case OpType::S:
264  case OpType::Sdg:
265  case OpType::V:
266  case OpType::Vdg:
267  case OpType::X:
268  case OpType::Y:
269  case OpType::Z: {
270  std::pair<Pauli, bool> new_basis = conjugate_Pauli(type, point[i].p);
271  point[i].p = new_basis.first;
272  point[i].phase ^= new_basis.second;
273  break;
274  }
275  case OpType::SWAP: {
276  pred_port = 1 - pred_port;
277  break;
278  }
279  default: {
280  commute = circ.commutes_with_basis(
281  pred, point[i].p, PortType::Source, pred_port);
282  break;
283  }
284  }
285  point[i].e = circ.get_nth_in_edge(pred, pred_port);
286  }
287  }
288  // Check for matching interactions
289  for (const std::pair<const IVertex, std::list<InteractionPoint>> &pair :
290  candidates[0]) {
291  auto found = candidates[1].find(pair.first);
292  if (found != candidates[1].end()) {
293  std::optional<std::pair<InteractionPoint, InteractionPoint>>
294  insert_point = valid_insertion_point(pair.second, found->second);
295  if (insert_point) {
296  InteractionMatch match = {
297  insert_point->first, insert_point->second,
298  point_lookup.at(insert_point->first.e),
299  point_lookup.at(insert_point->second.e)};
300  if (!allow_swaps) {
301  if (match.point0.p != match.rev0.p && match.point1.p != match.rev1.p)
302  continue;
303  }
304  return match;
305  }
306  }
307  }
308  return std::nullopt;
309 }

311 void CliffordReductionPass::process_new_interaction(const Vertex &inter) {
312  // Process the vertex as well as any new 2qb Cliffords that get inserted
313  // while doing so (and so on until there are none left to process).
314  std::list<Vertex> to_process = {inter};
315  while (!to_process.empty()) {
316  Vertex v = to_process.front();
317  to_process.pop_front();
318  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
319  Pauli basis0 = *op->commuting_basis(0);
320  Pauli basis1 = *op->commuting_basis(1);
321  EdgeVec ins = circ.get_in_edges(v);
322  RevInteractionPoint rip0 = {ins.at(0), basis0, false};
323  RevInteractionPoint rip1 = {ins.at(1), basis1, false};
324  std::optional<InteractionMatch> match = search_back_for_match(rip0, rip1);
325  if (match) {
326  Circuit replacement = interaction_replacement(*match);
327  Subcircuit site;
328  site.q_in_hole = site.q_out_hole = {match->point0.e, match->point1.e};
329  Subcircuit inserted = substitute(replacement, site);
330  const Vertex &source = match->point0.source;
331  Circuit source_locals =
332  local_cliffords(circ.get_OpType_from_Vertex(source));
333  Subcircuit source_site;
334  source_site.q_in_hole = circ.get_in_edges(source);
335  source_site.q_out_hole = {
336  circ.get_nth_out_edge(source, 0), circ.get_nth_out_edge(source, 1)};
337  source_site.verts.insert(source);
338  substitute(source_locals, source_site);
339  Circuit v_locals = local_cliffords(op->get_type());
340  Subcircuit v_site;
341  v_site.q_in_hole = circ.get_in_edges(v);
342  v_site.q_out_hole = {
343  circ.get_nth_out_edge(v, 0), circ.get_nth_out_edge(v, 1)};
344  v_site.verts.insert(v);
345  substitute(v_locals, v_site);
346  for (const Vertex &new_v : inserted.verts) {
347  if (circ.n_in_edges(new_v) == 2 &&
348  circ.get_OpType_from_Vertex(new_v) != OpType::SWAP) {
349  to_process.push_back(new_v);
350  break;
351  }
352  }
353  success = true;
354  } else {
355  std::vector<std::optional<Edge>> outs = circ.get_linear_out_edges(v);
356  InteractionPoint ip0 = {*outs.at(0), v, basis0, false};
357  insert_interaction_point(ip0);
358  InteractionPoint ip1 = {*outs.at(1), v, basis1, false};
359  insert_interaction_point(ip1);
360  }
361  }
362 }

364 Subcircuit CliffordReductionPass::substitute(
365  const Circuit &to_insert, const Subcircuit &to_replace) {
366  unsigned q_width = to_replace.q_in_hole.size();
367  TKET_ASSERT(q_width == 2);
368 
369  // Construct tables of predecessors, successors, units, in-edges, out-edges.
370  // Only quantum circuit replacments here so don't care about classical stuff
371  std::vector<VertPort> preds(q_width);
372  std::vector<VertPort> succs(q_width);
373  std::vector<UnitID> units(q_width);
374  std::vector<Edge> in_edges(q_width);
375  std::vector<Edge> out_edges(q_width);
376  for (unsigned qi = 0; qi < q_width; ++qi) {
377  const Edge &in = to_replace.q_in_hole.at(qi);
378  const Edge &out = to_replace.q_out_hole.at(qi);
379  preds[qi] = {circ.source(in), circ.get_source_port(in)};
380  succs[qi] = {circ.target(out), circ.get_target_port(out)};
381  units[qi] = e_to_unit.at(in);
382  in_edges[qi] = in;
383  out_edges[qi] = out;
384  TKET_ASSERT(in == out || circ.target(in) == circ.source(out));
385  }
386 
387  // List of points that will be invalidated by the substitution.
388  std::list<InteractionPoint> invalidated_points;
389 
390  // Lists of points having the same "in"/"out" edge as the replacement:
391  // These are all invalidated (though the "in" ones can be replaced later
392  // with the new edge).
393  std::vector<std::list<InteractionPoint>> points_with_in(q_width);
394  std::vector<std::list<InteractionPoint>> points_with_out(q_width);
395  for (unsigned qi = 0; qi < q_width; ++qi) {
396  auto r = itable.get<TagEdge>().equal_range(in_edges[qi]);
397  for (auto it = r.first; it != r.second; it++) {
398  points_with_in[qi].push_back(*it);
399  invalidated_points.push_back(*it);
400  }
401  r = itable.get<TagEdge>().equal_range(out_edges[qi]);
402  for (auto it = r.first; it != r.second; it++) {
403  points_with_out[qi].push_back(*it);
404  invalidated_points.push_back(*it);
405  }
406  }
407 
408  // For any (e0, v0) in points_with_out, any point (e1, v0) where e1 is in
409  // the causal future of e0 is also invalidated. Calculate all the future
410  // edges.
411  EdgeList future_edges;
412  VertexSet v_frontier;
413  for (unsigned qi = 0; qi < q_width; ++qi) {
414  v_frontier.insert(circ.target(out_edges[qi]));
415  }
416  while (!v_frontier.empty()) {
417  EdgeSet out_edges;
418  for (auto v : v_frontier) {
419  if (v_to_depth.find(v) != v_to_depth.end()) {
420  EdgeVec v_out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
421  out_edges.insert(v_out_edges.begin(), v_out_edges.end());
422  }
423  }
424  future_edges.insert(future_edges.end(), out_edges.begin(), out_edges.end());
425  VertexSet new_v_frontier;
426  for (auto e : out_edges) {
427  new_v_frontier.insert(circ.target(e));
428  }
429  v_frontier = std::move(new_v_frontier);
430  }
431 
432  // Invalidate the (e1, v0) as above.
433  VertexSet invalid_sources;
434  for (unsigned qi = 0; qi < q_width; ++qi) {
435  for (auto ip : points_with_out[qi]) {
436  TKET_ASSERT(ip.e == out_edges[qi]);
437  invalid_sources.insert(ip.source);
438  }
439  }
440  for (auto e1 : future_edges) {
441  auto r = itable.get<TagEdge>().equal_range(e1);
442  for (auto it = r.first; it != r.second; ++it) {
443  if (invalid_sources.find(it->source) != invalid_sources.end()) {
444  invalidated_points.push_back(*it);
445  }
446  }
447  }
448 
449  // Erase the invalidated points from the table.
450  for (auto ip : invalidated_points) {
451  itable.erase(ip.key());
452  }
453 
454  // Erase edges from e_to_unit
455  for (unsigned qi = 0; qi < q_width; ++qi) {
456  e_to_unit.erase(in_edges[qi]);
457  e_to_unit.erase(out_edges[qi]);
458  // Depth of to_replace is at most 1, so there are no other edges
459  }
460 
461  // Remove replaced vertices from depth and units maps and erase all points
462  // from the itable that have a replaced vertex as source.
463  for (const Vertex &v : to_replace.verts) {
464  v_to_depth.erase(v);
465  v_to_units.erase(v);
466  auto r = itable.get<TagSource>().equal_range(v);
467  for (auto next = r.first; next != r.second; r.first = next) {
468  ++next;
469  itable.erase(itable.project<TagKey>(r.first));
470  }
471  }
472 
473  circ.substitute(to_insert, to_replace);
474 
475  // Update the tables of in and out edges, and amend the stored points
476  for (unsigned qi = 0; qi < q_width; ++qi) {
477  in_edges[qi] = circ.get_nth_out_edge(preds[qi].first, preds[qi].second);
478  out_edges[qi] = circ.get_nth_in_edge(succs[qi].first, succs[qi].second);
479  for (InteractionPoint &ip : points_with_in[qi]) {
480  ip.e = in_edges[qi];
481  }
482  }
483 
484  // Construct inserted Subcircuit, update e_to_unit and v_to_units, and
485  // add new vertices to v_to_depth, with (temporary) value 0.
486  Subcircuit inserted;
487  for (unsigned qi = 0; qi < q_width; ++qi) {
488  Edge in = in_edges[qi];
489  inserted.q_in_hole.push_back(in);
490  Edge out = out_edges[qi];
491  inserted.q_out_hole.push_back(out);
492  while (in != out) {
493  Vertex next = circ.target(in);
494  inserted.verts.insert(next);
495  e_to_unit.insert({in, units[qi]});
496  v_to_depth.insert({next, 0});
497  v_to_units[next].insert(units[qi]);
498  in = circ.get_nth_out_edge(next, circ.get_target_port(in));
499  }
500  e_to_unit.insert({in, units[qi]});
501  }
502 
503  // Now `v_to_depth` is 0 at all `inserted.verts`. Fix this and propagate
504  // updates to the depth map into the future cone, ensuring that the depths
505  // are strictly increasing along wires. Stop when we reach a vertex that
506  // isn't already in v_to_depth (because we haven't processed it yet).
507  for (unsigned qi = 0; qi < q_width; ++qi) {
508  Edge in = in_edges[qi];
509  Vertex next = circ.target(in);
510  if (next != succs[qi].first) {
511  if (v_to_depth.at(preds[qi].first) >= v_to_depth[next]) {
512  // We may have already set v_to_depth[next] to a higher value
513  // when tracing another qubit, so the check above is necessary.
514  v_to_depth[next] = v_to_depth.at(preds[qi].first) + 1;
515  }
516  std::function<bool(const Vertex &, const Vertex &)> c =
517  [&](const Vertex &a, const Vertex &b) {
518  unsigned deptha = v_to_depth.at(a);
519  unsigned depthb = v_to_depth.at(b);
520  if (deptha == depthb) {
521  unit_set_t unitsa = v_to_units.at(a);
522  unit_set_t unitsb = v_to_units.at(b);
523  return unitsa < unitsb;
524  }
525  return deptha < depthb;
526  };
527  std::set<Vertex> to_search;
528  to_search.insert(next);
529  while (!to_search.empty()) {
530  Vertex v = *std::min_element(to_search.begin(), to_search.end(), c);
531  to_search.erase(v);
532  unsigned v_depth = v_to_depth.at(v);
533  EdgeVec outs = circ.get_all_out_edges(v);
534  for (const Edge &e : outs) {
535  Vertex succ = circ.target(e);
536  std::map<Vertex, unsigned>::iterator succ_it = v_to_depth.find(succ);
537  if (succ_it != v_to_depth.end()) {
538  if (succ_it->second <= v_depth) {
539  succ_it->second = v_depth + 1;
540  if (v_depth >= current_depth) {
541  current_depth = v_depth + 1;
542  }
543  to_search.insert(succ);
544  }
545  }
546  }
547  }
548  }
549  }
550 
551  // Re-insert all the interaction points we erased above.
552  for (unsigned qi = 0; qi < q_width; ++qi) {
553  for (const InteractionPoint &ip : points_with_in[qi]) {
554  insert_interaction_point(ip);
555  }
556  }
557 
558  return inserted;
559 }

561 std::optional<Edge> CliffordReductionPass::find_earliest_successor(
562  const Edge &source, const EdgeSet &candidates) const {
563  typedef std::function<bool(Vertex, Vertex)> Comp;
564  Comp c = [&](Vertex a, Vertex b) {
565  unsigned deptha = v_to_depth.at(a);
566  unsigned depthb = v_to_depth.at(b);
567  if (deptha == depthb) {
568  unit_set_t unitsa = v_to_units.at(a);
569  unit_set_t unitsb = v_to_units.at(b);
570  return unitsa < unitsb;
571  }
572  return deptha < depthb;
573  };
574  std::set<Vertex, Comp> to_search(c);
575  to_search.insert(circ.target(source));
576  while (!to_search.empty()) {
577  Vertex v = *to_search.begin();
578  to_search.erase(to_search.begin());
579  EdgeVec outs = circ.get_all_out_edges(v);
580  for (const Edge &e : outs) {
581  if (candidates.find(e) != candidates.end()) return e;
582  Vertex succ = circ.target(e);
583  if (v_to_depth.find(succ) != v_to_depth.end()) to_search.insert(succ);
584  }
585  }
586  return std::nullopt;
587 }

589 std::optional<std::pair<InteractionPoint, InteractionPoint>>
590 CliffordReductionPass::valid_insertion_point(
591  const std::list<InteractionPoint> &seq0,
592  const std::list<InteractionPoint> &seq1) const {
593  // seq0 is chain of edges (in temporal order) from the first qubit
594  // likewise seq1 for the other qubit
595  InteractionPoint seq0max = seq0.back();
596  InteractionPoint seq1max = seq1.back();
597  if (circ.in_causal_order(
598  circ.source(seq1max.e), circ.target(seq0max.e), true, v_to_depth,
599  v_to_units, false)) {
600  // Search for any points in seq1 from future of seq0max
601  EdgeSet candidates;
602  std::map<Edge, InteractionPoint> lookup;
603  for (const InteractionPoint &ip : seq1) {
604  candidates.insert(ip.e);
605  lookup.insert({ip.e, ip});
606  }
607  std::optional<Edge> successor =
608  find_earliest_successor(seq0max.e, candidates);
609  if (!successor || successor == seq1.front().e) return std::nullopt;
610  Vertex v = circ.source(*successor);
611  port_t p = circ.get_source_port(*successor);
612  if (circ.get_OpType_from_Vertex(v) == OpType::SWAP) p = 1 - p;
613  return {{seq0max, lookup.at(circ.get_nth_in_edge(v, p))}};
614  } else if (circ.in_causal_order(
615  circ.source(seq0max.e), circ.target(seq1max.e), true,
616  v_to_depth, v_to_units, false)) {
617  // Search for any points in seq0 from future of seq1max
618  EdgeSet candidates;
619  std::map<Edge, InteractionPoint> lookup;
620  for (const InteractionPoint &ip : seq0) {
621  candidates.insert(ip.e);
622  lookup.insert({ip.e, ip});
623  }
624  std::optional<Edge> successor =
625  find_earliest_successor(seq1max.e, candidates);
626  if (!successor || successor == seq0.front().e) return std::nullopt;
627  Vertex v = circ.source(*successor);
628  port_t p = circ.get_source_port(*successor);
629  if (circ.get_OpType_from_Vertex(v) == OpType::SWAP) p = 1 - p;
630  return {{lookup.at(circ.get_nth_in_edge(v, p)), seq1max}};
631  } else {
632  // seq0max and seq1max are space-like separated
633  return {{seq0max, seq1max}};
634  }
635 }

648 bool CliffordReductionPass::reduce_circuit(Circuit &circ, bool allow_swaps) {
649  CliffordReductionPass context(circ, allow_swaps);
650 
651  SliceVec slices = circ.get_slices();
652 
653  for (const Vertex &in : circ.all_inputs()) {
654  context.v_to_depth.insert({in, 0});
655  }
656 
657  // Process all 2qb Clifford vertices.
658  for (const Slice &sl : slices) {
659  for (const Vertex &v : sl) {
660  context.v_to_depth.insert({v, context.current_depth});
661  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
662  if (!op->get_desc().is_gate()) continue;
663  EdgeVec ins = circ.get_in_edges(v);
664  std::vector<std::optional<Edge>> outs = circ.get_linear_out_edges(v);
665  OpType type = op->get_type();
666  std::list<InteractionPoint> new_points;
667  switch (type) {
668  case OpType::H:
669  case OpType::S:
670  case OpType::Sdg:
671  case OpType::V:
672  case OpType::Vdg:
673  case OpType::X:
674  case OpType::Y:
675  case OpType::Z: {
676  auto r = context.itable.get<TagEdge>().equal_range(ins[0]);
677  for (auto it = r.first; it != r.second; ++it) {
678  InteractionPoint ip = *it;
679  std::pair<Pauli, bool> new_basis =
680  conjugate_Pauli(type, ip.p, true);
681  ip.p = new_basis.first;
682  ip.phase ^= new_basis.second;
683  ip.e = *outs[0];
684  new_points.push_back(ip);
685  }
686  break;
687  }
688  case OpType::SWAP: {
689  auto r0 = context.itable.get<TagEdge>().equal_range(ins[0]);
690  for (auto it = r0.first; it != r0.second; ++it) {
691  InteractionPoint ip = *it;
692  ip.e = *outs[1];
693  new_points.push_back(ip);
694  }
695  auto r1 = context.itable.get<TagEdge>().equal_range(ins[1]);
696  for (auto it = r1.first; it != r1.second; ++it) {
697  InteractionPoint ip = *it;
698  ip.e = *outs[0];
699  new_points.push_back(ip);
700  }
701  break;
702  }
703  default: {
704  for (unsigned i = 0; i < ins.size(); ++i) {
705  auto r = context.itable.get<TagEdge>().equal_range(ins[i]);
706  for (auto it = r.first; it != r.second; ++it) {
707  InteractionPoint ip = *it;
708  if (circ.commutes_with_basis(v, ip.p, PortType::Target, i)) {
709  ip.e = *outs[i];
710  new_points.push_back(ip);
711  }
712  }
713  }
714  break;
715  }
716  }
717  for (const InteractionPoint &ip : new_points) {
718  context.itable.insert(ip);
719  }
720  switch (type) {
721  case OpType::CX:
722  case OpType::CY:
723  case OpType::CZ:
724  case OpType::ZZMax: {
725  context.process_new_interaction(v);
726  break;
727  }
728  default: {
729  break;
730  }
731  }
732  }
733  ++context.current_depth;
734  }
735 
736  if (allow_swaps) {
737  circ.replace_SWAPs();
738  }
739 
740  return context.success;
741 }
