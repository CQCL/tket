33static Circuit resynthesised_circuit(
34 const Circuit &circ,
35 std::optional<std::function<Circuit(const Circuit &)>> transform = nullptr,
36 bool allow_swaps = true) {
37 if (transform.has_value()) {
38 return (*transform)(circ);
39 } else {
40 Circuit circ1(circ);
41 // Ensure all recognized Clifford gates are Clifford types:
42 decompose_multi_qubits_CX().apply(circ1);
43 decompose_cliffords_std().apply(circ1);
44 remove_redundancies().apply(circ1);
45
46 // Convert to tableau and back:
47 const UnitaryTableau tab = circuit_to_unitary_tableau(circ1);
48 Circuit newcirc = unitary_tableau_to_circuit(tab);
49
50 // Simplify:
51 clifford_simp(allow_swaps).apply(newcirc);
52
53 return newcirc;
54 }
55}

57// Returns the two-qubit-gate reduction resulting from resynthesising the given
58// (convex, connected) vertex set using the given transform.
59static int n_2q_reduction(
60 const Circuit &circ, const VertexSet &verts,
61 std::optional<std::function<Circuit(const Circuit &)>> transform,
62 bool allow_swaps) {
63 Subcircuit subcircuit = circ.make_subcircuit(verts);
64 const Circuit subc = circ.subcircuit(subcircuit);
65 Circuit newsubc = resynthesised_circuit(subc, transform, allow_swaps);
66 return int(subc.count_n_qubit_gates(2)) - int(newsubc.count_n_qubit_gates(2));
67}

69static bool resynthesise_cliffords(
70 Circuit &circ,
71 std::optional<std::function<Circuit(const Circuit &)>> transform = nullptr,
72 bool allow_swaps = true) {
73 bool changed = false;
74 while (true) {
75 std::vector<VertexSet> subcircuits =
76 circ.get_subcircuits([](Op_ptr op) { return op->is_clifford(); });
77 if (subcircuits.empty()) {
78 break;
79 }
80 // Pick the one that reduces the 2q gate count the most.
81 VertexSet verts = *std::max_element(
82 subcircuits.begin(), subcircuits.end(),
83 [&circ, &transform, &allow_swaps](
84 const VertexSet &A, const VertexSet &B) {
85 return n_2q_reduction(circ, A, transform, allow_swaps) <
86 n_2q_reduction(circ, B, transform, allow_swaps);
87 });
88 Subcircuit subcircuit = circ.make_subcircuit(verts);
89 const Circuit subc = circ.subcircuit(subcircuit);
90 Circuit newsubc = resynthesised_circuit(subc, transform, allow_swaps);
91 if (newsubc.count_n_qubit_gates(2) < subc.count_n_qubit_gates(2)) {
92 circ.substitute(newsubc, subcircuit);
93 changed = true;
94 } else {
95 break;
96 }
97 }
98 return changed;
99}

101Transform clifford_resynthesis(
102 std::optional<std::function<Circuit(const Circuit &)>> transform,
103 bool allow_swaps) {
104 return Transform([transform, allow_swaps](Circuit &circ) {
105 return resynthesise_cliffords(circ, transform, allow_swaps);
106 });
