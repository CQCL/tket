30Transform sequence(std::vector<Transform> &tvec) {
31 return Transform([=](Circuit &circ, std::shared_ptr<unit_bimaps_t> maps) {
32 bool success = false;
33 for (std::vector<Transform>::const_iterator it = tvec.begin();
34 it != tvec.end(); ++it) {
35 success = it->apply_fn(circ, maps) || success;
36 }
37 return success;
38 });

41Transform repeat(const Transform &trans) {
42 return Transform([=](Circuit &circ, std::shared_ptr<unit_bimaps_t> maps) {
43 bool success = false;

49Transform repeat_with_metric(
50 const Transform &trans, const Transform::Metric &eval) {
51 return Transform([=](Circuit &circ, std::shared_ptr<unit_bimaps_t> maps) {
52 bool success = false;
53 int currentVal = eval(circ);
54 Circuit *currentCircuit = &circ;
55 Circuit newCircuit = circ;
56 trans.apply_fn(newCircuit, maps);
57 int newVal = eval(newCircuit);
58 while (newVal < currentVal) {
59 currentCircuit = &newCircuit;
60 currentVal = newVal;
61 success = true;
62 trans.apply_fn(newCircuit, maps);
63 newVal = eval(newCircuit);
64 }
65 if (&circ != currentCircuit) circ = *currentCircuit;
66 return success;
67 });
68}

70Transform repeat_while(const Transform &cond, const Transform &body) {

74 success = true;
