22 void to_json(nlohmann::json& j, const Command& com) {
23  const Op_ptr op = com.get_op_ptr();
24  const std::optional<std::string> opgroup = com.get_opgroup();
25  j["op"] = op;
26  if (opgroup) {
27  j["opgroup"] = opgroup.value();
28  }
29 
30  const op_signature_t& sig = op->get_signature();
31  const unit_vector_t& args = com.get_args();
32 
33  if (sig.size() != args.size()) {
34  JsonError("Number of args does not match signature of op.");
35  }
36 
37  nlohmann::json j_args;
38 
39  for (std::size_t i = 0; i < sig.size(); i++) {
40  switch (sig[i]) {
41  case EdgeType::WASM: {
42  const WasmState& wb = static_cast<const WasmState&>(args[i]);
43  j_args.push_back(wb);
44  break;
45  }
46  case EdgeType::Quantum: {
47  const Qubit& qb = static_cast<const Qubit&>(args[i]);
48  j_args.push_back(qb);
49  break;
50  }
51  case EdgeType::Classical:
52  case EdgeType::Boolean: {
53  const Bit& cb = static_cast<const Bit&>(args[i]);
54  j_args.push_back(cb);
55  break;
56  }
57  default: {
58  TKET_ASSERT(!"command to json found invalid edge type in signature");
59  }
60  }
61  }
62 
63  j["args"] = j_args;
64 }
65 void from_json(const nlohmann::json& j, Command& com) {
66  const auto op = j.at("op").get<Op_ptr>();
67  std::optional<std::string> opgroup;
68  if (j.contains("opgroup")) {
69  opgroup = j.at("opgroup").get<std::string>();
70  }
71  const op_signature_t& sig = op->get_signature();
72 
73  const nlohmann::json& j_args = j.at("args");
74  if (sig.size() != j_args.size()) {
75  JsonError("Number of args does not match signature of op.");
76  }
77  unit_vector_t args;
78  for (std::size_t i = 0; i < sig.size(); i++) {
79  switch (sig[i]) {
80  case EdgeType::WASM: {
81  args.push_back(j_args[i].get<WasmState>());
82  break;
83  }
84  case EdgeType::Quantum: {
85  args.push_back(j_args[i].get<Qubit>());
86  break;
87  }
88  case EdgeType::Classical:
89  case EdgeType::Boolean: {
90  args.push_back(j_args[i].get<Bit>());
91  break;
92  }
93  default: {
94  TKET_ASSERT(!"command from json found invalid edge type in signature");
95  }
96  }
97  }
98  com = Command(op, args, opgroup);
