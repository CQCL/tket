44 void from_json(const nlohmann::json& j, Command& com) {
45  const auto op = j.at("op").get<Op_ptr>();
46  std::optional<std::string> opgroup;
47  if (j.contains("opgroup")) {
48  opgroup = j.at("opgroup").get<std::string>();
49  }
50  const op_signature_t& sig = op->get_signature();
51 
52  const nlohmann::json& j_args = j.at("args");
53  if (sig.size() != j_args.size()) {
54  JsonError("Number of args does not match signature of op.");
55  }
56  unit_vector_t args;
57  for (size_t i = 0; i < sig.size(); i++) {
58  if (sig[i] == EdgeType::Quantum) {
59  args.push_back(j_args[i].get<Qubit>());
60  } else {
61  args.push_back(j_args[i].get<Bit>());
62  }
63  }
64 
65  com = Command(op, args, opgroup);
