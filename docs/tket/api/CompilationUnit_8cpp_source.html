83 void CompilationUnit::empty_cache() const { cache_ = {}; }

85 void CompilationUnit::initialize_cache() const {
86  if (!cache_.empty())
87  throw std::logic_error("PredicateCache must be empty to be initialized");
88  for (const TypePredicatePair& pr : target_preds) {
89  const Predicate& p = *(pr.second);
90  std::type_index ti = typeid(p);
91  if (cache_.find(ti) != cache_.end())
92  throw std::logic_error("Duplicate verify type in Predicate list");
93  bool to_cache = calc_predicate(p);
94  cache_.insert({ti, {pr.second, to_cache}});
95  }
96 }
