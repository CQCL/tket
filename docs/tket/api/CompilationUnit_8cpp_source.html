84 void CompilationUnit::initialize_cache() const {
85  if (!cache_.empty())
86  throw std::logic_error("PredicateCache must be empty to be initialized");
87  for (const TypePredicatePair& pr : target_preds) {
88  const Predicate& p = *(pr.second);
89  std::type_index ti = typeid(p);
90  if (cache_.find(ti) != cache_.end())
91  throw std::logic_error("Duplicate verify type in Predicate list");
92  bool to_cache = calc_predicate(p);
93  cache_.insert({ti, {pr.second, to_cache}});
94  }
95 }
