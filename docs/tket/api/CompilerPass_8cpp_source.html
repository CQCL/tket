36 std::string BasePass::to_string() const {
37  std::string str = "Preconditions:\n";
38  for (const TypePredicatePair& pp : precons_) {
39  str += (" " + pp.second->to_string() + "\n");
40  }
41  str += "Specific Postconditions:\n";
42  for (const TypePredicatePair& pp : postcons_.specific_postcons_) {
43  str += (" " + pp.second->to_string() + "\n");
44  }
45  str += "Generic Postconditions:\n";
46  for (const std::pair<const std::type_index, Guarantee>& pg :
47  postcons_.generic_postcons_) {
48  str += (" " + predicate_name(pg.first) + " ");
49  str += (pg.second == Guarantee::Clear) ? "Clear\n" : "Preserve\n";
50  }
51  str += "Default Postcondition: ";
52  str += (postcons_.default_postcon_ == Guarantee::Clear) ? "Clear\n"
53  : "Preserve\n";
54  return str;
55 };

57 std::optional<PredicatePtr> BasePass::unsatisfied_precondition(
58  const CompilationUnit& c_unit, SafetyMode safe_mode) const {
59  for (const TypePredicatePair& pp : precons_) {
60  PredicateCache::const_iterator cache_iter = c_unit.cache_.find(pp.first);
61  if (cache_iter == c_unit.cache_.end()) { // cache does not contain
62  // predicate
63  if (!c_unit.calc_predicate(*pp.second)) return pp.second;
64  c_unit.cache_.insert({pp.first, {pp.second, true}});
65  } else {
66  /* if a Predicate is not `true` in the cache or implied by a set Predicate
67  in the cache then it is assumed to be `false` */
68  if (cache_iter->second.second) {
69  if (!cache_iter->second.first->implies(*pp.second)) {
70  if (!c_unit.calc_predicate(*pp.second)) return pp.second;
71  }
72  } else {
73  if (!c_unit.calc_predicate(*pp.second)) return pp.second;
74  }
75  }
76  }
77  if (safe_mode == SafetyMode::Audit) {
78  for (const TypePredicatePair& pp : precons_) {
79  if (!c_unit.calc_predicate(*pp.second)) return pp.second;
80  }
81  }
82  return {};
83 }

85 void BasePass::update_cache(
86  const CompilationUnit& c_unit, SafetyMode safe_mode) const {
87  if (postcons_.default_postcon_ == Guarantee::Clear) {
88  for (PredicateCache::iterator it = c_unit.cache_.begin();
89  it != c_unit.cache_.end(); ++it) {
90  it->second.second = false;
91  }
92  }
93  for (const std::pair<const std::type_index, Guarantee>& pg :
94  postcons_.generic_postcons_) {
95  if (pg.second == Guarantee::Clear) {
96  PredicateCache::iterator cache_iter = c_unit.cache_.find(pg.first);
97  if (cache_iter != c_unit.cache_.end()) cache_iter->second.second = false;
98  }
99  }
100  for (const TypePredicatePair& pp : postcons_.specific_postcons_) {
101  if (safe_mode == SafetyMode::Audit && !pp.second->verify(c_unit.circ_))
102  throw UnsatisfiedPredicate(pp.second->to_string());
103  std::pair<PredicatePtr, bool> cache_pair{pp.second, true};
104  c_unit.cache_[pp.first] = cache_pair;
105  }
106 }

112 Guarantee BasePass::get_guarantee(
113  const std::type_index& ti, const PassConditions& conditions) {
114  PredicateClassGuarantees::const_iterator class_guar_iter =
115  conditions.second.generic_postcons_.find(ti);
116  if (class_guar_iter == conditions.second.generic_postcons_.end()) {
117  return conditions.second.default_postcon_;

119  return class_guar_iter->second;

123 static PredicateClassGuarantees match_class_guarantees(
124  const PassConditions& lhs, const PassConditions& rhs) {
125  PredicateClassGuarantees cgs;
126  for (const std::pair<const std::type_index, Guarantee>& postcon :
127  rhs.second.generic_postcons_) {
128  if (postcon.second == Guarantee::Preserve) {
129  if (BasePass::get_guarantee(postcon.first, lhs) == Guarantee::Preserve)
130  cgs.insert(postcon);
131  else
132  cgs.insert({postcon.first, Guarantee::Clear});
133  } else
134  cgs.insert(postcon);
135  }
136  return cgs;
137 }

139 PassConditions BasePass::match_passes(
140  const PassConditions& lhs, const PassConditions& rhs) {
141  PredicatePtrMap new_precons = lhs.first;
142  for (const TypePredicatePair& precon : rhs.first) {
143  PredicatePtrMap::const_iterator data_guar_iter =
144  lhs.second.specific_postcons_.find(precon.first);
145  if (data_guar_iter == lhs.second.specific_postcons_.end()) {
146  if (get_guarantee(precon.first, lhs) == Guarantee::Clear) {
147  throw IncompatibleCompilerPasses(precon.first);
148  } else {
149  PredicatePtrMap::iterator new_pre_it = new_precons.find(precon.first);
150  if (new_pre_it == new_precons.end())
151  new_precons.insert(precon);
152  else {
153  PredicatePtr to_put_in = new_pre_it->second->meet(*precon.second);
154  TypePredicatePair tpp = CompilationUnit::make_type_pair(to_put_in);
155  new_precons[tpp.first] = tpp.second;
156  }
157  }
158  } else {
159  if (!data_guar_iter->second->implies(*precon.second)) {
160  throw IncompatibleCompilerPasses(precon.first);
161  }
162  }
163  }
164  PostConditions new_postcons;
165  new_postcons.specific_postcons_ = rhs.second.specific_postcons_;
166  for (const TypePredicatePair& postcon : lhs.second.specific_postcons_) {
167  PredicatePtrMap::iterator specific_post_it =
168  new_postcons.specific_postcons_.find(postcon.first);
169  if (specific_post_it == new_postcons.specific_postcons_.end()) {
170  if (get_guarantee(postcon.first, rhs) == Guarantee::Preserve)
171  new_postcons.specific_postcons_.insert(postcon);
172  }
173  }
174  new_postcons.generic_postcons_ = match_class_guarantees(lhs, rhs);
175  PredicateClassGuarantees others = match_class_guarantees(rhs, lhs);
176  new_postcons.generic_postcons_.insert(others.begin(), others.end());
177 
178  if (rhs.second.default_postcon_ == Guarantee::Clear)
179  new_postcons.default_postcon_ = Guarantee::Clear;
180  else
181  new_postcons.default_postcon_ = lhs.second.default_postcon_;
182 
183  return {new_precons, new_postcons};
184 }

190 bool StandardPass::apply(
191  CompilationUnit& c_unit, SafetyMode safe_mode,
192  const PassCallback& before_apply, const PassCallback& after_apply) const {
193  before_apply(c_unit, this->get_config());
194  std::optional<PredicatePtr> unsatisfied_precon =
195  unsatisfied_precondition(c_unit, safe_mode);
196  if (unsatisfied_precon)
197  throw UnsatisfiedPredicate(
198  unsatisfied_precon.value()
199  ->to_string()); // just raise warning in super-unsafe mode
200  // Allow trans_ to update the initial and final map
201  bool changed = trans_.apply_fn(c_unit.circ_, c_unit.maps);
202  update_cache(c_unit, safe_mode);
203  after_apply(c_unit, this->get_config());
204  return changed;
205 }

220 PassPtr operator>>(const PassPtr& lhs, const PassPtr& rhs) {
221  PassConditions pre_post_cons = BasePass::match_passes(lhs, rhs);
222  SequencePass new_pass;
223  new_pass.precons_ = pre_post_cons.first;
224  new_pass.postcons_ = pre_post_cons.second;
225  new_pass.seq_ = {lhs, rhs};
226  PassPtr sequence = std::make_shared<SequencePass>(new_pass);
227  return sequence;
228 }

230 SequencePass::SequencePass(const std::vector<PassPtr>& ptvec) {
231  if (ptvec.size() == 0)
232  throw std::logic_error("Cannot generate CompilerPass from empty list");
233  std::vector<PassPtr>::const_iterator iter = ptvec.begin();
234  PassConditions conditions = (*iter)->get_conditions();
235  for (++iter; iter != ptvec.end(); ++iter) {
236  const PassConditions next_cons = (*iter)->get_conditions();
237  conditions = match_passes(conditions, next_cons);
238  }
239  this->precons_ = conditions.first;
240  this->postcons_ = conditions.second;
241  this->seq_ = ptvec;
242 }

281 bool RepeatWithMetricPass::apply(
282  CompilationUnit& c_unit, SafetyMode safe_mode,
283  const PassCallback& before_apply, const PassCallback& after_apply) const {
284  before_apply(c_unit, this->get_config());
285  bool success = false;
286  unsigned currentVal = metric_(c_unit.get_circ_ref());
287  CompilationUnit* c_unit_current = &c_unit;
288  CompilationUnit c_unit_new = c_unit;
289  pass_->apply(
290  c_unit_new, safe_mode); // I can't make it apply the pass to a copy
291  // without copying the whole CompilationUnit
292  unsigned newVal = metric_(c_unit_new.get_circ_ref());
293  while (newVal < currentVal) {
294  c_unit_current = &c_unit_new;
295  currentVal = newVal;
296  success = true;
297  pass_->apply(c_unit_new, safe_mode, before_apply, after_apply);
298  newVal = metric_(c_unit_new.get_circ_ref());
299  }
300  if (&c_unit != c_unit_current) c_unit = *c_unit_current;
301  after_apply(c_unit, this->get_config());
302  return success;
303 }
