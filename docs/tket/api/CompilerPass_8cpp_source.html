37std::string BasePass::to_string() const {
38 std::string str = "Preconditions:\n";
39 for (const TypePredicatePair& pp : precons_) {
40 str += (" " + pp.second->to_string() + "\n");
41 }
42 str += "Specific Postconditions:\n";
43 for (const TypePredicatePair& pp : postcons_.specific_postcons_) {
44 str += (" " + pp.second->to_string() + "\n");
45 }
46 str += "Generic Postconditions:\n";
47 for (const std::pair<const std::type_index, Guarantee>& pg :
48 postcons_.generic_postcons_) {
49 str += (" " + predicate_name(pg.first) + " ");
50 str += (pg.second == Guarantee::Clear) ? "Clear\n" : "Preserve\n";
51 }
52 str += "Default Postcondition: ";
53 str += (postcons_.default_postcon_ == Guarantee::Clear) ? "Clear\n"
54 : "Preserve\n";
55 return str;
56};

58std::optional<PredicatePtr> BasePass::unsatisfied_precondition(
59 const CompilationUnit& c_unit, SafetyMode safe_mode) const {
60 for (const TypePredicatePair& pp : precons_) {
61 PredicateCache::const_iterator cache_iter = c_unit.cache_.find(pp.first);
62 if (cache_iter == c_unit.cache_.end()) { // cache does not contain
63 // predicate
64 if (!c_unit.calc_predicate(*pp.second)) return pp.second;
65 c_unit.cache_.insert({pp.first, {pp.second, true}});
66 } else {
67 /* if a Predicate is not `true` in the cache or implied by a set Predicate
68 in the cache then it is assumed to be `false` */
69 if (cache_iter->second.second) {
70 if (!cache_iter->second.first->implies(*pp.second)) {
71 if (!c_unit.calc_predicate(*pp.second)) return pp.second;
72 }
73 } else {
74 if (!c_unit.calc_predicate(*pp.second)) return pp.second;
75 }
76 }
77 }
78 if (safe_mode == SafetyMode::Audit) {
79 for (const TypePredicatePair& pp : precons_) {
80 if (!c_unit.calc_predicate(*pp.second)) return pp.second;
81 }
82 }
83 return {};
84}

86void BasePass::update_cache(
87 const CompilationUnit& c_unit, SafetyMode safe_mode) const {
88 if (postcons_.default_postcon_ == Guarantee::Clear) {
89 for (PredicateCache::iterator it = c_unit.cache_.begin();
90 it != c_unit.cache_.end(); ++it) {
91 it->second.second = false;
92 }
93 }
94 for (const std::pair<const std::type_index, Guarantee>& pg :
95 postcons_.generic_postcons_) {
96 if (pg.second == Guarantee::Clear) {
97 PredicateCache::iterator cache_iter = c_unit.cache_.find(pg.first);
98 if (cache_iter != c_unit.cache_.end()) cache_iter->second.second = false;
99 }
100 }
101 for (const TypePredicatePair& pp : postcons_.specific_postcons_) {
102 if (safe_mode == SafetyMode::Audit && !pp.second->verify(c_unit.circ_))
103 throw UnsatisfiedPredicate(pp.second->to_string());
104 std::pair<PredicatePtr, bool> cache_pair{pp.second, true};
105 c_unit.cache_[pp.first] = cache_pair;
106 }
107}

113Guarantee BasePass::get_guarantee(
114 const std::type_index& ti, const PassConditions& conditions) {
115 PredicateClassGuarantees::const_iterator class_guar_iter =
116 conditions.second.generic_postcons_.find(ti);
117 if (class_guar_iter == conditions.second.generic_postcons_.end()) {
118 return conditions.second.default_postcon_;
119 } else {
120 return class_guar_iter->second;
121 }
122}
123
124static PredicateClassGuarantees match_class_guarantees(
125 const PassConditions& lhs, const PassConditions& rhs) {
126 PredicateClassGuarantees cgs;
127 for (const std::pair<const std::type_index, Guarantee>& postcon :
128 rhs.second.generic_postcons_) {
129 if (postcon.second == Guarantee::Preserve) {
130 if (BasePass::get_guarantee(postcon.first, lhs) == Guarantee::Preserve)
131 cgs.insert(postcon);
132 else
133 cgs.insert({postcon.first, Guarantee::Clear});
134 } else
135 cgs.insert(postcon);
136 }
137 return cgs;
138}
139
140PassConditions BasePass::match_passes(
141 const PassConditions& lhs, const PassConditions& rhs, bool strict) {
142 PredicatePtrMap new_precons = lhs.first;
143 for (const TypePredicatePair& precon : rhs.first) {
144 PredicatePtrMap::const_iterator data_guar_iter =
145 lhs.second.specific_postcons_.find(precon.first);
146 if (data_guar_iter == lhs.second.specific_postcons_.end()) {
147 if (strict && get_guarantee(precon.first, lhs) == Guarantee::Clear) {
148 throw IncompatibleCompilerPasses(precon.first);
149 } else {
150 PredicatePtrMap::iterator new_pre_it = new_precons.find(precon.first);
151 if (new_pre_it == new_precons.end())
152 new_precons.insert(precon);
153 else {
154 PredicatePtr to_put_in = new_pre_it->second->meet(*precon.second);
155 TypePredicatePair tpp = CompilationUnit::make_type_pair(to_put_in);
156 new_precons[tpp.first] = tpp.second;
157 }
158 }
159 } else {
160 if (strict && !data_guar_iter->second->implies(*precon.second)) {
161 throw IncompatibleCompilerPasses(precon.first);
162 }
163 }
164 }
165 PostConditions new_postcons;
166 new_postcons.specific_postcons_ = rhs.second.specific_postcons_;
167 for (const TypePredicatePair& postcon : lhs.second.specific_postcons_) {
168 PredicatePtrMap::iterator specific_post_it =
169 new_postcons.specific_postcons_.find(postcon.first);
170 if (specific_post_it == new_postcons.specific_postcons_.end()) {
171 if (get_guarantee(postcon.first, rhs) == Guarantee::Preserve)
172 new_postcons.specific_postcons_.insert(postcon);
173 }
174 }
175 new_postcons.generic_postcons_ = match_class_guarantees(lhs, rhs);
176 PredicateClassGuarantees others = match_class_guarantees(rhs, lhs);
177 new_postcons.generic_postcons_.insert(others.begin(), others.end());
178
179 if (rhs.second.default_postcon_ == Guarantee::Clear)
180 new_postcons.default_postcon_ = Guarantee::Clear;
181 else
182 new_postcons.default_postcon_ = lhs.second.default_postcon_;
183
184 return {new_precons, new_postcons};
185}

191bool StandardPass::apply(
192 CompilationUnit& c_unit, SafetyMode safe_mode,
193 const PassCallback& before_apply, const PassCallback& after_apply) const {
194 before_apply(c_unit, this->get_config());
195 std::optional<PredicatePtr> unsatisfied_precon =
196 unsatisfied_precondition(c_unit, safe_mode);
197 if (unsatisfied_precon)
198 throw UnsatisfiedPredicate(
199 unsatisfied_precon.value()
200 ->to_string()); // just raise warning in super-unsafe mode
201 // Allow trans_ to update the initial and final map
202 bool changed = trans_.apply_fn(c_unit.circ_, c_unit.maps);
203 update_cache(c_unit, safe_mode);
204 after_apply(c_unit, this->get_config());
205 return changed;
206}

221PassPtr operator>>(const PassPtr& lhs, const PassPtr& rhs) {
222 PassConditions pre_post_cons = BasePass::match_passes(lhs, rhs);
223 SequencePass new_pass;
224 new_pass.precons_ = pre_post_cons.first;
225 new_pass.postcons_ = pre_post_cons.second;
226 new_pass.seq_ = {lhs, rhs};
227 PassPtr sequence = std::make_shared<SequencePass>(new_pass);
228 return sequence;
229}

231SequencePass::SequencePass(const std::vector<PassPtr>& ptvec, bool strict) {
232 if (ptvec.size() == 0)
233 throw std::logic_error("Cannot generate CompilerPass from empty list");
234 std::vector<PassPtr>::const_iterator iter = ptvec.begin();
235 PassConditions conditions = (*iter)->get_conditions();
236 for (++iter; iter != ptvec.end(); ++iter) {
237 const PassConditions next_cons = (*iter)->get_conditions();
238 conditions = match_passes(conditions, next_cons, strict);
239 }
240 this->precons_ = conditions.first;
241 this->postcons_ = conditions.second;
242 this->seq_ = ptvec;
243}

283bool RepeatWithMetricPass::apply(
284 CompilationUnit& c_unit, SafetyMode safe_mode,
285 const PassCallback& before_apply, const PassCallback& after_apply) const {
286 before_apply(c_unit, this->get_config());
287 bool success = false;
288 unsigned currentVal = metric_(c_unit.get_circ_ref());
289 CompilationUnit* c_unit_current = &c_unit;
290 CompilationUnit c_unit_new = c_unit;
291 pass_->apply(
292 c_unit_new, safe_mode); // I can't make it apply the pass to a copy
293 // without copying the whole CompilationUnit
294 unsigned newVal = metric_(c_unit_new.get_circ_ref());
295 while (newVal < currentVal) {
296 c_unit_current = &c_unit_new;
297 currentVal = newVal;
298 success = true;
299 pass_->apply(c_unit_new, safe_mode, before_apply, after_apply);
300 newVal = metric_(c_unit_new.get_circ_ref());
301 }
302 if (&c_unit != c_unit_current) c_unit = *c_unit_current;
303 after_apply(c_unit, this->get_config());
304 return success;
305}

330 const PassCallback& before_apply, const PassCallback& after_apply) const {
331 before_apply(c_unit, this->get_config());

640 }

644 for (const auto& j_entry : content.at("sequence")) {
645 seq.push_back(

652 content.at("body"), custom_deserialise, custom_map_deserialise));

660 pp = std::make_shared<RepeatUntilSatisfiedPass>(body, pred);
661 } else {
