56enum class SafetyMode {
57 Audit, /* Check after every pass that the cache is being updated correctly */
58 Default, /* Check composition and check precons and postcons */

62/* Guarantee composition for CompilerPass composition:
63 Given A, B, ..., Z CompilerPasss, the Guarantee of Predicate 'p' for A >> B
64 >> ... >> Z is the last non-"Preserve" Guarantee for 'p' in list. If all
65 "Preserve", Guarantee 'p' := "Preserve"
66 */

194 explicit SequencePass(const std::vector<PassPtr>& ptvec, bool strict = true);

197 const PassCallback& before_apply = trivial_callback,
198 const PassCallback& after_apply = trivial_callback) const override {
199 before_apply(c_unit, this->get_config());
200 bool success = false;

212 private:

216/* Repeats a Pass until it returns `false`, or if `strict_check` is `true`
217 * until it stops modifying the circuit.
218 */
219class RepeatPass : public BasePass {
220 public:
221 explicit RepeatPass(const PassPtr& pass, bool strict_check = false);
222 bool apply(
223 CompilationUnit& c_unit, SafetyMode safe_mode = SafetyMode::Default,
224 const PassCallback& before_apply = trivial_callback,
225 const PassCallback& after_apply = trivial_callback) const override {
226 before_apply(c_unit, this->get_config());
227 bool success = false;
228 if (strict_check_) {
229 Circuit c0 = c_unit.get_circ_ref();
230 bool keep_going = true;
231 while (keep_going) {
232 bool rv = pass_->apply(c_unit, safe_mode, before_apply, after_apply);
233 if (rv) {
234 const Circuit& c1 = c_unit.get_circ_ref();
235 if (c0 == c1) {
236 keep_going = false;
237 } else {
238 keep_going = true;
239 success = true;
240 c0 = c1;
241 }
242 } else {
243 keep_going = false;
244 }
245 }
246 } else {
247 while (pass_->apply(c_unit, safe_mode, before_apply, after_apply))
248 success = true;
249 }
250 after_apply(c_unit, this->get_config());
251 return success;
252 }
253 std::string to_string() const override;
254 nlohmann::json get_config() const override;
255 PassPtr get_pass() const { return pass_; }
256
257 private:
258 PassPtr pass_;
259 bool strict_check_;
260};
