217 /* Repeats a Pass until it returns `false`, or if `strict_check` is `true`
218  * until it stops modifying the circuit.
219  */
220 class RepeatPass : public BasePass {
221  public:
222  explicit RepeatPass(const PassPtr& pass, bool strict_check = false);
223  bool apply(
224  CompilationUnit& c_unit, SafetyMode safe_mode = SafetyMode::Default,
225  const PassCallback& before_apply = trivial_callback,
226  const PassCallback& after_apply = trivial_callback) const override {
227  before_apply(c_unit, this->get_config());
228  bool success = false;
229  if (strict_check_) {
230  Circuit c0 = c_unit.get_circ_ref();
231  bool keep_going = true;
232  while (keep_going) {
233  bool rv = pass_->apply(c_unit, safe_mode, before_apply, after_apply);
234  if (rv) {
235  const Circuit& c1 = c_unit.get_circ_ref();
236  if (c0 == c1) {
237  keep_going = false;
238  } else {
239  keep_going = true;
240  success = true;
241  c0 = c1;
242  }
243  } else {
244  keep_going = false;
245  }
246  }
247  } else {
248  while (pass_->apply(c_unit, safe_mode, before_apply, after_apply))
249  success = true;
250  }
251  after_apply(c_unit, this->get_config());
252  return success;
253  }
254  std::string to_string() const override;
255  nlohmann::json get_config() const override;
256  PassPtr get_pass() const { return pass_; }
257 
258  private:
259  PassPtr pass_;
260  bool strict_check_;
261 };
