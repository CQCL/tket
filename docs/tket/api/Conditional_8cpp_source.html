72 std::string Conditional::get_command_str(const unit_vector_t& args) const {
73  std::stringstream out;
74  out << "IF ([";
75  if (width_ > 0) {
76  out << args.at(0).repr();
77  for (unsigned i = 1; i < width_; ++i) {
78  out << ", " << args.at(i).repr();
79  }
80  }
81  out << "] == " << value_;
82  unit_vector_t inner_args(args.begin() + width_, args.end());
83  out << ") THEN " << op_->get_command_str(inner_args);
84  return out.str();
85 }
