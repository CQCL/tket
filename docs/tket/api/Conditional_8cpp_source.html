79std::string Conditional::get_command_str(const unit_vector_t& args) const {
80 std::stringstream out;
81 out << "IF ([";
82 if (width_ > 0) {
83 out << args.at(0).repr();
84 for (unsigned i = 1; i < width_; ++i) {
85 out << ", " << args.at(i).repr();
86 }
87 }
88 out << "] == " << value_;
89 unit_vector_t inner_args(args.begin() + width_, args.end());
90 out << ") THEN " << op_->get_command_str(inner_args);
91 return out.str();
92}
