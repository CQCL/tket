100void conjugate_PauliTensor(
101 SpPauliStabiliser& qpt, OpType op, const Qubit& q, bool reverse) {
102 QubitPauliMap::iterator it = qpt.string.find(q);
103 if (it == qpt.string.end()) {
104 return;
105 }
106 std::pair<Pauli, bool> conj = conjugate_Pauli(op, it->second, reverse);
107 it->second = conj.first;
108 if (conj.second) {
109 qpt.coeff = (qpt.coeff + 2) % 4;
110 }
111}

113void conjugate_PauliTensor(

117 {{Pauli::I, Pauli::I}, {Pauli::I, Pauli::I, false}},

119 {{Pauli::I, Pauli::Y}, {Pauli::Z, Pauli::Y, false}},
120 {{Pauli::I, Pauli::Z}, {Pauli::Z, Pauli::Z, false}},
121 {{Pauli::X, Pauli::I}, {Pauli::X, Pauli::X, false}},
122 {{Pauli::X, Pauli::X}, {Pauli::X, Pauli::I, false}},
123 {{Pauli::X, Pauli::Y}, {Pauli::Y, Pauli::Z, false}},
124 {{Pauli::X, Pauli::Z}, {Pauli::Y, Pauli::Y, true}},
125 {{Pauli::Y, Pauli::I}, {Pauli::Y, Pauli::X, false}},
126 {{Pauli::Y, Pauli::X}, {Pauli::Y, Pauli::I, false}},
127 {{Pauli::Y, Pauli::Y}, {Pauli::X, Pauli::Z, true}},
128 {{Pauli::Y, Pauli::Z}, {Pauli::X, Pauli::Y, false}},
129 {{Pauli::Z, Pauli::I}, {Pauli::Z, Pauli::I, false}},
130 {{Pauli::Z, Pauli::X}, {Pauli::Z, Pauli::X, false}},
131 {{Pauli::Z, Pauli::Y}, {Pauli::I, Pauli::Y, false}},
132 {{Pauli::Z, Pauli::Z}, {Pauli::I, Pauli::Z, false}},
133 };
134 if (op != OpType::CX) {
135 throw BadOpType("Conjugations of Pauli strings only defined for CXs", op);
136 }
137 Pauli p0 = qpt.get(q0);
138 Pauli p1 = qpt.get(q1);
139 std::tuple<Pauli, Pauli, bool> conj = cx_conj_lut.at({p0, p1});
140 qpt.set(q0, std::get<0>(conj));
141 qpt.set(q1, std::get<1>(conj));
142 if (std::get<2>(conj)) {
143 qpt.coeff = (qpt.coeff + 2) % 4;
144 }
145}

147void conjugate_PauliTensor(
148 SpPauliStabiliser& qpt, OpType op, const Qubit& q0, const Qubit& q1,
149 const Qubit& q2) {
150 /* XXPhase3 gates used for conjugations always implicitly use angle π/2
151 * i.e. XXPhase3(1/2). Note that up to phase the 3-qb gate is self-inverse:
152 * XXPhase3(1/2) == exp(π/2i) XXPhase3(-1/2).
153 *
154 * We conjugate XXPhase3(1/2) by conjugating its CX-equivalent circuit
155 * __________
156 * --| |-- ------- X--H--C--H--X--
157 * | XXPhase3 | | |
158 * --| (1/2) |-- = exp(-3π/4i) --H--C--C--H--+-----X--
159 * | | | |
160 * --|__________|-- -----X--------X-----X--
161 *
162 * We can safely ignore phase differences as for each conjugated XXPhase3,
163 * we insert XXPhase3(1/2) and its dagger XXPhase3(-1/2), so that the phases
164 * always cancel out.
165 */
166 if (op != OpType::XXPhase3) {
167 throw BadOpType(
168 "3qb-Conjugations of Pauli strings only defined for XXPhase3", op);
169 }
170 SpPauliStabiliser xxi_i({{q0, Pauli::X}, {q1, Pauli::X}}, 1);
171 SpPauliStabiliser xix_i({{q0, Pauli::X}, {q2, Pauli::X}}, 1);
172 SpPauliStabiliser ixx_i({{q1, Pauli::X}, {q2, Pauli::X}}, 1);
173 if (!xxi_i.commutes_with(qpt)) qpt = qpt * xxi_i;
174 if (!xix_i.commutes_with(qpt)) qpt = qpt * xix_i;
175 if (!ixx_i.commutes_with(qpt)) qpt = qpt * ixx_i;
176}
