26ConjugationBox::ConjugationBox(
27 const Op_ptr &compute, const Op_ptr &action,
28 const std::optional<Op_ptr> uncompute)
29 : Box(OpType::ConjugationBox),
30 compute_(compute),
31 action_(action),
32 uncompute_(uncompute) {
33 op_signature_t compute_sig = compute_->get_signature();
34 op_signature_t action_sig = action_->get_signature();
35 unsigned compute_size_ = compute_sig.size();
36 unsigned action_size_ = action_sig.size();
37 unsigned compute_n_qubits_ =
38 std::count(compute_sig.begin(), compute_sig.end(), EdgeType::Quantum);
39 unsigned action_n_qubits_ =
40 std::count(action_sig.begin(), action_sig.end(), EdgeType::Quantum);
41 unsigned uncompute_size_ = 0, uncompute_n_qubits_ = 0;
42 if (uncompute_ != std::nullopt) {
43 op_signature_t uncompute_sig = uncompute_.value()->get_signature();
44 uncompute_size_ = uncompute_sig.size();
45 uncompute_n_qubits_ = std::count(
46 uncompute_sig.begin(), uncompute_sig.end(), EdgeType::Quantum);
47 }
48 if (compute_size_ != compute_n_qubits_ || action_size_ != action_n_qubits_ ||
49 uncompute_size_ != uncompute_n_qubits_) {
50 throw std::invalid_argument(
51 "ConjugationBox only supports quantum operations");
52 }
53 if (compute_size_ != action_size_ ||
54 (uncompute_ != std::nullopt && uncompute_size_ != compute_size_)) {
55 throw std::invalid_argument(
56 "Operations provided to ConjugationBox need to have the same number of "
57 "qubits");
58 }
59 signature_ = op_signature_t(compute_size_, EdgeType::Quantum);
60}

73Op_ptr ConjugationBox::transpose() const {
74 return std::make_shared<ConjugationBox>(
75 (uncompute_ == std::nullopt) ? compute_->dagger()->transpose()
76 : uncompute_.value()->transpose(),
77 action_->transpose(), compute_->transpose());

80void ConjugationBox::generate_circuit() const {
81 Circuit circ(signature_.size());
82 std::vector<unsigned> args(circ.n_qubits());
83 std::iota(args.begin(), args.end(), 0);
84 circ.add_op<unsigned>(compute_, args);
85 circ.add_op<unsigned>(action_, args);
86 if (uncompute_ != std::nullopt) {
87 circ.add_op<unsigned>(uncompute_.value(), args);
88 } else {
89 circ.add_op<unsigned>(compute_->dagger(), args);
90 }
91 circ_ = std::make_shared<Circuit>(circ);
92}

108SymSet ConjugationBox::free_symbols() const {
109 SymSet compute_syms = compute_->free_symbols();
110 SymSet action_syms = action_->free_symbols();
111 SymSet uncompute_syms;
112 if (uncompute_.has_value()) {
113 SymSet s = uncompute_.value()->free_symbols();
114 uncompute_syms.insert(s.begin(), s.end());
115 }
116 SymSet sym_set;
117 sym_set.insert(compute_syms.begin(), compute_syms.end());
118 sym_set.insert(action_syms.begin(), action_syms.end());
119 sym_set.insert(uncompute_syms.begin(), uncompute_syms.end());
120 return sym_set;
121}

123bool ConjugationBox::is_equal(const Op &op_other) const {
124 const ConjugationBox &other = dynamic_cast<const ConjugationBox &>(op_other);
125 if (id_ == other.get_id()) return true;
126 // if only one of them has uncompute_, compare the uncompute_ with
127 // the other's compute_.dagger().
128 return *compute_ == *other.compute_ && *action_ == *other.action_ &&
129 ((uncompute_ == std::nullopt && other.uncompute_ == std::nullopt) ||
130 (uncompute_ != std::nullopt && other.uncompute_ != std::nullopt &&
131 *uncompute_.value() == *other.uncompute_.value()) ||
132 (uncompute_ == std::nullopt && other.uncompute_ != std::nullopt &&
133 *compute_->dagger() == *other.uncompute_.value()) ||
134 (uncompute_ != std::nullopt && other.uncompute_ == std::nullopt &&
135 *uncompute_.value() == *other.compute_->dagger()));
136}

138nlohmann::json ConjugationBox::to_json(const Op_ptr &op) {
139 const auto &box = static_cast<const ConjugationBox &>(*op);
140 nlohmann::json j = core_box_json(box);
141 j["compute"] = box.get_compute();
142 j["action"] = box.get_action();
143 // set j["uncompute"] to null
144 j["uncompute"] = nlohmann::json();
145 std::optional<Op_ptr> uncompute = box.get_uncompute();
146 if (uncompute != std::nullopt) {
147 j["uncompute"] = uncompute.value();
148 }
149 return j;
150}

152Op_ptr ConjugationBox::from_json(const nlohmann::json &j) {
153 std::optional<Op_ptr> uncompute = std::nullopt;
154 if (j.contains("uncompute") && !j.at("uncompute").is_null()) {
155 uncompute = j.at("uncompute").get<Op_ptr>();
156 }
157 ConjugationBox box = ConjugationBox(
158 j.at("compute").get<Op_ptr>(), j.at("action").get<Op_ptr>(), uncompute);
159 return set_box_id(
160 box,
161 boost::lexical_cast<boost::uuids::uuid>(j.at("id").get<std::string>()));
162}
