39 Transform remove_discarded_ops() {
40  return Transform([](Circuit &circ) {
41  // We want to keep all vertices that have an Output or ClOutput in their
42  // causal future. Start by constructing this set, then remove the
43  // remainder.
44  VertexSet keep;
45  for (auto v_end : circ.all_outputs()) {
46  if (circ.get_OpType_from_Vertex(v_end) != OpType::Discard) {
47  // Trace back from v, adding to keep-set. We can ignore vertices
48  // that are already in the keep-set.
49  VertexSet v_frontier;
50  keep.insert(v_end);
51  v_frontier.insert(v_end);
52  while (!v_frontier.empty()) {
53  VertexSet new_v_frontier;
54  for (auto v : v_frontier) {
55  for (const Vertex &v0 : circ.get_predecessors(v)) {
56  if (keep.find(v0) == keep.end()) {
57  keep.insert(v0);
58  new_v_frontier.insert(v0);
59  }
60  }
61  }
62  v_frontier = std::move(new_v_frontier);
63  }
64  }
65  }
66  // Now remove all operations not in the keep-set.
67  VertexList to_remove;
68  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
69  if (keep.find(v) == keep.end()) {
70  OpType optype = circ.get_OpType_from_Vertex(v);
71  if (is_gate_type(optype) || is_box_type(optype)) {
72  to_remove.push_back(v);
73  }
74  }
75  }
76  circ.remove_vertices(
77  to_remove, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
78  return !to_remove.empty();
79  });
80 }

82 static std::optional<Eigen::MatrixXcd> op_unitary(Op_ptr op) {
83  try {
84  return op->get_unitary();
85  } catch (BadOpType &) {
86  std::stringstream ss;
87  ss << "Attempting to compute unitary for invalid type: " << op->get_name();
88  tket_log()->warn(ss.str());
89  return std::nullopt;
90  } catch (SymbolsNotSupported &) {
91  std::stringstream ss;
92  ss << "Attempting to compute unitary for symbolic operation: "
93  << op->get_name();
94  tket_log()->warn(ss.str());
95  return std::nullopt;
96  }
97  // Any other exception is unexpected.
98 }

103 static std::optional<unsigned> unique_unit_row(
104  const Eigen::MatrixXcd U, unsigned j) {
105  unsigned n = U.rows();
106  for (unsigned i = 0; i < n; i++) {
107  double a = std::abs(U(i, j));
108  if (std::abs(a - 1) < EPS) {
109  return i;
110  } else if (a >= EPS) {
111  return std::nullopt;
112  }
113  }
114  // If we get here there is a logical error or numerical instability.
115  std::stringstream ss;
116  ss << "Not unitary:" << std::endl << U;
117  throw std::logic_error(ss.str());
118 }

123 static VertexSet known_inputs_only(
124  const Circuit &circ, const VertexSet &F,
125  const std::map<Edge, bool> &qvals) {
126  VertexSet v_frontier;
127  for (const Vertex &v : F) {
128  EdgeVec v_q_inedges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
129  if (std::all_of(
130  v_q_inedges.begin(), v_q_inedges.end(),
131  [&qvals](const Edge &e) { return qvals.find(e) != qvals.end(); })) {
132  v_frontier.insert(v);
133  }
134  }
135  return v_frontier;
136 }

138 Transform simplify_initial(
139  AllowClassical allow_classical, CreateAllQubits create_all_qubits,
140  std::shared_ptr<const Circuit> xcirc) {
141  return Transform([allow_classical, create_all_qubits, xcirc](Circuit &circ) {
142  if (create_all_qubits == CreateAllQubits::Yes) {
143  circ.qubit_create_all();
144  }
145 
146  // Find all Create and Reset vertices.
147  VertexSet zeroing_vertices;
148  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
149  OpType optype = circ.get_OpType_from_Vertex(v);
150  if (optype == OpType::Create || optype == OpType::Reset) {
151  zeroing_vertices.insert(v);
152  }
153  }
154 
155  // Partial map from quantum edges to values.
156  std::map<Edge, bool> qvals;
157 
158  // Assign 0 values to all edges coming out of zeroing vertices, and
159  // construct the set of their target vertices.
160  VertexSet F;
161  for (auto z : zeroing_vertices) {
162  EdgeVec z_outedges = circ.get_all_out_edges(z);
163  TKET_ASSERT(z_outedges.size() == 1);
164  Edge z_out = z_outedges[0];
165  TKET_ASSERT(circ.get_edgetype(z_out) == EdgeType::Quantum);
166  qvals[z_out] = false;
167  F.insert(circ.target(z_out));
168  }
169 
170  // Construct frontier of vertices with all-known input values.
171  VertexSet v_frontier = known_inputs_only(circ, F, qvals);
172 
173  // Partial map from vertices to sequences of X gates to replace them.
174  std::map<Vertex, std::vector<bool>> reductions;
175 
176  // Partial map from Measure vertices to bits to set after measure.
177  std::map<Vertex, bool> measurebits;
178 
179  while (!v_frontier.empty()) {
180  // Simplify vertices in the frontier that we can, and move it on.
181  F.clear();
182  for (const Vertex &v : v_frontier) {
183  // If there are any Boolean inputs to v, skip it.
184  if (circ.n_in_edges_of_type(v, EdgeType::Boolean) != 0) {
185  continue;
186  }
187 
188  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
189  std::optional<Eigen::MatrixXcd> U = op_unitary(op);
190  bool is_measure = (op->get_type() == OpType::Measure);
191 
192  if (!U && (!is_measure || allow_classical == AllowClassical::No)) {
193  continue;
194  }
195 
196  // Compute input state.
197  EdgeVec v_q_inedges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
198  unsigned n_q = v_q_inedges.size();
199  std::vector<bool> v_invals(n_q);
200  for (unsigned i = 0; i < n_q; i++) {
201  v_invals[i] = qvals[v_q_inedges[i]];
202  }
203 
204  if (U) {
205  // Compute relevant column J of U.
206  unsigned J = 0, pow2 = 1u << n_q;
207  for (unsigned i = 0; i < n_q; i++) {
208  pow2 >>= 1;
209  if (v_invals[i]) {
210  J |= pow2;
211  }
212  }
213 
214  // Check if there is a unique I s.t. |U(I,J)| == 1.
215  std::optional<unsigned> I = unique_unit_row(*U, J);
216  if (!I) continue;
217 
218  // Convert I to a vector of bool
219  std::vector<bool> v_outvals(n_q);
220  for (unsigned i = 0; i < n_q; i++) {
221  v_outvals[i] = (*I >> (n_q - 1 - i)) & 1;
222  }
223 
224  // Label out-edges of v; construct equivalent X-gate rep.
225  EdgeVec v_outedges = circ.get_all_out_edges(v);
226  TKET_ASSERT(v_outedges.size() == n_q);
227  std::vector<bool> x_gates(n_q);
228  for (unsigned i = 0; i < n_q; i++) {
229  bool outval = v_outvals[i];
230  qvals[v_outedges[i]] = outval;
231  x_gates[i] = v_invals[i] ^ outval;
232  }
233 
234  // Record vertex for later replacement with X-gates.
235  reductions[v] = x_gates;
236 
237  // Add all successors of v to potential next frontier F.
238  for (const Edge &e : v_outedges) {
239  F.insert(circ.target(e));
240  }
241  } else {
242  TKET_ASSERT(allow_classical == AllowClassical::Yes);
243  TKET_ASSERT(n_q == 1);
244  measurebits[v] = v_invals[0];
245  }
246  }
247 
248  // Replace v_frontier with vertices from F having all-known inputs.
249  v_frontier = known_inputs_only(circ, F, qvals);
250  }
251 
252  // Perform substitutions.
253  VertexList bin;
254  for (const auto &[v, x_gates] : reductions) {
255  unsigned n_q = x_gates.size();
256  Circuit xs_circ(n_q);
257  for (unsigned i = 0; i < n_q; i++) {
258  if (x_gates[i]) {
259  if (xcirc) {
260  unit_map_t qm;
261  qm.insert({Qubit(i), Qubit(0)});
262  xs_circ.append_with_map(*xcirc, qm);
263  } else {
264  xs_circ.add_op<unsigned>(OpType::X, {i});
265  }
266  }
267  }
268  EdgeVec v_in_edges = circ.get_in_edges(v); // all Quantum
269  EdgeVec v_out_edges = circ.get_all_out_edges(v); // all Quantum
270  TKET_ASSERT(v_in_edges.size() == n_q);
271  TKET_ASSERT(v_out_edges.size() == n_q);
272  Subcircuit subc{v_in_edges, v_out_edges, {v}};
273  circ.substitute(xs_circ, subc, Circuit::VertexDeletion::No);
274  bin.push_back(v);
275  }
276  for (const auto &[v, bitval] : measurebits) {
277  Circuit newc(1, 1);
278  // Replace the measure with a set-bit.
279  std::vector<bool> values = {bitval};
280  Op_ptr setbitop = std::make_shared<SetBitsOp>(values);
281  newc.add_op<Bit>(setbitop, {Bit(0)});
282  EdgeVec q_in_edges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
283  TKET_ASSERT(q_in_edges.size() == 1);
284  EdgeVec q_out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
285  TKET_ASSERT(q_out_edges.size() == 1);
286  EdgeVec c_in_edges = circ.get_in_edges_of_type(v, EdgeType::Classical);
287  TKET_ASSERT(c_in_edges.size() == 1);
288  EdgeVec c_out_edges = circ.get_out_edges_of_type(v, EdgeType::Classical);
289  TKET_ASSERT(c_out_edges.size() == 1);
290  EdgeVec b_out_edges = circ.get_out_edges_of_type(v, EdgeType::Boolean);
291  Subcircuit subc{q_in_edges, q_out_edges, c_in_edges,
292  c_out_edges, b_out_edges, {v}};
293  circ.substitute(newc, subc, Circuit::VertexDeletion::No);
294  bin.push_back(v);
295  }
296  bool changed = !bin.empty();
297  circ.remove_vertices(
298  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
299 
300  return changed;
301  });
302 }

304 std::optional<std::shared_ptr<const ClassicalTransformOp>> classical_transform(
305  Op_ptr op) {
306  std::optional<Eigen::MatrixXcd> U = op_unitary(op);
307  if (!U) return std::nullopt;
308  unsigned n = op->get_desc().n_qubits().value();
309  unsigned pow2n = 1u << n;
310  TKET_ASSERT(U->cols() == pow2n);
311  std::vector<uint32_t> values(pow2n);
312  for (uint32_t J = 0; J < pow2n; J++) {
313  // Look at the column U[*,J]. Is there a unique nonzero element?
314  std::optional<unsigned> I = unique_unit_row(*U, J);
315  if (!I) return std::nullopt;
316  values[reverse_bits(J, n)] = reverse_bits(*I, n);
317  }
318  return std::make_shared<ClassicalTransformOp>(n, values);
319 }

321 Transform simplify_measured() {
322  return Transform([](Circuit &circ) {
323  // First construct the set of all Measure vertices that are followed by
324  // Discard vertices (and no Boolean out-edges).
325  VertexSet M;
326  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
327  if (circ.get_OpType_from_Vertex(v) == OpType::Measure) {
328  if (circ.n_out_edges_of_type(v, EdgeType::Boolean) == 0) {
329  EdgeVec m_q_outs = circ.get_out_edges_of_type(v, EdgeType::Quantum);
330  TKET_ASSERT(m_q_outs.size() == 1);
331  Edge m_q_out = m_q_outs[0];
332  Vertex v1 = circ.target(m_q_out);
333  if (circ.get_OpType_from_Vertex(v1) == OpType::Discard) {
334  M.insert(v);
335  }
336  }
337  }
338  }
339  bool changed = false;
340  bool carry_on;
341  do {
342  carry_on = false;
343  VertexList bin;
344  // Find all classical maps all of whose successors are in M
345  for (const Vertex &v : M) {
346  VertexVec preds = circ.get_predecessors(v);
347  for (const Vertex &v0 : preds) {
348  // Any Boolean inputs?
349  if (circ.n_in_edges_of_type(v0, EdgeType::Boolean) != 0) {
350  continue;
351  }
352  // No. Are all successors in M?
353  VertexVec succs = circ.get_successors(v0);
354  if (std::any_of(succs.begin(), succs.end(), [&](const Vertex &u) {
355  return M.find(u) == M.end();
356  })) {
357  continue;
358  }
359  // Yes. Is it a classical map?
360  Op_ptr op = circ.get_Op_ptr_from_Vertex(v0);
361  std::optional<std::shared_ptr<const ClassicalTransformOp>> cm =
362  classical_transform(op);
363  if (!cm) continue;
364  // Yes. Remove v0.
365  unsigned n_qb = succs.size();
366  circ.remove_vertex(
367  v0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
368  bin.push_back(v0);
369  // Insert *cm on the classical target wires.
370  EdgeVec cl_edges(n_qb);
371  for (unsigned i = 0; i < n_qb; i++) {
372  EdgeVec m_c_outs =
373  circ.get_out_edges_of_type(succs[i], EdgeType::Classical);
374  TKET_ASSERT(m_c_outs.size() == 1);
375  cl_edges[i] = m_c_outs[0];
376  }
377  Subcircuit cl_subc{{}, {}, cl_edges, cl_edges, {}, {}};
378  Circuit cl_circ(0, n_qb);
379  std::vector<unsigned> args(n_qb);
380  std::iota(args.begin(), args.end(), 0);
381  cl_circ.add_op<unsigned>(*cm, args);
382  circ.substitute(cl_circ, cl_subc, Circuit::VertexDeletion::No);
383  changed = true;
384  carry_on = true;
385  }
386  }
387  circ.remove_vertices(
388  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
389  } while (carry_on);
390  return changed;
391  });
392 }

394 std::pair<Circuit, Circuit> separate_classical(const Circuit &circ) {
395  // Initialize the two circuits to return.
396  qubit_vector_t qubits = circ.all_qubits();
397  bit_vector_t bits = circ.all_bits();
398  Circuit c0, c1;
399  for (const Qubit &qb : qubits) {
400  c0.add_qubit(qb);
401  }
402  for (const Bit &b : bits) {
403  c0.add_bit(b);
404  c1.add_bit(b);
405  }
406 
407  // Get the command list.
408  std::vector<Command> cmds = circ.get_commands();
409 
410  // Find the final Measure (or ClInput if no Measure) on each bit.
411  VertexVec c_in = circ.c_inputs();
412  unsigned n_bits = bits.size();
413  TKET_ASSERT(n_bits == c_in.size());
414  std::map<Bit, Vertex> final_vert_on_bit;
415  for (unsigned i = 0; i < n_bits; i++) {
416  final_vert_on_bit[bits[i]] = c_in[i];
417  }
418  for (const Command &cmd : cmds) {
419  OpType optype = cmd.get_op_ptr()->get_type();
420  if (optype == OpType::Measure) {
421  bit_vector_t cmd_bits = cmd.get_bits();
422  TKET_ASSERT(cmd_bits.size() == 1);
423  final_vert_on_bit[cmd_bits[0]] = cmd.get_vertex();
424  }
425  }
426 
427  // Construct the set of final vertices.
428  VertexSet final_verts;
429  for (const Bit &b : bits) {
430  final_verts.insert(final_vert_on_bit[b]);
431  }
432 
433  // Construct the set of vertices to go in c1.
434  // Initially include final_verts; later exclude them.
435  VertexSet c1_verts(final_verts);
436  for (const Command &cmd : cmds) {
437  Vertex v = cmd.get_vertex();
438  VertexVec preds = circ.get_predecessors(v);
439  if (std::all_of(preds.begin(), preds.end(), [&c1_verts](const Vertex &v) {
440  return c1_verts.find(v) != c1_verts.end();
441  })) {
442  c1_verts.insert(v);
443  }
444  }
445  for (const Vertex &v : final_verts) {
446  c1_verts.erase(v);
447  }
448 
449  // Step through the circuit, filling in c0 and c1.
450  for (const Command &cmd : cmds) {
451  Op_ptr op = cmd.get_op_ptr();
452  unit_vector_t args = cmd.get_args();
453  if (c1_verts.find(cmd.get_vertex()) == c1_verts.end()) {
454  c0.add_op(op, args);
455  } else {
456  c1.add_op(op, args);
457  }
458  }
459 
460  return {c0, c1};
461 }
