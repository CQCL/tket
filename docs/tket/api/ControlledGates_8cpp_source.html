46 // `n` = size of incrementer, circuit returned is of size `n+1`
47 /* this is slightly less efficient than perhaps it could be -- asymptotically it
48  is still good. In an ideal world, this would decompose the incrementers
49  smarter for the "even" case */
50 Circuit incrementer_borrow_1_qubit(unsigned n) {
51  bool is_odd = n % 2;
52  Circuit circ(n + 1);
53  if (n < 6) {
54  if (n > 4) circ.append_qubits(C4X_normal_decomp(), {0, 1, 2, 3, 4});
55  if (n > 3) circ.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
56  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {0, 1, 2});
57  if (n > 1) circ.add_op<unsigned>(OpType::CX, {0, 1});
58  if (n > 0) circ.add_op<unsigned>(OpType::X, {0});
59  return circ;
60  }
61  unsigned j;
62  unsigned k;
63  // j is bottom qubits, k is top qubits
64  // k + j = n + 1 (total no. of qbs)
65  if (is_odd) {
66  /* if the number of bits we are incrementing is odd, we can just split the
67  incrementer in 2 and use the `incrementer_borrow_n_qubits` method twice */
68  j = (n + 1) / 2, k = (n + 1) / 2;
69  } else {
70  /* otherwise, we will also have to pull out a cnx */
71  j = n / 2 + 1, k = n / 2;
72  }
73 
74  Circuit top_incrementer = incrementer_borrow_n_qubits(k);
75  std::vector<unsigned> top_qbs(2 * k);
76  for (unsigned i = 0; i != k; ++i) {
77  top_qbs[2 * i] = i + k; // borrowed qubits
78  top_qbs[2 * i + 1] = i; // qbs we are trying to increment
79  }
80 
81  Circuit cnx_top;
82  std::vector<unsigned> cnx1_qbs;
83  if (k == 3) { // code is unreachable if k<3
84  cnx_top = C3X_normal_decomp();
85  cnx1_qbs = {0, 1, 2, n};
86  } else if (k == 4) {
87  cnx_top = C4X_normal_decomp();
88  cnx1_qbs = {0, 1, 2, 3, n};
89  } else {
90  cnx_top = lemma72(k); // k controls on cnx
91  cnx1_qbs.resize(
92  2 * k - 2); // size of replacement using borrowed qbs = 2*k-1
93  std::iota(cnx1_qbs.begin(), cnx1_qbs.end(), 0);
94  cnx1_qbs.push_back(n); // target is last qubit
95  }
96 
97  Circuit bottom_incrementer;
98  std::vector<unsigned> bot_qbs;
99  if (is_odd) {
100  bottom_incrementer = incrementer_borrow_n_qubits(j);
101  bot_qbs.resize(2 * j);
102  for (unsigned i = 0; i != j; ++i) {
103  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
104  if (i != 0)
105  bot_qbs[2 * i + 1] =
106  i + j -
107  1; // 3,5...n //other qbs we are actually trying to increment
108  }
109  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
110  } else {
111  if (j == 4) { // code is unreachable if j<4
112  bottom_incrementer.add_blank_wires(4);
113  bottom_incrementer.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
114  bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
115  bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
116  bottom_incrementer.add_op<unsigned>(OpType::X, {0});
117  bot_qbs = {n, n - 3, n - 2, n - 1};
118  } else if (j == 5) {
119  bottom_incrementer.add_blank_wires(5);
120  bottom_incrementer.append_qubits(C4X_normal_decomp(), {0, 1, 2, 3, 4});
121  bottom_incrementer.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
122  bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
123  bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
124  bottom_incrementer.add_op<unsigned>(OpType::X, {0});
125  bot_qbs = {n, n - 4, n - 3, n - 2, n - 1};
126  } else {
127  // insert peeled-out cnx
128  Circuit cnx_bot = lemma72(j - 1);
129  std::vector<unsigned> cnx2_qbs(
130  2 * j - 3); // lemma 7.2 uses 2j-3 qubits for a (j-1)-controlled X
131  for (unsigned i = 0; i < j - 2; ++i) {
132  cnx2_qbs[i] = k + i;
133  }
134  cnx2_qbs[j - 2] = n; // replace the last control
135  for (unsigned i = 0; i < j - 3; ++i) {
136  cnx2_qbs[j + i - 1] = i;
137  }
138  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
139 
140  circ.append_qubits(cnx_bot, cnx2_qbs);
141  bottom_incrementer = incrementer_borrow_n_qubits(
142  j - 1); // insert incrementer over remaining qubits
143  bot_qbs.resize(2 * j - 2);
144  for (unsigned i = 0; i != j - 1; ++i) {
145  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
146  if (i != 0)
147  bot_qbs[2 * i + 1] =
148  i + k -
149  1; // 3,5...n //other qbs we are actually trying to increment
150  }
151  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
152  }
153  }
154 
155  circ.append_qubits(bottom_incrementer, bot_qbs);
156  circ.add_op<unsigned>(
157  OpType::X,
158  {n}); // to convert controlled-incrementer to larger incrementer
159  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
160  circ.append_qubits(cnx_top, cnx1_qbs);
161  if (!is_odd && j > 5) {
162  // insert peeled-out cnx
163  Circuit cnx_bot = lemma72(j - 1);
164  std::vector<unsigned> cnx2_qbs(2 * j - 3);
165  for (unsigned i = 0; i < j - 1; ++i) {
166  cnx2_qbs[i] = k + i;
167  }
168  cnx2_qbs[j - 2] = n; // the last control
169  for (unsigned i = 0; i < j - 3; ++i) {
170  cnx2_qbs[j + i - 1] = i;
171  }
172  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
173  circ.append_qubits(cnx_bot, cnx2_qbs);
174  }
175  circ.append_qubits(bottom_incrementer, bot_qbs);
176  circ.add_op<unsigned>(OpType::X, {n});
177  circ.append_qubits(cnx_top, cnx1_qbs);
178  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
179  circ.append_qubits(top_incrementer, top_qbs);
180  return circ;
181 }

183 // an optimised version of
184 // https://algassert.com/circuits/2015/06/12/Constructing-Large-Increment-Gates.html
185 /* every second qubit (0,2,4...) is a borrowed qubit */
186 Circuit incrementer_borrow_n_qubits(unsigned n) {
187  const unsigned N = 2 * n;
188  Circuit circ(N);
189  /* deal with small cases where borrowing qubits is unnecessary */
190  if (n < 6) {
191  if (n > 4) circ.append_qubits(C4X_normal_decomp(), {1, 3, 5, 7, 9});
192  if (n > 3) circ.append_qubits(C3X_normal_decomp(), {1, 3, 5, 7});
193  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {1, 3, 5});
194  if (n > 1) circ.add_op<unsigned>(OpType::CX, {1, 3});
195  if (n > 0) circ.add_op<unsigned>(OpType::X, {1});
196  return circ;
197  }
198 
199  for (unsigned i = 1; i < N; ++i) {
200  if (i % 2)
201  circ.add_op<unsigned>(OpType::CX, {0, i});
202  else
203  circ.add_op<unsigned>(OpType::X, {i});
204  }
205 
206  circ.add_op<unsigned>(OpType::X, {N - 1});
207 
208  for (unsigned i = 2; i < N; ++(++i)) {
209  std::vector<unsigned> ladder_down_qbs = {i - 2, i - 1, i};
210  circ.append_qubits(ladder_down(), ladder_down_qbs);
211  }
212  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
213  for (unsigned i = N - 2; i > 1; --(--i)) {
214  std::vector<unsigned> tof_up_qbs = {i - 2, i - 1, i};
215  circ.add_op<unsigned>(OpType::CCX, tof_up_qbs);
216  }
217 
218  for (unsigned i = 2; i < N; ++(++i)) {
219  std::vector<unsigned> ladder_down_2_qbs = {i - 2, i - 1, i};
220  circ.append_qubits(ladder_down_2(), ladder_down_2_qbs);
221  }
222  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
223  for (unsigned i = N - 2; i > 1; --(--i)) {
224  std::vector<unsigned> ladder_up_qbs = {i - 2, i - 1, i};
225  circ.append_qubits(ladder_up(), ladder_up_qbs);
226  }
227  for (unsigned i = 1; i < N; ++(++i))
228  circ.add_op<unsigned>(OpType::CX, {0, i});
229  return circ;
230 }

232 // decompose CnX gate using
233 // https://algassert.com/circuits/2015/06/22/Using-Quantum-Gates-instead-of-Ancilla-Bits.html
234 // `n` = no. of controls
235 Circuit CnX_normal_decomp(unsigned n) {
236  /* handle low qb edge cases */
237  bool insert_c4xs; // dictate whether to add C4Xs or n>4 controlled Xs
238  // when bootstrapping
239  switch (n) {
240  case 0: {
241  return X();
242  }
243  case 1: {
244  return CX();
245  }
246  case 2: {
247  return CCX_normal_decomp();
248  }
249  case 3: {
250  return C3X_normal_decomp();
251  }
252  case 4: {
253  return C4X_normal_decomp();
254  }
255  case 5: {
256  insert_c4xs = true;
257  break;
258  }
259  default: {
260  insert_c4xs = false;
261  break;
262  }
263  }
264 
265  Circuit circ(n + 1);
266  std::vector<unsigned> cnx_qbs(n - 1);
267  std::iota(cnx_qbs.begin(), cnx_qbs.end(), 0);
268  cnx_qbs.push_back(n);
269  // first, bootstrap an ancilla qubit
270  circ.add_op<unsigned>(OpType::H, {n});
271  Vertex cnx1;
272  if (insert_c4xs)
273  circ.append_qubits(C4X_normal_decomp(), {cnx_qbs});
274  else {
275  cnx1 = circ.add_op<unsigned>(
276  OpType::CnX,
277  {cnx_qbs}); /* the CnXs can be decomposed using lemma 7.3 */
278  }
279  circ.add_op<unsigned>(OpType::Tdg, {n});
280  Vertex cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
281  if (!insert_c4xs) {
282  Edge e1 = circ.get_nth_in_edge(cx, 0);
283  lemma73(circ, {e1, cnx1}); // replace first CnX using lemma 7.3
284  }
285  circ.add_op<unsigned>(OpType::T, {n});
286  Vertex cnx2;
287  if (insert_c4xs)
288  circ.append_qubits(C4X_normal_decomp(), {cnx_qbs});
289  else {
290  cnx2 = circ.add_op<unsigned>(OpType::CnX, {cnx_qbs});
291  }
292  circ.add_op<unsigned>(OpType::Tdg, {n});
293  cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
294  Edge e2 = circ.get_nth_in_edge(cx, 0);
295  if (!insert_c4xs) lemma73(circ, {e2, cnx2});
296  circ.add_op<unsigned>(OpType::T, {n});
297  circ.add_op<unsigned>(OpType::H, {n});
298 
299  // add incremented shift pattern
300  Circuit incrementer = incrementer_borrow_1_qubit(n);
301  circ.append(incrementer);
302 
303  // z rotation layer #1
304  std::vector<Op_ptr> z_rots(n);
305  double angle = -0.25;
306  for (unsigned i = 0; i < n - 1; ++i) {
307  unsigned m = n - 1 - i;
308  z_rots[i] = get_op_ptr(OpType::Rz, angle);
309  circ.add_op<unsigned>(z_rots[i], {m});
310  angle /= 2;
311  }
312 
313  // decremented shift pattern
314  for (unsigned i = 0; i < n; ++i) {
315  circ.add_op<unsigned>(OpType::X, {i});
316  }
317  circ.append(incrementer);
318  for (unsigned i = 0; i < n; ++i) {
319  circ.add_op<unsigned>(OpType::X, {i});
320  }
321 
322  // z rotation layer #2
323  for (unsigned i = 0; i < n - 1; ++i) {
324  unsigned m = n - 1 - i;
325  Expr ang = z_rots[i]->get_params()[0];
326  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {m});
327  }
328  Expr ang = z_rots[n - 2]->get_params()[0];
329  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {0});
330 
331  const Op_ptr ccx = get_op_ptr(OpType::CCX);
332  circ.substitute_all(CCX_normal_decomp(), ccx);
333 
334  circ.add_phase(std::pow(0.5, n + 1));
335  return circ;
336 }

338 // unsigned -> which column the change is in
339 static unsigned find_first_differing_val(
340  const std::deque<bool>& d1, const std::deque<bool>& d2) {
341  unsigned N = d1.size();
342  if (N != d2.size())
343  throw ControlDecompError(
344  "Error in `find_first_differing_val`: Deques are of differing "
345  "sizes");
346  for (unsigned i = 0; i < N; ++i) {
347  if (d1[i] != d2[i]) return i;
348  }
349  throw ControlDecompError(
350  "Error in `find_first_differing_val`: No change between deques");
351 }

353 // optimal decomposition of CnRy and CnZ for 2 < n < 8 according to 1995
354 // paper... can do better with ZH calculus?
355 static Circuit lemma71(
356  unsigned arity, const Circuit& v_rep, const Circuit& v_dg_rep) {
357  unsigned m_controls = arity - 1;
358  if (m_controls < 2)
359  throw Unsupported(
360  "No point using Lemma 7.1 to decompose a gate with less than 2 "
361  "controls");
362 
363  GrayCode gc = gen_graycode(m_controls);
364 
365  Circuit rep(arity);
366 
367  unsigned control_qb = 0;
368  unsigned last = 0;
369  rep.append_with_map(
370  v_rep, {{Qubit(0), Qubit(0)}, {Qubit(1), Qubit(m_controls)}});
371  // we ignore the 0...0 term, and the first one is always trivial
372  // so start from 2
373  for (unsigned i = 2; i < gc.size(); ++i) {
374  unsigned change = find_first_differing_val(gc[i], gc[i - 1]);
375  for (unsigned j = 1; j < gc[i].size(); ++j) {
376  if (gc[i][j] == 1) last = j;
377  }
378  if (change < control_qb) {
379  rep.add_op<unsigned>(OpType::CX, {change, control_qb});
380  }
381  if (change == control_qb) {
382  throw ControlDecompError("Error in graycode iteration");
383  }
384  if (change > control_qb) {
385  rep.add_op<unsigned>(OpType::CX, {control_qb, change});
386  }
387 
388  if ((i % 2) == 0) {
389  rep.append_with_map(
390  v_dg_rep, {{Qubit(0), Qubit(last)}, {Qubit(1), Qubit(m_controls)}});
391  } else
392  rep.append_with_map(
393  v_rep, {{Qubit(0), Qubit(last)}, {Qubit(1), Qubit(m_controls)}});
394  control_qb = last;
395  }
396  auto [vit, vend] = boost::vertices(rep.dag);
397  VertexSet bin;
398  for (auto next = vit; vit != vend; vit = next) {
399  ++next;
400  Vertex v = *vit;
401  if (!bin.contains(v)) {
402  Op_ptr op = rep.get_Op_ptr_from_Vertex(v);
403  OpType optype = op->get_type();
404  if (is_multi_qubit_type(optype) && optype != OpType::CX) {
405  Circuit replacement = with_CX(as_gate_ptr(op));
406  rep.substitute(replacement, v, Circuit::VertexDeletion::No);
407  bin.insert(v);
408  }
409  }
410  }
411  rep.remove_vertices(
412  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
413  return rep;
414 }

416 //`control_m` = number of controls
417 static Circuit lemma72(unsigned control_m) {
418  if (control_m < 3)
419  throw Unsupported(
420  "Cannot decompose a gate with " + std::to_string(control_m) +
421  " controls using Lemma 7.2");
422  const unsigned n = 2 * control_m - 1;
423 
424  Circuit ccx_circ(n);
425  unsigned diff = n - control_m;
426  for (unsigned i = control_m - 1; i > 1; --i) {
427  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
428  }
429  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
430  for (unsigned i = 2; i < control_m; ++i) {
431  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
432  }
433  for (unsigned i = control_m - 2; i > 1; --i) {
434  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
435  }
436  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
437  for (unsigned i = 2; i < control_m - 1; ++i) {
438  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
439  }
440  if (ccx_circ.count_gates(OpType::CCX) != 4 * (control_m - 2))
441  throw ControlDecompError("Error in Lemma 7.2: CCX gate count is incorrect");
442  return ccx_circ;
443 }

445 /* this is specifically for performing corollary 7.4 via lemma 7.3 & lemma 7.2 -
446  * optimal decomp of a CnX gate */
447 /* for corollary 7.4, n >= 7 */
448 // this is a decomposition of a CnX gate using one dirty ancilla
449 static void lemma73(Circuit& circ, const std::pair<Edge, Vertex>& pairy) {
450  Vertex original_cnx = pairy.second;
451  Edge original_spare_edge = pairy.first;
452  EdgeVec in_edges = circ.get_in_edges(original_cnx);
453  const unsigned N =
454  in_edges.size() + 1; // number of qubits in new circuit to replace
455  if (N < 5)
456  throw Unsupported(
457  "Lemma 7.3 cannot decompose CnX with n = " + std::to_string(N - 1));
458 
459  EdgeVec out_edges = circ.get_all_out_edges(original_cnx);
460 
461  in_edges.insert(in_edges.begin() + in_edges.size() - 1, original_spare_edge);
462  out_edges.insert(
463  out_edges.begin() + out_edges.size() - 1, original_spare_edge);
464 
465  Subcircuit to_delete{in_edges, out_edges, {original_cnx}};
466  bool odd_N = N % 2;
467  unsigned m1 = (N + 1) / 2; // number of controls on the first type of CnX
468  unsigned m2 = N - m1 - 1; // number of controls on the second type of CnX
469 
470  // make new circuit to substitute later
471  Circuit new_circ(N);
472  const Op_ptr cnx_op1 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m1 + 1);
473  const Op_ptr cnx_op2 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m2 + 1);
474  std::vector<unsigned> qbs_m1(m1 + 1);
475  std::iota(qbs_m1.begin(), qbs_m1.begin() + m1, 0);
476  qbs_m1[m1] = N - 1;
477  std::vector<unsigned> qbs_m2(m2 + 1);
478  std::iota(qbs_m2.begin(), qbs_m2.end(), N - (m2 + 1));
479 
480  // add ladder of CnXs to the correct qubits
481  Vertex a = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
482  Vertex b = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
483  Vertex c = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
484  Vertex d = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
485 
486  /* first, replace vertex a, putting circuit at back of new_circ */
487  unsigned cutsize = odd_N ? N : N - 1;
488  EdgeVec cut1(cutsize);
489  VertexVec out_verts = new_circ.q_outputs();
490  if (odd_N) {
491  for (unsigned i = 0; i < N - 2; ++i)
492  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
493  cut1[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
494  cut1[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
495  } else {
496  for (unsigned i = 0; i < cutsize; ++i)
497  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
498  }
499 
500  Circuit a_replacement;
501  if (m1 == 1) {
502  a_replacement = CX();
503  } else if (m1 == 2) {
504  a_replacement = CCX();
505  } else
506  a_replacement = lemma72(m1); // returns circuit of size 2*m1 - 1
507  new_circ.cut_insert(a_replacement, cut1);
508  new_circ.remove_vertex(
509  a, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
510 
511  VertexSet normal_decomp_vertices;
512 
513  Circuit b_replacement;
514  if (m2 == 1) {
515  b_replacement = CX();
516  } else if (m2 == 2) {
517  b_replacement = CCX();
518  } else

627 static void lemma79(
628  Circuit& replacement, unsigned N, const Expr& alpha, const Expr& theta,
629  const Expr& beta, candidate_t& CCX_candidates) {
630  replacement.add_blank_wires(N);
631 
632  // Add Controlled C
633  if (!equiv_0(beta - alpha, 8)) {
634  replacement.add_op<unsigned>(
635  OpType::CRz, {(beta - alpha) / 2.}, {N - 2, N - 1});
636  }
637  // Add the first CnX
638  std::vector<unsigned> cnx_qbs(N - 1);
639  std::iota(cnx_qbs.begin(), --cnx_qbs.end(), 0);
640  cnx_qbs[N - 2] = N - 1;
641  const Op_ptr cnx = get_op_ptr(OpType::CnX, std::vector<Expr>{}, N - 1);
642  Vertex firstCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
643 
644  // Add Controlled B
645  VertexVec vBs;
646  if (!equiv_0(alpha + beta, 8)) {
647  Vertex vB1 = replacement.add_op<unsigned>(
648  OpType::CRz, {(-alpha - beta) / 2.}, {N - 2, N - 1});
649  vBs.push_back(vB1);
650  }
651  if (!equiv_0(theta, 8)) {
652  Vertex vB2 = replacement.add_op<unsigned>(
653  OpType::CRy, {-theta / 2.}, {N - 2, N - 1});
654  vBs.push_back(vB2);
655  }
656  // At least one of vB1 and vB2 should be set, otherwise it implies that the
657  // SU(2) is identity
658  if (vBs.empty()) {
659  throw ControlDecompError(
660  "Unknown error in Lemma 7.9: identity not rejected");
661  }
662 
663  // Add the second CnX
664  Vertex secondCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
665 
666  // Add Controlled A
667  if (!equiv_0(theta, 8)) {
668  replacement.add_op<unsigned>(OpType::CRy, {theta / 2.}, {N - 2, N - 1});
669  }
670  if (!equiv_0(alpha, 4)) {
671  replacement.add_op<unsigned>(OpType::CRz, {alpha}, {N - 2, N - 1});
672  }
673  Edge first_e = replacement.get_nth_in_edge(vBs[0], 0);
674  Edge second_e = replacement.get_nth_out_edge(vBs.back(), 0);
675  CCX_candidates.push_back({first_e, firstCnX});
676  CCX_candidates.push_back({second_e, secondCnX});
677 }

679 static Circuit CU_to_CU3(const Eigen::Matrix2cd& u) {
680  Circuit c(2);
681  std::vector<double> tk1_angles = tk1_angles_from_unitary(u);
682  Expr theta = tk1_angles[1];
683  Expr phi = tk1_angles[0] - 0.5;
684  Expr lambda = tk1_angles[2] + 0.5;
685  Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
686  c.add_op<unsigned>(OpType::U1, t, {0});
687  c.add_op<unsigned>(OpType::CU3, {theta, phi, lambda}, {0, 1});
688  c.remove_noops();
689  return c;

943 Circuit CnSU2_linear_decomp(
944  unsigned n, const Expr& alpha, const Expr& theta, const Expr& beta) {
945  // W == I iff one of the following two conditions is met
946  // 1. t/2 is even, and (a + b)/2 is even
947  // 2. t/2 is odd, and (a + b)/2 is odd
948  // check if SU(2) is identity
949  if ((equiv_0(theta / 2., 2) && equiv_0((alpha + beta) / 2., 2)) ||
950  (equiv_val(theta / 2., 1., 2) && equiv_val((alpha + beta) / 2., 1., 2))) {
951  return Circuit(n + 1);
952  }
953 
954  Circuit circ;
955 
956  if (n == 0) {
957  // add tk1
958  circ.add_blank_wires(1);
959  circ.add_op<unsigned>(OpType::TK1, {alpha + 0.5, theta, beta - 0.5}, {0});
960  return circ;
961  }
962 
963  // SU(2) matrix W expressed as Rz(a)Ry(t)Rz(b)
964  Expr a = alpha;
965  Expr b = beta;
966  Expr t = theta;
967 
968  // Lemma 4.3: W = A*X*B*X*C
969  // By lemma 5.4, C is identity iff W can be expressed as Rz(a')Ry(t')Rz(a')
970  // We handle the following two cases
971  // if (a-b)/2 is even, a' = (a + b)/2, t' = t
972  // if (a-b)/2 is odd, a' = (a + b)/2, t' = - t
973  if (equiv_0((a - b) / 2., 2)) {
974  a = (a + b) / 2.;
975  b = a;
976  } else if (equiv_val((a - b) / 2., 1., 2)) {
977  a = (a + b) / 2.;
978  b = a;
979  t = -t;
980  }
981 
982  // We test whether W can be expressed as a single Ry(t'')
983  if (equiv_0((a - b) / 2., 2)) {
984  if (equiv_val((a + b) / 2., 1., 2)) {
985  // if (a-b)/2 is even, and (a+b)/2 is odd, t'' = 2-t
986  a = 0.;
987  b = 0.;
988  t = 2. - t;
989  } else if (equiv_0((a + b) / 2., 2)) {
990  // if (a-b)/2 is even, and (a+b)/2 is even, t'' = t
991  a = 0.;
992  b = 0.;
993  }
994  } else if (equiv_val((a - b) / 2., 1., 2)) {
995  if (equiv_val((a + b) / 2., 1., 2)) {
996  // if (a-b)/2 is odd, and (a+b)/2 is odd, t'' = 2-t
997  a = 0.;
998  b = 0.;
999  t = 2. + t;
1000  } else if (equiv_0((a + b) / 2., 2)) {
1001  // if (a-b)/2 is odd, and (a+b)/2 is even, t'' = -t
1002  a = 0.;
1003  b = 0.;
1004  t = -t;
1005  }
1006  }
1007  if (n == 1) {
1008  circ.add_blank_wires(2);
1009  if (!equiv_0(b - a, 8)) {
1010  circ.add_op<unsigned>(OpType::Rz, {(b - a) / 2.}, {1});
1011  }
1012  circ.add_op<unsigned>(OpType::CX, {0, 1});
1013  if (!equiv_0(a + b, 8)) {
1014  circ.add_op<unsigned>(OpType::Rz, {(-a - b) / 2.}, {1});
1015  }
1016  if (!equiv_0(t, 8)) {
1017  circ.add_op<unsigned>(OpType::Ry, {-t / 2.}, {1});
1018  }
1019  circ.add_op<unsigned>(OpType::CX, {0, 1});
1020  if (!equiv_0(t, 8)) {
1021  circ.add_op<unsigned>(OpType::Ry, {t / 2.}, {1});
1022  }
1023  if (!equiv_0(a, 4)) {
1024  circ.add_op<unsigned>(OpType::Rz, {a}, {1});
1025  }
1026  return circ;
1027  }
1028  // Using lemma 7.9 for n >= 2
1029  candidate_t ct;
1030  lemma79(circ, n + 1, a, t, b, ct);
1031  for (const std::pair<Edge, Vertex>& pairy : ct) {
1032  Vertex original_cnx = pairy.second;
1033  unsigned cnx_arity = circ.n_in_edges(original_cnx);
1034  switch (cnx_arity) {
1035  case 2: {
1036  circ.dag[original_cnx] = {get_op_ptr(OpType::CX)};
1037  break;
1038  }
1039  case 3: {
1040  circ.dag[original_cnx] = {get_op_ptr(OpType::CCX)};
1041  break;
1042  }
1043  default: {
1044  lemma73(circ, pairy);
1045  }
1046  }
1047  }
1048  return circ;
1049 }
