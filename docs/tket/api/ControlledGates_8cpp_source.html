45// `n` = size of incrementer, circuit returned is of size `n+1`
46/* this is slightly less efficient than perhaps it could be -- asymptotically it
47 is still good. In an ideal world, this would decompose the incrementers
48 smarter for the "even" case */
49Circuit incrementer_borrow_1_qubit(unsigned n) {
50 bool is_odd = n % 2;
51 Circuit circ(n + 1);
52 if (n < 6) {
53 if (n > 4) circ.append_qubits(C4X_normal_decomp(), {0, 1, 2, 3, 4});
54 if (n > 3) circ.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
55 if (n > 2) circ.add_op<unsigned>(OpType::CCX, {0, 1, 2});
56 if (n > 1) circ.add_op<unsigned>(OpType::CX, {0, 1});
57 if (n > 0) circ.add_op<unsigned>(OpType::X, {0});
58 return circ;
59 }
60 unsigned j;
61 unsigned k;
62 // j is bottom qubits, k is top qubits
63 // k + j = n + 1 (total no. of qbs)
64 if (is_odd) {
65 /* if the number of bits we are incrementing is odd, we can just split the
66 incrementer in 2 and use the `incrementer_borrow_n_qubits` method twice */
67 j = (n + 1) / 2, k = (n + 1) / 2;
68 } else {
69 /* otherwise, we will also have to pull out a cnx */
70 j = n / 2 + 1, k = n / 2;
71 }
72
73 Circuit top_incrementer = incrementer_borrow_n_qubits(k);
74 std::vector<unsigned> top_qbs(2 * k);
75 for (unsigned i = 0; i != k; ++i) {
76 top_qbs[2 * i] = i + k; // borrowed qubits
77 top_qbs[2 * i + 1] = i; // qbs we are trying to increment
78 }
79
80 Circuit cnx_top;
81 std::vector<unsigned> cnx1_qbs;
82 if (k == 3) { // code is unreachable if k<3
83 cnx_top = C3X_normal_decomp();
84 cnx1_qbs = {0, 1, 2, n};
85 } else if (k == 4) {
86 cnx_top = C4X_normal_decomp();
87 cnx1_qbs = {0, 1, 2, 3, n};
88 } else {
89 cnx_top = lemma72(k); // k controls on cnx
90 cnx1_qbs.resize(
91 2 * k - 2); // size of replacement using borrowed qbs = 2*k-1
92 std::iota(cnx1_qbs.begin(), cnx1_qbs.end(), 0);
93 cnx1_qbs.push_back(n); // target is last qubit
94 }
95
96 Circuit bottom_incrementer;
97 std::vector<unsigned> bot_qbs;
98 if (is_odd) {
99 bottom_incrementer = incrementer_borrow_n_qubits(j);
100 bot_qbs.resize(2 * j);
101 for (unsigned i = 0; i != j; ++i) {
102 bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
103 if (i != 0)
104 bot_qbs[2 * i + 1] =
105 i + j -
106 1; // 3,5...n //other qbs we are actually trying to increment
107 }
108 bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
109 } else {
110 if (j == 4) { // code is unreachable if j<4
111 bottom_incrementer.add_blank_wires(4);
112 bottom_incrementer.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
113 bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
114 bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
115 bottom_incrementer.add_op<unsigned>(OpType::X, {0});
116 bot_qbs = {n, n - 3, n - 2, n - 1};
117 } else if (j == 5) {
118 bottom_incrementer.add_blank_wires(5);
119 bottom_incrementer.append_qubits(C4X_normal_decomp(), {0, 1, 2, 3, 4});
120 bottom_incrementer.append_qubits(C3X_normal_decomp(), {0, 1, 2, 3});
121 bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
122 bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
123 bottom_incrementer.add_op<unsigned>(OpType::X, {0});
124 bot_qbs = {n, n - 4, n - 3, n - 2, n - 1};
125 } else {
126 // insert peeled-out cnx
127 Circuit cnx_bot = lemma72(j - 1);
128 std::vector<unsigned> cnx2_qbs(
129 2 * j - 3); // lemma 7.2 uses 2j-3 qubits for a (j-1)-controlled X
130 for (unsigned i = 0; i < j - 2; ++i) {
131 cnx2_qbs[i] = k + i;
132 }
133 cnx2_qbs[j - 2] = n; // replace the last control
134 for (unsigned i = 0; i < j - 3; ++i) {
135 cnx2_qbs[j + i - 1] = i;
136 }
137 cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
138
139 circ.append_qubits(cnx_bot, cnx2_qbs);
140 bottom_incrementer = incrementer_borrow_n_qubits(
141 j - 1); // insert incrementer over remaining qubits
142 bot_qbs.resize(2 * j - 2);
143 for (unsigned i = 0; i != j - 1; ++i) {
144 bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
145 if (i != 0)
146 bot_qbs[2 * i + 1] =
147 i + k -
148 1; // 3,5...n //other qbs we are actually trying to increment
149 }
150 bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
151 }
152 }
153
154 circ.append_qubits(bottom_incrementer, bot_qbs);
155 circ.add_op<unsigned>(
156 OpType::X,
157 {n}); // to convert controlled-incrementer to larger incrementer
158 for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
159 circ.append_qubits(cnx_top, cnx1_qbs);
160 if (!is_odd && j > 5) {
161 // insert peeled-out cnx
162 Circuit cnx_bot = lemma72(j - 1);
163 std::vector<unsigned> cnx2_qbs(2 * j - 3);
164 for (unsigned i = 0; i < j - 1; ++i) {
165 cnx2_qbs[i] = k + i;
166 }
167 cnx2_qbs[j - 2] = n; // the last control
168 for (unsigned i = 0; i < j - 3; ++i) {
169 cnx2_qbs[j + i - 1] = i;
170 }
171 cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
172 circ.append_qubits(cnx_bot, cnx2_qbs);
173 }
174 circ.append_qubits(bottom_incrementer, bot_qbs);
175 circ.add_op<unsigned>(OpType::X, {n});
176 circ.append_qubits(cnx_top, cnx1_qbs);
177 for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
178 circ.append_qubits(top_incrementer, top_qbs);
179 return circ;
180}

182// an optimised version of
183// https://algassert.com/circuits/2015/06/12/Constructing-Large-Increment-Gates.html
184/* every second qubit (0,2,4...) is a borrowed qubit */
185Circuit incrementer_borrow_n_qubits(unsigned n) {
186 const unsigned N = 2 * n;
187 Circuit circ(N);
188 /* deal with small cases where borrowing qubits is unnecessary */
189 if (n < 6) {
190 if (n > 4) circ.append_qubits(C4X_normal_decomp(), {1, 3, 5, 7, 9});
191 if (n > 3) circ.append_qubits(C3X_normal_decomp(), {1, 3, 5, 7});
192 if (n > 2) circ.add_op<unsigned>(OpType::CCX, {1, 3, 5});
193 if (n > 1) circ.add_op<unsigned>(OpType::CX, {1, 3});
194 if (n > 0) circ.add_op<unsigned>(OpType::X, {1});
195 return circ;
196 }
197
198 for (unsigned i = 1; i < N; ++i) {
199 if (i % 2)
200 circ.add_op<unsigned>(OpType::CX, {0, i});
201 else
202 circ.add_op<unsigned>(OpType::X, {i});
203 }
204
205 circ.add_op<unsigned>(OpType::X, {N - 1});
206
207 for (unsigned i = 2; i < N; ++(++i)) {
208 std::vector<unsigned> ladder_down_qbs = {i - 2, i - 1, i};
209 circ.append_qubits(ladder_down(), ladder_down_qbs);
210 }
211 circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
212 for (unsigned i = N - 2; i > 1; --(--i)) {
213 std::vector<unsigned> tof_up_qbs = {i - 2, i - 1, i};
214 circ.add_op<unsigned>(OpType::CCX, tof_up_qbs);
215 }
216
217 for (unsigned i = 2; i < N; ++(++i)) {
218 std::vector<unsigned> ladder_down_2_qbs = {i - 2, i - 1, i};
219 circ.append_qubits(ladder_down_2(), ladder_down_2_qbs);
220 }
221 circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
222 for (unsigned i = N - 2; i > 1; --(--i)) {
223 std::vector<unsigned> ladder_up_qbs = {i - 2, i - 1, i};
224 circ.append_qubits(ladder_up(), ladder_up_qbs);
225 }
226 for (unsigned i = 1; i < N; ++(++i))
227 circ.add_op<unsigned>(OpType::CX, {0, i});
228 return circ;
229}

231// decompose CnX gate using
232// https://algassert.com/circuits/2015/06/22/Using-Quantum-Gates-instead-of-Ancilla-Bits.html
233// `n` = no. of controls
234Circuit CnX_normal_decomp(unsigned n) {
235 /* handle low qb edge cases */
236 bool insert_c4xs; // dictate whether to add C4Xs or n>4 controlled Xs
237 // when bootstrapping
238 switch (n) {
239 case 0: {
240 return X();
241 }
242 case 1: {
243 return CX();
244 }
245 case 2: {
246 return CCX_normal_decomp();
247 }
248 case 3: {
249 return C3X_normal_decomp();
250 }
251 case 4: {
252 return C4X_normal_decomp();
253 }
254 case 5: {
255 insert_c4xs = true;
256 break;
257 }
258 default: {
259 insert_c4xs = false;
260 break;
261 }
262 }
263
264 Circuit circ(n + 1);
265 std::vector<unsigned> cnx_qbs(n - 1);
266 std::iota(cnx_qbs.begin(), cnx_qbs.end(), 0);
267 cnx_qbs.push_back(n);
268 // first, bootstrap an ancilla qubit
269 circ.add_op<unsigned>(OpType::H, {n});
270 Vertex cnx1;
271 if (insert_c4xs)
272 circ.append_qubits(C4X_normal_decomp(), {cnx_qbs});
273 else {
274 cnx1 = circ.add_op<unsigned>(
275 OpType::CnX,
276 {cnx_qbs}); /* the CnXs can be decomposed using lemma 7.3 */
277 }
278 circ.add_op<unsigned>(OpType::Tdg, {n});
279 Vertex cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
280 if (!insert_c4xs) {
281 Edge e1 = circ.get_nth_in_edge(cx, 0);
282 lemma73(circ, {e1, cnx1}); // replace first CnX using lemma 7.3
283 }
284 circ.add_op<unsigned>(OpType::T, {n});
285 Vertex cnx2;
286 if (insert_c4xs)
287 circ.append_qubits(C4X_normal_decomp(), {cnx_qbs});
288 else {
289 cnx2 = circ.add_op<unsigned>(OpType::CnX, {cnx_qbs});
290 }
291 circ.add_op<unsigned>(OpType::Tdg, {n});
292 cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
293 Edge e2 = circ.get_nth_in_edge(cx, 0);
294 if (!insert_c4xs) lemma73(circ, {e2, cnx2});
295 circ.add_op<unsigned>(OpType::T, {n});
296 circ.add_op<unsigned>(OpType::H, {n});
297
298 // add incremented shift pattern
299 Circuit incrementer = incrementer_borrow_1_qubit(n);
300 circ.append(incrementer);
301
302 // z rotation layer #1
303 std::vector<Op_ptr> z_rots(n);
304 double angle = -0.25;
305 for (unsigned i = 0; i < n - 1; ++i) {
306 unsigned m = n - 1 - i;
307 z_rots[i] = get_op_ptr(OpType::Rz, angle);
308 circ.add_op<unsigned>(z_rots[i], {m});
309 angle /= 2;
310 }
311
312 // decremented shift pattern
313 for (unsigned i = 0; i < n; ++i) {
314 circ.add_op<unsigned>(OpType::X, {i});
315 }
316 circ.append(incrementer);
317 for (unsigned i = 0; i < n; ++i) {
318 circ.add_op<unsigned>(OpType::X, {i});
319 }
320
321 // z rotation layer #2
322 for (unsigned i = 0; i < n - 1; ++i) {
323 unsigned m = n - 1 - i;
324 Expr ang = z_rots[i]->get_params()[0];
325 circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {m});
326 }
327 Expr ang = z_rots[n - 2]->get_params()[0];
328 circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {0});
329
330 const Op_ptr ccx = get_op_ptr(OpType::CCX);
331 circ.substitute_all(CCX_normal_decomp(), ccx);
332
333 circ.add_phase(std::pow(0.5, n + 1));
334 return circ;
335}

337// unsigned -> which column the change is in
338static unsigned find_first_differing_val(
339 const std::deque<bool>& d1, const std::deque<bool>& d2) {
340 unsigned N = d1.size();
341 if (N != d2.size())
342 throw ControlDecompError(
343 "Error in `find_first_differing_val`: Deques are of differing "
344 "sizes");
345 for (unsigned i = 0; i < N; ++i) {
346 if (d1[i] != d2[i]) return i;
347 }
348 throw ControlDecompError(
349 "Error in `find_first_differing_val`: No change between deques");
350}

352// optimal decomposition of CnRy and CnZ for 2 < n < 8 according to 1995
353// paper... can do better with ZH calculus?
354static Circuit lemma71(
355 unsigned arity, const Circuit& v_rep, const Circuit& v_dg_rep) {
356 unsigned m_controls = arity - 1;
357 if (m_controls < 2)
358 throw Unsupported(
359 "No point using Lemma 7.1 to decompose a gate with less than 2 "
360 "controls");
361
362 GrayCode gc = gen_graycode(m_controls);
363
364 Circuit rep(arity);
365
366 unsigned control_qb = 0;
367 unsigned last = 0;
368 rep.append_with_map(
369 v_rep, {{Qubit(0), Qubit(0)}, {Qubit(1), Qubit(m_controls)}});
370 // we ignore the 0...0 term, and the first one is always trivial
371 // so start from 2
372 for (unsigned i = 2; i < gc.size(); ++i) {
373 unsigned change = find_first_differing_val(gc[i], gc[i - 1]);
374 for (unsigned j = 1; j < gc[i].size(); ++j) {
375 if (gc[i][j] == 1) last = j;
376 }
377 if (change < control_qb) {
378 rep.add_op<unsigned>(OpType::CX, {change, control_qb});
379 }
380 if (change == control_qb) {
381 throw ControlDecompError("Error in graycode iteration");
382 }
383 if (change > control_qb) {
384 rep.add_op<unsigned>(OpType::CX, {control_qb, change});
385 }
386
387 if ((i % 2) == 0) {
388 rep.append_with_map(
389 v_dg_rep, {{Qubit(0), Qubit(last)}, {Qubit(1), Qubit(m_controls)}});
390 } else
391 rep.append_with_map(
392 v_rep, {{Qubit(0), Qubit(last)}, {Qubit(1), Qubit(m_controls)}});
393 control_qb = last;
394 }
395 auto [vit, vend] = boost::vertices(rep.dag);
396 VertexSet bin;
397 for (auto next = vit; vit != vend; vit = next) {
398 ++next;
399 Vertex v = *vit;
400 if (!bin.contains(v)) {
401 Op_ptr op = rep.get_Op_ptr_from_Vertex(v);
402 OpType optype = op->get_type();
403 if (is_multi_qubit_type(optype) && optype != OpType::CX) {
404 Circuit replacement = with_CX(as_gate_ptr(op));
405 rep.substitute(replacement, v, Circuit::VertexDeletion::No);
406 bin.insert(v);
407 }
408 }
409 }
410 rep.remove_vertices(
411 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
412 return rep;
413}

415//`control_m` = number of controls
416static Circuit lemma72(unsigned control_m) {
417 if (control_m < 3)
418 throw Unsupported(
419 "Cannot decompose a gate with " + std::to_string(control_m) +
420 " controls using Lemma 7.2");
421 const unsigned n = 2 * control_m - 1;
422
423 Circuit ccx_circ(n);
424 unsigned diff = n - control_m;
425 for (unsigned i = control_m - 1; i > 1; --i) {
426 ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
427 }
428 ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
429 for (unsigned i = 2; i < control_m; ++i) {
430 ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
431 }
432 for (unsigned i = control_m - 2; i > 1; --i) {
433 ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
434 }
435 ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
436 for (unsigned i = 2; i < control_m - 1; ++i) {
437 ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
438 }
439 if (ccx_circ.count_gates(OpType::CCX) != 4 * (control_m - 2))
440 throw ControlDecompError("Error in Lemma 7.2: CCX gate count is incorrect");
441 return ccx_circ;
442}

444/* this is specifically for performing corollary 7.4 via lemma 7.3 & lemma 7.2 -
445 * optimal decomp of a CnX gate */
446/* for corollary 7.4, n >= 7 */
447// this is a decomposition of a CnX gate using one dirty ancilla
448static void lemma73(Circuit& circ, const std::pair<Edge, Vertex>& pairy) {
449 Vertex original_cnx = pairy.second;
450 Edge original_spare_edge = pairy.first;
451 EdgeVec in_edges = circ.get_in_edges(original_cnx);
452 const unsigned N =
453 in_edges.size() + 1; // number of qubits in new circuit to replace
454 if (N < 5)
455 throw Unsupported(
456 "Lemma 7.3 cannot decompose CnX with n = " + std::to_string(N - 1));
457
458 std::vector<std::optional<Edge>> out_edges =
459 circ.get_linear_out_edges(original_cnx);
460
461 in_edges.insert(in_edges.begin() + in_edges.size() - 1, original_spare_edge);
462 out_edges.insert(
463 out_edges.begin() + out_edges.size() - 1, original_spare_edge);
464
465 Subcircuit to_delete{in_edges, out_edges, {}, {original_cnx}};
466 bool odd_N = N % 2;
467 unsigned m1 = (N + 1) / 2; // number of controls on the first type of CnX
468 unsigned m2 = N - m1 - 1; // number of controls on the second type of CnX
469
470 // make new circuit to substitute later
471 Circuit new_circ(N);
472 const Op_ptr cnx_op1 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m1 + 1);
473 const Op_ptr cnx_op2 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m2 + 1);
474 std::vector<unsigned> qbs_m1(m1 + 1);
475 std::iota(qbs_m1.begin(), qbs_m1.begin() + m1, 0);
476 qbs_m1[m1] = N - 1;
477 std::vector<unsigned> qbs_m2(m2 + 1);
478 std::iota(qbs_m2.begin(), qbs_m2.end(), N - (m2 + 1));
479
480 // add ladder of CnXs to the correct qubits
481 Vertex a = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
482 Vertex b = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
483 Vertex c = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
484 Vertex d = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
485
486 /* first, replace vertex a, putting circuit at back of new_circ */
487 unsigned cutsize = odd_N ? N : N - 1;
488 EdgeVec cut1(cutsize);
489 VertexVec out_verts = new_circ.q_outputs();
490 if (odd_N) {
491 for (unsigned i = 0; i < N - 2; ++i)
492 cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
493 cut1[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
494 cut1[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
495 } else {
496 for (unsigned i = 0; i < cutsize; ++i)
497 cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
498 }
499
500 Circuit a_replacement;
501 if (m1 == 1) {
502 a_replacement = CX();
503 } else if (m1 == 2) {
504 a_replacement = CCX();
505 } else
506 a_replacement = lemma72(m1); // returns circuit of size 2*m1 - 1
507 new_circ.cut_insert(a_replacement, cut1);
508 new_circ.remove_vertex(
509 a, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
510
511 VertexSet normal_decomp_vertices;
512
513 Circuit b_replacement;
514 if (m2 == 1) {
515 b_replacement = CX();
516 } else if (m2 == 2) {
517 b_replacement = CCX();
518 } else
519 b_replacement = lemma72(m2); // returns circuit of size 2*m2 - 1
520
521 unsigned b_qubits = b_replacement.n_qubits();
522
523 // reassign cut to back of circuit
524 EdgeVec frontier(N);
525 for (unsigned i = 0; i < N; ++i)
526 frontier[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
527
528 EdgeVec cut2(b_qubits);
529 for (unsigned i = N - m2 - 1; i < N - 1; ++i)
530 cut2[i - (N - m2 - 1)] =
531 frontier[i]; // N-1 - (N-m2-1) = m2 (all the controls)
532 for (unsigned i = 0; i < b_qubits - (m2 + 1); ++i)
533 cut2[i + m2] = frontier[i]; // empty wires
534 cut2[b_qubits - 1] = frontier[N - 1]; // target
535
536 new_circ.cut_insert(b_replacement, cut2);
537 new_circ.remove_vertex(
538 b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
539 Vertex last_path = out_verts[N - 1];
540 Edge init_back_edge = new_circ.get_nth_in_edge(last_path, 0);
541 Vertex init_back_path = new_circ.source(init_back_edge);
542 normal_decomp_vertices.insert(init_back_path);
543 init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
544 init_back_path = new_circ.source(init_back_edge);
545 OpType backstop = new_circ.get_OpType_from_Vertex(init_back_path);
546 while ((backstop != OpType::CCX) && !is_initial_q_type(backstop)) {
547 init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
548 init_back_path = new_circ.source(init_back_edge);
549 backstop = new_circ.get_OpType_from_Vertex(init_back_path);
550 }
551 normal_decomp_vertices.insert(init_back_path);
552 /* now, replace vertex c */
553 EdgeVec cut3(cutsize);
554 if (odd_N) {
555 for (unsigned i = 0; i < N - 2; ++i)
556 cut3[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
557 cut3[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
558 cut3[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
559 } else
560 for (unsigned i = 0; i < cutsize; ++i)
561 cut3[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
562
563 new_circ.cut_insert(a_replacement, cut3);
564 new_circ.remove_vertex(
565 c, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
566
567 /* now, replace vertex d */
568 for (unsigned i = 0; i < N; ++i)
569 frontier[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
570 EdgeVec cut4(b_qubits);
571 for (unsigned i = N - m2 - 1; i < N - 1; ++i)
572 cut4[i - (N - m2 - 1)] =
573 frontier[i]; // N-1 - (N-m2-1) = m2 (all the controls)
574 for (unsigned i = 0; i < b_qubits - (m2 + 1); ++i)
575 cut4[i + m2] = frontier[i]; // empty wires
576 cut4[b_qubits - 1] = frontier[N - 1]; // target
577
578 new_circ.cut_insert(b_replacement, cut4);
579 new_circ.remove_vertex(
580 d, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
581
582 last_path = out_verts[N - 1];
583 init_back_edge = new_circ.get_nth_in_edge(last_path, 0);
584 init_back_path = new_circ.source(init_back_edge);
585 normal_decomp_vertices.insert(init_back_path);
586 init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
587 init_back_path = new_circ.source(init_back_edge);
588 backstop = new_circ.get_OpType_from_Vertex(init_back_path);
589 while ((backstop != OpType::CCX) && !is_initial_q_type(backstop)) {
590 init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
591 init_back_path = new_circ.source(init_back_edge);
592 backstop = new_circ.get_OpType_from_Vertex(init_back_path);
593 }
594 normal_decomp_vertices.insert(init_back_path);
595
596 if (m1 > 2 && m2 > 2 && (new_circ.count_gates(OpType::CCX) != 8 * N - 40))
597 throw ControlDecompError("Error in Lemma 7.3: CCX gate count is incorrect");
598 /* now, replace CCXs with either CX circuit modulo phase shift or just typical
599 * Toffoli decomposition */
600
601 auto [vit, vend] = boost::vertices(new_circ.dag);
602 for (auto next = vit; vit != vend; vit = next) {
603 ++next;
604 if (new_circ.get_OpType_from_Vertex(*vit) == OpType::CCX) {
605 Subcircuit sub = new_circ.singleton_subcircuit(*vit);
606 if (normal_decomp_vertices.find(*vit) == normal_decomp_vertices.end()) {
607 new_circ.substitute(
608 CCX_modulo_phase_shift(), sub, Circuit::VertexDeletion::Yes);
609 } else {
610 new_circ.substitute(
611 CCX_normal_decomp(), sub, Circuit::VertexDeletion::Yes);
612 }
613 }
614 }
615 if (m1 > 2 && m2 > 2 && new_circ.count_gates(OpType::CX) != 24 * N - 108)
616 throw ControlDecompError("Error in Lemma 7.3: CX gate count is incorrect");
617
618 circ.substitute(new_circ, to_delete, Circuit::VertexDeletion::Yes);
619}

622// Assume the unitary is not identity
623// Linear decomposition for multi-controlled special unitaries
624static void lemma79(
625 Circuit& replacement, unsigned N, const Expr& alpha, const Expr& theta,
626 const Expr& beta, candidate_t& CCX_candidates) {
627 replacement.add_blank_wires(N);
628
629 // Add Controlled C
630 if (!equiv_0(beta - alpha, 8)) {
631 replacement.add_op<unsigned>(
632 OpType::CRz, {(beta - alpha) / 2.}, {N - 2, N - 1});
633 }
634 // Add the first CnX
635 std::vector<unsigned> cnx_qbs(N - 1);
636 std::iota(cnx_qbs.begin(), --cnx_qbs.end(), 0);
637 cnx_qbs[N - 2] = N - 1;
638 const Op_ptr cnx = get_op_ptr(OpType::CnX, std::vector<Expr>{}, N - 1);
639 Vertex firstCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
640
641 // Add Controlled B
642 VertexVec vBs;
643 if (!equiv_0(alpha + beta, 8)) {
644 Vertex vB1 = replacement.add_op<unsigned>(
645 OpType::CRz, {(-alpha - beta) / 2.}, {N - 2, N - 1});
646 vBs.push_back(vB1);
647 }
648 if (!equiv_0(theta, 8)) {
649 Vertex vB2 = replacement.add_op<unsigned>(
650 OpType::CRy, {-theta / 2.}, {N - 2, N - 1});
651 vBs.push_back(vB2);
652 }
653 // At least one of vB1 and vB2 should be set, otherwise it implies that the
654 // SU(2) is identity
655 if (vBs.empty()) {
656 throw ControlDecompError(
657 "Unknown error in Lemma 7.9: identity not rejected");
658 }
659
660 // Add the second CnX
661 Vertex secondCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
662
663 // Add Controlled A
664 if (!equiv_0(theta, 8)) {
665 replacement.add_op<unsigned>(OpType::CRy, {theta / 2.}, {N - 2, N - 1});
666 }
667 if (!equiv_0(alpha, 4)) {
668 replacement.add_op<unsigned>(OpType::CRz, {alpha}, {N - 2, N - 1});
669 }
670 Edge first_e = replacement.get_nth_in_edge(vBs[0], 0);
671 Edge second_e = replacement.get_nth_out_edge(vBs.back(), 0);
672 CCX_candidates.push_back({first_e, firstCnX});
673 CCX_candidates.push_back({second_e, secondCnX});
674}

676static Circuit CU_to_CU3(const Eigen::Matrix2cd& u) {
677 Circuit c(2);
678 std::vector<double> tk1_angles = tk1_angles_from_unitary(u);
679 Expr theta = tk1_angles[1];
680 Expr phi = tk1_angles[0] - 0.5;
681 Expr lambda = tk1_angles[2] + 0.5;
682 Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
683 c.add_op<unsigned>(OpType::U1, t, {0});
684 c.add_op<unsigned>(OpType::CU3, {theta, phi, lambda}, {0, 1});
685 c.remove_noops();
686 return c;
687}

689Circuit CnU_gray_code_decomp(unsigned n, const Eigen::Matrix2cd& u) {
690 if (n == 0) {
691 // Synthesise U using tk1 and phase
692 Circuit cnu_circ(1);
693 std::vector<double> tk1_angles = tk1_angles_from_unitary(u);
694 cnu_circ.add_op<unsigned>(
695 OpType::TK1, {tk1_angles[0], tk1_angles[1], tk1_angles[2]}, {0});
696 cnu_circ.add_phase(tk1_angles[3]);
697 return cnu_circ;
698 }
699 if (n == 1) {
700 return CU_to_CU3(u);
701 }
702
703 Eigen::Matrix2cd v_matrix = nth_root(u, 1ULL << (n - 1));
704 Eigen::Matrix2cd v_matrix_dag = v_matrix.adjoint();
705 Circuit v_rep = CU_to_CU3(v_matrix);
706 Circuit v_dg_rep = CU_to_CU3(v_matrix_dag);
707 return lemma71(n + 1, v_rep, v_dg_rep);
708}

710Circuit CnU_gray_code_decomp(unsigned n, const Gate_ptr& gate) {
711 static const std::map<OpType, OpType> cu_type_map = {
712 {OpType::Rx, OpType::CRx},
713 {OpType::Ry, OpType::CRy},
714 {OpType::Rz, OpType::CRz},
715 {OpType::U1, OpType::CU1}};
716
717 auto it = cu_type_map.find(gate->get_type());
718 if (it == cu_type_map.end()) {
719 throw Unsupported(
720 "The implementation currently only supports Rx, Ry, Rz, U1");
721 }
722
723 const OpType& cu_type = it->second;
724
725 if (n == 0) {
726 Circuit cnu_circ(1);
727 cnu_circ.add_op<unsigned>(gate, {0});
728 return cnu_circ;
729 }
730
731 Expr angle = gate->get_params()[0];
732 if (n == 1) {
733 Circuit cnu_circ(2);
734 cnu_circ.add_op<unsigned>(cu_type, angle, {0, 1});
735 return cnu_circ;
736 }
737 Expr param;
738 std::optional<double> reduced = eval_expr_mod(angle, 4);
739 if (reduced)
740 param = reduced.value();
741 else
742 param = angle;
743
744 param = param / (1ULL << (n - 1));
745 Circuit v_rep(2);
746 Circuit v_dg_rep(2);
747 v_rep.add_op<unsigned>(cu_type, param, {0, 1});
748 v_dg_rep.add_op<unsigned>(cu_type, -param, {0, 1});
749 return lemma71(n + 1, v_rep, v_dg_rep);
750}

964Circuit CnX_vchain_decomp(unsigned n, bool zeroed_ancillas) {
965 // Multi-controlled Toffoli decompositions that return clean ancilla qubits
966 switch (n) {
967 case 0: {
968 return X();
969 }
970 case 1: {
971 return CX();
972 }
973 case 2: {
974 return CCX_normal_decomp();
975 }
976 }
977
978 const unsigned n_ancillas = (n - 1) / 2;
979 const unsigned n_qubits = n + 1 + n_ancillas;
980 Circuit circ(n_qubits);
981
982 // Index of ii^th ancilla qubit
983 auto a = [&n, &n_qubits](int ii) {
984 if (ii >= 0) {
985 return n + 1 + ii;
986 }
987 // Allow negative indices
988 return n_qubits + ii;
989 };
990
991 using namespace Maslov2015;
992
993 if (zeroed_ancillas) {
994 // Decomposition from Proposition 4 of the paper:
995 // (1) a chain of C3X and/or CCX gates (up to relative phase) where each
996 // target is an ancilla qubit
997 // (2) a CCX gate targeting qubit n
998 // (3) the inverse of (1)
999
1000 // Build the chain (1)
1001 Circuit rtl_chain(n_qubits);
1002 if (n == 3) {
1003 // Edge case (circuit 4, page 10)
1004 rtl_chain.append_qubits(RTL(), {0, 1, a(0)});
1005 } else {
1006 rtl_chain.append_qubits(RT3L(), {0, 1, 2, a(0)});
1007 const unsigned num_rt3l = n_ancillas - 1 - (n % 2);
1008 for (unsigned ii = 0; ii < num_rt3l; ii++) {
1009 rtl_chain.append_qubits(
1010 RT3L(), {a(ii), 2 * ii + 3, 2 * ii + 4, a(ii + 1)});
1011 }
1012 if (n % 2 == 1) {
1013 // For odd n, we have one control qubit that was not used in the
1014 // preceding loop
1015 rtl_chain.append_qubits(RTL(), {a(-2), n - 2, a(-1)});
1016 }
1017 }
1018
1019 circ.append(rtl_chain);
1020 circ.append_qubits(CCX_normal_decomp(), {a(-1), n - 1, n});
1021 circ.append(rtl_chain.dagger());
1022 return circ;
1023 } // if (zeroed_ancillas)
1024
1025 // Decomposition for arbitrarily initialized ancillas
1026 if (n == 3) {
1027 // Edge case (circuit 5, page 10)
1028 Circuit rtl(n_qubits);
1029 rtl.append_qubits(RTL(), {0, 1, a(0)});
1030 Circuit srts(n_qubits);
1031 srts.append_qubits(SRTS(), {2, a(0), 3});
1032
1033 circ.append(rtl);
1034 circ.append(srts);
1035 circ.append(rtl.dagger());
1036 circ.append(srts.dagger());
1037 } else {
1038 // Decomposition from Proposition 5 of the paper:
1039 // (1) a relative-phase CCX gate targeting qubit n
1040 // (2) a chain of relative-phase CCX and C3X gates targeting all but one
1041 // ancilla
1042 // (3) a relative-phase C3X gate targeting the last ancilla
1043 // (4) inverse of (2)
1044 // (5) inverse of (1)
1045 // (6) copy of (2)
1046 // (7) inverse of (3)
1047 // (8) inverse of (2)
1048 Circuit srts(n_qubits);
1049 srts.append_qubits(SRTS(), {0, a(0), n});
1050
1051 Circuit rts_chain(n_qubits);
1052 const int nmod2 = n % 2;
1053 if (nmod2 == 1) {
1054 rts_chain.append_qubits(RTS(), {a(1), 1, a(0)});
1055 }
1056 for (unsigned ii = nmod2; ii < n_ancillas - 1; ii++) {
1057 const unsigned ctrl = 2 * ii + (1 - nmod2);
1058 rts_chain.append_qubits(RT3S(), {a(ii + 1), ctrl, ctrl + 1, a(ii)});
1059 }
1060
1061 Circuit rt3l(n_qubits);
1062 rt3l.append_qubits(RT3L(), {n - 3, n - 2, n - 1, a(-1)});
1063
1064 circ.append(srts);
1065 circ.append(rts_chain);
1066 circ.append(rt3l);
1067 circ.append(rts_chain.dagger());
1068 circ.append(srts.dagger());
1069 circ.append(rts_chain);
1070 circ.append(rt3l.dagger());
1071 circ.append(rts_chain.dagger());
1072 }
1073
1074 return circ;
1075}

1109// Add an incrementer without toggling the least significant bit
1110// Apply to qubits {0,...,n-1}, assume n > 1
1111static void add_qn(Circuit& circ, unsigned n) {
1112 TKET_ASSERT(n > 1);

1182Circuit CnSU2_linear_decomp(
1183 unsigned n, const Expr& alpha, const Expr& theta, const Expr& beta) {
1184 // W == I iff one of the following two conditions is met
1185 // 1. t/2 is even, and (a + b)/2 is even
1186 // 2. t/2 is odd, and (a + b)/2 is odd
1187 // check if SU(2) is identity
1188 if ((equiv_0(theta / 2., 2) && equiv_0((alpha + beta) / 2., 2)) ||
1189 (equiv_val(theta / 2., 1., 2) && equiv_val((alpha + beta) / 2., 1., 2))) {
1190 return Circuit(n + 1);
1191 }
1192
1193 Circuit circ;
1194
1195 if (n == 0) {
1196 // add tk1
1197 circ.add_blank_wires(1);
1198 circ.add_op<unsigned>(OpType::TK1, {alpha + 0.5, theta, beta - 0.5}, {0});
1199 return circ;
1200 }
1201
1202 // SU(2) matrix W expressed as Rz(a)Ry(t)Rz(b)
1203 Expr a = alpha;
1204 Expr b = beta;
1205 Expr t = theta;
1206
1207 // Lemma 4.3: W = A*X*B*X*C
1208 // By lemma 5.4, C is identity iff W can be expressed as Rz(a')Ry(t')Rz(a')
1209 // We handle the following two cases
1210 // if (a-b)/2 is even, a' = (a + b)/2, t' = t
1211 // if (a-b)/2 is odd, a' = (a + b)/2, t' = - t
1212 if (equiv_0((a - b) / 2., 2)) {
1213 a = (a + b) / 2.;
1214 b = a;
1215 } else if (equiv_val((a - b) / 2., 1., 2)) {
1216 a = (a + b) / 2.;
1217 b = a;
1218 t = -t;
1219 }
1220
1221 // We test whether W can be expressed as a single Ry(t'')
1222 if (equiv_0((a - b) / 2., 2)) {
1223 if (equiv_val((a + b) / 2., 1., 2)) {
1224 // if (a-b)/2 is even, and (a+b)/2 is odd, t'' = 2-t
1225 a = 0.;
1226 b = 0.;
1227 t = 2. - t;
1228 } else if (equiv_0((a + b) / 2., 2)) {
1229 // if (a-b)/2 is even, and (a+b)/2 is even, t'' = t
1230 a = 0.;
1231 b = 0.;
1232 }
1233 } else if (equiv_val((a - b) / 2., 1., 2)) {
1234 if (equiv_val((a + b) / 2., 1., 2)) {
1235 // if (a-b)/2 is odd, and (a+b)/2 is odd, t'' = 2-t
1236 a = 0.;
1237 b = 0.;
1238 t = 2. + t;
1239 } else if (equiv_0((a + b) / 2., 2)) {
1240 // if (a-b)/2 is odd, and (a+b)/2 is even, t'' = -t
1241 a = 0.;
1242 b = 0.;
1243 t = -t;
1244 }
1245 }
1246 if (n == 1) {
1247 circ.add_blank_wires(2);
1248 if (!equiv_0(b - a, 8)) {
1249 circ.add_op<unsigned>(OpType::Rz, {(b - a) / 2.}, {1});
1250 }
1251 circ.add_op<unsigned>(OpType::CX, {0, 1});
1252 if (!equiv_0(a + b, 8)) {
1253 circ.add_op<unsigned>(OpType::Rz, {(-a - b) / 2.}, {1});
1254 }
1255 if (!equiv_0(t, 8)) {
1256 circ.add_op<unsigned>(OpType::Ry, {-t / 2.}, {1});
1257 }
1258 circ.add_op<unsigned>(OpType::CX, {0, 1});
1259 if (!equiv_0(t, 8)) {
1260 circ.add_op<unsigned>(OpType::Ry, {t / 2.}, {1});
1261 }
1262 if (!equiv_0(a, 4)) {
1263 circ.add_op<unsigned>(OpType::Rz, {a}, {1});
1264 }
1265 return circ;
1266 }
1267 // Using lemma 7.9 for n >= 2
1268 candidate_t ct;
1269 lemma79(circ, n + 1, a, t, b, ct);
1270 for (const std::pair<Edge, Vertex>& pairy : ct) {
1271 Vertex original_cnx = pairy.second;
1272 unsigned cnx_arity = circ.n_in_edges(original_cnx);
1273 switch (cnx_arity) {
1274 case 2: {
1275 circ.dag[original_cnx] = {get_op_ptr(OpType::CX)};
1276 break;
1277 }
1278 case 3: {
1279 circ.dag[original_cnx] = {get_op_ptr(OpType::CCX)};
1280 break;
1281 }
1282 default: {
1283 lemma73(circ, pairy);
1284 }
1285 }
1286 }
1287 return circ;
1288}
