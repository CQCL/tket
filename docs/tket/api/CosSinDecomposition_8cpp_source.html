25csd_t CS_decomp(const Eigen::MatrixXcd &u) {
26 if (!is_unitary(u)) {
27 throw std::invalid_argument("Matrix for CS decomposition is not unitary");
28 }
29 unsigned N = u.rows();
30 if (N % 2 != 0) {
31 throw std::invalid_argument(
32 "Matrix for CS decomposition has odd dimensions");
33 }
34 unsigned n = N / 2;
35 Eigen::MatrixXcd u00 = u.topLeftCorner(n, n);
36 Eigen::MatrixXcd u01 = u.topRightCorner(n, n);
37 Eigen::MatrixXcd u10 = u.bottomLeftCorner(n, n);
38 Eigen::MatrixXcd u11 = u.bottomRightCorner(n, n);
39
40 Eigen::JacobiSVD<Eigen::MatrixXcd, Eigen::NoQRPreconditioner> svd(
41 u00, Eigen::ComputeFullU | Eigen::ComputeFullV);
42 Eigen::MatrixXcd l0 = svd.matrixU().rowwise().reverse();
43 Eigen::MatrixXcd r0_dag = svd.matrixV().rowwise().reverse();
44 Eigen::MatrixXd c = svd.singularValues().reverse().asDiagonal();
45 Eigen::MatrixXcd r0 = r0_dag.adjoint();
46
47 // Now u00 = l0 c r0; l0 and r0 are unitary, and c is diagonal with positive
48 // non-decreasing entries. Because u00 is a submatrix of a unitary matrix, its
49 // singular values (the entries of c) are all <= 1.
50
51 Eigen::HouseholderQR<Eigen::MatrixXcd> qr = (u10 * r0_dag).householderQr();
52 Eigen::MatrixXcd l1 = qr.householderQ();
53 Eigen::MatrixXcd S = qr.matrixQR().triangularView<Eigen::Upper>();
54
55 // Now u10 r0* = l1 S; l1 is unitary, and S is upper triangular.
56 //
57 // Claim: S is diagonal.
58 // Proof: Since u is unitary, we have
59 // I = u00* u00 + u10* u10
60 // = (l0 c r0)* (l0 c r0) + (l1 S r0)* (l1 S r0)
61 // = r0* c l0* l0 c r0 + r0* S* l1* l1 S r0
62 // = r0* c^2 r0 + r0* S* S r0
63 // = r0* (c^2 + S* S) r0
64 // So I = c^2 + (S* S), so (S* S) = I - c^2 is a diagonal matrix with non-
65 // increasing entries in the range [0,1). As S is upper triangular, this
66 // implies that S must be diagonal. (Proof by induction over the dimension n
67 // of S: consider the two cases S_00 = 0 and S_00 != 0 and reduce to the n-1
68 // case.)
69 //
70 // We want S to be real. This is not guaranteed, though since it is diagonal
71 // it can be made so by adjusting l1. In fact, it seems that Eigen always does
72 // give us a real S. We will not assume this, but will log an informational
73 // message and do the adjustment ourselves if that behaviour does change.
74
75 if (!S.imag().isZero()) {
76 tket_log()->info(
77 "Eigen surprisingly returned a non-real diagonal R in QR "
78 "decomposition; adjusting Q and R to make it real.");
79 for (unsigned j = 0; j < n; j++) {
80 std::complex<double> z = S(j, j);
81 double r = std::abs(z);
82 if (r > EPS) {
83 std::complex<double> w = std::conj(z) / r;
84 S(j, j) *= w;
85 l1.col(j) /= w;
86 }
87 }
88 }
89
90 // Now S is real and diagonal, and c^2 + S^2 = I.
91
92 Eigen::MatrixXd s = S.real();
93
94 // Make all entries in s non-negative.
95 for (unsigned j = 0; j < n; j++) {
96 if (s(j, j) < 0) {
97 s(j, j) = -s(j, j);
98 l1.col(j) = -l1.col(j);
99 }
100 }
101
102 // Finally compute r1, being careful not to divide by small things.
103 Eigen::MatrixXcd r1 = Eigen::MatrixXcd::Zero(n, n);
104 for (unsigned i = 0; i < n; i++) {
105 if (s(i, i) > c(i, i)) {
106 r1.row(i) = -(l0.adjoint() * u01).row(i) / s(i, i);
107 } else {
108 r1.row(i) = (l1.adjoint() * u11).row(i) / c(i, i);
109 }
110 }
111 return {l0, l1, r0, r1, c, s};
112}
