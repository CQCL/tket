82 // Adds a new cycle to bt.key_to_cycle
83 std::pair<unsigned, std::set<unsigned>> CycleFinder::make_cycle(
84  const Vertex& v, const EdgeVec& out_edges, const CutFrontier& cut) {
85  // Make a new boundary, add it to "all_boundaries", update "boundary_key" map
86  std::vector<edge_pair_t> new_cycle_boundary;
87  std::vector<UnitID> new_boundary_uids;
88  std::set<unsigned> old_boundary_keys;
89  std::set<unsigned> banned_keys;
90 
91  // Get UnitID, add to uids for new boundary, keep note of all old boundary
92  // keys for merging set new boundary key, add new edges to new boundary update
93  // cycle out edges if the edge can't be found
94  for (const Edge& e : out_edges) {
95  UnitID uid = unitid_from_unit_frontier(cut.u_frontier, e);
96  new_boundary_uids.push_back(uid);
97  old_boundary_keys.insert(bt.uid_to_key[uid]);
98 
99  // boundary key associated with given edge e not included
100  // i.e. e's associated in edge isn't a cycle out edge
101  // i.e. some other non-cycle gate has been passed
102  if (cycle_out_edges.find(circ.get_last_edge(v, e)) ==
103  cycle_out_edges.end()) {
104  banned_keys.insert(bt.uid_to_key[uid]);
105  }
106  bt.uid_to_key[uid] = bt.key;
107  new_cycle_boundary.push_back({circ.get_last_edge(v, e), e});
108  update_cycle_out_edges(uid, e);
109  }
110 
111  std::vector<unsigned> op_indices(out_edges.size());
112  std::iota(op_indices.begin(), op_indices.end(), 0);
113  // Edges in port ordering
114  Cycle new_cycle(
115  new_cycle_boundary, {{circ.get_OpType_from_Vertex(v), op_indices, v}});
116  bt.key_to_cycle[bt.key] = {new_cycle};
117 
118  bt.history.push_back(new_boundary_uids);
119  for (const unsigned& key : banned_keys) {
120  erase_keys(key, old_boundary_keys);
121  }
122  std::pair<unsigned, std::set<unsigned>> return_keys = {
123  bt.key, old_boundary_keys};
124  bt.key++;
125  return return_keys;
126 }
127 
128 // "old_keys" returned gives keys for boundaries to be merged together
129 // these "old_keys" are keys corresponding to boundaries UnitIDs in new boundary
130 // were in previously if two keys in passed old_keys have overlapping UnitIDs,
131 // then they cannot be merged in this case, remove the 'earlier' boundary from
132 // keys to be merged
133 void CycleFinder::order_keys(
134  unsigned& new_key, std::set<unsigned>& old_keys) const {
135  std::set<unsigned> bad_keys;
136  for (std::set<unsigned>::iterator it = old_keys.begin(); it != old_keys.end();
137  ++it) {
138  std::set<unsigned>::iterator jt = it;
139  std::advance(jt, 1);
140  // if either std::vector<UnitID> has common UnitIDs then one is causally
141  // blocked. remove smaller key from keys from set as not a candidate for
142  // merging into
143  for (; jt != old_keys.end(); ++jt)
144  for (const UnitID& u0 : bt.history[*it])
145  for (const UnitID& u1 : bt.history[*jt])
146  if (u0 == u1) {
147  bad_keys.insert(*it);
148  goto new_loop;
149  }
150 
151  new_loop : {}
152  }
153 
154  for (const unsigned& key : bad_keys) old_keys.erase(key);
155  old_keys.insert(new_key);
156 }

158 void Cycle::update_coms_indices(
159  const std::map<unsigned, unsigned>& new_indices) {
160  for (CycleCom& com : coms_) {
161  for (unsigned& index : com.indices) {
162  index = new_indices.at(index);
163  }
164  }
165 }
166 
167 // new cycle is made
168 // it may need to be immediately merged into other cycles
169 // 1) check for overlap of UnitID between cycles of "old_keys"
170 // if overlap, remove lowest value key from set
171 // 2) add "new_key" to "old_keys", merge all boundaries into corresponding
172 // boundary to highest value "old_key"
173 void Cycle::merge(Cycle& new_cycle) {
174  // iterate through edges in boundary of new_cycle
175  std::map<unsigned, unsigned> new_indices;
176  for (unsigned i = 0; i < new_cycle.boundary_edges_.size(); i++) {
177  bool match = false;
178  for (unsigned j = 0; j < boundary_edges_.size(); j++) {
179  // if in edge of boundary of new_cycle matches out of edge *this, update
180  // *this out edge
181  if (boundary_edges_[j].second == new_cycle.boundary_edges_[i].first) {
182  boundary_edges_[j].second = new_cycle.boundary_edges_[i].second;
183  // As CycleCom are labelled by basic indexing system, where indexing is
184  // related to position edge has in boundary Update CycleCom in new_cycle
185  // s.t. edges with index i now have index j update commands in new_cycle
186  // to match edgesindexing of *this for now, store new indexing
187  new_indices[i] = j;
188  match = true;
189  break;
190  }
191  }
192  if (!match) {
193  boundary_edges_.push_back(new_cycle.boundary_edges_[i]);
194  new_indices[i] = boundary_edges_.size() - 1;
195  }
196  }
197  // update CycleCom indices in new_cycle
198  new_cycle.update_coms_indices(new_indices);
199  coms_.insert(coms_.end(), new_cycle.coms_.begin(), new_cycle.coms_.end());
200 }

202 void CycleFinder::merge_cycles(unsigned new_key, std::set<unsigned>& old_keys) {
203  // boundary_keys is a std::set<unsigned> so is ordered
204  order_keys(new_key, old_keys);
205  // all cycles corresponding to keys in boundary_keys should be merged into
206  // boundary with smallest value
207  std::set<unsigned>::iterator it = old_keys.begin();
208  unsigned base_key = *it;
209  std::advance(it, 1);
210  for (; it != old_keys.end(); ++it) {
211  unsigned merge_key = *it;
212  bt.key_to_cycle[base_key].merge(bt.key_to_cycle[merge_key]);
213  bt.key_to_cycle.erase(merge_key);
214  // update bt.history for "base_key"
215  for (const UnitID& u0 : bt.history[merge_key]) {
216  bt.uid_to_key[u0] = base_key;
217  for (const UnitID& u1 : bt.history[base_key])
218  if (u0 == u1) break;
219  bt.history[base_key].push_back(u0);
220  }
221  }
222 }

224 void CycleFinder::extend_cycles(const CutFrontier& cut) {
225  // For each vertex in slice
226  // Make a new "cycle"
227  // If any in edge to new cycle matches an out edge to a previous cycle,
228  // attempt to merge cycles together
229  for (const Vertex& v : *cut.slice) {
230  EdgeVec in_edges = circ.get_in_edges_of_type(v, EdgeType::Quantum);
231  EdgeVec out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
232  // Compare EdgeType::Quantum "in_edges" of vertex in slice to collection of
233  // out edges from "active" boundaries If an "in_edge" is not equivalent to
234  // an out edge from "active" boundary, implies vertex needs to start a new
235  // boundary
236  std::pair<unsigned, std::set<unsigned>> all_keys =
237  make_cycle(v, out_edges, cut);
238  if (all_keys.second.size() > 0) {
239  merge_cycles(all_keys.first, all_keys.second);
240  }
241  }
242 }

244 std::vector<Cycle> CycleFinder::get_cycles() {
245  std::function<bool(Op_ptr)> skip_func = [&](Op_ptr op) {
246  return (cycle_types_.find(op->get_type()) == cycle_types_.end());
247  };
248 
249  Circuit::SliceIterator slice_iter(circ, skip_func);
250  bt.key = 0;
251 
252  // initialization
253  if (!(*slice_iter).empty()) {
254  for (const std::pair<UnitID, Edge>& pair :
255  slice_iter.cut_.u_frontier->get<TagKey>()) {
256  Edge in_edge = pair.second;
257  if (circ.get_edgetype(in_edge) != EdgeType::Classical) {
258  Vertex in_vert = circ.source(pair.second);
259 
260  // if vertex is type from types, then vertex is in slice and need
261  // in_edge. else can ignore.
262  if (cycle_types_.find(circ.get_OpType_from_Vertex(in_vert)) !=
263  cycle_types_.end()) {
264  in_edge = circ.get_last_edge(in_vert, pair.second);
265  }
266 
267  cycle_out_edges.insert({in_edge, pair.first});
268  bt.uid_to_key.insert({pair.first, bt.key});
269  Cycle new_cycle({{in_edge, in_edge}}, {{}});
270  bt.key_to_cycle[bt.key] = new_cycle;
271  bt.history.push_back({pair.first});
272  bt.key++;
273  }
274  }
275  // extend cycles automatically merges cycles that can be merge due to
276  // overlapping multi-qubit gates
277  extend_cycles(slice_iter.cut_);
278  cycle_out_edges = {};
279  for (const std::pair<UnitID, Edge>& pair :
280  slice_iter.cut_.u_frontier->get<TagKey>()) {
281  cycle_out_edges.insert({pair.second, pair.first});
282  }
283  }
284  while (!slice_iter.finished()) {
285  slice_iter.cut_ = circ.next_cut(
286  slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
287  if (!(*slice_iter).empty()) {
288  extend_cycles(slice_iter.cut_);
289  }
290  }
291 
292  // Skim Cycle type from CycleHistory.key_to_cycle
293  // Discard any Cycles with only Input Gates
294  std::vector<Cycle> output_cycles;
295  for (std::pair<unsigned, Cycle> entry : bt.key_to_cycle) {
296  if (entry.second.coms_.size() == 0) {
297  throw CycleError(std::string("Cycle with no internal gates."));
298  }
299  if (entry.second.boundary_edges_[0].first ==
300  entry.second.boundary_edges_[0].second) {
301  continue;
302  }
303  if (entry.second.size() > circ.n_qubits()) {
304  throw CycleError(
305  std::string("Cycle has a larger frame than Circuit has qubits."));
306  }
307  output_cycles.push_back(entry.second);
308  }
309  return output_cycles;
310 }
