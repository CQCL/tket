48void CycleFinder::update_cycle_out_edges(const UnitID& uid, const Edge& e) {
49 for (std::pair<const Edge, UnitID>& pair : cycle_out_edges) {
50 if (pair.second == uid) {
51 cycle_out_edges.erase(pair.first);
52 cycle_out_edges[e] = uid;
53 return;
54 }
55 }
56 throw CycleError(
57 std::string(
58 "UnitID " + uid.repr() + " not in std::map<Edge, UnitID> object."));

85// Adds a new cycle to this->cycle_history.key_to_cycle
86std::pair<unsigned, std::set<unsigned>> CycleFinder::make_cycle(
87 const Vertex& v, const EdgeVec& out_edges, const CutFrontier& cut) {
88 // Make a new boundary, add it to "all_boundaries", update "boundary_key" map
89 std::vector<edge_pair_t> new_cycle_boundary;
90 std::set<UnitID> new_boundary_uids;
91 std::set<unsigned> old_boundary_keys;
92 std::set<unsigned> not_mergeable_keys;
93 std::set<UnitID> not_mergeable_uids;
94
95 // Get UnitID, add to uids for new boundary, keep note of all old boundary
96 // keys for merging set new boundary key, add new edges to new boundary update
97 // cycle out edges if the edge can't be found
98
99 for (const Edge& e : out_edges) {
100 UnitID uid = unitid_from_unit_frontier(cut.u_frontier, e);
101 new_boundary_uids.insert(uid);
102 old_boundary_keys.insert(this->cycle_history.uid_to_key[uid]);
103 // boundary key associated with given edge e not included
104 // i.e. e's associated in edge isn't a cycle out edge
105 // i.e. some other non-cycle gate has been passed
106
107 // if the edge going into the vertex is not in a previous cycle
108 // then it can't be merged with that cycle
109 if (cycle_out_edges.find(circ.get_last_edge(v, e)) ==
110 cycle_out_edges.end()) {
111 not_mergeable_keys.insert(this->cycle_history.uid_to_key[uid]);
112 }
113 this->cycle_history.uid_to_key[uid] = this->cycle_history.key;
114 new_cycle_boundary.push_back({circ.get_last_edge(v, e), e});
115 update_cycle_out_edges(uid, e);
116 }
117
118 for (const unsigned& key : old_boundary_keys) {
119 for (const UnitID& uid : this->cycle_history.history[key]) {
120 if (not_mergeable_uids.find(uid) != not_mergeable_uids.end()) {
121 not_mergeable_keys.insert(key);
122 break;
123 }
124 }
125 }
126
127 std::vector<unsigned> op_indices(out_edges.size());
128 std::iota(op_indices.begin(), op_indices.end(), 0);
129 // Edges in port ordering
130 Cycle new_cycle(
131 new_cycle_boundary, {{circ.get_OpType_from_Vertex(v), op_indices, v}});
132 this->cycle_history.key_to_cycle[this->cycle_history.key] = {new_cycle};
133
134 this->cycle_history.history.push_back(
135 std::vector<UnitID>(new_boundary_uids.begin(), new_boundary_uids.end()));
136 for (const unsigned& key : not_mergeable_keys) {
137 erase_keys(key, old_boundary_keys);
138 }
139
140 not_mergeable_keys.clear();
141
142 // For each candidate cycle "cycle" given as an unsigned key
143 // For all UnitID "uid" in "v" that are not in "cycle"
144 // For all cycles between cycle_history.uid_to_key[uid] and "cycle"
145 // If a cycle contains both "uid" and any "uid" from "cycle"
146 // Set the key as not to be merged
147 for (const unsigned& candidate_cycle_key : old_boundary_keys) {
148 std::vector<UnitID> candidate_cycle_uid =
149 this->cycle_history.history[candidate_cycle_key];
150 std::set<UnitID> not_present_uids;
151 for (const UnitID& candidate_uids : new_boundary_uids) {
152 if (std::find(
153 candidate_cycle_uid.begin(), candidate_cycle_uid.end(),
154 candidate_uids) == candidate_cycle_uid.end()) {
155 not_present_uids.insert(candidate_uids);
156 }
157 }
158 // not_present_uids now contains UnitID in the new cycle that are not in the
159 // cycle represented by "candidate_cycle_key"
160
161 // We now iterate through all cycles between these UnitIDs' most recent
162 // cycles and our target cycle
163
164 // If any of these cycles have both a non-present uid and any uid in the
165 // target cycle then this key is not mergeable and would lead to a "cycle in
166 // the DAG" (different type of cycle)
167 for (const UnitID& not_present_uid : not_present_uids) {
168 for (unsigned i = candidate_cycle_key;
169 i < this->cycle_history.uid_to_key[not_present_uid]; i++) {
170 std::vector<UnitID> history_uids = this->cycle_history.history[i];
171 if (std::find(
172 history_uids.begin(), history_uids.end(), not_present_uid) !=
173 history_uids.end()) {
174 std::vector<UnitID> intersection;
175 std::vector<UnitID> candidates =
176 cycle_history.history[candidate_cycle_key];
177 std::set_intersection(
178 candidates.begin(), candidates.end(), history_uids.begin(),
179 history_uids.end(), std::back_inserter(intersection));
180 if (!intersection.empty()) {
181 not_mergeable_keys.insert(candidate_cycle_key);
182 break;
183 }
184 }
185 }
186 }
187 }
188
189 for (const unsigned& key : not_mergeable_keys) {
190 erase_keys(key, old_boundary_keys);
191 }
192
193 std::pair<unsigned, std::set<unsigned>> return_keys = {
194 this->cycle_history.key, old_boundary_keys};
195 this->cycle_history.key++;
196 return return_keys;
197}
198
199// "old_keys" returned gives keys for boundaries to be merged together
200// these "old_keys" are keys corresponding to boundaries UnitIDs in new boundary
201// were in previously if two keys in passed old_keys have overlapping UnitIDs,
202// then they cannot be merged in this case, remove the 'earlier' boundary from
203// keys to be merged
204void CycleFinder::order_keys(
205 unsigned& new_key, std::set<unsigned>& old_keys) const {
206 std::set<unsigned> bad_keys;
207 for (std::set<unsigned>::iterator it = old_keys.begin(); it != old_keys.end();
208 ++it) {
209 std::set<unsigned>::iterator jt = it;
210 std::advance(jt, 1);
211 // if either std::vector<UnitID> has common UnitIDs then one is causally
212 // blocked. remove smaller key from keys from set as not a candidate for
213 // merging into
214 for (; jt != old_keys.end(); ++jt)
215 for (const UnitID& u0 : this->cycle_history.history[*it])
216 for (const UnitID& u1 : this->cycle_history.history[*jt])
217 if (u0 == u1) {
218 bad_keys.insert(*it);
219 goto new_loop;
220 }
221
222 new_loop: {}
223 }
224
225 for (const unsigned& key : bad_keys) old_keys.erase(key);
226 old_keys.insert(new_key);
227}

229void Cycle::update_coms_indices(
230 const std::map<unsigned, unsigned>& new_indices) {
231 for (CycleCom& com : coms_) {
232 for (unsigned& index : com.indices) {
233 index = new_indices.at(index);
234 }
235 }
236}
237
238// new cycle is made
239// it may need to be immediately merged into other cycles
240// 1) check for overlap of UnitID between cycles of "old_keys"
241// if overlap, remove lowest value key from set
242// 2) add "new_key" to "old_keys", merge all boundaries into corresponding
243// boundary to highest value "old_key"
244void Cycle::merge(Cycle& new_cycle) {
245 // iterate through edges in boundary of new_cycle
246 std::map<unsigned, unsigned> new_indices;
247 for (unsigned i = 0; i < new_cycle.boundary_edges_.size(); i++) {
248 bool match = false;
249 for (unsigned j = 0; j < boundary_edges_.size(); j++) {
250 // if in edge of boundary of new_cycle matches out of edge *this, update
251 // *this out edge
252 if (boundary_edges_[j].second == new_cycle.boundary_edges_[i].first) {
253 boundary_edges_[j].second = new_cycle.boundary_edges_[i].second;
254 // As CycleCom are labelled by basic indexing system, where indexing is
255 // related to position edge has in boundary Update CycleCom in new_cycle
256 // s.t. edges with index i now have index j update commands in new_cycle
257 // to match edgesindexing of *this for now, store new indexing
258 new_indices[i] = j;
259 match = true;
260 break;
261 }
262 }
263 if (!match) {
264 boundary_edges_.push_back(new_cycle.boundary_edges_[i]);
265 new_indices[i] = boundary_edges_.size() - 1;
266 }
267 }
268 // update CycleCom indices in new_cycle
269 new_cycle.update_coms_indices(new_indices);
270 coms_.insert(coms_.end(), new_cycle.coms_.begin(), new_cycle.coms_.end());
271}

273void CycleFinder::merge_cycles(unsigned new_key, std::set<unsigned>& old_keys) {
274 // boundary_keys is a std::set<unsigned> so is ordered
275 order_keys(new_key, old_keys);
276 // all cycles corresponding to keys in boundary_keys should be merged into
277 // boundary with smallest value
278 std::set<unsigned>::iterator it = old_keys.begin();
279 unsigned base_key = *it;
280 std::advance(it, 1);
281 for (; it != old_keys.end(); ++it) {
282 unsigned merge_key = *it;
283 this->cycle_history.key_to_cycle[base_key].merge(
284 this->cycle_history.key_to_cycle[merge_key]);
285 this->cycle_history.key_to_cycle.erase(merge_key);
286 // update this->cycle_history.history for "base_key"
287 for (const UnitID& u0 : this->cycle_history.history[merge_key]) {
288 this->cycle_history.uid_to_key[u0] = base_key;
289 for (const UnitID& u1 : this->cycle_history.history[base_key])
290 if (u0 == u1) break;
291 this->cycle_history.history[base_key].push_back(u0);
292 }
293 }

296void CycleFinder::extend_cycles(const CutFrontier& cut) {
297 // For each vertex in slice
298 // Make a new "cycle"
299 // If any in edge to new cycle matches an out edge to a previous cycle,
300 // attempt to merge cycles together
301 for (const Vertex& v : *cut.slice) {
302 EdgeVec out_edges = circ.get_out_edges_of_type(v, EdgeType::Quantum);
303 // Compare EdgeType::Quantum "in_edges" of vertex in slice to collection of
304 // out edges from "active" boundaries If an "in_edge" is not equivalent to
305 // an out edge from "active" boundary, implies vertex needs to start a new
306 // boundary
307 std::pair<unsigned, std::set<unsigned>> all_keys =
308 make_cycle(v, out_edges, cut);
309 if (all_keys.second.size() > 0) {
310 merge_cycles(all_keys.first, all_keys.second);
311 }
312 }

315std::vector<Cycle> CycleFinder::get_cycles() {
316 std::function<bool(Op_ptr)> skip_func = [&](Op_ptr op) {
317 return (cycle_types_.find(op->get_type()) == cycle_types_.end());
318 };
319
320 SliceIterator slice_iter(circ, skip_func);
321 this->cycle_history.key = 0;
322
323 // initialization
324 if (!(*slice_iter).empty()) {
325 for (const std::pair<UnitID, Edge>& pair :
326 slice_iter.cut_.u_frontier->get<TagKey>()) {
327 Edge in_edge = pair.second;
328 if (circ.get_edgetype(in_edge) != EdgeType::Classical) {
329 Vertex in_vert = circ.source(pair.second);
330 // if vertex is type from types, then vertex is in slice and need
331 // in_edge. else can ignore.
332 if (cycle_types_.find(circ.get_OpType_from_Vertex(in_vert)) !=
333 cycle_types_.end()) {
334 in_edge = circ.get_last_edge(in_vert, pair.second);
335 }
336 cycle_out_edges.insert({in_edge, pair.first});
337 this->cycle_history.uid_to_key.insert(
338 {pair.first, this->cycle_history.key});
339 Cycle new_cycle({{in_edge, in_edge}}, {{}});
340 this->cycle_history.key_to_cycle[this->cycle_history.key] = new_cycle;
341 this->cycle_history.history.push_back({pair.first});
342 this->cycle_history.key++;
343 }
344 }
345 // extend cycles automatically merges cycles that can be merge due to
346 // overlapping multi-qubit gates
347 this->extend_cycles(slice_iter.cut_);
348 cycle_out_edges = {};
349 for (const std::pair<UnitID, Edge>& pair :
350 slice_iter.cut_.u_frontier->get<TagKey>()) {
351 cycle_out_edges.insert({pair.second, pair.first});
352 }
353 }
354 while (!slice_iter.finished()) {
355 slice_iter.cut_ = circ.next_cut(
356 slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
357 if (!(*slice_iter).empty()) {
358 this->extend_cycles(slice_iter.cut_);
359 }
360 }
361
362 // Skim Cycle type from CycleHistory.key_to_cycle
363 // Discard any Cycles with only Input Gates
364 std::vector<Cycle> output_cycles;
365 for (std::pair<unsigned, Cycle> entry : this->cycle_history.key_to_cycle) {
366 if (entry.second.coms_.size() == 0) {
367 throw CycleError(std::string("Cycle with no internal gates."));
368 }
369 if (entry.second.boundary_edges_[0].first ==
370 entry.second.boundary_edges_[0].second) {
371 continue;
372 }
373 if (entry.second.size() > circ.n_qubits()) {
374 throw CycleError(
375 std::string("Cycle has a larger frame than Circuit has qubits."));
376 }
377 output_cycles.push_back(entry.second);
378 }
379 return output_cycles;
380}
