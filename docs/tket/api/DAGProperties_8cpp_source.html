38bool is_valid(const DAG &G) {
39 BGL_FORALL_VERTICES(v, G, DAG) {
40 EdgeSet q_in, c_in, b_in, w_in;
41 BGL_FORALL_INEDGES(v, e, G, DAG) {
42 switch (G[e].type) {
43 case EdgeType::Quantum:
44 q_in.insert(e);
45 break;
46 case EdgeType::Classical:
47 c_in.insert(e);
48 break;
49 case EdgeType::Boolean:
50 b_in.insert(e);
51 break;
52 case EdgeType::WASM:
53 w_in.insert(e);
54 break;
55 default:
56 CHECK(!"found unknown edge type in is_valid check");
57 }
58 }
59 EdgeSet q_out, c_out, b_out, w_out;
60 BGL_FORALL_OUTEDGES(v, e, G, DAG) {
61 switch (G[e].type) {
62 case EdgeType::Quantum:
63 q_out.insert(e);
64 break;
65 case EdgeType::Classical:
66 c_out.insert(e);
67 break;
68 case EdgeType::Boolean:
69 b_out.insert(e);
70 break;
71 case EdgeType::WASM:
72 w_out.insert(e);
73 break;
74 default:
75 CHECK(!"found unknown edge type in is_valid check");
76 }
77 }
78 std::set<port_t> in_ports, q_in_ports, q_out_ports, c_in_ports, c_out_ports,
79 b_in_ports, w_in_ports, w_out_ports;
80 for (const auto &e : q_in) {
81 port_t p = G[e].ports.second;
82 in_ports.insert(p);
83 q_in_ports.insert(p);
84 }
85 for (const auto &e : q_out) {
86 q_out_ports.insert(G[e].ports.first);
87 }
88 for (const auto &e : c_in) {
89 port_t p = G[e].ports.second;
90 in_ports.insert(p);
91 c_in_ports.insert(p);
92 }
93 for (const auto &e : c_out) {
94 c_out_ports.insert(G[e].ports.first);
95 }
96 for (const auto &e : b_in) {
97 port_t p = G[e].ports.second;
98 in_ports.insert(p);
99 b_in_ports.insert(p);
100 }
101 for (const auto &e : w_in) {
102 port_t p = G[e].ports.second;
103 in_ports.insert(p);
104 w_in_ports.insert(p);
105 }
106 for (const auto &e : w_out) {
107 w_out_ports.insert(G[e].ports.first);
108 }
109
110 // Now check the required properties.
111
112 CHECK(
113 in_ports.size() == q_in_ports.size() + c_in_ports.size() +
114 b_in_ports.size() + w_in_ports.size());
115
116 // Every Boolean out port matches a Classical out port.
117 for (const Edge &e : b_out) {
118 port_t p = G[e].ports.first;
119 CHECK(std::any_of(c_out.cbegin(), c_out.cend(), [&](const Edge &f) {
120 return G[f].ports.first == p;
121 }));
122 }
123
124 // get size and empty only once
125
126 unsigned q_in_deg = q_in.size();
127 unsigned q_out_deg = q_out.size();
128
129 unsigned c_in_deg = c_in.size();
130 unsigned c_out_deg = c_out.size();
131
132 unsigned b_in_deg = b_in.size();
133
134 unsigned w_in_deg = w_in.size();
135 unsigned w_out_deg = w_out.size();
136
137 bool q_in_empty = q_in.empty();
138 bool q_out_empty = q_out.empty();
139
140 bool c_in_empty = c_in.empty();
141 bool c_out_empty = c_out.empty();
142
143 bool b_in_empty = b_in.empty();
144 bool b_out_empty = b_out.empty();
145
146 bool w_in_empty = w_in.empty();
147 bool w_out_empty = w_out.empty();
148
149 // check that ports maps to edges
150
151 CHECK(q_in_ports.size() == q_in_deg);
152 CHECK(q_out_ports.size() == q_out_deg);
153
154 CHECK(w_in_ports.size() == w_in_deg);
155 CHECK(w_out_ports.size() == w_out_deg);
156
157 CHECK(c_in_ports.size() == c_in_deg);
158 CHECK(c_out_ports.size() == c_out_deg);
159
160 CHECK(b_in_ports.size() == b_in_deg);
161
162 if (c_in_empty && c_out_empty && q_in_empty && !q_out_empty) {
163 // quantum in vertex
164 CHECK(b_in_empty);
165 CHECK(b_out_empty);
166 CHECK(w_in_empty);
167 CHECK(w_out_empty);
168 CHECK(q_in_deg == 0 && q_out_deg == 1);
169 } else if (c_in_empty && c_out_empty && !q_in_empty && q_out_empty) {
170 // quantum out vertex
171 CHECK(b_in_empty);
172 CHECK(b_out_empty);
173 CHECK(w_in_empty);
174 CHECK(w_out_empty);
175 CHECK(q_in_deg == 1 && q_out_deg == 0);
176 } else if (
177 c_in_empty && c_out_empty && !q_in_empty && !q_out_empty &&
178 !b_in_empty) {
179 // conditional quantum vertex, can have bool edges
180 CHECK(b_out_empty);
181 CHECK(w_in_empty);
182 CHECK(w_out_empty);
183 CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
184 } else if (
185 c_in_empty && c_out_empty && !q_in_empty && !q_out_empty &&
186 b_in_empty) {
187 // quantum vertex
188 CHECK(b_out_empty);
189 CHECK(w_in_empty);
190 CHECK(w_out_empty);
191 CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
192 } else if (w_in_empty && !w_out_empty) {
193 // wasm out vertex
194 CHECK(b_in_empty);
195 CHECK(b_out_empty);
196 CHECK(c_in_empty);
197 CHECK(c_out_empty);
198 CHECK(q_in_empty);
199 CHECK(q_out_empty);
200 CHECK(w_in_deg == 0 && w_out_deg == 1);
201 } else if (!w_in_empty && w_out_empty) {
202 // wasm in vertex
203 CHECK(b_in_empty);
204 CHECK(b_out_empty);
205 CHECK(c_in_empty);
206 CHECK(c_out_empty);
207 CHECK(q_in_empty);
208 CHECK(q_out_empty);
209 CHECK(w_in_deg == 1 && w_out_deg == 0);
210 } else if (!w_in_empty && !w_out_empty) {
211 // wasm vertex
212 CHECK(q_in_empty);
213 CHECK(q_out_empty);
214 CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
215 CHECK(w_in_ports == w_out_ports); // bijection between in and out ports
216 } else if (c_in_empty && !c_out_empty) {
217 // classical in vertex, can have bool out edges
218 CHECK(b_in_empty);
219 CHECK(w_in_empty);
220 CHECK(w_out_empty);
221 CHECK(q_in_empty);
222 CHECK(q_out_empty);
223 CHECK(c_in_deg == 0 && c_out_deg == 1);
224 } else if (!c_in_empty && c_out_empty) {
225 // classical out vertex, can't have bool out edges
226 CHECK(b_out_empty);
227 CHECK(w_in_empty);
228 CHECK(w_out_empty);
229 CHECK(q_in_empty);
230 CHECK(q_out_empty);
231 CHECK(c_in_deg == 1 && c_out_deg == 0);
232 } else if (q_in_empty && q_out_empty && !c_in_empty && !c_out_empty) {
233 // classical vertex, can have bool in and out edges
234 CHECK(w_in_empty);
235 CHECK(w_out_empty);
236 CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
237 } else if (!q_in_empty && !q_out_empty && !c_in_empty && !c_out_empty) {
238 // mixed vertex, can have bool in and out edges
239 CHECK(w_in_empty);
240 CHECK(w_out_empty);
241 CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
242 CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
243 } else if (q_in_empty && q_out_empty && c_in_empty && c_out_empty) {
244 // unconected vertex, can't have bool in and out edges
245 CHECK(w_in_empty);
246 CHECK(w_out_empty);
247 CHECK(b_in_empty);
248 CHECK(b_out_empty);
249 } else {
250 TKET_ASSERT(!"Found invalid circuit");
251 }
252 }
253 return true;
254}
