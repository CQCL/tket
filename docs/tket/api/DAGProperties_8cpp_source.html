38 bool is_valid(const DAG &G) {
39  BGL_FORALL_VERTICES(v, G, DAG) {
40  EdgeSet q_in, c_in, b_in;
41  BGL_FORALL_INEDGES(v, e, G, DAG) {
42  switch (G[e].type) {
43  case EdgeType::Quantum:
44  q_in.insert(e);
45  break;
46  case EdgeType::Classical:
47  c_in.insert(e);
48  break;
49  case EdgeType::Boolean:
50  b_in.insert(e);
51  break;
52  default:
53  CHECK(!"unknown edge type");
54  }
55  }
56  EdgeSet q_out, c_out, b_out;
57  BGL_FORALL_OUTEDGES(v, e, G, DAG) {
58  switch (G[e].type) {
59  case EdgeType::Quantum:
60  q_out.insert(e);
61  break;
62  case EdgeType::Classical:
63  c_out.insert(e);
64  break;
65  case EdgeType::Boolean:
66  b_out.insert(e);
67  break;
68  default:
69  CHECK(!"unknown edge type");
70  }
71  }
72  std::set<port_t> in_ports, q_in_ports, q_out_ports, c_in_ports, c_out_ports,
73  b_in_ports;
74  for (const auto &e : q_in) {
75  port_t p = G[e].ports.second;
76  in_ports.insert(p);
77  q_in_ports.insert(p);
78  }
79  for (const auto &e : q_out) {
80  q_out_ports.insert(G[e].ports.first);
81  }
82  for (const auto &e : c_in) {
83  port_t p = G[e].ports.second;
84  in_ports.insert(p);
85  c_in_ports.insert(p);
86  }
87  for (const auto &e : c_out) {
88  c_out_ports.insert(G[e].ports.first);
89  }
90  for (const auto &e : b_in) {
91  port_t p = G[e].ports.second;
92  in_ports.insert(p);
93  b_in_ports.insert(p);
94  }
95 
96  // Now check the required properties.
97 
98  CHECK(
99  in_ports.size() ==
100  q_in_ports.size() + c_in_ports.size() + b_in_ports.size());
101 
102  // Every Boolean out port matches a Classical out port.
103  for (const Edge &e : b_out) {
104  port_t p = G[e].ports.first;
105  CHECK(std::any_of(c_out.cbegin(), c_out.cend(), [&](const Edge &f) {
106  return G[f].ports.first == p;
107  }));
108  }
109 
110  if (c_in.empty() && c_out.empty()) {
111  // It is a Quantum vertex.
112  unsigned in_deg = q_in.size();
113  unsigned out_deg = q_out.size();
114  CHECK(q_in_ports.size() == in_deg);
115  CHECK(q_out_ports.size() == out_deg);
116  CHECK(
117  (in_deg == 0 && out_deg == 1) || (in_deg == 1 && out_deg == 0) ||
118  (q_in_ports == q_out_ports)); // bijection between in and out ports
119  CHECK(b_out.empty());
120  } else if (q_in.empty() && q_out.empty()) {
121  // It is a Classical vertex.
122  unsigned in_deg = c_in.size();
123  unsigned out_deg = c_out.size();
124  CHECK(c_in_ports.size() == in_deg);
125  CHECK(c_out_ports.size() == out_deg);
126  CHECK(
127  (in_deg == 0 && out_deg == 1) || (in_deg == 1 && out_deg == 0) ||
128  (c_in_ports == c_out_ports)); // bijection between in and out ports
129  } else {
130  // Check that it is a Measure vertex.
131  CHECK(
132  q_in.size() == 1 && q_out.size() == 1 && c_in.size() == 1 &&
133  c_out.size() == 1);
134  CHECK(q_in_ports == q_out_ports && c_in_ports == c_out_ports);
135  }
136  }
137  return true;
138 }
