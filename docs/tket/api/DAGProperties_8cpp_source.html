39 bool is_valid(const DAG &G) {
40  BGL_FORALL_VERTICES(v, G, DAG) {
41  EdgeSet q_in, c_in, b_in, w_in;
42  BGL_FORALL_INEDGES(v, e, G, DAG) {
43  switch (G[e].type) {
44  case EdgeType::Quantum:
45  q_in.insert(e);
46  break;
47  case EdgeType::Classical:
48  c_in.insert(e);
49  break;
50  case EdgeType::Boolean:
51  b_in.insert(e);
52  break;
53  case EdgeType::WASM:
54  w_in.insert(e);
55  break;
56  default:
57  CHECK(!"found unknown edge type in is_valid check");
58  }
59  }
60  EdgeSet q_out, c_out, b_out, w_out;
61  BGL_FORALL_OUTEDGES(v, e, G, DAG) {
62  switch (G[e].type) {
63  case EdgeType::Quantum:
64  q_out.insert(e);
65  break;
66  case EdgeType::Classical:
67  c_out.insert(e);
68  break;
69  case EdgeType::Boolean:
70  b_out.insert(e);
71  break;
72  case EdgeType::WASM:
73  w_out.insert(e);
74  break;
75  default:
76  CHECK(!"found unknown edge type in is_valid check");
77  }
78  }
79  std::set<port_t> in_ports, q_in_ports, q_out_ports, c_in_ports, c_out_ports,
80  b_in_ports, w_in_ports, w_out_ports;
81  for (const auto &e : q_in) {
82  port_t p = G[e].ports.second;
83  in_ports.insert(p);
84  q_in_ports.insert(p);
85  }
86  for (const auto &e : q_out) {
87  q_out_ports.insert(G[e].ports.first);
88  }
89  for (const auto &e : c_in) {
90  port_t p = G[e].ports.second;
91  in_ports.insert(p);
92  c_in_ports.insert(p);
93  }
94  for (const auto &e : c_out) {
95  c_out_ports.insert(G[e].ports.first);
96  }
97  for (const auto &e : b_in) {
98  port_t p = G[e].ports.second;
99  in_ports.insert(p);
100  b_in_ports.insert(p);
101  }
102  for (const auto &e : w_in) {
103  port_t p = G[e].ports.second;
104  in_ports.insert(p);
105  w_in_ports.insert(p);
106  }
107  for (const auto &e : w_out) {
108  w_out_ports.insert(G[e].ports.first);
109  }
110 
111  // Now check the required properties.
112 
113  CHECK(
114  in_ports.size() == q_in_ports.size() + c_in_ports.size() +
115  b_in_ports.size() + w_in_ports.size());
116 
117  // Every Boolean out port matches a Classical out port.
118  for (const Edge &e : b_out) {
119  port_t p = G[e].ports.first;
120  CHECK(std::any_of(c_out.cbegin(), c_out.cend(), [&](const Edge &f) {
121  return G[f].ports.first == p;
122  }));
123  }
124 
125  // get size and empty only once
126 
127  unsigned q_in_deg = q_in.size();
128  unsigned q_out_deg = q_out.size();
129 
130  unsigned c_in_deg = c_in.size();
131  unsigned c_out_deg = c_out.size();
132 
133  unsigned b_in_deg = b_in.size();
134 
135  unsigned w_in_deg = w_in.size();
136  unsigned w_out_deg = w_out.size();
137 
138  bool q_in_empty = q_in.empty();
139  bool q_out_empty = q_out.empty();
140 
141  bool c_in_empty = c_in.empty();
142  bool c_out_empty = c_out.empty();
143 
144  bool b_in_empty = b_in.empty();
145  bool b_out_empty = b_out.empty();
146 
147  bool w_in_empty = w_in.empty();
148  bool w_out_empty = w_out.empty();
149 
150  // check that ports maps to edges
151 
152  CHECK(q_in_ports.size() == q_in_deg);
153  CHECK(q_out_ports.size() == q_out_deg);
154 
155  CHECK(w_in_ports.size() == w_in_deg);
156  CHECK(w_out_ports.size() == w_out_deg);
157 
158  CHECK(c_in_ports.size() == c_in_deg);
159  CHECK(c_out_ports.size() == c_out_deg);
160 
161  CHECK(b_in_ports.size() == b_in_deg);
162 
163  if (c_in_empty && c_out_empty && q_in_empty && !q_out_empty) {
164  // quantum in vertex
165  CHECK(b_in_empty);
166  CHECK(b_out_empty);
167  CHECK(w_in_empty);
168  CHECK(w_out_empty);
169  CHECK(q_in_deg == 0 && q_out_deg == 1);
170  } else if (c_in_empty && c_out_empty && !q_in_empty && q_out_empty) {
171  // quantum out vertex
172  CHECK(b_in_empty);
173  CHECK(b_out_empty);
174  CHECK(w_in_empty);
175  CHECK(w_out_empty);
176  CHECK(q_in_deg == 1 && q_out_deg == 0);
177  } else if (
178  c_in_empty && c_out_empty && !q_in_empty && !q_out_empty &&
179  !b_in_empty) {
180  // conditional quantum vertex, can have bool edges
181  CHECK(b_out_empty);
182  CHECK(w_in_empty);
183  CHECK(w_out_empty);
184  CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
185  } else if (
186  c_in_empty && c_out_empty && !q_in_empty && !q_out_empty &&
187  b_in_empty) {
188  // quantum vertex
189  CHECK(b_out_empty);
190  CHECK(w_in_empty);
191  CHECK(w_out_empty);
192  CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
193  } else if (w_in_empty && !w_out_empty) {
194  // wasm out vertex
195  CHECK(b_in_empty);
196  CHECK(b_out_empty);
197  CHECK(c_in_empty);
198  CHECK(c_out_empty);
199  CHECK(q_in_empty);
200  CHECK(q_out_empty);
201  CHECK(w_in_deg == 0 && w_out_deg == 1);
202  } else if (!w_in_empty && w_out_empty) {
203  // wasm in vertex
204  CHECK(b_in_empty);
205  CHECK(b_out_empty);
206  CHECK(c_in_empty);
207  CHECK(c_out_empty);
208  CHECK(q_in_empty);
209  CHECK(q_out_empty);
210  CHECK(w_in_deg == 1 && w_out_deg == 0);
211  } else if (!w_in_empty && !w_out_empty) {
212  // wasm vertex
213  CHECK(q_in_empty);
214  CHECK(q_out_empty);
215  CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
216  CHECK(w_in_ports == w_out_ports); // bijection between in and out ports
217  } else if (c_in_empty && !c_out_empty) {
218  // classical in vertex, can have bool out edges
219  CHECK(b_in_empty);
220  CHECK(w_in_empty);
221  CHECK(w_out_empty);
222  CHECK(q_in_empty);
223  CHECK(q_out_empty);
224  CHECK(c_in_deg == 0 && c_out_deg == 1);
225  } else if (!c_in_empty && c_out_empty) {
226  // classical out vertex, can't have bool out edges
227  CHECK(b_out_empty);
228  CHECK(w_in_empty);
229  CHECK(w_out_empty);
230  CHECK(q_in_empty);
231  CHECK(q_out_empty);
232  CHECK(c_in_deg == 1 && c_out_deg == 0);
233  } else if (q_in_empty && q_out_empty && !c_in_empty && !c_out_empty) {
234  // classical vertex, can have bool in and out edges
235  CHECK(w_in_empty);
236  CHECK(w_out_empty);
237  CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
238  } else if (!q_in_empty && !q_out_empty && !c_in_empty && !c_out_empty) {
239  // mixed vertex, can have bool in and out edges
240  CHECK(w_in_empty);
241  CHECK(w_out_empty);
242  CHECK(c_in_ports == c_out_ports); // bijection between in and out ports
243  CHECK(q_in_ports == q_out_ports); // bijection between in and out ports
244  } else if (q_in_empty && q_out_empty && c_in_empty && c_out_empty) {
245  // unconected vertex, can't have bool in and out edges
246  CHECK(w_in_empty);
247  CHECK(w_out_empty);
248  CHECK(b_in_empty);
249  CHECK(b_out_empty);
250  } else {
251  TKET_ASSERT(!"Found invalid circuit");
252  }
253  }
254  return true;
255 }
