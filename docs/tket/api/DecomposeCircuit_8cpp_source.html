31typedef std::map<Qubit, unsigned> QMap;

33// No checks because only used internally
34static QMap get_qmap_no_checks(
35 const Circuit& circ,
36 const std::vector<unsigned>& parent_circuit_qubit_indices) {
37 std::map<Qubit, unsigned> qmap;
38 unsigned index = 0;
39 for (const Qubit& q : circ.all_qubits()) {
40 TKET_ASSERT(qmap.count(q) == 0);
41 qmap[q] = parent_circuit_qubit_indices[index];
42 ++index;
43 }
44 TKET_ASSERT(qmap.size() <= parent_circuit_qubit_indices.size());
45 return qmap;
46}

48// Already known to be a box, with a nonempty Op ptr.
49// If possible (if the op is able to calculate its own unitary matrix),
50// fill the node triplets with the raw unitary matrix represented by this box.
51// Return true if it found the triplets.
52static bool fill_triplets_directly_from_box(
53 GateNode& node, const std::shared_ptr<const Box>& box_ptr,
54 double abs_epsilon) {
55 std::optional<Eigen::MatrixXcd> u = box_ptr->get_box_unitary();
56 if (!u.has_value()) {
57 return false;
58 }
59 node.triplets = tket::get_triplets(*u, abs_epsilon);
60 return true;
61}

63static void add_global_phase(const Circuit& circ, GateNodesBuffer& buffer) {
64 const auto global_phase = eval_expr(circ.get_phase());
65 if (!global_phase) {
66 throw SymbolsNotSupported("Circuit has symbolic global phase");
67 }
68 const double phase_value = global_phase.value();
69 buffer.add_global_phase(phase_value);
70}

82static void fill_qubit_indices(
83 const unit_vector_t& args, const QMap& qmap, GateNode& node) {
84 TKET_ASSERT(args.size() <= qmap.size());
85 node.qubit_indices.resize(args.size());
86 for (unsigned ii = 0; ii < args.size(); ++ii) {
87 node.qubit_indices[ii] = qmap.at(Qubit(args[ii]));
88 }
89}

91static void decompose_circuit_recursive(
92 const Circuit& circ, GateNodesBuffer& buffer,
93 const std::vector<unsigned>& parent_circuit_qubit_indices,
94 double abs_epsilon) {
95 const auto qmap = get_qmap_no_checks(circ, parent_circuit_qubit_indices);
96 unit_vector_t args;
97 GateNode node;
98
99 for (const Command& command : circ) {
100 const Op_ptr current_op = command.get_op_ptr();
101 TKET_ASSERT(current_op.get());
102 const OpType current_type = current_op->get_type();
103 if (is_classical_type(current_type) || is_projective_type(current_type) ||
104 current_type == OpType::Conditional) {
105 throw GateUnitaryMatrixError(
106 "Unsupported OpType " + current_op->get_name(),
107 GateUnitaryMatrixError::Cause::GATE_NOT_IMPLEMENTED);
108 }
109 if (current_type == OpType::noop || current_type == OpType::Barrier) {
110 continue;
111 }
112 const OpDesc desc = current_op->get_desc();
113 args = command.get_args();
114 fill_qubit_indices(args, qmap, node);
115 if (desc.is_gate()) {
116 const Gate* gate = dynamic_cast<const Gate*>(current_op.get());
117 TKET_ASSERT(gate);
118 node.triplets =
119 GateUnitaryMatrix::get_unitary_triplets(*gate, abs_epsilon);
120 buffer.push(node);
121 continue;
122 }
123 if (!desc.is_box()) {
124 throw_with_op_error_message(
125 desc.name(), qmap, circ, "This is not a gate or box type.");
126 }
127 std::shared_ptr<const Box> box_ptr =
128 std::dynamic_pointer_cast<const Box>(current_op);
129 TKET_ASSERT(box_ptr.get());
130
131 node.triplets.clear();
132 const bool found_unitary =
133 fill_triplets_directly_from_box(node, box_ptr, abs_epsilon);
134
135 if (!node.triplets.empty()) {
136 TKET_ASSERT(found_unitary);
137 buffer.push(node);
138 continue;
139 }
140 TKET_ASSERT(!found_unitary);
141 // Break this box down, recursively.
142 std::shared_ptr<Circuit> box_circ = box_ptr->to_circuit();
143 if (!box_circ.get()) {
144 throw_with_op_error_message(
145 desc.name(), qmap, circ,
146 "This is a box, which couldn't be "
147 "broken down into a circuit");
148 }
149 decompose_circuit_recursive(
150 *box_circ, buffer, node.qubit_indices, abs_epsilon);
151 }
152 add_global_phase(circ, buffer);
153}

155void decompose_circuit(
156 const Circuit& circ, GateNodesBuffer& buffer, double abs_epsilon) {
157 // The qubits are just [0,1,2,...].
158 std::vector<unsigned> iota(circ.n_qubits());
159 std::iota(iota.begin(), iota.end(), 0);
160
161 decompose_circuit_recursive(circ, buffer, iota, abs_epsilon);
162 buffer.flush();
163}
