34 // No checks because only used internally
35 static QMap get_qmap_no_checks(
36  const Circuit& circ,
37  const std::vector<unsigned>& parent_circuit_qubit_indices) {
38  std::map<Qubit, unsigned> qmap;
39  unsigned index = 0;
40  for (const Qubit& q : circ.all_qubits()) {
41  TKET_ASSERT(qmap.count(q) == 0);
42  qmap[q] = parent_circuit_qubit_indices[index];
43  ++index;
44  }
45  TKET_ASSERT(qmap.size() <= parent_circuit_qubit_indices.size());
46  return qmap;
47 }

49 // Already known to be a box, with a nonempty Op ptr.
50 // If possible (if the op is able to calculate its own unitary matrix),
51 // fill the node triplets with the raw unitary matrix represented by this box.
52 // Return true if it found the triplets.
53 static bool fill_triplets_directly_from_box(
54  GateNode& node, const std::shared_ptr<const Box>& box_ptr, OpType type,
55  double abs_epsilon) {
56  switch (type) {
57  case OpType::Unitary1qBox: {
58  auto u1q_ptr = dynamic_cast<const Unitary1qBox*>(box_ptr.get());
59  TKET_ASSERT(u1q_ptr);
60  node.triplets = tket::get_triplets(u1q_ptr->get_matrix(), abs_epsilon);
61  return true;
62  }
63  case OpType::Unitary2qBox: {
64  auto u2q_ptr = dynamic_cast<const Unitary2qBox*>(box_ptr.get());
65  TKET_ASSERT(u2q_ptr);
66  node.triplets = tket::get_triplets(u2q_ptr->get_matrix(), abs_epsilon);
67  return true;
68  }
69  case OpType::Unitary3qBox: {
70  auto u3q_ptr = dynamic_cast<const Unitary3qBox*>(box_ptr.get());
71  TKET_ASSERT(u3q_ptr);
72  node.triplets = tket::get_triplets(u3q_ptr->get_matrix(), abs_epsilon);
73  return true;
74  }
75  case OpType::PauliExpBox: {
76  auto pauli_box_ptr = dynamic_cast<const PauliExpBox*>(box_ptr.get());
77  TKET_ASSERT(pauli_box_ptr);
78  node.triplets = get_triplets(*pauli_box_ptr);
79  return true;
80  }
81  case OpType::ExpBox: {
82  auto exp_box_ptr = dynamic_cast<const ExpBox*>(box_ptr.get());
83  TKET_ASSERT(exp_box_ptr);
84  const auto numerical_data = exp_box_ptr->get_matrix_and_phase();
85  Eigen::Matrix4cd matrix =
86  numerical_data.first * (i_ * numerical_data.second);
87  matrix = matrix.exp();
88  node.triplets = tket::get_triplets(matrix);
89  return true;
90  }
91  default:
92  return false;
93  }
94 }

115 static void fill_qubit_indices(
116  const unit_vector_t& args, const QMap& qmap, GateNode& node) {
117  TKET_ASSERT(args.size() <= qmap.size());

124 static void decompose_circuit_recursive(
125  const Circuit& circ, GateNodesBuffer& buffer,
126  const std::vector<unsigned>& parent_circuit_qubit_indices,
127  double abs_epsilon) {
128  const auto qmap = get_qmap_no_checks(circ, parent_circuit_qubit_indices);
129  unit_vector_t args;
130  GateNode node;
131 
132  for (const Command& command : circ) {
133  const Op_ptr current_op = command.get_op_ptr();
134  TKET_ASSERT(current_op.get());
135  const OpType current_type = current_op->get_type();
136  if (is_classical_type(current_type) || is_projective_type(current_type) ||
137  current_type == OpType::Conditional) {
138  throw GateUnitaryMatrixError(
139  "Unsupported OpType " + current_op->get_name(),
140  GateUnitaryMatrixError::Cause::GATE_NOT_IMPLEMENTED);
141  }
142  if (current_type == OpType::noop || current_type == OpType::Barrier) {
143  continue;
144  }
145  const OpDesc desc = current_op->get_desc();
146  args = command.get_args();
147  fill_qubit_indices(args, qmap, node);
148  if (desc.is_gate()) {
149  const Gate* gate = dynamic_cast<const Gate*>(current_op.get());
150  TKET_ASSERT(gate);
151  node.triplets =
152  GateUnitaryMatrix::get_unitary_triplets(*gate, abs_epsilon);
153  buffer.push(node);
154  continue;
155  }
156  if (!desc.is_box()) {
157  throw_with_op_error_message(
158  desc.name(), qmap, circ, "This is not a gate or box type.");
159  }
160  std::shared_ptr<const Box> box_ptr =
161  std::dynamic_pointer_cast<const Box>(current_op);
162  TKET_ASSERT(box_ptr.get());
163 
164  node.triplets.clear();
165  const bool found_unitary = fill_triplets_directly_from_box(
166  node, box_ptr, current_op->get_type(), abs_epsilon);
167 
168  if (!node.triplets.empty()) {
169  TKET_ASSERT(found_unitary);
170  buffer.push(node);
171  continue;
172  }
173  TKET_ASSERT(!found_unitary);
174  // Break this box down, recursively.
175  std::shared_ptr<Circuit> box_circ = box_ptr->to_circuit();
176  if (!box_circ.get()) {
177  throw_with_op_error_message(
178  desc.name(), qmap, circ,
179  "This is a box, which couldn't be "
180  "broken down into a circuit");
181  }
182  decompose_circuit_recursive(
183  *box_circ, buffer, node.qubit_indices, abs_epsilon);
184  }
185  add_global_phase(circ, buffer);
186 }

188 void decompose_circuit(
189  const Circuit& circ, GateNodesBuffer& buffer, double abs_epsilon) {
190  // The qubits are just [0,1,2,...].
191  std::vector<unsigned> iota(circ.n_qubits());
192  std::iota(iota.begin(), iota.end(), 0);
193 
194  decompose_circuit_recursive(circ, buffer, iota, abs_epsilon);
195  buffer.flush();
196 }
