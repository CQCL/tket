107 static bool convert_to_zyz(Circuit &circ) {
108  static const Expr half = SymEngine::div(Expr(1), Expr(2));
109  bool success = decompose_single_qubits_TK1().apply(circ);
110  VertexList bin;
111  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
112  if (circ.n_in_edges(v) != 1) continue;
113  const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
114  if (op->get_type() == OpType::TK1) {
115  std::vector<Expr> params = op->get_params();
116  Circuit replacement(1);
117  Expr a = params[2] + half;
118  Expr b = params[1];
119  Expr c = params[0] - half;
120  if (!equiv_0(a, 4)) {
121  replacement.add_op<unsigned>(OpType::Rz, a, {0});
122  }
123  if (!equiv_0(b, 4)) {
124  replacement.add_op<unsigned>(OpType::Ry, b, {0});
125  }
126  if (!equiv_0(c, 4)) {
127  replacement.add_op<unsigned>(OpType::Rz, c, {0});
128  }
129  Subcircuit sub = {
130  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
131  bin.push_back(v);
132  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
133  success = true;
134  }
135  }
136  circ.remove_vertices(
137  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
138  return success;

202 Transform decompose_ZYZ_to_TK1() {
203  return Transform([](Circuit &circ) {
204  bool success = false;
205  static const Expr zero(0);
206  static const Expr half = SymEngine::div(Expr(1), Expr(2));
207  VertexList bin;
208  VertexVec inputs = circ.q_inputs();
209  for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
210  Edge e = circ.get_nth_out_edge(*i, 0);
211  Vertex v = circ.target(e);
212  while (!is_final_q_type(circ.get_OpType_from_Vertex(v))) {
213  if (circ.get_OpType_from_Vertex(v) == OpType::Rz) {
214  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
215  Expr angle_1 = v_g->get_params()[0];
216  Edge e1 = circ.get_next_edge(v, e);
217  Vertex v2 = circ.target(e1);
218  if (circ.get_OpType_from_Vertex(v2) == OpType::Ry) {
219  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
220  Expr angle_2 = v2_g->get_params()[0];
221  Edge e2 = circ.get_next_edge(v2, e1);
222  Vertex v3 = circ.target(e2);
223  bin.push_back(v2);
224  circ.remove_vertex(
225  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
226  Expr angle_3 = zero;
227  if (circ.get_OpType_from_Vertex(v3) == OpType::Rz) {
228  const Op_ptr v3_g = circ.get_Op_ptr_from_Vertex(v3);
229  angle_3 = v3_g->get_params()[0];
230  circ.remove_vertex(
231  v3, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
232  bin.push_back(v3);
233  }
234  std::vector<Expr> new_params = {
235  angle_3 + half, angle_2, angle_1 - half};
236  circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
237  } else {
238  circ.dag[v] = {get_op_ptr(OpType::TK1, {zero, zero, angle_1})};
239  }
240  } else if (circ.get_OpType_from_Vertex(v) == OpType::Ry) {
241  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
242  Expr angle_2 = v_g->get_params()[0];
243  Expr angle_3 = zero;
244  Edge e1 = circ.get_next_edge(v, e);
245  Vertex v2 = circ.target(e1);
246  if (circ.get_OpType_from_Vertex(v2) == OpType::Rz) {
247  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
248  angle_3 = v2_g->get_params()[0];
249  circ.remove_vertex(
250  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
251  bin.push_back(v2);
252  }
253  std::vector<Expr> new_params = {angle_3 + half, angle_2, -half};
254  circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
255  }
256  e = circ.get_next_edge(v, e);
257  v = circ.target(e);
258  }
259  }
260  circ.remove_vertices(
261  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
262  return success;
263  });
264 }

266 static Circuit tk1_angles_to_circ(Expr a, Expr b, Expr c) {
267  Circuit circ(1);
268 
269  std::optional<double> ea = eval_expr(a);
270  std::optional<double> eb = eval_expr(b);
271  std::optional<double> ec = eval_expr(c);
272 
273  // Remove global phases
274  if (ea && *ea >= 2) {
275  a -= 2;
276  circ.add_phase(1);
277  }
278  if (eb && *eb >= 2) {
279  b -= 2;
280  circ.add_phase(1);
281  }
282  if (ec && *ec >= 2) {
283  c -= 2;
284  circ.add_phase(1);
285  }
286 
287  if (!equiv_0(a, 2) || !equiv_0(b, 2) || !equiv_0(c, 2)) {
288  circ.add_op<unsigned>(OpType::TK1, {c, b, a}, {0});
289  }
290 
291  return circ;
292 }

294 Transform decompose_ZXZ_to_TK1() {
295  return Transform([](Circuit &circ) {
296  bool success = false;
297  VertexList bin;
298  VertexVec inputs = circ.q_inputs();
299  for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
300  Edge e = circ.get_nth_out_edge(*i, 0);
301  Vertex v = circ.target(e);
302  // Angles for current TK1
303  std::array<Expr, 3> curr_angles;
304  // Index from 0 <= curr_ind <= 2 into `curr_angles` collecting TK1 angles
305  unsigned curr_ind = 0;
306  // The beginning edge of a sequence of Rx and Rz gates
307  std::optional<Edge> first_edge;
308  // Vertices of the Rx/Rz sequence. Empty iff first_edge == std::nullopt
309  VertexSet curr_vs;
310  while (true) {
311  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
312  bool repeat_loop = false;
313  bool substitute_vertex = false;
314  TKET_ASSERT(curr_ind < 3);
315  switch (op->get_type()) {
316  case OpType::Rz: {
317  if (curr_ind == 1) {
318  curr_angles[curr_ind++] = 0;
319  }
320  TKET_ASSERT(op->get_params().size() == 1);
321  curr_angles[curr_ind++] = op->get_params().at(0);
322  substitute_vertex = true;
323  break;
324  }
325  case OpType::Rx: {
326  if (curr_ind != 1) {
327  curr_angles[curr_ind++] = 0;
328  }
329  TKET_ASSERT(op->get_params().size() == 1);
330  if (curr_ind < 3) {
331  curr_angles[curr_ind++] = op->get_params().at(0);
332  substitute_vertex = true;
333  } else {
334  repeat_loop = true;
335  }
336  break;
337  }
338  default: {
339  while (curr_ind < 3) {
340  curr_angles[curr_ind++] = 0;
341  }
342  }
343  }
344  if (substitute_vertex) {
345  curr_vs.insert(v);
346  if (!first_edge) {
347  first_edge = e;
348  }
349  }
350  // Substitute sequence of RzRxRz with TK1
351  auto substitute = [&]() {
352  if (first_edge) {
353  success = true;
354  Circuit sub = tk1_angles_to_circ(
355  curr_angles[0], curr_angles[1], curr_angles[2]);
356  Subcircuit hole{{*first_edge}, {e}, curr_vs};
357  // Backup
358  port_t backup_p = circ.get_target_port(e);
359  // Substitute
360  circ.substitute(sub, hole, Circuit::VertexDeletion::No);
361  // Restore
362  e = circ.get_nth_in_edge(v, backup_p);
363  // Reset all tracking variables
364  bin.insert(bin.end(), curr_vs.begin(), curr_vs.end());
365  std::fill(curr_angles.begin(), curr_angles.end(), 0);
366  curr_vs.clear();
367  first_edge.reset();
368  }
369  curr_ind = 0;
370  };
371  // Depending on `substitute_vertex`, we either place the TK1 before
372  // moving the edge forward or after
373  if (curr_ind == 3 && !substitute_vertex) {
374  substitute();
375  }
376  if (is_final_q_type(op->get_type())) {
377  TKET_ASSERT(!substitute_vertex);
378  break;
379  }
380  if (!repeat_loop) {
381  // Move edge forward
382  e = circ.get_next_edge(v, e);
383  v = circ.target(e);
384  }
385  if (curr_ind == 3 && substitute_vertex) {
386  substitute();
387  }
388  }
389  }
390  circ.remove_vertices(
391  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
392  return success;
393  });
394 }

460 Transform decompose_ZX_to_HQS1() {
461  return Transform([](Circuit &circ) {
462  bool success = false;
463  VertexList to_bin;
464  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
465  if (circ.get_OpType_from_Vertex(v) == OpType::Rx) {
466  success = true;
467  const Op_ptr g = circ.get_Op_ptr_from_Vertex(v);
468  Expr theta = g->get_params()[0];
469  Vertex prev_vert = *circ.get_predecessors(v).begin();
470  Vertex next_vert = *circ.get_successors(v).begin();
471  if (circ.get_OpType_from_Vertex(prev_vert) == OpType::Rz &&
472  circ.get_OpType_from_Vertex(next_vert) == OpType::Rz) {
473  const Op_ptr prev_g = circ.get_Op_ptr_from_Vertex(prev_vert);
474  const Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next_vert);
475  Expr phi = next_g->get_params()[0];
476  std::vector<Expr> params{theta, phi};
477  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
478  circ.remove_vertex(
479  next_vert, Circuit::GraphRewiring::Yes,
480  Circuit::VertexDeletion::No);
481  to_bin.push_back(next_vert);
482  Expr new_param = prev_g->get_params()[0] + phi;
483  circ.dag[prev_vert].op = get_op_ptr(OpType::Rz, new_param);
484  } else {
485  // if no Rz, initialise a PhasedX op with theta=Rx.params[0],phi=0
486  Expr phi(0);
487  std::vector<Expr> params{theta, phi};
488  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
489  }
490  }
491  }
492  circ.remove_vertices(
493  to_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
494  remove_redundancies().apply(circ);
495  return success;

499 // Decompose CX into MolmerSorensen as:
500 // ---C--- -V-S-|-H-
501 // | --> XX(pi/4)
502 // ---X--- -----|-Vdg-
503 Transform decompose_MolmerSorensen() {
504  return Transform([](Circuit &circ) {
505  bool success = false;
506  VertexList bin;
507  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
508  if (circ.get_OpType_from_Vertex(v) != OpType::CX) continue;
509  EdgeVec outs = circ.get_all_out_edges(v);
510  if (outs.size() == 2) {
511  Vertex next = circ.target(outs[0]);
512  // Is the next operation equivalent to an Rx, up to phase?
513  Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next);
514  OpType next_type = next_g->get_type();
515  if (is_single_qubit_type(next_type) && !is_projective_type(next_type)) {
516  std::vector<Expr> angles = as_gate_ptr(next_g)->get_tk1_angles();
517  if (equiv_0(angles[0], 2) && equiv_0(angles[2], 2)) {
518  Expr angle = angles[1];
519  Expr phase = angles[3];
520  if (!equiv_0(angles[0], 4)) phase += 1;
521  if (!equiv_0(angles[2], 4)) phase += 1;
522  Edge next_e = circ.get_nth_out_edge(next, 0);
523  Vertex last = circ.target(next_e);
524  if (circ.get_OpType_from_Vertex(last) == OpType::CX &&
525  circ.get_nth_in_edge(last, 1) == outs[1]) {
526  // Recognise exp(-i XX * angle * pi/2)
527  const Op_ptr op_ptr = get_op_ptr(OpType::XXPhase, angle);
528  circ.dag[v] = {op_ptr};
529  bin.push_back(next);
530  circ.remove_vertex(
531  next, Circuit::GraphRewiring::Yes,
532  Circuit::VertexDeletion::No);
533  bin.push_back(last);
534  circ.remove_vertex(
535  last, Circuit::GraphRewiring::Yes,
536  Circuit::VertexDeletion::No);
537  circ.add_phase(phase);
538  success = true;
539  continue;
540  }
541  }
542  }
543  // Replace remaining CX gates
544  Subcircuit sub = {{circ.get_in_edges(v)}, {outs}, {v}};
545  bin.push_back(v);
546  circ.substitute(
547  CircPool::CX_using_XXPhase_1(), sub, Circuit::VertexDeletion::No);
548  success = true;
549  }
550  }
551  circ.remove_vertices(
552  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
553  return success;
554  });
555 }

578 static double get_CX_fidelity(const std::array<double, 3> &k, unsigned nb_cx) {
579  TKET_ASSERT(nb_cx < 4);
580  auto [a, b, c] = k;
581 
582  // gate fidelity achievable with 0,...,3 cnots
583  // this is fully determined by the information content k and is optimal
584  // see PhysRevA 71.062331 (2005) for more details on this
585  switch (nb_cx) {
586  case 0:
587  return trace_fidelity(a, b, c);
588  case 1:
589  return trace_fidelity(0.5 - a, b, c);
590  case 2:
591  return trace_fidelity(0, 0, c);

597 // Try to decompose a TK2 gate using different gate sets, find the one with
598 // the highest fidelity.
599 // If no fidelities are provided, (best_optype, n_gates) is left unchanged.
600 static double best_noise_aware_decomposition(
601  const std::array<double, 3> &angles, const TwoQbFidelities &fid,
602  OpType &best_optype, unsigned &n_gates) {
603  double max_fid = 0.;
604 
605  // Try decomposition using CX or equivalent gates.
606  double cx_fid = std::max(
607  fid.CX_fidelity ? fid.CX_fidelity.value() : 0.,
608  fid.ZZMax_fidelity ? fid.ZZMax_fidelity.value() : 0.);
609  bool zzmax_is_better = false;
610  if (cx_fid < EPS) {
611  if (!fid.ZZPhase_fidelity) {
612  // No fidelity is defined, so default to CX
613  cx_fid = 1.;
614  }
615  } else {
616  zzmax_is_better = fid.CX_fidelity < fid.ZZMax_fidelity;
617  }
618  if (cx_fid > EPS) {
619  for (unsigned n_cx = 0; n_cx <= 3; ++n_cx) {
620  double ncx_fid = get_CX_fidelity(angles, n_cx) * pow(cx_fid, n_cx);
621  if (ncx_fid > max_fid) {
622  max_fid = ncx_fid;
623  best_optype = zzmax_is_better ? OpType::ZZMax : OpType::CX;
624  n_gates = n_cx;
625  }
626  }
627  }
628 
629  // Try decomposition using ZZPhase(α)
630  if (fid.ZZPhase_fidelity) {
631  double zz_fid = 1.;
632  // If ZZMax is available, ZZPhase is only interesting when used once.
633  // (two ZZPhase can always be written using two ZZmax)
634  unsigned max_nzz = fid.ZZMax_fidelity ? 1 : 3;
635  for (unsigned n_zz = 0; n_zz <= max_nzz; ++n_zz) {
636  if (n_zz > 0) {
637  double gate_fid = (*fid.ZZPhase_fidelity)(angles[n_zz - 1]);
638  if (gate_fid < 0 || gate_fid > 1) {
639  throw std::domain_error(
640  "ZZPhase_fidelity returned a value outside of [0, 1].");
641  }
642  zz_fid *= gate_fid;
643  }
644  double nzz_fid = get_ZZPhase_fidelity(angles, n_zz) * zz_fid;
645  // Use ZZPhase if fidelity is greater or it is equal but uses fewer gates
646  if (nzz_fid - max_fid > EPS ||
647  (nzz_fid - max_fid > -EPS && n_zz < n_gates)) {
648  max_fid = nzz_fid;
649  best_optype = OpType::ZZPhase;
650  n_gates = n_zz;
651  }
652  }
653  }
654 
655  return max_fid;
656 }

658 // Try to decompose a TK2 gate exactly using different gate sets.
659 // The fidelities are used as an indication of which gate set is usable, but
660 // the actual values of the fidelities will be ignored.
661 //
662 // Relies on default values of best_optype and n_gates if no optimisation can be
663 // performed.
664 static void best_exact_decomposition(
665  const std::array<Expr, 3> &angles, const TwoQbFidelities &fid,
666  OpType &best_optype, unsigned &n_gates) {
667  // Prefer CX/ZZMax when possible.
668  if (fid.CX_fidelity || fid.ZZMax_fidelity) {
669  bool zzmax_is_better =
670  !fid.CX_fidelity || fid.CX_fidelity < fid.ZZMax_fidelity;
671  best_optype = zzmax_is_better ? OpType::ZZMax : OpType::CX;
672  } else if (fid.ZZPhase_fidelity) {
673  // Only ZZPhase has fidelity, so use ZZPhase.
674  best_optype = OpType::ZZPhase;
675  }
676 
677  if (best_optype == OpType::CX || best_optype == OpType::ZZMax) {
678  // Reduce n_gates if possible.
679  if (equiv_0(angles[2], 4)) {
680  n_gates = 2;
681  }
682  } else if (best_optype == OpType::ZZPhase) {
683  // Reduce n_gates if possible.
684  if (equiv_0(angles[2], 4)) {
685  n_gates = 2;
686  if (equiv_0(angles[1], 4)) {
687  n_gates = 1;
688  }
689  }
690  }
691 
692  // Finally, handle the only case where ZZPhase is preferable over ZZMax.
693  if (fid.ZZPhase_fidelity && equiv_0(angles[2], 4) && equiv_0(angles[1], 4) &&
694  n_gates > 1) {
695  n_gates = 1;
696  best_optype = OpType::ZZPhase;
697  }
698 }

718 static Circuit TK2_replacement(
719  std::array<Expr, 3> angles, const TwoQbFidelities &fid, bool allow_swaps) {
720  if (!in_weyl_chamber(angles)) {
721  throw std::domain_error("TK2 params are not normalised to Weyl chamber.");
722  }
723  OpType best_optype = OpType::CX; // default to using CX
724  unsigned n_gates = 3; // default to 3x CX
725  bool implicit_swap = false; // default to no implicit swap
726 
727  // Only used when allow_swaps == true.
728  Circuit pre, post;
729  std::array<Expr, 3> angles_swapped;
730  if (allow_swaps) {
731  // Swapped circuit
732  Circuit swap_circ(2);
733  angles_swapped = angles;
734  for (unsigned i = 0; i < 3; ++i) {
735  angles_swapped[i] += 0.5;
736  }
737  std::tie(pre, angles_swapped, post) = normalise_TK2_angles(
738  angles_swapped[0], angles_swapped[1], angles_swapped[2]);
739  pre.add_phase(0.25);
740  }
741 
742  // Try to evaluate exprs to doubles.
743  std::array<double, 3> angles_eval;
744  std::array<double, 3> angles_eval_swapped;
745  unsigned last_angle = 0;
746  for (; last_angle < 3; ++last_angle) {
747  std::optional<double> eval = eval_expr_mod(angles[last_angle]);
748  if (eval) {
749  angles_eval[last_angle] = *eval;
750  } else {
751  break;
752  }
753  if (allow_swaps) {
754  eval = eval_expr_mod(angles_swapped[last_angle]);
755  TKET_ASSERT(eval);
756  angles_eval_swapped[last_angle] = *eval;
757  }
758  }
759 
760  if (last_angle <= 2) {
761  // Not all angles could be resolved numerically.
762  // For symbolic angles, we can only provide an exact decomposition.
763  best_exact_decomposition(angles, fid, best_optype, n_gates);
764  if (allow_swaps) {
765  OpType best_optype_swapped = OpType::CX; // default to using CX
766  unsigned n_gates_swapped = 3; // default to 3x CX
767  best_exact_decomposition(
768  angles_swapped, fid, best_optype_swapped, n_gates_swapped);
769  if (n_gates_swapped < n_gates) {
770  n_gates = n_gates_swapped;
771  best_optype = best_optype_swapped;
772  angles = angles_swapped;
773  implicit_swap = true;
774  }
775  }
776  } else {
777  // For non-symbolic angles, we can find the optimal number of gates
778  // using the gate fidelities provided.
779  double max_fid =
780  best_noise_aware_decomposition(angles_eval, fid, best_optype, n_gates);
781  if (allow_swaps) {
782  OpType best_optype_swapped = OpType::CX; // default to using CX
783  unsigned n_gates_swapped = 3; // default to 3x CX
784  double max_fid_swapped = best_noise_aware_decomposition(
785  angles_eval_swapped, fid, best_optype_swapped, n_gates_swapped);
786  if (max_fid_swapped > max_fid ||
787  ((max_fid_swapped - max_fid) < EPS && n_gates_swapped < n_gates)) {
788  n_gates = n_gates_swapped;
789  best_optype = best_optype_swapped;
790  angles = angles_swapped;
791  implicit_swap = true;
792  }
793  }
794  }
795 
796  // Build circuit for substitution.
797  Circuit sub(2);
798  switch (best_optype) {
799  case OpType::ZZMax:
800  case OpType::CX: {
801  switch (n_gates) {
802  case 0:
803  break;
804  case 1: {
805  sub.append(CircPool::approx_TK2_using_1xCX());
806  break;
807  }
808  case 2: {
809  sub.append(CircPool::approx_TK2_using_2xCX(angles[0], angles[1]));
810  break;
811  }
812  case 3: {
813  sub.append(CircPool::TK2_using_3xCX(angles[0], angles[1], angles[2]));
814  break;
815  }
816  default:
817  TKET_ASSERT(!"Number of CX invalid in decompose_TK2");
818  }
819  if (best_optype == OpType::ZZMax) {
820  decompose_CX_to_HQS2().apply(sub);
821  }
822  break;
823  }
824  case OpType::ZZPhase: {
825  switch (n_gates) {
826  case 0:
827  break;
828  case 1: {
829  sub.append(CircPool::approx_TK2_using_1xZZPhase(angles[0]));
830  break;
831  }
832  case 2: {
833  sub.append(
834  CircPool::approx_TK2_using_2xZZPhase(angles[0], angles[1]));
835  break;
836  }
837  case 3: {
838  sub.append(
839  CircPool::TK2_using_ZZPhase(angles[0], angles[1], angles[2]));
840  break;
841  }
842  default:
843  TKET_ASSERT(!"Number of ZZPhase invalid in decompose_TK2");
844  }
845  break;
846  }
847  default:
848  throw BadOpType(
849  "Unrecognised target OpType in decompose_TK2", best_optype);
850  }
851 
852  if (implicit_swap) {
853  Circuit swap(2);
854  swap.add_op<unsigned>(OpType::SWAP, {0, 1});
855  sub = pre >> sub >> post >> swap;
856  sub.replace_SWAPs();
857  }
858 
859  return sub;
860 }

1588 // Any PhasedX or NPhasedX gate is replaced by an NPhasedX that is global
1589 Transform globalise_PhasedX(bool squash) {
1590  // The key bit: choose the decomposition strategy depending on the current
1591  // beta angles.
1592  //
1593  // Given the set of PhasedX gates to synthesise, choose which decomposition
1594  // strategy. Valid strategies are 0, 1, 2, corresponding to the number of
1595  // NPhasedX gates to be inserted.
1596  //
1597  // The current strategy is rather simple: it chooses to insert a single
1598  // NPhasedX whenever it would solve the current problem and there are further
1599  // PhasedX left (meaning that the rest of the computation can be deferred till
1600  // later), otherwise inserts 2x PhasedX.
1601  auto choose_strategy = [](const PhasedXFrontier &frontier, unsigned n_target,
1602  unsigned n_all) {
1603  if (n_all == 0 || n_target == 0) {
1604  return 0;
1605  }
1606  if (n_target == 1 && n_all == 1) {
1607  return 1;
1608  }
1609  if (n_target == 1 && frontier.are_phasedx_left()) {
1610  return 1;
1611  }
1612  return 2;
1613  };
1614 
1615  // the actual transform
1616  return Transform([squash, choose_strategy](Circuit &circ) {
1617  // if we squash, we start by removing all NPhasedX gates
1618  if (squash) {
1619  Transforms::decompose_NPhasedX().apply(circ);
1620  }
1621 
1622  std::vector<unsigned> range_qbs(circ.n_qubits());
1623  std::iota(range_qbs.begin(), range_qbs.end(), 0);
1624  PhasedXFrontier frontier(circ);
1625 
1626  // find a total ordering of boundary gates (aka non-NPhasedX multi-qb gates)
1627  auto is_boundary = [&circ](Vertex v) {
1628  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
1629  return PhasedXFrontier::is_interval_boundary(op);
1630  };
1631  // get a lexicographic ordering of boundary vertices
1632  auto vertices_in_order = circ.vertices_in_order();
1633  auto r = vertices_in_order | boost::adaptors::filtered(is_boundary);
1634  OptVertexVec boundary_gates(r.begin(), r.end());
1635  // Add sentinel to process the gates after last boundary_gate.
1636  // This is required to force flush the frontier after the last multi-qubit
1637  // gate.
1638  OptVertex optv;
1639  boundary_gates.push_back(optv);
1640 
1641  // whether transform is successful (always true if squash=true)
1642  bool success = squash;
1643 
1644  // Loop through each multi-qb gate.
1645  // At each iteration, decide how to decompose the single-qb gates
1646  // immediately preceding the current multi-qb gate into global gates.
1647  //
1648  // The rationale: defer any computation until just before the next multi-qb
1649  // gate. At that point, the single-qb gates on the qubits where the multi-qb
1650  // gate is acting HAVE TO be decomposed. Figure out how to do that and take
1651  // care of the garbage you might have created on other qubits at a later
1652  // point.
1653 
1654  for (OptVertex v : boundary_gates) {
1655  // the qubits whose intervals must be decomposed into global gates
1656  std::set<unsigned> curr_qubits;
1657  while (true) {
1658  if (v) {
1659  curr_qubits = frontier.qubits_ending_in(*v);
1660  } else {
1661  curr_qubits.clear();
1662  for (unsigned i = 0; i < circ.n_qubits(); ++i) {
1663  curr_qubits.insert(curr_qubits.end(), i);
1664  }
1665  }
1666  if (squash) {
1667  frontier.squash_intervals();
1668  }
1669  OptVertexVec all_phasedx = frontier.get_all_beta_vertices();
1670  OptVertexVec curr_phasedx;
1671  for (unsigned q : curr_qubits) {
1672  curr_phasedx.push_back(all_phasedx[q]);
1673  }
1674 
1675  if (all_nullopt(curr_phasedx)) {
1676  // there is nothing to decompose anymore, move to next boundary_gate
1677  break;
1678  }
1679  if (all_equal(all_phasedx)) {
1680  // this is already a global NPhasedX gate, leave untouched
1681  frontier.skip_global_gates(1);
1682  continue;
1683  }
1684 
1685  // find best decomposition strategy
1686  unsigned n_curr_betas = n_distinct_beta(circ, curr_phasedx);
1687  unsigned n_all_betas = n_distinct_beta(circ, all_phasedx);
1688  unsigned strategy;
1689  if (squash) {
1690  strategy = choose_strategy(frontier, n_curr_betas, n_all_betas);
1691  } else {
1692  // if we don't squash we decompose each NPhasedX with 2x global
1693  strategy = 2;
1694  }
1695  switch (strategy) {
1696  case 0:
1697  // do nothing
1698  break;
1699  case 1:
1700  // insert one single global NPhasedX
1701  TKET_ASSERT(curr_qubits.size() > 0);
1702  frontier.insert_1_phasedx(*(curr_qubits.begin()));
1703  success = true;
1704  break;
1705  case 2:
1706  // insert two global NPhasedX
1707  frontier.insert_2_phasedx();
1708  success = true;
1709  break;
1710  default:
1711  TKET_ASSERT(!"Invalid strategy in replace_non_global_phasedx");
1712  }
1713  }
1714  if (v) {
1715  frontier.next_multiqb(*v);
1716  }
1717  }
1718  TKET_ASSERT(frontier.is_finished());
1719 
1720  success |= absorb_Rz_NPhasedX().apply(circ);
1721 
1722  return success;
1723  });

1821  std::map<unsigned, Circuit> &cnx_cache) {

1823  qubit_vector_t original_args = cmd.get_qubits();
1824  bool conditional = cmd.get_op_ptr()->get_type() == OpType::Conditional;
1825 
1826  // Compute the decomposition
1827  auto cache = cnx_cache.find((unsigned)original_args.size());

1829  cnx = cache->second;
1830  } else {
1831  cnx = multi_controlled_to_2q(
1832  get_op_ptr(OpType::CnX, std::vector<Expr>(), original_args.size()));
1833  cnx_cache.insert({(unsigned)original_args.size(), cnx});

1835

1837  if (new_args != std::nullopt) {
1838  // The cnx circuit must only use default registers for this renaming to
1839  // work.
1840  TKET_ASSERT(cnx.is_simple());

1843  auto it =
1844  find(original_args.begin(), original_args.end(), new_args.value()[i]);
1845  int index = it - original_args.begin();
1846  qmap.insert({Qubit(i), Qubit(index)});
1847  }
1848  cnx.rename_units(qmap);
1849  }
1850 
1851  if (as_dagger) {
1852  cnx = cnx.dagger();
1853  }
1854  if (conditional) {
1855  circ.substitute_conditional(
1856  cnx, cmd.get_vertex(), Circuit::VertexDeletion::No);
1857  } else {
1858  circ.substitute(cnx, cmd.get_vertex(), Circuit::VertexDeletion::No);
1859  }

1862 Transform cnx_pairwise_decomposition() {
1863  return Transform([](Circuit &circ) {
1864  bool success = commute_through_multis().apply(circ);
1865  success |= remove_redundancies().apply(circ);
1866 
1867  // Cache CnX decompositions
1868  std::map<unsigned, Circuit> cnx_cache;
1869  // Replaced vertices to delete at the end
1870  VertexList bin;
1871 
1872  // Find all CnX gates, including conditional ones
1873  std::vector<Command> commands = circ.get_commands();
1874  std::vector<int> cnx_indices;
1875  for (unsigned i = 0; i < commands.size(); i++) {
1876  Op_ptr op = commands[i].get_op_ptr();
1877  bool conditional = op->get_type() == OpType::Conditional;
1878  if (conditional) {
1879  const Conditional &cond = static_cast<const Conditional &>(*op);
1880  op = cond.get_op();
1881  }
1882 
1883  if (op->get_type() == OpType::CnX || op->get_type() == OpType::CCX) {
1884  if (op->n_qubits() < 3) {
1885  // Decompose as CX or X without further optimisation
1886  Circuit replacement = multi_controlled_to_2q(op);
1887  if (conditional) {
1888  circ.substitute_conditional(
1889  replacement, commands[i].get_vertex(),
1890  Circuit::VertexDeletion::No);
1891  } else {
1892  circ.substitute(
1893  replacement, commands[i].get_vertex(),
1894  Circuit::VertexDeletion::No);
1895  }
1896  bin.push_back(commands[i].get_vertex());
1897  } else {
1898  cnx_indices.push_back(i);
1899  }
1900  }
1901  }
1902 
1903  // If the number of CnX is odd, decompose the first one as it is
1904  unsigned itr = 0;
1905  if (cnx_indices.size() % 2 == 1) {
1906  substitute_cnx(
1907  commands[cnx_indices[0]], std::nullopt, false, circ, cnx_cache);
1908  bin.push_back(commands[cnx_indices[0]].get_vertex());
1909  itr = 1;
1910  }
1911 
1912  // For the rest of CnX, we reorder the control qubits of each pair
1913  // to maximise the chance of gate cancelling.
1914  for (; itr + 1 < cnx_indices.size(); itr += 2) {
1915  Command &cmd1 = commands[cnx_indices[itr]];
1916  Command &cmd2 = commands[cnx_indices[itr + 1]];
1917 
1918  qubit_vector_t args1 = cmd1.get_qubits();
1919  qubit_vector_t args2 = cmd2.get_qubits();
1920 
1921  qubit_vector_t new_args1 = args1;
1922  qubit_vector_t new_args2 = args2;
1923 
1924  bool cmd1_as_dagger = false;
1925  bool cmd2_as_dagger = false;
1926 
1927  if (args1.size() == 3 && args2.size() == 3) {
1928  // CCX gates are handled differently, there
1929  // is a maximum of 1 CX reduction (needs to be realised with
1930  // CliffordSimp), when they have args: [a,b,c] and [c,a,b]
1931  std::set<Qubit> args1_set(args1.begin(), args1.end());
1932  std::set<Qubit> args2_set(args2.begin(), args2.end());
1933  // Check if they are acting on the same set of qubits
1934  if (args1_set == args2_set) {
1935  // If they have the same target, we decompose them as CCX.dagger and
1936  // CCX
1937  if (new_args1.back() == new_args2.back()) {
1938  cmd1_as_dagger = true;
1939  new_args2 = new_args1;
1940  } else {
1941  // We find qubits a,b,c
1942  // such that the first CCX acts on [a,b,c]
1943  // and the second CCX acts on [c,a,b] without changing the semantics
1944  Qubit a = (args1[0] == args2.back()) ? args1[1] : args1[0];
1945  new_args1 = {a, args2.back(), args1.back()};
1946  new_args2 = {args1.back(), a, args2.back()};
1947  }
1948  }
1949  // If the two CCX gates are not acting on the same set of qubits
1950  // we don't change ther arguments, and decompose them as they are.
1951  } else {
1952  // The more general case
1953  // we try to reorder the control args of the two gates to maximise
1954  // cancellation. The idea is to move the common control qubits to the
1955  // end of the control qubit lists e.g. C4X [q0, q1, q2, q3, q4] and C4X
1956  // [q1, q2, q4, q7, q6] will be reordered as C4X [q0, q3, q1, q2, q4]
1957  // and C4X [q4, q7, q1, q2, q6]
1958  qubit_vector_t common_ctrls;
1959  qubit_vector_t ctrl_args1 = args1;
1960  qubit_vector_t ctrl_args2 = args2;
1961  Qubit target1 = ctrl_args1.back();
1962  Qubit target2 = ctrl_args2.back();
1963  // Remove the target qubits so they only have control qubits
1964  ctrl_args1.pop_back();
1965  ctrl_args2.pop_back();
1966  // common_ctrls = ctrl_args1 intersect ctrl_args2
1967  std::sort(ctrl_args1.begin(), ctrl_args1.end());
1968  std::sort(ctrl_args2.begin(), ctrl_args2.end());
1969  std::set_intersection(
1970  ctrl_args1.begin(), ctrl_args1.end(), ctrl_args2.begin(),
1971  ctrl_args2.end(), std::back_inserter(common_ctrls));
1972 
1973  // reorder control qubits such that
1974  // new_args1 = (ctrl_args1 - common_ctrls) + common_ctrls + target1
1975  // new_args2 = (ctrl_args2 - common_ctrls) + common_ctrls + target2
1976  new_args1.clear();
1977  new_args2.clear();
1978  std::set_difference(
1979  ctrl_args1.begin(), ctrl_args1.end(), common_ctrls.begin(),
1980  common_ctrls.end(), std::back_inserter(new_args1));
1981  std::set_difference(
1982  ctrl_args2.begin(), ctrl_args2.end(), common_ctrls.begin(),
1983  common_ctrls.end(), std::back_inserter(new_args2));
1984  new_args1.insert(
1985  new_args1.end(), common_ctrls.begin(), common_ctrls.end());
1986  new_args2.insert(
1987  new_args2.end(), common_ctrls.begin(), common_ctrls.end());
1988  new_args1.push_back(target1);
1989  new_args2.push_back(target2);
1990  cmd2_as_dagger = true;
1991  }
1992  substitute_cnx(cmd1, new_args1, cmd1_as_dagger, circ, cnx_cache);
1993  substitute_cnx(cmd2, new_args2, cmd2_as_dagger, circ, cnx_cache);
1994  bin.push_back(cmd1.get_vertex());
1995  bin.push_back(cmd2.get_vertex());
1996  }
1997 
1998  circ.remove_vertices(
1999  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
2000  success |= remove_redundancies().apply(circ);
2001  success |= !bin.empty();
2002  return success;
2003  });
2004 }
