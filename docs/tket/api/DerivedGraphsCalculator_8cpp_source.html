26 void DerivedGraphsCalculator::fill_mid_vertices_for_length_two_paths(
27  const NeighboursData& ndata, VertexWSM v) {
28  m_mid_vertices_for_length_two_paths.clear();
29  const std::vector<std::pair<VertexWSM, WeightWSM>>& neighbours =
30  ndata.get_neighbours_and_weights(v);
31  for (const std::pair<VertexWSM, WeightWSM>& entry : neighbours) {
32  const VertexWSM& v1 = entry.first;
33  for (const std::pair<VertexWSM, WeightWSM>& v2_entry :
34  ndata.get_neighbours_and_weights(v1)) {
35  const VertexWSM& v2 = v2_entry.first;
36  if (v2 == v) {
37  continue;
38  }
39  // Notice that v1 occurs in sorted order!
40  m_mid_vertices_for_length_two_paths[v2].push_back(v1);
41  }
42  }
43 }

45 void DerivedGraphsCalculator::fill_d2_neighbours_and_counts(
46  DerivedGraphStructs::NeighboursAndCounts& depth_2_neighbours_and_counts) {
47  depth_2_neighbours_and_counts.clear();
48  for (const auto& entry : m_mid_vertices_for_length_two_paths) {
49  const VertexWSM& v2 = entry.first;
50  const auto& v1_set = entry.second;
51  // Notice, the v2 vertices already are in sorted order!
52  TKET_ASSERT(is_sorted_and_unique(v1_set));
53  depth_2_neighbours_and_counts.emplace_back(v2, v1_set.size());
54  }
55 }

57 void DerivedGraphsCalculator::fill_d3_neighbours_and_counts_map(
58  const NeighboursData& ndata) {
59  m_depth_3_neighbours_and_counts_map.clear();
60  for (const auto& entry : m_mid_vertices_for_length_two_paths) {
61  const VertexWSM& v2 = entry.first;
62  const auto& v1_set = entry.second;
63 
64  // Now, v--v1--v2--v3 will be either a path, or a triangle (if v3=v).
65  // v--v1--v2--v3 is constructed so that each vertex is a neighbour
66  // of the previous. Thus it's a valid path as long as v2 != v
67  // (already checked and excluded) and v3 != v1.
68  // Note that v3 IS allowed to be a neighbour of v,
69  // as long as it isn't v1.
70  // Let |{v1}|=N.
71  for (const std::pair<VertexWSM, WeightWSM>& v3_entry :
72  ndata.get_neighbours_and_weights(v2)) {
73  // The weight is simply ignored.
74  const VertexWSM& v3 = v3_entry.first;
75  if (std::binary_search(v1_set.cbegin(), v1_set.cend(), v3)) {
76  // It's a path of the form v--v1--v2--(v1)'.
77  // So we contribute N-1, since v1 != (v1)' is the only restriction.
78  m_depth_3_neighbours_and_counts_map[v3] += v1_set.size() - 1;
79  } else {
80  // It's a path of the form v--v1--v2--v3, so contributes the whole N.
81  m_depth_3_neighbours_and_counts_map[v3] += v1_set.size();
82  }
83  }
84  }
85 }

87 void DerivedGraphsCalculator::fill_remaining_d3_data(
88  VertexWSM v, DerivedGraphStructs::Count& triangle_count,
89  DerivedGraphStructs::NeighboursAndCounts& depth_3_neighbours_and_counts) {
90  depth_3_neighbours_and_counts.clear();
91  triangle_count = 0;
92  for (const auto& entry : m_depth_3_neighbours_and_counts_map) {
93  if (entry.first == v) {
94  triangle_count = entry.second;
95  } else {
96  depth_3_neighbours_and_counts.emplace_back(entry);
97  }
98  }
99 }
