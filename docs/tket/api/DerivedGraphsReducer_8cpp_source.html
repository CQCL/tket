31 bool DerivedGraphsReducer::check(std::pair<VertexWSM, VertexWSM> assignment) {
32  const DerivedGraphs::VertexData pattern_vdata =
33  m_derived_pattern_graphs.get_data(assignment.first);
34  const DerivedGraphs::VertexData target_vdata =
35  m_derived_target_graphs.get_data(assignment.second);
36 
37  return pattern_vdata.triangle_count <= target_vdata.triangle_count &&
38  pattern_vdata.d2_neighbours->size() <=
39  target_vdata.d2_neighbours->size() &&
40  pattern_vdata.d3_neighbours->size() <=
41  target_vdata.d3_neighbours->size() &&
42 
43  // It happens that the sorted degree sequences filter algorithm
44  // is identical to the algorithm here (the pattern edge weight must be
45  // <= the target edge weight).
46  FilterUtils::compatible_sorted_degree_sequences(
47  *pattern_vdata.d2_sorted_counts_iter,
48  *target_vdata.d2_sorted_counts_iter) &&
49  FilterUtils::compatible_sorted_degree_sequences(
50  *pattern_vdata.d3_sorted_counts_iter,
51  *target_vdata.d3_sorted_counts_iter);
52 }

54 ReductionResult DerivedGraphsReducer::reduce_with_derived_data(
55  const DerivedGraphStructs::NeighboursAndCounts&
56  pattern_derived_neighbours_data,
57  const DerivedGraphStructs::NeighboursAndCounts&
58  target_derived_neighbours_data,
59  VertexWSM root_pattern_vertex, DomainsAccessor& accessor,
60  std::set<VertexWSM>& work_set) {
61  // If we do create a new assignment, we will continue
62  // so that this assignment at least is fully processed.
63  // This is simpler than trying to split the work up
64  // and resume halfway through (although, really, that's what we should do
65  // to squeeze the maximum performance from lazy evaluation).
66  bool found_new_assignment = false;
67 
68  for (const std::pair<VertexWSM, DerivedGraphStructs::Count>& p_entry :
69  pattern_derived_neighbours_data) {
70  // This PV is a neighbour of the root PV, in some derived graph.
71  const VertexWSM& pv = p_entry.first;
72 
73  // This is Dom(pv), which we're reducing.
74  const std::set<VertexWSM>& domain = accessor.get_domain(pv);
75 
76  if (other_vertex_reduction_can_be_skipped_by_symmetry(
77  domain, accessor, root_pattern_vertex, pv)) {
78  continue;
79  }
80 
81  // This is the edge weight of PV--(root PV) in the derived graph.
82  const DerivedGraphStructs::Count& p_count = p_entry.second;
83 
84  fill_intersection(
85  domain, target_derived_neighbours_data, work_set,
86  // How do we convert a (Vertex, edge weight) pair into a vertex?
87  [](const std::pair<VertexWSM, DerivedGraphStructs::Count>& pair) {
88  return pair.first;
89  },
90 
91  // To get the next pair (tv2, count) after tv1
92  // (a kind of "lower bound" function),
93  // we add the extra condition that count >= p-count.
94  // (Since the lexicographic ordering is being used).
95  // (Since, edges in the derived p-graph must map to edges with equal
96  // or greater counts in the target graph).
97  // Luckily, this fits in exactly with the general framework
98  // of the fast set intersection.
99  [p_count](VertexWSM tv) { return std::make_pair(tv, p_count); });
100 
101  switch (accessor.overwrite_domain_with_set_swap(pv, work_set)) {
102  case ReductionResult::NOGOOD:
103  return ReductionResult::NOGOOD;
104  case ReductionResult::NEW_ASSIGNMENTS:
105  found_new_assignment = true;
106  break;
107  case ReductionResult::SUCCESS:
108  break;
109  }
110  }
111  if (found_new_assignment) {
112  return ReductionResult::NEW_ASSIGNMENTS;
113  }
114  return ReductionResult::SUCCESS;
115 }

117 ReductionResult DerivedGraphsReducer::reduce(
118  std::pair<VertexWSM, VertexWSM> assignment, DomainsAccessor& accessor,
119  std::set<VertexWSM>& work_set) {
120  const auto pattern_vdata =
121  m_derived_pattern_graphs.get_data(assignment.first);
122  const auto target_vdata = m_derived_target_graphs.get_data(assignment.second);
123 
124  // All the d2, d3 pattern neighbours must map to target neighbours.
125  // Furthermore, the edge weights must increase in the target graph.
126  // Even if a new assignment is found, let's reduce with all graphs
127  // before returning. Thus, we know at least that the assignment is fully
128  // processed.
129  const auto d2_result = reduce_with_derived_data(
130  *pattern_vdata.d2_neighbours, *target_vdata.d2_neighbours,
131  assignment.first, accessor, work_set);
132 
133  if (d2_result == ReductionResult::NOGOOD) {
134  return ReductionResult::NOGOOD;
135  }
136  const auto d3_result = reduce_with_derived_data(
137  *pattern_vdata.d3_neighbours, *target_vdata.d3_neighbours,
138  assignment.first, accessor, work_set);
139 
140  if (d3_result == ReductionResult::NOGOOD) {
141  return ReductionResult::NOGOOD;
142  }
143  if (d2_result == ReductionResult::NEW_ASSIGNMENTS ||
144  d3_result == ReductionResult::NEW_ASSIGNMENTS) {
145  return ReductionResult::NEW_ASSIGNMENTS;
146  }
147  return ReductionResult::SUCCESS;
148 }
