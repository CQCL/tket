21// simple get key wrapped in std::optional
22template <typename T>
23static std::optional<typename T::mapped_type> maybe_get(
24 const T& error_map, const typename T::key_type& key) {
25 const auto& it = error_map.find(key);
26 return (it != error_map.end()) ? std::make_optional(it->second)
27 : std::nullopt;
28}

53gate_error_t DeviceCharacterisation::get_error(
54 const Node& n, const OpType& op) const {
55 std::optional<op_errors_t> maybe_dict = maybe_get(op_node_errors_, n);
56 if (maybe_dict) {
57 std::optional<gate_error_t> maybe_err = maybe_get(*maybe_dict, op);
58 if (maybe_err) {
59 return *maybe_err;
60 }
61 }
62 return get_error(n);
63}
