24DiagonalBox::DiagonalBox(const Eigen::VectorXcd &diagonal, bool upper_triangle)
25 : Box(OpType::DiagonalBox),
26 diagonal_(diagonal),
27 upper_triangle_(upper_triangle) {
28 std::size_t length = diagonal.size();
29 if (length < 2 || (length & (length - 1)) != 0) {
30 throw std::invalid_argument(
31 "The size of the diagonal operator passed to DiagonalBox is not a "
32 "power of 2.");
33 }
34 // check the operator is unitary
35 for (unsigned i = 0; i < diagonal.size(); i++) {
36 if (std::abs(1 - std::abs(diagonal[i])) > EPS) {
37 throw std::invalid_argument(
38 "The input diagonal passed to DiagonalBox is not unitary.");
39 }
40 }
41}

71static Circuit diagonal_circ(
72 const Eigen::VectorXcd &diagonal, bool upper_triangle) {
73 unsigned n_qubits = (unsigned)log2(diagonal.size());
74 Circuit circ(n_qubits);
75 Eigen::VectorXcd d(diagonal);
76 unsigned n_ctrl_qubits = n_qubits - 1;
77 while (d.size() != 1) {
78 ctrl_op_map_t multip_rz;
79 Eigen::VectorXcd new_d(d.size() / 2);
80 for (unsigned i = 0; i < new_d.size(); i++) {
81 Complex a;
82 Complex b;
83 if (upper_triangle) {
84 a = d[2 * i];
85 b = d[2 * i + 1];
86 } else {
87 a = d[i];
88 b = d[i + (unsigned)d.size() / 2];
89 }
90 // convert diag[a,b] into a p*Rz(alpha)
91 double a_phase = std::arg(a);
92 double b_phase = std::arg(b);
93 double alpha = (b_phase - a_phase) / PI;
94 Complex p = std::exp((b_phase + a_phase) * 0.5 * i_);
95 std::vector<bool> bitstr = dec_to_bin(i, n_ctrl_qubits);
96 if (std::abs(alpha) > EPS) {
97 multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, alpha)});
98 }
99 new_d[i] = p;
100 }
101 if (!multip_rz.empty()) {
102 std::vector<unsigned> args(n_ctrl_qubits);
103 if (upper_triangle) {
104 std::iota(std::begin(args), std::end(args), 0);
105 args.push_back(n_ctrl_qubits);
106 } else {
107 std::iota(std::begin(args), std::end(args), n_qubits - n_ctrl_qubits);
108 args.push_back(n_qubits - n_ctrl_qubits - 1);
109 }
110 circ.add_box(MultiplexedRotationBox(multip_rz), args);
111 }
112 d = new_d;
113 n_ctrl_qubits--;
114 }
115 circ.add_phase(std::arg(d[0]) / PI);
116 return circ;
117};

123bool DiagonalBox::is_equal(const Op &op_other) const {
124 const DiagonalBox &other = dynamic_cast<const DiagonalBox &>(op_other);
125 if (id_ == other.get_id()) return true;
126 return upper_triangle_ == other.upper_triangle_ &&
127 diagonal_.isApprox(other.diagonal_);
128}

130nlohmann::json DiagonalBox::to_json(const Op_ptr &op) {
