26 DiagonalBox::DiagonalBox(const Eigen::VectorXcd &diagonal, bool upper_triangle)
27  : Box(OpType::DiagonalBox),
28  diagonal_(diagonal),
29  upper_triangle_(upper_triangle) {
30  std::size_t length = diagonal.size();
31  if (length < 2 || (length & (length - 1)) != 0) {
32  throw std::invalid_argument(
33  "The size of the diagonal operator passed to DiagonalBox is not a "
34  "power of 2.");
35  }
36  // check the operator is unitary
37  for (unsigned i = 0; i < diagonal.size(); i++) {
38  if (std::abs(1 - std::abs(diagonal[i])) > EPS) {
39  throw std::invalid_argument(
40  "The input diagonal passed to DiagonalBox is not unitary.");
41  }
42  }
43 }

73 static Circuit diagonal_circ(
74  const Eigen::VectorXcd &diagonal, bool upper_triangle) {
75  unsigned n_qubits = (unsigned)log2(diagonal.size());
76  Circuit circ(n_qubits);
77  Eigen::VectorXcd d(diagonal);
78  unsigned n_ctrl_qubits = n_qubits - 1;
79  while (d.size() != 1) {
80  ctrl_op_map_t multip_rz;
81  Eigen::VectorXcd new_d(d.size() / 2);
82  for (unsigned i = 0; i < new_d.size(); i++) {
83  Complex a;
84  Complex b;
85  if (upper_triangle) {
86  a = d[2 * i];
87  b = d[2 * i + 1];
88  } else {
89  a = d[i];
90  b = d[i + (unsigned)d.size() / 2];
91  }
92  // convert diag[a,b] into a p*Rz(alpha)
93  double a_phase = std::arg(a);
94  double b_phase = std::arg(b);
95  double alpha = (b_phase - a_phase) / PI;
96  Complex p = std::exp((b_phase + a_phase) * 0.5 * i_);
97  std::vector<bool> bitstr = dec_to_bin(i, n_ctrl_qubits);
98  if (std::abs(alpha) > EPS) {
99  multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, alpha)});
100  }
101  new_d[i] = p;
102  }
103  if (!multip_rz.empty()) {
104  std::vector<unsigned> args(n_ctrl_qubits);
105  if (upper_triangle) {
106  std::iota(std::begin(args), std::end(args), 0);
107  args.push_back(n_ctrl_qubits);
108  } else {
109  std::iota(std::begin(args), std::end(args), n_qubits - n_ctrl_qubits);
110  args.push_back(n_qubits - n_ctrl_qubits - 1);
111  }
112  circ.add_box(MultiplexedRotationBox(multip_rz), args);
113  }
114  d = new_d;
115  n_ctrl_qubits--;
116  }
117  circ.add_phase(std::arg(d[0]) / PI);
118  return circ;
119 };

125 bool DiagonalBox::is_equal(const Op &op_other) const {
126  const DiagonalBox &other = dynamic_cast<const DiagonalBox &>(op_other);
