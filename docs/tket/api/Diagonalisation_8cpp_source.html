25 void check_easy_diagonalise(
26  std::list<std::pair<QubitPauliTensor, Expr>> &gadgets,
27  std::set<Qubit> &qubits, Circuit &circ) {
28  Conjugations conjugations;
29  std::set<Qubit>::iterator qb_iter = qubits.begin();
30  for (std::set<Qubit>::iterator next = qb_iter; qb_iter != qubits.end();
31  qb_iter = next) {
32  ++next;
33  Pauli p1 = Pauli::I;
34  bool remove_qb = true;
35  for (const std::pair<QubitPauliTensor, Expr> &pgp : gadgets) {
36  std::map<tket::Qubit, tket::Pauli>::const_iterator map_iter =
37  pgp.first.string.map.find(*qb_iter);
38  if (map_iter == pgp.first.string.map.end()) continue;
39  Pauli p2 = map_iter->second;
40  if (p2 == Pauli::I) continue;
41  if (p1 == Pauli::I) {
42  p1 = p2;
43  } else if (p1 != p2) {
44  remove_qb = false;
45  break;
46  }
47  }
48  if (remove_qb) {
49  switch (p1) {
50  case Pauli::I:
51  break;
52  case Pauli::X:
53  conjugations.push_back({OpType::H, {*qb_iter}});
54  circ.add_op<Qubit>(OpType::H, {*qb_iter});
55  break;
56  case Pauli::Y:
57  conjugations.push_back({OpType::Vdg, {*qb_iter}});
58  circ.add_op<Qubit>(OpType::V, {*qb_iter});
59  break;
60  case Pauli::Z:
61  break;
62  default:
63  throw UnknownPauli();
64  }
65  qubits.erase(qb_iter);
66  }
67  }
68  for (std::list<std::pair<QubitPauliTensor, Expr>>::iterator iter =
69  gadgets.begin();
70  iter != gadgets.end(); ++iter) {
71  apply_conjugations(iter->first, conjugations);
72  }
73 }

75 std::optional<std::pair<Pauli, Pauli>> check_pair_compatibility(
76  const Qubit &qb1, const Qubit &qb2,
77  const std::list<std::pair<QubitPauliTensor, Expr>> &gadgets) {
78  if (qb1 == qb2) return std::nullopt;
79 
80  /* Do exhaustive search for a Pauli A and Pauli B that
81  satisfy Theorem */
82  std::list<Pauli> paulis{Pauli::Z, Pauli::X, Pauli::Y};
83  for (Pauli pauli1 : paulis) {
84  for (Pauli pauli2 : paulis) {
85  bool found_pair = true;
86  for (const std::pair<QubitPauliTensor, Expr> &pgp : gadgets) {
87  Pauli inner_p_1;
88  QubitPauliMap::const_iterator iter1 = pgp.first.string.map.find(qb1);
89  if (iter1 == pgp.first.string.map.end())
90  inner_p_1 = Pauli::I;
91  else
92  inner_p_1 = iter1->second;
93 
94  Pauli inner_p_2;
95  QubitPauliMap::const_iterator iter2 = pgp.first.string.map.find(qb2);
96  if (iter2 == pgp.first.string.map.end())
97  inner_p_2 = Pauli::I;
98  else
99  inner_p_2 = iter2->second;
100 
101  if (inner_p_1 == Pauli::I || inner_p_1 == pauli1) {
102  if (!(inner_p_2 == Pauli::I || inner_p_2 == pauli2)) {
103  found_pair = false;
104  break;
105  }
106  }
107  if (inner_p_2 == Pauli::I || inner_p_2 == pauli2) {
108  if (!(inner_p_1 == Pauli::I || inner_p_1 == pauli1)) {
109  found_pair = false;
110  break;
111  }
112  }
113  }
114  if (found_pair) {
115  return std::make_pair(pauli1, pauli2);
116  }
117  }
118  }
119  return std::nullopt;
120 }

122 void greedy_diagonalise(
123  const std::list<std::pair<QubitPauliTensor, Expr>> &gadgets,
124  std::set<Qubit> &qubits, Conjugations &conjugations, Circuit &circ,
125  CXConfigType cx_config) {
126  unsigned total_counter = UINT_MAX;
127  QubitPauliMap to_diag;
128  for (std::list<std::pair<QubitPauliTensor, Expr>>::const_iterator pgp_iter =
129  gadgets.begin();
130  pgp_iter != gadgets.end(); ++pgp_iter) {
131  unsigned support_counter = 0;
132  QubitPauliMap to_diag_candidates;
133  for (const Qubit &qb : qubits) {
134  QubitPauliMap::const_iterator pauli_iter =
135  pgp_iter->first.string.map.find(qb);
136  if (pauli_iter == pgp_iter->first.string.map.end()) continue;
137  if (pauli_iter->second != Pauli::I) {
138  ++support_counter;
139  to_diag_candidates.insert(*pauli_iter);
140  }
141  }
142  if (support_counter < total_counter && support_counter > 1) {
143  total_counter = support_counter;
144  to_diag = to_diag_candidates;
145  }
146  }
147  if (to_diag.empty()) {
148  throw std::logic_error("Brute Force Diagonalise can't find a candidate!");
149  }
150  for (QubitPauliMap::iterator qb_p_iter = to_diag.begin();
151  qb_p_iter != to_diag.end(); ++qb_p_iter) {
152  const Qubit &qb = qb_p_iter->first;
153  Pauli p = qb_p_iter->second;
154  switch (p) {
155  case Pauli::X: {
156  conjugations.push_back({OpType::H, {qb}});
157  circ.add_op<Qubit>(OpType::H, {qb});
158  break;
159  }
160  case Pauli::Y: {
161  conjugations.push_back({OpType::Vdg, {qb}});
162  circ.add_op<Qubit>(OpType::V, {qb});
163  break;
164  }
165  case Pauli::Z: {
166  break;
167  }
168  case Pauli::I:
169  default:
170  throw UnknownPauli();
171  }
172  }
173  qubit_vector_t diag_qubits;
174  for (const auto &[k, v] : to_diag) diag_qubits.push_back(k);
175  unsigned n_qubits = diag_qubits.size();
176  Qubit first_qb = diag_qubits[0];
177 
178  switch (cx_config) {
179  case CXConfigType::Snake: {
180  for (unsigned i = n_qubits - 1; i > 0; --i) {
181  Qubit qb = diag_qubits[i], before = diag_qubits[i - 1];
182  conjugations.push_back({OpType::CX, {qb, before}});
183  circ.add_op<Qubit>(OpType::CX, {qb, before});
184  }
185  break;
186  }
187  case CXConfigType::Star: {
188  for (unsigned i = 1; i < n_qubits; ++i) {
189  Qubit qb = diag_qubits[i];
190  conjugations.push_back({OpType::CX, {qb, first_qb}});
191  circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
192  }
193  break;
194  }
195  case CXConfigType::Tree: {
196  unsigned complete_layers = floor(log2(n_qubits));
197  unsigned dense_end = pow(2, complete_layers);
198  for (unsigned j = 0; j < n_qubits - dense_end; j++) {
199  circ.add_op<Qubit>(
200  OpType::CX,
201  {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]});
202  conjugations.push_back(
203  {OpType::CX,
204  {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]}});
205  }
206  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
207  for (unsigned j = 0; j < dense_end; j += 2 * step_size) {
208  circ.add_op<Qubit>(
209  OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]});
210  conjugations.push_back(
211  {OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]}});
212  }
213  }
214  break;
215  }
216  case CXConfigType::MultiQGate: {
217  int sign_correction = 1;
218  for (int q = n_qubits - 1; q > 0; q -= 2) {
219  Qubit qb = diag_qubits[q];
220  if (q - 1 > 0) {
221  Qubit before = diag_qubits[q - 1];
222  circ.add_op<Qubit>(OpType::H, {qb});
223  circ.add_op<Qubit>(OpType::H, {before});
224  circ.add_op<Qubit>(OpType::XXPhase3, 0.5, {qb, before, first_qb});
225  conjugations.push_back({OpType::H, {qb}});
226  conjugations.push_back({OpType::H, {before}});
227  conjugations.push_back({OpType::XXPhase3, {qb, before, first_qb}});
228  sign_correction *= -1;
229  } else {
230  circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
231  conjugations.push_back({OpType::CX, {qb, first_qb}});
232  }
233  }
234  if (sign_correction < 0) {
235  circ.add_op<Qubit>(OpType::X, {first_qb});
236  conjugations.push_back({OpType::X, {first_qb}});
237  }
238  break;
239  }
240  default:
241  throw UnknownCXConfigType();
242  }
243  qubits.erase(first_qb);
244 }

246 /* Diagonalise a set of Pauli Gadgets simultaneously using Cliffords*/
247 Circuit mutual_diagonalise(
248  std::list<std::pair<QubitPauliTensor, Expr>> &gadgets,
249  std::set<Qubit> qubits, CXConfigType cx_config) {
250  Circuit cliff_circ;
251  for (const Qubit &qb : qubits) {
252  cliff_circ.add_qubit(qb);
253  }
254  check_easy_diagonalise(gadgets, qubits, cliff_circ);
255  while (!qubits.empty()) {
256  Conjugations conjugations;
257  Qubit qb_a;
258  Qubit qb_b;
259  std::pair<Pauli, Pauli> pauli_pair;
260  bool found_match = false;
261  /* First, try to find some qubits we can reduce using only 1 CX */
262  for (const Qubit &qb1 : qubits) {
263  for (const Qubit &qb2 : qubits) {
264  std::optional<std::pair<Pauli, Pauli>> compatible =
265  check_pair_compatibility(qb1, qb2, gadgets);
266  if (compatible.has_value()) {
267  pauli_pair = *compatible;
268  qb_a = qb1;
269  qb_b = qb2;
270  found_match = true;
271  break;
272  }
273  }
274  if (found_match) break;
275  }
276  if (found_match) {
277  Pauli p1 = pauli_pair.first;
278  Pauli p2 = pauli_pair.second;
279  switch (p1) {
280  case Pauli::X: {
281  conjugations.push_back({OpType::H, {qb_a}});
282  cliff_circ.add_op<Qubit>(OpType::H, {qb_a});
283  break;
284  }
285  case Pauli::Y: {
286  conjugations.push_back({OpType::Vdg, {qb_a}});
287  cliff_circ.add_op<Qubit>(OpType::V, {qb_a});
288  break;
289  }
290  case Pauli::Z: {
291  break;
292  }
293  case Pauli::I:
294  default:
295  throw UnknownPauli();
296  }
297  switch (p2) {
298  case Pauli::X: {
299  conjugations.push_back({OpType::H, {qb_b}});
300  cliff_circ.add_op<Qubit>(OpType::H, {qb_b});
301  break;
302  }
303  case Pauli::Y: {
304  conjugations.push_back({OpType::Vdg, {qb_b}});
305  cliff_circ.add_op<Qubit>(OpType::V, {qb_b});
306  break;
307  }
308  case Pauli::Z: {
309  break;
310  }
311  case Pauli::I:
312  default:
313  throw UnknownPauli();
314  }
315  conjugations.push_back({OpType::CX, {qb_a, qb_b}});
316  cliff_circ.add_op<Qubit>(OpType::CX, {qb_a, qb_b});
317  qubits.erase(qb_b);
318  }
319  /* If we can't, do it with `n-1` CXs, where `n` := no. of undiagonalised
320  * qubits */
321  if (!found_match) {
322  greedy_diagonalise(gadgets, qubits, conjugations, cliff_circ, cx_config);
323  }
324  for (std::list<std::pair<QubitPauliTensor, Expr>>::iterator iter =
325  gadgets.begin();
326  iter != gadgets.end(); ++iter) {
327  apply_conjugations(iter->first, conjugations);
328  }
329  // we may have made some easy-to-remove qubits
330  check_easy_diagonalise(gadgets, qubits, cliff_circ);
331  }
332  return cliff_circ;
333 }

335 void apply_conjugations(
336  QubitPauliTensor &qps, const Conjugations &conjugations) {
337  for (const auto &optype_qubit_pair : conjugations) {
338  OpType ot = optype_qubit_pair.first;
339  const qubit_vector_t &qbs = optype_qubit_pair.second;
340  if (!optypeinfo().at(ot).signature ||
341  optypeinfo().at(ot).signature->size() != qbs.size())
342  throw std::logic_error("Incompatible qubit count for conjugations");
343  switch (ot) {
344  case OpType::H:
345  case OpType::S:
346  case OpType::Sdg:
347  case OpType::V:
348  case OpType::Vdg:
349  case OpType::X:
350  case OpType::Z:
351  conjugate_PauliTensor(qps, ot, qbs[0]);
352  break;
353  case OpType::CX:
354  conjugate_PauliTensor(qps, ot, qbs[0], qbs[1]);
355  break;
356  case OpType::XXPhase3:
357  conjugate_PauliTensor(qps, ot, qbs[0], qbs[1], qbs[2]);
358  break;
359  default:
360  throw UnknownOpType();
361  }
362  }
363 }
