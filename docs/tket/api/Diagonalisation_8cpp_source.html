25 void check_easy_diagonalise(
26  std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> &qubits,
27  Circuit &circ) {
28  Conjugations conjugations;
29  std::set<Qubit>::iterator qb_iter = qubits.begin();
30  for (std::set<Qubit>::iterator next = qb_iter; qb_iter != qubits.end();
31  qb_iter = next) {
32  ++next;
33  Pauli p1 = Pauli::I;
34  bool remove_qb = true;
35  for (const SpSymPauliTensor &gadget : gadgets) {
36  Pauli p2 = gadget.get(*qb_iter);
37  if (p2 == Pauli::I) continue;
38  if (p1 == Pauli::I) {
39  p1 = p2;
40  } else if (p1 != p2) {
41  remove_qb = false;
42  break;
43  }
44  }
45  if (remove_qb) {
46  switch (p1) {
47  case Pauli::I:
48  break;
49  case Pauli::X:
50  conjugations.push_back({OpType::H, {*qb_iter}});
51  circ.add_op<Qubit>(OpType::H, {*qb_iter});
52  break;
53  case Pauli::Y:
54  conjugations.push_back({OpType::Vdg, {*qb_iter}});
55  circ.add_op<Qubit>(OpType::V, {*qb_iter});
56  break;
57  case Pauli::Z:
58  break;
59  default:
60  throw std::logic_error(
61  "Unknown Pauli encountered in checking diagonalisation");
62  }
63  qubits.erase(qb_iter);
64  }
65  }
66  for (SpSymPauliTensor &gadget : gadgets) {
67  apply_conjugations(gadget, conjugations);
68  }
69 }

71 std::optional<std::pair<Pauli, Pauli>> check_pair_compatibility(
72  const Qubit &qb1, const Qubit &qb2,
73  const std::list<SpSymPauliTensor> &gadgets) {
74  if (qb1 == qb2) return std::nullopt;
75 
76  /* Do exhaustive search for a Pauli A and Pauli B that
77  satisfy Theorem */
78  std::list<Pauli> paulis{Pauli::Z, Pauli::X, Pauli::Y};
79  for (Pauli pauli1 : paulis) {
80  for (Pauli pauli2 : paulis) {
81  bool found_pair = true;
82  for (const SpSymPauliTensor &gadget : gadgets) {
83  Pauli inner_p_1 = gadget.get(qb1);
84  Pauli inner_p_2 = gadget.get(qb2);
85 
86  if (inner_p_1 == Pauli::I || inner_p_1 == pauli1) {
87  if (!(inner_p_2 == Pauli::I || inner_p_2 == pauli2)) {
88  found_pair = false;
89  break;
90  }
91  }
92  if (inner_p_2 == Pauli::I || inner_p_2 == pauli2) {
93  if (!(inner_p_1 == Pauli::I || inner_p_1 == pauli1)) {
94  found_pair = false;
95  break;
96  }
97  }
98  }
99  if (found_pair) {
100  return std::make_pair(pauli1, pauli2);
101  }
102  }
103  }
104  return std::nullopt;
105 }

107 void greedy_diagonalise(
108  const std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> &qubits,
109  Conjugations &conjugations, Circuit &circ, CXConfigType cx_config) {
110  unsigned total_counter = UINT_MAX;
111  QubitPauliMap to_diag;
112  for (const SpSymPauliTensor &gadget : gadgets) {
113  unsigned support_counter = 0;
114  QubitPauliMap to_diag_candidates;
115  for (const Qubit &qb : qubits) {
116  Pauli p = gadget.get(qb);
117  if (p != Pauli::I) {
118  ++support_counter;
119  to_diag_candidates.insert({qb, p});
120  }
121  }
122  if (support_counter < total_counter && support_counter > 1) {
123  total_counter = support_counter;
124  to_diag = to_diag_candidates;
125  }
126  }
127  if (to_diag.empty()) {
128  throw std::logic_error("Brute Force Diagonalise can't find a candidate!");
129  }
130  for (const std::pair<const Qubit, Pauli> &qp : to_diag) {
131  const Qubit &qb = qp.first;
132  switch (qp.second) {
133  case Pauli::X: {
134  conjugations.push_back({OpType::H, {qb}});
135  circ.add_op<Qubit>(OpType::H, {qb});
136  break;
137  }
138  case Pauli::Y: {
139  conjugations.push_back({OpType::Vdg, {qb}});
140  circ.add_op<Qubit>(OpType::V, {qb});
141  break;
142  }
143  case Pauli::Z: {
144  break;
145  }
146  case Pauli::I:
147  default:
148  throw std::logic_error("Unknown Pauli in greedy diagonalisation.");
149  }
150  }
151  qubit_vector_t diag_qubits;
152  for (const auto &[k, v] : to_diag) diag_qubits.push_back(k);
153  unsigned n_qubits = diag_qubits.size();
154  Qubit first_qb = diag_qubits[0];
155 
156  switch (cx_config) {
157  case CXConfigType::Snake: {
158  for (unsigned i = n_qubits - 1; i > 0; --i) {
159  Qubit qb = diag_qubits[i], before = diag_qubits[i - 1];
160  conjugations.push_back({OpType::CX, {qb, before}});
161  circ.add_op<Qubit>(OpType::CX, {qb, before});
162  }
163  break;
164  }
165  case CXConfigType::Star: {
166  for (unsigned i = 1; i < n_qubits; ++i) {
167  Qubit qb = diag_qubits[i];
168  conjugations.push_back({OpType::CX, {qb, first_qb}});
169  circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
170  }
171  break;
172  }
173  case CXConfigType::Tree: {
174  unsigned complete_layers = floor(log2(n_qubits));
175  unsigned dense_end = pow(2, complete_layers);
176  for (unsigned j = 0; j < n_qubits - dense_end; j++) {
177  circ.add_op<Qubit>(
178  OpType::CX,
179  {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]});
180  conjugations.push_back(
181  {OpType::CX,
182  {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]}});
183  }
184  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
185  for (unsigned j = 0; j < dense_end; j += 2 * step_size) {
186  circ.add_op<Qubit>(
187  OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]});
188  conjugations.push_back(
189  {OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]}});
190  }
191  }
192  break;
193  }
194  case CXConfigType::MultiQGate: {
195  int sign_correction = 1;
196  for (int q = n_qubits - 1; q > 0; q -= 2) {
197  Qubit qb = diag_qubits[q];
198  if (q - 1 > 0) {
199  Qubit before = diag_qubits[q - 1];
200  circ.add_op<Qubit>(OpType::H, {qb});
201  circ.add_op<Qubit>(OpType::H, {before});
202  circ.add_op<Qubit>(OpType::XXPhase3, 0.5, {qb, before, first_qb});
203  conjugations.push_back({OpType::H, {qb}});
204  conjugations.push_back({OpType::H, {before}});
205  conjugations.push_back({OpType::XXPhase3, {qb, before, first_qb}});
206  sign_correction *= -1;
207  } else {
208  circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
209  conjugations.push_back({OpType::CX, {qb, first_qb}});
210  }
211  }
212  if (sign_correction < 0) {
213  circ.add_op<Qubit>(OpType::X, {first_qb});
214  conjugations.push_back({OpType::X, {first_qb}});
215  }
216  break;
217  }
218  default:
219  throw std::logic_error("Unknown CXConfigType in greedy diagonalisation.");
220  }
221  qubits.erase(first_qb);
222 }

224 /* Diagonalise a set of Pauli Gadgets simultaneously using Cliffords*/
225 Circuit mutual_diagonalise(
226  std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> qubits,
227  CXConfigType cx_config) {
228  Circuit cliff_circ;
229  for (const Qubit &qb : qubits) {
230  cliff_circ.add_qubit(qb);
231  }
232  check_easy_diagonalise(gadgets, qubits, cliff_circ);
233  while (!qubits.empty()) {
234  Conjugations conjugations;
235  Qubit qb_a;
236  Qubit qb_b;
237  std::pair<Pauli, Pauli> pauli_pair;
238  bool found_match = false;
239  /* First, try to find some qubits we can reduce using only 1 CX */
240  for (const Qubit &qb1 : qubits) {
241  for (const Qubit &qb2 : qubits) {
242  std::optional<std::pair<Pauli, Pauli>> compatible =
243  check_pair_compatibility(qb1, qb2, gadgets);
244  if (compatible.has_value()) {
245  pauli_pair = *compatible;
246  qb_a = qb1;
247  qb_b = qb2;
248  found_match = true;
249  break;
250  }
251  }
252  if (found_match) break;
253  }
254  if (found_match) {
255  Pauli p1 = pauli_pair.first;
256  Pauli p2 = pauli_pair.second;
257  switch (p1) {
258  case Pauli::X: {
259  conjugations.push_back({OpType::H, {qb_a}});
260  cliff_circ.add_op<Qubit>(OpType::H, {qb_a});
261  break;
262  }
263  case Pauli::Y: {
264  conjugations.push_back({OpType::Vdg, {qb_a}});
265  cliff_circ.add_op<Qubit>(OpType::V, {qb_a});
266  break;
267  }
268  case Pauli::Z: {
269  break;
270  }
271  case Pauli::I:
272  default:
273  throw std::logic_error("Unknown Pauli in mutual diagonalisation.");
274  }
275  switch (p2) {
276  case Pauli::X: {
277  conjugations.push_back({OpType::H, {qb_b}});
278  cliff_circ.add_op<Qubit>(OpType::H, {qb_b});
279  break;
280  }
281  case Pauli::Y: {
282  conjugations.push_back({OpType::Vdg, {qb_b}});
283  cliff_circ.add_op<Qubit>(OpType::V, {qb_b});
284  break;
285  }
286  case Pauli::Z: {
287  break;
288  }
289  case Pauli::I:
290  default:
291  throw std::logic_error("Unknown Pauli in mutual diagonalisation.");
292  }
293  conjugations.push_back({OpType::CX, {qb_a, qb_b}});
294  cliff_circ.add_op<Qubit>(OpType::CX, {qb_a, qb_b});
295  qubits.erase(qb_b);
296  }
297  /* If we can't, do it with `n-1` CXs, where `n` := no. of undiagonalised
298  * qubits */
299  if (!found_match) {
300  greedy_diagonalise(gadgets, qubits, conjugations, cliff_circ, cx_config);
301  }
302  for (SpSymPauliTensor &gadget : gadgets) {
303  apply_conjugations(gadget, conjugations);
304  }
305  // we may have made some easy-to-remove qubits
306  check_easy_diagonalise(gadgets, qubits, cliff_circ);
307  }
308  return cliff_circ;
309 }

311 void apply_conjugations(
312  SpSymPauliTensor &qps, const Conjugations &conjugations) {
313  SpPauliStabiliser stab(qps.string);
314  for (const auto &optype_qubit_pair : conjugations) {
315  OpType ot = optype_qubit_pair.first;
316  const qubit_vector_t &qbs = optype_qubit_pair.second;
317  if (!optypeinfo().at(ot).signature ||
318  optypeinfo().at(ot).signature->size() != qbs.size())
319  throw std::logic_error("Incompatible qubit count for conjugations");
320  switch (ot) {
321  case OpType::H:
322  case OpType::S:
323  case OpType::Sdg:
324  case OpType::V:
325  case OpType::Vdg:
326  case OpType::X:
327  case OpType::Z:
328  conjugate_PauliTensor(stab, ot, qbs[0]);
329  break;
330  case OpType::CX:
331  conjugate_PauliTensor(stab, ot, qbs[0], qbs[1]);
332  break;
333  case OpType::XXPhase3:
334  conjugate_PauliTensor(stab, ot, qbs[0], qbs[1], qbs[2]);
335  break;
336  default:
337  throw std::logic_error(
338  "Unknown OpType received when applying conjugations.");
339  }
340  }
341  qps.string = stab.string;
342  qps.coeff *= cast_coeff<quarter_turns_t, Expr>(stab.coeff);
343 }

345 std::pair<Circuit, Qubit> reduce_pauli_to_z(
346  const SpPauliStabiliser &pauli, CXConfigType cx_config) {
347  Circuit circ;
348  qubit_vector_t qubits;
349  for (const std::pair<const Qubit, Pauli> &qp : pauli.string) {
350  circ.add_qubit(qp.first);
351  if (qp.second != Pauli::I) qubits.push_back(qp.first);
352  switch (qp.second) {
353  case Pauli::X: {
354  circ.add_op<Qubit>(OpType::H, {qp.first});
355  break;
356  }
357  case Pauli::Y: {
358  circ.add_op<Qubit>(OpType::V, {qp.first});
359  break;
360  }
361  default: {
362  break;
363  }
364  }
365  }
366  unsigned n_qubits = qubits.size();
367  if (n_qubits == 0) throw std::logic_error("Cannot reduce identity to Z");
368  switch (cx_config) {
369  case CXConfigType::Snake: {
370  for (unsigned i = n_qubits - 1; i != 0; --i) {
371  circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.at(i - 1)});
372  }
373  break;
374  }
375  case CXConfigType::Star: {
376  for (unsigned i = n_qubits - 1; i != 0; --i) {
377  circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.front()});
378  }
379  break;
380  }
381  case CXConfigType::Tree: {
382  for (unsigned step_size = 1; step_size < n_qubits; step_size *= 2) {
383  for (unsigned i = 0; step_size + i < n_qubits; i += 2 * step_size) {
384  circ.add_op<Qubit>(
385  OpType::CX, {qubits.at(step_size + i), qubits.at(i)});
386  }
387  }
388  break;
389  }
390  case CXConfigType::MultiQGate: {
391  bool flip_phase = false;
392  for (unsigned i = n_qubits - 1; i != 0; --i) {
393  if (i == 1) {
394  circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.front()});
395  } else {
400  circ.add_op<Qubit>(OpType::H, {qubits.at(i)});
401  circ.add_op<Qubit>(OpType::H, {qubits.at(i - 1)});
402  circ.add_op<Qubit>(
403  OpType::XXPhase3, 0.5,
404  {qubits.at(i), qubits.at(i - 1), qubits.front()});
405  --i;
406  flip_phase = !flip_phase;
407  }
408  }
409  if (flip_phase) circ.add_op<Qubit>(OpType::X, {qubits.front()});
410  break;
411  }
412  }
413  return {circ, qubits.front()};
414 }

416 static void reduce_shared_qs_by_CX_snake(
417  Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
418  SpPauliStabiliser &pauli1) {
419  unsigned match_size = match.size();
420  while (match_size > 1) { // We allow one match left over
421  auto it = --match.end();
422  Qubit to_eliminate = *it;
423  match.erase(it);
424  Qubit helper = *match.rbegin();
425  // extend CX snake
426  circ.add_op<Qubit>(OpType::CX, {to_eliminate, helper});
427  pauli0.string.erase(to_eliminate);
428  pauli1.string.erase(to_eliminate);
429  match_size--;
430  }
431 }

433 static void reduce_shared_qs_by_CX_star(
434  Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
435  SpPauliStabiliser &pauli1) {
436  std::set<Qubit>::iterator iter = match.begin();
437  for (std::set<Qubit>::iterator next = match.begin(); match.size() > 1;
438  iter = next) {
439  ++next;
440  Qubit to_eliminate = *iter;
441  circ.add_op<Qubit>(OpType::CX, {to_eliminate, *match.rbegin()});
442  pauli0.string.erase(to_eliminate);
443  pauli1.string.erase(to_eliminate);
444  match.erase(iter);
445  }
446 }

448 static void reduce_shared_qs_by_CX_tree(
449  Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
450  SpPauliStabiliser &pauli1) {
451  while (match.size() > 1) {
452  std::set<Qubit> remaining;
453  std::set<Qubit>::iterator it = match.begin();
454  while (it != match.end()) {
455  Qubit maintained = *it;
456  it++;
457  remaining.insert(maintained);
458  if (it != match.end()) {
459  Qubit to_eliminate = *it;
460  it++;
461  circ.add_op<Qubit>(OpType::CX, {to_eliminate, maintained});
462  pauli0.string.erase(to_eliminate);
463  pauli1.string.erase(to_eliminate);
464  }
465  }
466  match = remaining;
467  }
468 }

470 static void reduce_shared_qs_by_CX_multiqgate(
471  Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
472  SpPauliStabiliser &pauli1) {
473  if (match.size() <= 1) {
474  return;
475  }
476  // last qubit is target
477  Qubit target = *match.rbegin();
478  while (match.size() > 1) {
479  std::set<Qubit>::iterator iter = match.begin();
480  if (match.size() == 2) {
481  // use CX
482  Qubit to_eliminate = *iter;
483  match.erase(iter);
484  pauli0.string.erase(to_eliminate);
485  pauli1.string.erase(to_eliminate);
486 
487  circ.add_op<Qubit>(OpType::CX, {to_eliminate, target});
488  } else {
489  // use XXPhase3
490  Qubit to_eliminate1 = *iter;
491  match.erase(iter++);
492  pauli0.string.erase(to_eliminate1);
493  pauli1.string.erase(to_eliminate1);
494 
495  Qubit to_eliminate2 = *iter;
496  match.erase(iter);
497  pauli0.string.erase(to_eliminate2);
498  pauli1.string.erase(to_eliminate2);
499 
500  circ.add_op<Qubit>(OpType::H, {to_eliminate1});
501  circ.add_op<Qubit>(OpType::H, {to_eliminate2});
502  circ.add_op<Qubit>(
503  OpType::XXPhase3, 0.5, {to_eliminate1, to_eliminate2, target});
504  circ.add_op<Qubit>(OpType::X, {target});
505  }
506  }

509 std::pair<Circuit, std::optional<Qubit>> reduce_overlap_of_paulis(
510  SpPauliStabiliser &pauli0, SpPauliStabiliser &pauli1,
511  CXConfigType cx_config) {
512  /*
513  * Cowtan, Dilkes, Duncan, Simmons, Sivarajah: Phase Gadget Synthesis for
514  * Shallow Circuits, Lemma 4.9
515  * Let s and t be Pauli strings; then there exists a Clifford unitary U such
516  * that
517  * P(a, s) . P(b, t) = U . P(a, s') . P(b, t') . U^\dagger
518  * where s' and t' are Pauli strings with intersection at most 1.
519  *
520  * Follows the procedure to reduce the intersection of the gadgets and then
521  * synthesises the remainder individually.
522  */
523 
524  /*
525  * Step 1: Identify qubits in both pauli0 and pauli1 which either match or
526  * don't
527  */
528  std::set<Qubit> match = pauli0.common_qubits(pauli1);
529  std::set<Qubit> mismatch = pauli0.conflicting_qubits(pauli1);
530 
531  /*
532  * Step 2: Build the unitary U that minimises the intersection of the gadgets.
533  */
534  Circuit u;
535  for (const std::pair<const Qubit, Pauli> &qp : pauli0.string)
536  u.add_qubit(qp.first);
537  for (const std::pair<const Qubit, Pauli> &qp : pauli1.string) {
538  if (!u.contains_unit(qp.first)) u.add_qubit(qp.first);
539  }
540 
541  /*
542  * Step 2.i: Remove (almost) all matches by converting to Z basis and applying
543  * CXs
544  */
545  for (const Qubit &qb : match) {
546  switch (pauli0.get(qb)) {
547  case Pauli::X:
548  u.add_op<Qubit>(OpType::H, {qb});
549  pauli0.set(qb, Pauli::Z);
550  pauli1.set(qb, Pauli::Z);
551  break;
552  case Pauli::Y:
553  u.add_op<Qubit>(OpType::V, {qb});
554  pauli0.set(qb, Pauli::Z);
555  pauli1.set(qb, Pauli::Z);
556  break;
557  default:
558  break;
559  }
560  }
561  switch (cx_config) {
562  case CXConfigType::Snake: {
563  reduce_shared_qs_by_CX_snake(u, match, pauli0, pauli1);
564  break;
565  }
566  case CXConfigType::Star: {
567  reduce_shared_qs_by_CX_star(u, match, pauli0, pauli1);
568  break;
569  }
570  case CXConfigType::Tree: {
571  reduce_shared_qs_by_CX_tree(u, match, pauli0, pauli1);
572  break;
573  }
574  case CXConfigType::MultiQGate: {
575  reduce_shared_qs_by_CX_multiqgate(u, match, pauli0, pauli1);
576  break;
577  }
578  default:
579  throw std::logic_error(
580  "Unknown CXConfigType received when decomposing gadget.");
581  }
582  /*
583  * Step 2.ii: Convert mismatches to Z in pauli0 and X in pauli1
584  */
585  for (const Qubit &qb : mismatch) {
586  switch (pauli0.get(qb)) {
587  case Pauli::X: {
588  switch (pauli1.get(qb)) {
589  case Pauli::Y:
590  u.add_op<Qubit>(OpType::Sdg, {qb});
591  u.add_op<Qubit>(OpType::Vdg, {qb});
592  break;
593  case Pauli::Z:
594  u.add_op<Qubit>(OpType::H, {qb});
595  break;
596  default:
597  TKET_ASSERT(false);
598  break; // Cannot hit this case
599  }
600  break;
601  }
602  case Pauli::Y: {
603  switch (pauli1.get(qb)) {
604  case Pauli::X:
605  u.add_op<Qubit>(OpType::V, {qb});
606  break;
607  case Pauli::Z:
608  u.add_op<Qubit>(OpType::V, {qb});
609  u.add_op<Qubit>(OpType::S, {qb});
610  break;
611  default:
612  TKET_ASSERT(false);
613  break; // Cannot hit this case
614  }
615  break;
616  }
617  default: { // Necessarily Z
618  if (pauli1.get(qb) == Pauli::Y) u.add_op<Qubit>(OpType::Sdg, {qb});
619  // No need to act if already X
620  }
621  }
622  pauli0.set(qb, Pauli::Z);
623  pauli1.set(qb, Pauli::X);
624  }
625 
626  /*
627  * Step 2.iii: Remove the final matching qubit against a mismatch if one
628  * exists, otherwise remove into another qubit
629  */
630  if (!match.empty()) {
631  Qubit last_match = *match.begin();
632  if (!mismatch.empty()) {
633  Qubit mismatch_used =
634  *mismatch.rbegin(); // Prefer to use the one that may be left over
635  // after reducing pairs
636  u.add_op<Qubit>(OpType::S, {mismatch_used});
637  u.add_op<Qubit>(OpType::CX, {last_match, mismatch_used});
638  u.add_op<Qubit>(OpType::Sdg, {mismatch_used});
639  pauli0.string.erase(last_match);
640  pauli1.string.erase(last_match);
641  } else {
642  std::optional<std::pair<Qubit, Pauli>> other;
643  for (const std::pair<const Qubit, Pauli> &qp : pauli0.string) {
644  if (qp.first != last_match && qp.second != Pauli::I) {
645  other = qp;
646  pauli0.string.erase(last_match);
647  break;
648  }
649  }
650  if (!other) {
651  for (const std::pair<const Qubit, Pauli> &qp : pauli1.string) {
652  if (qp.first != last_match && qp.second != Pauli::I) {
653  other = qp;
654  pauli1.string.erase(last_match);
655  break;
656  }
657  }
658  if (!other)
659  throw std::logic_error(
660  "Cannot reduce identical Paulis to different qubits");
661  }
662  if (other->second == Pauli::X) {
663  u.add_op<Qubit>(OpType::H, {other->first});
664  u.add_op<Qubit>(OpType::CX, {last_match, other->first});
665  u.add_op<Qubit>(OpType::H, {other->first});
666  } else {
667  u.add_op<Qubit>(OpType::CX, {last_match, other->first});
668  }
669  }
670  }
671 
672  /*
673  * Step 2.iv: Reduce pairs of mismatches to different qubits.
674  * Allow both gadgets to build a remaining qubit if it exists.
675  */
676  std::optional<Qubit> last_mismatch = std::nullopt;
677  std::set<Qubit>::iterator mis_it = mismatch.begin();
678  while (mis_it != mismatch.end()) {
679  Qubit z_in_0 = *mis_it;
680  mis_it++;
681  if (mis_it != mismatch.end()) {
682  Qubit x_in_1 = *mis_it;
683  u.add_op<Qubit>(OpType::CX, {x_in_1, z_in_0});
684  pauli0.string.erase(x_in_1);
685  pauli1.string.erase(z_in_0);
686  mis_it++;
687  } else {
688  last_mismatch = z_in_0;
689  }
690  }
691 
692  return {u, last_mismatch};
693 }

695 std::pair<Circuit, Qubit> reduce_anticommuting_paulis_to_z_x(
696  SpPauliStabiliser pauli0, SpPauliStabiliser pauli1,
697  CXConfigType cx_config) {
698  std::pair<Circuit, std::optional<Qubit>> reduced_overlap =
699  reduce_overlap_of_paulis(pauli0, pauli1, cx_config);
700  Circuit &u = reduced_overlap.first;
701  if (!reduced_overlap.second)
702  throw std::logic_error("No overlap for anti-commuting paulis");
703  Qubit &last_mismatch = *reduced_overlap.second;
704 
711  pauli0.string.erase(last_mismatch);
712  pauli0.compress();
713  if (!pauli0.string.empty()) {
714  std::pair<Circuit, Qubit> diag0 = reduce_pauli_to_z(pauli0, cx_config);
715  u.append(diag0.first);
716  u.add_op<Qubit>(OpType::CX, {diag0.second, last_mismatch});
717  }
718  pauli1.compress();
719  pauli1.string.erase(last_mismatch);
720  if (!pauli1.string.empty()) {
721  std::pair<Circuit, Qubit> diag1 = reduce_pauli_to_z(pauli1, cx_config);
722  u.append(diag1.first);
723  u.add_op<Qubit>(OpType::H, {last_mismatch});
724  u.add_op<Qubit>(OpType::CX, {diag1.second, last_mismatch});
725  u.add_op<Qubit>(OpType::H, {last_mismatch});
726  }
727 
728  return {u, last_mismatch};
729 }
