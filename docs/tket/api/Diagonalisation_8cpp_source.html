24void check_easy_diagonalise(
25 std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> &qubits,
26 Circuit &circ) {
27 Conjugations conjugations;
28 std::set<Qubit>::iterator qb_iter = qubits.begin();
29 for (std::set<Qubit>::iterator next = qb_iter; qb_iter != qubits.end();
30 qb_iter = next) {
31 ++next;
32 Pauli p1 = Pauli::I;
33 bool remove_qb = true;
34 for (const SpSymPauliTensor &gadget : gadgets) {
35 Pauli p2 = gadget.get(*qb_iter);
36 if (p2 == Pauli::I) continue;
37 if (p1 == Pauli::I) {
38 p1 = p2;
39 } else if (p1 != p2) {
40 remove_qb = false;
41 break;
42 }
43 }
44 if (remove_qb) {
45 switch (p1) {
46 case Pauli::I:
47 break;
48 case Pauli::X:
49 conjugations.push_back({OpType::H, {*qb_iter}});
50 circ.add_op<Qubit>(OpType::H, {*qb_iter});
51 break;
52 case Pauli::Y:
53 conjugations.push_back({OpType::Vdg, {*qb_iter}});
54 circ.add_op<Qubit>(OpType::V, {*qb_iter});
55 break;
56 case Pauli::Z:
57 break;
58 default:
59 throw std::logic_error(
60 "Unknown Pauli encountered in checking diagonalisation");
61 }
62 qubits.erase(qb_iter);
63 }
64 }
65 for (SpSymPauliTensor &gadget : gadgets) {
66 apply_conjugations(gadget, conjugations);
67 }
68}

70std::optional<std::pair<Pauli, Pauli>> check_pair_compatibility(
71 const Qubit &qb1, const Qubit &qb2,
72 const std::list<SpSymPauliTensor> &gadgets) {
73 if (qb1 == qb2) return std::nullopt;
74
75 /* Do exhaustive search for a Pauli A and Pauli B that
76 satisfy Theorem */
77 std::list<Pauli> paulis{Pauli::Z, Pauli::X, Pauli::Y};
78 for (Pauli pauli1 : paulis) {
79 for (Pauli pauli2 : paulis) {
80 bool found_pair = true;
81 for (const SpSymPauliTensor &gadget : gadgets) {
82 Pauli inner_p_1 = gadget.get(qb1);
83 Pauli inner_p_2 = gadget.get(qb2);
84
85 if (inner_p_1 == Pauli::I || inner_p_1 == pauli1) {
86 if (!(inner_p_2 == Pauli::I || inner_p_2 == pauli2)) {
87 found_pair = false;
88 break;
89 }
90 }
91 if (inner_p_2 == Pauli::I || inner_p_2 == pauli2) {
92 if (!(inner_p_1 == Pauli::I || inner_p_1 == pauli1)) {
93 found_pair = false;
94 break;
95 }
96 }
97 }
98 if (found_pair) {
99 return std::make_pair(pauli1, pauli2);
100 }
101 }
102 }
103 return std::nullopt;
104}

106void greedy_diagonalise(
107 const std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> &qubits,
108 Conjugations &conjugations, Circuit &circ, CXConfigType cx_config) {
109 unsigned total_counter = UINT_MAX;
110 QubitPauliMap to_diag;
111 for (const SpSymPauliTensor &gadget : gadgets) {
112 unsigned support_counter = 0;
113 QubitPauliMap to_diag_candidates;
114 for (const Qubit &qb : qubits) {
115 Pauli p = gadget.get(qb);
116 if (p != Pauli::I) {
117 ++support_counter;
118 to_diag_candidates.insert({qb, p});
119 }
120 }
121 if (support_counter < total_counter && support_counter > 1) {
122 total_counter = support_counter;
123 to_diag = to_diag_candidates;
124 }
125 }
126 if (to_diag.empty()) {
127 throw std::logic_error("Brute Force Diagonalise can't find a candidate!");
128 }
129 for (const std::pair<const Qubit, Pauli> &qp : to_diag) {
130 const Qubit &qb = qp.first;
131 switch (qp.second) {
132 case Pauli::X: {
133 conjugations.push_back({OpType::H, {qb}});
134 circ.add_op<Qubit>(OpType::H, {qb});
135 break;
136 }
137 case Pauli::Y: {
138 conjugations.push_back({OpType::Vdg, {qb}});
139 circ.add_op<Qubit>(OpType::V, {qb});
140 break;
141 }
142 case Pauli::Z: {
143 break;
144 }
145 case Pauli::I:
146 default:
147 throw std::logic_error("Unknown Pauli in greedy diagonalisation.");
148 }
149 }
150 qubit_vector_t diag_qubits;
151 for (const auto &[k, v] : to_diag) diag_qubits.push_back(k);
152 unsigned n_qubits = diag_qubits.size();
153 Qubit first_qb = diag_qubits[0];
154
155 switch (cx_config) {
156 case CXConfigType::Snake: {
157 for (unsigned i = n_qubits - 1; i > 0; --i) {
158 Qubit qb = diag_qubits[i], before = diag_qubits[i - 1];
159 conjugations.push_back({OpType::CX, {qb, before}});
160 circ.add_op<Qubit>(OpType::CX, {qb, before});
161 }
162 break;
163 }
164 case CXConfigType::Star: {
165 for (unsigned i = 1; i < n_qubits; ++i) {
166 Qubit qb = diag_qubits[i];
167 conjugations.push_back({OpType::CX, {qb, first_qb}});
168 circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
169 }
170 break;
171 }
172 case CXConfigType::Tree: {
173 unsigned complete_layers = floor(log2(n_qubits));
174 unsigned dense_end = pow(2, complete_layers);
175 for (unsigned j = 0; j < n_qubits - dense_end; j++) {
176 circ.add_op<Qubit>(
177 OpType::CX,
178 {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]});
179 conjugations.push_back(
180 {OpType::CX,
181 {diag_qubits[dense_end + j], diag_qubits[dense_end - 1 - j]}});
182 }
183 for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
184 for (unsigned j = 0; j < dense_end; j += 2 * step_size) {
185 circ.add_op<Qubit>(
186 OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]});
187 conjugations.push_back(
188 {OpType::CX, {diag_qubits[j + step_size], diag_qubits[j]}});
189 }
190 }
191 break;
192 }
193 case CXConfigType::MultiQGate: {
194 int sign_correction = 1;
195 for (int q = n_qubits - 1; q > 0; q -= 2) {
196 Qubit qb = diag_qubits[q];
197 if (q - 1 > 0) {
198 Qubit before = diag_qubits[q - 1];
199 circ.add_op<Qubit>(OpType::H, {qb});
200 circ.add_op<Qubit>(OpType::H, {before});
201 circ.add_op<Qubit>(OpType::XXPhase3, 0.5, {qb, before, first_qb});
202 conjugations.push_back({OpType::H, {qb}});
203 conjugations.push_back({OpType::H, {before}});
204 conjugations.push_back({OpType::XXPhase3, {qb, before, first_qb}});
205 sign_correction *= -1;
206 } else {
207 circ.add_op<Qubit>(OpType::CX, {qb, first_qb});
208 conjugations.push_back({OpType::CX, {qb, first_qb}});
209 }
210 }
211 if (sign_correction < 0) {
212 circ.add_op<Qubit>(OpType::X, {first_qb});
213 conjugations.push_back({OpType::X, {first_qb}});
214 }
215 break;
216 }
217 default:
218 throw std::logic_error("Unknown CXConfigType in greedy diagonalisation.");
219 }
220 qubits.erase(first_qb);
221}

223/* Diagonalise a set of Pauli Gadgets simultaneously using Cliffords*/
224Circuit mutual_diagonalise(
225 std::list<SpSymPauliTensor> &gadgets, std::set<Qubit> qubits,
226 CXConfigType cx_config) {
227 Circuit cliff_circ;
228 for (const Qubit &qb : qubits) {
229 cliff_circ.add_qubit(qb);
230 }
231 check_easy_diagonalise(gadgets, qubits, cliff_circ);
232 while (!qubits.empty()) {
233 Conjugations conjugations;
234 Qubit qb_a;
235 Qubit qb_b;
236 std::pair<Pauli, Pauli> pauli_pair;
237 bool found_match = false;
238 /* First, try to find some qubits we can reduce using only 1 CX */
239 for (const Qubit &qb1 : qubits) {
240 for (const Qubit &qb2 : qubits) {
241 std::optional<std::pair<Pauli, Pauli>> compatible =
242 check_pair_compatibility(qb1, qb2, gadgets);
243 if (compatible.has_value()) {
244 pauli_pair = *compatible;
245 qb_a = qb1;
246 qb_b = qb2;
247 found_match = true;
248 break;
249 }
250 }
251 if (found_match) break;
252 }
253 if (found_match) {
254 Pauli p1 = pauli_pair.first;
255 Pauli p2 = pauli_pair.second;
256 switch (p1) {
257 case Pauli::X: {
258 conjugations.push_back({OpType::H, {qb_a}});
259 cliff_circ.add_op<Qubit>(OpType::H, {qb_a});
260 break;
261 }
262 case Pauli::Y: {
263 conjugations.push_back({OpType::Vdg, {qb_a}});
264 cliff_circ.add_op<Qubit>(OpType::V, {qb_a});
265 break;
266 }
267 case Pauli::Z: {
268 break;
269 }
270 case Pauli::I:
271 default:
272 throw std::logic_error("Unknown Pauli in mutual diagonalisation.");
273 }
274 switch (p2) {
275 case Pauli::X: {
276 conjugations.push_back({OpType::H, {qb_b}});
277 cliff_circ.add_op<Qubit>(OpType::H, {qb_b});
278 break;
279 }
280 case Pauli::Y: {
281 conjugations.push_back({OpType::Vdg, {qb_b}});
282 cliff_circ.add_op<Qubit>(OpType::V, {qb_b});
283 break;
284 }
285 case Pauli::Z: {
286 break;
287 }
288 case Pauli::I:
289 default:
290 throw std::logic_error("Unknown Pauli in mutual diagonalisation.");
291 }
292 conjugations.push_back({OpType::CX, {qb_a, qb_b}});
293 cliff_circ.add_op<Qubit>(OpType::CX, {qb_a, qb_b});
294 qubits.erase(qb_b);
295 }
296 /* If we can't, do it with `n-1` CXs, where `n` := no. of undiagonalised
297 * qubits */
298 if (!found_match) {
299 greedy_diagonalise(gadgets, qubits, conjugations, cliff_circ, cx_config);
300 }
301 for (SpSymPauliTensor &gadget : gadgets) {
302 apply_conjugations(gadget, conjugations);
303 }
304 // we may have made some easy-to-remove qubits
305 check_easy_diagonalise(gadgets, qubits, cliff_circ);
306 }
307 return cliff_circ;
308}

310void apply_conjugations(
311 SpSymPauliTensor &qps, const Conjugations &conjugations) {
312 SpPauliStabiliser stab(qps.string);
313 for (const auto &optype_qubit_pair : conjugations) {
314 OpType ot = optype_qubit_pair.first;
315 const qubit_vector_t &qbs = optype_qubit_pair.second;
316 if (!optypeinfo().at(ot).signature ||
317 optypeinfo().at(ot).signature->size() != qbs.size())
318 throw std::logic_error("Incompatible qubit count for conjugations");
319 switch (ot) {
320 case OpType::H:
321 case OpType::S:
322 case OpType::Sdg:
323 case OpType::V:
324 case OpType::Vdg:
325 case OpType::X:
326 case OpType::Z:
327 conjugate_PauliTensor(stab, ot, qbs[0]);
328 break;
329 case OpType::CX:
330 conjugate_PauliTensor(stab, ot, qbs[0], qbs[1]);
331 break;
332 case OpType::XXPhase3:
333 conjugate_PauliTensor(stab, ot, qbs[0], qbs[1], qbs[2]);
334 break;
335 default:
336 throw std::logic_error(
337 "Unknown OpType received when applying conjugations.");
338 }
339 }
340 qps.string = stab.string;
341 qps.coeff *= cast_coeff<quarter_turns_t, Expr>(stab.coeff);
342}

344std::pair<Circuit, Qubit> reduce_pauli_to_z(
345 const SpPauliStabiliser &pauli, CXConfigType cx_config) {
346 Circuit circ;
347 qubit_vector_t qubits;
348 for (const std::pair<const Qubit, Pauli> &qp : pauli.string) {
349 circ.add_qubit(qp.first);
350 if (qp.second != Pauli::I) qubits.push_back(qp.first);
351 switch (qp.second) {
352 case Pauli::X: {
353 circ.add_op<Qubit>(OpType::H, {qp.first});
354 break;
355 }
356 case Pauli::Y: {
357 circ.add_op<Qubit>(OpType::V, {qp.first});
358 break;
359 }
360 default: {
361 break;
362 }
363 }
364 }
365 unsigned n_qubits = qubits.size();
366 if (n_qubits == 0) throw std::logic_error("Cannot reduce identity to Z");
367 switch (cx_config) {
368 case CXConfigType::Snake: {
369 for (unsigned i = n_qubits - 1; i != 0; --i) {
370 circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.at(i - 1)});
371 }
372 break;
373 }
374 case CXConfigType::Star: {
375 for (unsigned i = n_qubits - 1; i != 0; --i) {
376 circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.front()});
377 }
378 break;
379 }
380 case CXConfigType::Tree: {
381 for (unsigned step_size = 1; step_size < n_qubits; step_size *= 2) {
382 for (unsigned i = 0; step_size + i < n_qubits; i += 2 * step_size) {
383 circ.add_op<Qubit>(
384 OpType::CX, {qubits.at(step_size + i), qubits.at(i)});
385 }
386 }
387 break;
388 }
389 case CXConfigType::MultiQGate: {
390 bool flip_phase = false;
391 for (unsigned i = n_qubits - 1; i != 0; --i) {
392 if (i == 1) {
393 circ.add_op<Qubit>(OpType::CX, {qubits.at(i), qubits.front()});
394 } else {
399 circ.add_op<Qubit>(OpType::H, {qubits.at(i)});
400 circ.add_op<Qubit>(OpType::H, {qubits.at(i - 1)});
401 circ.add_op<Qubit>(
402 OpType::XXPhase3, 0.5,
403 {qubits.at(i), qubits.at(i - 1), qubits.front()});
404 --i;
405 flip_phase = !flip_phase;
406 }
407 }
408 if (flip_phase) circ.add_op<Qubit>(OpType::X, {qubits.front()});
409 break;
410 }
411 }
412 return {circ, qubits.front()};
413}

415static void reduce_shared_qs_by_CX_snake(
416 Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
417 SpPauliStabiliser &pauli1) {
418 unsigned match_size = match.size();
419 while (match_size > 1) { // We allow one match left over
420 auto it = --match.end();
421 Qubit to_eliminate = *it;
422 match.erase(it);
423 Qubit helper = *match.rbegin();
424 // extend CX snake
425 circ.add_op<Qubit>(OpType::CX, {to_eliminate, helper});
426 pauli0.string.erase(to_eliminate);
427 pauli1.string.erase(to_eliminate);
428 match_size--;
429 }
430}

432static void reduce_shared_qs_by_CX_star(
433 Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
434 SpPauliStabiliser &pauli1) {
435 std::set<Qubit>::iterator iter = match.begin();
436 for (std::set<Qubit>::iterator next = match.begin(); match.size() > 1;
437 iter = next) {
438 ++next;
439 Qubit to_eliminate = *iter;
440 circ.add_op<Qubit>(OpType::CX, {to_eliminate, *match.rbegin()});
441 pauli0.string.erase(to_eliminate);
442 pauli1.string.erase(to_eliminate);
443 match.erase(iter);
444 }
445}

447static void reduce_shared_qs_by_CX_tree(
448 Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
449 SpPauliStabiliser &pauli1) {
450 while (match.size() > 1) {
451 std::set<Qubit> remaining;
452 std::set<Qubit>::iterator it = match.begin();
453 while (it != match.end()) {
454 Qubit maintained = *it;
455 it++;
456 remaining.insert(maintained);
457 if (it != match.end()) {
458 Qubit to_eliminate = *it;
459 it++;
460 circ.add_op<Qubit>(OpType::CX, {to_eliminate, maintained});
461 pauli0.string.erase(to_eliminate);
462 pauli1.string.erase(to_eliminate);
463 }
464 }
465 match = remaining;
466 }

469static void reduce_shared_qs_by_CX_multiqgate(
470 Circuit &circ, std::set<Qubit> &match, SpPauliStabiliser &pauli0,
471 SpPauliStabiliser &pauli1) {
472 if (match.size() <= 1) {
473 return;
474 }
475 // last qubit is target
476 Qubit target = *match.rbegin();
477 while (match.size() > 1) {
478 std::set<Qubit>::iterator iter = match.begin();
479 if (match.size() == 2) {
480 // use CX
481 Qubit to_eliminate = *iter;
482 match.erase(iter);
483 pauli0.string.erase(to_eliminate);
484 pauli1.string.erase(to_eliminate);
485
486 circ.add_op<Qubit>(OpType::CX, {to_eliminate, target});
487 } else {
488 // use XXPhase3
489 Qubit to_eliminate1 = *iter;
490 match.erase(iter++);
491 pauli0.string.erase(to_eliminate1);
492 pauli1.string.erase(to_eliminate1);
493
494 Qubit to_eliminate2 = *iter;
495 match.erase(iter);
496 pauli0.string.erase(to_eliminate2);
497 pauli1.string.erase(to_eliminate2);
498
499 circ.add_op<Qubit>(OpType::H, {to_eliminate1});
500 circ.add_op<Qubit>(OpType::H, {to_eliminate2});
501 circ.add_op<Qubit>(
502 OpType::XXPhase3, 0.5, {to_eliminate1, to_eliminate2, target});
503 circ.add_op<Qubit>(OpType::X, {target});
504 }
505 }

508std::pair<Circuit, std::optional<Qubit>> reduce_overlap_of_paulis(
509 SpPauliStabiliser &pauli0, SpPauliStabiliser &pauli1,
510 CXConfigType cx_config, bool allow_matching_final) {
511 /*
512 * Cowtan, Dilkes, Duncan, Simmons, Sivarajah: Phase Gadget Synthesis for
513 * Shallow Circuits, Lemma 4.9
514 * Let s and t be Pauli strings; then there exists a Clifford unitary U such
515 * that
516 * P(a, s) . P(b, t) = U . P(a, s') . P(b, t') . U^\dagger
517 * where s' and t' are Pauli strings with intersection at most 1.
518 *
519 * Follows the procedure to reduce the intersection of the gadgets and then
520 * synthesises the remainder individually.
521 */
522
523 /*
524 * Step 1: Identify qubits in both pauli0 and pauli1 which either match or
525 * don't
526 */
527 std::set<Qubit> match = pauli0.common_qubits(pauli1);
528 std::set<Qubit> mismatch = pauli0.conflicting_qubits(pauli1);
529
530 /*
531 * Step 2: Build the unitary U that minimises the intersection of the gadgets.
532 */
533 Circuit u;
534 for (const std::pair<const Qubit, Pauli> &qp : pauli0.string)
535 u.add_qubit(qp.first);
536 for (const std::pair<const Qubit, Pauli> &qp : pauli1.string) {
537 if (!u.contains_unit(qp.first)) u.add_qubit(qp.first);
538 }
539
540 /*
541 * Step 2.i: Remove (almost) all matches by converting to Z basis and applying
542 * CXs
543 */
544 for (const Qubit &qb : match) {
545 switch (pauli0.get(qb)) {
546 case Pauli::X:
547 u.add_op<Qubit>(OpType::H, {qb});
548 pauli0.set(qb, Pauli::Z);
549 pauli1.set(qb, Pauli::Z);
550 break;
551 case Pauli::Y:
552 u.add_op<Qubit>(OpType::V, {qb});
553 pauli0.set(qb, Pauli::Z);
554 pauli1.set(qb, Pauli::Z);
555 break;
556 default:
557 break;
558 }
559 }
560 switch (cx_config) {
561 case CXConfigType::Snake: {
562 reduce_shared_qs_by_CX_snake(u, match, pauli0, pauli1);
563 break;
564 }
565 case CXConfigType::Star: {
566 reduce_shared_qs_by_CX_star(u, match, pauli0, pauli1);
567 break;
568 }
569 case CXConfigType::Tree: {
570 reduce_shared_qs_by_CX_tree(u, match, pauli0, pauli1);
571 break;
572 }
573 case CXConfigType::MultiQGate: {
574 reduce_shared_qs_by_CX_multiqgate(u, match, pauli0, pauli1);
575 break;
576 }
577 default:
578 throw std::logic_error(
579 "Unknown CXConfigType received when decomposing gadget.");
580 }
581 /*
582 * Step 2.ii: Convert mismatches to Z in pauli0 and X in pauli1
583 */
584 for (const Qubit &qb : mismatch) {
585 switch (pauli0.get(qb)) {
586 case Pauli::X: {
587 switch (pauli1.get(qb)) {
588 case Pauli::Y:
589 u.add_op<Qubit>(OpType::Sdg, {qb});
590 u.add_op<Qubit>(OpType::Vdg, {qb});
591 break;
592 case Pauli::Z:
593 u.add_op<Qubit>(OpType::H, {qb});
594 break;
595 default:
596 TKET_ASSERT(false);
597 break; // Cannot hit this case
598 }
599 break;
600 }
601 case Pauli::Y: {
602 switch (pauli1.get(qb)) {
603 case Pauli::X:
604 u.add_op<Qubit>(OpType::V, {qb});
605 break;
606 case Pauli::Z:
607 u.add_op<Qubit>(OpType::V, {qb});
608 u.add_op<Qubit>(OpType::S, {qb});
609 break;
610 default:
611 TKET_ASSERT(false);
612 break; // Cannot hit this case
613 }
614 break;
615 }
616 default: { // Necessarily Z
617 if (pauli1.get(qb) == Pauli::Y) u.add_op<Qubit>(OpType::Sdg, {qb});
618 // No need to act if already X
619 }
620 }
621 pauli0.set(qb, Pauli::Z);
622 pauli1.set(qb, Pauli::X);
623 }
624
625 /*
626 * Step 2.iii: Remove the final matching qubit against a mismatch if one
627 * exists, otherwise remove into another qubit or allow final overlap to be
628 * matching
629 */
630 std::optional<Qubit> last_overlap = std::nullopt;
631 if (!match.empty()) {
632 Qubit last_match = *match.begin();
633 if (!mismatch.empty()) {
634 Qubit mismatch_used =
635 *mismatch.rbegin(); // Prefer to use the one that may be left over
636 // after reducing pairs
637 u.add_op<Qubit>(OpType::S, {mismatch_used});
638 u.add_op<Qubit>(OpType::CX, {last_match, mismatch_used});
639 u.add_op<Qubit>(OpType::Sdg, {mismatch_used});
640 pauli0.string.erase(last_match);
641 pauli1.string.erase(last_match);
642 } else if (!allow_matching_final) {
643 std::optional<std::pair<Qubit, Pauli>> other;
644 for (const std::pair<const Qubit, Pauli> &qp : pauli0.string) {
645 if (qp.first != last_match && qp.second != Pauli::I) {
646 other = qp;
647 pauli0.string.erase(last_match);
648 break;
649 }
650 }
651 if (!other) {
652 for (const std::pair<const Qubit, Pauli> &qp : pauli1.string) {
653 if (qp.first != last_match && qp.second != Pauli::I) {
654 other = qp;
655 pauli1.string.erase(last_match);
656 break;
657 }
658 }
659 if (!other)
660 throw std::logic_error(
661 "Cannot reduce identical Paulis to different qubits");
662 }
663 if (other->second == Pauli::X) {
664 u.add_op<Qubit>(OpType::H, {other->first});
665 u.add_op<Qubit>(OpType::CX, {last_match, other->first});
666 u.add_op<Qubit>(OpType::H, {other->first});
667 } else {
668 u.add_op<Qubit>(OpType::CX, {last_match, other->first});
669 }
670 } else {
671 last_overlap = last_match;
672 }
673 }
674
675 /*
676 * Step 2.iv: Reduce pairs of mismatches to different qubits.
677 * Allow both gadgets to build a remaining qubit if it exists.
678 */
679 std::set<Qubit>::iterator mis_it = mismatch.begin();
680 while (mis_it != mismatch.end()) {
681 Qubit z_in_0 = *mis_it;
682 mis_it++;
683 if (mis_it != mismatch.end()) {
684 Qubit x_in_1 = *mis_it;
685 u.add_op<Qubit>(OpType::CX, {x_in_1, z_in_0});
686 pauli0.string.erase(x_in_1);
687 pauli1.string.erase(z_in_0);
688 mis_it++;
689 } else {
690 last_overlap = z_in_0;
691 }
692 }
693
694 return {u, last_overlap};
695}

697std::pair<Circuit, Qubit> reduce_anticommuting_paulis_to_z_x(
698 SpPauliStabiliser pauli0, SpPauliStabiliser pauli1,
699 CXConfigType cx_config) {
700 std::pair<Circuit, std::optional<Qubit>> reduced_overlap =
701 reduce_overlap_of_paulis(pauli0, pauli1, cx_config);
702 Circuit &u = reduced_overlap.first;
703 if (!reduced_overlap.second)
704 throw std::logic_error("No overlap for anti-commuting paulis");
705 Qubit &last_mismatch = *reduced_overlap.second;
706
713 pauli0.string.erase(last_mismatch);
714 pauli0.compress();
715 if (!pauli0.string.empty()) {
716 std::pair<Circuit, Qubit> diag0 = reduce_pauli_to_z(pauli0, cx_config);
717 u.append(diag0.first);
718 u.add_op<Qubit>(OpType::CX, {diag0.second, last_mismatch});
719 }
720 pauli1.compress();
721 pauli1.string.erase(last_mismatch);
722 if (!pauli1.string.empty()) {
723 std::pair<Circuit, Qubit> diag1 = reduce_pauli_to_z(pauli1, cx_config);
724 u.append(diag1.first);
725 u.add_op<Qubit>(OpType::H, {last_mismatch});
726 u.add_op<Qubit>(OpType::CX, {diag1.second, last_mismatch});
727 u.add_op<Qubit>(OpType::H, {last_mismatch});
728 }
729
730 return {u, last_mismatch};
731}

733std::tuple<Circuit, Qubit, Qubit> reduce_commuting_paulis_to_zi_iz(
734 SpPauliStabiliser pauli0, SpPauliStabiliser pauli1,
735 CXConfigType cx_config) {
736 std::pair<Circuit, std::optional<Qubit>> reduced_overlap =
737 reduce_overlap_of_paulis(pauli0, pauli1, cx_config);
738 Circuit &u = reduced_overlap.first;
739 if (reduced_overlap.second)
740 throw std::logic_error("Overlap remaining for commuting paulis");
741
745 std::pair<Circuit, Qubit> diag0 = reduce_pauli_to_z(pauli0, cx_config);
746 u.append(diag0.first);
747 std::pair<Circuit, Qubit> diag1 = reduce_pauli_to_z(pauli1, cx_config);
748 u.append(diag1.first);
749
750 return {u, diag0.second, diag1.second};
751}
