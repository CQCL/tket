49template <typename T>
50class DirectedGraphBase : public AbstractGraph<T> {
51 protected:
52 using ConnGraph = boost::adjacency_list<
53 boost::vecS, boost::vecS, boost::bidirectionalS, T, WeightedEdge>;
54 using UndirectedConnGraph = boost::adjacency_list<
55 boost::setS, boost::vecS, boost::undirectedS, T, WeightedEdge>;

72 explicit DirectedGraphBase(const std::vector<T>& nodes)
73 : AbstractGraph<T>(nodes), graph() {

80 explicit DirectedGraphBase(const std::vector<Connection>& edges) : graph() {
81 for (auto [node1, node2] : edges) {
82 if (!node_exists(node1)) {
83 add_node(node1);
84 }
85 if (!node_exists(node2)) {
86 add_node(node2);
87 }
88 if (node1 == node2) {
89 throw std::invalid_argument(
90 "An edge can not be added from a node to itself.");
91 }
92 add_connection(node1, node2);
93 }
94 }

97 bool edge_exists(const T& node1, const T& node2) const override {
98 if (!node_exists(node1) || !node_exists(node2)) {
99 throw NodeDoesNotExistError(
100 "The nodes passed to DirectedGraph::edge_exists must exist");
101 }
102 auto [_, exists] =
103 boost::edge(to_vertices(node1), to_vertices(node2), graph);
104 return exists;
105 }
106
108 void add_node(const T& node) {
109 nodes_.insert(node);
110 Vertex v = boost::add_vertex(node, graph);
111 node_to_vertex.left.insert({node, v});
112 }
113
115 void remove_node(const T& node) {
116 if (!node_exists(node)) {
117 throw NodeDoesNotExistError(
118 "The node passed to DirectedGraph::remove_node must exist!");
119 }
120 nodes_.erase(node);
121 Vertex v = to_vertices(node);
122 boost::clear_vertex(v, graph);
123 utils::remove_vertex_with_map(v, graph, node_to_vertex.right);
124 }
125
127 void add_connection(const T& node1, const T& node2, unsigned weight = 1) {
128 if (!node_exists(node1) || !node_exists(node2)) {
129 throw NodeDoesNotExistError(
130 "The nodes passed to DirectedGraph::add_connection must exist");
131 }
132 if (node1 == node2) {
133 throw std::invalid_argument(
134 "A connection can not be added between a node to itself.");
135 }
136 boost::add_edge(
137 to_vertices(node1), to_vertices(node2), WeightedEdge(weight), graph);
138 }
139
141 void remove_connection(const Connection& edge) {
142 if (!node_exists(edge.first) || !node_exists(edge.second)) {
143 throw NodeDoesNotExistError(
144 "Trying to remove an edge with non-existent vertices");
145 }
146 auto [e, exists] =
147 boost::edge(to_vertices(edge.first), to_vertices(edge.second), graph);
148 if (!exists) {
149 throw EdgeDoesNotExistError(
150 "The edge (" + edge.first.repr() + ", " + edge.second.repr() +
151 ") cannot be removed as it does not exist");
152 }
153 utils::remove_edge_with_map(e, graph, node_to_vertex.right);
154 }
155
157 void remove_connection(const T& node1, const T& node2) {
158 remove_connection({node1, node2});
159 }

166 unsigned get_connection_weight(const T& node1, const T& node2) const {
167 if (!node_exists(node1) || !node_exists(node2)) {
168 throw NodeDoesNotExistError(
169 "Trying to retrieve edge weight from non-existent vertices");
170 }
171 auto [e, exists] =
172 boost::edge(to_vertices(node1), to_vertices(node2), graph);
173 if (!exists) {
174 return 0;
175 }
176
177 return graph[e].weight;

291 std::vector<T> get_path(const T& root, const T& target) const {
292 if (!node_exists(root) || !node_exists(target)) {
293 throw NodeDoesNotExistError(
294 "Trying to get path between non-existent vertices");
295 }
296 using parent_vec = typename BFS<UndirectedConnGraph>::parent_vec;
297
298 const UndirectedConnGraph& g = get_undirected_connectivity();
299 auto bfs = run_bfs(to_vertices(root), g);
300
301 auto to_node = [&g](UndirectedVertex v) { return g[v]; };
302 parent_vec path = bfs.path_to_root(to_vertices(target));
303 std::vector<T> converted_path(path.size());
304 std::transform(path.begin(), path.end(), converted_path.begin(), to_node);
305 return converted_path;
306 }
307
309 std::set<T> get_neighbour_nodes(const T& node) const {
310 if (!node_exists(node)) {
311 throw NodeDoesNotExistError(
312 "Trying to get neighbours from non-existent vertex");
313 }
314 std::set<T> neighbours;
315 for (auto [it, end] = boost::out_edges(to_vertices(node), graph); it != end;
316 ++it) {
317 neighbours.insert(get_node(boost::target(*it, graph)));
318 }
319 for (auto [it, end] = boost::in_edges(to_vertices(node), graph); it != end;
320 ++it) {
321 neighbours.insert(get_node(boost::source(*it, graph)));
322 }
323 return neighbours;
324 }
325
326 bool operator==(const DirectedGraphBase<T>& other) const {
327 if (nodes_ != other.nodes_) return false;
328 for (const T& u : nodes_) {
329 for (const T& v : nodes_) {
330 if (this->edge_exists(u, v)) {
331 if (!other.edge_exists(u, v))
332 return false;
333 else if (
334 this->get_connection_weight(u, v) !=
335 other.get_connection_weight(u, v))
336 return false;
337 } else if (other.edge_exists(u, v))
338 return false;
339 }
340 }
341 return true;
342 }

344 protected:

347 return boost::make_iterator_range(boost::edges(graph));
348 }
349
351 auto get_vertices_it() const {
352 return boost::make_iterator_range(boost::vertices(graph));
353 }

355 using vertex_bimap = boost::bimap<T, Vertex>;
356 using left_map = typename vertex_bimap::left_map;
357 using right_map = typename vertex_bimap::right_map;
358 /* get node from vertex */
359 const T& get_node(Vertex v) const { return graph[v]; }
360
361 left_map& to_vertices() { return node_to_vertex.left; }
362 const left_map& to_vertices() const { return node_to_vertex.left; }
363 Vertex to_vertices(const T& node) const { return to_vertices().at(node); }
364
365 right_map& from_vertices() { return node_to_vertex.right; }
366 const right_map& from_vertices() const { return node_to_vertex.right; }
367 T from_vertices(Vertex v) const { return from_vertices().at(v); }
368
369 ConnGraph graph;
370 vertex_bimap node_to_vertex;

385template <typename T>
386class DirectedGraph : public DirectedGraphBase<T> {
387 private:
388 using Base = DirectedGraphBase<T>;
389
390 public:
391 using Base::Base;
392 using Base::get_all_nodes_vec;
393 using Base::n_nodes;
394 using Base::node_exists;
395 /* useful type synonyms */
396 using UndirectedConnGraph = typename Base::UndirectedConnGraph;
397 using ConnGraph = typename Base::ConnGraph;
398 using Connection = typename Base::Connection;
399 using Vertex = typename Base::Vertex;
400
404 const std::vector<std::size_t>& get_distances(const T& root) const& {
405 // We cache distances. A value of zero in the cache implies that the nodes
406 // are disconnected (unless they are equal).
407 if (distance_cache.find(root) == distance_cache.end()) {
408 distance_cache[root] = Base::get_distances(root);
409 }
410 return distance_cache[root];
411 }
412
416 std::vector<std::size_t>&& get_distances(const T& root) const&& {
417 if (distance_cache.find(root) == distance_cache.end()) {
418 distance_cache[root] = Base::get_distances(root);
419 }
420 return std::move(distance_cache[root]);
421 }
422
423 unsigned get_distance(const T& node1, const T& node2) const override {
424 if (node1 == node2) {
425 return 0;
426 }
427 std::size_t d;
428 if (distance_cache.find(node1) != distance_cache.end()) {
429 d = distance_cache[node1][this->to_vertices(node2)];
430 } else if (distance_cache.find(node2) != distance_cache.end()) {
431 d = distance_cache[node2][this->to_vertices(node1)];
432 } else {
433 distance_cache[node1] = Base::get_distances(node1);
434 d = distance_cache[node1][this->to_vertices(node2)];
435 }
436 if (d == 0) {
437 throw NodesNotConnected(node1, node2);
438 }
439 return d;
440 }
441
442 unsigned get_diameter() override {
443 unsigned N = n_nodes();
444 if (N == 0) {
445 throw std::logic_error("Graph is empty.");
446 }
447 if (!this->diameter_) {
448 this->diameter_ = 0;
449 const std::vector<T> nodes = get_all_nodes_vec();
450 for (unsigned i = 0; i < N; i++) {
451 for (unsigned j = i + 1; j < N; j++) {
452 unsigned d = get_distance(nodes[i], nodes[j]);
453 if (d > *this->diameter_) this->diameter_ = d;
454 }
455 }
456 }
457 return *this->diameter_;
458 }
459
461 std::vector<T> nodes_at_distance(const T& root, std::size_t distance) const {
462 auto dists = get_distances(root);
463 std::vector<T> out;
464 for (unsigned i = 0; i < dists.size(); ++i) {
465 if (dists[i] == distance) {
466 out.push_back(this->get_node(i));
467 }
468 }
469 return out;
470 }
471
473 const UndirectedConnGraph& get_undirected_connectivity() const& {
474 // we cache the undirected graph
475 if (!undir_graph) {
476 undir_graph = Base::get_undirected_connectivity();
477 }
478 return undir_graph.value();
479 }
480
482 UndirectedConnGraph&& get_undirected_connectivity() const&& {
483 if (!undir_graph) {
484 undir_graph = Base::get_undirected_connectivity();
485 }
486 return std::move(undir_graph.value());
487 }
488
489 // The following functions invalidate caching.
490
492 void remove_node(const T& node) {
493 invalidate_cache();
494 Base::remove_node(node);
495 }
496
498 void add_node(const T& node) {
499 invalidate_cache();
500 Base::add_node(node);
501 }
502
504 void remove_stray_nodes() {
505 invalidate_cache();
506 Base::remove_stray_nodes();
507 }
508
510 void add_connection(const T& node1, const T& node2, unsigned weight = 1) {
511 invalidate_cache();
512 Base::add_connection(node1, node2, weight);
513 }
514
516 void remove_connection(const Connection& edge) {
517 invalidate_cache();
518 Base::remove_connection(edge);
519 }
520
522 void remove_connection(const T& node1, const T& node2) {
523 invalidate_cache();
524 Base::remove_connection(node1, node2);
525 }
526
527 private:
528 inline void invalidate_cache() {
529 distance_cache.clear();
530 undir_graph = std::nullopt;
531 }
532 mutable std::map<T, std::vector<std::size_t>> distance_cache;
533 mutable std::optional<UndirectedConnGraph> undir_graph;
534};
