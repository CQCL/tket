296  std::vector<T> get_path(const T& root, const T& target) const {
297  if (!node_exists(root) || !node_exists(target)) {
298  throw NodeDoesNotExistError(
299  "Trying to get path between non-existent vertices");
300  }
301  using parent_vec = typename BFS<UndirectedConnGraph>::parent_vec;
302 
303  const UndirectedConnGraph& g = get_undirected_connectivity();
304  auto bfs = run_bfs(to_vertices(root), g);
305 
306  auto to_node = [&g](UndirectedVertex v) { return g[v]; };
307  parent_vec path = bfs.path_to_root(to_vertices(target));
308  std::vector<T> converted_path(path.size());
309  std::transform(path.begin(), path.end(), converted_path.begin(), to_node);
310  return converted_path;
311  }

314  std::set<T> get_neighbour_nodes(const T& node) const {
315  if (!node_exists(node)) {
316  throw NodeDoesNotExistError(
317  "Trying to get neighbours from non-existent vertex");
318  }
319  std::set<T> neighbours;
320  for (auto [it, end] = boost::out_edges(to_vertices(node), graph); it != end;
321  ++it) {
322  neighbours.insert(get_node(boost::target(*it, graph)));
323  }
324  for (auto [it, end] = boost::in_edges(to_vertices(node), graph); it != end;
325  ++it) {
326  neighbours.insert(get_node(boost::source(*it, graph)));
327  }
328  return neighbours;
329  }

331  bool operator==(const DirectedGraphBase<T>& other) const {
332  if (nodes_ != other.nodes_) return false;
333  for (const T& u : nodes_) {
334  for (const T& v : nodes_) {
335  if (this->edge_exists(u, v)) {
336  if (!other.edge_exists(u, v))
337  return false;
338  else if (
339  this->get_connection_weight(u, v) !=
340  other.get_connection_weight(u, v))
341  return false;
342  } else if (other.edge_exists(u, v))
343  return false;
344  }
345  }
346  return true;
347  }

390 template <typename T>
391 class DirectedGraph : public DirectedGraphBase<T> {
392  private:
393  using Base = DirectedGraphBase<T>;
394 
395  public:
396  using Base::Base;
397  using Base::get_all_nodes_vec;
398  using Base::n_nodes;
399  using Base::node_exists;
400  /* useful type synonyms */
401  using UndirectedConnGraph = typename Base::UndirectedConnGraph;
402  using ConnGraph = typename Base::ConnGraph;
403  using Connection = typename Base::Connection;
404  using Vertex = typename Base::Vertex;
405 
409  const std::vector<std::size_t>& get_distances(const T& root) const& {
410  // We cache distances. A value of zero in the cache implies that the nodes
411  // are disconnected (unless they are equal).
412  if (distance_cache.find(root) == distance_cache.end()) {
413  distance_cache[root] = Base::get_distances(root);
414  }
415  return distance_cache[root];
416  }
417 
421  std::vector<std::size_t>&& get_distances(const T& root) const&& {
422  if (distance_cache.find(root) == distance_cache.end()) {
423  distance_cache[root] = Base::get_distances(root);
424  }
425  return std::move(distance_cache[root]);
426  }
427 
428  unsigned get_distance(const T& node1, const T& node2) const override {
429  if (node1 == node2) {
430  return 0;
431  }
432  size_t d;
433  if (distance_cache.find(node1) != distance_cache.end()) {
434  d = distance_cache[node1][this->to_vertices(node2)];
435  } else if (distance_cache.find(node2) != distance_cache.end()) {
436  d = distance_cache[node2][this->to_vertices(node1)];
437  } else {
438  distance_cache[node1] = Base::get_distances(node1);
439  d = distance_cache[node1][this->to_vertices(node2)];
440  }
441  if (d == 0) {
442  throw NodesNotConnected(node1, node2);
443  }
444  return d;
445  }
446 
447  unsigned get_diameter() override {
448  unsigned N = n_nodes();
449  if (N == 0) {
450  throw std::logic_error("Graph is empty.");
451  }
452  if (!this->diameter_) {
453  this->diameter_ = 0;
454  const std::vector<T> nodes = get_all_nodes_vec();
455  for (unsigned i = 0; i < N; i++) {
456  for (unsigned j = i + 1; j < N; j++) {
457  unsigned d = get_distance(nodes[i], nodes[j]);
458  if (d > *this->diameter_) this->diameter_ = d;
459  }
460  }
461  }
462  return *this->diameter_;
463  }
464 
466  std::vector<T> nodes_at_distance(const T& root, std::size_t distance) const {
467  auto dists = get_distances(root);
468  std::vector<T> out;
469  for (unsigned i = 0; i < dists.size(); ++i) {
470  if (dists[i] == distance) {
471  out.push_back(this->get_node(i));
472  }
473  }
474  return out;
475  }
