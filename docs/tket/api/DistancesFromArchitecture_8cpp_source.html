23void DistancesFromArchitecture::register_shortest_path(
24 const std::vector<std::size_t>& path) {
25 // To avoid quadratic growth for really long paths,
26 // just do various slices.
27 if (path.size() <= 5) {
28 register_shortest_path_with_limits(path, 0, path.size());
29 return;
30 }
31 const std::size_t middle = path.size() / 2;
32 if (path.size() <= 10) {
33 register_shortest_path_with_limits(path, 0, middle);
34 register_shortest_path_with_limits(path, middle, path.size());
35 register_edge(path[middle - 1], path[middle]);
36 return;
37 }
38 register_shortest_path_with_limits(path, 0, 5);
39 register_shortest_path_with_limits(path, path.size() - 5, path.size());
40 if (path.size() >= 15) {
41 register_shortest_path_with_limits(path, middle - 2, middle + 3);
42 }
43}

45void DistancesFromArchitecture::register_shortest_path_with_limits(
46 const std::vector<std::size_t>& path, std::size_t begin, std::size_t end) {
47 for (std::size_t ii = begin; ii < end; ++ii) {
48 for (std::size_t jj = ii + 1; jj < end; ++jj) {
49 m_cached_distances[get_swap(path[ii], path[jj])] = jj - ii;
50 }
51 }
52}

54void DistancesFromArchitecture::register_edge(
55 std::size_t vertex1, std::size_t vertex2) {
56 m_cached_distances[get_swap(vertex1, vertex2)] = 1;
57}

59std::size_t DistancesFromArchitecture::operator()(
60 std::size_t vertex1, std::size_t vertex2) {
61 if (vertex1 == vertex2) {
62 return 0;
63 }
64 // Automatically set to zero if it doesn't exist yet.
65 auto& distance_entry = m_cached_distances[get_swap(vertex1, vertex2)];
66 if (distance_entry == 0) {
67 const auto& arch = m_arch_mapping.get_architecture();
68 distance_entry = arch.get_distance(
69 m_arch_mapping.get_node(vertex1), m_arch_mapping.get_node(vertex2));
70
71 // This message should no longer be triggered for disconnected
72 // architectures, since get_distance now should throw if v1, v2 are in
73 // different connected components. However, leave the check in, in case some
74 // other bizarre error causes distance zero to be returned.
75 // GCOVR_EXCL_START
76 TKET_ASSERT(
77 distance_entry > 0 ||
78 AssertMessage() << "DistancesFromArchitecture: architecture has "
79 << arch.n_nodes() << " vertices, "
80 << arch.n_connections() << " edges; "
81 << " and d(" << vertex1 << "," << vertex2
82 << ")=0. "
83 "Is the graph connected?");
84 // GCOVR_EXCL_STOP
85 }
86 return distance_entry;
87}
