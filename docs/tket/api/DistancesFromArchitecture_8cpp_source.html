26 void DistancesFromArchitecture::register_shortest_path(
27  const std::vector<size_t>& path) {
28  // To avoid quadratic growth for really long paths,
29  // just do various slices.
30  if (path.size() <= 5) {
31  register_shortest_path_with_limits(path, 0, path.size());
32  return;
33  }
34  const size_t middle = path.size() / 2;
35  if (path.size() <= 10) {
36  register_shortest_path_with_limits(path, 0, middle);
37  register_shortest_path_with_limits(path, middle, path.size());
38  register_edge(path[middle - 1], path[middle]);
39  return;
40  }
41  register_shortest_path_with_limits(path, 0, 5);
42  register_shortest_path_with_limits(path, path.size() - 5, path.size());
43  if (path.size() >= 15) {
44  register_shortest_path_with_limits(path, middle - 2, middle + 3);
45  }
46 }

48 void DistancesFromArchitecture::register_shortest_path_with_limits(
49  const std::vector<size_t>& path, size_t begin, size_t end) {
50  for (size_t ii = begin; ii < end; ++ii) {
51  for (size_t jj = ii + 1; jj < end; ++jj) {
52  m_cached_distances[get_swap(path[ii], path[jj])] = jj - ii;
53  }
54  }
55 }

61 size_t DistancesFromArchitecture::operator()(size_t vertex1, size_t vertex2) {
62  if (vertex1 == vertex2) {
63  return 0;
64  }
65  // Automatically set to zero if it doesn't exist yet.
66  auto& distance_entry = m_cached_distances[get_swap(vertex1, vertex2)];
67  if (distance_entry == 0) {
68  const auto& arch = m_arch_mapping.get_architecture();
69  distance_entry = arch.get_distance(
70  m_arch_mapping.get_node(vertex1), m_arch_mapping.get_node(vertex2));
71 
72  // This message should no longer be triggered for disconnected
73  // architectures, since get_distance now should throw if v1, v2 are in
74  // different connected components. However, leave the check in, in case some
75  // other bizarre error causes distance zero to be returned.
76  // GCOVR_EXCL_START
77  TKET_ASSERT(
78  distance_entry > 0 ||
79  AssertMessage() << "DistancesFromArchitecture: architecture has "
80  << arch.n_nodes() << " vertices, "
81  << arch.n_connections() << " edges; "
82  << " and d(" << vertex1 << "," << vertex2
83  << ")=0. "
84  "Is the graph connected?");
85  // GCOVR_EXCL_STOP
86  }
87  return distance_entry;
88 }
