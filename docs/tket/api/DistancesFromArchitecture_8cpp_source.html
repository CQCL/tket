26 void DistancesFromArchitecture::register_shortest_path(
27  const std::vector<std::size_t>& path) {
28  // To avoid quadratic growth for really long paths,
29  // just do various slices.
30  if (path.size() <= 5) {
31  register_shortest_path_with_limits(path, 0, path.size());
32  return;
33  }
34  const std::size_t middle = path.size() / 2;
35  if (path.size() <= 10) {
36  register_shortest_path_with_limits(path, 0, middle);
37  register_shortest_path_with_limits(path, middle, path.size());
38  register_edge(path[middle - 1], path[middle]);
39  return;
40  }
41  register_shortest_path_with_limits(path, 0, 5);
42  register_shortest_path_with_limits(path, path.size() - 5, path.size());
43  if (path.size() >= 15) {
44  register_shortest_path_with_limits(path, middle - 2, middle + 3);
45  }
46 }

48 void DistancesFromArchitecture::register_shortest_path_with_limits(
49  const std::vector<std::size_t>& path, std::size_t begin, std::size_t end) {
50  for (std::size_t ii = begin; ii < end; ++ii) {
51  for (std::size_t jj = ii + 1; jj < end; ++jj) {
52  m_cached_distances[get_swap(path[ii], path[jj])] = jj - ii;
53  }
54  }
55 }

62 std::size_t DistancesFromArchitecture::operator()(
63  std::size_t vertex1, std::size_t vertex2) {
64  if (vertex1 == vertex2) {
65  return 0;
66  }
67  // Automatically set to zero if it doesn't exist yet.
68  auto& distance_entry = m_cached_distances[get_swap(vertex1, vertex2)];
69  if (distance_entry == 0) {
70  const auto& arch = m_arch_mapping.get_architecture();
71  distance_entry = arch.get_distance(
72  m_arch_mapping.get_node(vertex1), m_arch_mapping.get_node(vertex2));
73 
74  // This message should no longer be triggered for disconnected
75  // architectures, since get_distance now should throw if v1, v2 are in
76  // different connected components. However, leave the check in, in case some
77  // other bizarre error causes distance zero to be returned.
78  // GCOVR_EXCL_START
79  TKET_ASSERT(
80  distance_entry > 0 ||
81  AssertMessage() << "DistancesFromArchitecture: architecture has "
82  << arch.n_nodes() << " vertices, "
83  << arch.n_connections() << " edges; "
84  << " and d(" << vertex1 << "," << vertex2
85  << ")=0. "
86  "Is the graph connected?");
87  // GCOVR_EXCL_STOP
88  }
89  return distance_entry;
90 }
