48 ReductionResult DistancesReducer::reduce(
49  std::pair<VertexWSM, VertexWSM> assignment, DomainsAccessor& accessor,
50  std::set<VertexWSM>& work_set) {
51  const auto& pv_at_distance_d = m_pattern_near_ndata.get_vertices_at_distance(
52  assignment.first, m_distance);
53 
54  if (pv_at_distance_d.empty()) {
55  // Nothing to check!
56  return ReductionResult::SUCCESS;
57  }
58 
59  // To save space, we don't explicitly construct the union
60  // {v : dist(TV,v) <= j}, which could be large.
61  // For each PV' with dist(PV, PV')=d, we intersect Dom(PV')
62  // with the disjoint UNION of A(1), A(2), ..., A(d),
63  // where A(j) = {v : dist(TV,v)=j}.
64  // We build it up by combining
65  // Dom(PV') intersect A(j) (which are disjoint), directly.
66  const std::vector<std::pair<VertexWSM, WeightWSM>>& tv_neighbours =
67  m_target_ndata.get_neighbours_and_weights(assignment.second);
68 
69  auto result = ReductionResult::SUCCESS;
70 
71  for (VertexWSM new_pv : pv_at_distance_d) {
72  const std::set<VertexWSM>& current_domain = accessor.get_domain(new_pv);
73  if (other_vertex_reduction_can_be_skipped_by_symmetry(
74  current_domain, accessor, assignment.first, new_pv)) {
75  continue;
76  }
77  // This is for d=1, obviously a special case.
78  fill_intersection_ignoring_second_elements(
79  current_domain, tv_neighbours, work_set);
80 
81  auto size_so_far = work_set.size();
82  unsigned next_work_vector_index = 0;
83 
84  // Add all the other intersected sets.
85  for (unsigned jj = 2; jj <= m_distance; ++jj) {
86  const auto& new_tv_list =
87  m_target_near_ndata.get_vertices_at_distance(assignment.second, jj);
88 
89  // The constructor already resized the vectors_list,
90  // to ensure a valid index.
91  std::vector<VertexWSM>& work_vector =
92  m_work_vectors_list[next_work_vector_index];
93 
94  fill_intersection(current_domain, new_tv_list, work_vector);
95  if (!work_vector.empty()) {
96  // If it was empty, just reuse for the next distance.
97  size_so_far += work_vector.size();
98  ++next_work_vector_index;
99  }
100  }
101  if (size_so_far == current_domain.size()) {
102  // No change!
103  continue;
104  }
105 
106  if (size_so_far == 0) {
107  return ReductionResult::NOGOOD;
108  }
109 
110  // Now, take the union of all the disjoint sets.
111  for (unsigned index = 0; index < next_work_vector_index; ++index) {
112  for (VertexWSM tv : m_work_vectors_list[index]) {
113  work_set.insert(tv);
114  }
115  }
116  // Now "m_work_vector" is the final intersection,
117  // although NOT sorted; but that doesn't matter.
118  switch (accessor.overwrite_domain_with_set_swap(new_pv, work_set)) {
119  case ReductionResult::NEW_ASSIGNMENTS:
120  // Normally we would return now,
121  // BUT the assignment PV->TV hasn't been fully processed
122  // until ALL PV' with dist(PV, PV')=d have been considered.
123  // Thus, continue, to finish off this assignment,
124  // rather than trying to store partial data
125  // and resuming later.
126  result = ReductionResult::NEW_ASSIGNMENTS;
127  break;
128  case ReductionResult::NOGOOD:
129  // We should already have handled nogoods.
130  TKET_ASSERT(false);
131  case ReductionResult::SUCCESS:
132  break;
133  }
134  }
135  return result;
136 }
