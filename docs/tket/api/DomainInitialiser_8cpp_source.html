44 bool DomainInitialiser::degree_sequence_initialisation(
45  InitialDomains& initial_domains,
46  const NeighboursData& pattern_neighbours_data,
47  const NeighboursData& target_neighbours_data) {
48  initial_domains.resize(
49  pattern_neighbours_data.get_number_of_nonisolated_vertices());
50  for (std::set<VertexWSM>& domain : initial_domains) {
51  domain.clear();
52  }
53 
54  // Get the target degree sequences.
55  const auto number_of_tv =
56  target_neighbours_data.get_number_of_nonisolated_vertices();
57 
58  // FIRST: the target vertex;
59  // SECOND: its degree sequence.
60  typedef std::pair<VertexWSM, std::vector<std::size_t>> DegSeqData;
61  std::vector<DegSeqData> target_degree_sequences;
62 
63  target_degree_sequences.reserve(number_of_tv);
64  for (unsigned tv = 0; tv < number_of_tv; ++tv) {
65  target_degree_sequences.emplace_back(
66  tv, target_neighbours_data.get_sorted_degree_sequence_expensive(tv));
67  }
68 
69  // Sort by decreasing sequence length.
70  std::sort(
71  target_degree_sequences.begin(), target_degree_sequences.end(),
72  [](const DegSeqData& lhs, const DegSeqData& rhs) -> bool {
73  return lhs.second.size() > rhs.second.size() ||
74  // This is not crucial, it just ensures exact behaviour for
75  // nonstable sorts:
76  (lhs.second.size() == rhs.second.size() &&
77  lhs.first < rhs.first);
78  });
79 
80  // TODO: get TV data lazily (maybe some will be unused).
81  std::vector<std::size_t> pattern_sequence;
82  for (unsigned pv = 0; pv < initial_domains.size(); ++pv) {
83  std::set<VertexWSM>& domain = initial_domains[pv];
84  pattern_sequence =
85  pattern_neighbours_data.get_sorted_degree_sequence_expensive(pv);
86 
87  // Now, which target vertices have compatible degree sequences?
88  for (const DegSeqData& deg_seq_data : target_degree_sequences) {
89  const auto& target_sequence = deg_seq_data.second;
90  if (target_sequence.size() < pattern_sequence.size()) {
91  break;
92  }
93  if (FilterUtils::compatible_sorted_degree_sequences(
94  pattern_sequence, target_sequence)) {
95  domain.insert(deg_seq_data.first);
96  }
97  }
98  if (domain.empty()) {
99  return false;
100  }
101  }
102  return true;
103 }

105 bool DomainInitialiser::distance_counts_reduction(
106  InitialDomains& initial_domains,
107  NearNeighboursData& pattern_near_neighbours_data,
108  NearNeighboursData& target_near_neighbours_data, unsigned max_path_length) {
109  if (max_path_length <= 1) {
110  // Neighbour counts are already included in degree sequences.
111  return true;
112  }
113  // Easier to build them all up, then erase them later in a separate pass.
114  std::vector<VertexWSM> tv_to_erase;
115 
116  for (unsigned pv = 0; pv < initial_domains.size(); ++pv) {
117  std::set<VertexWSM>& domain = initial_domains[pv];
118  tv_to_erase.clear();
119  for (VertexWSM tv : domain) {
120  // If f is a valid monomorphism then
121  // dist(PV, PV')=d ==> dist(f(PV), f(PV')) <= d.
122  // Now TV=f(PV) is the value to be checked;
123  // so for each d, just count how many PV' are at that exact distance d,
124  // and check that there are at least that many TV' with
125  // dist(TV, TV') <= d.
126  for (unsigned distance = 2; distance <= max_path_length; ++distance) {
127  const auto number_of_pv_at_distance_d =
128  pattern_near_neighbours_data.get_vertices_at_distance(pv, distance)
129  .size();
130 
131  if (number_of_pv_at_distance_d > 0) {
132  const auto number_of_tv_at_distance_le_d =
133  target_near_neighbours_data.get_n_vertices_at_max_distance(
134  tv, distance);
135  if (number_of_pv_at_distance_d > number_of_tv_at_distance_le_d) {
136  tv_to_erase.push_back(tv);
137  break;
138  }
139  }
140  }
141  }
142  if (tv_to_erase.size() == domain.size()) {
143  return false;
144  }
145  for (VertexWSM tv : tv_to_erase) {
146  TKET_ASSERT(domain.erase(tv) == 1);
147  }
148  }
149  return true;
150 }
