78 bool DomainsAccessor::alldiff_reduce_current_node(
79  std::size_t n_assignments_already_processed) {
80  NodesRawData::NodeData& node = m_raw_data.get_current_node_nonconst();
81  TKET_ASSERT(!node.nogood);
82  std::vector<std::pair<VertexWSM, VertexWSM>>& new_assignments =
83  node.new_assignments;
84 
85  while (n_assignments_already_processed < new_assignments.size()) {
86  const std::pair<VertexWSM, VertexWSM> assignment =
87  new_assignments[n_assignments_already_processed];
88  ++n_assignments_already_processed;
89 
90  for (unsigned pv = 0; pv < m_raw_data.domains_data.size(); ++pv) {
91  if (pv == assignment.first) {
92  continue;
93  }
94  NodesRawData::DomainData& data_for_this_pv = m_raw_data.domains_data[pv];
95  std::set<VertexWSM>& existing_domain =
96  data_for_this_pv.entries.top().domain;
97  auto iter = existing_domain.find(assignment.second);
98  if (iter == existing_domain.end()) {
99  continue;
100  }
101  if (existing_domain.size() == 1) {
102  // Erasing TV would make a nogood.
103  return false;
104  }
105  if (existing_domain.size() == 2) {
106  // Erasing TV would make a new assignment.
107  VertexWSM tv_other = *existing_domain.cbegin();
108  if (tv_other == assignment.second) {
109  tv_other = *existing_domain.crbegin();
110  }
111  new_assignments.emplace_back(pv, tv_other);
112  }
113  // Now, we've taken care of everything EXCEPT
114  // erasing TV from the domain.
115  if (data_for_this_pv.entries.top().node_index ==
116  m_raw_data.current_node_index()) {
117  // Erase in-place.
118  existing_domain.erase(iter);
119  } else {
120  // We must make a new domain and copy the data across.
121  data_for_this_pv.entries.push();
122  data_for_this_pv.entries.top().node_index =
123  m_raw_data.current_node_index();
124  data_for_this_pv.entries.top().domain =
125  data_for_this_pv.entries.one_below_top().domain;
126  TKET_ASSERT(
127  data_for_this_pv.entries.top().domain.erase(assignment.second) ==
128  1);
129  }
130  }
131  }
132  return true;
133 }

135 ReductionResult DomainsAccessor::overwrite_domain_with_set_swap(
136  VertexWSM pv, std::set<VertexWSM>& new_domain) {
137  if (new_domain.empty()) {
138  return ReductionResult::NOGOOD;
139  }
140  auto& data_for_this_pv = m_raw_data.domains_data.at(pv);
141  auto& existing_domain_data = data_for_this_pv.entries.top();
142  std::set<VertexWSM>& existing_domain = existing_domain_data.domain;
143 
144  // We should do a detailed assert that "new_domain" is a subset
145  // of "existing_domain", but this is expensive; thus we do a very
146  // cheap assert.
147  TKET_ASSERT(existing_domain.size() >= new_domain.size());
148  TKET_ASSERT(existing_domain.count(*new_domain.cbegin()) != 0);
149 
150  if (existing_domain.size() == new_domain.size()) {
151  // No change, no action required (even if the new domain has size 1;
152  // it's not a NEW assignment).
153  return ReductionResult::SUCCESS;
154  }
155  auto result = ReductionResult::SUCCESS;
156 
157  if (new_domain.size() == 1) {
158  result = ReductionResult::NEW_ASSIGNMENTS;
159  auto& current_node = m_raw_data.get_current_node_nonconst();
160  current_node.new_assignments.emplace_back(pv, *new_domain.cbegin());
161  }
162  // We must be careful; "existing_domain" might be shared with previous nodes.
163  if (existing_domain_data.node_index == m_raw_data.current_node_index()) {
164  // We can just overwrite the domain in-place; the caller will do that.
165  existing_domain.swap(new_domain);
166  } else {
167  // We need to make a new domain object.
168  data_for_this_pv.entries.push();
169  data_for_this_pv.entries.top().node_index = m_raw_data.current_node_index();
170  data_for_this_pv.entries.top().domain.swap(new_domain);
171  }
172  return result;
173 }

175 DomainsAccessor::IntersectionResult
176 DomainsAccessor::intersect_domain_with_complement_set(
177  VertexWSM pattern_v, const std::set<VertexWSM>& forbidden_target_vertices) {
178  IntersectionResult result;
179  {
180  const std::set<VertexWSM>& current_domain = get_domain(pattern_v);
181 
182  if (disjoint(current_domain, forbidden_target_vertices)) {
183  result.changed = false;
184  result.new_domain_size = current_domain.size();
185  result.reduction_result = ReductionResult::SUCCESS;
186  return result;
187  }
188  }
189  // The domain definitely is changing.
190  result.changed = true;
191  auto& data_for_this_pv = m_raw_data.domains_data.at(pattern_v);
192  if (data_for_this_pv.entries.top().node_index ==
193  m_raw_data.current_node_index()) {
194  // Domain is not shared; we can simply overwrite.
195  std::set<VertexWSM>& domain = data_for_this_pv.entries.top().domain;
196 
197  for (VertexWSM tv : forbidden_target_vertices) {
198  if (domain.erase(tv) == 1 && domain.empty()) {
199  result.reduction_result = ReductionResult::NOGOOD;
200  return result;
201  }
202  }
203  } else {
204  // Shared domain; copy on write:
205  data_for_this_pv.entries.push();
206  data_for_this_pv.entries.top().node_index = m_raw_data.current_node_index();
207 
208  const std::set<VertexWSM>& old_domain =
209  data_for_this_pv.entries.one_below_top().domain;
210  std::set<VertexWSM>& new_domain = data_for_this_pv.entries.top().domain;
211 
212  // Is it quicker to build it up as here, or copy then grind it down?
213  // Depends on all the sizes, and how many end up being erased...
214  new_domain.clear();
215 
216  for (VertexWSM old_tv : old_domain) {
217  if (forbidden_target_vertices.count(old_tv) == 0) {
218  new_domain.insert(old_tv);
219  }
220  }
221 
222  if (new_domain.empty()) {
223  // The caller will move up, no further action here.
224  result.reduction_result = ReductionResult::NOGOOD;
225  return result;
226  }
227  }
228 
229  // At this stage, we've created and filled the new domain object
230  // (necessarily different from the old one).
231  const std::set<VertexWSM>& domain = data_for_this_pv.entries.top().domain;
232  result.new_domain_size = domain.size();
233  TKET_ASSERT(!domain.empty());
234  if (domain.size() > 1) {
235  result.reduction_result = ReductionResult::SUCCESS;
236  } else {
237  // domain size 1, so a new assignment (as we KNOW that the domain has
238  // changed).
239  m_raw_data.get_current_node_nonconst().new_assignments.emplace_back(
240  pattern_v, *domain.cbegin());
241  result.reduction_result = ReductionResult::NEW_ASSIGNMENTS;
242  }
243  return result;
244 }

246 std::string DomainsAccessor::str(bool full) const {
247  std::stringstream ss;
248  if (full) {
249  ss << "\n@@@@@ ALL NODES: (curr.node index="
250  << m_raw_data.current_node_index() << ")";
251  for (unsigned node_index = 0; node_index < m_raw_data.nodes_data.size();
252  ++node_index) {
253  const auto& node = m_raw_data.nodes_data[node_index];
254  ss << "\n+++ node " << node_index << ":" << node.str();
255  }
256  ss << "\nDOMAINS: ";
257  for (unsigned pv = 0; pv < m_raw_data.domains_data.size(); ++pv) {
258  ss << "\nDOM(" << pv << "):" << m_raw_data.domains_data[pv].str();
259  }
260  ss << "\n";
261  return ss.str();
262  }
263  ss << "\ncurr.node index=" << m_raw_data.current_node_index()
264  << "; DOMAINS: ";
265  for (unsigned pv = 0; pv < m_raw_data.domains_data.size(); ++pv) {
266  ss << "\n DOM(" << pv << "): ";
267  const auto& dom_data = m_raw_data.domains_data[pv].entries.top();
268  ss << "(since index " << dom_data.node_index
269  << "): " << tket::WeightedSubgraphMonomorphism::str(dom_data.domain);
270  }
271  ss << "\n";
272  return ss.str();
273 }
