54nlohmann::json DummyBox::to_json(const Op_ptr &op) {
55 const auto &box = static_cast<const DummyBox &>(*op);
56 nlohmann::json j = core_box_json(box);
57 j["n_qubits"] = box.get_n_qubits();
58 j["n_bits"] = box.get_n_bits();
59 j["resource_data"] = box.get_resource_data();
60 return j;
61}

63Op_ptr DummyBox::from_json(const nlohmann::json &j) {
64 DummyBox box = DummyBox(
65 j.at("n_qubits").get<unsigned>(), j.at("n_bits").get<unsigned>(),
66 j.at("resource_data").get<ResourceData>());
67 return set_box_id(
68 box,
69 boost::lexical_cast<boost::uuids::uuid>(j.at("id").get<std::string>()));
