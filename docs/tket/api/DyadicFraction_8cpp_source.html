26 DyadicFraction::DyadicFraction(UInt x) : m_value(x), m_exponent(0) {}

28 // We have 0 < x <= y, but x*y may overflow,
29 // and x,y have already had all their trailing zeros removed.
30 // If x*y DOES overflow, right shift x,y as little as possible,
31 // until x*y can safely be calculated,
32 // and return the total right shifts.
33 static unsigned remove_bits_lossy(
34  DyadicFraction::UInt& x, unsigned x_bit_length, DyadicFraction::UInt& y,
35  unsigned y_bit_length) {
36  unsigned total_bitlength = x_bit_length + y_bit_length;
37 
38  // OK, some slight inaccuracy here;
39  // x*y would overflow <==> BitLength(xy) > 64,
40  // but of course we're using BitLength(x) + BitLength(y)
41  // as an estimate (upper bound) for BitLength(xy).
42  // Might be out by 1, so we might remove 1 extra bit;
43  // but don't worry about it, insignificant.
44  if (total_bitlength <= 64) {
45  return 0;
46  }
47  unsigned remaining_shift = total_bitlength - 64;
48  TKET_ASSERT(remaining_shift >= 1);
49  // We're definitely going to shift by exactly this much,
50  // so might as well store it now.
51  const auto shift_to_return = remaining_shift;
52 
53  // First, y is larger, so put all the reduction onto y, until
54  // x,y are equal in length.
55  const unsigned ly_minus_lx = y_bit_length - x_bit_length;
56  if (ly_minus_lx >= remaining_shift) {
57  // We're lucky! We need only shift y.
58  y >>= remaining_shift;
59  return shift_to_return;
60  }
61  // Shift y to get an equal bitlength with x.
62  y >>= ly_minus_lx;
63  remaining_shift -= ly_minus_lx;
64  // Now, split the shifts equally between x,y,
65  // since they now have equal bitlength.
66  const unsigned common_shift = remaining_shift / 2;
67  x >>= common_shift;
68  y >>= common_shift;
69  remaining_shift -= common_shift * 2;
70  TKET_ASSERT(remaining_shift <= 1);
71  if (remaining_shift == 0) {
72  return shift_to_return;
73  }
74  // Must be exactly 1 bit remaining. Are we lucky, did we create a zero?
75  if ((x & 1) == 0) {
76  x >>= 1;
77  } else {
78  y >>= 1;
79  }
80  return shift_to_return;
81 }

83 DyadicFraction& DyadicFraction::mult(UInt x) {
84  if (m_value == 0 || x == 1) {
85  return *this;
86  }
87  if (x == 0) {
88  m_value = 0;
89  m_exponent = 0;
90  return *this;
91  }
92  auto current_bit_length = BitFunctions::get_bit_length(m_value);
93  auto x_bit_length = BitFunctions::get_bit_length(x);
94 
95  // Of course, the bit length almost has the log property, but not quite:
96  // Bitlength(xy) <= Bitlength(x) + Bitlength(y),
97  // but inequality can hold, i.e. 8,15, 8*15=120 < 128
98  // have bitlengths 4,4,7 respectively.
99  // We're just using the total bitlength to get an upper bound
100  // on the size of the product.
101  auto total_bit_length = current_bit_length + x_bit_length;
102  if (total_bit_length <= 64) {
103  m_value *= x;
104  return *this;
105  }
106  // First, losslessly remove bits.
107  const auto current_trailing_zeros =
108  BitFunctions::get_number_of_rightmost_zero_bits(m_value);
109 
110  const auto x_trailing_zeros =
111  BitFunctions::get_number_of_rightmost_zero_bits(x);
112 
113  current_bit_length -= current_trailing_zeros;
114  m_value >>= current_trailing_zeros;
115  x_bit_length -= x_trailing_zeros;
116  x >>= x_trailing_zeros;
117 
118  // We've divided nx by 2^a.2^b, so we must compensate.
119  // (But we haven't actually multiplied by x yet!)
120  m_exponent += current_trailing_zeros + x_trailing_zeros;
121 
122  // We've losslessly removed as many bits as we can,
123  // but maybe still not enough.
124 
125  if (m_value < x) {
126  m_exponent +=
127  remove_bits_lossy(m_value, current_bit_length, x, x_bit_length);
128  } else {
129  m_exponent +=
130  remove_bits_lossy(x, x_bit_length, m_value, current_bit_length);
131  }
132  m_value *= x;
133  return *this;
134 }

165 bool DyadicFraction::operator<(const DyadicFraction& other) const {
166  // We're asking if 2^n.x < 2^m.y, i.e. x < 2^d.y ?
167  const auto exponent_diff = other.m_exponent - m_exponent;
168  if (exponent_diff >= 0) {
169  // d >= 0.
170  auto y = other.m_value;
171  const unsigned y_bitlength = BitFunctions::get_bit_length(y);
172  if (y_bitlength + (unsigned)exponent_diff > 64) {
173  // 2^d.y would overflow, so definitely bigger than x.
174  return true;
175  }
176  // We can shift without overflow.
177  y <<= (unsigned)exponent_diff;
178  return m_value < y;
179  }
180  // Now we are asking if 2^d.x < y for d>0.
181  auto x = m_value;
182  const unsigned x_shift = -exponent_diff;
183  const unsigned x_bitlength = BitFunctions::get_bit_length(m_value);
184  if (x_shift + x_bitlength > 64) {
185  // 2^d.x overflows.
186  return false;
187  }
188  x <<= x_shift;
189  return x < other.m_value;
190 }
