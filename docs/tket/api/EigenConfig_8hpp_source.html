44template <typename _Scalar, int _Rows, int _Cols>
45void to_json(nlohmann::json& j, const Matrix<_Scalar, _Rows, _Cols>& matrix) {
46 for (Index i = 0; i < matrix.rows(); ++i) {
47 nlohmann::json row = nlohmann::json::array();
48 for (Index j = 0; j < matrix.cols(); ++j) {
49 row.push_back(matrix(i, j));
50 }
51 j.push_back(row);
52 }
53}

55template <typename _Scalar, int _Rows, int _Cols>
56void from_json(const nlohmann::json& j, Matrix<_Scalar, _Rows, _Cols>& matrix) {
57 // resize for dynamically-sized matrices
58 if (j.size() == 0) return;
59 matrix.resize(j.size(), j.at(0).size());
60 for (std::size_t i = 0; i < j.size(); ++i) {
61 const auto& j_row = j.at(i);
62 for (std::size_t j = 0; j < j_row.size(); ++j) {
63 matrix(i, j) =
64 j_row.at(j).get<typename Matrix<_Scalar, _Rows, _Cols>::Scalar>();
65 }
66 }
67}
