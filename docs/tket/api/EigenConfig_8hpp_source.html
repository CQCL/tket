47 template <typename _Scalar, int _Rows, int _Cols>
48 void to_json(nlohmann::json& j, const Matrix<_Scalar, _Rows, _Cols>& matrix) {
49  for (Index i = 0; i < matrix.rows(); ++i) {
50  nlohmann::json row = nlohmann::json::array();
51  for (Index j = 0; j < matrix.cols(); ++j) {
52  row.push_back(matrix(i, j));
53  }
54  j.push_back(row);
55  }
56 }

58 template <typename _Scalar, int _Rows, int _Cols>
59 void from_json(const nlohmann::json& j, Matrix<_Scalar, _Rows, _Cols>& matrix) {
60  // resize for dynamically-sized matrices
61  if (j.size() == 0) return;
62  matrix.resize(j.size(), j.at(0).size());
63  for (std::size_t i = 0; i < j.size(); ++i) {
64  const auto& j_row = j.at(i);
65  for (std::size_t j = 0; j < j_row.size(); ++j) {
66  matrix(i, j) =
67  j_row.at(j).get<typename Matrix<_Scalar, _Rows, _Cols>::Scalar>();
68  }
69  }
70 }
