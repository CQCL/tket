29double fmodn(double x, unsigned n) {
30 x /= n;
31 x -= floor(x);
32 return n * x;

35bool approx_eq(double x, double y, unsigned mod, double tol) {
36 double r = fmodn(x - y, mod);
37 return r < tol || r > mod - tol;

78std::optional<double> eval_expr_mod(const Expr& e, unsigned n) {
79 std::optional<double> reduced_val = eval_expr(e);
80 if (!reduced_val) return std::nullopt;
81 double val = reduced_val.value();
82 double val4 = 4 * val;
83 long nearest_val4 = std::lrint(val4);
84 if (std::abs(val4 - nearest_val4) < 4 * EPS) {
85 val = nearest_val4 * 0.25;
86 }
87 return fmodn(val, n);
88}

90// Evaluate cos(pi x / 12). If x is close to a multiple of pi/12, it is clamped
91// to that exact multiple and the return value is exact.
92// Is is assumed that 0 <= x < 24.
93static Expr cos_pi_by_12_times(double x) {
94 static const double pi_by_12 = PI / 12;
95 static const Expr pi_by_12_sym =
96 SymEngine::div(SymEngine::pi, SymEngine::integer(12));
97 int n(x + 0.5); // nearest integer to x (assuming x >= 0)
98 if (std::abs(x - n) < EPS) {
99 Expr z = SymEngine::cos(n * pi_by_12_sym);
100 return z;
101 } else {
102 return Expr(cos(pi_by_12 * x));
103 }
104}

106Expr cos_halfpi_times(const Expr& e) {
107 std::optional<double> x = eval_expr_mod(e / 2); // >= 0
108 if (x) {
109 return cos_pi_by_12_times(12 * x.value());
110 } else {
111 return SymEngine::cos(SymEngine::expand(e * SymEngine::pi / 2));
112 }
113}

119Expr minus_times(const Expr& e) {
120 Expr e1 = -e;
121 Expr e2 = SymEngine::expand(e1);
122
123 unsigned e1_size = e1.get_basic()->dumps().size();
124 unsigned e2_size = e2.get_basic()->dumps().size();
125
126 return e2_size < e1_size ? e2 : e1;
127}

129bool equiv_expr(const Expr& e0, const Expr& e1, unsigned n, double tol) {
130 std::optional<double> eval0 = eval_expr(e0);
131 std::optional<double> eval1 = eval_expr(e1);
132 if (!eval0 || !eval1) return e0 == e1;
133 return approx_eq(eval0.value(), eval1.value(), n, tol);
134}

136bool equiv_val(const Expr& e, double x, unsigned n, double tol) {
137 std::optional<double> eval = eval_expr(e);
138 if (!eval) return false;
139 return approx_eq(eval.value(), x, n, tol);
140}

146std::optional<unsigned> equiv_Clifford(const Expr& e, unsigned n, double tol) {
147 std::optional<double> eval = eval_expr_mod(e, n);
148 if (!eval) return std::nullopt;
149 double v_mod_n = eval.value();
150 unsigned nearest = lround(v_mod_n * 2);
151 if (std::abs(v_mod_n - (nearest * 0.5)) < tol)
152 return nearest;
153 else
154 return std::nullopt;
155}
