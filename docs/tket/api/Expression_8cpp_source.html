79 std::optional<double> eval_expr_mod(const Expr& e, unsigned n) {
80  std::optional<double> reduced_val = eval_expr(e);
81  if (!reduced_val) return std::nullopt;
82  double val = reduced_val.value();
83  double val4 = 4 * val;
84  long nearest_val4 = std::lrint(val4);
85  if (std::abs(val4 - nearest_val4) < 4 * EPS) {
86  val = nearest_val4 * 0.25;
87  }
88  return fmodn(val, n);
89 }

91 // Evaluate cos(pi x / 12). If x is close to a multiple of pi/12, it is clamped
92 // to that exact multiple and the return value is exact.
93 // Is is assumed that 0 <= x < 24.
94 static Expr cos_pi_by_12_times(double x) {
95  static const double pi_by_12 = PI / 12;
96  static const Expr pi_by_12_sym =
97  SymEngine::div(SymEngine::pi, SymEngine::integer(12));
98  int n(x + 0.5); // nearest integer to x (assuming x >= 0)
99  if (std::abs(x - n) < EPS) {
100  Expr z = SymEngine::cos(n * pi_by_12_sym);
101  return z;
102  } else {
103  return Expr(cos(pi_by_12 * x));
104  }
105 }

130 bool equiv_expr(const Expr& e0, const Expr& e1, unsigned n, double tol) {
131  std::optional<double> eval0 = eval_expr(e0);
132  std::optional<double> eval1 = eval_expr(e1);
133  if (!eval0 || !eval1) return e0 == e1;
134  return approx_eq(eval0.value(), eval1.value(), n, tol);
135 }

147 std::optional<unsigned> equiv_Clifford(const Expr& e, unsigned n, double tol) {
148  std::optional<double> eval = eval_expr_mod(e, n);
149  if (!eval) return std::nullopt;
150  double v_mod_n = eval.value();
151  unsigned nearest = lround(v_mod_n * 2);
152  if (std::abs(v_mod_n - (nearest * 0.5)) < tol)
153  return nearest;
154  else
155  return std::nullopt;
156 }
