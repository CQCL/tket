24 bool FilterUtils::compatible_sorted_degree_sequences(
25  const std::vector<std::size_t>& pattern_v_deg_seq,
26  const std::vector<std::size_t>& target_v_deg_seq) {
27  // If a pattern vertex v_P has deg sequence p(0) <= p(1) <= p(2) <= ... <=
28  // p(n) in the pattern graph, and target vertex v_T has deg seq t(0) <= t(1)
29  // <= ... <= t(m) in the target graph, then:
30  //
31  // v_P can be mapped to v_T (at least locally) <==>
32  //
33  // there exists injective F : {0,1,...,n} -> {0,1,...,m} such that p(i) <=
34  // t(F(i)) for all i
35  // [obvious by defn: F IS the vertex mapping]
36  //
37  // <==>
38  //
39  // (p(i)) can be placed inside a dominating subsequence of (t(j)),
40  // i.e. we can make F be INCREASING
41  //
42  // [Pr.: <== is trivial; for ==>: can do bubble sort:
43  // let F be given, inj but maybe not increasing. If F(i+1) < F(i), then
44  // p(i) <= p(i+1) <= t(F(i+1)) <= t(F(i)).
45  // Thus both p(i), p(i+1) are dominated by t(F(i+1)), t(F(i))
46  // so we can just swap F(i+1), F(i) and F remains valid].
47  //
48  // <==>
49  //
50  // we can construct a solution greedily, i.e. DEFINE G(0) = min { j : p(0) <=
51  // t(j)}, and G(i+1) = min { j > G(i) : p(i+1) <= t(j)}, and just set F=G
52  // (assuming that G(0),...,G(n) exist).
53  //
54  // [Pr.: <== triv: if the alg succeeds, G is strictly increasing by
55  // construction.
56  // ==>: let F incr be given. Start trying to construct G by the alg.
57  // We have G(0) <= F(0). If at stage k, G(k) has been constructed AND G(i)
58  // <= F(i) for all i <= k, and k<n, consider trying to find G(k+1). Let S =
59  // { j > G(k) : p(k+1) <= t(j)}. By definition, F(k+1) is in S, since
60  // F(k+1) > F(k) >= G(k). Thus S is nonempty, so a value G(k+1) > G(k)
61  // exists. By construction G(k+1) is the MIN possible solution, so G(k+1)
62  // <= F(k+1). Thus by induction, G will be fully constructed and G(i) <=
63  // F(i) for all i. Now, by defn p(k) <= t(G(k)) for all k, so G really does
64  // work.]
65  //
66  //
67  if (pattern_v_deg_seq.size() > target_v_deg_seq.size()) {
68  return false;
69  }
70  if (pattern_v_deg_seq.empty()) {
71  return true;
72  }
73 
74  // This will remain the iterator to the first elem
75  // to which we can map.
76  auto start_t_citer = target_v_deg_seq.cbegin();
77 
78  for (std::size_t next_p_index_to_dominate = 0;
79  next_p_index_to_dominate < pattern_v_deg_seq.size();
80  ++next_p_index_to_dominate) {
81  const std::size_t& p_degree = pattern_v_deg_seq[next_p_index_to_dominate];
82  const auto t_citer =
83  std::lower_bound(start_t_citer, target_v_deg_seq.cend(), p_degree);
84  if (t_citer == target_v_deg_seq.cend()) {
85  // Every "t" degree is < the "p" degree.
86  return false;
87  }
88  // t_citer exactly points towards G(k).
89  // How many t-values are still available (including this G(k) we're about to
90  // assign?)
91  const std::size_t remaining_t_values = target_v_deg_seq.cend() - t_citer;
92  const std::size_t remaining_p_values =
93  pattern_v_deg_seq.size() - next_p_index_to_dominate;
94  if (remaining_t_values < remaining_p_values) {
95  return false;
96  }
97  // A slight optim is possible: if remaining_t_values == remaining_p_values
98  // we can just go through linearly, no binary search.
99  // We could also have been fancy and searched up to somewhere before
100  // target_v_deg_seq.cend() earlier.
101  start_t_citer = t_citer + 1;
102  }
103  return true;
104 }

106 bool FilterUtils::compatible_sorted_degree_counts(
107  const DegreeCounts& degree_counts1, const DegreeCounts& degree_counts2) {
108  if (degree_counts1.empty()) {
109  return true;
110  }
111  auto counts_to_satisfy = degree_counts1.back();
112  TKET_ASSERT(counts_to_satisfy.first >= 1);
113  TKET_ASSERT(counts_to_satisfy.second >= 1);
114  if (degree_counts2.empty()) {
115  return false;
116  }
117  auto next_counts_sink = degree_counts2.back();
118  unsigned index1 = degree_counts1.size() - 1;
119  unsigned index2 = degree_counts2.size() - 1;
120 
121  // Break out when we've cleared all the pattern vertices to be matched.
122  for (;;) {
123  while (next_counts_sink.first < counts_to_satisfy.first) {
124  if (index2 == 0) {
125  // We can't move down any further.
126  return false;
127  }
128  --index2;
129  next_counts_sink = degree_counts2[index2];
130  TKET_ASSERT(next_counts_sink.first >= 1);
131  TKET_ASSERT(next_counts_sink.second >= 1);
132  }
133  if (counts_to_satisfy.second <= next_counts_sink.second) {
134  next_counts_sink.second -= counts_to_satisfy.second;
135  if (index1 == 0) {
136  break;
137  }
138  --index1;
139  counts_to_satisfy = degree_counts1[index1];
140  TKET_ASSERT(counts_to_satisfy.first >= 1);
141  TKET_ASSERT(counts_to_satisfy.second >= 1);
142  continue;
143  }
144  counts_to_satisfy.second -= next_counts_sink.second;
145  if (index2 == 0) {
146  // We can't move down any further.
147  return false;
148  }
149  --index2;
150  next_counts_sink = degree_counts2[index2];
151  TKET_ASSERT(next_counts_sink.first >= 1);
152  TKET_ASSERT(next_counts_sink.second >= 1);
153  }
154  return true;
155 }
