31ZXVertSeqSet Flow::odd(const ZXVert& v, const ZXDiagram& diag) const {
32 sequenced_map_t<ZXVert, unsigned> parities;
33 ZXVertSeqSet cv = c(v);
34 for (const ZXVert& u : cv.get<TagSeq>()) {
35 for (const ZXVert& n : diag.neighbours(u)) {
36 if (diag.get_zxtype(n) == ZXType::Output) continue;
37 sequenced_map_t<ZXVert, unsigned>::iterator found =
38 parities.get<TagKey>().find(n);
39 if (found == parities.get<TagKey>().end()) {
40 parities.insert({n, 1});
41 } else {
42 parities.replace(found, {n, found->second + 1});
43 }
44 }
45 }
46 ZXVertSeqSet odds;
47 for (const std::pair<ZXVert, unsigned>& p : parities.get<TagSeq>()) {
48 if (p.second % 2 == 1) {
49 odds.insert(p.first);
50 }
51 }
52 return odds;
53}

57void Flow::verify(const ZXDiagram& diag) const {
58 if (!diag.is_MBQC())
59 throw ZXError("Verifying a flow for a diagram that is not in MBQC form");
60 std::set<ZXVert> output_set;
61 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
62 output_set.insert(diag.neighbours(o).at(0));
63 }
64 BGL_FORALL_VERTICES(u, *diag.graph, ZXGraph) {
65 ZXType type = diag.get_zxtype(u);
66 if (is_boundary_type(type) || output_set.find(u) != output_set.end())
67 continue;
68 ZXVertSeqSet uc = c(u);
69 ZXVertSeqSet uodd = odd(u, diag);
70 for (const ZXVert& v : uc.get<TagSeq>()) {
71 ZXType vt = diag.get_zxtype(v);
72 if (u != v && vt != ZXType::PX && vt != ZXType::PY && d(u) <= d(v))
73 throw ZXError("A qubit has an X correction in its past");
74 if (u != v && vt == ZXType::PY && d(u) <= d(v) &&
75 uodd.find(v) == uodd.end())
76 throw ZXError("A past Y vertex receives an X correction");
77 }
78 for (const ZXVert& v : uodd.get<TagSeq>()) {
79 ZXType vt = diag.get_zxtype(v);
80 if (u != v && vt != ZXType::PY && vt != ZXType::PZ && d(u) <= d(v))
81 throw ZXError("A qubit has a Z correction in its past");
82 if (u != v && vt == ZXType::PY && d(u) <= d(v) && uc.find(v) == uc.end())
83 throw ZXError("A past Y vertex receives a Z correction");
84 }
85 bool self_x = (uc.find(u) != uc.end());
86 bool self_z = (uodd.find(u) != uodd.end());
87 switch (type) {
88 case ZXType::XY: {
89 if (self_x || !self_z)
90 throw ZXError("XY vertex must be corrected with a Z");
91 break;
92 }
93 case ZXType::XZ: {
94 if (!self_x || !self_z)
95 throw ZXError("XZ vertex must be corrected with a Y");
96 break;
97 }
98 case ZXType::YZ: {
99 if (!self_x || self_z)
100 throw ZXError("YZ vertex must be corrected with an X");
101 break;
102 }
103 case ZXType::PX: {
104 if (!self_z) throw ZXError("PX vertex must be corrected with a Y or Z");
105 break;
106 }
107 case ZXType::PY: {
108 if (self_x == self_z)
109 throw ZXError("PY vertex must be corrected with an X or Z");
110 break;
111 }
112 case ZXType::PZ: {
113 if (!self_x)
114 throw ZXError("PZ vertex must be corrected with an X or Y");
115 break;
116 }
117 default:
118 throw ZXError("Invalid ZXType for MBQC diagram");
119 }
120 }

123void Flow::focus(const ZXDiagram& diag) {
124 std::set<ZXVert> output_set;
125 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
126 output_set.insert(diag.neighbours(o).at(0));
127 }
128 std::map<unsigned, ZXVertVec> order;
129 for (const std::pair<const ZXVert, unsigned>& p : d_) {
130 auto found = order.find(p.second);
131 if (found == order.end())
132 order.insert({p.second, {p.first}});
133 else
134 found->second.push_back(p.first);
135 }
136
137 for (const std::pair<const unsigned, ZXVertVec>& p : order) {
138 for (const ZXVert& u : p.second) {
139 if (output_set.find(u) != output_set.end()) continue;
140 ZXVertSeqSet uc = c(u);
141 ZXVertSeqSet uodd = odd(u, diag);
142 sequenced_map_t<ZXVert, unsigned> parities;
143 for (const ZXVert& v : uc.get<TagSeq>()) parities.insert({v, 1});
144 for (const ZXVert& v : uc.get<TagSeq>()) {
145 if (v == u) continue;
146 ZXType vtype = diag.get_zxtype(v);
147 if ((vtype != ZXType::XY && vtype != ZXType::PX &&
148 vtype != ZXType::PY) ||
149 (vtype == ZXType::PY && uodd.find(v) == uodd.end())) {
150 ZXVertSeqSet cv = c(v);
151 for (const ZXVert& w : cv.get<TagSeq>()) {
152 auto found = parities.get<TagKey>().find(w);
153 if (found == parities.get<TagKey>().end())
154 parities.insert({w, 1});
155 else
156 parities.replace(found, {w, found->second + 1});
157 }
158 }
159 }
160 for (const ZXVert& v : uodd.get<TagSeq>()) {
161 if (v == u) continue;
162 ZXType vtype = diag.get_zxtype(v);
163 if ((output_set.find(v) == output_set.end() && vtype != ZXType::XZ &&
164 vtype != ZXType::YZ && vtype != ZXType::PY &&
165 vtype != ZXType::PZ) ||
166 (vtype == ZXType::PY && uc.find(v) == uc.end())) {
167 ZXVertSeqSet cv = c(v);
168 for (const ZXVert& w : cv.get<TagSeq>()) {
169 auto found = parities.get<TagKey>().find(w);
170 if (found == parities.get<TagKey>().end())
171 parities.insert({w, 1});
172 else
173 parities.replace(found, {w, found->second + 1});
174 }
175 }
176 }
177 ZXVertSeqSet new_c;
178 for (const std::pair<const ZXVert, unsigned> p : parities.get<TagSeq>()) {
179 if (p.second % 2 == 1) new_c.insert(p.first);
180 }
181 c_.at(u) = new_c;
182 }
183 }

186Flow Flow::identify_causal_flow(const ZXDiagram& diag) {
187 // Check diagram has the expected form for causal flow
188 if (!diag.is_MBQC())
189 throw ZXError("ZXDiagram must be in MBQC form to identify causal flow");
190 std::set<ZXVert> input_set;
191 for (const ZXVert& i : diag.get_boundary(ZXType::Input)) {
192 input_set.insert(diag.neighbours(i).at(0));
193 }
194 std::set<ZXVert> output_set;
195 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
196 output_set.insert(diag.neighbours(o).at(0));
197 }
198 BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
199 ZXType vtype = diag.get_zxtype(v);
200 if (!is_boundary_type(vtype) && output_set.find(v) == output_set.end() &&
201 vtype != ZXType::XY)
202 throw ZXError(
203 "Causal flow is only defined when all measured vertices are XY");
204 }
205
206 // solved contains all vertices for which we have found corrections
207 ZXVertSeqSet solved;
208 // correctors are those vertices that have been solved but are not yet
209 // fl.c(u) for some u
210 ZXVertSeqSet correctors;
211 // past[v] is undefined if v is not yet solved
212 // past[v] is the number of neighbours of v that are still unsolved
213 // When past[v] drops to 1, we can correct the unsolved vertex using an X on
214 // v and Z on all of its other neighbours
215 std::map<ZXVert, unsigned> past;
216 Flow fl{{}, {}};
217
218 // Outputs are trivially solved
219 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
220 // ZX Diagrams requires each output to have a unique edge to another vertex
221 ZXVert n = diag.neighbours(o).at(0);
222 past[n] = diag.degree(n) - 1;
223 solved.insert(o);
224 solved.insert(n);
225 fl.c_.insert({n, {}});
226 fl.d_.insert({n, 0});
227 // Add output to correctors if it is not an input
228 if (input_set.find(n) == input_set.end()) correctors.insert(n);
229 }
230
231 unsigned depth = 1;
232
233 do {
234 ZXVertSeqSet new_correctors;
235 for (const ZXVert& v : correctors.get<TagSeq>()) {
236 // Determine whether |N(v) cap unsolved| == 1 to find u
237 ZXVert u;
238 unsigned n_found = 0;
239 for (const ZXVert& vn : diag.neighbours(v)) {
240 if (solved.find(vn) == solved.end()) {
241 u = vn;
242 ++n_found;
243 }
244 }
245 if (n_found != 1) continue;
246
247 // Can correct u by firing stabilizer of v
248 fl.c_.insert({u, {v}});
249 fl.d_.insert({u, depth});
250 solved.insert(u);
251
252 // Determine any new correctors
253 n_found = 0;
254 bool in = false;
255 for (const ZXVert& un : diag.neighbours(u)) {
256 if (diag.get_zxtype(un) == ZXType::Input) {
257 in = true;
258 solved.insert(un);
259 continue;
260 }
261 if (solved.find(un) == solved.end()) {
262 ++n_found;
263 }
264 // Another neighbour of un has been solved, so check if it can now
265 // correct something
266 auto it = past.find(un);
267 if (it != past.end() && it->second > 0) {
268 --it->second;
269 if (it->second == 1) new_correctors.insert(un);
270 }
271 }
272 // u is a new corrector if u notin I and |N(u) cap unsolved| == 1
273 if (!in) {
274 past.insert({u, n_found});
275 if (n_found == 1) new_correctors.insert(u);
276 }
277 }
278 correctors = new_correctors;
279 ++depth;
280 } while (!correctors.empty());
281 if (solved.size() != diag.n_vertices())
282 throw ZXError("ZXDiagram does not have causal flow");
283 return fl;
284}

286std::map<ZXVert, ZXVertSeqSet> Flow::gauss_solve_correctors(
287 const ZXDiagram& diag, const boost::bimap<ZXVert, unsigned>& correctors,
288 const boost::bimap<ZXVert, unsigned>& preserve, const ZXVertVec& to_solve,
289 const boost::bimap<ZXVert, unsigned>& ys) {
290 unsigned n_correctors = correctors.size();
291 unsigned n_preserve = preserve.size();
292 unsigned n_to_solve = to_solve.size();
293 unsigned n_ys = ys.size();
294 MatrixXb mat = MatrixXb::Zero(n_preserve + n_ys, n_correctors + n_to_solve);
295 // Build adjacency matrix
296 for (boost::bimap<ZXVert, unsigned>::const_iterator it = correctors.begin(),
297 end = correctors.end();
298 it != end; ++it) {
299 for (const ZXVert& n : diag.neighbours(it->left)) {
300 auto in_past = preserve.left.find(n);
301 if (in_past != preserve.left.end()) {
302 mat(in_past->second, it->right) = true;
303 } else {
304 auto in_ys = ys.left.find(n);
305 if (in_ys != ys.left.end()) {
306 mat(n_preserve + in_ys->second, it->right) = true;
307 }
308 }
309 }
310 }
311 for (boost::bimap<ZXVert, unsigned>::const_iterator it = ys.begin(),
312 end = ys.end();
313 it != end; ++it) {
314 auto found = correctors.left.find(it->left);
315 if (found != correctors.left.end())
316 mat(n_preserve + it->right, found->second) = true;
317 }
318 // Add rhs
319 for (unsigned i = 0; i < n_to_solve; ++i) {
320 ZXVert v = to_solve.at(i);
321 switch (diag.get_zxtype(v)) {
322 case ZXType::XY:
323 case ZXType::PX: {
324 mat(preserve.left.at(v), n_correctors + i) = true;
325 break;
326 }
327 case ZXType::XZ: {
328 mat(preserve.left.at(v), n_correctors + i) = true;
329 }
330 // fall through
331 case ZXType::YZ:
332 case ZXType::PZ: {
333 for (const ZXVert& n : diag.neighbours(v)) {
334 auto found = preserve.left.find(n);
335 if (found != preserve.left.end())
336 mat(found->second, n_correctors + i) = true;
337 else {
338 found = ys.left.find(n);
339 if (found != ys.left.end())
340 mat(n_preserve + found->second, n_correctors + i) = true;
341 }
342 }
343 break;
344 }
345 case ZXType::PY: {
346 mat(n_preserve + ys.left.at(v), n_correctors + i) = true;
347 break;
348 }
349 default: {
350 throw ZXError(
351 "Internal error in flow identification: non-MBQC vertex found");
352 }
353 }
354 }
355
356 // Gaussian elimination
357 std::vector<std::pair<unsigned, unsigned>> row_ops =
358 gaussian_elimination_row_ops(
359 mat.block(0, 0, n_preserve + n_ys, n_correctors));
360 for (const std::pair<unsigned, unsigned>& op : row_ops) {
361 for (unsigned j = 0; j < n_correctors + n_to_solve; ++j) {
362 mat(op.second, j) ^= mat(op.first, j);
363 }
364 }
365
366 // Back substitution
367 // For each row i, pick a corrector j for which mat(i,j) == true, else
368 // determine that row i has zero lhs
369 std::map<unsigned, ZXVert> row_corrector;
370 for (unsigned i = 0; i < n_preserve + n_ys; ++i) {
371 for (unsigned j = 0; j < n_correctors; ++j) {
372 if (mat(i, j)) {
373 row_corrector.insert({i, correctors.right.at(j)});
374 break;
375 }
376 }
377 }
378 // For each past i, scan down column of rhs and for each mat(j,CI+i) == true,
379 // add corrector from row j or try next i if row j has zero lhs
380 std::map<ZXVert, ZXVertSeqSet> solved_flow;
381 for (unsigned i = 0; i < n_to_solve; ++i) {
382 bool fail = false;
383 ZXVertSeqSet c_i;
384 for (unsigned j = 0; j < n_preserve + n_ys; ++j) {
385 if (mat(j, n_correctors + i)) {
386 auto found = row_corrector.find(j);
387 if (found == row_corrector.end()) {
388 fail = true;
389 break;
390 } else {
391 c_i.insert(found->second);
392 }
393 }
394 }
395 if (!fail) {
396 ZXVert v = to_solve.at(i);
397 ZXType vt = diag.get_zxtype(v);
398 if (vt == ZXType::XZ || vt == ZXType::YZ || vt == ZXType::PZ)
399 c_i.insert(v);
400 solved_flow.insert({v, c_i});
401 }
402 }
403 return solved_flow;
404}

406Flow Flow::identify_pauli_flow(const ZXDiagram& diag) {
407 // Check diagram has the expected form for pauli flow
408 if (!diag.is_MBQC())
409 throw ZXError("ZXDiagram must be in MBQC form to identify Pauli flow");
410
411 ZXVertSeqSet solved;
412 std::set<ZXVert> inputs;
413 Flow fl{{}, {}};
414
415 // Tag input measurements
416 for (const ZXVert& i : diag.get_boundary(ZXType::Input)) {
417 ZXVert ni = diag.neighbours(i).at(0);
418 inputs.insert(ni);
419 ZXType nt = diag.get_zxtype(ni);
420 if (nt == ZXType::XZ || nt == ZXType::YZ || nt == ZXType::PZ)
421 throw ZXError(
422 "Inputs measured in XZ, YZ, or Z cannot be corrected with Pauli "
423 "flow");
424 }
425
426 // Indexing of correctors in binary matrix can be preserved between rounds as
427 // we will only ever add new correctors
428 boost::bimap<ZXVert, unsigned> correctors;
429 unsigned corrector_i = 0;
430
431 BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
432 switch (diag.get_zxtype(v)) {
433 case ZXType::Output: {
434 ZXVert n = diag.neighbours(v).at(0);
435 // Outputs are trivially solved
436 solved.insert(v);
437 if (diag.get_zxtype(n) != ZXType::Input) {
438 solved.insert(n);
439 fl.c_.insert({n, {}});
440 fl.d_.insert({n, 0});
441 }
442 // n is either an Input or PX, in which case it will be added to the
443 // correctors in the PX case
444 break;
445 }
446 case ZXType::PX:
447 case ZXType::PY: {
448 // Can use non-input Xs and Ys to correct
449 if (inputs.find(v) == inputs.end()) {
450 correctors.insert({v, corrector_i});
451 ++corrector_i;
452 }
453 break;
454 }
455 default:
456 break;
457 }
458 }
459
460 unsigned depth = 1;
461
462 unsigned n_solved = 0;
463 do {
464 // Construct Gaussian elimination problem
465 boost::bimap<ZXVert, unsigned> preserve;
466 boost::bimap<ZXVert, unsigned> unsolved_ys;
467 ZXVertVec to_solve;
468 BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
469 ZXType type = diag.get_zxtype(v);
470 if (solved.get<TagKey>().find(v) == solved.get<TagKey>().end() &&
471 type != ZXType::Input) {
472 to_solve.push_back(v);
473 if (type == ZXType::PY)
474 unsolved_ys.insert({v, (unsigned)unsolved_ys.size()});
475 else if (type != ZXType::PZ)
476 preserve.insert({v, (unsigned)preserve.size()});
477 }
478 }
479
480 std::map<ZXVert, ZXVertSeqSet> new_corrections = gauss_solve_correctors(
481 diag, correctors, preserve, to_solve, unsolved_ys);
482
483 n_solved = new_corrections.size();
484
485 for (const ZXVert& v : to_solve) {
486 auto nc = new_corrections.find(v);
487 if (nc != new_corrections.end()) {
488 fl.c_.insert(*nc);
489 fl.d_.insert({nc->first, depth});
490 solved.insert(nc->first);
491 if (inputs.find(nc->first) == inputs.end())
492 correctors.insert({nc->first, (unsigned)correctors.size()});
493 }
494 }
495
496 ++depth;
497 } while (n_solved != 0);
498
499 if (solved.size() + inputs.size() != diag.n_vertices())
500 throw ZXError("ZXDiagram does not have pauli flow");
501
502 return fl;
503}

505std::set<ZXVertSeqSet> Flow::identify_focussed_sets(const ZXDiagram& diag) {
506 // Check diagram has the expected form for pauli flow
507 if (!diag.is_MBQC())
508 throw ZXError("ZXDiagram must be in MBQC form to identify gflow");
509
510 std::set<ZXVert> inputs;
511 std::set<ZXVert> outputs;
512
513 // Tag input measurements
514 for (const ZXVert& i : diag.get_boundary(ZXType::Input)) {
515 ZXVert ni = diag.neighbours(i).at(0);
516 inputs.insert(ni);
517 ZXType nt = diag.get_zxtype(ni);
518 if (nt == ZXType::XZ || nt == ZXType::YZ || nt == ZXType::PY)
519 throw ZXError(
520 "Inputs measured in XZ, YZ, or Y cannot be corrected with Pauli "
521 "flow");
522 }
523 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
524 ZXVert no = diag.neighbours(o).at(0);
525 outputs.insert(no);
526 }
527
528 // Build Gaussian elimination problem
529 boost::bimap<ZXVert, unsigned> correctors;
530 boost::bimap<ZXVert, unsigned> preserve;
531 boost::bimap<ZXVert, unsigned> ys;
532 unsigned n_correctors = 0;
533 unsigned n_preserve = 0;
534 unsigned n_ys = 0;
535
536 BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
537 switch (diag.get_zxtype(v)) {
538 case ZXType::XY: {
539 preserve.insert({v, n_preserve});
540 ++n_preserve;
541 // Can use non-input Xs and Ys to correct
542 if (inputs.find(v) == inputs.end()) {
543 correctors.insert({v, n_correctors});
544 ++n_correctors;
545 }
546 break;
547 }
548 case ZXType::PX: {
549 // Nonmeasured vertices also covered by PX
550 // Only need to preserve measured vertices
551 if (outputs.find(v) == outputs.end()) {
552 preserve.insert({v, n_preserve});
553 ++n_preserve;
554 }
555 // Can use non-input Xs and Ys to correct
556 if (inputs.find(v) == inputs.end()) {
557 correctors.insert({v, n_correctors});
558 ++n_correctors;
559 }
560 break;
561 }
562 case ZXType::PY: {
563 ys.insert({v, n_ys});
564 ++n_ys;
565 // Can use non-input Xs and Ys to correct
566 if (inputs.find(v) == inputs.end()) {
567 correctors.insert({v, n_correctors});
568 ++n_correctors;
569 }
570 break;
571 }
572 default:
573 break;
574 }
575 }
576
577 MatrixXb mat = MatrixXb::Zero(n_preserve + n_ys, n_correctors);
578
579 // Build adjacency matrix
580 for (boost::bimap<ZXVert, unsigned>::const_iterator it = correctors.begin(),
581 end = correctors.end();
582 it != end; ++it) {
583 for (const ZXVert& n : diag.neighbours(it->left)) {
584 auto in_preserve = preserve.left.find(n);
585 if (in_preserve != preserve.left.end()) {
586 mat(in_preserve->second, it->right) = true;
587 } else {
588 auto in_ys = ys.left.find(n);
589 if (in_ys != ys.left.end()) {
590 mat(n_preserve + in_ys->second, it->right) = true;
591 }
592 }
593 }
594 }
595 for (boost::bimap<ZXVert, unsigned>::const_iterator it = ys.begin(),
596 end = ys.end();
597 it != end; ++it) {
598 auto found = correctors.left.find(it->left);
599 if (found != correctors.left.end())
600 mat(n_preserve + it->right, found->second) = true;
601 }
602
603 // Gaussian elimination
604 std::vector<std::pair<unsigned, unsigned>> row_ops =
605 gaussian_elimination_row_ops(mat);
606 for (const std::pair<unsigned, unsigned>& op : row_ops) {
607 for (unsigned j = 0; j < n_correctors; ++j) {
608 mat(op.second, j) ^= mat(op.first, j);
609 }
610 }
611
612 // Back substitution
613 // For each column j, it either a leading column (the first column for which
614 // mat(i,j) == true for a given i, so set row_corrector[i] = j; by Gaussian
615 // Elimination this is the only entry in the column) or it describes the
616 // focussed set generator {j} + {row_corrector[i] | mat(i,j) == true}
617 std::set<ZXVertSeqSet> focussed;
618 std::map<unsigned, ZXVert> row_corrector;
619 for (boost::bimap<ZXVert, unsigned>::const_iterator it = correctors.begin(),
620 end = correctors.end();
621 it != end; ++it) {
622 ZXVertSeqSet fset{it->left};
623 bool new_row_corrector = false;
624 for (unsigned i = 0; i < n_preserve + n_ys; ++i) {
625 if (mat(i, it->right)) {
626 auto inserted = row_corrector.insert({i, it->left});
627 if (inserted.second) {
628 // New row_corrector, so move to next column
