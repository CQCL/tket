62 // Focusses a flow according to Lemma B.5, Simmons "Relating Measurement

66 // Attempts to identify a causal flow for a diagram
67 // Follows Algorithm 1 from Mhalla & Perdrix "Finding Optimal Flows
68 // Efficiently" https://arxiv.org/pdf/0709.2670.pdf O(n^2 log n) for n
69 // vertices
70 static Flow identify_causal_flow(const ZXDiagram& diag);
71 // Attempts to identify a Pauli flow for a diagram
72 // Follows Algorithm 1 from Simmons "Relating Measurement Patterns to Circuits
73 // via Pauli Flow" https://arxiv.org/pdf/2109.05654.pdf O(n^4) for n vertices
74 static Flow identify_pauli_flow(const ZXDiagram& diag);
75
76 // Attempts to identify focussed sets according to Lemma B.10, Simmons
77 // "Relating Measurement Patterns to Circuits via Pauli Flow"
78 // https://arxiv.org/pdf/2109.05654.pdf
79 static std::set<ZXVertSeqSet> identify_focussed_sets(const ZXDiagram& diag);
80
81 friend Circuit tket::zx_to_circuit(const zx::ZXDiagram& diag);
82
83 // Internal state is public so binders can expose maps for easy iteration
84 // Correction sets
85 std::map<ZXVert, ZXVertSeqSet> c_;
86 // Approximate the partial order by recording the depth from outputs
87 std::map<ZXVert, unsigned> d_;
88
89 private:
90 // Solve for corrections using Gaussian elimination and back substitution
91 // Used within identify_pauli_flow
92 // correctors are those vertices which may be included in the correction sets
93 // preserve are those vertices which may not be included in the odd
94 // neighbourhood (unless being corrected) to_solve are those vertices that are

101 const boost::bimap<ZXVert, unsigned>& ys);
