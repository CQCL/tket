46 // Wires Identity gates into each cycle edge. Identity gates then relabelled
47 // with Ops from OpTypeSet to create instances of Frame Randomisation
48 void add_noop_frames(std::vector<Cycle>& cycles, Circuit& circ) {
49  std::map<Edge, Edge> replacement_rewiring_edges;
50  for (Cycle& full_cycle : cycles) {
51  std::vector<edge_pair_t> cycle = full_cycle.boundary_edges_;
52  // wire "noop" vertices into each cycle edge
53  // these are later relabelled with frame gates
54  // also wire "barrier" vertices between "noop" vertices and cycle gates
55  // i.e. something like this:
56  // ---C---
57  // |
58  // ---X---
59  // to
60  // -noop--B--C--B--noop-
61  // A | A
62  // -noop--R--X--R--noop-
63  // R R
64  // I I
65  // E E
66  // R R
67 
68  EdgeVec barrier_ins;
69  EdgeVec barrier_outs;
70  for (const std::pair<Edge, Edge>& boundary : cycle) {
71  Vertex input_noop_vert = circ.add_vertex(OpType::noop);
72  Vertex output_noop_vert = circ.add_vertex(OpType::noop);
73  Edge in_edge = boundary.first;
74  // boundary out edges can be equal to other boundary in edges
75  // rewiring a vertex into these out edges replace the edge
76  // first see if its been rewired and use new edge if necessary
77  std::map<Edge, Edge>::iterator it =
78  replacement_rewiring_edges.find(in_edge);
79  if (it != replacement_rewiring_edges.end()) {
80  in_edge = (*it).second;
81  }
82 
83  circ.rewire(input_noop_vert, {in_edge}, {EdgeType::Quantum});
84  circ.rewire(output_noop_vert, {boundary.second}, {EdgeType::Quantum});
85 
86  // Can guarantee both have one output edge only as just rewired
87  Edge input_noop_out_edge =
88  circ.get_out_edges_of_type(input_noop_vert, EdgeType::Quantum)[0];
89  Edge output_noop_in_edge =
90  circ.get_in_edges_of_type(output_noop_vert, EdgeType::Quantum)[0];
91 
92  barrier_ins.push_back(input_noop_out_edge);
93  barrier_outs.push_back(output_noop_in_edge);
94 
95  replacement_rewiring_edges[boundary.second] =
96  circ.get_out_edges_of_type(output_noop_vert, EdgeType::Quantum)[0];
97  full_cycle.add_vertex_pair({input_noop_vert, output_noop_vert});
98  }
99  std::vector<EdgeType> sig(barrier_ins.size(), EdgeType::Quantum);
100  Op_ptr o_ptr = std::make_shared<MetaOp>(OpType::Barrier, sig);
101  Vertex input_barrier_vert = circ.add_vertex(o_ptr);
102  Vertex output_barrier_vert = circ.add_vertex(o_ptr);
103  circ.rewire(input_barrier_vert, barrier_ins, sig);
104  circ.rewire(output_barrier_vert, barrier_outs, sig);
105  }
106 }

108 std::vector<std::vector<OpTypeVector>> get_all_frame_permutations(
109  const unsigned& max_frame_size, const OpTypeSet& frame_types) {
110  OpTypeVector base_otv(frame_types.begin(), frame_types.end());
111  std::sort(base_otv.begin(), base_otv.end());
112  std::vector<OpTypeVector> base_frame;
113  for (const OpType& ot : base_otv) {
114  base_frame.push_back({ot});
115  }
116  std::sort(base_frame.begin(), base_frame.end());
117  std::vector<std::vector<OpTypeVector>> out({base_frame});
118 
119  for (unsigned i = 1; i < max_frame_size; i++) {
120  std::vector<OpTypeVector> new_ops;
121  for (const OpTypeVector& base_ops : out[0]) {
122  for (const OpTypeVector& merge_ops : out[i - 1]) {
123  OpTypeVector new_base(base_ops);
124  std::copy(
125  merge_ops.begin(), merge_ops.end(), std::back_inserter(new_base));
126  new_ops.push_back(new_base);
127  }
128  }
129  out.push_back(new_ops);
130  }
131  return out;
132 }

165 std::pair<OpTypeVector, std::vector<Vertex>> FrameRandomisation::get_out_frame(
166  const OpTypeVector& in_frame, const Cycle& cycle) {
167  OpTypeVector out_frame = in_frame;
168  for (const CycleCom& cycle_op : cycle.coms_) {
169  OpTypeVector current_frame;
170  for (const unsigned& u : cycle_op.indices)
171  current_frame.push_back(out_frame[u]);
172  if (!is_initial_q_type(cycle_op.type)) {
173  current_frame =
174  (frame_cycle_conjugates_.at(cycle_op.type)).at(current_frame);
175  }
176  for (unsigned i = 0; i < current_frame.size(); i++) {
177  out_frame[cycle_op.indices[i]] = current_frame[i];
178  }
179  }
180  return {out_frame, {}};
181 }

183 std::pair<OpTypeVector, std::vector<Vertex>>
184 PauliFrameRandomisation::get_out_frame(
185  const OpTypeVector& in_frame, const Cycle& cycle) {
186  QubitPauliMap qpm;
187  for (unsigned i = 0; i < in_frame.size(); i++) {
188  switch (in_frame[i]) {
189  case OpType::noop:
190  qpm[Qubit("frame", i)] = Pauli::I;
191  break;
192  case OpType::X:
193  qpm[Qubit("frame", i)] = Pauli::X;
194  break;
195  case OpType::Y:
196  qpm[Qubit("frame", i)] = Pauli::Y;
197  break;
198  case OpType::Z:
199  qpm[Qubit("frame", i)] = Pauli::Z;
200  break;
201  default: {
202  throw FrameRandomisationError(std::string(
203  "Frame OpType " + OpDesc(in_frame[i]).name() +
204  " not a Pauli OpType."));
205  }
206  }
207  }
208 
209  QubitPauliTensor qpt(qpm);
210 
211  for (const CycleCom& cycle_op : cycle.coms_) {
212  switch (cycle_op.type) {
213  case OpType::noop:
214  case OpType::Input:
215  case OpType::Create:
216  case OpType::Output:
217  case OpType::Discard:
218  break;
219  case OpType::H:
220  case OpType::S:
221  case OpType::X:
222  case OpType::V:
223  case OpType::Z:
224  case OpType::Y:
225  conjugate_PauliTensor(
226  qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), false);
227  break;
228  case OpType::Vdg:
229  case OpType::Sdg:
230  conjugate_PauliTensor(
231  qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), true);
232  break;
233  case OpType::CX:
234  conjugate_PauliTensor(
235  qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]),
236  Qubit("frame", cycle_op.indices[1]));
237  break;
238  default: {
239  throw FrameRandomisationError(std::string(
240  "Cycle OpType " + OpDesc(cycle_op.type).name() +
241  " not supported for PauliFrameRandomisation."));
242  }
243  }
244  }
245 
246  OpTypeVector out_frame(in_frame.size());
247  for (const auto& entry : qpt.string.map) {
248  switch (entry.second) {
249  case Pauli::I:
250  out_frame[entry.first.index()[0]] = OpType::noop;
251  break;
252  case Pauli::X:
253  out_frame[entry.first.index()[0]] = OpType::X;
254  break;
255  case Pauli::Y:
256  out_frame[entry.first.index()[0]] = OpType::Y;
257  break;
258  case Pauli::Z:
259  out_frame[entry.first.index()[0]] = OpType::Z;
260  break;
261  }
262  }
263  return {out_frame, {}};
264 }

266 std::pair<OpTypeVector, std::vector<Vertex>>
267 UniversalFrameRandomisation::get_out_frame(
268  const OpTypeVector& in_frame, const Cycle& cycle) {
269  QubitPauliMap qpm;
270  std::vector<Vertex> to_dagger;
271  for (unsigned i = 0; i < in_frame.size(); i++) {
272  switch (in_frame[i]) {
273  case OpType::noop:
274  qpm[Qubit("frame", i)] = Pauli::I;
275  break;
276  case OpType::X:
277  qpm[Qubit("frame", i)] = Pauli::X;
278  break;
279  case OpType::Y:
280  qpm[Qubit("frame", i)] = Pauli::Y;
281  break;
282  case OpType::Z:
283  qpm[Qubit("frame", i)] = Pauli::Z;
284  break;
285  default: {
286  throw FrameRandomisationError(std::string(
287  "Frame OpType " + OpDesc(in_frame[i]).name() +
288  " not a Pauli OpType."));
289  }
290  }
291  }
292 
293  QubitPauliTensor qpt(qpm);
294 
295  for (const CycleCom& cycle_op : cycle.coms_) {
296  if (cycle_op.type == OpType::Rz) {
297  Pauli frame_type = qpt.string.map[Qubit("frame", cycle_op.indices[0])];
298  if (frame_type == Pauli::X || frame_type == Pauli::Y) {
299  to_dagger.push_back(cycle_op.address);
300  }
301  // don't conjugate Pauli Tensor as no frame changes for Rz
302  }
303  if (cycle_op.type == OpType::H) {
304  conjugate_PauliTensor(
305  qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), false);
306  }
307  if (cycle_op.type == OpType::CX) {
308  conjugate_PauliTensor(
309  qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]),
310  Qubit("frame", cycle_op.indices[1]));
311  }
312  }
313 
314  OpTypeVector out_frame(in_frame.size());
315  for (const auto& entry : qpt.string.map) {
316  switch (entry.second) {
317  case Pauli::I:
318  out_frame[entry.first.index()[0]] = OpType::noop;
319  break;
320  case Pauli::X:
321  out_frame[entry.first.index()[0]] = OpType::X;
322  break;
323  case Pauli::Y:
324  out_frame[entry.first.index()[0]] = OpType::Y;
325  break;
326  case Pauli::Z:
327  out_frame[entry.first.index()[0]] = OpType::Z;
328  break;
329  }
330  }
331  return {out_frame, to_dagger};
332 }

334 std::vector<Circuit> FrameRandomisation::label_frames(
335  const std::vector<std::vector<OpTypeVector>>& all_frame_ops,
336  const std::vector<Cycle>& cycles) {
337  std::vector<Circuit> output_circuit_list;
338  // make instances of all circuits using
339  for (const std::vector<OpTypeVector>& cycle_frames : all_frame_ops) {
340  std::vector<Vertex> dagger_vertices;
341  if (cycle_frames.size() != cycles.size()) {
342  throw FrameRandomisationError(
343  std::string("Length of combination of Frame Permutations does not "
344  "equal number of Cycles."));
345  }
346  for (unsigned i = 0; i < cycle_frames.size(); i++) {
347  if (cycle_frames[i].size() != cycles[i].size()) {
348  throw FrameRandomisationError(
349  std::string("Size of frame does not match the number "
350  "of qubits in Cycles."));
351  }
352  OpTypeVector in_frame = cycle_frames[i];
353  std::pair<OpTypeVector, std::vector<Vertex>> frame_and_vertices =
354  get_out_frame(in_frame, cycles[i]);
355  dagger_vertices.insert(
356  dagger_vertices.end(), frame_and_vertices.second.begin(),
357  frame_and_vertices.second.end());
358  assign_vertices(
359  in_frame, frame_and_vertices.first, cycles[i].get_frame());
360  }
361 
362  for (const Vertex& v : dagger_vertices) {
363  circuit_.set_vertex_Op_ptr(
364  v, circuit_.get_Op_ptr_from_Vertex(v)->dagger());
365  }
366  output_circuit_list.push_back(Circuit(circuit_));
367 
368  for (const Vertex& v : dagger_vertices) {
369  circuit_.set_vertex_Op_ptr(
370  v, circuit_.get_Op_ptr_from_Vertex(v)->dagger());
371  }
372  }
373  return output_circuit_list;
374 }

480 std::vector<Circuit> PowerCycle::sample_cycles(
481  const Circuit& circ, unsigned total_cycles, unsigned samples) {
482  this->circuit_ = circ;
483  std::vector<Circuit> out;
484  std::vector<Cycle> all_cycles = get_cycles(circuit_);
485  if (all_cycles.size() == 0) {
486  throw FrameRandomisationError(
487  std::string("Circuit has no gates with OpType in Clifford gates."));
488  }
489  if (all_cycles.size() > 1) {
490  throw FrameRandomisationError(
491  std::string("Circuit has non-Clifford gates."));
492  }
493 
494  add_noop_frames(all_cycles, circuit_);
495  std::vector<unsigned> frame_sizes = get_frame_sizes(all_cycles).first;
496  std::vector<std::vector<OpTypeVector>> all_samples =
497  get_all_samples(samples, frame_sizes);
498  for (const std::vector<OpTypeVector>& sample : all_samples) {
499  if (sample.size() > 1) {
500  throw FrameRandomisationError(
501  std::string("Frames have been sampled for more than one cycle."));
502  }
503  OpTypeVector in_frame = sample[0];
504  OpTypeVector noop_frame;
505  for (unsigned i = 0; i < in_frame.size(); i++) {
506  noop_frame.push_back(OpType::noop);
507  }
508  std::pair<OpTypeVector, std::vector<Vertex>> frame_and_vertices =
509  get_out_frame(in_frame, all_cycles[0]);
510  assign_vertices(
511  in_frame, frame_and_vertices.first, all_cycles[0].get_frame());
512  Circuit full_circ = circuit_;
513  for (unsigned i = 0; i < total_cycles - 1; i++) {
514  frame_and_vertices =
515  get_out_frame(frame_and_vertices.first, all_cycles[0]);
516  assign_vertices(
517  noop_frame, frame_and_vertices.first, all_cycles[0].get_frame());
518  full_circ.append(circuit_);
519  }
520  out.push_back(full_circ);
521  }
522  return out;
523 }
