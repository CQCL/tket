45// Wires Identity gates into each cycle edge. Identity gates then relabelled
46// with Ops from OpTypeSet to create instances of Frame Randomisation
47void add_noop_frames(std::vector<Cycle>& cycles, Circuit& circ) {
48 std::map<Edge, Edge> replacement_rewiring_edges;
49 for (Cycle& full_cycle : cycles) {
50 std::vector<edge_pair_t> cycle = full_cycle.boundary_edges_;
51 // wire "noop" vertices into each cycle edge
52 // these are later relabelled with frame gates
53 // also wire "barrier" vertices between "noop" vertices and cycle gates
54 // i.e. something like this:
55 // ---C---
56 // |
57 // ---X---
58 // to
59 // -noop--B--C--B--noop-
60 // A | A
61 // -noop--R--X--R--noop-
62 // R R
63 // I I
64 // E E
65 // R R
66
67 EdgeVec barrier_ins;
68 EdgeVec barrier_outs;
69 for (const std::pair<Edge, Edge>& boundary : cycle) {
70 Vertex input_noop_vert = circ.add_vertex(OpType::noop);
71 Vertex output_noop_vert = circ.add_vertex(OpType::noop);
72 Edge in_edge = boundary.first;
73 // boundary out edges can be equal to other boundary in edges
74 // rewiring a vertex into these out edges replace the edge
75 // first see if its been rewired and use new edge if necessary
76 std::map<Edge, Edge>::iterator it =
77 replacement_rewiring_edges.find(in_edge);
78 if (it != replacement_rewiring_edges.end()) {
79 in_edge = (*it).second;
80 }
81
82 // boundary in edges can be equal to other boundary out edges
83 // rewiring a vertex into these in edges replace the edge
84 // first see if its been rewired and use new edge if necessary
85 Edge out_edge = boundary.second;
86 it = replacement_rewiring_edges.find(out_edge);
87 if (it != replacement_rewiring_edges.end()) {
88 out_edge = (*it).second;
89 }
90
91 circ.rewire(input_noop_vert, {in_edge}, {EdgeType::Quantum});
92 circ.rewire(output_noop_vert, {out_edge}, {EdgeType::Quantum});
93
94 // Can guarantee both have one output edge only as just rewired
95 Edge input_noop_out_edge =
96 circ.get_out_edges_of_type(input_noop_vert, EdgeType::Quantum)[0];
97 Edge output_noop_in_edge =
98 circ.get_in_edges_of_type(output_noop_vert, EdgeType::Quantum)[0];
99
100 barrier_ins.push_back(input_noop_out_edge);
101 barrier_outs.push_back(output_noop_in_edge);
102
103 replacement_rewiring_edges.insert(
104 {out_edge,
105 circ.get_out_edges_of_type(output_noop_vert, EdgeType::Quantum)[0]});
106 replacement_rewiring_edges.insert(
107 {in_edge,
108 circ.get_in_edges_of_type(input_noop_vert, EdgeType::Quantum)[0]});
109 full_cycle.add_vertex_pair({input_noop_vert, output_noop_vert});
110 }
111 std::vector<EdgeType> sig(barrier_ins.size(), EdgeType::Quantum);
112 Op_ptr o_ptr = std::make_shared<BarrierOp>(sig);
113 Vertex input_barrier_vert = circ.add_vertex(o_ptr);
114 Vertex output_barrier_vert = circ.add_vertex(o_ptr);
115 circ.rewire(input_barrier_vert, barrier_ins, sig);
116 circ.rewire(output_barrier_vert, barrier_outs, sig);
117 }
118 std::vector<Command> commands = circ.get_commands();
119}

121std::vector<std::vector<OpTypeVector>> get_all_frame_permutations(
122 const unsigned& max_frame_size, const OpTypeSet& frame_types) {
123 OpTypeVector base_otv(frame_types.begin(), frame_types.end());
124 std::sort(base_otv.begin(), base_otv.end());
125 std::vector<OpTypeVector> base_frame;
126 for (const OpType& ot : base_otv) {
127 base_frame.push_back({ot});
128 }
129 std::sort(base_frame.begin(), base_frame.end());
130 std::vector<std::vector<OpTypeVector>> out({base_frame});
131
132 for (unsigned i = 1; i < max_frame_size; i++) {
133 std::vector<OpTypeVector> new_ops;
134 for (const OpTypeVector& base_ops : out[0]) {
135 for (const OpTypeVector& merge_ops : out[i - 1]) {
136 OpTypeVector new_base(base_ops);
137 std::copy(
138 merge_ops.begin(), merge_ops.end(), std::back_inserter(new_base));
139 new_ops.push_back(new_base);
140 }
141 }
142 out.push_back(new_ops);
143 }
144 return out;
145}

147std::vector<std::vector<OpTypeVector>> combine_vectors(
148 const std::vector<std::vector<OpTypeVector>>& vec_0,
149 const std::vector<OpTypeVector>& vec_1) {
150 std::vector<std::vector<OpTypeVector>> new_vec;
151
152 for (const std::vector<OpTypeVector>& base_frame : vec_0) {
153 for (const OpTypeVector& new_frame : vec_1) {
154 std::vector<OpTypeVector> copy = base_frame;
155 copy.push_back(new_frame);
156 new_vec.push_back(copy);
157 }
158 }
159 return new_vec;
160}

162std::vector<std::vector<OpTypeVector>> get_all_permutation_combinations(
163 const std::vector<unsigned>& frame_sizes,
164 const std::vector<std::vector<OpTypeVector>>& frame_permutations) {
165 std::vector<std::vector<OpTypeVector>> all_combinations;
166
167 for (const OpTypeVector& frame : frame_permutations[frame_sizes[0] - 1]) {
168 all_combinations.push_back({frame});
169 }
170
171 for (unsigned i = 1; i < frame_sizes.size(); i++) {
172 all_combinations = combine_vectors(
173 all_combinations, frame_permutations[frame_sizes[i] - 1]);
174 }
175 return all_combinations;
176}

178std::pair<OpTypeVector, std::vector<Vertex>> FrameRandomisation::get_out_frame(
179 const OpTypeVector& in_frame, const Cycle& cycle) {
180 OpTypeVector out_frame = in_frame;
181 for (const CycleCom& cycle_op : cycle.coms_) {
182 OpTypeVector current_frame;
183 for (const unsigned& u : cycle_op.indices)
184 current_frame.push_back(out_frame[u]);
185 if (!is_initial_q_type(cycle_op.type)) {
186 current_frame =
187 (frame_cycle_conjugates_.at(cycle_op.type)).at(current_frame);
188 }
189 for (unsigned i = 0; i < current_frame.size(); i++) {
190 out_frame[cycle_op.indices[i]] = current_frame[i];
191 }
192 }
193 return {out_frame, {}};
194}

196std::pair<OpTypeVector, std::vector<Vertex>>
197PauliFrameRandomisation::get_out_frame(
198 const OpTypeVector& in_frame, const Cycle& cycle) {
199 QubitPauliMap qpm;
200 for (unsigned i = 0; i < in_frame.size(); i++) {
201 switch (in_frame[i]) {
202 case OpType::noop:
203 qpm[Qubit("frame", i)] = Pauli::I;
204 break;
205 case OpType::X:
206 qpm[Qubit("frame", i)] = Pauli::X;
207 break;
208 case OpType::Y:
209 qpm[Qubit("frame", i)] = Pauli::Y;
210 break;
211 case OpType::Z:
212 qpm[Qubit("frame", i)] = Pauli::Z;
213 break;
214 default: {
215 throw FrameRandomisationError(
216 std::string(
217 "Frame OpType " + OpDesc(in_frame[i]).name() +
218 " not a Pauli OpType."));
219 }
220 }
221 }
222
223 SpPauliStabiliser qpt(qpm);
224
225 for (const CycleCom& cycle_op : cycle.coms_) {
226 switch (cycle_op.type) {
227 case OpType::noop:
228 case OpType::Input:
229 case OpType::Create:
230 case OpType::Output:
231 case OpType::Discard:
232 break;
233 case OpType::H:
234 case OpType::S:
235 case OpType::X:
236 case OpType::V:
237 case OpType::Z:
238 case OpType::Y:
239 conjugate_PauliTensor(
240 qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), false);
241 break;
242 case OpType::Vdg:
243 case OpType::Sdg:
244 conjugate_PauliTensor(
245 qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), true);
246 break;
247 case OpType::CX:
248 conjugate_PauliTensor(
249 qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]),
250 Qubit("frame", cycle_op.indices[1]));
251 break;
252 default: {
253 throw FrameRandomisationError(
254 std::string(
255 "Cycle OpType " + OpDesc(cycle_op.type).name() +
256 " not supported for PauliFrameRandomisation."));
257 }
258 }
259 }
260
261 OpTypeVector out_frame(in_frame.size(), OpType::noop);
262 for (const auto& entry : qpt.string) {
263 switch (entry.second) {
264 case Pauli::I:
265 out_frame[entry.first.index()[0]] = OpType::noop;
266 break;
267 case Pauli::X:
268 out_frame[entry.first.index()[0]] = OpType::X;
269 break;
270 case Pauli::Y:
271 out_frame[entry.first.index()[0]] = OpType::Y;
272 break;
273 case Pauli::Z:
274 out_frame[entry.first.index()[0]] = OpType::Z;
275 break;
276 }
277 }
278 return {out_frame, {}};
279}

281std::pair<OpTypeVector, std::vector<Vertex>>
282UniversalFrameRandomisation::get_out_frame(
283 const OpTypeVector& in_frame, const Cycle& cycle) {
284 QubitPauliMap qpm;
285 std::vector<Vertex> to_dagger;
286 for (unsigned i = 0; i < in_frame.size(); i++) {
287 switch (in_frame[i]) {
288 case OpType::noop:
289 qpm[Qubit("frame", i)] = Pauli::I;
290 break;
291 case OpType::X:
292 qpm[Qubit("frame", i)] = Pauli::X;
293 break;
294 case OpType::Y:
295 qpm[Qubit("frame", i)] = Pauli::Y;
296 break;
297 case OpType::Z:
298 qpm[Qubit("frame", i)] = Pauli::Z;
299 break;
300 default: {
301 throw FrameRandomisationError(
302 std::string(
303 "Frame OpType " + OpDesc(in_frame[i]).name() +
304 " not a Pauli OpType."));
305 }
306 }
307 }
308
309 SpPauliStabiliser qpt(qpm);
310
311 for (const CycleCom& cycle_op : cycle.coms_) {
312 if (cycle_op.type == OpType::Rz) {
313 Pauli frame_type = qpt.get(Qubit("frame", cycle_op.indices[0]));
314 if (frame_type == Pauli::X || frame_type == Pauli::Y) {
315 to_dagger.push_back(cycle_op.address);
316 }
317 // don't conjugate Pauli Tensor as no frame changes for Rz
318 }
319 if (cycle_op.type == OpType::H) {
320 conjugate_PauliTensor(
321 qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]), false);
322 }
323 if (cycle_op.type == OpType::CX) {
324 conjugate_PauliTensor(
325 qpt, cycle_op.type, Qubit("frame", cycle_op.indices[0]),
326 Qubit("frame", cycle_op.indices[1]));
327 }
328 }
329
330 OpTypeVector out_frame(in_frame.size(), OpType::noop);
331 for (const auto& entry : qpt.string) {
332 switch (entry.second) {
333 case Pauli::I:
334 out_frame[entry.first.index()[0]] = OpType::noop;
335 break;
336 case Pauli::X:
337 out_frame[entry.first.index()[0]] = OpType::X;
338 break;
339 case Pauli::Y:
340 out_frame[entry.first.index()[0]] = OpType::Y;
341 break;
342 case Pauli::Z:
343 out_frame[entry.first.index()[0]] = OpType::Z;
344 break;
345 }
346 }
347 return {out_frame, to_dagger};
348}

350std::vector<Circuit> FrameRandomisation::label_frames(
351 const std::vector<std::vector<OpTypeVector>>& all_frame_ops,
352 const std::vector<Cycle>& cycles) {
353 std::vector<Circuit> output_circuit_list;
354 // make instances of all circuits using
355 for (const std::vector<OpTypeVector>& cycle_frames : all_frame_ops) {
356 std::vector<Vertex> dagger_vertices;
357 if (cycle_frames.size() != cycles.size()) {
358 throw FrameRandomisationError(
359 std::string(
360 "Length of combination of Frame Permutations does not "
361 "equal number of Cycles."));
362 }
363 for (unsigned i = 0; i < cycle_frames.size(); i++) {
364 if (cycle_frames[i].size() != cycles[i].size()) {
365 throw FrameRandomisationError(
366 std::string(
367 "Size of frame does not match the number "
368 "of qubits in Cycles."));
369 }
370 OpTypeVector in_frame = cycle_frames[i];
371 std::pair<OpTypeVector, std::vector<Vertex>> frame_and_vertices =
372 get_out_frame(in_frame, cycles[i]);
373 dagger_vertices.insert(
374 dagger_vertices.end(), frame_and_vertices.second.begin(),
375 frame_and_vertices.second.end());
376 assign_vertices(
377 in_frame, frame_and_vertices.first, cycles[i].get_frame());
378 }
379
380 for (const Vertex& v : dagger_vertices) {
381 circuit_.set_vertex_Op_ptr(
382 v, circuit_.get_Op_ptr_from_Vertex(v)->dagger());
383 }
384 output_circuit_list.push_back(Circuit(circuit_));
385
386 for (const Vertex& v : dagger_vertices) {
387 circuit_.set_vertex_Op_ptr(
388 v, circuit_.get_Op_ptr_from_Vertex(v)->dagger());
389 }
390 }
391 return output_circuit_list;
392}

394std::pair<std::vector<unsigned>, unsigned> get_frame_sizes(
395 const std::vector<Cycle>& cycles) {
396 unsigned max_frame_size = 0;
397 std::vector<unsigned> frame_sizes;
398 for (const Cycle& cycle : cycles) {
399 unsigned frame_size = cycle.size();
400 frame_sizes.push_back(frame_size);
401 if (frame_size > max_frame_size) {
402 max_frame_size = frame_size;
403 }
404 }
405 return {frame_sizes, max_frame_size};

408std::vector<Circuit> FrameRandomisation::get_all_circuits(const Circuit& circ) {
409 // Get all cycle boundaries
410 // Get all sizes
411 // Get all possible frames for all possible cycles
412 // Apply every permutation of frames to cycles
413 // Return as vector
414 // get all cycles in circuit, wire noop vertices for relabelling into
415 // boundaries
416 this->circuit_ = circ;
417 std::vector<Cycle> all_cycles = get_cycles(circuit_);
418 if (all_cycles.size() == 0) {
419 throw FrameRandomisationError(
420 std::string("Circuit has no gates with OpType in Cycle OpTypes."));
421 }
422 add_noop_frames(all_cycles, circuit_);
423 std::pair<std::vector<unsigned>, unsigned> frame_sizes =
424 get_frame_sizes(all_cycles);
425
426 // work out all possible permutations of given ops for all frame sizes
427 std::vector<std::vector<OpTypeVector>> all_frame_perms =
428 get_all_frame_permutations(frame_sizes.second, frame_types_);
429 // combine all these permutations
430 std::vector<std::vector<OpTypeVector>> all_permutation_combinations =
431 get_all_permutation_combinations(frame_sizes.first, all_frame_perms);
432 std::vector<Circuit> output_circuit_list =
433 label_frames(all_permutation_combinations, all_cycles);
434 return output_circuit_list;
435}

457void FrameRandomisation::assign_vertices(
458 const OpTypeVector& in_frame, const OpTypeVector& out_frame,
459 const std::vector<std::pair<Vertex, Vertex>>& frame_vertices) {
460 if (in_frame.size() != out_frame.size() ||
461 in_frame.size() != frame_vertices.size()) {
462 throw FrameRandomisationError(
463 std::string(
464 "Number of gates in sampled frame doesn't match "
465 "number of qubits in frame"));
466 }
467 for (unsigned i = 0; i < frame_vertices.size(); i++) {
468 circuit_.set_vertex_Op_ptr(
469 frame_vertices[i].first, get_op_ptr(in_frame[i]));
470 circuit_.set_vertex_Op_ptr(
471 frame_vertices[i].second, get_op_ptr(out_frame[i]));
472 }

499std::vector<Circuit> PowerCycle::sample_cycles(
500 const Circuit& circ, unsigned total_cycles, unsigned samples) {
501 this->circuit_ = circ;
502 std::vector<Circuit> out;
503 std::vector<Cycle> all_cycles = get_cycles(circuit_);
504 if (all_cycles.size() == 0) {
505 throw FrameRandomisationError(
506 std::string("Circuit has no gates with OpType in Clifford gates."));
507 }
508 if (all_cycles.size() > 1) {
509 throw FrameRandomisationError(
510 std::string("Circuit has non-Clifford gates."));
511 }
512
513 add_noop_frames(all_cycles, circuit_);
514 std::vector<unsigned> frame_sizes = get_frame_sizes(all_cycles).first;
515 std::vector<std::vector<OpTypeVector>> all_samples =
516 get_all_samples(samples, frame_sizes);
517 for (const std::vector<OpTypeVector>& sample : all_samples) {
518 if (sample.size() > 1) {
519 throw FrameRandomisationError(
520 std::string("Frames have been sampled for more than one cycle."));
521 }
522 OpTypeVector in_frame = sample[0];
523 OpTypeVector noop_frame;
524 for (unsigned i = 0; i < in_frame.size(); i++) {
525 noop_frame.push_back(OpType::noop);
526 }
527 std::pair<OpTypeVector, std::vector<Vertex>> frame_and_vertices =
528 get_out_frame(in_frame, all_cycles[0]);
529 assign_vertices(
530 in_frame, frame_and_vertices.first, all_cycles[0].get_frame());
531 Circuit full_circ = circuit_;
532 for (unsigned i = 0; i < total_cycles - 1; i++) {
533 frame_and_vertices =
534 get_out_frame(frame_and_vertices.first, all_cycles[0]);
535 assign_vertices(
536 noop_frame, frame_and_vertices.first, all_cycles[0].get_frame());
537 full_circ.append(circuit_);
538 }
539 out.push_back(full_circ);
540 }
541 return out;
542}
