19 GraphPlacement::Frontier::Frontier(const Circuit& _circ) : circ(_circ) {
20  VertexVec input_slice;
21  quantum_in_edges = std::make_shared<unit_frontier_t>();
22  boolean_in_edges = std::make_shared<b_frontier_t>();
23 
24  for (const Qubit& qb : circ.all_qubits()) {
25  Vertex input = circ.get_in(qb);
26  input_slice.push_back(input);
27  Edge candidate = circ.get_nth_out_edge(input, 0);
28  quantum_in_edges->insert({qb, circ.skip_irrelevant_edges(candidate)});
29  }
30  for (const Bit& bit : circ.all_bits()) {
31  Vertex input = circ.get_in(bit);
32  EdgeVec candidates = circ.get_nth_b_out_bundle(input, 0);
33  boolean_in_edges->insert({bit, candidates});
34  }
35 
36  CutFrontier next_cut = circ.next_cut(quantum_in_edges, boolean_in_edges);
37  slice = next_cut.slice;
38  quantum_out_edges = next_cut.u_frontier;
39 }

41 void GraphPlacement::Frontier::next_slicefrontier() {
42  quantum_in_edges = std::make_shared<unit_frontier_t>();
43  boolean_in_edges = std::make_shared<b_frontier_t>();
44  for (const std::pair<UnitID, Edge>& pair : quantum_out_edges->get<TagKey>()) {
45  Edge new_e = circ.skip_irrelevant_edges(pair.second);
46  quantum_in_edges->insert({pair.first, new_e});
47  Vertex targ = circ.target(new_e);
48  EdgeVec targ_classical_ins =
49  circ.get_in_edges_of_type(targ, EdgeType::Boolean);
50  boolean_in_edges->insert(
