24struct GateNodesBuffer::Impl {

30 Impl(Eigen::MatrixXcd& matr, double abs_eps)

33 number_of_qubits(get_number_of_qubits(matr.rows())),
34 global_phase(0.0) {

36 throw std::invalid_argument("Matrix has zero cols");

40 void push(const GateNode&);

47void GateNodesBuffer::Impl::push(const GateNode& node) {
48 // Later, we might add fancy optimisation here:
49 // storing the gate for later use, looking for other compatible gates
50 // acting on the same qubits to merge with this, etc. etc.
51 node.apply_full_unitary(matrix, number_of_qubits);
52}
53
54void GateNodesBuffer::Impl::flush() {
55 if (global_phase != 0.0) {
56 const auto factor = std::polar(1.0, PI * global_phase);
57 matrix *= factor;
58 global_phase = 0.0;
59 }
60}
