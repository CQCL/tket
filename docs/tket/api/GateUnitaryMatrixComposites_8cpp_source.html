71Eigen::Matrix4cd GateUnitaryMatrixImplementations::PhasedISWAP(
72 double p, double t) {
73 auto matr = ISWAP(t);
74 const auto exp_term = std::polar(1.0, -2 * PI * p);
75 matr(2, 1) *= exp_term;
76 matr(1, 2) *= std::conj(exp_term);
77 return matr;
78}

85Eigen::Matrix2cd GateUnitaryMatrixImplementations::PhasedX(
86 double alpha, double beta) {
87 const auto rz_beta_matr = Rz(beta);
88 return rz_beta_matr * Rx(alpha) * rz_beta_matr.conjugate();
89}

91Eigen::MatrixXcd GateUnitaryMatrixImplementations::NPhasedX(
92 unsigned number_of_qubits, double alpha, double beta) {
93 const auto phasedx_matr = PhasedX(alpha, beta);
94 Eigen::MatrixXcd U = Eigen::MatrixXcd::Identity(1, 1);
95 for (unsigned i = 0; i < number_of_qubits; i++) {
96 Eigen::MatrixXcd V = Eigen::kroneckerProduct(phasedx_matr, U);
97 U = std::move(V);
98 }
99 return U;
100}
