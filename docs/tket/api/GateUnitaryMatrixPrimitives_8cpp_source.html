89 Matrix8cd GateUnitaryMatrixImplementations::XXPhase3(double alpha) {
90  Eigen::Matrix2cd PauliI, PauliX;
91  PauliI = Eigen::Matrix2cd::Identity();
92  PauliX << 0, 1, 1, 0;
93 
94  Eigen::Matrix4cd XI = Eigen::kroneckerProduct(PauliX, PauliI);
95  Eigen::Matrix4cd IX = Eigen::kroneckerProduct(PauliI, PauliX);
96  Matrix8cd XXI = Eigen::kroneckerProduct(PauliX, XI);
97  Matrix8cd IXX = Eigen::kroneckerProduct(IX, PauliX);
98  Matrix8cd XIX = Eigen::kroneckerProduct(XI, PauliX);
99 
100  Matrix8cd exponent = -0.5 * alpha * PI * i_ * (XXI + IXX + XIX);
101  Matrix8cd matr = exponent.exp();
102 
103  return matr;
104 }

134 // Returns either 0 or 1.
135 static unsigned number_of_bits_mod_2_in_binary_expansion(unsigned ii) {
136  unsigned result = 0;
137  while (ii > 0) {
138  // Standard trick: if the bits of i are abc...z10...0,
139  // then ~ii = ~(abc...z) 01...1
140  unsigned next_bit = (~ii + 1) & ii;
141  result = 1 - result;
142  ii &= ~next_bit;
143  }
144  return result;
145 }

147 Eigen::VectorXcd GateUnitaryMatrixImplementations::PhaseGadget_diagonal_entries(
148  unsigned int number_of_qubits, double alpha) {
149  // We will count the bits in the binary expansion and reduce mod 2.
150  const unsigned size = get_matrix_size(number_of_qubits);
151  Eigen::VectorXcd result(size);
152 
153  std::array<std::complex<double>, 2> diagonal_terms;
154  diagonal_terms[1] = std::polar(1.0, 0.5 * PI * alpha);
155  diagonal_terms[0] = std::conj(diagonal_terms[1]);
156 
157  for (unsigned ii = 0; ii < size; ++ii) {
158  // Could also do this iteratively/recursively,
159  // but maybe not much faster (surely no more than 20 bits, so the
160  // number_of_bits function is effectively constant time
161  // with a not too large constant)
162  result[ii] = diagonal_terms[number_of_bits_mod_2_in_binary_expansion(ii)];
163  }
164  return result;
165 }
