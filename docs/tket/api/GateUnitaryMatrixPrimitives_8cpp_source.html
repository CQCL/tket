75Eigen::Matrix4cd GateUnitaryMatrixImplementations::AAMS(
76 double theta, double phi0, double phi1) {
77 Eigen::Matrix4cd matr;
78 const double angle = 0.5 * PI * theta;
79 const double cc = cos(angle);
80 const double ss = sin(angle);
81 matr << cc, 0, 0, -i_ * std::polar(1., -PI * (phi0 + phi1)) * ss, 0, cc,
82 -i_ * std::polar(1., PI * (phi1 - phi0)) * ss, 0, 0,
83 -i_ * std::polar(1., PI * (phi0 - phi1)) * ss, cc, 0,
84 -i_ * std::polar(1., PI * (phi0 + phi1)) * ss, 0, 0, cc;
85 return matr;

97 const double angle = 0.5 * PI * alpha;
98 Eigen::Matrix4cd matr = cos(angle) * Eigen::Matrix4cd::Identity();
99 matr(3, 0) = matr(2, 1) = matr(1, 2) = matr(0, 3) = i_ * (-sin(angle));

117Matrix8cd GateUnitaryMatrixImplementations::XXPhase3(double alpha) {
118 Eigen::Matrix2cd PauliI, PauliX;
119 PauliI = Eigen::Matrix2cd::Identity();
120 PauliX << 0, 1, 1, 0;
121
122 Eigen::Matrix4cd XI = Eigen::kroneckerProduct(PauliX, PauliI);
123 Eigen::Matrix4cd IX = Eigen::kroneckerProduct(PauliI, PauliX);
124 Matrix8cd XXI = Eigen::kroneckerProduct(PauliX, XI);
125 Matrix8cd IXX = Eigen::kroneckerProduct(IX, PauliX);
126 Matrix8cd XIX = Eigen::kroneckerProduct(XI, PauliX);
127
128 Matrix8cd exponent = -0.5 * alpha * PI * i_ * (XXI + IXX + XIX);
129 Matrix8cd matr = exponent.exp();
130
131 return matr;
132}

134Eigen::Matrix4cd GateUnitaryMatrixImplementations::ESWAP(double alpha) {
135 Eigen::Matrix4cd matr = Eigen::Matrix4cd::Identity();
136 const double angle = 0.5 * PI * alpha;
137 const double cc = cos(angle);
138 const double ss = sin(angle);
139 matr(0, 0) = matr(3, 3) = std::complex(cc, -ss);
140 matr(1, 1) = matr(2, 2) = cc;
141 matr(1, 2) = matr(2, 1) = -i_ * ss;
142 return matr;

155Eigen::MatrixXcd GateUnitaryMatrixImplementations::PhaseGadget(
156 unsigned int number_of_qubits, double alpha) {
157 const auto diagonal_entries =
158 PhaseGadget_diagonal_entries(number_of_qubits, alpha);
159 return diagonal_entries.asDiagonal();
160}

162// Returns either 0 or 1.
163static unsigned number_of_bits_mod_2_in_binary_expansion(unsigned ii) {
164 unsigned result = 0;
165 while (ii > 0) {
166 // Standard trick: if the bits of i are abc...z10...0,
167 // then ~ii = ~(abc...z) 01...1
168 unsigned next_bit = (~ii + 1) & ii;
169 result = 1 - result;
170 ii &= ~next_bit;
171 }
172 return result;
173}

175Eigen::VectorXcd GateUnitaryMatrixImplementations::PhaseGadget_diagonal_entries(
176 unsigned int number_of_qubits, double alpha) {
177 // We will count the bits in the binary expansion and reduce mod 2.
178 const unsigned size = get_matrix_size(number_of_qubits);
179 Eigen::VectorXcd result(size);
180
181 std::array<std::complex<double>, 2> diagonal_terms;
182 diagonal_terms[1] = std::polar(1.0, 0.5 * PI * alpha);
183 diagonal_terms[0] = std::conj(diagonal_terms[1]);
184
185 for (unsigned ii = 0; ii < size; ++ii) {
186 // Could also do this iteratively/recursively,
187 // but maybe not much faster (surely no more than 20 bits, so the
188 // number_of_bits function is effectively constant time
189 // with a not too large constant)
190 result[ii] = diagonal_terms[number_of_bits_mod_2_in_binary_expansion(ii)];
191 }
192 return result;
193}
