33Eigen::MatrixXcd
34GateUnitaryMatrixUtils::get_multi_controlled_gate_dense_unitary(
35 const Eigen::MatrixXcd& u, unsigned int number_of_qubits) {
36 unsigned int size;
37 try {
38 size = get_matrix_size(number_of_qubits);
39 } catch (const std::exception& e) {
40 throw GateUnitaryMatrixError(
41 e.what(), GateUnitaryMatrixError::Cause::TOO_MANY_QUBITS);
42 }
43 const auto throw_with_message = [&](const std::string& message) {
44 std::stringstream ss;
45 ss << "multi_controlled_gate with " << number_of_qubits
46 << " qubits "
47 "(final matrix size "
48 << size << "x" << size
49 << "), "
50 "for unitary matrix U with "
51 << u.cols() << " cols, " << u.rows() << ": " << message;
52 throw GateUnitaryMatrixError(
53 ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
54 };
55 if (u.cols() != u.rows()) {
56 throw_with_message("matrix U not square");
57 }
58 if (u.cols() == 0) {
59 throw_with_message("zero size matrix U");
60 }
61 if (!(number_of_qubits >= 1 && size >= 2)) {
62 throw_with_message("must have at least 1 qubit");
63 }
64 if (!(size >= u.cols())) {
65 throw_with_message("input U is too large for the final number of qubits");
66 }
67 // A trick: we should check that U is of size 2^k * 2^k.
68 // But actually, we know the full size is 2^N * 2^N for N >= k,
69 // so we just check for factors.
70 const unsigned int ratio = size / u.cols();
71 if (ratio * u.cols() != size) {
72 std::stringstream ss;
73 ss << "input U number of columns is not a power of 2 (" << u.cols()
74 << " doesn't divide " << size << ")";
75 throw_with_message(ss.str());
76 }
77 Eigen::MatrixXcd matr = Eigen::MatrixXcd::Identity(size, size);
78 matr.block(size - u.cols(), size - u.cols(), u.cols(), u.cols()) = u;
79 return matr;
80}

82std::string GateUnitaryMatrixUtils::get_error_prefix(
83 const std::string& name, unsigned number_of_qubits,
84 const std::vector<double>& parameters) {
85 std::stringstream ss;
86 ss << "GateUnitaryMatrix for op " << name << " acting on " << number_of_qubits
87 << " qubits, taking " << parameters.size() << " parameters:\n";
88
89 for (unsigned nn = 0; nn < parameters.size(); ++nn) {
90 if (nn >= 10) {
91 ss << "...";
92 break;
93 }
94 ss << "param[" << nn << "] = " << parameters[nn] << "\n";

121std::vector<double> GateUnitaryMatrixUtils::get_checked_parameters(
122 const Gate& gate) {
123 const std::vector<Expr> parameter_expressions = gate.get_params();
124 const unsigned int number_of_qubits = gate.n_qubits();
125 std::vector<double> parameters(parameter_expressions.size());
126 for (unsigned nn = 0; nn < parameters.size(); ++nn) {
127 const auto optional_value = eval_expr(parameter_expressions[nn]);
128 if (!optional_value) {
129 std::stringstream ss;
130 ss << get_error_prefix(gate.get_name(), number_of_qubits, parameters)
131 << "parameter[" << nn << "] is symbolic";
132 throw GateUnitaryMatrixError(
133 ss.str(), GateUnitaryMatrixError::Cause::SYMBOLIC_PARAMETERS);
134 }
135 const double value = optional_value.value();
136 if (!std::isfinite(value)) {
137 std::stringstream ss;
138 ss << get_error_prefix(gate.get_name(), number_of_qubits, parameters)
139 << "parameter[" << nn << "] has non-finite value " << value;
140 throw GateUnitaryMatrixError(
141 ss.str(), GateUnitaryMatrixError::Cause::NON_FINITE_PARAMETER);
142 }
143 parameters[nn] = value;
144 }
145 return parameters;
146}
