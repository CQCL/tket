32 Eigen::MatrixXcd
33 GateUnitaryMatrixUtils::get_multi_controlled_gate_dense_unitary(
34  const Eigen::MatrixXcd& u, unsigned int number_of_qubits) {
35  unsigned int size;
36  try {
37  size = get_matrix_size(number_of_qubits);
38  } catch (const std::exception& e) {
39  throw GateUnitaryMatrixError(
40  e.what(), GateUnitaryMatrixError::Cause::TOO_MANY_QUBITS);
41  }
42  const auto throw_with_message = [&](const std::string& message) {
43  std::stringstream ss;
44  ss << "multi_controlled_gate with " << number_of_qubits
45  << " qubits "
46  "(final matrix size "
47  << size << "x" << size
48  << "), "
49  "for unitary matrix U with "
50  << u.cols() << " cols, " << u.rows() << ": " << message;
51  throw GateUnitaryMatrixError(
52  ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
53  };
54  if (u.cols() != u.rows()) {
55  throw_with_message("matrix U not square");
56  }
57  if (u.cols() == 0) {
58  throw_with_message("zero size matrix U");
59  }
60  if (!(number_of_qubits >= 1 && size >= 2)) {
61  throw_with_message("must have at least 1 qubit");
62  }
63  if (!(size >= u.cols())) {
64  throw_with_message("input U is too large for the final number of qubits");
65  }
66  // A trick: we should check that U is of size 2^k * 2^k.
67  // But actually, we know the full size is 2^N * 2^N for N >= k,
68  // so we just check for factors.
69  const unsigned int ratio = size / u.cols();
70  if (ratio * u.cols() != size) {
71  std::stringstream ss;
72  ss << "input U number of columns is not a power of 2 (" << u.cols()
73  << " doesn't divide " << size << ")";
74  throw_with_message(ss.str());
75  }
76  Eigen::MatrixXcd matr = Eigen::MatrixXcd::Identity(size, size);
77  matr.block(size - u.cols(), size - u.cols(), u.cols(), u.cols()) = u;
78  return matr;
79 }

120 std::vector<double> GateUnitaryMatrixUtils::get_checked_parameters(
121  const Gate& gate) {
122  const std::vector<Expr> parameter_expressions = gate.get_params();
123  const unsigned int number_of_qubits = gate.n_qubits();
124  std::vector<double> parameters(parameter_expressions.size());
125  for (unsigned nn = 0; nn < parameters.size(); ++nn) {
126  const auto optional_value = eval_expr(parameter_expressions[nn]);
127  if (!optional_value) {
128  std::stringstream ss;
129  ss << get_error_prefix(gate.get_name(), number_of_qubits, parameters)
130  << "parameter[" << nn << "] is symbolic";
131  throw GateUnitaryMatrixError(
132  ss.str(), GateUnitaryMatrixError::Cause::SYMBOLIC_PARAMETERS);
133  }
134  const double value = optional_value.value();
135  if (!std::isfinite(value)) {
136  std::stringstream ss;
137  ss << get_error_prefix(gate.get_name(), number_of_qubits, parameters)
138  << "parameter[" << nn << "] has non-finite value " << value;
139  throw GateUnitaryMatrixError(
140  ss.str(), GateUnitaryMatrixError::Cause::NON_FINITE_PARAMETER);
141  }
142  parameters[nn] = value;
143  }
144  return parameters;
145 }
