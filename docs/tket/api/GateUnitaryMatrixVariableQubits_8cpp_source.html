24 GateUnitaryMatrixVariableQubits::GateUnitaryMatrixVariableQubits(

36  break;

40  default:
41  known_type = false;

53 Eigen::MatrixXcd GateUnitaryMatrixVariableQubits::get_dense_unitary(
54  unsigned number_of_qubits, const std::vector<double>& parameters) const {
55  // This class is internal only, so an assert is OK.
56  TKET_ASSERT(known_type);
57  TKET_ASSERT(parameters.size() == number_of_parameters);
58  switch (parameters.size()) {
59  case 0:
60  TKET_ASSERT(
61  op_type == OpType::CnX || op_type == OpType::CnZ ||
62  op_type == OpType::CnY);
63  if (op_type == OpType::CnX) {
64  return GateUnitaryMatrixImplementations::CnX(number_of_qubits);
65  } else if (op_type == OpType::CnZ) {
66  return GateUnitaryMatrixImplementations::CnZ(number_of_qubits);
67  } else {
68  return GateUnitaryMatrixImplementations::CnY(number_of_qubits);
69  }
70  case 1:
71  if (op_type == OpType::CnRy) {
72  return GateUnitaryMatrixImplementations::CnRy(
73  number_of_qubits, parameters[0]);
74  } else {
75  TKET_ASSERT(op_type == OpType::PhaseGadget);
76  return GateUnitaryMatrixImplementations::PhaseGadget(
77  number_of_qubits, parameters[0]);
78  }
79  case 2:
80  TKET_ASSERT(op_type == OpType::NPhasedX);
81  return GateUnitaryMatrixImplementations::NPhasedX(
82  number_of_qubits, parameters[0], parameters[1]);
83  default:
84  TKET_ASSERT(false);
85  }
86 }
