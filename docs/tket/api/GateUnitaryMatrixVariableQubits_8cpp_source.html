24GateUnitaryMatrixVariableQubits::GateUnitaryMatrixVariableQubits(
25 OpType op_type_)
26 : op_type(op_type_), known_type(true), number_of_parameters(0) {
27 switch (op_type) {
28 case OpType::CnRy:
29 case OpType::CnRx:
30 case OpType::CnRz:
31 // Fall through.
32 case OpType::PhaseGadget:
33 number_of_parameters = 1;
34 break;
35 case OpType::CnX:
36 case OpType::CnZ:
37 case OpType::CnY:
38 break;
39 case OpType::NPhasedX:
40 number_of_parameters = 2;
41 break;
42 default:
43 known_type = false;
44 }
45}

55Eigen::MatrixXcd GateUnitaryMatrixVariableQubits::get_dense_unitary(
56 unsigned number_of_qubits, const std::vector<double>& parameters) const {
57 // This class is internal only, so an assert is OK.
58 TKET_ASSERT(known_type);
59 TKET_ASSERT(parameters.size() == number_of_parameters);
60 switch (parameters.size()) {
61 case 0:
62 TKET_ASSERT(
63 op_type == OpType::CnX || op_type == OpType::CnZ ||
64 op_type == OpType::CnY);
65 if (op_type == OpType::CnX) {
66 return GateUnitaryMatrixImplementations::CnX(number_of_qubits);
67 } else if (op_type == OpType::CnZ) {
68 return GateUnitaryMatrixImplementations::CnZ(number_of_qubits);
69 } else {
70 return GateUnitaryMatrixImplementations::CnY(number_of_qubits);
71 }
72 case 1:
73 if (op_type == OpType::CnRy) {
74 return GateUnitaryMatrixImplementations::CnRy(
75 number_of_qubits, parameters[0]);
76 } else if (op_type == OpType::CnRx) {
77 return GateUnitaryMatrixImplementations::CnRx(
78 number_of_qubits, parameters[0]);
79 } else if (op_type == OpType::CnRz) {
80 return GateUnitaryMatrixImplementations::CnRz(
81 number_of_qubits, parameters[0]);
82 } else {
83 TKET_ASSERT(op_type == OpType::PhaseGadget);
84 return GateUnitaryMatrixImplementations::PhaseGadget(
85 number_of_qubits, parameters[0]);
86 }
87 case 2:
88 TKET_ASSERT(op_type == OpType::NPhasedX);
89 return GateUnitaryMatrixImplementations::NPhasedX(
90 number_of_qubits, parameters[0], parameters[1]);
91 default:
92 TKET_ASSERT(false);
93 }
94}
