154 // It's already been checked not to be one of the special cases
155 // having a variable number of qubits.
156 static Eigen::MatrixXcd get_unitary_for_ordinary_fixed_size_case(
157  OpType op_type, unsigned number_of_qubits,
158  const std::vector<double>& parameters) {
159  const Eigen::MatrixXcd matr =
160  get_unitary_or_throw(op_type, number_of_qubits, parameters);
161 
162  TKET_ASSERT(matr.cols() == matr.rows());
163  const auto expected_number_of_qubits = get_number_of_qubits(matr.cols());
164  if (expected_number_of_qubits == number_of_qubits) {
165  return matr;
166  }
167  std::stringstream ss;
168  ss << GateUnitaryMatrixUtils::get_error_prefix(
169  op_type, number_of_qubits, parameters)
170  << "wrong number of qubits (expected " << expected_number_of_qubits << ")";
171  throw GateUnitaryMatrixError(
172  ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
173 }

175 Eigen::MatrixXcd GateUnitaryMatrix::get_unitary(
176  OpType op_type, unsigned number_of_qubits,
177  const std::vector<double>& parameters) {
178  const internal::GateUnitaryMatrixVariableQubits variable_qubits_data(op_type);
179  if (!variable_qubits_data.is_known_type()) {
180  return get_unitary_for_ordinary_fixed_size_case(
181  op_type, number_of_qubits, parameters);
182  }
183  const auto expected_number_of_parameters =
184  variable_qubits_data.get_number_of_parameters();
185  if (parameters.size() == expected_number_of_parameters) {
186  return variable_qubits_data.get_dense_unitary(number_of_qubits, parameters);
187  }
188  std::stringstream ss;
189  ss << GateUnitaryMatrixUtils::get_error_prefix(
190  op_type, number_of_qubits, parameters)
191  << "wrong number of parameters (expected " << expected_number_of_parameters
192  << ")";
193  throw GateUnitaryMatrixError(
194  ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
195 }

197 Eigen::MatrixXcd GateUnitaryMatrix::get_unitary(const Gate& gate) {
198  const auto parameters = GateUnitaryMatrixUtils::get_checked_parameters(gate);
199  return get_unitary(gate.get_type(), gate.n_qubits(), parameters);

202 std::vector<TripletCd> GateUnitaryMatrix::get_unitary_triplets(
203  const Gate& gate, double abs_epsilon) {
204  auto triplets = internal::GateUnitarySparseMatrix::get_unitary_triplets(
205  gate, abs_epsilon);
206  if (triplets.empty()) {
207  // Not recognised as a specific sparse type, so just get the dense matrix
208  const auto unitary_matr = get_unitary(gate);
209  triplets = get_triplets(unitary_matr, abs_epsilon);
210  }
211  return triplets;
212 }
