159// It's already been checked not to be one of the special cases
160// having a variable number of qubits.
161static Eigen::MatrixXcd get_unitary_for_ordinary_fixed_size_case(
162 OpType op_type, unsigned number_of_qubits,
163 const std::vector<double>& parameters) {
164 const Eigen::MatrixXcd matr =
165 get_unitary_or_throw(op_type, number_of_qubits, parameters);
166
167 TKET_ASSERT(matr.cols() == matr.rows());
168 const auto expected_number_of_qubits = get_number_of_qubits(matr.cols());
169 if (expected_number_of_qubits == number_of_qubits) {
170 return matr;
171 }
172 std::stringstream ss;
173 ss << GateUnitaryMatrixUtils::get_error_prefix(
174 op_type, number_of_qubits, parameters)
175 << "wrong number of qubits (expected " << expected_number_of_qubits << ")";
176 throw GateUnitaryMatrixError(
177 ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
178}

180Eigen::MatrixXcd GateUnitaryMatrix::get_unitary(
181 OpType op_type, unsigned number_of_qubits,
182 const std::vector<double>& parameters) {
183 const internal::GateUnitaryMatrixVariableQubits variable_qubits_data(op_type);
184 if (!variable_qubits_data.is_known_type()) {
185 return get_unitary_for_ordinary_fixed_size_case(
186 op_type, number_of_qubits, parameters);
187 }
188 const auto expected_number_of_parameters =
189 variable_qubits_data.get_number_of_parameters();
190 if (parameters.size() == expected_number_of_parameters) {
191 return variable_qubits_data.get_dense_unitary(number_of_qubits, parameters);
192 }
193 std::stringstream ss;
194 ss << GateUnitaryMatrixUtils::get_error_prefix(
195 op_type, number_of_qubits, parameters)
196 << "wrong number of parameters (expected " << expected_number_of_parameters
197 << ")";
198 throw GateUnitaryMatrixError(
199 ss.str(), GateUnitaryMatrixError::Cause::INPUT_ERROR);
200}

202Eigen::MatrixXcd GateUnitaryMatrix::get_unitary(const Gate& gate) {
203 const auto parameters = GateUnitaryMatrixUtils::get_checked_parameters(gate);
204 return get_unitary(gate.get_type(), gate.n_qubits(), parameters);
205}

207std::vector<TripletCd> GateUnitaryMatrix::get_unitary_triplets(
208 const Gate& gate, double abs_epsilon) {
209 auto triplets = internal::GateUnitarySparseMatrix::get_unitary_triplets(
210 gate, abs_epsilon);
211 if (triplets.empty()) {
212 // Not recognised as a specific sparse type, so just get the dense matrix
213 const auto unitary_matr = get_unitary(gate);
214 triplets = get_triplets(unitary_matr, abs_epsilon);
215 }
216 return triplets;
217}
