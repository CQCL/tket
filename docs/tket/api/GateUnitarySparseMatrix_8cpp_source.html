28 // Given a controlled type, which can be written as
29 // a number of control qubits applied to a more primitive type,
30 // return the primitive type. (e.g. CnX returns X).
31 // Returns "noop" if it's not a controlled type we know how to deal with.
32 // We are excluding things like, e.g., CX, which are only 4x4 and too small
33 // to be worth dealing with specially.
34 static OpType get_primitive_type(OpType type_without_controls) {
35  switch (type_without_controls) {
36  case OpType::CnX:
37  // fall through
38  case OpType::CCX:
39  return OpType::X;
40 
41  case OpType::CnZ:
42  return OpType::Z;
43 
44  case OpType::CnY:
45  return OpType::Y;
46 
47  case OpType::CnRy:
48  return OpType::Ry;
49 
50  default:
51  return OpType::noop;
52  }
53 }

55 // We have a type acting on 1 qubit.
56 // Convert this to a type acting on n qubits (given by the Gate object)
57 // by adding controls.
58 static std::vector<TripletCd>
59 convert_1qb_type_to_controlled_type_and_get_triplets(
60  const Gate& gate, OpType one_qubit_type, double abs_epsilon) {
61  const Gate new_gate(one_qubit_type, gate.get_params(), 1);
62  const auto small_unitary = GateUnitaryMatrix::get_unitary(new_gate);
63  auto triplets = get_triplets(small_unitary, abs_epsilon);
64 
65  // e.g., if CnX or CnRy for n=3, then U is 2x2, but we are embedding into
66  // the bottom right corner of an 8x8 identity matrix
67  unsigned full_matr_size = get_matrix_size(gate.n_qubits());
68  unsigned translation = full_matr_size - small_unitary.rows();
69  if (translation > 0) {
70  for (auto& triplet : triplets) {
71  triplet = TripletCd(
72  triplet.row() + translation, triplet.col() + translation,
73  triplet.value());
74  }
75  triplets.reserve(triplets.size() + translation);
76  for (unsigned ii = 0; ii < translation; ++ii) {
77  triplets.emplace_back(ii, ii, 1.0);
78  }
79  }
80  return triplets;
81 }

97 FixedTripletsWithNoParameters::FixedTripletsWithNoParameters() {
98  bridge_triplets.reserve(8);
99  cswap_triplets.reserve(8);
100  const auto& bridge_cols =
101  GateUnitaryMatrixImplementations::get_bridge_columns();
102  const auto& cswap_cols =
103  GateUnitaryMatrixImplementations::get_cswap_columns();
104  for (unsigned ii = 0; ii < 8; ++ii) {
105  bridge_triplets.emplace_back(ii, bridge_cols[ii], 1.0);
106  cswap_triplets.emplace_back(ii, cswap_cols[ii], 1.0);
107  }

122 static std::vector<TripletCd> get_phase_gadget_triplets(
123  unsigned number_of_qubits, double param) {
124  // All diagonal entries have abs value 1,
125  // so no point in abs_epsilon.
126  const auto entries =
127  GateUnitaryMatrixImplementations::PhaseGadget_diagonal_entries(
128  number_of_qubits, param);
129  std::vector<TripletCd> triplets(entries.rows());
130  for (unsigned ii = 0; ii < entries.rows(); ++ii) {
131  triplets[ii] = TripletCd(ii, ii, entries(ii));
132  }
133  return triplets;
134 }

136 static std::vector<TripletCd> get_triplets_for_noncontrolled_gate(
137  const Gate& gate) {
138  switch (gate.get_type()) {
139  case OpType::CSWAP:
140  return FixedTripletsWithNoParameters::get(gate).cswap_triplets;
141  case OpType::BRIDGE:
142  return FixedTripletsWithNoParameters::get(gate).bridge_triplets;
143  case OpType::PhaseGadget: {
144  const auto params = GateUnitaryMatrixUtils::get_checked_parameters(gate);
145  GateUnitaryMatrixUtils::check_and_throw_upon_wrong_number_of_parameters(
146  gate.get_type(), gate.n_qubits(), params, 1);
147  return get_phase_gadget_triplets(gate.n_qubits(), params[0]);
148  }
149  default:
150  return {};
151  }
152 }

154 std::vector<TripletCd> GateUnitarySparseMatrix::get_unitary_triplets(
155  const Gate& gate, double abs_epsilon) {
156  const auto type = gate.get_type();
157  const auto primitive_type = get_primitive_type(type);
158  if (primitive_type != OpType::noop) {
159  try {
160  return convert_1qb_type_to_controlled_type_and_get_triplets(
161  gate, primitive_type, abs_epsilon);
162  } catch (const GateUnitaryMatrixError& e) {
163  // GCOVR_EXCL_START
164  TKET_ASSERT(
165  AssertMessage()
166  << "Converting " << gate.get_name()
167  << " to sparse unitary, via adding controls to gate type "
168  << OpDesc(primitive_type).name() << ": " << e.what());
169  // GCOVR_EXCL_STOP
170  }
171  }
172  return get_triplets_for_noncontrolled_gate(gate);
173 }
