28// Given a controlled type, which can be written as
29// a number of control qubits applied to a more primitive type,
30// return the primitive type. (e.g. CnX returns X).
31// Returns "noop" if it's not a controlled type we know how to deal with.
32// We are excluding things like, e.g., CX, which are only 4x4 and too small
33// to be worth dealing with specially.
34static OpType get_primitive_type(OpType type_without_controls) {
35 switch (type_without_controls) {
36 case OpType::CnX:
37 // fall through
38 case OpType::CCX:
39 return OpType::X;
40
41 case OpType::CnZ:
42 return OpType::Z;
43
44 case OpType::CnY:
45 return OpType::Y;
46
47 case OpType::CnRy:
48 return OpType::Ry;
49
50 case OpType::CnRx:
51 return OpType::Rx;
52
53 case OpType::CnRz:
54 return OpType::Rz;
55
56 default:
57 return OpType::noop;
58 }
59}

61// We have a type acting on 1 qubit.
62// Convert this to a type acting on n qubits (given by the Gate object)
63// by adding controls.
64static std::vector<TripletCd>
65convert_1qb_type_to_controlled_type_and_get_triplets(
66 const Gate& gate, OpType one_qubit_type, double abs_epsilon) {
67 const Gate new_gate(one_qubit_type, gate.get_params(), 1);
68 const auto small_unitary = GateUnitaryMatrix::get_unitary(new_gate);
69 auto triplets = get_triplets(small_unitary, abs_epsilon);
70
71 // e.g., if CnX or CnRy for n=3, then U is 2x2, but we are embedding into
72 // the bottom right corner of an 8x8 identity matrix
73 unsigned full_matr_size = get_matrix_size(gate.n_qubits());
74 unsigned translation = full_matr_size - small_unitary.rows();
75 if (translation > 0) {
76 for (auto& triplet : triplets) {
77 triplet = TripletCd(
78 triplet.row() + translation, triplet.col() + translation,
79 triplet.value());
80 }
81 triplets.reserve(triplets.size() + translation);
82 for (unsigned ii = 0; ii < translation; ++ii) {
83 triplets.emplace_back(ii, ii, 1.0);
84 }
85 }
86 return triplets;
87}

90struct FixedTripletsWithNoParameters {
91 std::vector<TripletCd> bridge_triplets;
92 std::vector<TripletCd> cswap_triplets;

94 FixedTripletsWithNoParameters();

96 // It just so happens that these gates all take
97 // the same number of qubits and parameters.
98 // Therefore, test the gate for this, as well as
99 // returning the const data for use.
100 static const FixedTripletsWithNoParameters& get(const Gate& gate);
101};

103FixedTripletsWithNoParameters::FixedTripletsWithNoParameters() {
104 bridge_triplets.reserve(8);
105 cswap_triplets.reserve(8);
106 const auto& bridge_cols =
107 GateUnitaryMatrixImplementations::get_bridge_columns();
108 const auto& cswap_cols =
109 GateUnitaryMatrixImplementations::get_cswap_columns();
110 for (unsigned ii = 0; ii < 8; ++ii) {
111 bridge_triplets.emplace_back(ii, bridge_cols[ii], 1.0);
112 cswap_triplets.emplace_back(ii, cswap_cols[ii], 1.0);
113 }
114}

116const FixedTripletsWithNoParameters& FixedTripletsWithNoParameters::get(
117 const Gate& gate) {

120 GateUnitaryMatrixUtils::check_and_throw_upon_wrong_number_of_parameters(
121 gate.get_type(), gate.n_qubits(),
122 GateUnitaryMatrixUtils::get_checked_parameters(gate), 0);

128static std::vector<TripletCd> get_phase_gadget_triplets(
129 unsigned number_of_qubits, double param) {
130 // All diagonal entries have abs value 1,
131 // so no point in abs_epsilon.
132 const auto entries =
133 GateUnitaryMatrixImplementations::PhaseGadget_diagonal_entries(
134 number_of_qubits, param);
135 std::vector<TripletCd> triplets(entries.rows());
136 for (unsigned ii = 0; ii < entries.rows(); ++ii) {
137 triplets[ii] = TripletCd(ii, ii, entries(ii));
138 }
139 return triplets;
140}

142static std::vector<TripletCd> get_triplets_for_noncontrolled_gate(
143 const Gate& gate) {
144 switch (gate.get_type()) {
145 case OpType::CSWAP:
146 return FixedTripletsWithNoParameters::get(gate).cswap_triplets;
147 case OpType::BRIDGE:
148 return FixedTripletsWithNoParameters::get(gate).bridge_triplets;
149 case OpType::PhaseGadget: {
150 const auto params = GateUnitaryMatrixUtils::get_checked_parameters(gate);
151 GateUnitaryMatrixUtils::check_and_throw_upon_wrong_number_of_parameters(
152 gate.get_type(), gate.n_qubits(), params, 1);
153 return get_phase_gadget_triplets(gate.n_qubits(), params[0]);
154 }
155 default:
156 return {};
157 }
158}

160std::vector<TripletCd> GateUnitarySparseMatrix::get_unitary_triplets(
161 const Gate& gate, double abs_epsilon) {
162 const auto type = gate.get_type();
163 const auto primitive_type = get_primitive_type(type);
164 if (primitive_type != OpType::noop) {
165 try {
166 return convert_1qb_type_to_controlled_type_and_get_triplets(
167 gate, primitive_type, abs_epsilon);
168 } catch (const GateUnitaryMatrixError& e) {
169 // GCOVR_EXCL_START
170 TKET_ASSERT(
171 AssertMessage()
172 << "Converting " << gate.get_name()
173 << " to sparse unitary, via adding controls to gate type "
174 << OpDesc(primitive_type).name() << ": " << e.what());
175 // GCOVR_EXCL_STOP
176 }
177 }
178 return get_triplets_for_noncontrolled_gate(gate);
179}
