36Op_ptr Gate::dagger() const {
37 OpType optype = get_type();
38 switch (optype) {
39 case OpType::H:
40 case OpType::X:
41 case OpType::Y:
42 case OpType::Z:
43 case OpType::SWAP:
44 case OpType::CH:
45 case OpType::CX:
46 case OpType::CY:
47 case OpType::CZ:
48 case OpType::CCX:
49 case OpType::noop:
50 case OpType::CSWAP:
51 case OpType::ECR:
52 case OpType::BRIDGE: {
53 return get_op_ptr(optype);
54 }
55 case OpType::CnX:
56 case OpType::CnZ:
57 case OpType::CnY: {
58 return get_op_ptr(optype, std::vector<Expr>(), n_qubits_);
59 }
60 case OpType::S: {
61 return get_op_ptr(OpType::Sdg);
62 }
63 case OpType::Sdg: {
64 return get_op_ptr(OpType::S);
65 }
66 case OpType::T: {
67 return get_op_ptr(OpType::Tdg);
68 }
69 case OpType::Tdg: {
70 return get_op_ptr(OpType::T);
71 }
72 case OpType::V: {
73 return get_op_ptr(OpType::Vdg);
74 }
75 case OpType::Vdg: {
76 return get_op_ptr(OpType::V);
77 }
78 case OpType::CV: {
79 return get_op_ptr(OpType::CVdg);
80 }
81 case OpType::CVdg: {
82 return get_op_ptr(OpType::CV);
83 }
84 case OpType::SX: {
85 return get_op_ptr(OpType::SXdg);
86 }
87 case OpType::SXdg: {
88 return get_op_ptr(OpType::SX);
89 }
90 case OpType::CSX: {
91 return get_op_ptr(OpType::CSXdg);
92 }
93 case OpType::CSXdg: {
94 return get_op_ptr(OpType::CSX);
95 }
96 case OpType::CS: {
97 return get_op_ptr(OpType::CSdg);
98 }
99 case OpType::CSdg: {
100 return get_op_ptr(OpType::CS);
101 }
102 case OpType::Phase:
103 case OpType::CRz:
104 case OpType::CRx:
105 case OpType::CRy:
106 case OpType::CU1:
107 case OpType::U1:
108 case OpType::Rz:
109 case OpType::Ry:
110 case OpType::Rx:
111 case OpType::PhaseGadget:
112 case OpType::CnRy:
113 case OpType::CnRx:
114 case OpType::CnRz:
115 case OpType::XXPhase:
116 case OpType::YYPhase:
117 case OpType::ZZPhase:
118 case OpType::XXPhase3:
119 case OpType::ISWAP:
120 case OpType::ESWAP: {
121 return get_op_ptr(optype, minus_times(params_[0]), n_qubits_);
122 }
123 case OpType::GPI: {
124 return get_op_ptr(optype, params_[0]);
125 }
126 case OpType::GPI2: {
127 return get_op_ptr(optype, params_[0] + 1);
128 }
129 case OpType::AAMS: {
130 return get_op_ptr(optype, {params_[0], params_[1] + 1, params_[2]});
131 }
132 case OpType::ZZMax: {
133 // ZZMax.dagger = ZZPhase(-0.5)
134 return get_op_ptr(OpType::ZZPhase, -0.5);
135 }
136 case OpType::FSim: {
137 // FSim(a,b).dagger() == FSim(-a,-b)
138 return get_op_ptr(
139 OpType::FSim, {minus_times(params_[0]), minus_times(params_[1])});
140 }
141 case OpType::Sycamore: {
142 return get_op_ptr(OpType::FSim, {-0.5, -1. / 6.});
143 }
144 case OpType::ISWAPMax: {
145 return get_op_ptr(OpType::ISWAP, 3.);
146 }
147 case OpType::U2: {
148 // U2(a,b).dagger() == U3(-pi/2,-b,-a)
149 return get_op_ptr(
150 OpType::U3, {-0.5, minus_times(params_[1]), minus_times(params_[0])});
151 }
152 case OpType::U3:
153 case OpType::CU3:
154 // U3(a,b,c).dagger() == U3(-a,-c.-b)
155 {
156 return get_op_ptr(
157 optype, {minus_times(params_[0]), minus_times(params_[2]),
158 minus_times(params_[1])});
159 }
160 case OpType::TK1:
161 // TK1(a,b,c).dagger() == TK1(-c,-b,-a)
162 {
163 return get_op_ptr(
164 OpType::TK1, {minus_times(params_[2]), minus_times(params_[1]),
165 minus_times(params_[0])});
166 }
167 case OpType::TK2:
168 return get_op_ptr(
169 OpType::TK2, {minus_times(params_[0]), minus_times(params_[1]),
170 minus_times(params_[2])});
171 case OpType::PhasedX:
172 case OpType::NPhasedX:
173 // PhasedX(a,b).dagger() == PhasedX(-a,b)
174 {
175 return get_op_ptr(
176 optype, {minus_times(params_[0]), params_[1]}, n_qubits_);
177 }
178 case OpType::PhasedISWAP:
179 // PhasedISWAP(a,b).dagger() == PhasedISWAP(a,-b)
180 {
181 return get_op_ptr(
182 OpType::PhasedISWAP, {params_[0], minus_times(params_[1])});
183 }
184 default: {
185 throw BadOpType("Cannot compute dagger", optype);
186 }
187 }
188}

192 switch (optype) {
193 case OpType::Phase:
194 case OpType::H:
195 case OpType::X:
196 case OpType::Z:
197 case OpType::SWAP:
198 case OpType::CH:
199 case OpType::CX:
200 case OpType::CZ:
201 case OpType::CV:
202 case OpType::CVdg:
203 case OpType::CSX:
204 case OpType::CSXdg:
205 case OpType::CS:
206 case OpType::CSdg:
207 case OpType::CCX:
208 case OpType::noop:
209 case OpType::CSWAP:
210 case OpType::BRIDGE:
211 case OpType::S:
212 case OpType::Sdg:
213 case OpType::T:
214 case OpType::V:
215 case OpType::Vdg:
216 case OpType::SX:
217 case OpType::SXdg:
218 case OpType::CRz:
219 case OpType::CRx:
220 case OpType::CnRz:
221 case OpType::CnRx:
222 case OpType::CU1:
223 case OpType::U1:
224 case OpType::Rz:
225 case OpType::Rx:
226 case OpType::PhaseGadget:
227 case OpType::XXPhase:
228 case OpType::YYPhase:
229 case OpType::ZZPhase:
230 case OpType::TK2:
231 case OpType::XXPhase3:
232 case OpType::ESWAP:
233 case OpType::FSim: {
234 return get_op_ptr(optype, params_);
235 }
236 case OpType::Y: {
237 return get_op_ptr(OpType::U3, {3, 0.5, 0.5});
238 }
239 case OpType::Ry:
240 case OpType::CRy:
241 case OpType::CnRy: {
242 return get_op_ptr(optype, minus_times(params_[0]), n_qubits_);
243 }
244 case OpType::CnX:
245 case OpType::CnZ: {
246 return get_op_ptr(optype, std::vector<Expr>(), n_qubits_);
247 }
248 case OpType::GPI:
249 case OpType::GPI2: {
250 return get_op_ptr(optype, -params_[0]);
251 }
252 case OpType::AAMS: {
253 return get_op_ptr(optype, {params_[0], -params_[1], -params_[2]});
254 }
255 case OpType::U2: {
256 // U2(a,b).transpose() == U2(b+1,a+1)
257 return get_op_ptr(OpType::U2, {params_[1] + 1., params_[0] + 1.});
258 }
259 case OpType::U3:
260 case OpType::CU3: {
261 // U3(a,b,c).transpose() == U3(-a,c,b)
262 return get_op_ptr(
263 OpType::U3, {minus_times(params_[0]), params_[2], params_[1]});
264 }
265 case OpType::TK1: {
266 // TK1(a,b,c).transpose() == TK1(c,b,a)
267 return get_op_ptr(OpType::TK1, {params_[2], params_[1], params_[0]});
268 }
269 case OpType::PhasedX:
270 case OpType::NPhasedX: {
271 // PhasedX(a,b).transpose() == PhasedX(a,-b)
272 return get_op_ptr(
273 optype, {params_[0], minus_times(params_[1])}, n_qubits_);
274 }
275 case OpType::PhasedISWAP: {
276 // PhasedISWAP(a,b).transpose() == PhasedISWAP(-a,b)
277 return get_op_ptr(
278 OpType::PhasedISWAP, {minus_times(params_[0]), params_[1]});
279 }
280
281 default: {
282 throw BadOpType("Cannot compute transpose", optype);
283 }
284 }

287static bool params_contain_nan(const std::vector<Expr>& params) {
288 static const std::regex nan_regex("\\bnan\\b");
289 for (const Expr& e : params) {
290 stringstream ss;
291 ss << e;
292 if (std::regex_search(ss.str(), nan_regex)) return true;
293 }
294 return false;
295}

297static double random_perturbation() {
298 static RNG rng;
299 int a = rng.get_size_t(10);
300 return (a - 5) * EPS;
301}

303// If `to_doubles` is true, when an expression contains no free symbols evaluate
304// it as a double. This is appropriate in the case where an expression has been
305// perturbed, when there is no point retaining exact values for symbolic
306// constants.
307static std::vector<Expr> subs_all_params(
308 const std::vector<Expr>& params, const SymEngine::map_basic_basic& sub_map,
309 bool to_doubles = false) {
310 std::vector<Expr> new_params;
311 for (const Expr& p : params) {
312 Expr psub = p.subs(sub_map);
313 if (to_doubles && expr_free_symbols(psub).empty()) {
314 new_params.push_back(SymEngine::eval_double(psub));
315 } else {
316 new_params.push_back(psub);
317 }
318 }
319 return new_params;
320}

322Op_ptr Gate::symbol_substitution(
323 const SymEngine::map_basic_basic& sub_map) const {
324 // Perform symbolic substitution, but catch the case where the returned
325 // expression is not a number, and in that case try to set a value by
326 // perturbing the inputs. This deals with cases where expressions contain
327 // terms that are undefined at specific values but where the singularity is
328 // removable at the op level. (Non-removable singularities, which ought
329 // strictly to fail substitution, will result in spectacularly wrong values,
330 // but are unlikely to arise in practice.)
331 //
332 // This is a partial workaround for issues arising from squashing symbolic
333 // rotations, where it is hard or impossible to handle special cases (e.g.
334 // where the general formula reduces to one involving atan2(0,0)).
335 //
336 // A proper solution to this problem may have to wait for TKET 2.
337 std::vector<Expr> new_params = subs_all_params(this->params_, sub_map);
338 if (!params_contain_nan(new_params)) { // happy path
339 return get_op_ptr(this->type_, new_params, this->n_qubits_);
340 }
341
342 // Try perturbing all values in the map. May need several attempts in case
343 // there are subexpressions on the boundary of validity, such as acos(1.0). If
344 // we fail after 1000 attempts, give up.
345 for (unsigned i = 0; i < 1000; i++) {
346 SymEngine::map_basic_basic new_sub_map;
347 for (const auto& pair : sub_map) {
348 new_sub_map[pair.first] = Expr(pair.second) + random_perturbation();
349 }
350 std::vector<Expr> new_params_1 =
351 subs_all_params(this->params_, new_sub_map, true);
352 if (!params_contain_nan(new_params_1)) {
353 return get_op_ptr(this->type_, new_params_1, this->n_qubits_);
354 }
355 }
356
357 // Something really is fishy.
358 std::stringstream msg;
359 msg << "Failed to substitute values { ";
360 for (const auto& pair : sub_map) {
361 msg << Expr(pair.first) << " --> " << Expr(pair.second) << ", ";
362 }
363 msg << "} in operation " << get_name() << ".";
364 throw SubstitutionFailure(msg.str());
365}

367std::optional<double> Gate::is_identity() const {
368 static const std::optional<double> notid;
369 const std::vector<Expr>& params = get_params();
370 switch (get_type()) {
371 case OpType::noop: {
372 return 0.;
373 }
374 case OpType::Phase: {
375 // This is _always_ the identity up to phase, but the method does not
376 // allow us to return a symbolic phase, so we must reject in that case.
377 std::optional<double> eval = eval_expr(params[0]);
378 if (!eval) return notid;
379 return eval.value();
380 }
381 case OpType::Rx:
382 case OpType::Ry:
383 case OpType::Rz:
384 case OpType::PhasedX:
385 case OpType::NPhasedX:
386 case OpType::XXPhase:
387 case OpType::YYPhase:
388 case OpType::ZZPhase:
389 case OpType::XXPhase3:
390 case OpType::ESWAP:
391 case OpType::AAMS: {
392 Expr e = params[0];
393 if (equiv_0(e, 4)) {
394 return 0.;
395 } else if (equiv_0(e + 2, 4)) {
396 return 1.;
397 } else
398 return notid;
399 }
400 case OpType::U1:
401 case OpType::CU1: {
402 return equiv_0(params[0]) ? 0. : notid;
403 }
404 case OpType::U3: {
405 Expr theta = params[0];
406 if (equiv_0(params[1] + params[2])) {
407 if (equiv_0(theta, 4)) {
408 return 0.;
409 } else if (equiv_0(theta + 2, 4)) {
410 return 1.;
411 } else
412 return notid;
413 } else
414 return notid;
415 }
416 case OpType::CU3: {
417 if (equiv_0(params[0], 4) && equiv_0(params[1] + params[2])) {
418 return 0.;
419 } else
420 return notid;
421 }
422 case OpType::TK1: {
423 Expr s = params[0] + params[2], t = params[1];
424 if (equiv_0(s) && equiv_0(t)) {
425 return (equiv_0(s, 4) ^ equiv_0(t, 4)) ? 1. : 0.;
426 } else
427 return notid;
428 }
429 case OpType::TK2: {
430 bool pi_phase = false;
431 for (const Expr& a : params) {
432 if (equiv_0(a + 2, 4)) {
433 pi_phase = !pi_phase;
434 } else if (!equiv_0(a, 4)) {
435 return notid;
436 }
437 }
438 return pi_phase ? 1. : 0.;
439 }
440 case OpType::CRz:
441 case OpType::CRx:
442 case OpType::CRy:
443 case OpType::PhaseGadget:
444 case OpType::ISWAP:
445 case OpType::CnRy:
446 case OpType::CnRx:
447 case OpType::CnRz: {
448 return equiv_0(params[0], 4) ? 0. : notid;
449 }
450 case OpType::FSim: {
451 return (equiv_0(params[0]) && equiv_0(params[1])) ? 0. : notid;
452 }
453 case OpType::PhasedISWAP: {
454 return (equiv_0(params[0], 1) && equiv_0(params[1], 4)) ? 0. : notid;
455 }
456 default:
457 return notid;
458 }
459}

461bool Gate::is_clifford() const {
462 if (is_clifford_type(type_)) return true;
463
464 switch (type_) {
465 case OpType::Rx:
466 case OpType::Ry:
467 case OpType::Rz:
468 case OpType::U1:
469 case OpType::U2:
470 case OpType::U3:
471 case OpType::TK1:
472 case OpType::TK2:
473 case OpType::XXPhase:
474 case OpType::YYPhase:
475 case OpType::ZZPhase:
476 case OpType::XXPhase3:
477 case OpType::PhasedX:
478 case OpType::NPhasedX:
479 return std::all_of(params_.begin(), params_.end(), [](const Expr& e) {
480 return equiv_0(4 * e);
481 });
482 case OpType::ISWAP:
483 case OpType::ESWAP:
484 return equiv_0(2 * params_.at(0));
485 case OpType::PhasedISWAP:
486 case OpType::FSim:
487 return equiv_0(4 * params_.at(0)) && equiv_0(2 * params_.at(1));
488 case OpType::GPI:
489 return equiv_0(8 * params_.at(0));
490 case OpType::GPI2:
491 return equiv_0(4 * params_.at(0));
492 case OpType::AAMS:
493 if (equiv_0(params_.at(0))) {
494 return true;
495 } else if (
496 !equiv_0(4 * params_.at(0)) || !equiv_0(8 * params_.at(1)) ||
497 !equiv_0(8 * params_.at(2))) {
498 return false;
499 } else if (equiv_0(2 * params_.at(0))) {
500 return true;
501 } else {
502 return equiv_0(4 * params_.at(1)) && equiv_0(4 * params_.at(2));
503 }
504 default:
505 return false;
506 }
507}

509bool Gate::has_symmetry(unsigned port1, unsigned port2) const {
510 const auto n_q = n_qubits();
511 if (port1 >= n_q || port2 >= n_q) {
512 throw std::out_of_range("port ids must be less than n_qubits");
513 }
514 if (port1 == port2) {
515 // exchanging with self is always symmetric
516 return true;
517 }
518 OpType optype = get_type();
519 switch (optype) {
520 // a two qubit gate G2 is symmetric
521 // if and only if its matrix form (g2_{ij}), i,j in 0,1,2,3, satisfies
522 // g2_01 == a02, g2_10 == g2_20, g2_13 == g2_23, g2_31 == g2_32,
523 // g2_11 == g2_22, and g2_12 == g2_21. (Follows from stipulation SWAP G2
524 // SWAP == G2) If G2 is a controlled one-qubit gate, G2 = CG1. Then G1
525 // must satisfy g1_00 == 1, g1_01 == g1_10 == 0
526 case OpType::CH:
527 case OpType::CX:
528 case OpType::CY:
529 case OpType::ECR:
530 case OpType::CV:
531 case OpType::CVdg:
532 case OpType::CSX:
533 case OpType::CSXdg:
534 case OpType::CRz:
535 case OpType::CRx:
536 case OpType::CRy: {
537 // not symmetric
538 return false;
539 }
540 case OpType::SWAP:
541 case OpType::ISWAP:
542 case OpType::ESWAP:
543 case OpType::PhasedISWAP:
544 case OpType::ISWAPMax:
545 case OpType::CZ:
546 case OpType::XXPhase:
547 case OpType::YYPhase:
548 case OpType::ZZPhase:
549 case OpType::ZZMax:
550 case OpType::FSim:
551 case OpType::Sycamore:
552 case OpType::TK2:
553 case OpType::CS:
554 case OpType::CSdg:
555 case OpType::CU1: {
556 // symmetric
557 return true;
558 }
559 case OpType::CU3: {
560 // symmetric if first param theta == 0
561 const auto theta = get_params()[0];
562 return equiv_0(theta);
563 }
564 // three qubit gates
565 case OpType::CCX: {
566 // 0 <-> 1 symmetry
567 return port1 + port2 == 1; // will be 2 or 3 for non-symmetric cases
568 }
569 case OpType::CSWAP: {
570 // 1 <-> 2 symmetry
571 return port1 + port2 == 3; // will be 1 or 2 for non-symmetric cases
572 }
573 case OpType::BRIDGE: {
574 // no symmetry
575 return false;
576 }
577 case OpType::XXPhase3: {
578 // completely symmetric
579 return true;
580 }
581 // n (+1) qubit gates
582 case OpType::CnX:
583 case OpType::CnY:
584 case OpType::CnRx:
585 case OpType::CnRz:
586 case OpType::CnRy: {
587 // symmetry on first n ports not on n+1
588 auto last_port = n_q - 1;
589 return not(port1 == last_port || port2 == last_port);
590 }
591 case OpType::CnZ:
592 case OpType::PhaseGadget:
593 case OpType::NPhasedX: {
594 // symmetry on all n (+1) ports
595 return true;
596 }
597 default: {
598 return false;
599 }
600 }

603std::string Gate::get_name(bool latex) const {
604 OpDesc desc = get_desc();
605 if (params_.size() > 0) {
606 std::stringstream name;
607 if (latex) {
608 name << desc.latex() << "(";
609 } else {
610 name << desc.name() << "(";
611 }
612 for (unsigned i = 0; i < params_.size(); ++i) {
613 std::optional<double> reduced =
614 eval_expr_mod(params_[i], desc.param_mod(i));
615 if (reduced) {
616 name << reduced.value();
617 } else {
618 name << params_[i];
619 }
620 if (i < params_.size() - 1) name << ", ";
621 }
622 name << ")";
623 return name.str();
624 } else {
625 return Op::get_name(latex);
626 }
627}

648bool Gate::is_equal(const Op& op_other) const {
649 const Gate& other = dynamic_cast<const Gate&>(op_other);
650
651 OpDesc desc = get_desc();
652 if (n_qubits() != other.n_qubits()) return false;
653 std::vector<Expr> params1 = this->get_params();
654 std::vector<Expr> params2 = other.get_params();
655 unsigned param_count = params1.size();
656 if (params2.size() != param_count) return false;
657 for (unsigned i = 0; i < param_count; i++) {
658 if (!equiv_expr(params1[i], params2[i], desc.param_mod(i))) return false;
659 }
660 return true;
661}

663std::vector<Expr> Gate::get_params_reduced() const {
664 OpDesc desc = get_desc();
665 unsigned n_params = desc.n_params();
666 std::vector<Expr> params(n_params);
667 for (unsigned i = 0; i < n_params; i++) {
668 Expr param = params_[i];
669 std::optional<double> e = eval_expr_mod(param, desc.param_mod(i));
670 if (e) {
671 params[i] = e.value();
672 } else {
673 params[i] = param;
674 }
675 }
676 return params;
677}

679std::vector<Expr> Gate::get_tk1_angles() const {
680 const Expr half =
681 SymEngine::div(SymEngine::integer(1), SymEngine::integer(2));
682 const Expr quarter =
683 SymEngine::div(SymEngine::integer(1), SymEngine::integer(4));
684 const Expr eighth =
685 SymEngine::div(SymEngine::integer(1), SymEngine::integer(8));
686 switch (get_type()) {
687 case OpType::noop: {
688 return {0, 0, 0, 0};
689 }
690 case OpType::Z: {
691 return {0, 0, 1, half};
692 }
693 case OpType::X: {
694 return {0, 1, 0, half};
695 }
696 case OpType::Y: {
697 return {half, 1, -half, half};
698 }
699 case OpType::S: {
700 return {0, 0, half, quarter};
701 }
702 case OpType::Sdg: {
703 return {0, 0, -half, -quarter};
704 }
705 case OpType::T: {
706 return {0, 0, quarter, eighth};
707 }
708 case OpType::Tdg: {
709 return {0, 0, -quarter, -eighth};
710 }
711 case OpType::V: {
712 return {0, half, 0, 0};
713 }

742 (params_.at(0) + params_.at(1)) / 2};
743 }

746 params_.at(1) + half, params_.at(0), params_.at(2) - half,
747 (params_.at(1) + params_.at(2)) / 2};
748 }
749 case OpType::GPI: {
750 return {2 * params_.at(0), 1, 0, half};
751 }
752 case OpType::GPI2: {
753 return {params_.at(0), half, -params_.at(0), 0};
754 }
755 case OpType::NPhasedX: {

757 throw BadOpType(
758 "OpType::NPhasedX can only be decomposed into a TK1 "
759 "if it acts on a single qubit",
760 OpType::NPhasedX);
761 }
762 return {params_.at(1), params_.at(0), minus_times(params_.at(1)), 0};
763 }

765 return {params_.at(1), params_.at(0), minus_times(params_.at(1)), 0};
766 }
767 case OpType::TK1: {
768 return {params_.at(0), params_.at(1), params_.at(2), 0};
769 }

791static std::optional<Pauli> tk1_commuting_basis(std::vector<Expr> params) {
792 const bool p0_is_zero = equiv_0(params[0], 2);
793 const bool p1_is_zero = equiv_0(params[1], 2);
794 const bool p2_is_zero = equiv_0(params[2], 2);
795 if (p1_is_zero && equiv_expr(params[0], -params[2], 2)) {
796 return Pauli::I;
797 } else if (p1_is_zero) {
798 return Pauli::Z;
799 } else if (p0_is_zero && p2_is_zero) {
800 return Pauli::X;
801 } else {
802 return std::nullopt;
803 }
804}

818static std::optional<Pauli> tk2_commuting_basis(std::vector<Expr> params) {
819 const bool p0_is_zero = equiv_0(params[0], 2);
820 const bool p1_is_zero = equiv_0(params[1], 2);
821 const bool p2_is_zero = equiv_0(params[2], 2);
822 if (p0_is_zero && p1_is_zero && p2_is_zero) {
823 return Pauli::I;
824 } else if (p1_is_zero && p2_is_zero) {
825 return Pauli::X;
826 } else if (p0_is_zero && p2_is_zero) {
827 return Pauli::Y;
828 } else if (p0_is_zero && p1_is_zero) {
829 return Pauli::Z;
830 } else {
831 return std::nullopt;
832 }
833}

969Op_ptr Gate::deserialize(const nlohmann::json& j) {
970 OpType optype = j.at("type").get<OpType>();
971 std::vector<Expr> params;
972 if (j.contains("params")) {
973 params = j.at("params").get<std::vector<Expr>>();
974 }
975 // if type has fixed number of qubits use it, otherwise it should have been
976 // stored
977 const auto& sig = optypeinfo().at(optype).signature;
978 unsigned n_qb;
979 if (sig) {
980 auto check_quantum = [](unsigned sum, const EdgeType& e) {
981 return sum + (e == EdgeType::Quantum ? 1 : 0);
982 };
983 n_qb = std::accumulate(sig->begin(), sig->end(), 0, check_quantum);
984 } else {
985 n_qb = j.at("n_qb").get<unsigned>();
986 }
987 return get_op_ptr(optype, params, n_qb);
988}
