36 Op_ptr Gate::dagger() const {
37  OpType optype = get_type();
38  switch (optype) {
39  case OpType::H:
40  case OpType::X:
41  case OpType::Y:
42  case OpType::Z:
43  case OpType::SWAP:
44  case OpType::CH:
45  case OpType::CX:
46  case OpType::CY:
47  case OpType::CZ:
48  case OpType::CCX:
49  case OpType::noop:
50  case OpType::CSWAP:
51  case OpType::ECR:
52  case OpType::BRIDGE: {
53  return get_op_ptr(optype);
54  }
55  case OpType::CnX:
56  case OpType::CnZ:
57  case OpType::CnY: {
58  return get_op_ptr(optype, std::vector<Expr>(), n_qubits_);
59  }
60  case OpType::S: {
61  return get_op_ptr(OpType::Sdg);
62  }
63  case OpType::Sdg: {
64  return get_op_ptr(OpType::S);
65  }
66  case OpType::T: {
67  return get_op_ptr(OpType::Tdg);
68  }
69  case OpType::Tdg: {
70  return get_op_ptr(OpType::T);
71  }
72  case OpType::V: {
73  return get_op_ptr(OpType::Vdg);
74  }
75  case OpType::Vdg: {
76  return get_op_ptr(OpType::V);
77  }
78  case OpType::CV: {
79  return get_op_ptr(OpType::CVdg);
80  }
81  case OpType::CVdg: {
82  return get_op_ptr(OpType::CV);
83  }
84  case OpType::SX: {
85  return get_op_ptr(OpType::SXdg);
86  }
87  case OpType::SXdg: {
88  return get_op_ptr(OpType::SX);
89  }
90  case OpType::CSX: {
91  return get_op_ptr(OpType::CSXdg);
92  }
93  case OpType::CSXdg: {
94  return get_op_ptr(OpType::CSX);
95  }
96  case OpType::Phase:
97  case OpType::CRz:
98  case OpType::CRx:
99  case OpType::CRy:
100  case OpType::CU1:
101  case OpType::U1:
102  case OpType::Rz:
103  case OpType::Ry:
104  case OpType::Rx:
105  case OpType::PhaseGadget:
106  case OpType::CnRy:
107  case OpType::XXPhase:
108  case OpType::YYPhase:
109  case OpType::ZZPhase:
110  case OpType::XXPhase3:
111  case OpType::ISWAP:
112  case OpType::ESWAP: {
113  return get_op_ptr(optype, minus_times(params_[0]), n_qubits_);
114  }
115  case OpType::ZZMax: {
116  // ZZMax.dagger = ZZPhase(-0.5)
117  return get_op_ptr(OpType::ZZPhase, -0.5);
118  }
119  case OpType::FSim: {
120  // FSim(a,b).dagger() == FSim(-a,-b)
121  return get_op_ptr(
122  OpType::FSim, {minus_times(params_[0]), minus_times(params_[1])});
123  }
124  case OpType::Sycamore: {
125  return get_op_ptr(OpType::FSim, {-0.5, -1. / 6.});
126  }
127  case OpType::ISWAPMax: {
128  return get_op_ptr(OpType::ISWAP, 3.);
129  }
130  case OpType::U2: {
131  // U2(a,b).dagger() == U3(-pi/2,-b,-a)
132  return get_op_ptr(
133  OpType::U3, {-0.5, minus_times(params_[1]), minus_times(params_[0])});
134  }
135  case OpType::U3:
136  case OpType::CU3:
137  // U3(a,b,c).dagger() == U3(-a,-c.-b)
138  {
139  return get_op_ptr(
140  optype, {minus_times(params_[0]), minus_times(params_[2]),
141  minus_times(params_[1])});
142  }
143  case OpType::TK1:
144  // TK1(a,b,c).dagger() == TK1(-c,-b,-a)
145  {
146  return get_op_ptr(
147  OpType::TK1, {minus_times(params_[2]), minus_times(params_[1]),
148  minus_times(params_[0])});
149  }
150  case OpType::TK2:
151  return get_op_ptr(
152  OpType::TK2, {minus_times(params_[0]), minus_times(params_[1]),
153  minus_times(params_[2])});
154  case OpType::PhasedX:
155  case OpType::NPhasedX:
156  // PhasedX(a,b).dagger() == PhasedX(-a,b)
157  {
158  return get_op_ptr(
159  optype, {minus_times(params_[0]), params_[1]}, n_qubits_);
160  }
161  case OpType::PhasedISWAP:
162  // PhasedISWAP(a,b).dagger() == PhasedISWAP(a,-b)
163  {
164  return get_op_ptr(
165  OpType::PhasedISWAP, {params_[0], minus_times(params_[1])});
166  }
167  default: {
168  throw BadOpType("Cannot compute dagger", optype);
169  }
170  }

186  case OpType::CSX:
187  case OpType::CSXdg:
188  case OpType::CCX:
189  case OpType::noop:
190  case OpType::CSWAP:
191  case OpType::BRIDGE:
192  case OpType::S:
193  case OpType::Sdg:
194  case OpType::T:
195  case OpType::V:
196  case OpType::Vdg:
197  case OpType::SX:
198  case OpType::SXdg:
199  case OpType::CRz:
200  case OpType::CRx:
201  case OpType::CU1:
202  case OpType::U1:
203  case OpType::Rz:
204  case OpType::Rx:
205  case OpType::PhaseGadget:
206  case OpType::XXPhase:
207  case OpType::YYPhase:
208  case OpType::ZZPhase:
209  case OpType::TK2:
210  case OpType::XXPhase3:
211  case OpType::ESWAP:
212  case OpType::FSim: {
213  return get_op_ptr(optype, params_);
214  }
215  case OpType::Y: {

229  return get_op_ptr(OpType::U2, {params_[1] + 1., params_[0] + 1.});
230  }
231  case OpType::U3:
232  case OpType::CU3: {
233  // U3(a,b,c).transpose() == U3(-a,c,b)
234  return get_op_ptr(
235  OpType::U3, {minus_times(params_[0]), params_[2], params_[1]});
236  }
237  case OpType::TK1: {
238  // TK1(a,b,c).transpose() == TK1(c,b,a)
239  return get_op_ptr(OpType::TK1, {params_[2], params_[1], params_[0]});
240  }
241  case OpType::PhasedX:
242  case OpType::NPhasedX: {
243  // PhasedX(a,b).transpose() == PhasedX(a,-b)
244  return get_op_ptr(
245  optype, {params_[0], minus_times(params_[1])}, n_qubits_);
246  }
247  case OpType::PhasedISWAP: {
248  // PhasedISWAP(a,b).transpose() == PhasedISWAP(-a,b)
249  return get_op_ptr(
250  OpType::PhasedISWAP, {minus_times(params_[0]), params_[1]});
251  }
252 
253  default: {
254  throw BadOpType("Cannot compute transpose", optype);
255  }
256  }

259 static bool params_contain_nan(const std::vector<Expr>& params) {
260  static const std::regex nan_regex("\\bnan\\b");

262  stringstream ss;

275 // If `to_doubles` is true, when an expression contains no free symbols evaluate
276 // it as a double. This is appropriate in the case where an expression has been
277 // perturbed, when there is no point retaining exact values for symbolic
278 // constants.
279 static std::vector<Expr> subs_all_params(
280  const std::vector<Expr>& params, const SymEngine::map_basic_basic& sub_map,
281  bool to_doubles = false) {
282  std::vector<Expr> new_params;
283  for (const Expr& p : params) {
284  Expr psub = p.subs(sub_map);
285  if (to_doubles && expr_free_symbols(psub).empty()) {
286  new_params.push_back(SymEngine::eval_double(psub));
287  } else {
288  new_params.push_back(psub);
289  }
290  }
291  return new_params;
292 }

294 Op_ptr Gate::symbol_substitution(
295  const SymEngine::map_basic_basic& sub_map) const {
296  // Perform symbolic substitution, but catch the case where the returned
297  // expression is not a number, and in that case try to set a value by
298  // perturbing the inputs. This deals with cases where expressions contain
299  // terms that are undefined at specific values but where the singularity is
300  // removable at the op level. (Non-removable singularities, which ought
301  // strictly to fail substitution, will result in spectacularly wrong values,
302  // but are unlikely to arise in practice.)
303  //
304  // This is a partial workaround for issues arising from squashing symbolic
305  // rotations, where it is hard or impossible to handle special cases (e.g.
306  // where the general formula reduces to one involving atan2(0,0)).
307  //
308  // A proper solution to this problem may have to wait for TKET 2.
309  std::vector<Expr> new_params = subs_all_params(this->params_, sub_map);
310  if (!params_contain_nan(new_params)) { // happy path
311  return get_op_ptr(this->type_, new_params, this->n_qubits_);
312  }
313 
314  // Try perturbing all values in the map. May need several attempts in case
315  // there are subexpressions on the boundary of validity, such as acos(1.0). If
316  // we fail after 1000 attempts, give up.
317  for (unsigned i = 0; i < 1000; i++) {
318  SymEngine::map_basic_basic new_sub_map;
319  for (const auto& pair : sub_map) {
320  new_sub_map[pair.first] = Expr(pair.second) + random_perturbation();
321  }
322  std::vector<Expr> new_params_1 =
323  subs_all_params(this->params_, new_sub_map, true);
324  if (!params_contain_nan(new_params_1)) {
325  return get_op_ptr(this->type_, new_params_1, this->n_qubits_);
326  }
327  }
328 
329  // Something really is fishy.
330  std::stringstream msg;
331  msg << "Failed to substitute values { ";
332  for (const auto& pair : sub_map) {
333  msg << Expr(pair.first) << " --> " << Expr(pair.second) << ", ";
334  }
335  msg << "} in operation " << get_name() << ".";
336  throw SubstitutionFailure(msg.str());
337 }

339 std::optional<double> Gate::is_identity() const {
340  static const std::optional<double> notid;
341  const std::vector<Expr>& params = get_params();
342  switch (get_type()) {
343  case OpType::noop: {
344  return 0.;
345  }
346  case OpType::Phase: {
347  // This is _always_ the identity up to phase, but the method does not
348  // allow us to return a symbolic phase, so we must reject in that case.
349  std::optional<double> eval = eval_expr(params[0]);
350  if (!eval) return notid;
351  return eval.value();
352  }
353  case OpType::Rx:
354  case OpType::Ry:
355  case OpType::Rz:
356  case OpType::PhasedX:
357  case OpType::NPhasedX:
358  case OpType::XXPhase:
359  case OpType::YYPhase:
360  case OpType::ZZPhase:
361  case OpType::XXPhase3:
362  case OpType::ESWAP: {
363  Expr e = params[0];
364  if (equiv_0(e, 4)) {
365  return 0.;
366  } else if (equiv_0(e + 2, 4)) {
367  return 1.;
368  } else
369  return notid;
370  }
371  case OpType::U1:
372  case OpType::CU1: {
373  return equiv_0(params[0]) ? 0. : notid;
374  }
375  case OpType::U3: {
376  Expr theta = params[0];
377  if (equiv_0(params[1] + params[2])) {
378  if (equiv_0(theta, 4)) {
379  return 0.;
380  } else if (equiv_0(theta + 2, 4)) {
381  return 1.;
382  } else
383  return notid;
384  } else
385  return notid;
386  }
387  case OpType::CU3: {
388  if (equiv_0(params[0], 4) && equiv_0(params[1] + params[2])) {
389  return 0.;
390  } else
391  return notid;
392  }
393  case OpType::TK1: {
394  Expr s = params[0] + params[2], t = params[1];
395  if (equiv_0(s) && equiv_0(t)) {
396  return (equiv_0(s, 4) ^ equiv_0(t, 4)) ? 1. : 0.;
397  } else
398  return notid;
399  }
400  case OpType::TK2: {
401  bool pi_phase = false;
402  for (const Expr& a : params) {
403  if (equiv_0(a + 2, 4)) {
404  pi_phase = !pi_phase;
405  } else if (!equiv_0(a, 4)) {
406  return notid;
407  }
408  }
409  return pi_phase ? 1. : 0.;
410  }
411  case OpType::CRz:
412  case OpType::CRx:
413  case OpType::CRy:
414  case OpType::PhaseGadget:
415  case OpType::ISWAP:
416  case OpType::CnRy: {
417  return equiv_0(params[0], 4) ? 0. : notid;
418  }
419  case OpType::FSim: {
420  return (equiv_0(params[0]) && equiv_0(params[1])) ? 0. : notid;
421  }
422  case OpType::PhasedISWAP: {
423  return (equiv_0(params[0], 1) && equiv_0(params[1], 4)) ? 0. : notid;
424  }
425  default:
426  return notid;
427  }
428 }

430 bool Gate::is_clifford() const {
431  if (is_clifford_type(type_)) return true;
432 
433  switch (type_) {
434  case OpType::Rx:
435  case OpType::Ry:
436  case OpType::Rz:
437  case OpType::U1:
438  case OpType::U2:
439  case OpType::U3:
440  case OpType::TK1:
441  case OpType::TK2:
442  case OpType::XXPhase:
443  case OpType::YYPhase:
444  case OpType::ZZPhase:
445  case OpType::XXPhase3:
446  case OpType::PhasedX:
447  case OpType::NPhasedX:
448  return std::all_of(params_.begin(), params_.end(), [](const Expr& e) {
449  return equiv_0(4 * e);
450  });
451  case OpType::ISWAP:
452  case OpType::ESWAP:
453  return equiv_0(2 * params_.at(0));
454  case OpType::PhasedISWAP:
455  case OpType::FSim:
456  return equiv_0(4 * params_.at(0)) && equiv_0(2 * params_.at(1));
457  default:
458  return false;
459  }

462 bool Gate::has_symmetry(unsigned port1, unsigned port2) const {
463  const auto n_q = n_qubits();
464  if (port1 >= n_q || port2 >= n_q) {
465  throw std::out_of_range("port ids must be less than n_qubits");
466  }
467  if (port1 == port2) {
468  // exchanging with self is always symmetric
469  return true;
470  }
471  OpType optype = get_type();
472  switch (optype) {
473  // a two qubit gate G2 is symmetric
474  // if and only if its matrix form (g2_{ij}), i,j in 0,1,2,3, satisfies
475  // g2_01 == a02, g2_10 == g2_20, g2_13 == g2_23, g2_31 == g2_32,
476  // g2_11 == g2_22, and g2_12 == g2_21. (Follows from stipulation SWAP G2
477  // SWAP == G2) If G2 is a controlled one-qubit gate, G2 = CG1. Then G1
478  // must satisfy g1_00 == 1, g1_01 == g1_10 == 0
479  case OpType::CH:
480  case OpType::CX:
481  case OpType::CY:
482  case OpType::ECR:
483  case OpType::CV:
484  case OpType::CVdg:
485  case OpType::CSX:
486  case OpType::CSXdg:
487  case OpType::CRz:
488  case OpType::CRx:
489  case OpType::CRy: {
490  // not symmetric
491  return false;
492  }
493  case OpType::SWAP:
494  case OpType::ISWAP:
495  case OpType::ESWAP:
496  case OpType::PhasedISWAP:
497  case OpType::ISWAPMax:
498  case OpType::CZ:
499  case OpType::XXPhase:
500  case OpType::YYPhase:
501  case OpType::ZZPhase:
502  case OpType::ZZMax:
503  case OpType::FSim:
504  case OpType::Sycamore:
505  case OpType::TK2:
506  case OpType::CU1: {
507  // symmetric
508  return true;
509  }
510  case OpType::CU3: {
511  // symmetric if first param theta == 0
512  const auto theta = get_params()[0];
513  return equiv_0(theta);
514  }
515  // three qubit gates
516  case OpType::CCX: {
517  // 0 <-> 1 symmetry
518  return port1 + port2 == 1; // will be 2 or 3 for non-symmetric cases
519  }
520  case OpType::CSWAP: {
521  // 1 <-> 2 symmetry
522  return port1 + port2 == 3; // will be 1 or 2 for non-symmetric cases
523  }
524  case OpType::BRIDGE: {
525  // no symmetry
526  return false;
527  }
528  case OpType::XXPhase3: {
529  // completely symmetric
530  return true;
531  }
532  // n (+1) qubit gates
533  case OpType::CnX:
534  case OpType::CnY:
535  case OpType::CnRy: {
536  // symmetry on first n ports not on n+1
537  auto last_port = n_q - 1;
538  return not(port1 == last_port || port2 == last_port);
539  }
540  case OpType::CnZ:
541  case OpType::PhaseGadget:
542  case OpType::NPhasedX: {
543  // symmetry on all n (+1) ports
544  return true;
545  }
546  default: {
547  return false;
548  }
549  }
550 }

552 std::string Gate::get_name(bool latex) const {
553  OpDesc desc = get_desc();
554  if (params_.size() > 0) {
555  std::stringstream name;
556  if (latex) {

572  return name.str();
573  } else {

575  }

597 bool Gate::is_equal(const Op& op_other) const {
598  const Gate& other = dynamic_cast<const Gate&>(op_other);
599 
600  OpDesc desc = get_desc();
601  if (n_qubits() != other.n_qubits()) return false;
602  std::vector<Expr> params1 = this->get_params();
603  std::vector<Expr> params2 = other.get_params();
604  unsigned param_count = params1.size();
605  if (params2.size() != param_count) return false;
606  for (unsigned i = 0; i < param_count; i++) {
607  if (!equiv_expr(params1[i], params2[i], desc.param_mod(i))) return false;
608  }
609  return true;
610 }

628 std::vector<Expr> Gate::get_tk1_angles() const {
629  const Expr half =
630  SymEngine::div(SymEngine::integer(1), SymEngine::integer(2));
631  const Expr quarter =
632  SymEngine::div(SymEngine::integer(1), SymEngine::integer(4));
633  const Expr eighth =
634  SymEngine::div(SymEngine::integer(1), SymEngine::integer(8));
635  switch (get_type()) {
636  case OpType::noop: {
637  return {0, 0, 0, 0};
638  }
639  case OpType::Z: {
640  return {0, 0, 1, half};
641  }
642  case OpType::X: {
643  return {0, 1, 0, half};
644  }
645  case OpType::Y: {
646  return {half, 1, -half, half};
647  }
648  case OpType::S: {
649  return {0, 0, half, quarter};
650  }
651  case OpType::Sdg: {
652  return {0, 0, -half, -quarter};
653  }
654  case OpType::T: {
655  return {0, 0, quarter, eighth};
656  }
657  case OpType::Tdg: {
658  return {0, 0, -quarter, -eighth};

734 static std::optional<Pauli> tk1_commuting_basis(std::vector<Expr> params) {
735  const bool p0_is_zero = equiv_0(params[0], 2);
736  const bool p1_is_zero = equiv_0(params[1], 2);
737  const bool p2_is_zero = equiv_0(params[2], 2);
738  if (p1_is_zero && equiv_expr(params[0], -params[2], 2)) {
739  return Pauli::I;

743  return Pauli::X;

763  const bool p1_is_zero = equiv_0(params[1], 2);
764  const bool p2_is_zero = equiv_0(params[2], 2);
765  if (p0_is_zero && p1_is_zero && p2_is_zero) {

772  return Pauli::Z;

774  return std::nullopt;
