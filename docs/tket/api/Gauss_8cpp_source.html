49bool DiagMatrix::is_id_until_columns(unsigned limit) const {
50 TKET_ASSERT(limit <= n_rows());
51
52 for (unsigned i = 0; i < n_rows(); ++i) {
53 if (_matrix(i, i) == 0) return false;
54 }
55
56 for (unsigned i = 0; i < n_rows(); ++i) {
57 for (unsigned j = 0; j < n_cols(); ++j) {
58 if ((i > j) && (_matrix(i, j) == 1)) return false;
59 }
60 }
61
62 for (unsigned i = 0; i < n_rows(); ++i) {
63 for (unsigned j = 0; j < n_cols(); ++j) {
64 if (j > limit) {
65 if ((i < j) && (_matrix(i, j) == 1)) return false;
66 }
67 }
68 }
69
70 return true;
71}
