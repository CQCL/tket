23 std::vector<VertexWSM> get_vertices(const GraphEdgeWeights& edges_and_weights) {
24  std::set<VertexWSM> vertices;
25  for (const auto& entry : edges_and_weights) {
26  if (!(entry.first.first < entry.first.second)) {
27  throw std::runtime_error(
28  "get_vertices called on invalid raw edges_and_weights");
29  }
30  vertices.insert(entry.first.first);
31  vertices.insert(entry.first.second);
32  }
33  return {vertices.cbegin(), vertices.cend()};
34 }

36 EdgeWSM get_edge(VertexWSM v1, VertexWSM v2) {
37  if (v1 > v2) {
38  std::swap(v1, v2);
39  }
40  if (!(v1 < v2)) {
41  throw std::runtime_error(
42  std::string("get_edge on equal vertex v1=") + std::to_string(v1));
43  }
44  return std::make_pair(v1, v2);
45 }
