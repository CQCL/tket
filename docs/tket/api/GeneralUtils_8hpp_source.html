128 template <typename UINT>
129 std::optional<UINT> get_checked_sum(UINT x, UINT y) {
130  if (x == 0) {
131  return y;
132  }
133  if (y == 0) {
134  return x;
135  }
136  const UINT sum = x + y;
137  // x,y >= 1, and if n = x add y (the C add operation)
138  // then n = x+y (mod M) (as ordinary integers),
139  // with M being the max possible value.
140  // If overflow didn't occur, then n=x+y is the correct value,
141  // and so n > x (of course!)
142  //
143  // If overflow occurs, then
144  // M <= x+y <= 2M (as ordinary integers).
145  // So, if x+y = 2M then x=y=M, and then
146  // n = 0 < x.
147  // Otherwise, x+y < 2M, so that 0 <= x+y-M < M,
148  // and thus n = x+y-M = x - (M-y) (as ordinary integers).
149  // Clearly, now n <= x (and also n <= y).
150  if (sum <= x) {
151  return {};
152  }
153  return sum;
154 }

163 template <typename UINT>
164 std::optional<UINT> get_checked_product(UINT x, UINT y) {
165  switch (x) {
166  case 0:
167  return 0;
168  case 1:
169  return y;
170  default:
171  break;
172  }
173  switch (y) {
174  case 0:
175  return 0;
176  case 1:
177  return x;
178  default:
179  break;
180  }
181  if (y > std::numeric_limits<UINT>::max() / x) {
182  // By definition of unsigned integer type division for x > 0,
183  // M div x = n
184  // (where div means the C++ integer division operation)
185  // if and only if n <= M/x < n+1 (as real numbers).
186  // Therefore nx <= M, (n+1)x > M (as real numbers),
187  // so yx <= M (as real numbers) <==> y <= n.
188  //
189  // (OK, int division can be a lot slower than multiplication.
190  // But I think it's unavoidable; we cannot just inspect
191  // x MULT y directly, as for addition.
192  // E.g., if x ~ sqrt(M), y = 1.5x, x even, then
193  // xy = 1.5x^2 ~ 1.5 M will be converted to ~0.5 M,
194  // which is far bigger than both x and y).
195  return {};
196  }
197  return x * y;
198 }
