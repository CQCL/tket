61// Also updates the number of colours used in "result".
62static void colour_single_component(
63 const AdjacencyData& adjacency_data,
64 const vector<set<std::size_t>>& connected_components,
65 const vector<set<std::size_t>>& cliques, std::size_t component_index,
66 GraphColouringResult& result) {
67 result.number_of_colours =
68 std::max(result.number_of_colours, cliques[component_index].size());
69
70 const ColouringPriority colouring_priority(
71 adjacency_data, connected_components[component_index],
72 cliques[component_index]);
73
74 const BruteForceColouring brute_force_colouring(
75 colouring_priority, result.number_of_colours);
76
77 const auto& partial_colour_map = brute_force_colouring.get_colours();
78
79 for (const auto& entry : partial_colour_map) {
80 const auto& vertex = entry.first;
81 const auto& colour = entry.second;
82 result.number_of_colours = std::max(result.number_of_colours, colour + 1);
83
84 // GCOVR_EXCL_START
85 try {
86 if (vertex >= result.colours.size()) {
87 throw runtime_error("illegal vertex index");
88 }
89 auto& colour_to_assign = result.colours[vertex];
90 if (colour_to_assign < result.colours.size()) {
91 stringstream ss;
92 ss << "colour already assigned! Existing colour " << colour_to_assign;
93 throw runtime_error(ss.str());
94 }
95 colour_to_assign = colour;
96 } catch (const exception& e) {
97 TKET_ASSERT(
98 AssertMessage() << "colouring single component " << component_index
99 << " returned vertex " << vertex << " with colour "
100 << colour << " : " << e.what());
101 }
102 // GCOVR_EXCL_STOP
103 }
104}

106// Check that everything was coloured,
107// and we do have the correct number of colours
108// (we might not have used all the colours we were allowed).
109// (Don't bother trying to remove colour gaps, there shouldn't be any).
110static void check_final_colouring(GraphColouringResult& result) {
111 result.number_of_colours = 0;
112 for (std::size_t i = 0; i < result.colours.size(); ++i) {
113 const auto colour = result.colours[i];
114 // GCOVR_EXCL_START
115 if (colour >= result.colours.size()) {
116 stringstream ss;
117 ss << "vertex " << i << " has unassigned or illegal colour " << colour;
118 throw runtime_error(ss.str());
119 }
120 // GCOVR_EXCL_STOP
121 result.number_of_colours = std::max(result.number_of_colours, colour + 1);
122 }
123}

125GraphColouringResult GraphColouringRoutines::get_colouring(
126 const AdjacencyData& adjacency_data) {
127 const auto connected_components =
128 GraphRoutines::get_connected_components(adjacency_data);
129 vector<set<std::size_t>> cliques(connected_components.size());
130 vector<std::size_t> component_indices(connected_components.size());
131
132 try {
133 for (std::size_t i = 0; i < connected_components.size(); ++i) {
134 const LargeCliquesResult cliques_in_this_component(
135 adjacency_data, connected_components[i]);
136
137 // GCOVR_EXCL_START
138 if (cliques_in_this_component.cliques.empty()) {
139 stringstream ss;
140 ss << "component " << i << " has " << connected_components[i].size()
141 << " vertices, but couldn't find a clique!";
142 throw runtime_error(ss.str());
143 }
144 // GCOVR_EXCL_STOP
145 cliques[i] = cliques_in_this_component.cliques[0];
146 component_indices[i] = i;
147 }
148
149 // Now, we might as well start with the component containing the LARGEST
150 // clique first (since, colouring becomes easier with more colours, and once
151 // we've coloured one component, no point in trying to colour the others
152 // with fewer colours).
153 std::sort(
154 component_indices.begin(), component_indices.end(),
155 [&cliques](std::size_t lhs, std::size_t rhs) {
156 return cliques[lhs].size() > cliques[rhs].size();
157 });
158
159 GraphColouringResult result;
160 result.colours.assign(
161 adjacency_data.get_number_of_vertices(),
162 std::numeric_limits<std::size_t>::max());
163
164 for (auto component_index : component_indices) {
165 colour_single_component(
166 adjacency_data, connected_components, cliques, component_index,
167 result);
168 }
169 check_final_colouring(result);
170 return result;
171 } catch (const exception& e) {
172 // GCOVR_EXCL_START
173 TKET_ASSERT(
174 AssertMessage() << "We had " << connected_components.size()
175 << " connected components, "
176 << adjacency_data.get_number_of_vertices()
177 << " vertices in total: " << e.what());
178 // Some compilers error with "non-void function does not
179 // return a value in all control paths..."
180 return GraphColouringResult();
181 // GCOVR_EXCL_STOP
182 }
183}
