28vector<set<std::size_t>> GraphRoutines::get_connected_components(
29 const AdjacencyData& adjacency_data) {
30 set<std::size_t> vertices_seen;
31 vector<set<std::size_t>> result;
32 const std::size_t number_of_vertices =
33 adjacency_data.get_number_of_vertices();
34
35 for (std::size_t v = 0; v < number_of_vertices; ++v) {
36 if (vertices_seen.count(v) != 0) {
37 continue;
38 }
39 set<std::size_t> vertices_seen_in_this_component;
40 vertices_seen_in_this_component.insert(v);
41
42 std::stack<std::size_t> vertices_to_examine_next;
43 vertices_to_examine_next.push(v);
44
45 while (!vertices_to_examine_next.empty()) {
46 const std::size_t this_vertex = vertices_to_examine_next.top();
47 vertices_to_examine_next.pop();
48 const auto& neighbouring_vertices =
49 adjacency_data.get_neighbours(this_vertex);
50
51 for (std::size_t j : neighbouring_vertices) {
52 if (vertices_seen_in_this_component.count(j) == 0) {
53 vertices_to_examine_next.push(j);
54 vertices_seen_in_this_component.insert(j);
55 }
56 }
57 }
58 // Now, push back the vertices seen.
59 result.emplace_back(vertices_seen_in_this_component);
60 for (std::size_t k : vertices_seen_in_this_component) {
61 vertices_seen.insert(k);
62 }
63 }
64 return result;
65}
