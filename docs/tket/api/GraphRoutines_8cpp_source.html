29 vector<set<std::size_t>> GraphRoutines::get_connected_components(
30  const AdjacencyData& adjacency_data) {
31  set<std::size_t> vertices_seen;
32  vector<set<std::size_t>> result;
33  const std::size_t number_of_vertices =
34  adjacency_data.get_number_of_vertices();
35 
36  for (std::size_t v = 0; v < number_of_vertices; ++v) {
37  if (vertices_seen.count(v) != 0) {
38  continue;
39  }
40  set<std::size_t> vertices_seen_in_this_component;
41  vertices_seen_in_this_component.insert(v);
42 
43  std::stack<std::size_t> vertices_to_examine_next;
44  vertices_to_examine_next.push(v);
45 
46  while (!vertices_to_examine_next.empty()) {
47  const std::size_t this_vertex = vertices_to_examine_next.top();
48  vertices_to_examine_next.pop();
49  const auto& neighbouring_vertices =
50  adjacency_data.get_neighbours(this_vertex);
51 
52  for (std::size_t j : neighbouring_vertices) {
53  if (vertices_seen_in_this_component.count(j) == 0) {
54  vertices_to_examine_next.push(j);
55  vertices_seen_in_this_component.insert(j);
56  }
57  }
58  }
59  // Now, push back the vertices seen.
60  result.emplace_back(vertices_seen_in_this_component);
61  for (std::size_t k : vertices_seen_in_this_component) {
62  vertices_seen.insert(k);
63  }
64  }
65  return result;
66 }
