28// convert a Clifford tableau to a vector of PauliNode_ptr
29static std::vector<PauliNode_ptr> get_nodes_from_tableau(
30 const UnitaryRevTableau& tab, unsigned n_qubits) {
31 std::vector<PauliNode_ptr> rows;
32 for (unsigned i = 0; i < n_qubits; i++) {
33 Qubit q(i);
34 SpPauliStabiliser z_stab = tab.get_zrow(q);
35 SpPauliStabiliser x_stab = tab.get_xrow(q);
36 bool z_sign = cast_coeff<quarter_turns_t, Complex>(z_stab.coeff) == 1.;
37 bool x_sign = cast_coeff<quarter_turns_t, Complex>(x_stab.coeff) == 1.;
38 TKET_ASSERT(z_stab.string.size() == n_qubits);
39 std::vector<Pauli> z_string;
40 std::vector<Pauli> x_string;
41 for (unsigned j = 0; j < n_qubits; j++) {
42 z_string.push_back(z_stab.string.at(Qubit(j)));
43 x_string.push_back(x_stab.string.at(Qubit(j)));
44 }
45 rows.push_back(
46 std::make_shared<PauliPropagation>(
47 z_string, x_string, z_sign, x_sign, i));
48 }
49 return rows;
50}

52std::tuple<std::vector<PauliNode_ptr>, std::vector<PauliNode_ptr>>
53gpg_from_unordered_set(const std::vector<SymPauliTensor>& unordered_set) {
54 std::vector<PauliNode_ptr> rotation_set;
55 unsigned n_qubits = unordered_set.at(0).string.size();
56 for (auto& pauli : unordered_set) {
57 TKET_ASSERT(pauli.string.size() == n_qubits);
58 rotation_set.push_back(
59 std::make_shared<PauliRotation>(pauli.string, true, pauli.coeff));
60 }
61 UnitaryRevTableau tab(n_qubits);
62 std::vector<PauliNode_ptr> rows = get_nodes_from_tableau(tab, n_qubits);
63 return {rotation_set, rows};
64}

66// given a stabiliser Pauli string and an angle, return a dense string and an
67// angle
68static std::pair<std::vector<Pauli>, Expr> dense_pauli(
69 const SpPauliStabiliser& pauli, const unsigned& n_qubits,
70 const Expr& angle) {
71 bool sign = cast_coeff<quarter_turns_t, Complex>(pauli.coeff) == 1.;
72 std::vector<Pauli> string(n_qubits, Pauli::I);
73 for (const auto& pair : pauli.string) {
74 string[pair.first.index().at(0)] = pair.second;
75 }
76 return {string, sign ? angle : -angle};
77}

79static bool strings_commute(
80 const std::vector<Pauli>& s1, const std::vector<Pauli>& s2) {
81 unsigned n_conflicts = 0;
82 TKET_ASSERT(s1.size() == s2.size());
83 for (unsigned i = 0; i < s1.size(); ++i) {
84 Pauli p = s1[i];
85 Pauli p2 = s2[i];
86 if (p != Pauli::I && p2 != Pauli::I && p != p2) n_conflicts++;
87 }
88 return (n_conflicts % 2) == 0;
89}

91static bool nodes_commute(const PauliNode_ptr& n1, const PauliNode_ptr& n2) {
92 CommuteInfo c1 = n1->get_commute_info();
93 CommuteInfo c2 = n2->get_commute_info();
94 // check if every string in n1 commutes with all strings in n2
95 for (const std::vector<Pauli>& p1 : c1.paulis) {
96 for (const std::vector<Pauli>& p2 : c2.paulis) {
97 if (!strings_commute(p1, p2)) return false;
98 }
99 }
100 // check if the bits commute
101 for (const std::pair<UnitID, BitType>& b1 : c1.bits_info) {
102 for (const std::pair<UnitID, BitType>& b2 : c2.bits_info) {
103 if (b1.first == b2.first) {
104 // if two nodes read the same bit it's OK
105 if (b1.second == BitType::READ && b2.second == BitType::READ) {
106 break;
107 }
108 return false;
109 }
110 }
111 }
112 return true;
113}

115GPGraph::GPGraph(qubit_vector_t qubits, bit_vector_t bits)
116 : n_qubits_(qubits.size()), n_bits_(bits.size()) {
117 for (const Qubit& q : qubits) {
118 TKET_ASSERT(q.reg_name() == q_default_reg());
119 TKET_ASSERT(q.index().at(0) < qubits.size());
120 }
121 for (const Bit& b : bits) {
122 TKET_ASSERT(b.reg_name() == c_default_reg());
123 TKET_ASSERT(b.index().at(0) < bits.size());
124 }
125 cliff_ = UnitaryRevTableau(n_qubits_);

146// Adapted from `PauliGraph`, when adding a node to the graph, we check if it
147// can be merged with an existing node.
148void GPGraph::apply_node_at_end(PauliNode_ptr& node) {
149 GPVertSet to_search = end_line_;
150 GPVertSet commuted;
151 GPVert new_vert = boost::add_vertex(graph_);
152 graph_[new_vert] = node;
153 while (!to_search.empty()) {
154 // Get next candidate parent
155 GPVert to_compare = *to_search.begin();
156 to_search.erase(to_search.begin());
157 // Check that we have already commuted past all of its children
158 bool ready = true;
159 for (const GPVert& child : get_successors(to_compare)) {
160 if (commuted.get<TagKey>().find(child) == commuted.get<TagKey>().end()) {
161 ready = false;
162 break;
163 }
164 }
165 if (!ready) continue;
166 // Check if we can commute past it
167 PauliNode_ptr compare_node = graph_[to_compare];
168 // merge two ConditionalBlocks if they share the same condition
169 // this sacrifices the ability to commute the node but can group operations
170 // for better optimisation
171 if (node->get_type() == PauliNodeType::ConditionalBlock &&
172 compare_node->get_type() == PauliNodeType::ConditionalBlock) {
173 const ConditionalBlock& block1 =
174 dynamic_cast<const ConditionalBlock&>(*node);
175 ConditionalBlock& block2 = dynamic_cast<ConditionalBlock&>(*compare_node);
176 if (block1.cond_bits() == block2.cond_bits() &&
177 block1.cond_value() == block2.cond_value()) {
178 block2.append(block1);
179 boost::clear_vertex(new_vert, graph_);
180 boost::remove_vertex(new_vert, graph_);
181 return;
182 }
183 }
184 if (nodes_commute(node, compare_node)) {
185 // Check if two pauli rotations can be merged
186 if (node->get_type() == PauliNodeType::PauliRotation &&
187 compare_node->get_type() == PauliNodeType::PauliRotation) {
188 const PauliRotation& rot1 = dynamic_cast<const PauliRotation&>(*node);
189 const PauliRotation& rot2 =
190 dynamic_cast<const PauliRotation&>(*compare_node);
191 if (rot1.string() == rot2.string()) {
192 boost::clear_vertex(new_vert, graph_);
193 boost::remove_vertex(new_vert, graph_);
194 Expr merged_angle = rot1.angle() + rot2.angle();
195 std::optional<unsigned> cl_ang = equiv_Clifford(merged_angle);
196 if (cl_ang) {
197 cliff_.apply_pauli_at_front(
198 SpPauliStabiliser(rot1.string()), *cl_ang);
199 start_line_.erase(to_compare);
200 for (const GPVert& v : get_predecessors(to_compare)) {
201 if (boost::out_degree(v, graph_) == 1) {
202 end_line_.insert(v);
203 }
204 }
205 end_line_.erase(to_compare);
206 boost::clear_vertex(to_compare, graph_);
207 boost::remove_vertex(to_compare, graph_);
208 } else {
209 graph_[to_compare] = std::make_shared<PauliRotation>(
210 rot1.string(), true, merged_angle);
211 }
212 return;
213 }
214 }
215 // Commute and continue searching
216 GPVertSet preds = get_predecessors(to_compare);
217 to_search.insert(preds.begin(), preds.end());
218 commuted.insert(to_compare);
219 } else {
220 // Does not commute - add dependency edge
221 boost::add_edge(to_compare, new_vert, graph_);
222 end_line_.erase(to_compare);
223 }
224 }
225 end_line_.insert(new_vert);
226 if (get_predecessors(new_vert).empty()) start_line_.insert(new_vert);
227}

229void GPGraph::apply_paulis_at_end(
230 const std::vector<std::pair<std::vector<Pauli>, Expr>>& rotations,
231 const qubit_vector_t& qbs, bool conditional,
232 std::vector<unsigned> cond_bits, unsigned cond_value) {
233 std::vector<std::tuple<std::vector<Pauli>, bool, Expr>> conj_rotations;
234 for (const auto& pair : rotations) {
235 const std::vector<Pauli>& paulis = pair.first;
236 const Expr& angle = pair.second;
237 // Note that global phase is ignored
238 if (static_cast<std::size_t>(std::count(
239 paulis.begin(), paulis.end(), Pauli::I)) == paulis.size())
240 continue;
241 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
242 if (cliff_angle && cliff_angle.value() == 0) continue;
243 QubitPauliMap qpm;
244 for (unsigned i = 0; i != qbs.size(); ++i)
245 qpm.insert({Qubit(qbs[i]), paulis[i]});
246 if (cliff_angle && !conditional) {
247 cliff_.apply_pauli_at_end(SpPauliStabiliser(qpm), *cliff_angle);
248 continue;
249 }
250 // if not clifford we conjugate the string with the end-circuit tableau
251 SpPauliStabiliser qpt = cliff_.get_row_product(SpPauliStabiliser(qpm));
252 auto [pauli_dense, theta] = dense_pauli(qpt, n_qubits_, angle);
253 conj_rotations.push_back({pauli_dense, true, theta});
254 }
255
256 // if conditional we add a ConditionalBlock otherwise we add individual
257 // rotations.
258 if (conditional) {
259 PauliNode_ptr node = std::make_shared<ConditionalBlock>(
260 conj_rotations, cond_bits, cond_value);
261 apply_node_at_end(node);
262 } else {
263 for (const auto& t : conj_rotations) {
264 PauliNode_ptr node = std::make_shared<PauliRotation>(
265 std::get<0>(t), std::get<1>(t), std::get<2>(t));
266 apply_node_at_end(node);
267 }
268 }

271void GPGraph::apply_gate_at_end(
272 const Command& cmd, bool conditional, std::vector<unsigned> cond_bits,
273 unsigned cond_value) {
274 const Op_ptr op = cmd.get_op_ptr();
275 unit_vector_t args = cmd.get_args();
276 qubit_vector_t qbs = cmd.get_qubits();
277 OpType type = op->get_type();
278
279 for (const UnitID& arg : args) {
280 if (arg.type() == UnitType::Qubit) {
281 auto it = end_measures_.left.find(arg.index().at(0));
282 if (it != end_measures_.left.end()) {
283 // the measurement is no longer end-circuit, we remove it from
284 // end_measures_ and add it as a MidMeasure node instead.
285 SpPauliStabiliser paulis = cliff_.get_zrow(Qubit(it->first));
286 auto [pauli_dense, angle] = dense_pauli(paulis, n_qubits_, 1.);
287 PauliNode_ptr node = std::make_shared<MidMeasure>(
288 pauli_dense, (angle == 1.), it->second);
289 apply_node_at_end(node);
290 end_measures_.left.erase(it);
291 }
292 } else if (arg.type() == UnitType::Bit) {
293 auto it = end_measures_.right.find(arg.index().at(0));
294 if (it != end_measures_.right.end()) {
295 SpPauliStabiliser paulis = cliff_.get_zrow(Qubit(it->second));
296 auto [pauli_dense, angle] = dense_pauli(paulis, n_qubits_, 1.);
297 PauliNode_ptr node =
298 std::make_shared<MidMeasure>(pauli_dense, (angle == 1.), it->first);
299 apply_node_at_end(node);
300 end_measures_.right.erase(it);
301 }
302 }
303 }
304
305 std::vector<std::pair<std::vector<Pauli>, Expr>> pauli_rots;
306 switch (type) {
307 case OpType::Conditional: {
308 const Conditional& cond = static_cast<const Conditional&>(*op);
309 unit_vector_t inner_args;
310 for (unsigned i = 0; i < cond.get_width(); ++i)
311 cond_bits.push_back(Bit(args.at(i)).index().at(0));
312 for (unsigned i = cond.get_width(); i < args.size(); ++i)
313 inner_args.push_back(args.at(i));
314 cond_value = (cond_value << cond.get_width()) + cond.get_value();
315 apply_gate_at_end(
316 Command(cond.get_op(), inner_args), true, cond_bits, cond_value);
317 return;
318 }
319 case OpType::Measure: {
320 end_measures_.insert(
321 {args.at(0).index().at(0), args.at(1).index().at(0)});
322 return;
323 }
324 case OpType::Reset: {
325 SpPauliStabiliser z_paulis = cliff_.get_zrow(qbs[0]);
326 auto [z_pauli_dense, z_angle] = dense_pauli(z_paulis, n_qubits_, 1.);
327 SpPauliStabiliser x_paulis = cliff_.get_xrow(qbs[0]);
328 auto [x_pauli_dense, x_angle] = dense_pauli(x_paulis, n_qubits_, 1.);
329 PauliNode_ptr node = std::make_shared<Reset>(
330 z_pauli_dense, x_pauli_dense, (z_angle == 1.), (x_angle == 1.));
331 apply_node_at_end(node);
332 return;
333 }
334 case OpType::Z: {
335 pauli_rots.push_back({{Pauli::Z}, 1});
336 break;
337 }
338 case OpType::X: {
339 pauli_rots.push_back({{Pauli::X}, 1});
340 break;
341 }
342 case OpType::Y: {
343 pauli_rots.push_back({{Pauli::Y}, 1});
344 break;
345 }
346 case OpType::S: {
347 pauli_rots.push_back({{Pauli::Z}, 0.5});
348 break;
349 }
350 case OpType::V: {
351 pauli_rots.push_back({{Pauli::X}, 0.5});
352 break;
353 }
354 case OpType::Sdg: {
355 pauli_rots.push_back({{Pauli::Z}, 1.5});
356 break;
357 }
358 case OpType::Vdg: {
359 pauli_rots.push_back({{Pauli::X}, 1.5});
360 break;
361 }
362 case OpType::H: {
363 pauli_rots.push_back({{Pauli::Y}, 0.5});
364 pauli_rots.push_back({{Pauli::X}, 1});
365 break;
366 }
367 case OpType::CX:
368 case OpType::CY:
369 case OpType::CZ: {
370 Pauli t = (type == OpType::CZ) ? Pauli::Z
371 : (type == OpType::CX) ? Pauli::X
372 : Pauli::Y;
373 pauli_rots.push_back({{Pauli::Z, Pauli::I}, 1.5});
374 pauli_rots.push_back({{Pauli::I, t}, 1.5});
375 pauli_rots.push_back({{Pauli::Z, t}, 0.5});
376 break;
377 }
378 case OpType::SWAP: {
379 pauli_rots.push_back({{Pauli::Z, Pauli::Z}, 0.5});
380 pauli_rots.push_back({{Pauli::X, Pauli::X}, 0.5});
381 pauli_rots.push_back({{Pauli::Y, Pauli::Y}, 0.5});
382 break;
383 }
384 case OpType::noop:
385 case OpType::Phase: {
386 // ignore global phase
387 return;
388 }
389 case OpType::Rz: {
390 pauli_rots.push_back({{Pauli::Z}, op->get_params().at(0)});
391 break;
392 }
393 case OpType::Rx: {
394 pauli_rots.push_back({{Pauli::X}, op->get_params().at(0)});
395 break;
396 }
397 case OpType::Ry: {
398 pauli_rots.push_back({{Pauli::Y}, op->get_params().at(0)});
399 break;
400 }
401 case OpType::PhasedX: {
402 Expr alpha = op->get_params().at(0);
403 Expr beta = op->get_params().at(1);
404 pauli_rots.push_back({{Pauli::Z}, -beta});
405 pauli_rots.push_back({{Pauli::X}, alpha});
406 pauli_rots.push_back({{Pauli::Z}, beta});
407 break;
408 }
409 case OpType::T: {
410 pauli_rots.push_back({{Pauli::Z}, 0.25});
411 break;
412 }
413 case OpType::Tdg: {
414 pauli_rots.push_back({{Pauli::Z}, -0.25});
415 break;
416 }
417 case OpType::ZZMax: {
418 pauli_rots.push_back({{Pauli::Z, Pauli::Z}, 0.5});
419 break;
420 }
421 case OpType::PhaseGadget:
422 case OpType::ZZPhase: {
423 Expr angle = op->get_params().at(0);
424 std::vector<Pauli> paulis(qbs.size(), Pauli::Z);
425 pauli_rots.push_back({paulis, angle});
426 break;
427 }
428 case OpType::XXPhase: {
429 Expr angle = op->get_params().at(0);
430 pauli_rots.push_back({{Pauli::X, Pauli::X}, angle});
431 break;
432 }
433 case OpType::YYPhase: {
434 Expr angle = op->get_params().at(0);
435 pauli_rots.push_back({{Pauli::Y, Pauli::Y}, angle});
436 break;
437 }
438 case OpType::PauliExpBox: {
439 const PauliExpBox& peb = static_cast<const PauliExpBox&>(*op);
440 pauli_rots.push_back({peb.get_paulis(), peb.get_phase()});
441 break;
442 }
443 case OpType::PauliExpPairBox: {
444 const PauliExpPairBox& peb = static_cast<const PauliExpPairBox&>(*op);
445 auto [paulis1, paulis2] = peb.get_paulis_pair();
446 auto [phase1, phase2] = peb.get_phase_pair();
447 pauli_rots.push_back({paulis1, phase1});
448 pauli_rots.push_back({paulis2, phase2});
449 break;
450 }
451 case OpType::PauliExpCommutingSetBox: {
452 const PauliExpCommutingSetBox& peb =
453 static_cast<const PauliExpCommutingSetBox&>(*op);
454 for (const SymPauliTensor& pt : peb.get_pauli_gadgets()) {
455 pauli_rots.push_back({pt.string, pt.coeff});
456 }
457 break;
458 }
459 default: {
460 if (qbs.empty()) {
461 // ops with no quantum dependencies
462 PauliNode_ptr node = std::make_shared<ClassicalNode>(args, op);
463 apply_node_at_end(node);
464 return;
465 }
466 throw BadOpType("GreedyPauliSimp doesn't support", type);
467 }
468 }
469 apply_paulis_at_end(pauli_rots, qbs, conditional, cond_bits, cond_value);
470}

472std::vector<GPVert> GPGraph::vertices_in_order() const {
473 GPVIndex index = boost::get(boost::vertex_index, graph_);
474 int i = 0;
475 BGL_FORALL_VERTICES(v, graph_, GPDAG) { boost::put(index, v, i++); }
476 std::vector<GPVert> vertices;
477 boost::topological_sort(graph_, std::back_inserter(vertices));
478 std::reverse(vertices.begin(), vertices.end());
479 return vertices;
480}

482std::tuple<
483 std::vector<std::vector<PauliNode_ptr>>, std::vector<PauliNode_ptr>,
484 boost::bimap<unsigned, unsigned>>
485GPGraph::get_sequence(std::shared_ptr<std::atomic<bool>> stop_flag) {
486 std::vector<GPVert> vertices = vertices_in_order();
487 auto it = vertices.begin();
488 std::vector<std::vector<PauliNode_ptr>> interior_nodes;
489 while (it != vertices.end()) {
490 if (stop_flag.get()->load()) {
491 return {};
492 }
493 const PauliNode_ptr& node = graph_[*it];
494 std::vector<PauliNode_ptr> commuting_set;
495 commuting_set.push_back(node);
496 ++it;
497 while (it != vertices.end()) {
498 const PauliNode_ptr& u = graph_[*it];
499 bool commutes_with_all = true;
500 for (const PauliNode_ptr& v : commuting_set) {
501 if (!nodes_commute(u, v)) {
502 commutes_with_all = false;
503 break;
504 }
505 }
506 if (!commutes_with_all) break;
507 commuting_set.push_back(u);
508 ++it;
509 }
510 interior_nodes.push_back(commuting_set);
511 }
512 // add clifford
513 std::vector<PauliNode_ptr> cliff_nodes =
514 get_nodes_from_tableau(cliff_, n_qubits_);
515 return {interior_nodes, cliff_nodes, end_measures_};
516}
