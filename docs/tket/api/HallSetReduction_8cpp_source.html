44 ReductionResult HallSetReduction::reduce(DomainsAccessor& accessor) {
45  const bool new_reduce_call = m_awaiting_initial_fill;
46  if (m_awaiting_initial_fill && !fill_initial_partition(accessor)) {
47  return ReductionResult::NOGOOD;
48  }
49  if (m_partition_ids.empty()) {
50  return ReductionResult::SUCCESS;
51  }
52  m_partition_ids_to_erase.clear();
53  m_new_partition_ids.clear();
54 
55  ReductionResult result_to_return = ReductionResult::SUCCESS;
56 
57  for (auto id : m_partition_ids) {
58  auto& partition = m_partitions_storage.get_nonconst_object(id);
59 
60  {
61  const WithinReduceLoopHandlePartitionRefill refill_object(
62  accessor, new_reduce_call, partition, m_partition_ids_to_erase, id,
63  m_max_number_of_consecutive_failed_attempts);
64 
65  if (refill_object.nogood) {
66  return ReductionResult::NOGOOD;
67  }
68  if (refill_object.should_skip) {
69  continue;
70  }
71  }
72 
73  const auto search_result =
74  partition.search_for_hall_set(accessor, m_union_of_domains);
75  if (search_result == Partition::SearchResult::NOGOOD) {
76  return ReductionResult::NOGOOD;
77  }
78  if (search_result == Partition::SearchResult::HALL_SET) {
79  const auto hall_set_reduction_result =
80  within_reduce_loop_handle_hall_set_reduction(accessor, partition, id);
81 
82  if (hall_set_reduction_result == ReductionResult::NOGOOD) {
83  return ReductionResult::NOGOOD;
84  }
85  if (hall_set_reduction_result == ReductionResult::NEW_ASSIGNMENTS) {
86  result_to_return = ReductionResult::NEW_ASSIGNMENTS;
87  }
88  }
89  }
90  for (auto id : m_partition_ids_to_erase) {
91  TKET_ASSERT(m_partition_ids.erase(id) == 1);
92  m_partitions_storage.release(id);
93  }
94  for (auto id : m_new_partition_ids) {
95  TKET_ASSERT(m_partition_ids.insert(id).second);
96  }
97  return result_to_return;
98 }

100 HallSetReduction::WithinReduceLoopHandlePartitionRefill::
101  WithinReduceLoopHandlePartitionRefill(
102  const DomainsAccessor& accessor, bool new_reduce_call,
103  Partition& partition,
104  std::vector<ReusableStorageId>& partition_ids_to_erase,
105  ReusableStorageId partition_id,
106  unsigned max_number_of_consecutive_failed_attempts)
107  : nogood(false), should_skip(false) {
108  if (new_reduce_call) {
109  return;
110  }
111  switch (partition.refill(accessor)) {
112  case Partition::FillResult::NOGOOD:
113  nogood = true;
114  break;
115 
116  case Partition::FillResult::UNCHANGED:
117  // No point in testing again, all data is unchanged
118  // since the last check.
119  should_skip = true;
120  ++partition.attempt_count;
121  if (partition.attempt_count >=
122  max_number_of_consecutive_failed_attempts) {
123  partition_ids_to_erase.push_back(partition_id);
124  }
125  break;
126 
127  case Partition::FillResult::CHANGED:
128  partition.attempt_count = 0;
129  break;
130 
131  case Partition::FillResult::ERASE_THIS_PARTITION:
132  should_skip = true;
133  partition_ids_to_erase.push_back(partition_id);
134  break;
135  }
136 }

138 ReductionResult HallSetReduction::within_reduce_loop_handle_hall_set_reduction(
139  DomainsAccessor& accessor, Partition& partition,
140  ReusableStorageId partition_id) {
141  // First, erase the TV from all other domains in this partition.
142  const auto reduction_data =
143  partition.reduce_with_hall_set(accessor, m_union_of_domains);
144 
145  if (reduction_data.result_to_return == ReductionResult::NOGOOD) {
146  return ReductionResult::NOGOOD;
147  }
148  if (reduction_data.should_move_hall_pv_to_another_partition) {
149  // Copy the Hall set data across to a new partition.
150  // Of course, this may invalidate the "partition" reference!
151  const auto new_id = m_partitions_storage.get_new_id();
152  m_new_partition_ids.push_back(new_id);
153 
154  auto& new_partition = m_partitions_storage.get_nonconst_object(new_id);
155  new_partition.attempt_count = 0;
156  new_partition.domains_data.resize(m_union_of_domains.size());
157 
158  // CARE! The old partition reference might now be invalid!
159  // So we have to get it again.
160  const auto& old_partition_again =
161  m_partitions_storage.get_object(partition_id);
162  for (unsigned ii = 0; ii < new_partition.domains_data.size(); ++ii) {
163  new_partition.domains_data[ii] = old_partition_again.domains_data.at(
164  old_partition_again.domains_data.size() - ii - 1);
165  }
166  }
167  if (reduction_data.should_erase_this_partition) {
168  m_partition_ids_to_erase.push_back(partition_id);
169  } else {
170  // The old partition isn't actually being erased, so we're
171  // retaining it; but we must delete the existing Hall set
172  // (at the back), which is no longer relevant.
173  auto& old_partition_again =
174  m_partitions_storage.get_nonconst_object(partition_id);
175  old_partition_again.domains_data.resize(
176  old_partition_again.domains_data.size() - m_union_of_domains.size());
177  }
178  return reduction_data.result_to_return;
179 }

181 bool HallSetReduction::fill_initial_partition(const DomainsAccessor& accessor) {
182  m_awaiting_initial_fill = false;
183  for (auto id : m_partition_ids) {
184  m_partitions_storage.release(id);
185  }
186  m_partition_ids.clear();
187  const auto new_id = m_partitions_storage.get_new_id();
188 
189  switch (
190  m_partitions_storage.get_nonconst_object(new_id).initial_fill(accessor)) {
191  case Partition::FillResult::CHANGED:
192  m_partition_ids.insert(new_id);
193  break;
194  case Partition::FillResult::ERASE_THIS_PARTITION:
195  m_partitions_storage.release(new_id);
196  break;
197  case Partition::FillResult::NOGOOD:
198  return false;
199  case Partition::FillResult::UNCHANGED:
200  // We just don't allow this to be returned for a new fill.
201  TKET_ASSERT(false);
202  }
203  return true;
204 }

206 void HallSetReduction::Partition::sort_and_remove_singleton_domains() {
207  std::sort(
208  domains_data.begin(), domains_data.end(),
209  [](const Data& lhs, const Data& rhs) {
210  return lhs.domain_size > rhs.domain_size ||
211  (lhs.domain_size == rhs.domain_size && lhs.pv > rhs.pv);
212  });
213 
214  while (!domains_data.empty() && domains_data.back().domain_size < 2) {
215  TKET_ASSERT(domains_data.back().domain_size == 1);
216  domains_data.pop_back();
217  }
218 }

220 HallSetReduction::Partition::FillResult
221 HallSetReduction::Partition::initial_fill(const DomainsAccessor& accessor) {
222  domains_data.clear();
223  for (VertexWSM pv : accessor.get_unassigned_pattern_vertices_superset()) {
224  const auto domain_size = accessor.get_domain(pv).size();
225  if (domain_size == 0) {
226  return FillResult::NOGOOD;
227  }
228  if (domain_size == 1) {
229  continue;
230  }
231  domains_data.emplace_back();
232  domains_data.back().domain_size = domain_size;
233  domains_data.back().pv = pv;
234  attempt_count = 0;
235  }
236  if (domains_data.size() >= 3) {
237  sort_and_remove_singleton_domains();
238  if (domains_data.size() >= 3) {
239  return FillResult::CHANGED;
240  }
241  }
242  return FillResult::ERASE_THIS_PARTITION;
243 }

245 HallSetReduction::Partition::FillResult HallSetReduction::Partition::refill(
246  const DomainsAccessor& accessor) {
247  bool changed = false;
248  for (auto& entry : domains_data) {
249  const auto new_domain_size = accessor.get_domain(entry.pv).size();
250  if (new_domain_size == 0) {
251  return FillResult::NOGOOD;
252  }
253  if (new_domain_size != entry.domain_size) {
254  changed = true;
255  entry.domain_size = new_domain_size;
256  }
257  }
258  if (!changed) {
259  return FillResult::UNCHANGED;
260  }
261  if (domains_data.size() >= 3) {
262  // You need at least 3 vertices for a useful Hall set.
263  // (If there are vertices pv1, pv2 with
264  // Dom(pv1) = Dom(pv2) = {a,b}, and no other domain including a or b,
265  // then Hall set reduction can never reduce them further).
266  // Note that singleton Hall sets Dom(pv) = {a} are treated
267  // separately (i.e., by alldiff propagation).
268  sort_and_remove_singleton_domains();
269  if (domains_data.size() >= 3) {
270  return FillResult::CHANGED;
271  }
272  }
273  return FillResult::ERASE_THIS_PARTITION;
274 }

276 HallSetReduction::Partition::SearchResult
277 HallSetReduction::Partition::search_for_hall_set(
278  const DomainsAccessor& accessor,
279  std::set<VertexWSM>& union_of_domains_work_set) const {
280  TKET_ASSERT(domains_data.size() > 2);
281 
282  // Note that only PV in this partition have domains which intersect
283  // any possible combined union. Thus, a useful Hall set needs at least one
284  // PV in this partition NOT to be included (for, they are the only PV which
285  // could be reduced).
286 
287  if (domains_data.back().domain_size >= domains_data.size()) {
288  // Even the smallest domain is too big to get a Hall set or nogood.
289  return SearchResult::NO_HALL_SET;
290  }
291  union_of_domains_work_set = accessor.get_domain(domains_data.back().pv);
292 
293  for (std::size_t next_size = 2; next_size < domains_data.size();
294  ++next_size) {
295  // First, is it POSSIBLE that adding the next PV will create
296  // a union of domains small enough? (The number of PVs must
297  // hit or overtake the union size. We can get a guaranteed
298  // lower bound on the union size. So, maybe we can prove
299  // that ALL unions will be too big, for the order
300  // in which we add them, and thus break out early).
301  auto union_size_lower_bound = union_of_domains_work_set.size();
302  bool should_add_pv = false;
303 
304  // If we kept on adding PVs from this point, how big would the union
305  // definitely get?
306  for (auto hypothetical_size = next_size;
307  hypothetical_size < domains_data.size(); ++hypothetical_size) {
308  // Consider the hypothetical union with the next domain.
309  union_size_lower_bound = std::max(
310  union_size_lower_bound,
311  domains_data[domains_data.size() - hypothetical_size].domain_size);
312 
313  if (union_size_lower_bound <= hypothetical_size) {
314  // The union possibly could be small enough at this point.
315  should_add_pv = true;
316  break;
317  }
318  }
319  if (!should_add_pv) {
320  // At every stage, the union of domains was guaranteed to be too big.
321  return SearchResult::NO_HALL_SET;
322  }
323 
324  // Now, actually add the next PV.
325  const VertexWSM& next_pv = domains_data[domains_data.size() - next_size].pv;
326  for (VertexWSM tv : accessor.get_domain(next_pv)) {
327  union_of_domains_work_set.insert(tv);
328  }
329  if (union_of_domains_work_set.size() < next_size) {
330  return SearchResult::NOGOOD;
331  }
332  if (union_of_domains_work_set.size() == next_size) {
333  return SearchResult::HALL_SET;
334  }
335  }
336  return SearchResult::NO_HALL_SET;
337 }

339 HallSetReduction::Partition::HallSetReductionData
340 HallSetReduction::Partition::reduce_with_hall_set(
341  DomainsAccessor& accessor, const std::set<VertexWSM>& union_of_domains) {
342  TKET_ASSERT(union_of_domains.size() > 1);
343  TKET_ASSERT(union_of_domains.size() < domains_data.size());
344 
345  HallSetReductionData reduction_data;
346  reduction_data.changed = false;
347 
348  // As long as we have at least 3 vertices, there's the chance
349  // of another Hall set.
350  // Note that all OTHER vertices will have domains
351  // disjoint from the Hall set at the end, so there's no point in keeping
352  // the Hall set in THIS partition.
353  reduction_data.should_move_hall_pv_to_another_partition =
354  union_of_domains.size() > 2;
355  const unsigned number_of_vertices_to_reduce =
356  domains_data.size() - union_of_domains.size();
357  unsigned number_of_newly_assigned_vertices = 0;
358 
359  for (unsigned ii = 0; ii < number_of_vertices_to_reduce; ++ii) {
360  const auto intersect_result = accessor.intersect_domain_with_complement_set(
361  domains_data[ii].pv, union_of_domains);
362 
363  if (intersect_result.reduction_result == ReductionResult::NOGOOD) {
364  reduction_data.result_to_return = ReductionResult::NOGOOD;
365  return reduction_data;
366  }
367  // Conveniently, the old domain size is already stored.
368  if (intersect_result.new_domain_size != domains_data[ii].domain_size) {
369  reduction_data.changed = true;
370  domains_data[ii].domain_size = intersect_result.new_domain_size;
371  }
372  // We continue to the end, just for this partition,
373  // so at least everything is fully reduced before we go round the loop
374  // again.
375  if (intersect_result.reduction_result == ReductionResult::NEW_ASSIGNMENTS) {
376  ++number_of_newly_assigned_vertices;
377  }
378  }
379 
380  if (number_of_newly_assigned_vertices > 0) {
381  reduction_data.result_to_return = ReductionResult::NEW_ASSIGNMENTS;
382  } else {
383  reduction_data.result_to_return = ReductionResult::SUCCESS;
384  }
385 
386  // How many vertices would be left in THIS partition
387  // after removing the newly assigned ones (and the old Hall set, which will
388  // be removed whatever happens)?
389  reduction_data.should_erase_this_partition =
390  domains_data.size() - union_of_domains.size() -
391  number_of_newly_assigned_vertices <
392  3;
393 
394  return reduction_data;
395 }
