21 GrayCode gen_graycode(unsigned m_controls) {
22  const unsigned n = m_controls;
23  if (m_controls == 0) return {};
24  GrayCode gc{{0}, {1}};
25 
26  for (unsigned i = 2; i < (1u << n); i = i << 1) {
27  for (unsigned j = 0; j < i; ++j) {
28  gc.push_back(gc[i - 1 - j]);
29  }
30  for (unsigned j = 0; j < i; ++j) {
31  gc[j].push_back(0);
32  }
33  for (unsigned j = i; j < 2 * i; ++j) {
34  gc[j].push_back(1);
35  }
36  }
37  return gc;
38 }

40 uint32_t reverse_bits(uint32_t v, unsigned w) {
41  uint32_t r = 0;
42  for (unsigned i = w; i--;) {

49 std::vector<bool> dec_to_bin(unsigned long long dec, unsigned width) {

51  std::vector<bool> bits(width);
