21GrayCode gen_graycode(unsigned m_controls) {
22 const unsigned n = m_controls;
23 if (m_controls == 0) return {};
24 GrayCode gc{{0}, {1}};
25
26 for (unsigned i = 2; i < (1u << n); i = i << 1) {
27 for (unsigned j = 0; j < i; ++j) {
28 gc.push_back(gc[i - 1 - j]);
29 }
30 for (unsigned j = 0; j < i; ++j) {
31 gc[j].push_back(0);
32 }
33 for (unsigned j = i; j < 2 * i; ++j) {
34 gc[j].push_back(1);
35 }
36 }
37 return gc;
38}

40uint64_t reverse_bits(uint64_t v, unsigned w) {
41 uint64_t r = 0;
42 for (unsigned i = w; i--;) {
43 r |= (v & 1) << i;
44 v >>= 1;
45 }
46 return r;
47}

49std::vector<bool> dec_to_bin(unsigned long long dec, unsigned width) {
50 auto bs = boost::dynamic_bitset<>(width, dec);
51 std::vector<bool> bits(width);
52 for (unsigned long long i = 0; i < width; i++) {
53 bits[width - i - 1] = bs[i];
54 }
55 return bits;
56}

59unsigned long long bin_to_dec(const std::vector<bool> &bin) {
60 unsigned long long res = 0;
61 for (unsigned long long i = 0; i < bin.size(); i++) {
62 if (bin[i]) {
63 res = res + (1ULL << (bin.size() - 1 - i));
64 }
65 }
66 return res;
