26LargeCliquesResult::LargeCliquesResult(
27 const AdjacencyData& adjacency_data,
28 const set<std::size_t>& vertices_in_component,
29 std::size_t internal_size_limit) {
30 // ...and, before swapping, everything in here will have size one greater than
31 // "result".
32 vector<set<std::size_t>> extended_result;
33
34 // At each stage, every set will have the same size, containing a clique of
35 // that size.
36 cliques.reserve(vertices_in_component.size());
37 for (std::size_t i : vertices_in_component) {
38 cliques.emplace_back(set{i});
39 }
40 bool hit_internal_limit = false;
41
42 // A little trick: the vertex indices can always be stored in order: v1 < v2 <
43 // ... Therefore, within each vertex set, if we only allow adding vertices
44 // with LARGER index than the largest already stored, we will automatically
45 // discard duplicates.
46 for (std::size_t counter = 0; counter <= vertices_in_component.size();
47 ++counter) {
48 // Extend the existing results.
49 for (const auto& clique : cliques) {
50 // Guaranteed to be nonempty.
51 const std::size_t largest_index = *clique.crbegin();
52
53 if (extended_result.size() >= internal_size_limit) {
54 hit_internal_limit = true;
55 break;
56 }
57 // We only have to check the neighbours of ONE vertex to form a larger
58 // clique.
59 const auto& neighbours = adjacency_data.get_neighbours(largest_index);
60 for (std::size_t new_v : neighbours) {
61 if (new_v <= largest_index) {
62 continue;
63 }
64 // We have a new vertex; does it adjoin EVERY existing vertex?
65 bool joins_every_vertex = true;
66 for (std::size_t existing_v : clique) {
67 if (adjacency_data.get_neighbours(existing_v).count(new_v) == 0) {
68 joins_every_vertex = false;
69 break;
70 }
71 }
72 if (joins_every_vertex) {
73 extended_result.emplace_back(clique);
74 extended_result.back().insert(new_v);
75 if (extended_result.size() >= internal_size_limit) {
76 hit_internal_limit = true;
77 break;
78 }
79 }
80 }
81 }
82 if (extended_result.empty()) {
83 // No extensions
84 cliques_are_definitely_max_size = !hit_internal_limit;
85 return;
86 }
87 cliques = extended_result;
88 // TODO: improve memory reallocation etc. by reusing rather than clearing
89 // (or, could be sneaky and do this within a single vector, jiggle with
90 // indices).
91 extended_result.clear();
92 }
93 cliques_are_definitely_max_size = false;
94}
