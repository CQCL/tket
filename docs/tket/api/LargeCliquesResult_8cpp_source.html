26 LargeCliquesResult::LargeCliquesResult(
27  const AdjacencyData& adjacency_data,
28  const set<size_t>& vertices_in_component, size_t internal_size_limit) {
29  // ...and, before swapping, everything in here will have size one greater than
30  // "result".
31  vector<set<size_t>> extended_result;
32 
33  // At each stage, every set will have the same size, containing a clique of
34  // that size.
35  cliques.reserve(vertices_in_component.size());
36  for (size_t i : vertices_in_component) {
37  cliques.emplace_back(set{i});
38  }
39  bool hit_internal_limit = false;
40 
41  // A little trick: the vertex indices can always be stored in order: v1 < v2 <
42  // ... Therefore, within each vertex set, if we only allow adding vertices
43  // with LARGER index than the largest already stored, we will automatically
44  // discard duplicates.
45  for (size_t counter = 0; counter <= vertices_in_component.size(); ++counter) {
46  // Extend the existing results.
47  for (const auto& clique : cliques) {
48  // Guaranteed to be nonempty.
49  const size_t largest_index = *clique.crbegin();
50 
51  if (extended_result.size() >= internal_size_limit) {
52  hit_internal_limit = true;
53  break;
54  }
55  // We only have to check the neighbours of ONE vertex to form a larger
56  // clique.
57  const auto& neighbours = adjacency_data.get_neighbours(largest_index);
58  for (size_t new_v : neighbours) {
59  if (new_v <= largest_index) {
60  continue;
61  }
62  // We have a new vertex; does it adjoin EVERY existing vertex?
63  bool joins_every_vertex = true;
64  for (size_t existing_v : clique) {
65  if (adjacency_data.get_neighbours(existing_v).count(new_v) == 0) {
66  joins_every_vertex = false;
67  break;
68  }
69  }
70  if (joins_every_vertex) {
71  extended_result.emplace_back(clique);
72  extended_result.back().insert(new_v);
73  if (extended_result.size() >= internal_size_limit) {
74  hit_internal_limit = true;
75  break;
76  }
77  }
78  }
79  }
80  if (extended_result.empty()) {
81  // No extensions
82  cliques_are_definitely_max_size = !hit_internal_limit;
83  return;
84  }
85  cliques = extended_result;
86  // TODO: improve memory reallocation etc. by reusing rather than clearing
87  // (or, could be sneaky and do this within a single vector, jiggle with
88  // indices).
89  extended_result.clear();
90  }
91  cliques_are_definitely_max_size = false;
92 }
