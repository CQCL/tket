21LexiRoute::LexiRoute(
22 const ArchitecturePtr& _architecture,
23 MappingFrontier_ptr& _mapping_frontier)
24 : architecture_(_architecture), mapping_frontier_(_mapping_frontier) {
25 // set initial logical->physical labelling
26 for (const Qubit& qb : this->mapping_frontier_->circuit_.all_qubits()) {
27 this->labelling_.insert({qb, qb});
28 Node n(qb);
29 // store which Node have been assigned to Circuit already
30 if (this->architecture_->node_exists(n)) {
31 this->assigned_nodes_.insert(n);
32 }
33 }
34}

36void LexiRoute::reassign_to_any_ancilla_node(const Node& reassign_node) {
46 TKET_ASSERT(!this->mapping_frontier_->ancilla_nodes_.empty());
47 auto ancilla_it = this->mapping_frontier_->ancilla_nodes_.begin();
48 Node ancilla_node = *ancilla_it;
49 this->mapping_frontier_->ancilla_nodes_.erase(ancilla_it);
50 this->mapping_frontier_->merge_ancilla(reassign_node, ancilla_node);
51 auto it = std::find_if(
52 this->labelling_.begin(), this->labelling_.end(),
53 [&reassign_node](const std::pair<const UnitID, UnitID>& pair_unitid) {
54 return pair_unitid.second == reassign_node;
55 });
56 TKET_ASSERT(it != this->labelling_.end());
57 this->labelling_[it->first] = ancilla_node;
58 unit_map_t relabel = {{UnitID(reassign_node), UnitID(ancilla_node)}};
59 this->mapping_frontier_->circuit_.rename_units(relabel);
60}

62bool LexiRoute::reassign_to_any_spare_node(const Node& reassign_node) {
68 std::set<Node> all_nodes_s;
69 for (const std::pair<const UnitID, UnitID>& p : this->labelling_) {
70 all_nodes_s.insert(Node(p.second));
71 }
72 std::set<Node> architecture_nodes = this->architecture_->nodes();
73 auto it = std::find_if(
74 architecture_nodes.begin(), architecture_nodes.end(),
75 [&all_nodes_s](const Node& node) {
76 return all_nodes_s.find(node) == all_nodes_s.end();
77 });
78
79 if (it == architecture_nodes.end()) {
80 return false;
81 }
82
91 Node spare_node = *it;
92
93 auto jt = std::find_if(
94 this->labelling_.begin(), this->labelling_.end(),
95 [&reassign_node](const std::pair<const UnitID, UnitID>& pair_unitid) {
96 return pair_unitid.second == reassign_node;
97 });
98 TKET_ASSERT(jt != this->labelling_.end());
99 this->labelling_[jt->first] = spare_node;
104 auto reassign_node_reassignable_it =
105 this->mapping_frontier_->reassignable_nodes_.find(reassign_node);
106 TKET_ASSERT(
107 reassign_node_reassignable_it !=
108 this->mapping_frontier_->reassignable_nodes_.end());
109 this->mapping_frontier_->reassignable_nodes_.erase(
110 reassign_node_reassignable_it);
111 this->mapping_frontier_->reassignable_nodes_.insert(spare_node);
112
117 auto reassign_node_boundary_it =
118 this->mapping_frontier_->linear_boundary->get<TagKey>().find(
119 reassign_node);
120 TKET_ASSERT(
121 reassign_node_boundary_it !=
122 this->mapping_frontier_->linear_boundary->get<TagKey>().end());
123 this->mapping_frontier_->linear_boundary->replace(
124 reassign_node_boundary_it,
125 {spare_node, reassign_node_boundary_it->second});
126
131 unit_map_t relabel = {{UnitID(reassign_node), UnitID(spare_node)}};
132 this->mapping_frontier_->circuit_.rename_units(relabel);
133
138 auto initial_reassign_it =
139 this->mapping_frontier_->bimaps_->initial.right.find(reassign_node);
140 TKET_ASSERT(
141 initial_reassign_it !=
142 this->mapping_frontier_->bimaps_->initial.right.end());
143 UnitID preserved_orig = initial_reassign_it->second;
144 std::pair<UnitID, UnitID> initial_replacement = {preserved_orig, spare_node};
145 this->mapping_frontier_->bimaps_->initial.right.erase(initial_reassign_it);
146 this->mapping_frontier_->bimaps_->initial.left.insert(initial_replacement);
147 auto final_reassign_it =
148 this->mapping_frontier_->bimaps_->final.left.find(preserved_orig);
149 TKET_ASSERT(
150 final_reassign_it != this->mapping_frontier_->bimaps_->final.left.end());
151 std::pair<UnitID, UnitID> final_replacement = {preserved_orig, spare_node};
152 this->mapping_frontier_->bimaps_->final.left.erase(final_reassign_it);
153 this->mapping_frontier_->bimaps_->final.left.insert(final_replacement);
154
158 this->assigned_nodes_.insert(spare_node);
159 return true;
160}

200
205 unit_map_t relabel = {{UnitID(assignee), UnitID(reassign_node)}};
206 this->mapping_frontier_->circuit_.rename_units(relabel);
207
212 auto initial_assignee_it =

214 TKET_ASSERT(

216 this->mapping_frontier_->bimaps_->initial.right.end());

233// "assignee" is Circuit UnitID being relabelled to UnitID "replacement"
234void LexiRoute::assign_valid_node(
235 const UnitID& assignee, const UnitID& replacement) {
247 if (this->mapping_frontier_->reassignable_nodes_.find(Node(replacement)) !=
248 this->mapping_frontier_->reassignable_nodes_.end()) {
253 this->reassign_node(Node(replacement), assignee);
254 return;
255 }
256 if (this->mapping_frontier_->ancilla_nodes_.find(Node(replacement)) !=
257 this->mapping_frontier_->ancilla_nodes_.end()) {
264 this->mapping_frontier_->merge_ancilla(assignee, replacement);
265 this->labelling_.erase(replacement);
266 this->labelling_[assignee] = replacement;
267 return;
268 }
269
274 this->labelling_[assignee] = replacement;
275 this->assigned_nodes_.insert(Node(replacement));
276
277 // bimaps are used by CompilationUnit to track how an original Circuit Qubit
278 // may be relabelled update this to reflect new assignment first do bimaps
279 // "RHS"
280 auto initial_assignee_it =
281 this->mapping_frontier_->bimaps_->initial.right.find(assignee);
282 TKET_ASSERT(
283 initial_assignee_it !=
284 this->mapping_frontier_->bimaps_->initial.right.end());
285 UnitID original = initial_assignee_it->second;
286 this->mapping_frontier_->bimaps_->initial.right.erase(initial_assignee_it);
287 this->mapping_frontier_->bimaps_->initial.left.insert(
288 {original, replacement});
289
290 // then bimaps "LHS"
291 auto final_assignee_it =
292 this->mapping_frontier_->bimaps_->final.left.find(original);
293 TKET_ASSERT(
294 final_assignee_it != this->mapping_frontier_->bimaps_->final.left.end());
295 this->mapping_frontier_->bimaps_->final.left.erase(final_assignee_it);
296 this->mapping_frontier_->bimaps_->final.left.insert({original, replacement});
297
298 // Update linear boundaries
299 auto assignee_boundary_it =
300 this->mapping_frontier_->linear_boundary->get<TagKey>().find(assignee);
301 TKET_ASSERT(
302 assignee_boundary_it !=
303 this->mapping_frontier_->linear_boundary->get<TagKey>().end());
304 this->mapping_frontier_->linear_boundary->replace(
305 assignee_boundary_it, {replacement, assignee_boundary_it->second});
306
307 unit_map_t relabel = {{UnitID(assignee), UnitID(replacement)}};
308 // Relabel Circuit UnitID
309 this->mapping_frontier_->circuit_.rename_units(relabel);
310}

312bool LexiRoute::assign_at_distance(
313 const UnitID& assignee, const Node& root, unsigned distances) {
321 node_set_t valid_nodes;
322 for (const Node& neighbour :
323 this->architecture_->nodes_at_distance(root, distances)) {
324 // A node is unassigned if it's empty or holding an ancilla, or can be
325 // reassigned
326 if (this->mapping_frontier_->reassignable_nodes_.find(neighbour) !=
327 this->mapping_frontier_->reassignable_nodes_.end() ||
328 this->assigned_nodes_.find(neighbour) == this->assigned_nodes_.end() ||
329 this->mapping_frontier_->ancilla_nodes_.find(neighbour) !=
330 this->mapping_frontier_->ancilla_nodes_.end()) {
331 valid_nodes.insert(neighbour);
332 }
333 }
334
338 if (valid_nodes.empty()) {
339 return false;
340 }
341 auto it = valid_nodes.begin();
345 if (valid_nodes.size() == 1) {
346 this->assign_valid_node(assignee, Node(*it));
347 return true;
348 }
353 lexicographical_distances_t winning_distances =
354 this->architecture_->get_distances(*it);
355 Node preserved_node = *it;
356 ++it;
357 while (it != valid_nodes.end()) {
358 lexicographical_distances_t comparison_distances =
359 this->architecture_->get_distances(*it);
360 if (comparison_distances < winning_distances) {
361 preserved_node = *it;
362 winning_distances = comparison_distances;
363 }
364 ++it;
365 }
369 this->assign_valid_node(assignee, preserved_node);
370 return true;
371}

373bool LexiRoute::update_labelling() {
374 // iterate through interacting qubits, assigning them to an Architecture
375 // Node if they aren't already
376 bool relabelled = false;
377 for (const auto& pair : this->interacting_uids_) {
378 bool uid_0_exist =
379 this->architecture_->node_exists(Node(this->labelling_[pair.first]));
380 bool uid_1_exist =
381 this->architecture_->node_exists(Node(this->labelling_[pair.second]));
382 if (!uid_0_exist || !uid_1_exist) {
383 relabelled = true;
384 }
385 if (!uid_0_exist && !uid_1_exist) {
398 if (this->assigned_nodes_.size() == 0) {
405 // Find Architecture Node with best out degree
406 std::set<Node> max_degree_nodes =
407 this->architecture_->max_degree_nodes();
408 auto it = max_degree_nodes.begin();
409
420 lexicographical_distances_t winning_distances =
421 this->architecture_->get_distances(*it);
422 Node preserved_node = Node(*it);
423 ++it;
424 for (; it != max_degree_nodes.end(); ++it) {
425 lexicographical_distances_t comparison_distances =
426 this->architecture_->get_distances(*it);
427 if (comparison_distances < winning_distances) {
428 preserved_node = Node(*it);
429 winning_distances = comparison_distances;
430 }
431 }
432 // assign unplaced Qubit to "best" Node
433 this->labelling_[pair.first] = preserved_node;
434 this->assigned_nodes_.insert(preserved_node);
435
436 // update bimaps
437 auto initial_assignee_it =
438 this->mapping_frontier_->bimaps_->initial.right.find(pair.first);
439 TKET_ASSERT(
440 initial_assignee_it !=
441 this->mapping_frontier_->bimaps_->initial.right.end());
442 UnitID original = initial_assignee_it->second;
443 this->mapping_frontier_->bimaps_->initial.right.erase(
444 initial_assignee_it);
445 this->mapping_frontier_->bimaps_->initial.left.insert(
446 {original, preserved_node});
447
448 auto final_assignee_it =
449 this->mapping_frontier_->bimaps_->final.left.find(original);
450 this->mapping_frontier_->bimaps_->final.left.erase(final_assignee_it);
451 this->mapping_frontier_->bimaps_->final.left.insert(
452 {original, preserved_node});
453
454 // this prompts next if statement to place "uid1"
455 uid_0_exist = true;
456
457 // update circuit with new labelling
458 unit_map_t relabel = {{UnitID(pair.first), UnitID(preserved_node)}};
459 this->mapping_frontier_->circuit_.rename_units(relabel);
460
461 // update linear boundary
462 auto pair_first_boundary_it =
463 this->mapping_frontier_->linear_boundary->get<TagKey>().find(
464 pair.first);
465 this->mapping_frontier_->linear_boundary->replace(
466 pair_first_boundary_it,
467 {preserved_node, pair_first_boundary_it->second});
468
469 // given best node, do something
470 } else {
477 auto root_it = this->assigned_nodes_.begin();
478 while (!uid_0_exist && root_it != this->assigned_nodes_.end()) {
479 Node root = *root_it;
480 uid_0_exist = this->assign_at_distance(pair.first, root, 1);
481 ++root_it;
482 }
483 if (!uid_0_exist) {
484 throw LexiRouteError(
485 "Unable to assign physical qubit - no free qubits remaining.");
486 }
487 }
488 }
489 if (!uid_0_exist && uid_1_exist) {
490 Node root(this->labelling_[pair.second]);
491 for (unsigned k = 1; k <= this->architecture_->get_diameter(); k++) {
492 if (this->assign_at_distance(pair.first, root, k)) {
493 uid_0_exist = true;
494 break;
495 }
496 }
497 if (!uid_0_exist) {
498 throw LexiRouteError(
499 "Unable to assign physical qubit - no free qubits remaining.");
500 }
501 }
502 if (uid_0_exist && !uid_1_exist) {
503 Node root(this->labelling_[pair.first]);
504 for (unsigned k = 1; k <= this->architecture_->get_diameter(); k++) {
505 if (this->assign_at_distance(pair.second, root, k)) {
506 uid_1_exist = true;
507 break;
508 }
509 }
510 if (!uid_1_exist) {
511 throw LexiRouteError(
512 "Unable to assign physical qubit - no free qubits remaining.");
513 }
514 }
515 }
516 return relabelled;
517}

524bool LexiRoute::set_interacting_uids(
525 AssignedOnly assigned_only, CheckRoutingValidity route_check,
526 CheckLabellingValidity label_check) {
527 // return types
528 this->interacting_uids_.clear();
529 bool all_placed = true;
530 for (auto it =
531 this->mapping_frontier_->linear_boundary->get<TagKey>().begin();
532 it != this->mapping_frontier_->linear_boundary->get<TagKey>().end();
533 ++it) {
534 Edge e0 = this->mapping_frontier_->circuit_.get_nth_out_edge(
535 it->second.first, it->second.second);
536 Vertex v0 = this->mapping_frontier_->circuit_.target(e0);
537 // should never be input vertex, so can always use in_edges
538 Op_ptr op = this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(v0);
539 if (op->get_type() != OpType::Barrier) {
540 int n_edges = this->mapping_frontier_->circuit_.n_in_edges_of_type(
541 v0, EdgeType::Quantum);
542 // make forwards = backwards
543 if (n_edges == 2) {
544 auto jt = it;
545 ++jt;
546 while (jt !=
547 this->mapping_frontier_->linear_boundary->get<TagKey>().end()) {
548 // i.e. if vertices match
549 Edge e1 = this->mapping_frontier_->circuit_.get_nth_out_edge(
550 jt->second.first, jt->second.second);
551 Vertex v1 = this->mapping_frontier_->circuit_.target(e1);
552 if (v0 == v1) {
553 // we can assume a qubit will only be in one interaction
554 // we can assume from how we iterate through pairs that each qubit
555 // will only be found in one match
556 bool node0_exists =
557 this->architecture_->node_exists(Node(it->first));
558 bool node1_exists =
559 this->architecture_->node_exists(Node(jt->first));
560 if (!node0_exists || !node1_exists || op->get_desc().is_box()) {
561 all_placed = false;
562 if (route_check == CheckRoutingValidity::Yes) return false;
563 }
564
565 if (assigned_only == AssignedOnly::No ||
566 (node0_exists && node1_exists)) {
567 interacting_uids_.insert({it->first, jt->first});
568 interacting_uids_.insert({jt->first, it->first});
569 }
570 }
571 ++jt;
572 }
573 }
574 }
575 }
576
577 // conditions for proceeding with labelling
578 if (label_check == CheckLabellingValidity::Yes) {
579 if (all_placed) {
580 return true;
581 } else {
582 return false;
583 }
584 }
585 // this should have left early when first found
586 if (route_check == CheckRoutingValidity::Yes) {
587 if (all_placed && interacting_uids_.size() > 0) {
588 return true;
589 }
590 return false;
591 }
592 // => either route_check true and all_placed so valid
593 // or !route_check and !label_check so return true and discard
594 return true;
595}

597swap_set_t LexiRoute::get_candidate_swaps() {
598 swap_set_t candidate_swaps;
599 for (const auto& interaction : this->interacting_uids_) {
600 Node assigned_first = Node(this->labelling_[interaction.first]);
601 std::vector<Node> adjacent_uids_0 =
602 this->architecture_->nodes_at_distance(assigned_first, 1);
603 TKET_ASSERT(adjacent_uids_0.size() != 0);
604 for (const Node& neighbour : adjacent_uids_0) {
605 if (candidate_swaps.find({neighbour, assigned_first}) ==
606 candidate_swaps.end()) {
607 candidate_swaps.insert({assigned_first, neighbour});
608 }
609 }
610 Node assigned_second = Node(this->labelling_[interaction.second]);
611 std::vector<Node> adjacent_uids_1 =
612 this->architecture_->nodes_at_distance(assigned_second, 1);
613 TKET_ASSERT(adjacent_uids_1.size() != 0);
614 for (const Node& neighbour : adjacent_uids_1) {
615 if (candidate_swaps.find({neighbour, assigned_second}) ==
616 candidate_swaps.end()) {
617 candidate_swaps.insert({assigned_second, neighbour});
618 }
619 }
620 }
621 return candidate_swaps;
622}

624bool is_vertex_CX(const Circuit& circ_, const Vertex& v) {
625 Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
626 while (op->get_type() == OpType::Conditional) {
627 op = static_cast<const Conditional&>(*op).get_op();
628 }
629 return op->get_type() == OpType::CX;
630}

632std::pair<bool, bool> LexiRoute::check_bridge(
633 const std::pair<Node, Node>& swap, unsigned lookahead) {
634 std::pair<bool, bool> output = {false, false};
635 // first confirm whether it even has an interaction
636 auto it = this->interacting_uids_.find(swap.first);
637 if (it != this->interacting_uids_.end()) { // => in interaction
638 if (this->architecture_->get_distance(swap.first, Node(it->second)) ==
639 2) { // => could be bridge
640 // below should always return correct object given prior checks
641 VertPort vp =
642 (*this->mapping_frontier_->linear_boundary->find(swap.first)).second;
643 Edge out_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
644 vp.first, vp.second);
645 output.first = is_vertex_CX(
646 this->mapping_frontier_->circuit_,
647 this->mapping_frontier_->circuit_.target(out_edge));
648 }
649 }
650 // repeat for second swap
651 it = this->interacting_uids_.find(swap.second);
652 if (it != this->interacting_uids_.end()) {
653 if (this->architecture_->get_distance(swap.second, Node(it->second)) == 2) {
654 VertPort vp =
655 (*this->mapping_frontier_->linear_boundary->find(swap.second)).second;
656 Edge out_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
657 vp.first, vp.second);
658 output.second = is_vertex_CX(
659 this->mapping_frontier_->circuit_,
660 this->mapping_frontier_->circuit_.target(out_edge));
661 }
662 }
663 if ((output.first && output.second) || (!output.first && !output.second)) {
664 return {0, 0};
665 }
666 // implies conditions are set to at least check if BRIDGE is better
667 swap_set_t candidate_swaps = {
668 swap,
669 {swap.first,
670 swap.first}}; // second swap here will just compare the base case
671
672 // as with best swap finder, we create a set of candidate swap gates and
673 // then find best, except with only 2 swap (best swap and no swap)
674 while (candidate_swaps.size() > 1 /*some lookahead parameter*/) {
675 this->mapping_frontier_->advance_next_2qb_slice(lookahead);
676 // true bool means it only sets interacting uids if both uids are in
677 // architecture
678 this->set_interacting_uids(
679 AssignedOnly::Yes, CheckRoutingValidity::No,
680 CheckLabellingValidity::No);
681 // if 0, just take first swap rather than place
682 if (this->interacting_uids_.size() == 0) {
683 candidate_swaps = {*candidate_swaps.begin()};
684 } else {
685 interacting_nodes_t convert_uids;
686 for (const auto& p : this->interacting_uids_) {
687 convert_uids.insert(
688 {Node(this->labelling_[p.first]),
689 Node(this->labelling_[p.second])});
690 }
691 LexicographicalComparison lookahead_lc(this->architecture_, convert_uids);
692 lookahead_lc.remove_swaps_lexicographical(candidate_swaps);
693 }
694 }
695 // condition implies bridge is chosen
696 // if both remained then lexicographically equivalent under given conditions
697 // so either can be added with same consequences (for given hyper
698 // parameters)
699 if (*candidate_swaps.begin() == swap) {
700 output = {0, 0};
701 }
702 return output;
703}

705// Returns the distance between n1 and p1 and the distance between n2 and p2,
706// distance ordered (greatest first)
707const std::pair<std::size_t, std::size_t> LexiRoute::pair_distances(
708 const Node& p0_first, const Node& p0_second, const Node& p1_first,
709 const Node& p1_second) const {
710 {
711 const bool valid = this->architecture_->node_exists(p0_first) &&
712 this->architecture_->node_exists(p0_second) &&
713 this->architecture_->node_exists(p1_first) &&
714 this->architecture_->node_exists(p1_second);
715 TKET_ASSERT(valid);
716 }
717 std::size_t curr_dist1 =
718 this->architecture_->get_distance(p0_first, p0_second);
719 std::size_t curr_dist2 =
720 this->architecture_->get_distance(p1_first, p1_second);
721 return (curr_dist1 > curr_dist2) ? std::make_pair(curr_dist1, curr_dist2)
722 : std::make_pair(curr_dist2, curr_dist1);
723}

725void LexiRoute::remove_swaps_decreasing(swap_set_t& swaps) {
726 swap_set_t remaining_swaps;
727 Node pair_first, pair_second;
728 for (const auto& swap : swaps) {
729 auto it = this->interacting_uids_.find(swap.first);
730 // => swap.first is in interaction
731 if (it != this->interacting_uids_.end()) {
732 // find its pair
733 pair_first = Node(it->second);
734 } else {
735 // => not interacting, assign pair to self (will give lexicographic
736 // distance 0)
737 pair_first = swap.first;
738 }
739 // => UnitID in SWAP are interacting
740 if (pair_first == swap.second) {
741 continue;
742 }
743 auto jt = this->interacting_uids_.find(swap.second);
744 // => swap.second is in interaction
745 if (jt != this->interacting_uids_.end()) {
746 pair_second = Node(jt->second);
747 } else {
748 pair_second = swap.second;
749 }
750 // => UnitID in SWAP are interacting
751 // Check should already be done with earlier continue
752 TKET_ASSERT(pair_second != swap.first);
753
754 const std::pair<std::size_t, std::size_t>& curr_dists =
755 this->pair_distances(swap.first, pair_first, swap.second, pair_second);
756 const std::pair<std::size_t, std::size_t>& news_dists =
757 this->pair_distances(swap.second, pair_first, swap.first, pair_second);
758 if (news_dists >= curr_dists) {
759 continue;
760 }
761 remaining_swaps.insert(swap);
762 }
763}

775bool LexiRoute::solve(unsigned lookahead) {
776 // work out if valid
777
778 bool all_labelled = this->set_interacting_uids(
779 AssignedOnly::No, CheckRoutingValidity::Yes, CheckLabellingValidity::No);
780 if (!all_labelled) {
781 return false;
782 }
783
784 // store a copy of the original this->mapping_frontier_->quantum_boundray
785 // this object will be updated and reset throughout the swap picking
786 // procedure so need to return it to original setting at end
787 unit_vertport_frontier_t copy;
788 for (const std::pair<UnitID, VertPort>& pair :
789 this->mapping_frontier_->linear_boundary->get<TagKey>()) {
790 copy.insert({pair.first, pair.second});
791 }
792 swap_set_t candidate_swaps = this->get_candidate_swaps();
793 this->remove_swaps_decreasing(candidate_swaps);
794 TKET_ASSERT(candidate_swaps.size() != 0);
795 // Only want to substitute a single swap
796 // check next layer of interacting qubits and remove swaps until only one
797 // lexicographically superior swap is left
798 unsigned counter = 0;
799 while (candidate_swaps.size() > 1 && counter < lookahead) {
800 // if 0, just take first swap rather than place
801 if (this->interacting_uids_.size() == 0) {
802 break;
803 } else {
804 interacting_nodes_t convert_uids;
805 for (const auto& p : this->interacting_uids_) {
806 convert_uids.insert(
807 {Node(this->labelling_[p.first]),
808 Node(this->labelling_[p.second])});
809 }
810 LexicographicalComparison lookahead_lc(this->architecture_, convert_uids);
811 lookahead_lc.remove_swaps_lexicographical(candidate_swaps);
812 }
813 counter++;
814 this->mapping_frontier_->advance_next_2qb_slice(lookahead);
815 // true bool means it only sets interacting uids if both uids are in
816 // architecture
817 this->set_interacting_uids(
818 AssignedOnly::Yes, CheckRoutingValidity::No,
819 CheckLabellingValidity::No);
820 }
821 // find best swap
822 auto it = candidate_swaps.end();
823 --it;
824
825 std::pair<Node, Node> chosen_swap = *it;
826 this->mapping_frontier_->set_linear_boundary(copy);
827
828 this->set_interacting_uids(
829 AssignedOnly::No, CheckRoutingValidity::No, CheckLabellingValidity::No);
830 std::pair<bool, bool> check = this->check_bridge(chosen_swap, lookahead);
831 // set for final time, to allow gates to be correctly inserted, but then
832 // leave as is insert gates
833 this->mapping_frontier_->set_linear_boundary(copy);
834 if (!check.first && !check.second) {
835 // update circuit with new swap
836 // final_labelling is initial labelling permuted by single swap
837
838 // if false, SWAP is identical to last SWAP added without any gates
839 // realised
840 if (!this->mapping_frontier_->add_swap(
841 chosen_swap.first, chosen_swap.second)) {
842 // only need to reset in bridge case
843 this->set_interacting_uids(
844 AssignedOnly::No, CheckRoutingValidity::No,
845 CheckLabellingValidity::No);
846 // if SWAP has both Nodes in interaction default takes first
847 // this could be inefficient compared to other SWAP, however
848 // this failsafe is expected to be called extremely rarely (once in
849 // thousands) on very dense circuits for very symmetrical architectures
850 // so doesn't largely matter
851 auto it = this->interacting_uids_.find(chosen_swap.first);
852 auto add_path_swaps = [this](const Node& source, const Node& target) {
853 auto path = this->architecture_->get_path(source, target);
854 auto path_it_0 = path.begin() + 1;
855 auto path_it_1 = path.begin();
856 // adds a SWAP between each pair of adjacent nodes on path
857 while (path_it_0 != path.end() - 1) {
858 this->mapping_frontier_->add_swap(*path_it_0, *path_it_1);
859 ++path_it_0;
860 ++path_it_1;
861 }
862 };
863 if (it != this->interacting_uids_.end()) {
864 add_path_swaps(chosen_swap.first, Node(it->second));
865 } else {
866 it = this->interacting_uids_.find(chosen_swap.second);
867 TKET_ASSERT(it != this->interacting_uids_.end());
868 add_path_swaps(chosen_swap.second, Node(it->second));
869 }
870 }
871
872 } else {
873 // only need to reset in bridge case
874 this->set_interacting_uids(
875 AssignedOnly::No, CheckRoutingValidity::No, CheckLabellingValidity::No);
876
877 auto add_ordered_bridge = [&](const Node& n) {
878 auto it0 = this->mapping_frontier_->linear_boundary->find(n);
879 // this should implicitly be the case if this logic is reached
880 TKET_ASSERT(it0 != this->mapping_frontier_->linear_boundary->end());
881
882 Node other_node = Node(this->interacting_uids_[n]);
883 auto it1 = this->mapping_frontier_->linear_boundary->find(other_node);
884 // this should implicitly be the case if this logic is reached
885 TKET_ASSERT(it1 != this->mapping_frontier_->linear_boundary->end());
886
887 auto path = this->architecture_->get_path(n, other_node);
888 Node central = Node(path[1]);
889
890 Edge n_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
891 it0->second.first, it0->second.second);
892 Edge other_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
893 it1->second.first, it1->second.second);
894
895 unsigned port0 =
896 this->mapping_frontier_->circuit_.get_target_port(n_edge);
897 unsigned port1 =
898 this->mapping_frontier_->circuit_.get_target_port(other_edge);
899 // compare port ordering to get control vs target
900 TKET_ASSERT(port0 != port1);
901 if (port0 < port1) {
902 this->mapping_frontier_->add_bridge(n, central, other_node);
903 } else {
904 this->mapping_frontier_->add_bridge(other_node, central, n);
905 }
906 };
907
908 if (check.first) {
909 add_ordered_bridge(chosen_swap.first);
910 }
911 if (check.second) {
912 add_ordered_bridge(chosen_swap.second);
913 }
914 }
915 return true;
916}
