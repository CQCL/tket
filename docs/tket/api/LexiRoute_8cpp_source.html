37 void LexiRoute::reassign_to_any_ancilla_node(const Node& reassign_node) {
47  TKET_ASSERT(!this->mapping_frontier_->ancilla_nodes_.empty());
48  auto ancilla_it = this->mapping_frontier_->ancilla_nodes_.begin();
49  Node ancilla_node = *ancilla_it;
50  this->mapping_frontier_->ancilla_nodes_.erase(ancilla_it);
51  this->mapping_frontier_->merge_ancilla(reassign_node, ancilla_node);
52  auto it = std::find_if(
53  this->labelling_.begin(), this->labelling_.end(),
54  [&reassign_node](const std::pair<const UnitID, UnitID>& pair_unitid) {

57  TKET_ASSERT(it != this->labelling_.end());
58  this->labelling_[it->first] = ancilla_node;
59  unit_map_t relabel = {{UnitID(reassign_node), UnitID(ancilla_node)}};
60  this->mapping_frontier_->circuit_.rename_units(relabel);

63 bool LexiRoute::reassign_to_any_spare_node(const Node& reassign_node) {
69  std::set<Node> all_nodes_s;
70  for (const std::pair<const UnitID, UnitID>& p : this->labelling_) {
71  all_nodes_s.insert(Node(p.second));
72  }
73  std::set<Node> architecture_nodes = this->architecture_->nodes();
74  auto it = std::find_if(
75  architecture_nodes.begin(), architecture_nodes.end(),
76  [&all_nodes_s](const Node& node) {
77  return all_nodes_s.find(node) == all_nodes_s.end();
78  });
79 
80  if (it == architecture_nodes.end()) {
81  return false;
82  }
83 
92  Node spare_node = *it;
93 
94  auto jt = std::find_if(
95  this->labelling_.begin(), this->labelling_.end(),
96  [&reassign_node](const std::pair<const UnitID, UnitID>& pair_unitid) {
97  return pair_unitid.second == reassign_node;
98  });
99  TKET_ASSERT(jt != this->labelling_.end());
100  this->labelling_[jt->first] = spare_node;
105  auto reassign_node_reassignable_it =
106  this->mapping_frontier_->reassignable_nodes_.find(reassign_node);
107  TKET_ASSERT(
108  reassign_node_reassignable_it !=
109  this->mapping_frontier_->reassignable_nodes_.end());
110  this->mapping_frontier_->reassignable_nodes_.erase(
111  reassign_node_reassignable_it);
112  this->mapping_frontier_->reassignable_nodes_.insert(spare_node);
113 
118  auto reassign_node_boundary_it =
119  this->mapping_frontier_->linear_boundary->get<TagKey>().find(
120  reassign_node);
121  TKET_ASSERT(
122  reassign_node_boundary_it !=
123  this->mapping_frontier_->linear_boundary->get<TagKey>().end());
124  this->mapping_frontier_->linear_boundary->replace(
125  reassign_node_boundary_it,
126  {spare_node, reassign_node_boundary_it->second});
127 
132  unit_map_t relabel = {{UnitID(reassign_node), UnitID(spare_node)}};
133  this->mapping_frontier_->circuit_.rename_units(relabel);
134 
139  auto initial_reassign_it =
140  this->mapping_frontier_->bimaps_->initial.right.find(reassign_node);
141  TKET_ASSERT(
142  initial_reassign_it !=
143  this->mapping_frontier_->bimaps_->initial.right.end());
144  UnitID preserved_orig = initial_reassign_it->second;
145  std::pair<UnitID, UnitID> initial_replacement = {preserved_orig, spare_node};
146  this->mapping_frontier_->bimaps_->initial.right.erase(initial_reassign_it);
147  this->mapping_frontier_->bimaps_->initial.left.insert(initial_replacement);
148  auto final_reassign_it =
149  this->mapping_frontier_->bimaps_->final.left.find(preserved_orig);
150  TKET_ASSERT(
151  final_reassign_it != this->mapping_frontier_->bimaps_->final.left.end());
152  std::pair<UnitID, UnitID> final_replacement = {preserved_orig, spare_node};
153  this->mapping_frontier_->bimaps_->final.left.erase(final_reassign_it);
154  this->mapping_frontier_->bimaps_->final.left.insert(final_replacement);
155 
159  this->assigned_nodes_.insert(spare_node);
160  return true;
161 }

234 // "assignee" is Circuit UnitID being relabelled to UnitID "replacement"
235 void LexiRoute::assign_valid_node(
236  const UnitID& assignee, const UnitID& replacement) {
248  if (this->mapping_frontier_->reassignable_nodes_.find(Node(replacement)) !=
249  this->mapping_frontier_->reassignable_nodes_.end()) {
254  this->reassign_node(Node(replacement), assignee);
255  return;
256  }
257  if (this->mapping_frontier_->ancilla_nodes_.find(Node(replacement)) !=
258  this->mapping_frontier_->ancilla_nodes_.end()) {
265  this->mapping_frontier_->merge_ancilla(assignee, replacement);
266  this->labelling_.erase(replacement);
267  this->labelling_[assignee] = replacement;
268  return;
269  }
270 
275  this->labelling_[assignee] = replacement;
276  this->assigned_nodes_.insert(Node(replacement));
277 
278  // bimaps are used by CompilationUnit to track how an original Circuit Qubit
279  // may be relabelled update this to reflect new assignment first do bimaps
280  // "RHS"
281  auto initial_assignee_it =
282  this->mapping_frontier_->bimaps_->initial.right.find(assignee);
283  TKET_ASSERT(
284  initial_assignee_it !=
285  this->mapping_frontier_->bimaps_->initial.right.end());
286  UnitID original = initial_assignee_it->second;
287  this->mapping_frontier_->bimaps_->initial.right.erase(initial_assignee_it);
288  this->mapping_frontier_->bimaps_->initial.left.insert(
289  {original, replacement});
290 
291  // then bimaps "LHS"
292  auto final_assignee_it =
293  this->mapping_frontier_->bimaps_->final.left.find(original);
294  TKET_ASSERT(
295  final_assignee_it != this->mapping_frontier_->bimaps_->final.left.end());
296  this->mapping_frontier_->bimaps_->final.left.erase(final_assignee_it);
297  this->mapping_frontier_->bimaps_->final.left.insert({original, replacement});
298 
299  // Update linear boundaries
300  auto assignee_boundary_it =
301  this->mapping_frontier_->linear_boundary->get<TagKey>().find(assignee);
302  TKET_ASSERT(
303  assignee_boundary_it !=
304  this->mapping_frontier_->linear_boundary->get<TagKey>().end());
305  this->mapping_frontier_->linear_boundary->replace(
306  assignee_boundary_it, {replacement, assignee_boundary_it->second});
307 
308  unit_map_t relabel = {{UnitID(assignee), UnitID(replacement)}};
309  // Relabel Circuit UnitID
310  this->mapping_frontier_->circuit_.rename_units(relabel);
311 }

313 bool LexiRoute::assign_at_distance(
314  const UnitID& assignee, const Node& root, unsigned distances) {
322  node_set_t valid_nodes;
323  for (const Node& neighbour :
324  this->architecture_->nodes_at_distance(root, distances)) {
325  // A node is unassigned if it's empty or holding an ancilla, or can be
326  // reassigned
327  if (this->mapping_frontier_->reassignable_nodes_.find(neighbour) !=
328  this->mapping_frontier_->reassignable_nodes_.end() ||
329  this->assigned_nodes_.find(neighbour) == this->assigned_nodes_.end() ||
330  this->mapping_frontier_->ancilla_nodes_.find(neighbour) !=
331  this->mapping_frontier_->ancilla_nodes_.end()) {
332  valid_nodes.insert(neighbour);
333  }
334  }
335 
339  if (valid_nodes.empty()) {
340  return false;
341  }
342  auto it = valid_nodes.begin();
346  if (valid_nodes.size() == 1) {
347  this->assign_valid_node(assignee, Node(*it));
348  return true;
349  }
354  lexicographical_distances_t winning_distances =
355  this->architecture_->get_distances(*it);
356  Node preserved_node = *it;
357  ++it;
358  while (it != valid_nodes.end()) {
359  lexicographical_distances_t comparison_distances =
360  this->architecture_->get_distances(*it);
361  if (comparison_distances < winning_distances) {
362  preserved_node = *it;
363  winning_distances = comparison_distances;
364  }
365  ++it;
366  }
370  this->assign_valid_node(assignee, preserved_node);
371  return true;
372 }

374 bool LexiRoute::update_labelling() {
375  // iterate through interacting qubits, assigning them to an Architecture
376  // Node if they aren't already
377  bool relabelled = false;
378  for (const auto& pair : this->interacting_uids_) {
379  bool uid_0_exist =
380  this->architecture_->node_exists(Node(this->labelling_[pair.first]));
381  bool uid_1_exist =
382  this->architecture_->node_exists(Node(this->labelling_[pair.second]));
383  if (!uid_0_exist || !uid_1_exist) {
384  relabelled = true;
385  }
386  if (!uid_0_exist && !uid_1_exist) {
399  if (this->assigned_nodes_.size() == 0) {
406  // Find Architecture Node with best out degree
407  std::set<Node> max_degree_nodes =
408  this->architecture_->max_degree_nodes();
409  auto it = max_degree_nodes.begin();
410 
421  lexicographical_distances_t winning_distances =
422  this->architecture_->get_distances(*it);
423  Node preserved_node = Node(*it);
424  ++it;
425  for (; it != max_degree_nodes.end(); ++it) {
426  lexicographical_distances_t comparison_distances =
427  this->architecture_->get_distances(*it);
428  if (comparison_distances < winning_distances) {
429  preserved_node = Node(*it);
430  winning_distances = comparison_distances;
431  }
432  }
433  // assign unplaced Qubit to "best" Node
434  this->labelling_[pair.first] = preserved_node;
435  this->assigned_nodes_.insert(preserved_node);
436 
437  // update bimaps
438  auto initial_assignee_it =
439  this->mapping_frontier_->bimaps_->initial.right.find(pair.first);
440  TKET_ASSERT(
441  initial_assignee_it !=
442  this->mapping_frontier_->bimaps_->initial.right.end());
443  UnitID original = initial_assignee_it->second;
444  this->mapping_frontier_->bimaps_->initial.right.erase(
445  initial_assignee_it);
446  this->mapping_frontier_->bimaps_->initial.left.insert(
447  {original, preserved_node});
448 
449  auto final_assignee_it =
450  this->mapping_frontier_->bimaps_->final.left.find(original);
451  this->mapping_frontier_->bimaps_->final.left.erase(final_assignee_it);
452  this->mapping_frontier_->bimaps_->final.left.insert(
453  {original, preserved_node});
454 
455  // this prompts next if statement to place "uid1"
456  uid_0_exist = true;
457 
458  // update circuit with new labelling
459  unit_map_t relabel = {{UnitID(pair.first), UnitID(preserved_node)}};
460  this->mapping_frontier_->circuit_.rename_units(relabel);
461 
462  // update linear boundary
463  auto pair_first_boundary_it =
464  this->mapping_frontier_->linear_boundary->get<TagKey>().find(
465  pair.first);
466  this->mapping_frontier_->linear_boundary->replace(
467  pair_first_boundary_it,
468  {preserved_node, pair_first_boundary_it->second});
469 
470  // given best node, do something
471  } else {
478  auto root_it = this->assigned_nodes_.begin();
479  while (!uid_0_exist && root_it != this->assigned_nodes_.end()) {
480  Node root = *root_it;
481  uid_0_exist = this->assign_at_distance(pair.first, root, 1);
482  ++root_it;
483  }
484  if (!uid_0_exist) {
485  throw LexiRouteError(
486  "Unable to assign physical qubit - no free qubits remaining.");
487  }
488  }
489  }
490  if (!uid_0_exist && uid_1_exist) {
491  Node root(this->labelling_[pair.second]);
492  for (unsigned k = 1; k <= this->architecture_->get_diameter(); k++) {
493  if (this->assign_at_distance(pair.first, root, k)) {
494  uid_0_exist = true;
495  break;
496  }
497  }
498  if (!uid_0_exist) {
499  throw LexiRouteError(
500  "Unable to assign physical qubit - no free qubits remaining.");
501  }
502  }
503  if (uid_0_exist && !uid_1_exist) {
504  Node root(this->labelling_[pair.first]);
505  for (unsigned k = 1; k <= this->architecture_->get_diameter(); k++) {
506  if (this->assign_at_distance(pair.second, root, k)) {
507  uid_1_exist = true;
508  break;
509  }
510  }
511  if (!uid_1_exist) {
512  throw LexiRouteError(
513  "Unable to assign physical qubit - no free qubits remaining.");
514  }
515  }
516  }
517  return relabelled;
518 }

525 bool LexiRoute::set_interacting_uids(
526  AssignedOnly assigned_only, CheckRoutingValidity route_check,
527  CheckLabellingValidity label_check) {
528  // return types
529  this->interacting_uids_.clear();
530  bool all_placed = true;
531  for (auto it =
532  this->mapping_frontier_->linear_boundary->get<TagKey>().begin();
533  it != this->mapping_frontier_->linear_boundary->get<TagKey>().end();
534  ++it) {
535  Edge e0 = this->mapping_frontier_->circuit_.get_nth_out_edge(
536  it->second.first, it->second.second);
537  Vertex v0 = this->mapping_frontier_->circuit_.target(e0);
538  // should never be input vertex, so can always use in_edges
539  Op_ptr op = this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(v0);
540  if (op->get_type() != OpType::Barrier) {
541  int n_edges = this->mapping_frontier_->circuit_.n_in_edges_of_type(
542  v0, EdgeType::Quantum);
543  // make forwards = backwards
544  if (n_edges == 2) {
545  auto jt = it;
546  ++jt;
547  while (jt !=
548  this->mapping_frontier_->linear_boundary->get<TagKey>().end()) {
549  // i.e. if vertices match
550  Edge e1 = this->mapping_frontier_->circuit_.get_nth_out_edge(
551  jt->second.first, jt->second.second);
552  Vertex v1 = this->mapping_frontier_->circuit_.target(e1);
553  if (v0 == v1) {
554  // we can assume a qubit will only be in one interaction
555  // we can assume from how we iterate through pairs that each qubit
556  // will only be found in one match
557  bool node0_exists =
558  this->architecture_->node_exists(Node(it->first));
559  bool node1_exists =
560  this->architecture_->node_exists(Node(jt->first));
561  if (!node0_exists || !node1_exists || op->get_desc().is_box()) {
562  all_placed = false;
563  if (route_check == CheckRoutingValidity::Yes) return false;
564  }
565 
566  if (assigned_only == AssignedOnly::No ||
567  (node0_exists && node1_exists)) {
568  interacting_uids_.insert({it->first, jt->first});
569  interacting_uids_.insert({jt->first, it->first});
570  }
571  }
572  ++jt;
573  }
574  }
575  }
576  }
577 
578  // conditions for proceeding with labelling
579  if (label_check == CheckLabellingValidity::Yes) {
580  if (all_placed) {
581  return true;
582  } else {
583  return false;
584  }
585  }
586  // this should have left early when first found
587  if (route_check == CheckRoutingValidity::Yes) {
588  if (all_placed && interacting_uids_.size() > 0) {
589  return true;
590  }
591  return false;
592  }
593  // => either route_check true and all_placed so valid
594  // or !route_check and !label_check so return true and discard
595  return true;
596 }

598 swap_set_t LexiRoute::get_candidate_swaps() {
599  swap_set_t candidate_swaps;
600  for (const auto& interaction : this->interacting_uids_) {
601  Node assigned_first = Node(this->labelling_[interaction.first]);
602  std::vector<Node> adjacent_uids_0 =
603  this->architecture_->nodes_at_distance(assigned_first, 1);
604  TKET_ASSERT(adjacent_uids_0.size() != 0);
605  for (const Node& neighbour : adjacent_uids_0) {
606  if (candidate_swaps.find({neighbour, assigned_first}) ==
607  candidate_swaps.end()) {
608  candidate_swaps.insert({assigned_first, neighbour});
609  }
610  }
611  Node assigned_second = Node(this->labelling_[interaction.second]);
612  std::vector<Node> adjacent_uids_1 =
613  this->architecture_->nodes_at_distance(assigned_second, 1);
614  TKET_ASSERT(adjacent_uids_1.size() != 0);
615  for (const Node& neighbour : adjacent_uids_1) {
616  if (candidate_swaps.find({neighbour, assigned_second}) ==
617  candidate_swaps.end()) {
618  candidate_swaps.insert({assigned_second, neighbour});
619  }
620  }
621  }
622  return candidate_swaps;
623 }

625 bool is_vertex_CX(const Circuit& circ_, const Vertex& v) {
626  OpType ot = circ_.get_OpType_from_Vertex(v);
627  if (ot != OpType::CX) {
628  if (ot == OpType::Conditional) {
629  const Conditional& b =
630  static_cast<const Conditional&>(*circ_.get_Op_ptr_from_Vertex(v));
631  if (b.get_op()->get_type() != OpType::CX) {
632  return false;
633  }
634  } else {
635  return false;
636  }
637  }
638  return true;

641 std::pair<bool, bool> LexiRoute::check_bridge(
642  const std::pair<Node, Node>& swap, unsigned lookahead) {
643  std::pair<bool, bool> output = {false, false};
644  // first confirm whether it even has an interaction
645  auto it = this->interacting_uids_.find(swap.first);
646  if (it != this->interacting_uids_.end()) { // => in interaction
647  if (this->architecture_->get_distance(swap.first, Node(it->second)) ==
648  2) { // => could be bridge
649  // below should always return correct object given prior checks
650  VertPort vp =
651  (*this->mapping_frontier_->linear_boundary->find(swap.first)).second;
652  Edge out_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
653  vp.first, vp.second);
654  output.first = is_vertex_CX(
655  this->mapping_frontier_->circuit_,
656  this->mapping_frontier_->circuit_.target(out_edge));
657  }
658  }
659  // repeat for second swap
660  it = this->interacting_uids_.find(swap.second);
661  if (it != this->interacting_uids_.end()) {
662  if (this->architecture_->get_distance(swap.second, Node(it->second)) == 2) {
663  VertPort vp =
664  (*this->mapping_frontier_->linear_boundary->find(swap.second)).second;
665  Edge out_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
666  vp.first, vp.second);
667  output.second = is_vertex_CX(
668  this->mapping_frontier_->circuit_,
669  this->mapping_frontier_->circuit_.target(out_edge));
670  }
671  }
672  if ((output.first && output.second) || (!output.first && !output.second)) {
673  return {0, 0};
674  }
675  // implies conditions are set to at least check if BRIDGE is better
676  swap_set_t candidate_swaps = {
677  swap,
678  {swap.first,
679  swap.first}}; // second swap here will just compare the base case
680 
681  // as with best swap finder, we create a set of candidate swap gates and
682  // then find best, except with only 2 swap (best swap and no swap)
683  while (candidate_swaps.size() > 1 /*some lookahead parameter*/) {
684  this->mapping_frontier_->advance_next_2qb_slice(lookahead);
685  // true bool means it only sets interacting uids if both uids are in
686  // architecture
687  this->set_interacting_uids(
688  AssignedOnly::Yes, CheckRoutingValidity::No,
689  CheckLabellingValidity::No);
690  // if 0, just take first swap rather than place
691  if (this->interacting_uids_.size() == 0) {
692  candidate_swaps = {*candidate_swaps.begin()};
693  } else {
694  interacting_nodes_t convert_uids;
695  for (const auto& p : this->interacting_uids_) {
696  convert_uids.insert(
697  {Node(this->labelling_[p.first]),
698  Node(this->labelling_[p.second])});
699  }
700  LexicographicalComparison lookahead_lc(this->architecture_, convert_uids);
701  lookahead_lc.remove_swaps_lexicographical(candidate_swaps);
702  }
703  }
704  // condition implies bridge is chosen
705  // if both remained then lexicographically equivalent under given conditions
706  // so either can be added with same consequences (for given hyper
707  // parameters)
708  if (*candidate_swaps.begin() == swap) {
709  output = {0, 0};
710  }
711  return output;
712 }

714 // Returns the distance between n1 and p1 and the distance between n2 and p2,
715 // distance ordered (greatest first)
716 const std::pair<size_t, size_t> LexiRoute::pair_distances(
717  const Node& p0_first, const Node& p0_second, const Node& p1_first,
718  const Node& p1_second) const {
719  {
720  const bool valid = this->architecture_->node_exists(p0_first) &&
721  this->architecture_->node_exists(p0_second) &&
722  this->architecture_->node_exists(p1_first) &&
723  this->architecture_->node_exists(p1_second);
724  TKET_ASSERT(valid);
725  }
726  size_t curr_dist1 = this->architecture_->get_distance(p0_first, p0_second);
727  size_t curr_dist2 = this->architecture_->get_distance(p1_first, p1_second);
728  return (curr_dist1 > curr_dist2) ? std::make_pair(curr_dist1, curr_dist2)
729  : std::make_pair(curr_dist2, curr_dist1);
730 }

732 void LexiRoute::remove_swaps_decreasing(swap_set_t& swaps) {
733  swap_set_t remaining_swaps;
734  Node pair_first, pair_second;
735  for (const auto& swap : swaps) {
736  auto it = this->interacting_uids_.find(swap.first);
737  // => swap.first is in interaction
738  if (it != this->interacting_uids_.end()) {
739  // find its pair
740  pair_first = Node(it->second);
741  } else {
742  // => not interacting, assign pair to self (will give lexicographic
743  // distance 0)
744  pair_first = swap.first;
745  }
746  // => UnitID in SWAP are interacting
747  if (pair_first == swap.second) {
748  continue;
749  }
750  auto jt = this->interacting_uids_.find(swap.second);
751  // => swap.second is in interaction
752  if (jt != this->interacting_uids_.end()) {
753  pair_second = Node(jt->second);
754  } else {
755  pair_second = swap.second;
756  }
757  // => UnitID in SWAP are interacting
758  // Check should alrady be done with earlier continue
759  TKET_ASSERT(pair_second != swap.first);
760 
761  const std::pair<size_t, size_t>& curr_dists =
762  this->pair_distances(swap.first, pair_first, swap.second, pair_second);
763  const std::pair<size_t, size_t>& news_dists =
764  this->pair_distances(swap.second, pair_first, swap.first, pair_second);
765  if (news_dists >= curr_dists) {
766  continue;
767  }
768  remaining_swaps.insert(swap);
769  }
770 }

782 bool LexiRoute::solve(unsigned lookahead) {
783  // work out if valid
784 
785  bool all_labelled = this->set_interacting_uids(
786  AssignedOnly::No, CheckRoutingValidity::Yes, CheckLabellingValidity::No);
787  if (!all_labelled) {
788  return false;
789  }
790 
791  // store a copy of the original this->mapping_frontier_->quantum_boundray
792  // this object will be updated and reset throughout the swap picking
793  // procedure so need to return it to original setting at end
794  unit_vertport_frontier_t copy;
795  for (const std::pair<UnitID, VertPort>& pair :
796  this->mapping_frontier_->linear_boundary->get<TagKey>()) {
797  copy.insert({pair.first, pair.second});
798  }
799  swap_set_t candidate_swaps = this->get_candidate_swaps();
800  this->remove_swaps_decreasing(candidate_swaps);
801  TKET_ASSERT(candidate_swaps.size() != 0);
802  // Only want to substitute a single swap
803  // check next layer of interacting qubits and remove swaps until only one
804  // lexicographically superior swap is left
805  unsigned counter = 0;
806  while (candidate_swaps.size() > 1 && counter < lookahead) {
807  // if 0, just take first swap rather than place
808  if (this->interacting_uids_.size() == 0) {
809  break;
810  } else {
811  interacting_nodes_t convert_uids;
812  for (const auto& p : this->interacting_uids_) {
813  convert_uids.insert(
814  {Node(this->labelling_[p.first]),
815  Node(this->labelling_[p.second])});
816  }
817  LexicographicalComparison lookahead_lc(this->architecture_, convert_uids);
818  lookahead_lc.remove_swaps_lexicographical(candidate_swaps);
819  }
820  counter++;
821  this->mapping_frontier_->advance_next_2qb_slice(lookahead);
822  // true bool means it only sets interacting uids if both uids are in
823  // architecture
824  this->set_interacting_uids(
825  AssignedOnly::Yes, CheckRoutingValidity::No,
826  CheckLabellingValidity::No);
827  }
828  // find best swap
829  auto it = candidate_swaps.end();
830  --it;
831 
832  std::pair<Node, Node> chosen_swap = *it;
833  this->mapping_frontier_->set_linear_boundary(copy);
834 
835  this->set_interacting_uids(
836  AssignedOnly::No, CheckRoutingValidity::No, CheckLabellingValidity::No);
837  std::pair<bool, bool> check = this->check_bridge(chosen_swap, lookahead);
838  // set for final time, to allow gates to be correctly inserted, but then
839  // leave as is insert gates
840  this->mapping_frontier_->set_linear_boundary(copy);
841  if (!check.first && !check.second) {
842  // update circuit with new swap
843  // final_labelling is initial labelling permuted by single swap
844 
845  // if false, SWAP is identical to last SWAP added without any gates
846  // realised
847  if (!this->mapping_frontier_->add_swap(
848  chosen_swap.first, chosen_swap.second)) {
849  // only need to reset in bridge case
850  this->set_interacting_uids(
851  AssignedOnly::No, CheckRoutingValidity::No,
852  CheckLabellingValidity::No);
853  // if SWAP has both Nodes in interaction default takes first
854  // this could be inefficient compared to other SWAP, however
855  // this failsafe is expected to be called extremely rarely (once in
856  // thousands) on very dense circuits for very symmetrical architectures
857  // so doesn't largely matter
858  auto it = this->interacting_uids_.find(chosen_swap.first);
859  auto add_path_swaps = [this](const Node& source, const Node& target) {
860  auto path = this->architecture_->get_path(source, target);
861  auto path_it_0 = path.begin() + 1;
862  auto path_it_1 = path.begin();
863  // adds a SWAP between each pair of adjacent nodes on path
864  while (path_it_0 != path.end() - 1) {
865  this->mapping_frontier_->add_swap(*path_it_0, *path_it_1);
866  ++path_it_0;
867  ++path_it_1;
868  }
869  };
870  if (it != this->interacting_uids_.end()) {
871  add_path_swaps(chosen_swap.first, Node(it->second));
872  } else {
873  it = this->interacting_uids_.find(chosen_swap.second);
874  TKET_ASSERT(it != this->interacting_uids_.end());
875  add_path_swaps(chosen_swap.second, Node(it->second));
876  }
877  }
878 
879  } else {
880  // only need to reset in bridge case
881  this->set_interacting_uids(
882  AssignedOnly::No, CheckRoutingValidity::No, CheckLabellingValidity::No);
883 
884  auto add_ordered_bridge = [&](const Node& n) {
885  auto it0 = this->mapping_frontier_->linear_boundary->find(n);
886  // this should implicitly be the case if this logic is reached
887  TKET_ASSERT(it0 != this->mapping_frontier_->linear_boundary->end());
888 
889  Node other_node = Node(this->interacting_uids_[n]);
890  auto it1 = this->mapping_frontier_->linear_boundary->find(other_node);
891  // this should implicitly be the case if this logic is reached
892  TKET_ASSERT(it1 != this->mapping_frontier_->linear_boundary->end());
893 
894  auto path = this->architecture_->get_path(n, other_node);
895  Node central = Node(path[1]);
896 
897  Edge n_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
898  it0->second.first, it0->second.second);
899  Edge other_edge = this->mapping_frontier_->circuit_.get_nth_out_edge(
900  it1->second.first, it1->second.second);
901 
902  unsigned port0 =
903  this->mapping_frontier_->circuit_.get_target_port(n_edge);
904  unsigned port1 =
905  this->mapping_frontier_->circuit_.get_target_port(other_edge);
906  // compare port ordering to get control vs target
907  TKET_ASSERT(port0 != port1);
908  if (port0 < port1) {
909  this->mapping_frontier_->add_bridge(n, central, other_node);
910  } else {
911  this->mapping_frontier_->add_bridge(other_node, central, n);
912  }
913  };
914 
915  if (check.first) {
916  add_ordered_bridge(chosen_swap.first);
917  }
918  if (check.second) {
919  add_ordered_bridge(chosen_swap.second);
920  }
921  }
922  return true;
923 }
