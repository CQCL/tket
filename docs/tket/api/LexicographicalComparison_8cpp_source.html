19 LexicographicalComparison::LexicographicalComparison(
20  const ArchitecturePtr& _architecture,
21  const interacting_nodes_t& _interacting_nodes)
22  : architecture_(_architecture), interacting_nodes_(_interacting_nodes) {
23  unsigned diameter = this->architecture_->get_diameter();
24 
25  lexicographical_distances_t distance_vector(diameter, 0);
26  for (const auto& interaction : this->interacting_nodes_) {
27  // If Node not in architecture, don't add
28  if (!this->architecture_->node_exists(interaction.first) ||
29  !this->architecture_->node_exists(interaction.second)) {
30  throw LexicographicalComparisonError(
31  "Constructor passed some interacting node not in architecture.");
32  }
33  // key->value already copied, assign reverse to map for later ease
34  this->interacting_nodes_[interaction.second] = interaction.first;
35  unsigned distance = this->architecture_->get_distance(
36  interaction.first, interaction.second);
37  if (distance > 0) {
38  ++distance_vector[diameter - distance];
39  }
40  }
41  this->lexicographical_distances = distance_vector;
42 }

44 void LexicographicalComparison::increment_distances(
45  lexicographical_distances_t& distances,
46  const std::pair<Node, Node>& interaction, int increment) const {
47  const unsigned distances_index =
48  this->architecture_->get_diameter() -
49  this->architecture_->get_distance(interaction.first, interaction.second);
50  if (distances[distances_index] == 0 && increment < 0) {
51  throw LexicographicalComparisonError(
52  "Negative increment value is larger than value held at index, "
53  "modification not allowed.");
54  }
55  distances[distances_index] += increment;
56 }

72 lexicographical_distances_t LexicographicalComparison::get_updated_distances(
73  const swap_t& swap) const {
74  // make a copy of base lexicographical distances
75  lexicographical_distances_t copy = this->lexicographical_distances;
76  if (swap.first == swap.second) {
77  return copy;
78  }
79  auto iq_it = this->interacting_nodes_.find(swap.first);
80  // first condition => first node not interacting with self, so update
81  // distances
82  if (iq_it != this->interacting_nodes_.end()) {
83  // update distances due to first swap node and qubit its interating with
84  // (assuming swap)
85  Node interacting = iq_it->second;
86  if (interacting != swap.second) {
87  increment_distances(copy, {swap.first, interacting}, -2);
88  // updates distances due to second swap node and qubit first is
89  // interacting with
90  increment_distances(copy, {swap.second, interacting}, 2);
91  }
92  }
93  iq_it = this->interacting_nodes_.find(swap.second);
94  // => second node not interacting with self
95  if (iq_it != this->interacting_nodes_.end()) {
96  Node interacting = iq_it->second;
97  if (interacting != swap.first) {
98  // update distances due to second node and qubit its interacting with
99  increment_distances(copy, {swap.second, interacting}, -2);
100  // update distannces due to frist node and qubit second node is
101  // interacting with
102  increment_distances(copy, {swap.first, interacting}, 2);
103  }
104  }
105  return copy;
106 }

120 void LexicographicalComparison::remove_swaps_lexicographical(
121  swap_set_t& candidate_swaps) const {
122  auto it = candidate_swaps.begin();
123  lexicographical_distances_t winning_distances =
124  this->get_updated_distances(*it);
125  swap_set_t preserved_swaps = {*it};
126  ++it;
127  for (; it != candidate_swaps.end(); ++it) {
128  lexicographical_distances_t comparison_distances =
129  this->get_updated_distances(*it);
130 
131  if (comparison_distances < winning_distances) {
132  preserved_swaps = {*it};
133  winning_distances = comparison_distances;
134  } else if (comparison_distances == winning_distances) {
135  preserved_swaps.insert(*it);
136  }
137  }
138  candidate_swaps = preserved_swaps;
139 }
