28 std::vector<qubit_vector_t> LinePlacement::interactions_to_lines(
29  const Circuit& circ_) const {
30  qubit_vector_t all_qubits_v = circ_.all_qubits();
31  std::vector<WeightedEdge> pattern_weighting =
32  this->default_pattern_weighting(circ_);
33  if (pattern_weighting.empty()) {
34  return {{}};
35  }
36  const QubitGraph q_graph =
37  this->construct_pattern_graph(pattern_weighting, all_qubits_v.size());
38  QubitGraph::UndirectedConnGraph uc_graph =
39  q_graph.get_undirected_connectivity();
40 
41  std::set<Qubit> all_qubits_s(all_qubits_v.begin(), all_qubits_v.end());
42 
43  std::vector<qubit_vector_t> found_lines;
44  unsigned found_line_size = 0;
45  do {
46  auto u_line = graphs::longest_simple_path(uc_graph);
47  qubit_vector_t found;
48  std::transform(
49  u_line.begin(), u_line.end(), std::back_inserter(found),
50  [&uc_graph](auto v) -> Qubit { return uc_graph[v]; });
51  found_line_size = found.size();
52  if (found_line_size > 1) {
53  found_lines.push_back(found);
54  for (const auto& vertex : u_line) {
55  boost::clear_vertex(vertex, uc_graph);
56  }
57  for (const Qubit& q : found) {
58  all_qubits_s.erase(q);
59  }
60  }
61  } while (found_line_size > 1);
62 
63  for (const Qubit& qb : all_qubits_v) {
64  if (all_qubits_s.find(qb) != all_qubits_s.end())
65  found_lines.push_back({qb});
66  }
67  return found_lines;
68 }

70 std::map<Qubit, Node> LinePlacement::assign_lines_to_target_graph(
71  std::vector<qubit_vector_t>& line_pattern, unsigned n_qubits) const {
72  unsigned n_unused_nodes = this->architecture_.n_nodes() - n_qubits;
73  // Sort lines from longest to shortest
74  std::sort(
75  line_pattern.begin(), line_pattern.end(),
76  [](qubit_vector_t x, qubit_vector_t y) {
77  size_t xsz = x.size(), ysz = y.size();
78  if (xsz > ysz) {
79  return true;
80  } else if (xsz < ysz) {
81  return false;
82  } else {
83  return std::lexicographical_compare(
84  x.begin(), x.end(), y.begin(), y.end());
85  }
86  });
87  // Remove single qubit lines
88  while (!line_pattern.empty() && line_pattern.back().size() < 2) {
89  n_unused_nodes++;
90  line_pattern.pop_back();
91  }
92  Architecture copy = this->architecture_;
93  node_set_t all_architecture_nodes = this->architecture_.nodes();
94  node_set_t bad_nodes;
95  for (const Node& node : all_architecture_nodes) {
96  if (this->architecture_.get_degree(node) == 0) {
97  bad_nodes.insert(node);
98  n_unused_nodes--;
99  }
100  }
101  node_set_t removed_nodes = copy.remove_worst_nodes(n_unused_nodes);
102  bad_nodes.insert(removed_nodes.begin(), removed_nodes.end());
103  node_set_t remaining_nodes;
104  std::set_difference(
105  all_architecture_nodes.begin(), all_architecture_nodes.end(),
106  bad_nodes.begin(), bad_nodes.end(),
107  std::inserter(remaining_nodes, remaining_nodes.begin()));
108 
109  // store lengths of lines to find
110  std::vector<unsigned> lengths;
111  for (const qubit_vector_t& line : line_pattern) {
112  lengths.push_back(line.size());
113  }
114  // get lines of Node to map Qubit to
115 
116  std::vector<node_vector_t> architecture_lines = copy.get_lines(lengths);
117  // and then produce a mapping between qubit and node lines
118  std::map<Qubit, Node> out_map;
119  for (unsigned i = 0; i < architecture_lines.size(); i++) {
120  qubit_vector_t qubit_line = line_pattern[i];
121  for (const Node& node : architecture_lines[i]) {
122  auto it = qubit_line.begin();
123  out_map.insert({*it, node});
124  qubit_line.erase(it);
125  }
126  }
127  return out_map;
128 }
