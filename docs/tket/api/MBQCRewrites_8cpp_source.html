22bool Rewrite::extend_for_PX_outputs_fun(ZXDiagram& diag) {
23 bool modified = false;
24 for (const ZXVert& o : diag.get_boundary(ZXType::Output)) {
25 ZXVert n = diag.neighbours(o).front();
26 if (diag.get_zxtype(n) != ZXType::Input &&
27 !(diag.get_zxtype(n) == ZXType::PX &&
28 !diag.get_vertex_ZXGen<CliffordGen>(n).get_param())) {
29 // Extend output
30 ZXGen_ptr px = std::make_shared<CliffordGen>(ZXType::PX, false);
31 ZXVert n1 = diag.add_vertex(px);
32 ZXVert n2 = diag.add_vertex(px);
33 diag.remove_wire(diag.adj_wires(o).front());
34 diag.add_wire(n, n1, ZXWireType::H);
35 diag.add_wire(n1, n2, ZXWireType::H);
36 diag.add_wire(n2, o);
37 modified = true;
38 }
39 }
40 return modified;
41}

47bool Rewrite::internalise_gadgets_fun(ZXDiagram& diag) {
48 std::set<ZXVert> out_vs;
49 for (const ZXVert& o : diag.get_boundary(ZXType::Output))
50 out_vs.insert(diag.other_end(diag.adj_wires(o).front(), o));
51 std::list<ZXVert> to_remove;
52 BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
53 if (diag.degree(v) == 1 && is_MBQC_type(diag.get_zxtype(v))) {
54 ZXVert axis = diag.neighbours(v).front();
55 if (out_vs.find(axis) != out_vs.end()) continue;
56 std::optional<unsigned> axis_clifford = std::nullopt;
57 Expr axis_XY_angle;
58 switch (diag.get_zxtype(axis)) {
59 case ZXType::XY: {
60 axis_XY_angle = diag.get_vertex_ZXGen<PhasedGen>(axis).get_param();
61 axis_clifford = equiv_Clifford(axis_XY_angle);
62 break;
63 }
64 case ZXType::PX: {
65 bool axis_param =
66 diag.get_vertex_ZXGen<CliffordGen>(axis).get_param();
67 axis_clifford = axis_param ? 2 : 0;
68 axis_XY_angle = axis_param ? 1. : 0.;
69 break;
70 }
71 case ZXType::PY: {
72 bool axis_param =
73 diag.get_vertex_ZXGen<CliffordGen>(axis).get_param();
74 axis_clifford = axis_param ? 3 : 1;
75 axis_XY_angle = axis_param ? 1.5 : 0.5;
76 break;
77 }
78 default:
79 continue;
80 }
81 switch (diag.get_zxtype(v)) {
82 case ZXType::XY: {
83 Expr current_param = diag.get_vertex_ZXGen<PhasedGen>(v).get_param();
84 if (!axis_clifford)
85 continue;
86 else if (*axis_clifford % 2 == 0) {
87 Expr new_param =
88 (*axis_clifford == 0) ? -current_param : current_param;
89 diag.set_vertex_ZXGen_ptr(
90 axis, std::make_shared<PhasedGen>(ZXType::YZ, new_param));
91 } else {
92 Expr new_param =
93 (*axis_clifford == 1) ? current_param : -current_param;
94 diag.set_vertex_ZXGen_ptr(
95 axis, std::make_shared<PhasedGen>(ZXType::XZ, new_param));
96 }
97 to_remove.push_back(v);
98 break;
99 }
100 case ZXType::YZ: {
101 Expr current_param = diag.get_vertex_ZXGen<PhasedGen>(v).get_param();
102 Expr new_param = axis_XY_angle - current_param;
103 diag.set_vertex_ZXGen_ptr(
104 axis, std::make_shared<PhasedGen>(ZXType::XY, new_param));
105 to_remove.push_back(v);
106 break;
107 }
108 case ZXType::XZ: {
109 Expr current_param = diag.get_vertex_ZXGen<PhasedGen>(v).get_param();
110 if (!axis_clifford)
111 continue;
112 else if (*axis_clifford % 2 == 0) {
113 Expr new_param = (*axis_clifford == 0) ? 0.5 - current_param
114 : current_param - 0.5;
115 diag.set_vertex_ZXGen_ptr(
116 axis, std::make_shared<PhasedGen>(ZXType::XZ, new_param));
117 } else {
118 Expr new_param = (*axis_clifford == 1) ? 0.5 - current_param
119 : current_param - 0.5;
120 diag.set_vertex_ZXGen_ptr(
121 axis, std::make_shared<PhasedGen>(ZXType::YZ, new_param));
122 }
123 to_remove.push_back(v);
124 break;
125 }
126 case ZXType::PX: {
127 bool current_param =
128 diag.get_vertex_ZXGen<CliffordGen>(v).get_param();
129 diag.set_vertex_ZXGen_ptr(
130 axis, std::make_shared<CliffordGen>(ZXType::PZ, current_param));
131 to_remove.push_back(v);
132 break;
133 }
134 case ZXType::PY: {
135 bool current_param =
136 diag.get_vertex_ZXGen<CliffordGen>(v).get_param();
137 if (!axis_clifford) {
138 Expr new_param =
139 current_param ? axis_XY_angle + 0.5 : axis_XY_angle - 0.5;
140 diag.set_vertex_ZXGen_ptr(
141 axis, std::make_shared<PhasedGen>(ZXType::XY, new_param));
142 } else if (*axis_clifford % 2 == 0) {
143 diag.set_vertex_ZXGen_ptr(
144 axis, std::make_shared<CliffordGen>(
145 ZXType::PY, current_param ^ (*axis_clifford == 0)));
146 } else {
147 diag.set_vertex_ZXGen_ptr(
148 axis, std::make_shared<CliffordGen>(
149 ZXType::PX, !current_param ^ (*axis_clifford == 1)));
150 }
151 to_remove.push_back(v);
152 break;
153 }
154 case ZXType::PZ: {
155 bool current_param =
156 diag.get_vertex_ZXGen<CliffordGen>(v).get_param();
157 if (!axis_clifford) {
158 Expr new_param = current_param ? axis_XY_angle + 1. : axis_XY_angle;
159 diag.set_vertex_ZXGen_ptr(
160 axis, std::make_shared<PhasedGen>(ZXType::XY, new_param));
161 } else if (*axis_clifford % 2 == 0) {
162 diag.set_vertex_ZXGen_ptr(
163 axis, std::make_shared<CliffordGen>(
164 ZXType::PX, !current_param ^ (*axis_clifford == 0)));
165 } else {
166 diag.set_vertex_ZXGen_ptr(
167 axis, std::make_shared<CliffordGen>(
168 ZXType::PY, !current_param ^ (*axis_clifford == 1)));
169 }
170 to_remove.push_back(v);
171 break;
172 }
173 default:
174 break;
175 }
176 }
177 }
178 for (const ZXVert& v : to_remove) diag.remove_vertex(v);
179 return !to_remove.empty();
180}
