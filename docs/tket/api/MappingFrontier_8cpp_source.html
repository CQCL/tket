22 // check if the maps are valid such that two maps should have the same set of
23 // keys (logical qubits), and the values (physical qubits) of two maps should be
24 // the same up to a permutation. All qubits in the circuit should also appear in
25 // the maps as physical qubits.
26 void MappingFrontier::assert_valid_maps() {
27  TKET_ASSERT(bimaps_->initial.size() == bimaps_->final.size());
28  auto left_it_1 = bimaps_->initial.left.begin();
29  auto left_it_2 = bimaps_->final.left.begin();
30  while (left_it_1 != bimaps_->initial.left.end()) {
31  TKET_ASSERT(left_it_1->first == left_it_2->first);
32  left_it_1++;
33  left_it_2++;
34  }
35  auto right_it_1 = bimaps_->initial.right.begin();
36  auto right_it_2 = bimaps_->final.right.begin();
37  while (right_it_1 != bimaps_->initial.right.end()) {
38  TKET_ASSERT(right_it_1->first == right_it_2->first);
39  right_it_1++;
40  right_it_2++;
41  }
42  for (const Qubit& q : circuit_.all_qubits()) {
43  TKET_ASSERT(bimaps_->initial.right.find(q) != bimaps_->initial.right.end());
44  }
45 }

63 static Bit get_bit_from_bool_frontier(
64  const std::shared_ptr<b_frontier_t>& b_frontier, const EdgeVec& ev) {
65  TKET_ASSERT(ev.size() > 0);
66  // condition is that if one Edge in EdgeVector ev is in a
67  // held bundle, then all are so return true
68  for (auto it = b_frontier->get<TagKey>().begin();
69  it != b_frontier->get<TagKey>().end(); ++it) {
70  for (const Edge& e0 : ev) {
71  for (const Edge& e1 : it->second) {
72  if (e0 == e1) {
73  return it->first;
74  }
75  }
76  }
77  }
84  TKET_ASSERT(false);

87 std::shared_ptr<unit_frontier_t> frontier_convert_vertport_to_edge(
88  const Circuit& circuit,
89  const std::shared_ptr<unit_vertport_frontier_t>& u_frontier) {
90  // make empty unit_frontier_t object
91  std::shared_ptr<unit_frontier_t> output_frontier =
92  std::make_shared<unit_frontier_t>();
93  // iterate through u_frontier, convert VertPort to Edge and insert
94  for (const std::pair<UnitID, VertPort>& pair : u_frontier->get<TagKey>()) {
95  output_frontier->insert(
96  {pair.first,
97  circuit.get_nth_out_edge(pair.second.first, pair.second.second)});
98  }
99  return output_frontier;
100 }

106 MappingFrontier::MappingFrontier(Circuit& _circuit) : circuit_(_circuit) {
107  this->linear_boundary = std::make_shared<unit_vertport_frontier_t>();
108  this->boolean_boundary = std::make_shared<b_frontier_t>();
109  this->bimaps_ = std::make_shared<unit_bimaps_t>();
110 
111  // Set up {UnitID, VertPort} objects for quantum and classical boundaries
112  for (const Qubit& qb : this->circuit_.all_qubits()) {
113  this->linear_boundary->insert({qb, {this->circuit_.get_in(qb), 0}});
114  this->bimaps_->initial.insert({qb, qb});
115  this->bimaps_->final.insert({qb, qb});
116  }
117  for (const Bit& bit : this->circuit_.all_bits()) {
118  Vertex bit_input = this->circuit_.get_in(bit);
119  EdgeVec bool_bundle = this->circuit_.get_nth_b_out_bundle(bit_input, 0);
120  // N.B. An Input Vertex may have boolean and classical out edges.
121  if (!bool_bundle.empty()) {
122  this->boolean_boundary->insert({bit, bool_bundle});
123  }
124  if (this->circuit_.n_out_edges_of_type(bit_input, EdgeType::Classical) >
125  0) {
126  this->linear_boundary->insert({bit, {bit_input, 0}});
127  }
128  }
129 }

192 void MappingFrontier::advance_next_2qb_slice(unsigned max_advance) {
193  bool boundary_updated = false;
194  unsigned loop = 0;
195  std::shared_ptr<unit_frontier_t> current_frontier =
196  frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary);
197 
198  // Get all vertices in first cut
199  VertexVec immediate_cut_vertices_v =
200  *(this->circuit_
201  .next_cut(current_frontier, std::make_shared<b_frontier_t>())
202  .slice);
203  do {
204  // each do section first finds the next set of edges after the held set
205  // for edges with target vertices with all their edges presented in the
206  // first set
207  loop++;
208  boundary_updated = false;
209  // produce next frontier object
210  std::shared_ptr<unit_frontier_t> next_frontier =
211  std::make_shared<unit_frontier_t>();
212 
213  for (const std::pair<UnitID, Edge>& pair :
214  current_frontier->get<TagKey>()) {
215  // if target_v not in immediate_cut_vertices, then do not pass it
216  Vertex target_v = this->circuit_.target(pair.second);
217  EdgeVec in_edges =
218  this->circuit_.get_in_edges_of_type(target_v, EdgeType::Quantum);
219  bool in_slice =
220  std::find(
221  immediate_cut_vertices_v.begin(), immediate_cut_vertices_v.end(),
222  target_v) != immediate_cut_vertices_v.end();
223  OpType ot = this->circuit_.get_OpType_from_Vertex(target_v);
224  if (((!in_slice && in_edges.size() > 1) || ot == OpType::Output ||
225  ot == OpType::ClOutput) &&
226  this->circuit_.get_OpType_from_Vertex(target_v) != OpType::Barrier) {
227  // Vertex either not allowed to pass, or is output vertex => update
228  // nothing
229  next_frontier->insert({pair.first, pair.second});
230  } else {
231  // vertex can be surpassed, so update linear_boundary and
232  // next_frontier with next edge
233  Edge next_edge = this->circuit_.get_next_edge(target_v, pair.second);
234  this->linear_boundary->replace(
235  this->linear_boundary->get<TagKey>().find(pair.first),
236  {pair.first,
237  {target_v, this->circuit_.get_source_port(next_edge)}});
238  next_frontier->insert({pair.first, next_edge});
239  }
240  }
241 
242  // Given new frontier, find the actual next cut
243  CutFrontier next_cut = this->circuit_.next_cut(
244  next_frontier, std::make_shared<b_frontier_t>());
245  // For each vertex in a slice, if its physically permitted, update
246  // linear_boundary with quantum out edges from vertex (i.e.
247  // next_cut.u_frontier)
248  for (const Vertex& vert : *next_cut.slice) {
249  // Output means we don't want to pass, so just leave
250  OpType ot = this->circuit_.get_OpType_from_Vertex(vert);
251  if (ot == OpType::Output || ot == OpType::ClOutput) {
252  continue;
253  }
254  EdgeVec in_edges =
255  this->circuit_.get_in_edges_of_type(vert, EdgeType::Quantum);
256  // More than 1 edge means we want to keep edges, so continue
257  if (in_edges.size() > 1) {
258  continue;
259  }
260  // can guarantee that we update now as non-updating cases have been
261  // continued
262  boundary_updated = true;
263  // push edge past single qubit vertex, repeat
264  UnitID uid = get_unitid_from_unit_frontier(
265  this->linear_boundary, {this->circuit_.source(in_edges[0]),
266  this->circuit_.get_source_port(in_edges[0])});
267 
268  Edge replacement_edge =
269  next_cut.u_frontier->get<TagKey>().find(uid)->second;
270 
271  Vertex source_vertex = this->circuit_.source(replacement_edge);
272  port_t source_port = this->circuit_.get_source_port(replacement_edge);
273 
274  this->linear_boundary->replace(
275  this->linear_boundary->get<TagKey>().find(uid),
276  {uid, {source_vertex, source_port}});
277  }
278  current_frontier = next_frontier;
279  } while (boundary_updated && loop <= max_advance);
280  return;
281 }

289 void MappingFrontier::advance_frontier_boundary(
290  const ArchitecturePtr& architecture) {
291  bool boundary_updated = false;
292  do {
298  boundary_updated = false;
299  std::shared_ptr<unit_frontier_t> l_frontier_edges =
300  frontier_convert_vertport_to_edge(
301  this->circuit_, this->linear_boundary);
302  CutFrontier next_cut =
303  this->circuit_.next_cut(l_frontier_edges, this->boolean_boundary);
312  for (const Vertex& vert : *next_cut.slice) {
317  std::vector<std::pair<UnitID, EdgeType>> in_uids;
318  std::vector<Node> nodes;
319  std::vector<Edge> all_in_edges = this->circuit_.get_in_edges(vert);
320  for (const Edge& edge : all_in_edges) {
321  UnitID uid;
322  EdgeType edge_type = this->circuit_.get_edgetype(edge);
323  switch (edge_type) {
324  case EdgeType::Quantum: {
325  uid = get_unitid_from_unit_frontier(
326  this->linear_boundary, {this->circuit_.source(edge),
327  this->circuit_.get_source_port(edge)});
328  // We use "nodes" to check if Quantum arguments respect the
329  // architecture
330  nodes.push_back(Node(uid));
331  break;
332  }
333  case EdgeType::Classical: {
334  uid = get_unitid_from_unit_frontier(
335  this->linear_boundary, {this->circuit_.source(edge),
336  this->circuit_.get_source_port(edge)});
337  break;
338  }
339  case EdgeType::Boolean: {
340  // N.B. function will find uid even if the bundle has more than the
341  // single "edge" passed
342  uid = get_bit_from_bool_frontier(this->boolean_boundary, {edge});
343  break;
344  }
345  default: {
346  TKET_ASSERT(false);
347  }
348  }
349  in_uids.push_back({uid, edge_type});
350  }
356  if (nodes.empty() ||
357  this->valid_boundary_operation(
358  architecture, this->circuit_.get_Op_ptr_from_Vertex(vert),
359  nodes)) {
360  boundary_updated = true;
361 
368  for (const std::pair<UnitID, EdgeType>& uid : in_uids) {
369  switch (uid.second) {
370  case EdgeType::Boolean:
371  break;
372  case EdgeType::Quantum:
373  case EdgeType::Classical: {
374  Edge replacement_edge =
375  next_cut.u_frontier->get<TagKey>().find(uid.first)->second;
376  this->linear_boundary->replace(
377  this->linear_boundary->get<TagKey>().find(uid.first),
378  {uid.first,
379  {this->circuit_.source(replacement_edge),
380  this->circuit_.get_source_port(replacement_edge)}});
381  break;
382  }
383  default: {
384  TKET_ASSERT(false);
385  }
386  }
387  }
388 
397  std::vector<EdgeVec> in_bundles = this->circuit_.get_b_in_bundles(vert);
398  std::vector<EdgeVec> out_bundles =
399  this->circuit_.get_b_out_bundles(vert);
400 
401  unsigned n_in_bundles = in_bundles.size();
402  unsigned n_out_bundles = out_bundles.size();
403 
404  TKET_ASSERT(n_out_bundles == n_in_bundles);
405  TKET_ASSERT(n_in_bundles == in_uids.size());
406 
428  for (port_t port = 0; port < n_in_bundles; port++) {
429  std::pair<UnitID, EdgeType> linear_uid = in_uids[port];
430  EdgeVec in_bundle = in_bundles[port];
431  EdgeVec out_bundle = out_bundles[port];
432  switch (linear_uid.second) {
433  // Linear & can't spawn Boolean bundle: pass
434  case EdgeType::Quantum:
435  break;
436  // Linear but can spawn Boolean bundle
437  // If Boolean edges spawned, add to boolean boundary
438  case EdgeType::Classical: {
439  EdgeVec out_bundle = out_bundles[port];
440  TKET_ASSERT(in_bundle.empty());
441  if (!out_bundle.empty()) {
442  this->boolean_boundary->insert(
443  {Bit(linear_uid.first), out_bundle});
444  }
445  break;
446  }
447  // Can spawn, erase or edit boolean bundles
448  case EdgeType::Boolean: {
449  TKET_ASSERT(!in_bundle.empty());
450  Bit bit = Bit(linear_uid.first);
451  auto boolean_it = this->boolean_boundary->get<TagKey>().find(bit);
452  TKET_ASSERT(
453  boolean_it != this->boolean_boundary->get<TagKey>().end());
454  // update "out bundle" with other boolean edges attached to other
455  // vertices
456  for (const Edge& edge : boolean_it->second) {
457  if (std::find(in_bundle.begin(), in_bundle.end(), edge) ==
458  in_bundle.end()) {
459  out_bundle.push_back(edge);
460  }
461  }
462  if (out_bundle.empty()) {
463  // => all edges in boolean bundle in boolean boundary attached
464  // to this vertex
465  // => vertex has no edges in output boolean bundle for thos port
466  // => can erase Bit from boolean boundary as its not longer used
467  this->boolean_boundary->erase(boolean_it);
468  } else {
469  // => either Vertex has edges in output boolean bundle
470  // => or there are other edges in boolean bundle held in boolean
471  // boundary that are not attached to this vertex
472  // => update boolean boundary
473  this->boolean_boundary->replace(boolean_it, {bit, out_bundle});
474  }
475  break;
476  }
477  default: {
478  TKET_ASSERT(false);
479  }
480  }
481  }
482  }
483  }
484  } while (boundary_updated);
490  return;
491 }

501 Subcircuit MappingFrontier::get_frontier_subcircuit(
502  unsigned _max_subcircuit_depth, unsigned _max_subcircuit_size) const {
503  CutFrontier current_cut = this->circuit_.next_cut(
504  frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary),
505  this->boolean_boundary);
506 
507  unsigned subcircuit_depth = 1;
508  VertexSet subcircuit_vertices(
509  current_cut.slice->begin(), current_cut.slice->end());
510  // add cuts of vertices to subcircuit_vertices until constraints met, or end
511  // of circuit reached
512  while (subcircuit_depth < _max_subcircuit_depth &&
513  unsigned(subcircuit_vertices.size()) < _max_subcircuit_size &&
514  current_cut.slice->size() > 0) {
515  current_cut =
516  this->circuit_.next_cut(current_cut.u_frontier, current_cut.b_frontier);
517  subcircuit_depth++;
518  subcircuit_vertices.insert(
519  current_cut.slice->begin(), current_cut.slice->end());
520  }
521  TKET_ASSERT(subcircuit_vertices.size() != 0);
522  return Subcircuit(
523  convert_u_frontier_to_edges(*frontier_convert_vertport_to_edge(
524  this->circuit_, this->linear_boundary)),
525  convert_u_frontier_to_edges(*current_cut.u_frontier),
526  subcircuit_vertices);
527 }

535 void MappingFrontier::update_bimaps(UnitID qubit, UnitID node) {
536  // Update initial map
537  auto init_it = this->bimaps_->initial.left.find(qubit);
538  TKET_ASSERT(init_it != this->bimaps_->initial.left.end());
539  this->bimaps_->initial.left.erase(init_it);
540  this->bimaps_->initial.left.insert({qubit, node});
541  // Update final map
542  auto final_it = this->bimaps_->final.left.find(qubit);
543  TKET_ASSERT(final_it != this->bimaps_->final.left.end());
544  this->bimaps_->final.left.erase(final_it);
545  this->bimaps_->final.left.insert({qubit, node});

548 void MappingFrontier::update_linear_boundary_uids(
549  const unit_map_t& relabelled_uids) {
550  for (const std::pair<const UnitID, UnitID>& label : relabelled_uids) {
551  // implies new labelling
552  if (label.first != label.second &&
553  this->reassignable_nodes_.find(Node(label.second)) ==
554  this->reassignable_nodes_.end()) {
555  // by type, label.first already assumed in circuit
556  // this condition means label.second also in circuit
557  // implies that a merging is done -> remove first qubit
558  if (this->linear_boundary->get<TagKey>().find(label.second) !=
559  this->linear_boundary->get<TagKey>().end()) {
560  // erase, assume updated already
561  this->linear_boundary->erase(label.first);
562  } else {
563  auto current_label_it =
564  this->linear_boundary->get<TagKey>().find(label.first);
565  // relabel "label.first" with "label.second"
566  this->linear_boundary->replace(
567  current_label_it, {label.second, current_label_it->second});
568  unit_map_t relabel = {label};
569  this->circuit_.rename_units(relabel);
570  }
571  }
572  }

575 void MappingFrontier::permute_subcircuit_q_out_hole(
576  const unit_map_t& final_permutation, Subcircuit& subcircuit) {
577  EdgeVec new_q_out_hole;
578  int i = 0;
579  if (this->linear_boundary->size() != final_permutation.size()) {
580  throw MappingFrontierError(
581  "Number of Qubits in mapping permutation does not match number of "
582  "Qubits in MappingFrontier boundary, for permuting Qubits as with "
583  "routed Subcircuit.");
584  }
585  for (const std::pair<UnitID, VertPort>& pair :
586  this->linear_boundary->get<TagKey>()) {
587  auto it = final_permutation.find(pair.first);
588  if (it == final_permutation.end()) {
589  throw MappingFrontierError("Qubit in boundary not in permutation.");
590  }
591  std::pair<UnitID, UnitID> uid_pair = *it;
592  if (uid_pair.first == uid_pair.second) {
593  new_q_out_hole.push_back(subcircuit.q_out_hole[i]);
594  } else {
595  int j = 0;
596  for (auto it = this->linear_boundary->get<TagKey>().begin();
597  it != this->linear_boundary->get<TagKey>().end(); ++it) {
598  if (it->first == uid_pair.second) {
599  new_q_out_hole.push_back(subcircuit.q_out_hole[j]);
600  break;
601  }
602  j++;
603  }
604  }
605  i++;
606  }
607  subcircuit.q_out_hole = new_q_out_hole;
608 }

639 bool MappingFrontier::add_swap(const UnitID& uid_0, const UnitID& uid_1) {
640  // get iterators to linear_boundary uids
641  auto uid0_in_it = this->linear_boundary->find(uid_0);
642  auto uid1_in_it = this->linear_boundary->find(uid_1);
643  // Add Qubit if not in MappingFrontier boundary (i.e. not in circuit)
644  if (uid0_in_it == this->linear_boundary->end()) {
645  this->add_ancilla(uid_0);
646  uid0_in_it = this->linear_boundary->find(uid_0);
647  }
648  if (uid1_in_it == this->linear_boundary->end()) {
649  this->add_ancilla(uid_1);
650  uid1_in_it = this->linear_boundary->find(uid_1);
651  }
652  this->reassignable_nodes_.erase(Node(uid_0));
653  this->reassignable_nodes_.erase(Node(uid_1));
654 
655  // Get predecessor edges to SWAP insert location
656  VertPort vp0 = uid0_in_it->second;
657  VertPort vp1 = uid1_in_it->second;
658  EdgeVec predecessors = {
659  this->circuit_.get_nth_out_edge(vp0.first, vp0.second),
660  this->circuit_.get_nth_out_edge(vp1.first, vp1.second)};
661 
669  Vertex source0 = this->circuit_.source(predecessors[0]);
670  Vertex source1 = this->circuit_.source(predecessors[1]);
671  if (source0 == source1) {
672  if (this->circuit_.get_OpType_from_Vertex(source0) == OpType::SWAP) {
673  return false;
674  }
675  }
676 
677  // update held ancillas
678  // the location/id of the "ancilla node" changes when a SWAP occurs
679  Node n0 = Node(uid_0);
680  Node n1 = Node(uid_1);
681  bool uid0_ancilla =
682  this->ancilla_nodes_.find(n0) != this->ancilla_nodes_.end();
683  bool uid1_ancilla =
684  this->ancilla_nodes_.find(n1) != this->ancilla_nodes_.end();
685 
686  // In this case we can still use the node, but we need to make sure that
687  // the wire it's on won't be reassigned
688  // This could reduce performance, but the "reassignable_nodes_" case
689  // deals with Qubit wires that have no multi-qubit quantum operations,
690  // so in practice this won't change the results for a meaningful circuit
691  if (this->reassignable_nodes_.find(n0) != this->reassignable_nodes_.end()) {
692  this->reassignable_nodes_.erase(n0);
693  }
694  if (this->reassignable_nodes_.find(n1) != this->reassignable_nodes_.end()) {
695  this->reassignable_nodes_.erase(n1);
696  }
697 
698  if (uid0_ancilla && !uid1_ancilla) {
699  this->ancilla_nodes_.erase(n0);
700  this->ancilla_nodes_.insert(n1);
701  }
702  if (!uid0_ancilla && uid1_ancilla) {
703  this->ancilla_nodes_.erase(n1);
704  this->ancilla_nodes_.insert(n0);
705  }
706 
707  // add SWAP vertex to circuit_ and rewire into predecessor
708  Vertex swap_v = this->circuit_.add_vertex(OpType::SWAP);
709  this->circuit_.rewire(
710  swap_v, predecessors, {EdgeType::Quantum, EdgeType::Quantum});
711 
712  // Update boundary to reflect new edges
713  EdgeVec successors = this->circuit_.get_all_out_edges(swap_v);
714  this->circuit_.dag[successors[0]].ports.first = 1;
715  this->circuit_.dag[successors[1]].ports.first = 0;
716 
717  this->linear_boundary->replace(
718  uid0_in_it, {uid_0, {this->circuit_.source(successors[1]), 0}});
719  this->linear_boundary->replace(
720  uid1_in_it, {uid_1, {this->circuit_.source(successors[0]), 1}});
721 
722  // update output vertices of quantum boundary of circuit to reflect changing
723  // qubit paths
724  auto uid0_circuit_boundary_it =
725  this->circuit_.boundary.get<TagID>().find(uid_0);
726  auto uid1_circuit_boundary_it =
727  this->circuit_.boundary.get<TagID>().find(uid_1);
728 
729  Vertex uid0_out = uid0_circuit_boundary_it->out_;
730  Vertex uid1_out = uid1_circuit_boundary_it->out_;
731  Vertex uid0_in = uid0_circuit_boundary_it->in_;
732  Vertex uid1_in = uid1_circuit_boundary_it->in_;
733 
734  this->circuit_.boundary.get<TagID>().erase(uid_0);
735  this->circuit_.boundary.get<TagID>().erase(uid_1);
736 
737  this->circuit_.boundary.get<TagID>().insert({uid_0, uid0_in, uid1_out});
738  this->circuit_.boundary.get<TagID>().insert({uid_1, uid1_in, uid0_out});
739 
740  std::map<Node, Node> final_map = {{n0, n1}, {n1, n0}};
741  update_maps(this->bimaps_, {}, final_map);
742  assert_valid_maps();
743  return true;

746 void MappingFrontier::add_bridge(
747  const UnitID& control, const UnitID& central, const UnitID& target) {
748  // get predecessors
749  auto control_in_it = this->linear_boundary->find(control);
750  auto central_in_it = this->linear_boundary->find(central);
751  auto target_in_it = this->linear_boundary->find(target);
752 
753  // by virtue of method, control and target qubit will always be in BRIDGE.
754  // However, distances used to check BRIDGE and find PATH may use
755  // central qubit that is unallocated, in which add it.
756  if (central_in_it == this->linear_boundary->end()) {
757  this->add_ancilla(central);
758  central_in_it = this->linear_boundary->find(central);
759  }
760 
761  Node central_node = Node(central);
762  // In this case we can still use the node, but we need to make sure that
763  // the wire it's on won't be reassigned
764  // This could reduce performance, but the "reassignable_nodes_" case
765  // deals with Qubit wires that have no multi-qubit quantum operations,
766  // so in practice this won't change the results for a meaningful circuit
767  if (this->reassignable_nodes_.find(central_node) !=
768  this->reassignable_nodes_.end()) {
769  this->reassignable_nodes_.erase(central_node);
770  }
771  VertPort vp_control = control_in_it->second;
772  VertPort vp_central = central_in_it->second;
773  VertPort vp_target = target_in_it->second;
774 
775  EdgeVec predecessors = {
776  this->circuit_.get_nth_out_edge(vp_control.first, vp_control.second),
777  this->circuit_.get_nth_out_edge(vp_central.first, vp_central.second),
778  this->circuit_.get_nth_out_edge(vp_target.first, vp_target.second),
779  }; // get cx vertex
780  // this should be guaranteeds by pre-checks
781  Vertex cx_v = this->circuit_.target(predecessors[0]);
782  // add bridge
783  Vertex bridge_v = this->circuit_.add_vertex(OpType::BRIDGE);
784  // add bridge vertex to circuit
785  this->circuit_.rewire(
786  bridge_v, predecessors,
787  {EdgeType::Quantum, EdgeType::Quantum, EdgeType::Quantum});
788  // remove old cx vertex
789  this->circuit_.remove_vertex(
790  cx_v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
791  assert_valid_maps();
792 }

794 void MappingFrontier::add_ancilla(const UnitID& ancilla) {
795  // Given the ancilla node, we 1. add ancilla to the circuit.
796  // 2. map ancilla to placeholder_q, which will be a qubit from the routing
797  // ancilla register that is not present anywhere in the circuit and bimaps.
798  // Add {placeholder_q, ancilla} to the bimaps.
799  register_t ancilla_reg = this->circuit_.get_reg(q_routing_ancilla_reg());
800  unsigned available_ancilla_idx = 0;
801  auto it = ancilla_reg.rbegin();
802  if (it != ancilla_reg.rend()) {
803  available_ancilla_idx = it->first + 1;
804  }
805  for (auto const& it : this->bimaps_->initial.left) {
806  if (it.first.reg_name() == q_routing_ancilla_reg() &&
807  it.first.index()[0] >= available_ancilla_idx) {
808  available_ancilla_idx = it.first.index()[0] + 1;
809  }
810  if (it.second.reg_name() == q_routing_ancilla_reg() &&
811  it.second.index()[0] >= available_ancilla_idx) {
812  available_ancilla_idx = it.second.index()[0] + 1;
813  }
814  }
815  Qubit placeholder_q(q_routing_ancilla_reg(), available_ancilla_idx);
816  Qubit ancilla_q(ancilla);
817  this->circuit_.add_qubit(ancilla_q);
818  this->linear_boundary->insert(
819  {ancilla_q, {this->circuit_.get_in(ancilla_q), 0}});
820 
821  this->bimaps_->initial.insert({placeholder_q, ancilla});
822  this->bimaps_->final.insert({placeholder_q, ancilla});
823  this->ancilla_nodes_.insert(Node(ancilla));
824 }

826 void MappingFrontier::merge_ancilla(
827  const UnitID& merge, const UnitID& ancilla) {
828  // "front" meaning causally ahead
829  // ancilla front, merge back
830 
831  auto rewire = [&](const UnitID& front, const UnitID& back) {
832  // get output and input vertices
833  Vertex back_v_in = this->circuit_.get_in(back);
834  Vertex back_v_out = this->circuit_.get_out(back);
835  Vertex front_v_out = this->circuit_.get_out(front);
836 
837  // find source vertex & port of merge_v_out
838  // output vertex, so can assume single edge
839  Edge back_out_edge = this->circuit_.get_nth_out_edge(back_v_in, 0);
840  Edge front_in_edge = this->circuit_.get_nth_in_edge(front_v_out, 0);
841  // Find port number
842  port_t back_target_port = this->circuit_.get_target_port(back_out_edge);
843  port_t front_source_port = this->circuit_.get_source_port(front_in_edge);
844  // Find vertices
845  Vertex back_v_target = this->circuit_.target(back_out_edge);
846  Vertex front_v_source = this->circuit_.source(front_in_edge);
847 
853  // remove and replace edges
854  this->circuit_.remove_edge(back_out_edge);
855  this->circuit_.remove_edge(front_in_edge);
861  this->circuit_.add_edge(
862  {front_v_source, front_source_port}, {back_v_target, back_target_port},
863  EdgeType::Quantum);
871  // instead of manually updating all boundaries, we change which output
872  // vertex the qubit paths to
873  Edge back_in_edge = this->circuit_.get_nth_in_edge(back_v_out, 0);
874  port_t back_source_port = this->circuit_.get_source_port(back_in_edge);
875  Vertex back_v_source = this->circuit_.source(back_in_edge);
876 
877  this->circuit_.remove_edge(back_in_edge);
878 
885  this->circuit_.add_edge(
886  {back_v_source, back_source_port}, {front_v_out, 0}, EdgeType::Quantum);
894  // remove empty vertex wire, relabel dag vertices
895  this->circuit_.dag[back_v_in].op = get_op_ptr(OpType::noop);
896  this->circuit_.dag[back_v_out].op = get_op_ptr(OpType::noop);
897 
898  TKET_ASSERT(this->circuit_.n_in_edges(back_v_out) == 0);
899  TKET_ASSERT(this->circuit_.n_out_edges(back_v_in) == 0);
900  TKET_ASSERT(this->circuit_.n_in_edges(front_v_out) == 1);
901  this->circuit_.remove_vertex(
902  back_v_in, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
903  this->circuit_.remove_vertex(
904  back_v_out, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
909  this->circuit_.boundary.get<TagID>().erase(back);
910  };
911 
918  auto merge_boundary_it = this->linear_boundary->get<TagKey>().find(merge);
919  auto ancilla_boundary_it = this->linear_boundary->get<TagKey>().find(ancilla);
920  TKET_ASSERT(
921  ancilla_boundary_it != this->linear_boundary->get<TagKey>().end());
922  TKET_ASSERT(merge_boundary_it != this->linear_boundary->get<TagKey>().end());
923 
924  // If a valid ancilla, this should always be true
925  VertPort ancilla_vp = ancilla_boundary_it->second;
926  TKET_ASSERT(
927  this->circuit_
928  .dag[this->circuit_.target(this->circuit_.get_nth_out_edge(
929  ancilla_vp.first, ancilla_vp.second))]
930  .op->get_type() == OpType::Output);
931 
932  VertPort merge_vp = merge_boundary_it->second;
933  Edge merge_edge =
934  this->circuit_.get_nth_out_edge(merge_vp.first, merge_vp.second);
935  Vertex merge_target = this->circuit_.target(merge_edge);
936  port_t merge_target_port = this->circuit_.get_target_port(merge_edge);
937 
938  this->linear_boundary->erase(merge_boundary_it);
939  OpType merge_target_optype = this->circuit_.dag[merge_target].op->get_type();
943  rewire(ancilla, merge);
944 
953  if (merge_target_optype != OpType::Output) {
954  Edge merge_source_edge =
955  this->circuit_.get_nth_in_edge(merge_target, merge_target_port);
956  Vertex merge_source = this->circuit_.source(merge_source_edge);
957  port_t merge_source_port =
958  this->circuit_.get_source_port(merge_source_edge);
959  this->linear_boundary->replace(
960  ancilla_boundary_it, {ancilla, {merge_source, merge_source_port}});
961  } else {
962  Vertex ancilla_out_vertex = this->circuit_.get_out(ancilla);
963  Edge ancilla_out_in_edge =
964  this->circuit_.get_nth_in_edge(ancilla_out_vertex, 0);
965  Vertex ancilla_out_source = this->circuit_.source(ancilla_out_in_edge);
966  port_t ancilla_out_source_port =
967  this->circuit_.get_source_port(ancilla_out_in_edge);
968  this->linear_boundary->replace(
969  ancilla_boundary_it,
970  {ancilla, {ancilla_out_source, ancilla_out_source_port}});
971  }
972 
973  // Update the qubit mappings
974  // let's call the arguments ancilla_node and merge_node
975  // e.g. before merge:
976  // initial := {ancilla_q:node_x, merge_q:some_uid}
977  // final := {ancilla_q:ancilla_node, merge_q:merge_node}
978  // e.g. after merge:
979  // initial := {merge_q:node_x}
980  // final := {merge_q:ancilla_node}
981  // Basically, in both qubit maps, erase the entry with qubit merge_q
982  // then replace the entry ancilla_q -> x with the merge_q -> x
983 
984  auto merge_it = this->bimaps_->initial.right.find(merge);
985  TKET_ASSERT(merge_it != this->bimaps_->initial.right.end());
986  UnitID merge_q = merge_it->second;
987  this->bimaps_->initial.right.erase(merge_it);
988  this->bimaps_->final.left.erase(merge_q);
989  // Find ancilla_q
990  auto final_it = this->bimaps_->final.right.find(ancilla);
991  TKET_ASSERT(final_it != this->bimaps_->final.right.end());
992  UnitID ancilla_q = final_it->second;
993  // Replace in final map
994  this->bimaps_->final.right.erase(final_it);
995  this->bimaps_->final.left.insert({merge_q, ancilla});
996  // Replace in initial map
997  auto init_it = this->bimaps_->initial.left.find(ancilla_q);
998  UnitID init_ancilla_node = init_it->second;
999  this->bimaps_->initial.left.erase(init_it);
1000  this->bimaps_->initial.left.insert({merge_q, init_ancilla_node});
1001 
1005  this->ancilla_nodes_.erase(Node(ancilla));
1006  this->reassignable_nodes_.erase(Node(ancilla));
1007  assert_valid_maps();
1008  return;
1009 }

1011 bool MappingFrontier::valid_boundary_operation(
1012  const ArchitecturePtr& architecture, const Op_ptr& op,
1013  const std::vector<Node>& uids) const {
1014  OpType ot = op->get_type();
1015 
1016  if (ot == OpType::Conditional) {
1017  Op_ptr cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
1018  // conditional boxes are never allowed, too
1019  ot = cond_op_ptr->get_type();
1020  while (ot == OpType::Conditional) {
1021  cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
1022  ot = cond_op_ptr->get_type();
1023  }
1024  }
1025  // Barriers are always allowed
1026  if (is_barrier_type(ot)) {
1027  return true;
1028  }
1029  // boxes are never allowed
1030  if (is_box_type(ot)) {
1031  return false;
1032  }
1033 
1034  // this currently allows unplaced single qubits gates
1035  // this should be changes in the future
1036  if (uids.size() == 1) {
1037  return true;
1038  }
1039 
1040  // allow two qubit gates only for placed and connected nodes
1041  if (uids.size() == 2) {
1042  bool n0 = architecture->node_exists(uids[0]);
1043  bool n1 = architecture->node_exists(uids[1]);
1044  if (n0 && n1) {
1045  bool bde = architecture->bidirectional_edge_exists(uids[0], uids[1]);
1046  if (bde) {
1047  return true;
1048  }
1049  }
1050  } else if (uids.size() == 3 && ot == OpType::BRIDGE) {
1051  bool con_0_exists =
1052  architecture->bidirectional_edge_exists(uids[0], uids[1]);
1053  bool con_1_exists =
1054  architecture->bidirectional_edge_exists(uids[2], uids[1]);
1055  if (architecture->node_exists(uids[0]) &&
1056  architecture->node_exists(uids[1]) &&
1057  architecture->node_exists(uids[2]) && con_0_exists && con_1_exists) {
1058  return true;
1059  }
1060  }
1061 
1062  return false;
1063 }
