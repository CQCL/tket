22// check if the maps are valid such that two maps should have the same set of
23// keys (logical qubits), and the values (physical qubits) of two maps should be
24// the same up to a permutation. All qubits in the circuit should also appear in
25// the maps as physical qubits.
26void MappingFrontier::assert_valid_maps() {
27 TKET_ASSERT(bimaps_->initial.size() == bimaps_->final.size());
28 auto left_it_1 = bimaps_->initial.left.begin();
29 auto left_it_2 = bimaps_->final.left.begin();
30 while (left_it_1 != bimaps_->initial.left.end()) {
31 TKET_ASSERT(left_it_1->first == left_it_2->first);
32 left_it_1++;
33 left_it_2++;
34 }
35 auto right_it_1 = bimaps_->initial.right.begin();
36 auto right_it_2 = bimaps_->final.right.begin();
37 while (right_it_1 != bimaps_->initial.right.end()) {
38 TKET_ASSERT(right_it_1->first == right_it_2->first);
39 right_it_1++;
40 right_it_2++;
41 }
42 for (const Qubit& q : circuit_.all_qubits()) {
43 TKET_ASSERT(bimaps_->initial.right.find(q) != bimaps_->initial.right.end());
44 }
45}

63static Bit get_bit_from_bool_frontier(
64 const std::shared_ptr<b_frontier_t>& b_frontier, const EdgeVec& ev) {
65 TKET_ASSERT(ev.size() > 0);
66 // condition is that if one Edge in EdgeVector ev is in a
67 // held bundle, then all are so return true
68 for (auto it = b_frontier->get<TagKey>().begin();
69 it != b_frontier->get<TagKey>().end(); ++it) {
70 for (const Edge& e0 : ev) {
71 for (const Edge& e1 : it->second) {
72 if (e0 == e1) {
73 return it->first;
74 }
75 }
76 }
77 }
84 TKET_ASSERT(false);

87std::shared_ptr<unit_frontier_t> frontier_convert_vertport_to_edge(
88 const Circuit& circuit,
89 const std::shared_ptr<unit_vertport_frontier_t>& u_frontier) {
90 // make empty unit_frontier_t object
91 std::shared_ptr<unit_frontier_t> output_frontier =
92 std::make_shared<unit_frontier_t>();
93 // iterate through u_frontier, convert VertPort to Edge and insert
94 for (const std::pair<UnitID, VertPort>& pair : u_frontier->get<TagKey>()) {
95 output_frontier->insert(
96 {pair.first,
97 circuit.get_nth_out_edge(pair.second.first, pair.second.second)});
98 }
99 return output_frontier;
100}

106MappingFrontier::MappingFrontier(Circuit& _circuit) : circuit_(_circuit) {
107 this->linear_boundary = std::make_shared<unit_vertport_frontier_t>();
108 this->boolean_boundary = std::make_shared<b_frontier_t>();
109 this->bimaps_ = std::make_shared<unit_bimaps_t>();
110
111 // Set up {UnitID, VertPort} objects for quantum and classical boundaries
112 for (const Qubit& qb : this->circuit_.all_qubits()) {
113 this->linear_boundary->insert({qb, {this->circuit_.get_in(qb), 0}});
114 this->bimaps_->initial.insert({qb, qb});
115 this->bimaps_->final.insert({qb, qb});
116 }
117 for (const Bit& bit : this->circuit_.all_bits()) {
118 Vertex bit_input = this->circuit_.get_in(bit);
119 EdgeVec bool_bundle = this->circuit_.get_nth_b_out_bundle(bit_input, 0);
120 // N.B. An Input Vertex may have boolean and classical out edges.
121 if (!bool_bundle.empty()) {
122 this->boolean_boundary->insert({bit, bool_bundle});
123 }
124 if (this->circuit_.n_out_edges_of_type(bit_input, EdgeType::Classical) >
125 0) {
126 this->linear_boundary->insert({bit, {bit_input, 0}});
127 }
128 }
129}

135MappingFrontier::MappingFrontier(
136 Circuit& _circuit, std::shared_ptr<unit_bimaps_t> _bimaps)
137 : circuit_(_circuit), bimaps_(_bimaps) {
138 // Check that the maps are valid
139 for (const Qubit& q : _circuit.all_qubits()) {
140 if (_bimaps->initial.right.find(q) == _bimaps->initial.right.end()) {
141 throw MappingFrontierError(
142 "Uid " + q.repr() + " not found in initial map.");
143 }
144 if (_bimaps->final.right.find(q) == _bimaps->final.right.end()) {
145 throw MappingFrontierError(
146 "Uid " + q.repr() + " not found in final map.");
147 }
148 }
149
150 this->linear_boundary = std::make_shared<unit_vertport_frontier_t>();
151 this->boolean_boundary = std::make_shared<b_frontier_t>();
152 // Set up {UnitID, VertPort} objects for quantum and classical boundaries
153 for (const Qubit& qb : this->circuit_.all_qubits()) {
154 this->linear_boundary->insert({qb, {this->circuit_.get_in(qb), 0}});
155 }
156 for (const Bit& bit : this->circuit_.all_bits()) {
157 Vertex bit_input = this->circuit_.get_in(bit);
158 EdgeVec bool_bundle = this->circuit_.get_nth_b_out_bundle(bit_input, 0);
159 // N.B. An Input Vertex may have boolean and classical out edges.
160 if (!bool_bundle.empty()) {
161 this->boolean_boundary->insert({bit, bool_bundle});
162 }
163 if (this->circuit_.n_out_edges_of_type(bit_input, EdgeType::Classical) >
164 0) {
165 this->linear_boundary->insert({bit, {bit_input, 0}});
166 }
167 }
168}

171 : circuit_(mapping_frontier.circuit_), bimaps_(mapping_frontier.bimaps_) {
172 this->linear_boundary = std::make_shared<unit_vertport_frontier_t>();
173 this->boolean_boundary = std::make_shared<b_frontier_t>();
174
175 for (const std::pair<UnitID, VertPort>& pair :
176 mapping_frontier.linear_boundary->get<TagKey>()) {
177 this->linear_boundary->insert({pair.first, pair.second});
178 }
179 for (const std::pair<Bit, EdgeVec>& pair :
180 mapping_frontier.boolean_boundary->get<TagKey>()) {
181 EdgeVec edges;
182 for (const Edge& edge : pair.second) {
183 edges.push_back(edge);
184 }
185 this->boolean_boundary->insert({pair.first, edges});
186 }
187 for (const Node& node : mapping_frontier.ancilla_nodes_) {
188 this->ancilla_nodes_.insert(node);
189 }

192void MappingFrontier::advance_next_2qb_slice(unsigned max_advance) {
193 bool boundary_updated = false;
194 unsigned loop = 0;
195 std::shared_ptr<unit_frontier_t> current_frontier =
196 frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary);
197
198 // Get all vertices in first cut
199 VertexVec immediate_cut_vertices_v =
200 *(this->circuit_
201 .next_cut(current_frontier, std::make_shared<b_frontier_t>())
202 .slice);
203 do {
204 // each do section first finds the next set of edges after the held set
205 // for edges with target vertices with all their edges presented in the
206 // first set
207 loop++;
208 boundary_updated = false;
209 // produce next frontier object
210 std::shared_ptr<unit_frontier_t> next_frontier =
211 std::make_shared<unit_frontier_t>();
212
213 for (const std::pair<UnitID, Edge>& pair :
214 current_frontier->get<TagKey>()) {
215 // if target_v not in immediate_cut_vertices, then do not pass it
216 Vertex target_v = this->circuit_.target(pair.second);
217 EdgeVec in_edges =
218 this->circuit_.get_in_edges_of_type(target_v, EdgeType::Quantum);
219 bool in_slice =
220 std::find(
221 immediate_cut_vertices_v.begin(), immediate_cut_vertices_v.end(),
222 target_v) != immediate_cut_vertices_v.end();
223 OpType ot = this->circuit_.get_OpType_from_Vertex(target_v);
224 if (((!in_slice && in_edges.size() > 1) || ot == OpType::Output ||
225 ot == OpType::ClOutput) &&
226 this->circuit_.get_OpType_from_Vertex(target_v) != OpType::Barrier) {
227 // Vertex either not allowed to pass, or is output vertex => update
228 // nothing
229 next_frontier->insert({pair.first, pair.second});
230 } else {
231 // vertex can be surpassed, so update linear_boundary and
232 // next_frontier with next edge
233 Edge next_edge = this->circuit_.get_next_edge(target_v, pair.second);
234 this->linear_boundary->replace(
235 this->linear_boundary->get<TagKey>().find(pair.first),
236 {pair.first,
237 {target_v, this->circuit_.get_source_port(next_edge)}});
238 next_frontier->insert({pair.first, next_edge});
239 }
240 }
241
242 // Given new frontier, find the actual next cut
243 CutFrontier next_cut = this->circuit_.next_cut(
244 next_frontier, std::make_shared<b_frontier_t>());
245 // For each vertex in a slice, if its physically permitted, update
246 // linear_boundary with quantum out edges from vertex (i.e.
247 // next_cut.u_frontier)
248 for (const Vertex& vert : *next_cut.slice) {
249 // Output means we don't want to pass, so just leave
250 OpType ot = this->circuit_.get_OpType_from_Vertex(vert);
251 if (ot == OpType::Output || ot == OpType::ClOutput) {
252 continue;
253 }
254 EdgeVec in_edges =
255 this->circuit_.get_in_edges_of_type(vert, EdgeType::Quantum);
256 // More than 1 edge means we want to keep edges, so continue
257 if (in_edges.size() > 1) {
258 continue;
259 }
260 // can guarantee that we update now as non-updating cases have been
261 // continued
262 boundary_updated = true;
263 // push edge past single qubit vertex, repeat
264 UnitID uid = get_unitid_from_unit_frontier(
265 this->linear_boundary, {this->circuit_.source(in_edges[0]),
266 this->circuit_.get_source_port(in_edges[0])});
267
268 Edge replacement_edge =
269 next_cut.u_frontier->get<TagKey>().find(uid)->second;
270
271 Vertex source_vertex = this->circuit_.source(replacement_edge);
272 port_t source_port = this->circuit_.get_source_port(replacement_edge);
273
274 this->linear_boundary->replace(
275 this->linear_boundary->get<TagKey>().find(uid),
276 {uid, {source_vertex, source_port}});
277 }
278 current_frontier = next_frontier;
279 } while (boundary_updated && loop <= max_advance);
280 return;
281}

289void MappingFrontier::advance_frontier_boundary(
290 const ArchitecturePtr& architecture) {
291 bool boundary_updated = false;
292 do {
298 boundary_updated = false;
299 std::shared_ptr<unit_frontier_t> l_frontier_edges =
300 frontier_convert_vertport_to_edge(
301 this->circuit_, this->linear_boundary);
302 CutFrontier next_cut =
303 this->circuit_.next_cut(l_frontier_edges, this->boolean_boundary);
312 for (const Vertex& vert : *next_cut.slice) {
317 std::vector<std::pair<UnitID, EdgeType>> in_uids;
318 std::vector<Node> nodes;
319 std::vector<Edge> all_in_edges = this->circuit_.get_in_edges(vert);
320 for (const Edge& edge : all_in_edges) {
321 UnitID uid;
322 EdgeType edge_type = this->circuit_.get_edgetype(edge);
323 switch (edge_type) {
324 case EdgeType::Quantum: {
325 uid = get_unitid_from_unit_frontier(
326 this->linear_boundary, {this->circuit_.source(edge),
327 this->circuit_.get_source_port(edge)});
328 // We use "nodes" to check if Quantum arguments respect the
329 // architecture
330 nodes.push_back(Node(uid));
331 break;
332 }
333 case EdgeType::Classical: {
334 uid = get_unitid_from_unit_frontier(
335 this->linear_boundary, {this->circuit_.source(edge),
336 this->circuit_.get_source_port(edge)});
337 break;
338 }
339 case EdgeType::Boolean: {
340 // N.B. function will find uid even if the bundle has more than the
341 // single "edge" passed
342 uid = get_bit_from_bool_frontier(this->boolean_boundary, {edge});
343 break;
344 }
345 default: {
346 TKET_ASSERT(false);
347 }
348 }
349 in_uids.push_back({uid, edge_type});
350 }
356 if (nodes.empty() ||
357 this->valid_boundary_operation(
358 architecture, this->circuit_.get_Op_ptr_from_Vertex(vert),
359 nodes)) {
360 boundary_updated = true;
361
368 for (const std::pair<UnitID, EdgeType>& uid : in_uids) {
369 switch (uid.second) {
370 case EdgeType::Boolean:
371 break;
372 case EdgeType::Quantum:
373 case EdgeType::Classical: {
374 Edge replacement_edge =
375 next_cut.u_frontier->get<TagKey>().find(uid.first)->second;
376 this->linear_boundary->replace(
377 this->linear_boundary->get<TagKey>().find(uid.first),
378 {uid.first,
379 {this->circuit_.source(replacement_edge),
380 this->circuit_.get_source_port(replacement_edge)}});
381 break;
382 }
383 default: {
384 TKET_ASSERT(false);
385 }
386 }
387 }
388
397 std::vector<EdgeVec> in_bundles = this->circuit_.get_b_in_bundles(vert);
398 std::vector<EdgeVec> out_bundles =
399 this->circuit_.get_b_out_bundles(vert);
400
401 unsigned n_in_bundles = in_bundles.size();
402 unsigned n_out_bundles = out_bundles.size();
403
404 TKET_ASSERT(n_out_bundles == n_in_bundles);
405 TKET_ASSERT(n_in_bundles == in_uids.size());
406
428 for (port_t port = 0; port < n_in_bundles; port++) {
429 std::pair<UnitID, EdgeType> linear_uid = in_uids[port];
430 EdgeVec in_bundle = in_bundles[port];
431 EdgeVec out_bundle = out_bundles[port];
432 switch (linear_uid.second) {
433 // Linear & can't spawn Boolean bundle: pass
434 case EdgeType::Quantum:
435 break;
436 // Linear but can spawn Boolean bundle
437 // If Boolean edges spawned, add to boolean boundary
438 case EdgeType::Classical: {
439 EdgeVec out_bundle = out_bundles[port];
440 TKET_ASSERT(in_bundle.empty());
441 if (!out_bundle.empty()) {
442 this->boolean_boundary->insert(
443 {Bit(linear_uid.first), out_bundle});
444 }
445 break;
446 }
447 // Can spawn, erase or edit boolean bundles
448 case EdgeType::Boolean: {
449 TKET_ASSERT(!in_bundle.empty());
450 Bit bit = Bit(linear_uid.first);
451 auto boolean_it = this->boolean_boundary->get<TagKey>().find(bit);
452 TKET_ASSERT(
453 boolean_it != this->boolean_boundary->get<TagKey>().end());
454 // update "out bundle" with other boolean edges attached to other
455 // vertices
456 for (const Edge& edge : boolean_it->second) {
457 if (std::find(in_bundle.begin(), in_bundle.end(), edge) ==
458 in_bundle.end()) {
459 out_bundle.push_back(edge);
460 }
461 }
462 if (out_bundle.empty()) {
463 // => all edges in boolean bundle in boolean boundary attached
464 // to this vertex
465 // => vertex has no edges in output boolean bundle for thos port
466 // => can erase Bit from boolean boundary as its not longer used
467 this->boolean_boundary->erase(boolean_it);
468 } else {
469 // => either Vertex has edges in output boolean bundle
470 // => or there are other edges in boolean bundle held in boolean
471 // boundary that are not attached to this vertex
472 // => update boolean boundary
473 this->boolean_boundary->replace(boolean_it, {bit, out_bundle});
474 }
475 break;
476 }
477 default: {
478 TKET_ASSERT(false);
479 }
480 }
481 }
482 }
483 }
484 } while (boundary_updated);
490 return;
491}

509Subcircuit MappingFrontier::get_frontier_subcircuit(
510 unsigned _max_subcircuit_depth, unsigned _max_subcircuit_size) const {
511 CutFrontier current_cut = this->circuit_.next_cut(
512 frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary),
513 this->boolean_boundary);
514
515 unsigned subcircuit_depth = 1;
516 VertexSet subcircuit_vertices(
517 current_cut.slice->begin(), current_cut.slice->end());
518 // add cuts of vertices to subcircuit_vertices until constraints met, or end
519 // of circuit reached
520 while (subcircuit_depth < _max_subcircuit_depth &&
521 unsigned(subcircuit_vertices.size()) < _max_subcircuit_size &&
522 current_cut.slice->size() > 0) {
523 current_cut =
524 this->circuit_.next_cut(current_cut.u_frontier, current_cut.b_frontier);
525 subcircuit_depth++;
526 subcircuit_vertices.insert(
527 current_cut.slice->begin(), current_cut.slice->end());
528 }
529 TKET_ASSERT(subcircuit_vertices.size() != 0);
530 EdgeVec out_edges = convert_u_frontier_to_edges(*current_cut.u_frontier);
531 std::vector<std::optional<Edge>> opt_out_edges{
532 out_edges.begin(), out_edges.end()};
533 return Subcircuit(
534 convert_u_frontier_to_edges(*frontier_convert_vertport_to_edge(
535 this->circuit_, this->linear_boundary)),
536 opt_out_edges, convert_b_frontier_to_edges(*current_cut.b_frontier),
537 subcircuit_vertices);
538}

546void MappingFrontier::update_bimaps(UnitID qubit, UnitID node) {
547 // Update initial map
548 auto init_it = this->bimaps_->initial.left.find(qubit);
549 TKET_ASSERT(init_it != this->bimaps_->initial.left.end());
550 this->bimaps_->initial.left.erase(init_it);
551 this->bimaps_->initial.left.insert({qubit, node});
552 // Update final map
553 auto final_it = this->bimaps_->final.left.find(qubit);
554 TKET_ASSERT(final_it != this->bimaps_->final.left.end());
555 this->bimaps_->final.left.erase(final_it);
556 this->bimaps_->final.left.insert({qubit, node});
557}

559void MappingFrontier::update_linear_boundary_uids(
560 const unit_map_t& relabelled_uids) {
561 for (const std::pair<const UnitID, UnitID>& label : relabelled_uids) {
562 // implies new labelling
563 if (label.first != label.second &&
564 this->reassignable_nodes_.find(Node(label.second)) ==
565 this->reassignable_nodes_.end()) {
566 // by type, label.first already assumed in circuit
567 // this condition means label.second also in circuit
568 // implies that a merging is done -> remove first qubit
569 if (this->linear_boundary->get<TagKey>().find(label.second) !=
570 this->linear_boundary->get<TagKey>().end()) {
571 // erase, assume updated already
572 this->linear_boundary->erase(label.first);
573 } else {
574 auto current_label_it =
575 this->linear_boundary->get<TagKey>().find(label.first);
576 // relabel "label.first" with "label.second"
577 this->linear_boundary->replace(
578 current_label_it, {label.second, current_label_it->second});
579 unit_map_t relabel = {label};
580 this->circuit_.rename_units(relabel);
581 }
582 }
583 }

586void MappingFrontier::permute_subcircuit_q_out_hole(
587 const unit_map_t& final_permutation, Subcircuit& subcircuit) {
588 std::vector<std::optional<Edge>> new_out_hole;
589 int i = 0;
590 if (this->linear_boundary->size() != final_permutation.size()) {
591 throw MappingFrontierError(
592 "Number of Qubits in mapping permutation does not match number of "
593 "Qubits in MappingFrontier boundary, for permuting Qubits as with "
594 "routed Subcircuit.");
595 }
596 for (const std::pair<UnitID, VertPort>& pair :
597 this->linear_boundary->get<TagKey>()) {
598 auto it = final_permutation.find(pair.first);
599 if (it == final_permutation.end()) {
600 throw MappingFrontierError("Qubit in boundary not in permutation.");
601 }
602 std::pair<UnitID, UnitID> uid_pair = *it;
603 if (uid_pair.first == uid_pair.second) {
604 new_out_hole.push_back(subcircuit.out_hole[i]);
605 } else {
606 int j = 0;
607 for (auto it = this->linear_boundary->get<TagKey>().begin();
608 it != this->linear_boundary->get<TagKey>().end(); ++it) {
609 if (it->first == uid_pair.second) {
610 new_out_hole.push_back(subcircuit.out_hole[j]);
611 break;
612 }
613 j++;
614 }
615 }
616 i++;
617 }
618 subcircuit.out_hole = new_out_hole;
619}

650bool MappingFrontier::add_swap(const UnitID& uid_0, const UnitID& uid_1) {
651 // get iterators to linear_boundary uids
652 auto uid0_in_it = this->linear_boundary->find(uid_0);
653 auto uid1_in_it = this->linear_boundary->find(uid_1);
654 // Add Qubit if not in MappingFrontier boundary (i.e. not in circuit)
655 if (uid0_in_it == this->linear_boundary->end()) {
656 this->add_ancilla(uid_0);
657 uid0_in_it = this->linear_boundary->find(uid_0);
658 }
659 if (uid1_in_it == this->linear_boundary->end()) {
660 this->add_ancilla(uid_1);
661 uid1_in_it = this->linear_boundary->find(uid_1);
662 }
663 this->reassignable_nodes_.erase(Node(uid_0));
664 this->reassignable_nodes_.erase(Node(uid_1));
665
666 // Get predecessor edges to SWAP insert location
667 VertPort vp0 = uid0_in_it->second;
668 VertPort vp1 = uid1_in_it->second;
669 EdgeVec predecessors = {
670 this->circuit_.get_nth_out_edge(vp0.first, vp0.second),
671 this->circuit_.get_nth_out_edge(vp1.first, vp1.second)};
672
680 Vertex source0 = this->circuit_.source(predecessors[0]);
681 Vertex source1 = this->circuit_.source(predecessors[1]);
682 if (source0 == source1) {
683 if (this->circuit_.get_OpType_from_Vertex(source0) == OpType::SWAP) {
684 return false;
685 }
686 }
687
688 // update held ancillas
689 // the location/id of the "ancilla node" changes when a SWAP occurs
690 Node n0 = Node(uid_0);
691 Node n1 = Node(uid_1);
692 bool uid0_ancilla =
693 this->ancilla_nodes_.find(n0) != this->ancilla_nodes_.end();
694 bool uid1_ancilla =
695 this->ancilla_nodes_.find(n1) != this->ancilla_nodes_.end();
696
697 // In this case we can still use the node, but we need to make sure that
698 // the wire it's on won't be reassigned
699 // This could reduce performance, but the "reassignable_nodes_" case
700 // deals with Qubit wires that have no multi-qubit quantum operations,
701 // so in practice this won't change the results for a meaningful circuit
702 if (this->reassignable_nodes_.find(n0) != this->reassignable_nodes_.end()) {
703 this->reassignable_nodes_.erase(n0);
704 }
705 if (this->reassignable_nodes_.find(n1) != this->reassignable_nodes_.end()) {
706 this->reassignable_nodes_.erase(n1);
707 }
708
709 if (uid0_ancilla && !uid1_ancilla) {
710 this->ancilla_nodes_.erase(n0);
711 this->ancilla_nodes_.insert(n1);
712 }
713 if (!uid0_ancilla && uid1_ancilla) {
714 this->ancilla_nodes_.erase(n1);
715 this->ancilla_nodes_.insert(n0);
716 }
717
718 // add SWAP vertex to circuit_ and rewire into predecessor
719 Vertex swap_v = this->circuit_.add_vertex(OpType::SWAP);
720 this->circuit_.rewire(
721 swap_v, predecessors, {EdgeType::Quantum, EdgeType::Quantum});
722
723 // Update boundary to reflect new edges
724 EdgeVec successors = this->circuit_.get_all_out_edges(swap_v);
725 this->circuit_.dag[successors[0]].ports.first = 1;
726 this->circuit_.dag[successors[1]].ports.first = 0;
727
728 this->linear_boundary->replace(
729 uid0_in_it, {uid_0, {this->circuit_.source(successors[1]), 0}});
730 this->linear_boundary->replace(
731 uid1_in_it, {uid_1, {this->circuit_.source(successors[0]), 1}});
732
733 // update output vertices of quantum boundary of circuit to reflect changing
734 // qubit paths
735 auto uid0_circuit_boundary_it =
736 this->circuit_.boundary.get<TagID>().find(uid_0);
737 auto uid1_circuit_boundary_it =
738 this->circuit_.boundary.get<TagID>().find(uid_1);
739
740 Vertex uid0_out = uid0_circuit_boundary_it->out_;
741 Vertex uid1_out = uid1_circuit_boundary_it->out_;
742 Vertex uid0_in = uid0_circuit_boundary_it->in_;
743 Vertex uid1_in = uid1_circuit_boundary_it->in_;
744
745 this->circuit_.boundary.get<TagID>().erase(uid_0);
746 this->circuit_.boundary.get<TagID>().erase(uid_1);
747
748 this->circuit_.boundary.get<TagID>().insert({uid_0, uid0_in, uid1_out});
749 this->circuit_.boundary.get<TagID>().insert({uid_1, uid1_in, uid0_out});
750
751 std::map<Node, Node> final_map = {{n0, n1}, {n1, n0}};
752 update_maps(this->bimaps_, {}, final_map);
753 assert_valid_maps();
754 return true;
755}

757void MappingFrontier::add_bridge(
758 const UnitID& control, const UnitID& central, const UnitID& target) {
759 // get predecessors
760 auto control_in_it = this->linear_boundary->find(control);
761 auto central_in_it = this->linear_boundary->find(central);
762 auto target_in_it = this->linear_boundary->find(target);
763
764 // by virtue of method, control and target qubit will always be in BRIDGE.
765 // However, distances used to check BRIDGE and find PATH may use
766 // central qubit that is unallocated, in which add it.
767 if (central_in_it == this->linear_boundary->end()) {
768 this->add_ancilla(central);
769 central_in_it = this->linear_boundary->find(central);
770 }
771
772 Node central_node = Node(central);
773 // In this case we can still use the node, but we need to make sure that
774 // the wire it's on won't be reassigned
775 // This could reduce performance, but the "reassignable_nodes_" case
776 // deals with Qubit wires that have no multi-qubit quantum operations,
777 // so in practice this won't change the results for a meaningful circuit
778 if (this->reassignable_nodes_.find(central_node) !=
779 this->reassignable_nodes_.end()) {
780 this->reassignable_nodes_.erase(central_node);
781 }
782 VertPort vp_control = control_in_it->second;
783 VertPort vp_central = central_in_it->second;
784 VertPort vp_target = target_in_it->second;
785
786 EdgeVec predecessors = {
787 this->circuit_.get_nth_out_edge(vp_control.first, vp_control.second),
788 this->circuit_.get_nth_out_edge(vp_central.first, vp_central.second),
789 this->circuit_.get_nth_out_edge(vp_target.first, vp_target.second),
790 }; // get cx vertex
791 // this should be guaranteeds by pre-checks
792 Vertex cx_v = this->circuit_.target(predecessors[0]);
793 // add bridge
794 Vertex bridge_v = this->circuit_.add_vertex(OpType::BRIDGE);
795 // add bridge vertex to circuit
796 this->circuit_.rewire(
797 bridge_v, predecessors,
798 {EdgeType::Quantum, EdgeType::Quantum, EdgeType::Quantum});
799 // remove old cx vertex
800 this->circuit_.remove_vertex(
801 cx_v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
802 assert_valid_maps();
803}

805void MappingFrontier::add_ancilla(const UnitID& ancilla) {
806 // Given the ancilla node, we 1. add ancilla to the circuit.
807 // 2. map ancilla to placeholder_q, which will be a qubit from the routing
808 // ancilla register that is not present anywhere in the circuit and bimaps.
809 // Add {placeholder_q, ancilla} to the bimaps.
810 register_t ancilla_reg = this->circuit_.get_reg(q_routing_ancilla_reg());
811 unsigned available_ancilla_idx = 0;
812 auto it = ancilla_reg.rbegin();
813 if (it != ancilla_reg.rend()) {
814 available_ancilla_idx = it->first + 1;
815 }
816 for (auto const& it : this->bimaps_->initial.left) {
817 if (it.first.reg_name() == q_routing_ancilla_reg() &&
818 it.first.index()[0] >= available_ancilla_idx) {
819 available_ancilla_idx = it.first.index()[0] + 1;
820 }
821 if (it.second.reg_name() == q_routing_ancilla_reg() &&
822 it.second.index()[0] >= available_ancilla_idx) {
823 available_ancilla_idx = it.second.index()[0] + 1;
824 }
825 }
826 Qubit placeholder_q(q_routing_ancilla_reg(), available_ancilla_idx);
827 Qubit ancilla_q(ancilla);
828 this->circuit_.add_qubit(ancilla_q);
829 this->linear_boundary->insert(
830 {ancilla_q, {this->circuit_.get_in(ancilla_q), 0}});
831
832 this->bimaps_->initial.insert({placeholder_q, ancilla});
833 this->bimaps_->final.insert({placeholder_q, ancilla});
834 this->ancilla_nodes_.insert(Node(ancilla));
835}

837void MappingFrontier::merge_ancilla(
838 const UnitID& merge, const UnitID& ancilla) {
839 // "front" meaning causally ahead
840 // ancilla front, merge back
841
842 auto rewire = [&](const UnitID& front, const UnitID& back) {
843 // get output and input vertices
844 Vertex back_v_in = this->circuit_.get_in(back);
845 Vertex back_v_out = this->circuit_.get_out(back);
846 Vertex front_v_out = this->circuit_.get_out(front);
847
848 // find source vertex & port of merge_v_out
849 // output vertex, so can assume single edge
850 Edge back_out_edge = this->circuit_.get_nth_out_edge(back_v_in, 0);
851 Edge front_in_edge = this->circuit_.get_nth_in_edge(front_v_out, 0);
852 // Find port number
853 port_t back_target_port = this->circuit_.get_target_port(back_out_edge);
854 port_t front_source_port = this->circuit_.get_source_port(front_in_edge);
855 // Find vertices
856 Vertex back_v_target = this->circuit_.target(back_out_edge);
857 Vertex front_v_source = this->circuit_.source(front_in_edge);
858
864 // remove and replace edges
865 this->circuit_.remove_edge(back_out_edge);
866 this->circuit_.remove_edge(front_in_edge);
872 this->circuit_.add_edge(
873 {front_v_source, front_source_port}, {back_v_target, back_target_port},
874 EdgeType::Quantum);
882 // instead of manually updating all boundaries, we change which output
883 // vertex the qubit paths to
884 Edge back_in_edge = this->circuit_.get_nth_in_edge(back_v_out, 0);
885 port_t back_source_port = this->circuit_.get_source_port(back_in_edge);
886 Vertex back_v_source = this->circuit_.source(back_in_edge);
887
888 this->circuit_.remove_edge(back_in_edge);
889
896 this->circuit_.add_edge(
897 {back_v_source, back_source_port}, {front_v_out, 0}, EdgeType::Quantum);
905 // remove empty vertex wire, relabel dag vertices
906 this->circuit_.dag[back_v_in].op = get_op_ptr(OpType::noop);
907 this->circuit_.dag[back_v_out].op = get_op_ptr(OpType::noop);
908
909 TKET_ASSERT(this->circuit_.n_in_edges(back_v_out) == 0);
910 TKET_ASSERT(this->circuit_.n_out_edges(back_v_in) == 0);
911 TKET_ASSERT(this->circuit_.n_in_edges(front_v_out) == 1);
912 this->circuit_.remove_vertex(
913 back_v_in, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
914 this->circuit_.remove_vertex(
915 back_v_out, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
920 this->circuit_.boundary.get<TagID>().erase(back);
921 };
922
929 auto merge_boundary_it = this->linear_boundary->get<TagKey>().find(merge);
930 auto ancilla_boundary_it = this->linear_boundary->get<TagKey>().find(ancilla);
931 TKET_ASSERT(
932 ancilla_boundary_it != this->linear_boundary->get<TagKey>().end());
933 TKET_ASSERT(merge_boundary_it != this->linear_boundary->get<TagKey>().end());
934
935 // If a valid ancilla, this should always be true
936 VertPort ancilla_vp = ancilla_boundary_it->second;
937 TKET_ASSERT(
938 this->circuit_
939 .dag[this->circuit_.target(this->circuit_.get_nth_out_edge(
940 ancilla_vp.first, ancilla_vp.second))]
941 .op->get_type() == OpType::Output);
942
943 VertPort merge_vp = merge_boundary_it->second;
944 Edge merge_edge =
945 this->circuit_.get_nth_out_edge(merge_vp.first, merge_vp.second);
946 Vertex merge_target = this->circuit_.target(merge_edge);
947 port_t merge_target_port = this->circuit_.get_target_port(merge_edge);
948
949 this->linear_boundary->erase(merge_boundary_it);
950 OpType merge_target_optype = this->circuit_.dag[merge_target].op->get_type();
954 rewire(ancilla, merge);
955
964 if (merge_target_optype != OpType::Output) {
965 Edge merge_source_edge =
966 this->circuit_.get_nth_in_edge(merge_target, merge_target_port);
967 Vertex merge_source = this->circuit_.source(merge_source_edge);
968 port_t merge_source_port =
969 this->circuit_.get_source_port(merge_source_edge);
970 this->linear_boundary->replace(
971 ancilla_boundary_it, {ancilla, {merge_source, merge_source_port}});
972 } else {
973 Vertex ancilla_out_vertex = this->circuit_.get_out(ancilla);
974 Edge ancilla_out_in_edge =
975 this->circuit_.get_nth_in_edge(ancilla_out_vertex, 0);
976 Vertex ancilla_out_source = this->circuit_.source(ancilla_out_in_edge);
977 port_t ancilla_out_source_port =
978 this->circuit_.get_source_port(ancilla_out_in_edge);
979 this->linear_boundary->replace(
980 ancilla_boundary_it,
981 {ancilla, {ancilla_out_source, ancilla_out_source_port}});
982 }
983
984 // Update the qubit mappings
985 // let's call the arguments ancilla_node and merge_node
986 // e.g. before merge:
987 // initial := {ancilla_q:node_x, merge_q:some_uid}
988 // final := {ancilla_q:ancilla_node, merge_q:merge_node}
989 // e.g. after merge:
990 // initial := {merge_q:node_x}
991 // final := {merge_q:ancilla_node}
992 // Basically, in both qubit maps, erase the entry with qubit merge_q
993 // then replace the entry ancilla_q -> x with the merge_q -> x
994
995 auto merge_it = this->bimaps_->initial.right.find(merge);
996 TKET_ASSERT(merge_it != this->bimaps_->initial.right.end());
997 UnitID merge_q = merge_it->second;
998 this->bimaps_->initial.right.erase(merge_it);
999 this->bimaps_->final.left.erase(merge_q);
1000 // Find ancilla_q
1001 auto final_it = this->bimaps_->final.right.find(ancilla);
1002 TKET_ASSERT(final_it != this->bimaps_->final.right.end());
1003 UnitID ancilla_q = final_it->second;
1004 // Replace in final map
1005 this->bimaps_->final.right.erase(final_it);
1006 this->bimaps_->final.left.insert({merge_q, ancilla});
1007 // Replace in initial map
1008 auto init_it = this->bimaps_->initial.left.find(ancilla_q);
1009 UnitID init_ancilla_node = init_it->second;
1010 this->bimaps_->initial.left.erase(init_it);
1011 this->bimaps_->initial.left.insert({merge_q, init_ancilla_node});
1012
1016 this->ancilla_nodes_.erase(Node(ancilla));
1017 this->reassignable_nodes_.erase(Node(ancilla));
1018 assert_valid_maps();
1019 return;
1020}

1022bool MappingFrontier::valid_boundary_operation(
1023 const ArchitecturePtr& architecture, const Op_ptr& op,
1024 const std::vector<Node>& uids) const {
1025 OpType ot = op->get_type();
1026
1027 if (ot == OpType::Conditional) {
1028 Op_ptr cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
1029 // conditional boxes are never allowed, too
1030 ot = cond_op_ptr->get_type();
1031 while (ot == OpType::Conditional) {
1032 cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
1033 ot = cond_op_ptr->get_type();
1034 }
1035 }
1036 // Barriers are always allowed
1037 if (is_barrier_type(ot)) {
1038 return true;
1039 }
1040 // boxes are never allowed
1041 if (is_box_type(ot)) {
1042 return false;
1043 }
1044
1045 // this currently allows unplaced single qubits gates
1046 // this should be changes in the future
1047 if (uids.size() == 1) {
1048 return true;
1049 }
1050
1051 // allow two qubit gates only for placed and connected nodes
1052 if (uids.size() == 2) {
1053 bool n0 = architecture->node_exists(uids[0]);
1054 bool n1 = architecture->node_exists(uids[1]);
1055 if (n0 && n1) {
1056 bool bde = architecture->bidirectional_edge_exists(uids[0], uids[1]);
1057 if (bde) {
1058 return true;
1059 }
1060 }
1061 } else if (uids.size() == 3 && ot == OpType::BRIDGE) {
1062 bool con_0_exists =
1063 architecture->bidirectional_edge_exists(uids[0], uids[1]);
1064 bool con_1_exists =
1065 architecture->bidirectional_edge_exists(uids[2], uids[1]);
1066 if (architecture->node_exists(uids[0]) &&
1067 architecture->node_exists(uids[1]) &&
1068 architecture->node_exists(uids[2]) && con_0_exists && con_1_exists) {
1069 return true;
1070 }
1071 }
1072
1073 return false;
1074}
