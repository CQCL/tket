38 static Bit get_bit_from_bool_frontier(
39  const std::shared_ptr<b_frontier_t>& b_frontier, const EdgeVec& ev) {
40  TKET_ASSERT(ev.size() > 0);
41  // condition is that if one Edge in EdgeVector ev is in a
42  // held bundle, then all are so return true
43  for (auto it = b_frontier->get<TagKey>().begin();
44  it != b_frontier->get<TagKey>().end(); ++it) {
45  for (const Edge& e0 : ev) {
46  for (const Edge& e1 : it->second) {
47  if (e0 == e1) {
48  return it->first;
49  }
50  }
51  }
52  }
59  TKET_ASSERT(false);
60 }

62 std::shared_ptr<unit_frontier_t> frontier_convert_vertport_to_edge(
63  const Circuit& circuit,
64  const std::shared_ptr<unit_vertport_frontier_t>& u_frontier) {
65  // make empty unit_frontier_t object
66  std::shared_ptr<unit_frontier_t> output_frontier =
67  std::make_shared<unit_frontier_t>();
68  // iterate through u_frontier, convert VertPort to Edge and insert
69  for (const std::pair<UnitID, VertPort>& pair : u_frontier->get<TagKey>()) {
70  output_frontier->insert(
71  {pair.first,
72  circuit.get_nth_out_edge(pair.second.first, pair.second.second)});
73  }
74  return output_frontier;
75 }

81 MappingFrontier::MappingFrontier(Circuit& _circuit) : circuit_(_circuit) {
82  this->linear_boundary = std::make_shared<unit_vertport_frontier_t>();
83  this->boolean_boundary = std::make_shared<b_frontier_t>();
84  this->bimaps_ = std::make_shared<unit_bimaps_t>();
85 
86  // Set up {UnitID, VertPort} objects for quantum and classical boundaries
87  for (const Qubit& qb : this->circuit_.all_qubits()) {
88  this->linear_boundary->insert({qb, {this->circuit_.get_in(qb), 0}});
89  this->bimaps_->initial.insert({qb, qb});
90  this->bimaps_->final.insert({qb, qb});
91  }
92  for (const Bit& bit : this->circuit_.all_bits()) {
93  Vertex bit_input = this->circuit_.get_in(bit);
94  EdgeVec bool_bundle = this->circuit_.get_nth_b_out_bundle(bit_input, 0);
95  // N.B. An Input Vertex may have boolean and classical out edges.
96  if (!bool_bundle.empty()) {
97  this->boolean_boundary->insert({bit, bool_bundle});
98  }
99  if (this->circuit_.n_out_edges_of_type(bit_input, EdgeType::Classical) >
100  0) {
101  this->linear_boundary->insert({bit, {bit_input, 0}});
102  }
103  }
104 }

167 void MappingFrontier::advance_next_2qb_slice(unsigned max_advance) {
168  bool boundary_updated = false;
169  unsigned loop = 0;
170  std::shared_ptr<unit_frontier_t> current_frontier =
171  frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary);
172 
173  // Get all vertices in first cut
174  VertexVec immediate_cut_vertices_v =
175  *(this->circuit_
176  .next_cut(current_frontier, std::make_shared<b_frontier_t>())
177  .slice);
178  do {
179  // each do section first finds the next set of edges after the held set
180  // for edges with target vertices with all their edges presented in the
181  // first set
182  loop++;
183  boundary_updated = false;
184  // produce next frontier object
185  std::shared_ptr<unit_frontier_t> next_frontier =
186  std::make_shared<unit_frontier_t>();
187 
188  for (const std::pair<UnitID, Edge>& pair :
189  current_frontier->get<TagKey>()) {
190  // if target_v not in immediate_cut_vertices, then do not pass it
191  Vertex target_v = this->circuit_.target(pair.second);
192  EdgeVec in_edges =
193  this->circuit_.get_in_edges_of_type(target_v, EdgeType::Quantum);
194  bool in_slice =
195  std::find(
196  immediate_cut_vertices_v.begin(), immediate_cut_vertices_v.end(),
197  target_v) != immediate_cut_vertices_v.end();
198  OpType ot = this->circuit_.get_OpType_from_Vertex(target_v);
199  if (((!in_slice && in_edges.size() > 1) || ot == OpType::Output ||
200  ot == OpType::ClOutput) &&
201  this->circuit_.get_OpType_from_Vertex(target_v) != OpType::Barrier) {
202  // Vertex either not allowed to pass, or is output vertex => update
203  // nothing
204  next_frontier->insert({pair.first, pair.second});
205  } else {
206  // vertex can be surpassed, so update linear_boundary and
207  // next_frontier with next edge
208  Edge next_edge = this->circuit_.get_next_edge(target_v, pair.second);
209  this->linear_boundary->replace(
210  this->linear_boundary->get<TagKey>().find(pair.first),
211  {pair.first,
212  {target_v, this->circuit_.get_source_port(next_edge)}});
213  next_frontier->insert({pair.first, next_edge});
214  }
215  }
216 
217  // Given new frontier, find the actual next cut
218  CutFrontier next_cut = this->circuit_.next_cut(
219  next_frontier, std::make_shared<b_frontier_t>());
220  // For each vertex in a slice, if its physically permitted, update
221  // linear_boundary with quantum out edges from vertex (i.e.
222  // next_cut.u_frontier)
223  for (const Vertex& vert : *next_cut.slice) {
224  // Output means we don't want to pass, so just leave
225  OpType ot = this->circuit_.get_OpType_from_Vertex(vert);
226  if (ot == OpType::Output || ot == OpType::ClOutput) {
227  continue;
228  }
229  EdgeVec in_edges =
230  this->circuit_.get_in_edges_of_type(vert, EdgeType::Quantum);
231  // More than 1 edge means we want to keep edges, so continue
232  if (in_edges.size() > 1) {
233  continue;
234  }
235  // can guarantee that we update now as non-updating cases have been
236  // continued
237  boundary_updated = true;
238  // push edge past single qubit vertex, repeat
239  UnitID uid = get_unitid_from_unit_frontier(
240  this->linear_boundary, {this->circuit_.source(in_edges[0]),
241  this->circuit_.get_source_port(in_edges[0])});
242 
243  Edge replacement_edge =
244  next_cut.u_frontier->get<TagKey>().find(uid)->second;
245 
246  Vertex source_vertex = this->circuit_.source(replacement_edge);
247  port_t source_port = this->circuit_.get_source_port(replacement_edge);
248 
249  this->linear_boundary->replace(
250  this->linear_boundary->get<TagKey>().find(uid),
251  {uid, {source_vertex, source_port}});
252  }
253  current_frontier = next_frontier;
254  } while (boundary_updated && loop <= max_advance);
255  return;
256 }

264 void MappingFrontier::advance_frontier_boundary(
265  const ArchitecturePtr& architecture) {
266  bool boundary_updated = false;
267  do {
273  boundary_updated = false;
274  std::shared_ptr<unit_frontier_t> l_frontier_edges =
275  frontier_convert_vertport_to_edge(
276  this->circuit_, this->linear_boundary);
277  CutFrontier next_cut =
278  this->circuit_.next_cut(l_frontier_edges, this->boolean_boundary);
287  for (const Vertex& vert : *next_cut.slice) {
292  std::vector<std::pair<UnitID, EdgeType>> in_uids;
293  std::vector<Node> nodes;
294  std::vector<Edge> all_in_edges = this->circuit_.get_in_edges(vert);
295  for (const Edge& edge : all_in_edges) {
296  UnitID uid;
297  EdgeType edge_type = this->circuit_.get_edgetype(edge);
298  switch (edge_type) {
299  case EdgeType::Quantum: {
300  uid = get_unitid_from_unit_frontier(
301  this->linear_boundary, {this->circuit_.source(edge),
302  this->circuit_.get_source_port(edge)});
303  // We use "nodes" to check if Quantum arguments respect the
304  // architecture
305  nodes.push_back(Node(uid));
306  break;
307  }
308  case EdgeType::Classical: {
309  uid = get_unitid_from_unit_frontier(
310  this->linear_boundary, {this->circuit_.source(edge),
311  this->circuit_.get_source_port(edge)});
312  break;
313  }
314  case EdgeType::Boolean: {
315  // N.B. function will find uid even if the bundle has more than the
316  // single "edge" passed
317  uid = get_bit_from_bool_frontier(this->boolean_boundary, {edge});
318  break;
319  }
320  default: {
321  TKET_ASSERT(false);
322  }
323  }
324  in_uids.push_back({uid, edge_type});
325  }
331  if (nodes.empty() ||
332  this->valid_boundary_operation(
333  architecture, this->circuit_.get_Op_ptr_from_Vertex(vert),
334  nodes)) {
335  boundary_updated = true;
336 
343  for (const std::pair<UnitID, EdgeType>& uid : in_uids) {
344  switch (uid.second) {
345  case EdgeType::Boolean:
346  break;
347  case EdgeType::Quantum:
348  case EdgeType::Classical: {
349  Edge replacement_edge =
350  next_cut.u_frontier->get<TagKey>().find(uid.first)->second;
351  this->linear_boundary->replace(
352  this->linear_boundary->get<TagKey>().find(uid.first),
353  {uid.first,
354  {this->circuit_.source(replacement_edge),
355  this->circuit_.get_source_port(replacement_edge)}});
356  break;
357  }
358  default: {
359  TKET_ASSERT(false);
360  }
361  }
362  }
363 
372  std::vector<EdgeVec> in_bundles = this->circuit_.get_b_in_bundles(vert);
373  std::vector<EdgeVec> out_bundles =
374  this->circuit_.get_b_out_bundles(vert);
375 
376  unsigned n_in_bundles = in_bundles.size();
377  unsigned n_out_bundles = out_bundles.size();
378 
379  TKET_ASSERT(n_out_bundles == n_in_bundles);
380  TKET_ASSERT(n_in_bundles == in_uids.size());
381 
403  for (port_t port = 0; port < n_in_bundles; port++) {
404  std::pair<UnitID, EdgeType> linear_uid = in_uids[port];
405  EdgeVec in_bundle = in_bundles[port];
406  EdgeVec out_bundle = out_bundles[port];
407  switch (linear_uid.second) {
408  // Linear & can't spawn Boolean bundle: pass
409  case EdgeType::Quantum:
410  break;
411  // Linear but can spawn Boolean bundle
412  // If Boolean edges spawned, add to boolean boundary
413  case EdgeType::Classical: {
414  EdgeVec out_bundle = out_bundles[port];
415  TKET_ASSERT(in_bundle.empty());
416  if (!out_bundle.empty()) {
417  this->boolean_boundary->insert(
418  {Bit(linear_uid.first), out_bundle});
419  }
420  break;
421  }
422  // Can spawn, erase or edit boolean bundles
423  case EdgeType::Boolean: {
424  TKET_ASSERT(!in_bundle.empty());
425  Bit bit = Bit(linear_uid.first);
426  auto boolean_it = this->boolean_boundary->get<TagKey>().find(bit);
427  TKET_ASSERT(
428  boolean_it != this->boolean_boundary->get<TagKey>().end());
429  // update "out bundle" with other boolean edges attached to other
430  // vertices
431  for (const Edge& edge : boolean_it->second) {
432  if (std::find(in_bundle.begin(), in_bundle.end(), edge) ==
433  in_bundle.end()) {
434  out_bundle.push_back(edge);
435  }
436  }
437  if (out_bundle.empty()) {
438  // => all edges in boolean bundle in boolean boundary attached
439  // to this vertex
440  // => vertex has no edges in output boolean bundle for thos port
441  // => can erase Bit from boolean boundary as its not longer used
442  this->boolean_boundary->erase(boolean_it);
443  } else {
444  // => either Vertex has edges in output boolean bundle
445  // => or there are other edges in boolean bundle held in boolean
446  // boundary that are not attached to this vertex
447  // => update boolean boundary
448  this->boolean_boundary->replace(boolean_it, {bit, out_bundle});
449  }
450  break;
451  }
452  default: {
453  TKET_ASSERT(false);
454  }
455  }
456  }
457  }
458  }
459  } while (boundary_updated);
465  return;
466 }

476 Subcircuit MappingFrontier::get_frontier_subcircuit(
477  unsigned _max_subcircuit_depth, unsigned _max_subcircuit_size) const {
478  CutFrontier current_cut = this->circuit_.next_cut(
479  frontier_convert_vertport_to_edge(this->circuit_, this->linear_boundary),
480  this->boolean_boundary);
481 
482  unsigned subcircuit_depth = 1;
483  VertexSet subcircuit_vertices(
484  current_cut.slice->begin(), current_cut.slice->end());
485  // add cuts of vertices to subcircuit_vertices until constraints met, or end
486  // of circuit reached
487  while (subcircuit_depth < _max_subcircuit_depth &&
488  unsigned(subcircuit_vertices.size()) < _max_subcircuit_size &&
489  current_cut.slice->size() > 0) {
490  current_cut =
491  this->circuit_.next_cut(current_cut.u_frontier, current_cut.b_frontier);
492  subcircuit_depth++;
493  subcircuit_vertices.insert(
494  current_cut.slice->begin(), current_cut.slice->end());
495  }
496  TKET_ASSERT(subcircuit_vertices.size() != 0);
497  return Subcircuit(
498  convert_u_frontier_to_edges(*frontier_convert_vertport_to_edge(
499  this->circuit_, this->linear_boundary)),
500  convert_u_frontier_to_edges(*current_cut.u_frontier),
501  subcircuit_vertices);
502 }

510 void MappingFrontier::update_bimaps(UnitID qubit, UnitID node) {
511  // Update initial map
512  auto init_it = this->bimaps_->initial.left.find(qubit);
513  TKET_ASSERT(init_it != this->bimaps_->initial.left.end());
514  this->bimaps_->initial.left.erase(init_it);
515  this->bimaps_->initial.left.insert({qubit, node});
516  // Update final map
517  auto final_it = this->bimaps_->final.left.find(qubit);
518  TKET_ASSERT(final_it != this->bimaps_->final.left.end());
519  this->bimaps_->final.left.erase(final_it);
520  this->bimaps_->final.left.insert({qubit, node});
521 }

523 void MappingFrontier::update_linear_boundary_uids(
524  const unit_map_t& relabelled_uids) {
525  for (const std::pair<const UnitID, UnitID>& label : relabelled_uids) {
526  // implies new labelling
527  if (label.first != label.second &&
528  this->reassignable_nodes_.find(Node(label.second)) ==
529  this->reassignable_nodes_.end()) {
530  // by type, label.first already assumed in circuit
531  // this condition means label.second also in circuit
532  // implies that a merging is done -> remove first qubit
533  if (this->linear_boundary->get<TagKey>().find(label.second) !=
534  this->linear_boundary->get<TagKey>().end()) {
535  // erase, assume updated already
536  this->linear_boundary->erase(label.first);
537  } else {
538  auto current_label_it =
539  this->linear_boundary->get<TagKey>().find(label.first);
540  // relabel "label.first" with "label.second"
541  this->linear_boundary->replace(
542  current_label_it, {label.second, current_label_it->second});
543  unit_map_t relabel = {label};
544  this->circuit_.rename_units(relabel);
545  }
546  }
547  }

550 void MappingFrontier::permute_subcircuit_q_out_hole(
551  const unit_map_t& final_permutation, Subcircuit& subcircuit) {
552  EdgeVec new_q_out_hole;
553  int i = 0;
554  if (this->linear_boundary->size() != final_permutation.size()) {
555  throw MappingFrontierError(
556  "Number of Qubits in mapping permutation does not match number of "
557  "Qubits in MappingFrontier boundary, for permuting Qubits as with "
558  "routed Subcircuit.");
559  }
560  for (const std::pair<UnitID, VertPort>& pair :
561  this->linear_boundary->get<TagKey>()) {
562  auto it = final_permutation.find(pair.first);
563  if (it == final_permutation.end()) {
564  throw MappingFrontierError("Qubit in boundary not in permutation.");
565  }
566  std::pair<UnitID, UnitID> uid_pair = *it;
567  if (uid_pair.first == uid_pair.second) {
568  new_q_out_hole.push_back(subcircuit.q_out_hole[i]);
569  } else {
570  int j = 0;
571  for (auto it = this->linear_boundary->get<TagKey>().begin();
572  it != this->linear_boundary->get<TagKey>().end(); ++it) {
573  if (it->first == uid_pair.second) {
574  new_q_out_hole.push_back(subcircuit.q_out_hole[j]);
575  break;
576  }
577  j++;
578  }
579  }
580  i++;
581  }
582  subcircuit.q_out_hole = new_q_out_hole;
583 }

614 bool MappingFrontier::add_swap(const UnitID& uid_0, const UnitID& uid_1) {
615  // get iterators to linear_boundary uids
616  auto uid0_in_it = this->linear_boundary->find(uid_0);
617  auto uid1_in_it = this->linear_boundary->find(uid_1);
618  // Add Qubit if not in MappingFrontier boundary (i.e. not in circuit)
619  if (uid0_in_it == this->linear_boundary->end()) {
620  this->add_ancilla(uid_0);
621  uid0_in_it = this->linear_boundary->find(uid_0);
622  }
623  if (uid1_in_it == this->linear_boundary->end()) {
624  this->add_ancilla(uid_1);
625  uid1_in_it = this->linear_boundary->find(uid_1);
626  }
627 
628  this->reassignable_nodes_.erase(Node(uid_0));
629  this->reassignable_nodes_.erase(Node(uid_1));
630 
631  // Get predecessor edges to SWAP insert location
632  VertPort vp0 = uid0_in_it->second;
633  VertPort vp1 = uid1_in_it->second;
634  EdgeVec predecessors = {
635  this->circuit_.get_nth_out_edge(vp0.first, vp0.second),
636  this->circuit_.get_nth_out_edge(vp1.first, vp1.second)};
637 
645  Vertex source0 = this->circuit_.source(predecessors[0]);
646  Vertex source1 = this->circuit_.source(predecessors[1]);
647  if (source0 == source1) {
648  if (this->circuit_.get_OpType_from_Vertex(source0) == OpType::SWAP) {
649  return false;
650  }
651  }
652 
653  // update held ancillas
654  // the location/id of the "ancilla node" changes when a SWAP occurs
655  Node n0 = Node(uid_0);
656  Node n1 = Node(uid_1);
657  bool uid0_ancilla =
658  this->ancilla_nodes_.find(n0) != this->ancilla_nodes_.end();
659  bool uid1_ancilla =
660  this->ancilla_nodes_.find(n1) != this->ancilla_nodes_.end();
661 
662  // In this case we can still use the node, but we need to make sure that
663  // the wire it's on won't be reassigned
664  // This could reduce performance, but the "reassignable_nodes_" case
665  // deals with Qubit wires that have no multi-qubit quantum operations,
666  // so in practice this won't change the results for a meaningful circuit
667  if (this->reassignable_nodes_.find(n0) != this->reassignable_nodes_.end()) {
668  this->reassignable_nodes_.erase(n0);
669  }
670  if (this->reassignable_nodes_.find(n1) != this->reassignable_nodes_.end()) {
671  this->reassignable_nodes_.erase(n1);
672  }
673 
674  if (uid0_ancilla && !uid1_ancilla) {
675  this->ancilla_nodes_.erase(n0);
676  this->ancilla_nodes_.insert(n1);
677  }
678  if (!uid0_ancilla && uid1_ancilla) {
679  this->ancilla_nodes_.erase(n1);
680  this->ancilla_nodes_.insert(n0);
681  }
682 
683  // add SWAP vertex to circuit_ and rewire into predecessor
684  Vertex swap_v = this->circuit_.add_vertex(OpType::SWAP);
685  this->circuit_.rewire(
686  swap_v, predecessors, {EdgeType::Quantum, EdgeType::Quantum});
687 
688  // Update boundary to reflect new edges
689  EdgeVec successors = this->circuit_.get_all_out_edges(swap_v);
690  this->circuit_.dag[successors[0]].ports.first = 1;
691  this->circuit_.dag[successors[1]].ports.first = 0;
692 
693  this->linear_boundary->replace(
694  uid0_in_it, {uid_0, {this->circuit_.source(successors[1]), 0}});
695  this->linear_boundary->replace(
696  uid1_in_it, {uid_1, {this->circuit_.source(successors[0]), 1}});
697 
698  // update output vertices of quantum boundary of circuit to reflect changing
699  // qubit paths
700  auto uid0_circuit_boundary_it =
701  this->circuit_.boundary.get<TagID>().find(uid_0);
702  auto uid1_circuit_boundary_it =
703  this->circuit_.boundary.get<TagID>().find(uid_1);
704 
705  Vertex uid0_out = uid0_circuit_boundary_it->out_;
706  Vertex uid1_out = uid1_circuit_boundary_it->out_;
707  Vertex uid0_in = uid0_circuit_boundary_it->in_;
708  Vertex uid1_in = uid1_circuit_boundary_it->in_;
709 
710  this->circuit_.boundary.get<TagID>().erase(uid_0);
711  this->circuit_.boundary.get<TagID>().erase(uid_1);
712 
713  this->circuit_.boundary.get<TagID>().insert({uid_0, uid0_in, uid1_out});
714  this->circuit_.boundary.get<TagID>().insert({uid_1, uid1_in, uid0_out});
715 
716  std::map<Node, Node> final_map = {{n0, n1}, {n1, n0}};
717  update_maps(this->bimaps_, {}, final_map);
718  return true;

721 void MappingFrontier::add_bridge(
722  const UnitID& control, const UnitID& central, const UnitID& target) {
723  // get predecessors
724  auto control_in_it = this->linear_boundary->find(control);
725  auto central_in_it = this->linear_boundary->find(central);
726  auto target_in_it = this->linear_boundary->find(target);
727 
728  // by virtue of method, control and target qubit will always be in BRIDGE.
729  // However, distances used to check BRIDGE and find PATH may use
730  // central qubit that is unallocated, in which add it.
731  if (central_in_it == this->linear_boundary->end()) {
732  this->add_ancilla(central);
733  central_in_it = this->linear_boundary->find(central);
734  }
735 
736  Node central_node = Node(central);
737  // In this case we can still use the node, but we need to make sure that
738  // the wire it's on won't be reassigned
739  // This could reduce performance, but the "reassignable_nodes_" case
740  // deals with Qubit wires that have no multi-qubit quantum operations,
741  // so in practice this won't change the results for a meaningful circuit
742  if (this->reassignable_nodes_.find(central_node) !=
743  this->reassignable_nodes_.end()) {
744  this->reassignable_nodes_.erase(central_node);
745  }
746  VertPort vp_control = control_in_it->second;
747  VertPort vp_central = central_in_it->second;
748  VertPort vp_target = target_in_it->second;
749 
750  EdgeVec predecessors = {
751  this->circuit_.get_nth_out_edge(vp_control.first, vp_control.second),
752  this->circuit_.get_nth_out_edge(vp_central.first, vp_central.second),
753  this->circuit_.get_nth_out_edge(vp_target.first, vp_target.second),
754  }; // get cx vertex
755  // this should be guaranteeds by pre-checks
756  Vertex cx_v = this->circuit_.target(predecessors[0]);
757  // add bridge
758  Vertex bridge_v = this->circuit_.add_vertex(OpType::BRIDGE);
759  // add bridge vertex to circuit
760  this->circuit_.rewire(
761  bridge_v, predecessors,
762  {EdgeType::Quantum, EdgeType::Quantum, EdgeType::Quantum});
763  // remove old cx vertex
764  this->circuit_.remove_vertex(
765  cx_v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
766 }

784 void MappingFrontier::merge_ancilla(
785  const UnitID& merge, const UnitID& ancilla) {
786  // "front" meaning causally ahead
787  // ancilla front, merge back
788 
789  auto rewire = [&](const UnitID& front, const UnitID& back) {
790  // get output and input vertices
791  Vertex back_v_in = this->circuit_.get_in(back);
792  Vertex back_v_out = this->circuit_.get_out(back);
793  Vertex front_v_out = this->circuit_.get_out(front);
794 
795  // find source vertex & port of merge_v_out
796  // output vertex, so can assume single edge
797  Edge back_out_edge = this->circuit_.get_nth_out_edge(back_v_in, 0);
798  Edge front_in_edge = this->circuit_.get_nth_in_edge(front_v_out, 0);
799  // Find port number
800  port_t back_target_port = this->circuit_.get_target_port(back_out_edge);
801  port_t front_source_port = this->circuit_.get_source_port(front_in_edge);
802  // Find vertices
803  Vertex back_v_target = this->circuit_.target(back_out_edge);
804  Vertex front_v_source = this->circuit_.source(front_in_edge);
805 
811  // remove and replace edges
812  this->circuit_.remove_edge(back_out_edge);
813  this->circuit_.remove_edge(front_in_edge);
819  this->circuit_.add_edge(
820  {front_v_source, front_source_port}, {back_v_target, back_target_port},
821  EdgeType::Quantum);
829  // instead of manually updating all boundaries, we change which output
830  // vertex the qubit paths to
831  Edge back_in_edge = this->circuit_.get_nth_in_edge(back_v_out, 0);
832  port_t back_source_port = this->circuit_.get_source_port(back_in_edge);
833  Vertex back_v_source = this->circuit_.source(back_in_edge);
834 
835  this->circuit_.remove_edge(back_in_edge);
836 
843  this->circuit_.add_edge(
844  {back_v_source, back_source_port}, {front_v_out, 0}, EdgeType::Quantum);
852  // remove empty vertex wire, relabel dag vertices
853  this->circuit_.dag[back_v_in].op = get_op_ptr(OpType::noop);
854  this->circuit_.dag[back_v_out].op = get_op_ptr(OpType::noop);
855 
856  TKET_ASSERT(this->circuit_.n_in_edges(back_v_out) == 0);
857  TKET_ASSERT(this->circuit_.n_out_edges(back_v_in) == 0);
858  TKET_ASSERT(this->circuit_.n_in_edges(front_v_out) == 1);
859  this->circuit_.remove_vertex(
860  back_v_in, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
861  this->circuit_.remove_vertex(
862  back_v_out, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
867  this->circuit_.boundary.get<TagID>().erase(back);
868  };
869 
876  auto merge_boundary_it = this->linear_boundary->get<TagKey>().find(merge);
877  auto ancilla_boundary_it = this->linear_boundary->get<TagKey>().find(ancilla);
878  TKET_ASSERT(
879  ancilla_boundary_it != this->linear_boundary->get<TagKey>().end());
880  TKET_ASSERT(merge_boundary_it != this->linear_boundary->get<TagKey>().end());
881 
882  // If a valid ancilla, this should always be true
883  VertPort ancilla_vp = ancilla_boundary_it->second;
884  TKET_ASSERT(
885  this->circuit_
886  .dag[this->circuit_.target(this->circuit_.get_nth_out_edge(
887  ancilla_vp.first, ancilla_vp.second))]
888  .op->get_type() == OpType::Output);
889 
890  VertPort merge_vp = merge_boundary_it->second;
891  Edge merge_edge =
892  this->circuit_.get_nth_out_edge(merge_vp.first, merge_vp.second);
893  Vertex merge_target = this->circuit_.target(merge_edge);
894  port_t merge_target_port = this->circuit_.get_target_port(merge_edge);
895 
896  this->linear_boundary->erase(merge_boundary_it);
897  OpType merge_target_optype = this->circuit_.dag[merge_target].op->get_type();
901  rewire(ancilla, merge);
902 
911  if (merge_target_optype != OpType::Output) {
912  Edge merge_source_edge =
913  this->circuit_.get_nth_in_edge(merge_target, merge_target_port);
914  Vertex merge_source = this->circuit_.source(merge_source_edge);
915  port_t merge_source_port =
916  this->circuit_.get_source_port(merge_source_edge);
917  this->linear_boundary->replace(
918  ancilla_boundary_it, {ancilla, {merge_source, merge_source_port}});
919  } else {
920  Vertex ancilla_out_vertex = this->circuit_.get_out(ancilla);
921  Edge ancilla_out_in_edge =
922  this->circuit_.get_nth_in_edge(ancilla_out_vertex, 0);
923  Vertex ancilla_out_source = this->circuit_.source(ancilla_out_in_edge);
924  port_t ancilla_out_source_port =
925  this->circuit_.get_source_port(ancilla_out_in_edge);
926  this->linear_boundary->replace(
927  ancilla_boundary_it,
928  {ancilla, {ancilla_out_source, ancilla_out_source_port}});
929  }
930 
931  // Update the qubit mappings
932  // let's call the arguments ancilla_node and merge_node
933  // e.g. before merge:
934  // initial := {ancilla_q:node_x, merge_q:some_uid}
935  // final := {ancilla_q:ancilla_node, merge_q:merge_node}
936  // e.g. after merge:
937  // initial := {merge_q:node_x}
938  // final := {merge_q:ancilla_node}
939  // Basically, in both qubit maps, erase the entry with qubit merge_q
940  // then replace the entry ancilla_q -> x with the merge_q -> x
941 
942  auto merge_it = this->bimaps_->initial.right.find(merge);
943  TKET_ASSERT(merge_it != this->bimaps_->initial.right.end());
944  UnitID merge_q = merge_it->second;
945  this->bimaps_->initial.right.erase(merge_it);
946  this->bimaps_->final.left.erase(merge_q);
947  // Find ancilla_q
948  auto final_it = this->bimaps_->final.right.find(ancilla);
949  UnitID ancilla_q = final_it->second;
950  // Replace in final map
951  this->bimaps_->final.right.erase(final_it);
952  this->bimaps_->final.left.insert({merge_q, ancilla});
953  // Replace in initial map
954  auto init_it = this->bimaps_->initial.left.find(ancilla_q);
955  UnitID init_ancilla_node = init_it->second;
956  this->bimaps_->initial.left.erase(init_it);
957  this->bimaps_->initial.left.insert({merge_q, init_ancilla_node});
958 
962  this->ancilla_nodes_.erase(Node(ancilla));
963  this->reassignable_nodes_.erase(Node(ancilla));
964 
965  return;
966 }

968 bool MappingFrontier::valid_boundary_operation(
969  const ArchitecturePtr& architecture, const Op_ptr& op,
970  const std::vector<Node>& uids) const {
971  // boxes are never allowed
972  OpType ot = op->get_type();
973  if (is_box_type(ot)) {
974  return false;
975  }
976 
977  if (ot == OpType::Conditional) {
978  Op_ptr cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
979  // conditional boxes are never allowed, too
980  OpType ot = cond_op_ptr->get_type();
981  while (ot == OpType::Conditional) {
982  cond_op_ptr = static_cast<const Conditional&>(*op).get_op();
983  ot = cond_op_ptr->get_type();
984  if (is_box_type(ot)) {
985  return false;
986  }
987  }
988  }
989 
990  // Barriers are allways allowed
991  if (ot == OpType::Barrier) {
992  return true;
993  }
994 
995  // this currently allows unplaced single qubits gates
996  // this should be changes in the future
997  if (uids.size() == 1) {
998  return true;
999  }
1000 
1001  // allow two qubit gates only for placed and connected nodes
1002  if (uids.size() == 2) {
1003  bool n0 = architecture->node_exists(uids[0]);
1004  bool n1 = architecture->node_exists(uids[1]);
1005  if (n0 && n1) {
1006  bool bde = architecture->bidirectional_edge_exists(uids[0], uids[1]);
1007  if (bde) {
1008  return true;
1009  }
1010  }
1011  } else if (uids.size() == 3 && ot == OpType::BRIDGE) {
1012  bool con_0_exists =
1013  architecture->bidirectional_edge_exists(uids[0], uids[1]);
1014  bool con_1_exists =
1015  architecture->bidirectional_edge_exists(uids[2], uids[1]);
1016  if (architecture->node_exists(uids[0]) &&
1017  architecture->node_exists(uids[1]) &&
1018  architecture->node_exists(uids[2]) && con_0_exists && con_1_exists) {
1019  return true;
1020  }
1021  }
1022 
1023  return false;
1024 }
