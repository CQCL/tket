24bool MappingManager::route_circuit(
25 Circuit& circuit,
26 const std::vector<RoutingMethodPtr>& routing_methods) const {
27 // we make bimaps if not present
28 unit_bimaps_t maps;
29 for (const Qubit& qubit : circuit.all_qubits()) {
30 maps.initial.left.insert({qubit, qubit});
31 maps.final.left.insert({qubit, qubit});
32 }
33 return this->route_circuit_with_maps(
34 circuit, routing_methods, std::make_shared<unit_bimaps_t>(maps));
35}

37bool MappingManager::route_circuit_with_maps(
38 Circuit& circuit, const std::vector<RoutingMethodPtr>& routing_methods,
39 std::shared_ptr<unit_bimaps_t> maps) const {
40 if (circuit.n_qubits() > this->architecture_->n_nodes()) {
41 std::string error_string =
42 "Circuit has" + std::to_string(circuit.n_qubits()) +
43 " logical qubits. Architecture has " +
44 std::to_string(this->architecture_->n_nodes()) +
45 " physical qubits. Circuit to be routed can not have more "
46 "qubits than the Architecture.";
47 throw MappingManagerError(error_string);
48 }
49
58 // get unplaced UnitID
59 std::set<UnitID> unplaced, placed;
60 std::set<Node> subarch_nodes = this->architecture_->nodes();
61 for (const Qubit& qubit : circuit.all_qubits()) {
62 Node n(qubit);
63 if (!this->architecture_->node_exists(n)) {
64 unplaced.insert(n);
65 } else {
66 subarch_nodes.erase(n);
67 }
68 }
69 // get qubit with no physical qubit constraint
70 Vertex v;
71 Edge e;
72 std::set<UnitID> to_assign;
73 for (const UnitID& unitid : unplaced) {
74 v = circuit.get_in(Qubit(unitid));
75 e = circuit.get_nth_out_edge(v, 0);
76 while (true) {
77 if (circuit.n_out_edges_of_type(v, EdgeType::Quantum) > 1 &&
78 circuit.get_OpType_from_Vertex(v) != OpType::Barrier)
79 break;
80 v = circuit.target(e);
81 if (circuit.detect_final_Op(v)) break;
82 e = circuit.get_next_edge(v, e);
83 }
84 if (circuit.detect_final_Op(v)) {
85 to_assign.insert(unitid);
86 }
87 }
88 // assign these edge cases to "bad" Architecture Node
89 bool modify_at_start = false;
90 std::set<Node> reassignable_nodes;
91 if (!to_assign.empty()) {
92 modify_at_start = true;
93 // Use a copy as it finds iteratively bad nodes by removing
94 std::vector<Node> subarch_nodes_v;
95 std::copy(
96 subarch_nodes.begin(), subarch_nodes.end(),
97 std::back_inserter(subarch_nodes_v));
98 // as we produce a subarch, these "bad" nodes may in practice be fine
99 // However we allow these single qubit assignments to be overwritten by
100 // dynamic assignment
101 Architecture subarc = this->architecture_->create_subarch(subarch_nodes_v);
102 std::set<Node> nodes = subarc.remove_worst_nodes(to_assign.size());
103 TKET_ASSERT(nodes.size() == to_assign.size());
104 // note that as we do this pre MappingFrontier we just need to update the
105 // Circuit and bimaps
106 std::map<Qubit, Node> relabelling_map;
107 auto it = nodes.begin();
108 auto jt = to_assign.begin();
109 while (it != nodes.end()) {
110 relabelling_map.insert({Qubit(*jt), Node(*it)});
111 reassignable_nodes.insert(Node(*it));
112 ++it;
113 ++jt;
114 }
115 circuit.rename_units(relabelling_map);
116 // relabel maps
117 for (const std::pair<const Qubit, Node>& q_n : relabelling_map) {
118 auto q_initial_it = maps->initial.right.find(q_n.first);
119 TKET_ASSERT(q_initial_it != maps->initial.right.end());
120 UnitID original_first = q_initial_it->second;
121 auto q_final_it = maps->final.left.find(original_first);
122 TKET_ASSERT(q_final_it != maps->final.left.end());
123 maps->initial.right.erase(q_initial_it);
124 maps->final.left.erase(q_final_it);
125 maps->initial.left.insert({original_first, q_n.second});
126 maps->final.left.insert({original_first, q_n.second});
127 }
128 }
129 // mapping_frontier tracks boundary between routed & un-routed in circuit
130 // when initialised, boundary is over output edges of input vertices
131 MappingFrontier_ptr mapping_frontier;
132 if (!maps->initial.empty() && !maps->final.empty()) {
133 mapping_frontier = std::make_shared<MappingFrontier>(circuit, maps);
134 } else {
135 mapping_frontier = std::make_shared<MappingFrontier>(circuit);
136 }
137 mapping_frontier->reassignable_nodes_ = reassignable_nodes;
138 // updates routed/un-routed boundary
139 mapping_frontier->advance_frontier_boundary(this->architecture_);
140
145 auto check_finish = [&mapping_frontier]() {
146 for (const std::pair<UnitID, VertPort>& pair :
147 mapping_frontier->linear_boundary->get<TagKey>()) {
148 Edge e = mapping_frontier->circuit_.get_nth_out_edge(
149 pair.second.first, pair.second.second);
150 Vertex v = mapping_frontier->circuit_.target(e);
151 OpType ot = mapping_frontier->circuit_.get_OpType_from_Vertex(v);
152 if (!is_final_q_type(ot) && ot != OpType::ClOutput) {
153 return false;
154 }
155 }
156 return true;
157 };
158
159 bool circuit_modified = !check_finish();
160 while (!check_finish()) {
161 // The order methods are passed in std::vector<RoutingMethod> is
162 // the order they are run
163 // If a method performs better but only on specific subcircuits,
164 // rank it earlier in the passed vector
165 bool valid_methods = false;
166 for (const auto& rm : routing_methods) {
167 // true => can use held routing method
168 std::pair<bool, unit_map_t> bool_map =
169 rm->routing_method(mapping_frontier, this->architecture_);
170 if (bool_map.first) {
171 valid_methods = true;
172 if (bool_map.second.size() > 0) {
173 std::map<Node, Node> node_map;
174 for (const auto& x : bool_map.second) {
175 node_map.insert({Node(x.first), Node(x.second)});
176 }
177 for (const std::pair<Node, Node>& swap :
178 BestTsaWithArch::get_swaps(*this->architecture_, node_map)) {
179 mapping_frontier->add_swap(swap.first, swap.second);
180 }
181 }
182 break;
183 }
184 }
185 if (!valid_methods) {
186 throw MappingManagerError(
187 "No RoutingMethod suitable to map given subcircuit.");
188 }
189 // find next routed/unrouted boundary given updates
190 mapping_frontier->advance_frontier_boundary(this->architecture_);
191 }
192 return (circuit_modified || modify_at_start);
193}
