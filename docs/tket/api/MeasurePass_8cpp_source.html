37bool check_only_end_measures(const Command& com, unit_set_t& measured_units) {
38 OpType optype = com.get_op_ptr()->get_type();
39 if (optype == OpType::Conditional) {
40 unit_vector_t all_args = com.get_args();
41 const Conditional& cond =
42 static_cast<const Conditional&>(*com.get_op_ptr());
43 unit_vector_t::iterator arg_it = all_args.begin();
44 for (unsigned i = 0; i < cond.get_width(); ++i) {
45 if (measured_units.find(*arg_it) != measured_units.end()) return false;
46 ++arg_it;
47 }
48 unit_vector_t new_args = {arg_it, all_args.end()};
49 Command new_com = {cond.get_op(), new_args};
50 return check_only_end_measures(new_com, measured_units);
51 } else if (optype == OpType::CircBox || optype == OpType::CustomGate) {
52 const Box& box = static_cast<const Box&>(*com.get_op_ptr());
53 unit_map_t interface;
54 unit_set_t inner_set;
55 unsigned q_count = 0;
56 unsigned b_count = 0;
57 for (const UnitID& u : com.get_args()) {
58 UnitID inner_unit = (u.type() == UnitType::Qubit)
59 ? static_cast<UnitID>(Qubit(q_count++))
60 : static_cast<UnitID>(Bit(b_count++));
61 interface.insert({inner_unit, u});
62 if (measured_units.find(u) != measured_units.end()) {
63 inner_set.insert(inner_unit);
64 }
65 }
66 for (const Command& c : *box.to_circuit()) {
67 if (!check_only_end_measures(c, inner_set)) return false;
68 }
69 for (const UnitID& u : inner_set) {
70 measured_units.insert(interface.at(u));
71 }
72 return true;
73 } else if (optype == OpType::Measure) {
74 std::pair<unit_set_t::iterator, bool> q_inserted =
75 measured_units.insert(com.get_args().at(0));
76 std::pair<unit_set_t::iterator, bool> c_inserted =
77 measured_units.insert(com.get_args().at(1));
78 return q_inserted.second && c_inserted.second;
79 } else {
80 for (const UnitID& a : com.get_args()) {
81 if (measured_units.find(a) != measured_units.end()) return false;
82 }
83 return true;
84 }
85}

112static std::optional<port_t> op_commutes(
113 Op_ptr op, port_t in_port, bool only_boxes) {
114 OpType optype = op->get_type();
115 if (optype == OpType::SWAP) {
116 if (only_boxes) return std::nullopt;
117 return 1 - in_port;
118 } else if (optype == OpType::Conditional) {
119 const Conditional& cond = static_cast<const Conditional&>(*op);
120 if (in_port < cond.get_width()) return std::nullopt;
121 std::optional<port_t> inner_port =
122 op_commutes(cond.get_op(), in_port - cond.get_width(), only_boxes);
123 // We can only commute through the conditional gate if it commutes to the
124 // same port for both values of the condition, e.g. we ban conditional swap
125 // gates
126 if (inner_port && (*inner_port + cond.get_width() == in_port))
127 return in_port;
128 else
129 return std::nullopt;
130 } else if (optype == OpType::CircBox || optype == OpType::CustomGate) {
131 const Box& box = static_cast<const Box&>(*op);
132 const Circuit& circ = *box.to_circuit();
133 Vertex inner_vertex = circ.all_inputs().at(in_port);
134 Edge inner_edge = circ.get_nth_out_edge(inner_vertex, 0);
135 inner_edge = follow_until_noncommuting(circ, inner_edge, only_boxes);
136 inner_vertex = circ.target(inner_edge);
137 if (!is_final_q_type(circ.get_OpType_from_Vertex(inner_vertex))) {
138 // The operation does not commute through the box
139 return std::nullopt;
140 }
141 auto outputs = circ.all_outputs();
142 port_t box_out_port = std::distance(
143 outputs.begin(),
144 std::find(outputs.begin(), outputs.end(), inner_vertex));
145 return box_out_port;
146 } else {
147 if (only_boxes) return std::nullopt;
148 if (op->commutes_with_basis(Pauli::Z, in_port)) return in_port;
149 return std::nullopt;
150 }
151}

153static Edge follow_until_noncommuting(
154 const Circuit& circ, Edge current_edge, bool only_boxes) {
155 port_t current_port = circ.get_target_port(current_edge);
156 Vertex current_vertex = circ.target(current_edge);
157 OpType current_optype = circ.get_OpType_from_Vertex(current_vertex);
158 while (!is_final_q_type(current_optype)) {
159 Op_ptr op = circ.get_Op_ptr_from_Vertex(current_vertex);
160 std::optional<port_t> next_port = op_commutes(op, current_port, only_boxes);
161 if (next_port.has_value()) {
162 current_port = next_port.value();
163 } else {
164 break;
165 }
166 // Proceed to successor
167 current_edge = circ.get_nth_out_edge(current_vertex, current_port);
168 current_vertex = circ.target(current_edge);
169 current_port = circ.get_target_port(current_edge);
170 current_optype = circ.get_OpType_from_Vertex(current_vertex);
171 }
172 return current_edge;
173}

175std::pair<bool, bool> run_delay_measures(
176 Circuit& circ, bool allow_partial, bool dry_run) {
177 if (allow_partial && dry_run) {
178 // All circuits are valid in a partial run
179 return {false, true};
180 }
181 // Collect the vertices to swap to the end of the circuit, along with their
182 // target edges
183 std::vector<std::pair<Vertex, Edge>> to_delay;
184 for (const Command& com : circ) {
185 Vertex v = com.get_vertex();
186 OpType optype = com.get_op_ptr()->get_type();
187 if (optype == OpType::Measure) {
188 Edge c_out_edge = circ.get_nth_out_edge(v, 1);
189 bool measurement_is_final = circ.detect_final_Op(circ.target(c_out_edge));
190 if (allow_partial) {
191 if (!measurement_is_final) continue;
192 } else {
193 if (!measurement_is_final ||
194 circ.n_out_edges_of_type(v, EdgeType::Boolean) != 0) {
195 if (dry_run) return {false, false};
196 throw CircuitInvalidity(
197 "Cannot commute Measure through classical operations to the end "
198 "of the circuit");
199 }
200 }
201 Edge out_edge = circ.get_nth_out_edge(v, 0);
202 Edge current_edge = follow_until_noncommuting(circ, out_edge, false);
203 Vertex current_vertex = circ.target(current_edge);
204 // If we haven't reached an output, we can't continue
205 if (!allow_partial &&
206 !is_final_q_type(circ.get_OpType_from_Vertex(current_vertex))) {
207 if (dry_run) return {false, false};
208 throw CircuitInvalidity(
209 "Cannot commute Measure through quantum gates to the end of the "
210 "circuit");
211 }
212 if (dry_run) continue;
213 // If the measure was already at the output, nothing to do
214 if (current_edge == out_edge) continue;
215 to_delay.push_back({v, current_edge});
216 } else {
217 // Measurements inside boxes are never modified, so there is no need to
218 // check them during a partial run.
219 if (allow_partial) continue;
220 // Raise an error if there are any boxes or conditionals with internal
221 // mid-measures.
222 unit_set_t measured_units;
223 if (!check_only_end_measures(com, measured_units)) {
224 if (dry_run) return {false, false};
225 throw optype == OpType::Conditional
226 ? CircuitInvalidity("Cannot delay measures inside a conditional")
227 : CircuitInvalidity("Cannot delay measures inside a circuit box");
228 }
229 // If the command has internal end measurements we allow them only if they
230 // are at the end of the full circuit. We can't delay the measures inside
231 // boxes and conditionals.
232 for (const auto& unit : measured_units) {
233 if (unit.type() == UnitType::Bit) continue;
234 auto args = com.get_args();
235 port_t out_port = std::distance(
236 args.begin(), std::find(args.begin(), args.end(), unit));
237 Edge current_edge = circ.get_nth_out_edge(v, out_port);
238 // Follow the edge through boxes
239 current_edge = follow_until_noncommuting(circ, current_edge, true);
240 Vertex current_vertex = circ.target(current_edge);
241 optype = circ.get_OpType_from_Vertex(current_vertex);
242 if (!is_final_q_type(optype)) {
243 if (dry_run) return {false, false};
244 throw optype == OpType::Conditional
245 ? CircuitInvalidity("Cannot delay measures inside a conditional")
246 : CircuitInvalidity("Cannot delay measures inside a circuit box");
247 }
248 }
249 }
250 }
251
252 if (to_delay.empty()) {
253 return {false, true};
254 }
255 // Swap all the measures to the end
256 for (auto& p : to_delay) {
257 Vertex v = p.first;
258 Edge current_edge = p.second;
259 Vertex current_vertex = circ.target(current_edge);
260
261 Edge in_edge = circ.get_nth_in_edge(v, 0);
262 Edge out_edge = circ.get_nth_out_edge(v, 0);
263 // Rewire measure
264 circ.add_edge(
265 {circ.source(in_edge), circ.get_source_port(in_edge)},
266 {circ.target(out_edge), circ.get_target_port(out_edge)},
267 EdgeType::Quantum);
268 circ.remove_edge(in_edge);
269 circ.remove_edge(out_edge);
270 circ.add_edge(
271 {circ.source(current_edge), circ.get_source_port(current_edge)}, {v, 0},
272 EdgeType::Quantum);
273 circ.add_edge({v, 0}, {current_vertex, 0}, EdgeType::Quantum);
274 circ.remove_edge(current_edge);
275 }
276 return {true, true};
277}
