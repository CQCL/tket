39 bool check_only_end_measures(const Command& com, unit_set_t& measured_units) {
40  OpType optype = com.get_op_ptr()->get_type();
41  if (optype == OpType::Conditional) {
42  unit_vector_t all_args = com.get_args();
43  const Conditional& cond =
44  static_cast<const Conditional&>(*com.get_op_ptr());
45  unit_vector_t::iterator arg_it = all_args.begin();
46  for (unsigned i = 0; i < cond.get_width(); ++i) {
47  if (measured_units.find(*arg_it) != measured_units.end()) return false;
48  ++arg_it;
49  }
50  unit_vector_t new_args = {arg_it, all_args.end()};
51  Command new_com = {cond.get_op(), new_args};
52  return check_only_end_measures(new_com, measured_units);
53  } else if (optype == OpType::CircBox || optype == OpType::CustomGate) {
54  const Box& box = static_cast<const Box&>(*com.get_op_ptr());
55  unit_map_t interface;
56  unit_set_t inner_set;
57  unsigned q_count = 0;
58  unsigned b_count = 0;
59  for (const UnitID& u : com.get_args()) {
60  UnitID inner_unit = (u.type() == UnitType::Qubit)
61  ? static_cast<UnitID>(Qubit(q_count++))
62  : static_cast<UnitID>(Bit(b_count++));
63  interface.insert({inner_unit, u});
64  if (measured_units.find(u) != measured_units.end()) {
65  inner_set.insert(inner_unit);
66  }
67  }
68  for (const Command& c : *box.to_circuit()) {
69  if (!check_only_end_measures(c, inner_set)) return false;
70  }
71  for (const UnitID& u : inner_set) {
72  measured_units.insert(interface.at(u));
73  }
74  return true;
75  } else if (optype == OpType::Measure) {
76  std::pair<unit_set_t::iterator, bool> q_inserted =
77  measured_units.insert(com.get_args().at(0));
78  std::pair<unit_set_t::iterator, bool> c_inserted =
79  measured_units.insert(com.get_args().at(1));
80  return q_inserted.second && c_inserted.second;
81  } else {
82  for (const UnitID& a : com.get_args()) {
83  if (measured_units.find(a) != measured_units.end()) return false;
84  }
85  return true;
86  }
87 }

115 static std::optional<port_t> op_commutes(
116  const Circuit& circ, Op_ptr op, port_t in_port, bool only_boxes) {
117  OpType optype = op->get_type();
118  if (optype == OpType::SWAP) {
119  if (only_boxes) return std::nullopt;
120  return 1 - in_port;
121  } else if (optype == OpType::Conditional) {
122  const Conditional& cond = static_cast<const Conditional&>(*op);
123  if (in_port < cond.get_width()) return std::nullopt;
124  return op_commutes(
125  circ, cond.get_op(), in_port - cond.get_width(), only_boxes);
126  } else if (optype == OpType::CircBox || optype == OpType::CustomGate) {
127  const Box& box = static_cast<const Box&>(*op);
128  const Circuit& circ = *box.to_circuit();
129  Vertex inner_vertex = circ.all_inputs().at(in_port);
130  Edge inner_edge = circ.get_nth_out_edge(inner_vertex, 0);
131  inner_edge = follow_until_noncommuting(circ, inner_edge, only_boxes);
132  inner_vertex = circ.target(inner_edge);
133  if (!is_final_q_type(circ.get_OpType_from_Vertex(inner_vertex))) {
134  // The operation does not commute through the box
135  return std::nullopt;
136  }
137  auto outputs = circ.all_outputs();
138  port_t box_out_port = std::distance(
139  outputs.begin(),
140  std::find(outputs.begin(), outputs.end(), inner_vertex));
141  return box_out_port;
142  } else {
143  if (only_boxes) return std::nullopt;
144  unsigned qubit_index = 0;
145  for (unsigned i = 0; i < in_port; ++i) {
146  EdgeType et = op->get_signature()[i];
147  if (et == EdgeType::Quantum) {
148  qubit_index++;
149  }
150  }
151  if (op->commutes_with_basis(Pauli::Z, qubit_index)) {
152  return in_port;
153  }
154  return std::nullopt;
155  }
156 }

158 static Edge follow_until_noncommuting(
159  const Circuit& circ, Edge current_edge, bool only_boxes) {
160  port_t current_port = circ.get_target_port(current_edge);
161  Vertex current_vertex = circ.target(current_edge);
162  OpType current_optype = circ.get_OpType_from_Vertex(current_vertex);
163  while (!is_final_q_type(current_optype)) {
164  Op_ptr op = circ.get_Op_ptr_from_Vertex(current_vertex);
165  std::optional<port_t> next_port =
166  op_commutes(circ, op, current_port, only_boxes);
167  if (next_port.has_value()) {
168  current_port = next_port.value();
169  } else {
170  break;
171  }
172  // Proceed to successor
173  current_edge = circ.get_nth_out_edge(current_vertex, current_port);
174  current_vertex = circ.target(current_edge);
175  current_port = circ.get_target_port(current_edge);
176  current_optype = circ.get_OpType_from_Vertex(current_vertex);
177  }
178  return current_edge;
179 }

181 std::pair<bool, bool> run_delay_measures(
182  Circuit& circ, bool allow_partial, bool dry_run) {
183  if (allow_partial && dry_run) {
184  // All circuits are valid in a partial run
185  return {false, true};
186  }
187  // Collect the vertices to swap to the end of the circuit, along with their
188  // target edges
189  std::vector<std::pair<Vertex, Edge>> to_delay;
190  for (const Command& com : circ) {
191  Vertex v = com.get_vertex();
192  OpType optype = com.get_op_ptr()->get_type();
193  if (optype == OpType::Measure) {
194  if (!allow_partial) {
195  Edge c_out_edge = circ.get_nth_out_edge(v, 1);
196  if (!circ.detect_final_Op(circ.target(c_out_edge)) ||
197  circ.n_out_edges_of_type(v, EdgeType::Boolean) != 0) {
198  if (dry_run) return {false, false};
199  throw CircuitInvalidity(
200  "Cannot commute Measure through classical operations to the end "
201  "of the circuit");
202  }
203  }
204  Edge out_edge = circ.get_nth_out_edge(v, 0);
205  Edge current_edge = follow_until_noncommuting(circ, out_edge, false);
206  Vertex current_vertex = circ.target(current_edge);
207  // If we haven't reached an output, we can't continue
208  if (!allow_partial &&
209  !is_final_q_type(circ.get_OpType_from_Vertex(current_vertex))) {
210  if (dry_run) return {false, false};
211  throw CircuitInvalidity(
212  "Cannot commute Measure through quantum gates to the end of the "
213  "circuit");
214  }
215  if (dry_run) continue;
216  // If the measure was already at the output, nothing to do
217  if (current_edge == out_edge) continue;
218  to_delay.push_back({v, current_edge});
219  } else {
220  // Measurements inside boxes are never modified, so there is no need to
221  // check them during a partial run.
222  if (allow_partial) continue;
223  // Raise an error if there are any boxes or conditionals with internal
224  // mid-measures.
225  unit_set_t measured_units;
226  if (!check_only_end_measures(com, measured_units)) {
227  if (dry_run) return {false, false};
228  throw optype == OpType::Conditional
229  ? CircuitInvalidity("Cannot delay measures inside a conditional")
230  : CircuitInvalidity("Cannot delay measures inside a circuit box");
231  }
232  // If the command has internal end measurements we allow them only if they
233  // are at the end of the full circuit. We can't delay the measures inside
234  // boxes and conditionals.
235  for (const auto& unit : measured_units) {
236  if (unit.type() == UnitType::Bit) continue;
237  auto args = com.get_args();
238  port_t out_port = std::distance(
239  args.begin(), std::find(args.begin(), args.end(), unit));
240  Edge current_edge = circ.get_nth_out_edge(v, out_port);
241  // Follow the edge through boxes
242  current_edge = follow_until_noncommuting(circ, current_edge, true);
243  Vertex current_vertex = circ.target(current_edge);
244  optype = circ.get_OpType_from_Vertex(current_vertex);
245  if (!is_final_q_type(optype)) {
246  if (dry_run) return {false, false};
247  throw optype == OpType::Conditional
248  ? CircuitInvalidity("Cannot delay measures inside a conditional")
249  : CircuitInvalidity("Cannot delay measures inside a circuit box");
250  }
251  }
252  }
253  }
254 
255  if (to_delay.empty()) {
256  return {false, true};
257  }
258  // Swap all the measures to the end
259  for (auto& p : to_delay) {
260  Vertex v = p.first;
261  Edge current_edge = p.second;
262  Vertex current_vertex = circ.target(current_edge);
263 
264  Edge in_edge = circ.get_nth_in_edge(v, 0);
265  Edge out_edge = circ.get_nth_out_edge(v, 0);
266  // Rewire measure
267  circ.add_edge(
268  {circ.source(in_edge), circ.get_source_port(in_edge)},
269  {circ.target(out_edge), circ.get_target_port(out_edge)},
270  EdgeType::Quantum);
271  circ.remove_edge(in_edge);
272  circ.remove_edge(out_edge);
273  circ.add_edge(
274  {circ.source(current_edge), circ.get_source_port(current_edge)}, {v, 0},
275  EdgeType::Quantum);
276  circ.add_edge({v, 0}, {current_vertex, 0}, EdgeType::Quantum);
277  circ.remove_edge(current_edge);
278  }
279  return {true, true};
280 }
