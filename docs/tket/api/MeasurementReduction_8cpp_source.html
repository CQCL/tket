21MeasurementSetup measurement_reduction(
22 const std::list<SpPauliString>& strings, PauliPartitionStrat strat,
23 GraphColourMethod method, CXConfigType cx_config) {
24 std::set<Qubit> qubits;
25 for (const SpPauliString& qpt : strings) {
26 for (const std::pair<const Qubit, Pauli>& qb_p : qpt.string)
27 qubits.insert(qb_p.first);
28 }
29
30 std::map<Qubit, unsigned> qb_location_map;
31 unsigned u = 0;
32 for (const Qubit& qb : qubits) {
33 qb_location_map[qb] = u;
34 ++u;
35 }
36
37 std::list<std::list<SpPauliString>> all_terms =
38 term_sequence(strings, strat, method);
39 MeasurementSetup ms;
40 unsigned i = 0;
41 for (const std::list<SpPauliString>& terms : all_terms) {
42 std::list<SpSymPauliTensor> gadgets;
43 for (const SpPauliString& string : terms) {
44 gadgets.push_back(string);
45 }
46
47 std::set<Qubit> mutable_qb_set(qubits);
48 Circuit cliff_circ = mutual_diagonalise(gadgets, mutable_qb_set, cx_config);
49 unsigned bit_count = 0;
50 for (const Qubit& qb : cliff_circ.all_qubits()) {
51 cliff_circ.add_bit(Bit(bit_count));
52 cliff_circ.add_measure(qb, Bit(bit_count));
53 ++bit_count;
54 }
55 ms.add_measurement_circuit(cliff_circ);
56
57 std::list<SpSymPauliTensor>::const_iterator gadgets_iter = gadgets.begin();
58 for (const SpPauliString& string : terms) {
59 SpPauliStabiliser stab(*gadgets_iter); // Force coeff to be real
60 std::vector<unsigned> bits;
61 for (const std::pair<const Qubit, Pauli>& qp_pair : stab.string) {
62 if (qp_pair.second == Pauli::Z)
63 bits.push_back(qb_location_map.at(qp_pair.first));
64 }
65 ms.add_result_for_term(string, {i, bits, stab.is_real_negative()});
66 ++gadgets_iter;
67 }
68 ++i;
69 }
70
71 return ms;
72}
