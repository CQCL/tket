19 MeasurementSetup measurement_reduction(
20  const std::list<SpPauliString>& strings, PauliPartitionStrat strat,
21  GraphColourMethod method, CXConfigType cx_config) {
22  std::set<Qubit> qubits;
23  for (const SpPauliString& qpt : strings) {
24  for (const std::pair<const Qubit, Pauli>& qb_p : qpt.string)
25  qubits.insert(qb_p.first);
26  }
27 
28  std::map<Qubit, unsigned> qb_location_map;
29  unsigned u = 0;
30  for (const Qubit& qb : qubits) {
31  qb_location_map[qb] = u;
32  ++u;
33  }
34 
35  std::list<std::list<SpPauliString>> all_terms =
36  term_sequence(strings, strat, method);
37  MeasurementSetup ms;
38  unsigned i = 0;
39  for (const std::list<SpPauliString>& terms : all_terms) {
40  std::list<SpSymPauliTensor> gadgets;
41  for (const SpPauliString& string : terms) {
42  gadgets.push_back(string);
43  }
44 
45  std::set<Qubit> mutable_qb_set(qubits);
46  Circuit cliff_circ = mutual_diagonalise(gadgets, mutable_qb_set, cx_config);
47  unsigned bit_count = 0;
48  for (const Qubit& qb : cliff_circ.all_qubits()) {
49  cliff_circ.add_bit(Bit(bit_count));
50  cliff_circ.add_measure(qb, Bit(bit_count));
51  ++bit_count;
52  }
53  ms.add_measurement_circuit(cliff_circ);
54 
55  std::list<SpSymPauliTensor>::const_iterator gadgets_iter = gadgets.begin();
56  for (const SpPauliString& string : terms) {
57  SpPauliStabiliser stab(*gadgets_iter); // Force coeff to be real
58  std::vector<unsigned> bits;
59  for (const std::pair<const Qubit, Pauli>& qp_pair : stab.string) {
60  if (qp_pair.second == Pauli::Z)
61  bits.push_back(qb_location_map.at(qp_pair.first));
62  }
63  ms.add_result_for_term(string, {i, bits, stab.is_real_negative()});
64  ++gadgets_iter;
65  }
66  ++i;
67  }
68 
69  return ms;
70 }
