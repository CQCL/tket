19 MeasurementSetup measurement_reduction(
20  const std::list<QubitPauliString>& strings, PauliPartitionStrat strat,
21  GraphColourMethod method, CXConfigType cx_config) {
22  std::set<Qubit> qubits;
23  for (const QubitPauliString& qpt : strings) {
24  for (const std::pair<const Qubit, Pauli>& qb_p : qpt.map)
25  qubits.insert(qb_p.first);
26  }
27 
28  std::map<Qubit, unsigned> qb_location_map;
29  unsigned u = 0;
30  for (const Qubit& qb : qubits) {
31  qb_location_map[qb] = u;
32  ++u;
33  }
34 
35  std::list<std::list<QubitPauliString>> all_terms =
36  term_sequence(strings, strat, method);
37  MeasurementSetup ms;
38  unsigned i = 0;
39  for (const std::list<QubitPauliString>& terms : all_terms) {
40  std::list<std::pair<QubitPauliTensor, Expr>> gadgets;
41  for (const QubitPauliString& string : terms) {
42  QubitPauliTensor qps(string);
43  gadgets.push_back({qps, 1.});
44  }
45 
46  std::set<Qubit> mutable_qb_set(qubits);
47  Circuit cliff_circ = mutual_diagonalise(gadgets, mutable_qb_set, cx_config);
48  unsigned bit_count = 0;
49  for (const Qubit& qb : cliff_circ.all_qubits()) {
50  cliff_circ.add_bit(Bit(bit_count));
51  cliff_circ.add_measure(qb, Bit(bit_count));
52  ++bit_count;
53  }
54  ms.add_measurement_circuit(cliff_circ);
55 
56  std::list<std::pair<QubitPauliTensor, Expr>>::const_iterator gadgets_iter =
57  gadgets.begin();
58  for (const QubitPauliString& string : terms) {
59  const std::pair<QubitPauliTensor, Expr>& new_gadget = *gadgets_iter;
60  std::vector<unsigned> bits;
61  for (const std::pair<const Qubit, Pauli>& qp_pair :
62  new_gadget.first.string.map) {
63  if (qp_pair.second == Pauli::Z)
64  bits.push_back(qb_location_map.at(qp_pair.first));
65  }
66  bool invert = (std::abs(new_gadget.first.coeff + Complex(1)) < EPS);
67  ms.add_result_for_term(string, {i, bits, invert});
68  ++gadgets_iter;
69  }
70  ++i;
71  }
72 
73  return ms;
74 }
