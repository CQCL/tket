52 bool MeasurementSetup::verify() const {
53  std::map<std::pair<unsigned, unsigned>, QubitPauliTensor> pauli_map;
54  // Identify Paulis measured onto each bit
55  for (unsigned circ_id = 0; circ_id < measurement_circs.size(); ++circ_id) {
56  Circuit circ = measurement_circs[circ_id];
57  std::map<Qubit, unsigned> readout = circ.qubit_readout();
58  // Remove Measures from circuit to allow CliffordTableau generation
59  for (const Vertex &out : circ.q_outputs()) {
60  Vertex pred = circ.get_predecessors(out).front();
61  if (circ.get_OpType_from_Vertex(pred) == OpType::Measure) {
62  circ.remove_vertex(
63  pred, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
64  }
65  }
66  CliffTableau tab = circuit_to_tableau(circ);
67  for (const Qubit &qb : tab.get_qubits()) {
68  pauli_map.insert({{circ_id, readout[qb]}, tab.get_zpauli(qb)});
69  }
70  }
71  for (const std::pair<const QubitPauliString, std::vector<MeasurementBitMap>>
72  &term : result_map) {
73  for (const MeasurementBitMap &bits : term.second) {
74  QubitPauliTensor total;
75  for (unsigned bit : bits.bits) {
76  total = total * pauli_map[{bits.circ_index, bit}];
77  }
78  if (bits.invert) total.coeff *= -1.;
79  QubitPauliTensor term_tensor(term.first);
80  if (total != term_tensor) {
81  std::stringstream out;
82  out << "Invalid MeasurementSetup: expecting to measure "
83  << term_tensor.to_str() << "; actually measured " << total.to_str();
84  tket_log()->error(out.str());
85  return false;
86  }
87  }
88  }
89  return true;
90 }

124 void to_json(nlohmann::json &j, const MeasurementSetup &setup) {
125  std::vector<std::pair<
126  QubitPauliString, std::vector<MeasurementSetup::MeasurementBitMap>>>
127  map_list;
128  for (const std::pair<
129  const QubitPauliString,
130  std::vector<MeasurementSetup::MeasurementBitMap>> &tensor_map :
131  setup.get_result_map()) {
132  map_list.push_back(tensor_map);
133  }
134  // sort the list for consistent serialisation
135  std::sort(map_list.begin(), map_list.end(), [](auto pair1, auto pair2) {
136  return pair1.first < pair2.first;
137  });
138  j["result_map"] = map_list;
139  j["circs"] = setup.get_circs();
140 }
