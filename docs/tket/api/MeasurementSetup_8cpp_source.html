47 bool MeasurementSetup::verify() const {
48  std::map<std::pair<unsigned, unsigned>, SpPauliStabiliser> pauli_map;
49  // Identify Paulis measured onto each bit
50  for (unsigned circ_id = 0; circ_id < measurement_circs.size(); ++circ_id) {
51  Circuit circ = measurement_circs[circ_id];
52  std::map<Qubit, unsigned> readout = circ.qubit_readout();
53  // Remove Measures from circuit to allow CliffordTableau generation
54  for (const Vertex &out : circ.q_outputs()) {
55  Vertex pred = circ.get_predecessors(out).front();
56  if (circ.get_OpType_from_Vertex(pred) == OpType::Measure) {
57  circ.remove_vertex(
58  pred, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
59  }
60  }
61  UnitaryRevTableau tab = circuit_to_unitary_rev_tableau(circ);
62  for (const Qubit &qb : tab.get_qubits()) {
63  pauli_map.insert({{circ_id, readout[qb]}, tab.get_zrow(qb)});
64  }
65  }
66  for (const std::pair<const SpPauliString, std::vector<MeasurementBitMap>>
67  &term : result_map) {
68  for (const MeasurementBitMap &bits : term.second) {
69  SpPauliStabiliser total;
70  for (unsigned bit : bits.bits) {
71  total = total * pauli_map[{bits.circ_index, bit}];
72  }
73  if (bits.invert) total.coeff = (total.coeff + 2) % 4;
74  SpPauliStabiliser term_tensor(term.first);
75  if (total != term_tensor) {
76  std::stringstream out;
77  out << "Invalid MeasurementSetup: expecting to measure "
78  << term_tensor.to_str() << "; actually measured " << total.to_str();
79  tket_log()->error(out.str());
80  return false;
81  }
82  }
83  }
84  return true;
85 }

119 void to_json(nlohmann::json &j, const MeasurementSetup &setup) {
120  std::vector<std::pair<
121  SpPauliString, std::vector<MeasurementSetup::MeasurementBitMap>>>
122  map_list;
123  for (const std::pair<
124  const SpPauliString,
125  std::vector<MeasurementSetup::MeasurementBitMap>> &tensor_map :
126  setup.get_result_map()) {
127  map_list.push_back(tensor_map);
128  }
129  // sort the list for consistent serialisation
130  std::sort(map_list.begin(), map_list.end(), [](auto pair1, auto pair2) {
131  return pair1.first < pair2.first;
132  });
133  std::vector<std::pair<
134  QubitPauliMap, std::vector<MeasurementSetup::MeasurementBitMap>>>
135  map_encoding;
136  for (const std::pair<
137  SpPauliString, std::vector<MeasurementSetup::MeasurementBitMap>>
138  &tensor_map : map_list) {
139  map_encoding.push_back({tensor_map.first.string, tensor_map.second});
140  }
141  // Convert SpPauliString to QubitPauliMap for backwards compatibility with
142  // before templated PauliTensor
143  j["result_map"] = map_encoding;
144  j["circs"] = setup.get_circs();
145 }
