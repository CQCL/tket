30// Where should isolated pattern vertices be assigned?
31// They might NOT have been isolated originally; it may be
32// that we deliberately erased some pattern edges.
33// Thus, we still want them connected to useful target components,
34// so assign to nonisolated target vertices first.
35static void assign_isolated_pattern_vertices(
36 boost::bimap<Qubit, Node>& map,
37 const RelabelledPatternGraph& relabelled_pattern_graph,
38 const RelabelledTargetGraph& relabelled_target_graph) {
39 if (relabelled_pattern_graph.get_relabelled_isolated_vertices().empty()) {
40 return;
41 }
42 // The PV assigned so far must be exactly the nonisolated ones.
43 TKET_ASSERT(
44 map.size() ==
45 relabelled_pattern_graph.get_relabelled_nonisolated_vertices().size());
46
47 // Also, all PV so far must have been assigned to nonisolated TV.
48 TKET_ASSERT(
49 map.size() <=
50 relabelled_target_graph.get_relabelled_nonisolated_vertices().size());
51
52 std::set<VertexWSM> unused_target_vertices;
53 bool refilled_with_isolated_tv;
54 if (map.size() <
55 relabelled_target_graph.get_relabelled_nonisolated_vertices().size()) {
56 refilled_with_isolated_tv = false;
57 unused_target_vertices =
58 relabelled_target_graph.get_relabelled_nonisolated_vertices();
59 for (const auto& entry : map.left) {
60 // We CANNOT have assigned a nonisolated pattern vertex
61 // to an isolated target vertex.
62 const VertexWSM new_pv =
63 relabelled_pattern_graph.get_relabelled_vertex(entry.first);
64 const VertexWSM new_tv =
65 relabelled_target_graph.get_relabelled_vertex(entry.second);
66 TKET_ASSERT(
67 relabelled_pattern_graph.get_relabelled_nonisolated_vertices().count(
68 new_pv) != 0);
69 TKET_ASSERT(
70 relabelled_target_graph.get_relabelled_nonisolated_vertices().count(
71 new_tv) != 0);
72 TKET_ASSERT(unused_target_vertices.erase(new_tv) == 1);
73 }
74 } else {
75 refilled_with_isolated_tv = true;
76 unused_target_vertices =
77 relabelled_target_graph.get_relabelled_isolated_vertices();
78 }
79 for (VertexWSM isolated_pv :
80 relabelled_pattern_graph.get_relabelled_isolated_vertices()) {
81 if (unused_target_vertices.empty()) {
82 // We cannot run out of vertices.
83 TKET_ASSERT(!refilled_with_isolated_tv);
84 refilled_with_isolated_tv = true;
85 unused_target_vertices =
86 relabelled_target_graph.get_relabelled_isolated_vertices();
87 TKET_ASSERT(!unused_target_vertices.empty());
88 }
89 const VertexWSM next_tv = *unused_target_vertices.cbegin();
90 unused_target_vertices.erase(next_tv);
91
92 map.insert(BimapValue(
93 relabelled_pattern_graph.get_original_vertices().at(isolated_pv),
94 relabelled_target_graph.get_original_vertices().at(next_tv)));
95 }
96}

98static void write_solver_solutions(
99 std::vector<boost::bimap<Qubit, Node>>& all_maps,
100 const std::vector<SolutionWSM>& solutions,
101 const RelabelledPatternGraph& relabelled_pattern_graph,
102 const RelabelledTargetGraph& relabelled_target_graph, bool return_best) {
103 TKET_ASSERT(all_maps.empty());
104 std::vector<unsigned> reordering = {};
105
106 for (unsigned ii = 0; ii < solutions.size(); ++ii) {
107 const auto& solution = solutions[ii];
108 boost::bimap<Qubit, Node> map;
109 for (const auto& relabelled_pv_tv : solution.assignments) {
110 map.insert(BimapValue(
111 relabelled_pattern_graph.get_original_vertices().at(
112 relabelled_pv_tv.first),
113 relabelled_target_graph.get_original_vertices().at(
114 relabelled_pv_tv.second)));
115 }
116 assign_isolated_pattern_vertices(
117 map, relabelled_pattern_graph, relabelled_target_graph);
118 TKET_ASSERT(
119 map.size() == relabelled_pattern_graph.get_original_vertices().size());
120 unsigned original_size = reordering.size();
121 for (unsigned i = 0;
122 i < reordering.size() && reordering.size() == original_size; i++) {
123 TKET_ASSERT(all_maps.size() == reordering.size());
124 if (solution.scalar_product > reordering[i]) {
125 TKET_ASSERT(reordering.size() >= i);
126 TKET_ASSERT(all_maps.size() >= i);
127 auto it = reordering.begin() + i;
128 reordering.insert(it, solution.scalar_product);
129 auto jt = all_maps.begin() + i;
130 all_maps.insert(jt, map);
131 }
132 }
133 // implies that solution.scalar_product was smallest so far, so add to end
134 if (original_size == reordering.size()) {
135 reordering.push_back(solution.scalar_product);
136 all_maps.push_back(map);
137 }
138 }
139 // in some cases we only want maps which are costed best and identically
140 if (return_best && !reordering.empty()) {
141 TKET_ASSERT(reordering.size() == all_maps.size());
142 WeightWSM best = reordering[0];
143 for (unsigned i = 0; i < reordering.size(); i++) {
144 if (reordering[i] != best) {
145 all_maps.resize(i);
146 break;
147 }
148 }
149 }
150}
155std::vector<boost::bimap<Qubit, Node>> get_weighted_subgraph_monomorphisms(
156 QubitGraph::UndirectedConnGraph& pattern_graph,
157 Architecture::UndirectedConnGraph& target_graph, unsigned max_matches,
158 unsigned timeout_ms, bool return_best) {
159 std::vector<boost::bimap<Qubit, Node>> all_maps;
160
161 const RelabelledPatternGraph relabelled_pattern_graph(pattern_graph);
162 const RelabelledTargetGraph relabelled_target_graph(target_graph);
163
164 if (relabelled_pattern_graph.get_relabelled_edges_and_weights().size() >
165 relabelled_target_graph.get_relabelled_edges_and_weights().size() ||
166 relabelled_pattern_graph.get_relabelled_nonisolated_vertices().size() >
167 relabelled_target_graph.get_relabelled_nonisolated_vertices()
168 .size() ||
169
170 relabelled_pattern_graph.get_relabelled_isolated_vertices().size() +
171 relabelled_pattern_graph.get_relabelled_nonisolated_vertices()
172 .size() >
173 relabelled_target_graph.get_relabelled_isolated_vertices().size() +
174 relabelled_target_graph.get_relabelled_nonisolated_vertices()
175 .size()) {
176 // The problem is trivially insoluble.
177 return all_maps;
178 }
179
180 if (relabelled_pattern_graph.get_relabelled_nonisolated_vertices().empty()) {
181 // Trivial special case: all pattern vertices are isolated!
182 all_maps.resize(1);
183 assign_isolated_pattern_vertices(
184 all_maps[0], relabelled_pattern_graph, relabelled_target_graph);
185 return all_maps;
186 }
187
188 MainSolverParameters solver_parameters;
189 solver_parameters.terminate_with_first_full_solution = false;
190 solver_parameters.for_multiple_full_solutions_the_max_number_to_obtain =
191 max_matches;
192 solver_parameters.timeout_ms = timeout_ms;
193 const MainSolver main_solver(
194 relabelled_pattern_graph.get_relabelled_edges_and_weights(),
195 relabelled_target_graph.get_relabelled_edges_and_weights(),
196 solver_parameters);
197 const auto& solution_data = main_solver.get_solution_data();
198 write_solver_solutions(
199 all_maps, solution_data.solutions, relabelled_pattern_graph,
200 relabelled_target_graph, return_best);
201 return all_maps;
202}
