32 // Traverse the DAG to the quantum frontier
33 // to find the UnitID associated with an VertPort
34 static UnitID get_unitid_from_vertex_port(
35  const MappingFrontier_ptr &frontier, const VertPort &vert_port) {
36  VertPort current_vert_port = vert_port;
37  while (true) {
38  auto it =
39  frontier->linear_boundary->get<TagValue>().find(current_vert_port);
40  if (it != frontier->linear_boundary->get<TagValue>().end()) {
41  return it->first;
42  }
43  Edge current_e = frontier->circuit_.get_nth_out_edge(
44  current_vert_port.first, current_vert_port.second);
45  Vertex prev_vert;
46  Edge prev_e;
47  std::tie(prev_vert, prev_e) =
48  frontier->circuit_.get_prev_pair(current_vert_port.first, current_e);
49  current_vert_port = {prev_vert, frontier->circuit_.get_source_port(prev_e)};
50  }
51 }

53 static bool is_multiq_quantum_gate(const Circuit &circ, const Vertex &vert) {
54  Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
55  return (
56  op->get_desc().is_gate() && circ.n_in_edges(vert) > 1 &&
57  circ.n_in_edges_of_type(vert, EdgeType::Quantum) ==
58  circ.n_in_edges(vert) &&
59  circ.n_out_edges_of_type(vert, EdgeType::Quantum) ==
60  circ.n_out_edges(vert));
61 }

63 // This method will try to commute a vertex to the quantum frontier
64 // If successful, returns the current in-edges and the target in-edges to
65 // rewire the vertex
66 static std::optional<std::pair<EdgeVec, EdgeVec>> try_find_commute_edges(
67  const Circuit &circ, const EdgeVec &frontier_edges, const Vertex &vert) {
68  // Initialize to be the in_edges for the given vertex
69  EdgeVec current_edges = circ.get_in_edges(vert);
70  EdgeVec initial_edges(current_edges.begin(), current_edges.end());
71 
72  // Record the colour of each port of the vertex.
73  std::vector<std::optional<Pauli>> colours;
74  for (const Edge &edge : current_edges) {
75  port_t target_port = circ.get_target_port(edge);
76  std::optional<Pauli> colour =
77  circ.commuting_basis(vert, PortType::Target, target_port);
78  colours.push_back(colour);
79  }
80  // Stores all edges which the vertex can be commuted to
81  EdgeVec dest_edges;
82  while (true) {
83  // The vertex can be commuted to the front
84  bool success = true;
85  for (unsigned i = 0; i < current_edges.size(); ++i) {
86  // Check if the edge is already in the quantum frontier
87  if (std::find(
88  frontier_edges.begin(), frontier_edges.end(), current_edges[i]) !=
89  frontier_edges.end()) {
90  dest_edges.push_back(current_edges[i]);
91  continue;
92  }
93  // Check prev_op is a gate
94  Vertex prev_vert = circ.source(current_edges[i]);
95  Op_ptr prev_op = circ.get_Op_ptr_from_Vertex(prev_vert);
96  if (!prev_op->get_desc().is_gate()) {
97  // not commute
98  return std::nullopt;
99  }
100 
101  // Check commute
102  port_t source_port = circ.get_source_port(current_edges[i]);
103  if (!circ.commutes_with_basis(
104  prev_vert, colours[i], PortType::Source, source_port)) {
105  // not commute
106  return std::nullopt;
107  } else {
108  // Update dest_edges
109  Vertex prev_prev_v;
110  Edge prev_e;
111  std::tie(prev_prev_v, prev_e) =
112  circ.get_prev_pair(prev_vert, current_edges[i]);
113  dest_edges.push_back(prev_e);
114  }
115  // Only true if all edges are in frontier
116  success = false;
117  }
118  if (success) {
119  std::pair<EdgeVec, EdgeVec> p(initial_edges, dest_edges);
120  return p;
121  } else {
122  current_edges = dest_edges;
123  dest_edges = {};
124  }
125  }
126 }

128 static void partial_rewire(
129  const Vertex &vert, Circuit &circ, EdgeVec &src_edges,
130  EdgeVec &dest_edges) {
131  // move the vertex to the frontier
132  // Notice that if one of the vertex's in edge is already a destination
133  // edge then the circuit::remove_vertex will delete the destination edge
134  // hence circuit::rewire would result in an error due to the missing edge.
135  // We need a partial rewire for that reason.
136  // Example:
137  // Moving the second vertex (CX gate) to the front we only need to rewire
138  // the "x" part.
139  // --o-----
140  // |
141  // --x--x--
142  // |
143  // -----o--
144 
145  for (unsigned i = 0; i < dest_edges.size(); i++) {
146  Edge &dest_in_edge = dest_edges[i];
147  Edge &curr_in_edge = src_edges[i];
148  // If the vertex is already connected to an edge in the frontier, do
149  // nothing.
150  if (dest_in_edge != curr_in_edge) {
151  // Add first edge
152  Vertex dest_prev_vert = circ.source(dest_in_edge);
153  circ.add_edge(
154  {dest_prev_vert, circ.get_source_port(dest_in_edge)},
155  {vert, circ.get_target_port(curr_in_edge)}, EdgeType::Quantum);
156  // Add second edge
157  Vertex curr_next_vert;
158  Edge curr_out_edge;
159  Vertex dest_next_vert = circ.target(dest_in_edge);
160  std::tie(curr_next_vert, curr_out_edge) =
161  circ.get_next_pair(vert, curr_in_edge);
162  circ.add_edge(
163  {vert, circ.get_source_port(curr_out_edge)},
164  {dest_next_vert, circ.get_target_port(dest_in_edge)},
165  EdgeType::Quantum);
166  // Add third edge
167  Vertex curr_prev_vert = circ.source(curr_in_edge);
168  circ.add_edge(
169  {curr_prev_vert, circ.get_source_port(curr_in_edge)},
170  {curr_next_vert, circ.get_target_port(curr_out_edge)},
171  EdgeType::Quantum);
172  // Remove edges
173  circ.remove_edge(dest_in_edge);
174  circ.remove_edge(curr_in_edge);
175  circ.remove_edge(curr_out_edge);
176  }
177  }

180 bool MultiGateReorder::solve(unsigned max_depth, unsigned max_size) {
181  // Assume the frontier has been advanced
182 
183  // store a copy of the original this->mapping_frontier_->linear_boundary
184  // this object will be updated and reset throughout the procedure
185  // so need to return it to original setting at end.
186  unit_vertport_frontier_t copy;
187  for (const std::pair<UnitID, VertPort> &pair :
188  this->mapping_frontier_->linear_boundary->get<TagKey>()) {
189  copy.insert({pair.first, pair.second});
190  }
191  // get subcircuit vertices in topological order
192  CutFrontier current_cut = this->mapping_frontier_->circuit_.next_cut(
193  frontier_convert_vertport_to_edge(
194  this->mapping_frontier_->circuit_,
195  this->mapping_frontier_->linear_boundary),
196  this->mapping_frontier_->boolean_boundary);
197  unsigned subcircuit_depth = 1;
198  std::vector<Vertex> subcircuit_vertices(
199  current_cut.slice->begin(), current_cut.slice->end());
200  // add cuts of vertices to subcircuit_vertices until constraints met, or end
201  // of circuit reached
202  while (subcircuit_depth < max_depth &&
203  unsigned(subcircuit_vertices.size()) < max_size &&
204  current_cut.slice->size() > 0) {
205  current_cut = this->mapping_frontier_->circuit_.next_cut(
206  current_cut.u_frontier, current_cut.b_frontier);
207  subcircuit_depth++;
208  subcircuit_vertices.insert(
209  subcircuit_vertices.end(), current_cut.slice->begin(),
210  current_cut.slice->end());
211  }
212  // For each of the multi-q vertices in the subcircuit, find its unitids
213  std::vector<std::pair<Vertex, std::vector<Node>>> vertex_nodes;
214  for (const Vertex &vert : subcircuit_vertices) {
215  if (!is_multiq_quantum_gate(this->mapping_frontier_->circuit_, vert)) {
216  continue;
217  }
218  std::vector<Node> nodes;
219  for (port_t port = 0;
220  port < this->mapping_frontier_->circuit_.n_ports(vert); ++port) {
221  nodes.push_back(Node(
222  get_unitid_from_vertex_port(this->mapping_frontier_, {vert, port})));
223  }
224  vertex_nodes.push_back({vert, nodes});
225  }
226 
227  bool modification_made = false;
228  for (auto it = vertex_nodes.begin(); it != vertex_nodes.end(); it++) {
229  const Vertex &vert = it->first;
230  const std::vector<Node> &nodes = it->second;
231  // determine whether a vertex has been advanced by the
232  // advance_frontier_boundary call in this loop by traversing to the right by
233  // max_depth steps to see if it hits a frontier vertex.
234  // checking one port is enough
235  bool advanced = false;
236  VertPort current_vert_port(vert, 0);
237  for (unsigned i = 0; i < max_depth; i++) {
238  if (this->mapping_frontier_->linear_boundary->get<TagValue>().find(
239  current_vert_port) !=
240  this->mapping_frontier_->linear_boundary->get<TagValue>().end()) {
241  advanced = true;
242  break;
243  }
244  if (this->mapping_frontier_->circuit_.detect_boundary_Op(
245  current_vert_port.first)) {
246  break;
247  }
248  Edge current_in_edge = this->mapping_frontier_->circuit_.get_nth_in_edge(
249  current_vert_port.first, current_vert_port.second);
250  Vertex next_vert;
251  Edge next_e;
252  std::tie(next_vert, next_e) =
253  this->mapping_frontier_->circuit_.get_next_pair(
254  current_vert_port.first, current_in_edge);
255  current_vert_port = {
256  next_vert, this->mapping_frontier_->circuit_.get_target_port(next_e)};
257  }
258  if (advanced) {
259  continue;
260  }
261 
262  // try to commute a vertex to the frontier if it's physically permitted
263  if (this->mapping_frontier_->valid_boundary_operation(
264  this->architecture_,
265  this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(vert),
266  nodes)) {
267  std::optional<std::pair<EdgeVec, EdgeVec>> commute_pairs =
268  try_find_commute_edges(
269  this->mapping_frontier_->circuit_, this->u_frontier_edges_, vert);
270 
271  if (commute_pairs != std::nullopt) {
272  modification_made = true;
273  partial_rewire(
274  vert, this->mapping_frontier_->circuit_, (*commute_pairs).first,
275  (*commute_pairs).second);
276  // Update the frontier
277  this->mapping_frontier_->advance_frontier_boundary(this->architecture_);
278  this->u_frontier_edges_ =
279  convert_u_frontier_to_edges(*frontier_convert_vertport_to_edge(
280  this->mapping_frontier_->circuit_,
281  this->mapping_frontier_->linear_boundary));
282  }
283  }
284  }
285  // Return the quantum boundary to its original setting
286  this->mapping_frontier_->set_linear_boundary(copy);
287  return modification_made;
288 }
