32// Traverse the DAG to the quantum frontier
33// to find the UnitID associated with an VertPort
34static UnitID get_unitid_from_vertex_port(
35 const MappingFrontier_ptr &frontier, const VertPort &vert_port) {
36 VertPort current_vert_port = vert_port;
37 while (true) {
38 auto it =
39 frontier->linear_boundary->get<TagValue>().find(current_vert_port);
40 if (it != frontier->linear_boundary->get<TagValue>().end()) {
41 return it->first;
42 }
43 Edge current_e = frontier->circuit_.get_nth_out_edge(
44 current_vert_port.first, current_vert_port.second);
45 Vertex prev_vert;
46 Edge prev_e;
47 std::tie(prev_vert, prev_e) =
48 frontier->circuit_.get_prev_pair(current_vert_port.first, current_e);
49 current_vert_port = {prev_vert, frontier->circuit_.get_source_port(prev_e)};
50 }
51}

53static bool is_multiq_quantum_gate(const Circuit &circ, const Vertex &vert) {
54 Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
55 return (
56 op->get_desc().is_gate() && circ.n_in_edges(vert) > 1 &&
57 circ.n_in_edges_of_type(vert, EdgeType::Quantum) ==
58 circ.n_in_edges(vert) &&
59 circ.n_out_edges_of_type(vert, EdgeType::Quantum) ==
60 circ.n_out_edges(vert));
61}

63// This method will try to commute a vertex to the quantum frontier
64// If successful, returns the current in-edges and the target in-edges to
65// rewire the vertex
66static std::optional<std::pair<EdgeVec, EdgeVec>> try_find_commute_edges(
67 const Circuit &circ, const EdgeVec &frontier_edges, const Vertex &vert) {
68 // Initialize to be the in_edges for the given vertex
69 EdgeVec current_edges = circ.get_in_edges(vert);
70 EdgeVec initial_edges(current_edges.begin(), current_edges.end());
71
72 // Record the colour of each port of the vertex.
73 std::vector<std::optional<Pauli>> colours;
74 for (const Edge &edge : current_edges) {
75 port_t target_port = circ.get_target_port(edge);
76 std::optional<Pauli> colour = circ.commuting_basis(vert, target_port);
77 colours.push_back(colour);
78 }
79 // Stores all edges which the vertex can be commuted to
80 EdgeVec dest_edges;
81 while (true) {
82 // The vertex can be commuted to the front
83 bool success = true;
84 for (unsigned i = 0; i < current_edges.size(); ++i) {
85 // Check if the edge is already in the quantum frontier
86 if (std::find(
87 frontier_edges.begin(), frontier_edges.end(), current_edges[i]) !=
88 frontier_edges.end()) {
89 dest_edges.push_back(current_edges[i]);
90 continue;
91 }
92 // Check prev_op is a gate
93 Vertex prev_vert = circ.source(current_edges[i]);
94 Op_ptr prev_op = circ.get_Op_ptr_from_Vertex(prev_vert);
95 if (!prev_op->get_desc().is_gate()) {
96 // not commute
97 return std::nullopt;
98 }
99
100 // Check commute
101 port_t source_port = circ.get_source_port(current_edges[i]);
102 if (!circ.commutes_with_basis(prev_vert, colours[i], source_port)) {
103 // not commute
104 return std::nullopt;
105 } else {
106 // Update dest_edges
107 Vertex prev_prev_v;
108 Edge prev_e;
109 std::tie(prev_prev_v, prev_e) =
110 circ.get_prev_pair(prev_vert, current_edges[i]);
111 dest_edges.push_back(prev_e);
112 }
113 // Only true if all edges are in frontier
114 success = false;
115 }
116 if (success) {
117 std::pair<EdgeVec, EdgeVec> p(initial_edges, dest_edges);
118 return p;
119 } else {
120 current_edges = dest_edges;
121 dest_edges = {};
122 }
123 }
124}

126static void partial_rewire(
127 const Vertex &vert, Circuit &circ, EdgeVec &src_edges,
128 EdgeVec &dest_edges) {
129 // move the vertex to the frontier
130 // Notice that if one of the vertex's in edge is already a destination
131 // edge then the circuit::remove_vertex will delete the destination edge
132 // hence circuit::rewire would result in an error due to the missing edge.
133 // We need a partial rewire for that reason.
134 // Example:
135 // Moving the second vertex (CX gate) to the front we only need to rewire
136 // the "x" part.
137 // --o-----
138 // |
139 // --x--x--
140 // |
141 // -----o--
142
143 for (unsigned i = 0; i < dest_edges.size(); i++) {
144 Edge &dest_in_edge = dest_edges[i];
145 Edge &curr_in_edge = src_edges[i];
146 // If the vertex is already connected to an edge in the frontier, do
147 // nothing.
148 if (dest_in_edge != curr_in_edge) {
149 // Add first edge
150 Vertex dest_prev_vert = circ.source(dest_in_edge);
151 circ.add_edge(
152 {dest_prev_vert, circ.get_source_port(dest_in_edge)},
153 {vert, circ.get_target_port(curr_in_edge)}, EdgeType::Quantum);
154 // Add second edge
155 Vertex curr_next_vert;
156 Edge curr_out_edge;
157 Vertex dest_next_vert = circ.target(dest_in_edge);
158 std::tie(curr_next_vert, curr_out_edge) =
159 circ.get_next_pair(vert, curr_in_edge);
160 circ.add_edge(
161 {vert, circ.get_source_port(curr_out_edge)},
162 {dest_next_vert, circ.get_target_port(dest_in_edge)},
163 EdgeType::Quantum);
164 // Add third edge
165 Vertex curr_prev_vert = circ.source(curr_in_edge);
166 circ.add_edge(
167 {curr_prev_vert, circ.get_source_port(curr_in_edge)},
168 {curr_next_vert, circ.get_target_port(curr_out_edge)},
169 EdgeType::Quantum);
170 // Remove edges
171 circ.remove_edge(dest_in_edge);
172 circ.remove_edge(curr_in_edge);
173 circ.remove_edge(curr_out_edge);
174 }
175 }

178bool MultiGateReorder::solve(unsigned max_depth, unsigned max_size) {
179 // Assume the frontier has been advanced
180
181 // store a copy of the original this->mapping_frontier_->linear_boundary
182 // this object will be updated and reset throughout the procedure
183 // so need to return it to original setting at end.
184 unit_vertport_frontier_t copy;
185 for (const std::pair<UnitID, VertPort> &pair :
186 this->mapping_frontier_->linear_boundary->get<TagKey>()) {
187 copy.insert({pair.first, pair.second});
188 }
189 // get subcircuit vertices in topological order
190 CutFrontier current_cut = this->mapping_frontier_->circuit_.next_cut(
191 frontier_convert_vertport_to_edge(
192 this->mapping_frontier_->circuit_,
193 this->mapping_frontier_->linear_boundary),
194 this->mapping_frontier_->boolean_boundary);
195 unsigned subcircuit_depth = 1;
196 std::vector<Vertex> subcircuit_vertices(
197 current_cut.slice->begin(), current_cut.slice->end());
198 // add cuts of vertices to subcircuit_vertices until constraints met, or end
199 // of circuit reached
200 while (subcircuit_depth < max_depth &&
201 unsigned(subcircuit_vertices.size()) < max_size &&
202 current_cut.slice->size() > 0) {
203 current_cut = this->mapping_frontier_->circuit_.next_cut(
204 current_cut.u_frontier, current_cut.b_frontier);
205 subcircuit_depth++;
206 subcircuit_vertices.insert(
207 subcircuit_vertices.end(), current_cut.slice->begin(),
208 current_cut.slice->end());
209 }
210 // For each of the multi-q vertices in the subcircuit, find its unitids
211 std::vector<std::pair<Vertex, std::vector<Node>>> vertex_nodes;
212 for (const Vertex &vert : subcircuit_vertices) {
213 if (!is_multiq_quantum_gate(this->mapping_frontier_->circuit_, vert)) {
214 continue;
215 }
216 std::vector<Node> nodes;
217 for (port_t port = 0;
218 port < this->mapping_frontier_->circuit_.n_ports(vert); ++port) {
219 nodes.push_back(Node(
220 get_unitid_from_vertex_port(this->mapping_frontier_, {vert, port})));
221 }
222 vertex_nodes.push_back({vert, nodes});
223 }
224
225 bool modification_made = false;
226 for (auto it = vertex_nodes.begin(); it != vertex_nodes.end(); it++) {
227 const Vertex &vert = it->first;
228 const std::vector<Node> &nodes = it->second;
229 // determine whether a vertex has been advanced by the
230 // advance_frontier_boundary call in this loop by traversing to the right by
231 // max_depth steps to see if it hits a frontier vertex.
232 // checking one port is enough
233 bool advanced = false;
234 VertPort current_vert_port(vert, 0);
235 for (unsigned i = 0; i < max_depth; i++) {
236 if (this->mapping_frontier_->linear_boundary->get<TagValue>().find(
237 current_vert_port) !=
238 this->mapping_frontier_->linear_boundary->get<TagValue>().end()) {
239 advanced = true;
240 break;
241 }
242 if (this->mapping_frontier_->circuit_.detect_boundary_Op(
243 current_vert_port.first)) {
244 break;
245 }
246 Edge current_in_edge = this->mapping_frontier_->circuit_.get_nth_in_edge(
247 current_vert_port.first, current_vert_port.second);
248 Vertex next_vert;
249 Edge next_e;
250 std::tie(next_vert, next_e) =
251 this->mapping_frontier_->circuit_.get_next_pair(
252 current_vert_port.first, current_in_edge);
253 current_vert_port = {
254 next_vert, this->mapping_frontier_->circuit_.get_target_port(next_e)};
255 }
256 if (advanced) {
257 continue;
258 }
259
260 // try to commute a vertex to the frontier if it's physically permitted
261 if (this->mapping_frontier_->valid_boundary_operation(
262 this->architecture_,
263 this->mapping_frontier_->circuit_.get_Op_ptr_from_Vertex(vert),
264 nodes)) {
265 std::optional<std::pair<EdgeVec, EdgeVec>> commute_pairs =
266 try_find_commute_edges(
267 this->mapping_frontier_->circuit_, this->u_frontier_edges_, vert);
268
269 if (commute_pairs != std::nullopt) {
270 modification_made = true;
271 partial_rewire(
272 vert, this->mapping_frontier_->circuit_, (*commute_pairs).first,
273 (*commute_pairs).second);
274 // Update the frontier
275 this->mapping_frontier_->advance_frontier_boundary(this->architecture_);
276 this->u_frontier_edges_ =
277 convert_u_frontier_to_edges(*frontier_convert_vertport_to_edge(
278 this->mapping_frontier_->circuit_,
279 this->mapping_frontier_->linear_boundary));
280 }
281 }
282 }
283 // Return the quantum boundary to its original setting
284 this->mapping_frontier_->set_linear_boundary(copy);
285 return modification_made;
286}
