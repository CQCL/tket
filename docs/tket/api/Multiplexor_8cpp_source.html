42 static Circuit multiplexor_sequential_decomp(
43  const ctrl_op_map_t &op_map, unsigned n_controls, unsigned n_targets) {
44  Circuit c(n_controls + n_targets);
45  std::vector<unsigned> qubits(n_controls + n_targets);
46  std::iota(std::begin(qubits), std::end(qubits), 0);
47  for (auto it = op_map.begin(); it != op_map.end(); it++) {
48  std::vector<unsigned> zero_ctrls;
49  for (unsigned i = 0; i < it->first.size(); i++) {
50  if (!it->first[i]) {
51  zero_ctrls.push_back(i);
52  c.add_op<unsigned>(OpType::X, {i});
53  }
54  }
55  QControlBox qcbox(it->second, n_controls);
56  c.add_box(qcbox, qubits);
57  for (unsigned i : zero_ctrls) {
58  c.add_op<unsigned>(OpType::X, {i});
59  }
60  }
61  return c;
62 }

100 static void recursive_demultiplex_rotation(
101  const std::vector<Expr> &angles, const OpType &axis, unsigned total_qubits,
102  Circuit &circ, const RecursionNodeType &node_type) {
103  unsigned n_rotations = angles.size();
104  unsigned n_qubits = (unsigned)log2(n_rotations) + 1;
105  unsigned mid = (unsigned)(n_rotations / 2);
106  std::vector<Expr> p_angles;
107  std::vector<Expr> q_angles;
108  for (unsigned i = 0; i < mid; i++) {
109  p_angles.push_back((angles[i] - angles[mid + i]) / 2);
110  q_angles.push_back((angles[i] + angles[mid + i]) / 2);
111  }
112  // UCR = CX P CX Q = Q CX P CX
113  // the left recursion child implements P CX Q, and the
114  // right recursion child implements Q CX P to cancel the CXs
115  if (node_type == RecursionNodeType::Right) {
116  std::swap(p_angles, q_angles);
117  }
118  if (q_angles.size() == 1) {
119  // base step
120  circ.add_op<unsigned>(axis, q_angles[0], {total_qubits - 1});
121  } else {
122  recursive_demultiplex_rotation(
123  q_angles, axis, total_qubits, circ, RecursionNodeType::Left);
124  }
125  circ.add_op<unsigned>(
126  OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
127  if (p_angles.size() == 1) {
128  // base step
129  circ.add_op<unsigned>(axis, p_angles[0], {total_qubits - 1});
130  } else {
131  recursive_demultiplex_rotation(
132  p_angles, axis, total_qubits, circ, RecursionNodeType::Right);
133  }
134  if (node_type == RecursionNodeType::Root) {
135  // for the root step, we implement UCR = CX P CX Q
136  circ.add_op<unsigned>(
137  OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
138  }
139 }

151 static std::tuple<Eigen::Matrix2cd, Eigen::Matrix2cd, double, double>
152 constant_demultiplex(const Eigen::Matrix2cd &a, const Eigen::Matrix2cd &b) {
153  Eigen::Matrix2cd X = a * b.adjoint();
154  // decompose X using eq(19)
155  std::vector<double> tk1_params = tk1_angles_from_unitary(X);
156  Eigen::Matrix2cd tk1_su2 = get_matrix_from_tk1_angles(
157  {tk1_params[0], tk1_params[1], tk1_params[2], 0.0});
158  Complex x0 = tk1_su2(0, 0);
159  double phi = tk1_params[3] * PI * 2;
160  // compute r matrix using eq(11) and eq(12)
161  double a0 = -PI / 2 - phi / 2 - std::arg(x0);
162  double a1 = PI / 2 - phi / 2 + std::arg(x0);
163  Complex r0 = std::exp(0.5 * i_ * a0);
164  Complex r1 = std::exp(0.5 * i_ * a1);
165  Eigen::Matrix2cd r = Eigen::Matrix2cd::Zero();
166  r(0, 0) = r0;
167  r(1, 1) = r1;
168  // a0 and a1 defines the R matrix in eq(3). And they are the z rotation angles
169  // activated by 0 and 1 respectively.
170  a0 = a0 / PI;
171  a1 = a1 / PI;
172  Eigen::ComplexEigenSolver<Eigen::Matrix2cd> eigen_solver(r * X * r);
173  Eigen::Matrix2cd u = eigen_solver.eigenvectors();
174  // the eigenvalues are guaranteed to be {i, -i}. We permute u so its
175  // eigenvalues have the order [i, -i].
176  if (std::abs(eigen_solver.eigenvalues()[0] + i_) < EPS) {
177  u.col(0).swap(u.col(1));
178  }
179  Eigen::Matrix2cd d = Eigen::Matrix2cd::Zero();
180  d(0, 0) = std::sqrt(i_);
181  d(1, 1) = std::sqrt(-i_);
182  Eigen::Matrix2cd v = d * u.adjoint() * r.adjoint() * b;
183  return {u, v, a0, a1};
184 }

193 static std::vector<Eigen::Matrix2cd> ucrz_angles_to_diagonal(
194  const std::vector<double> &angles) {
195  std::vector<Eigen::Matrix2cd> diag;
196  unsigned mid = (unsigned)(angles.size() / 2);
197  for (unsigned i = 0; i < mid; i++) {
198  Eigen::Matrix2cd u = Eigen::Matrix2cd::Zero();
199  u(0, 0) = std::exp(-0.5 * angles[i] * i_ * PI);
200  u(1, 1) = std::exp(-0.5 * angles[i + mid] * i_ * PI);
201  diag.push_back(u);
202  }
203  for (unsigned i = 0; i < mid; i++) {
204  Eigen::Matrix2cd u = Eigen::Matrix2cd::Zero();
205  u(0, 0) = std::exp(0.5 * angles[i] * i_ * PI);
206  u(1, 1) = std::exp(0.5 * angles[i + mid] * i_ * PI);
207  diag.push_back(u);
208  }
209  return diag;
210 }
242 static void recursive_demultiplex_u2(
243  std::vector<Eigen::Matrix2cd> &unitaries, unsigned total_qubits,
244  Circuit &circ, std::vector<std::vector<double>> &ucrzs,
245  const Eigen::Matrix2cd &left_compose,
246  const Eigen::Matrix2cd &right_compose) {
247  // The following two constant matrices are the bottom SQ unitaries resulted
248  // from decomposing the D gate (i.e. ZZPhase(-0.5)) using CX
249  const static Eigen::Matrix2cd U_MULT =
250  get_matrix_from_tk1_angles({0.5, 0.5, 0.5, 0.0});
251  const static Eigen::Matrix2cd V_MULT =
252  get_matrix_from_tk1_angles({0.5, 0.5, 0, 0.0});
253  unsigned n_unitaries = unitaries.size();
254  unsigned n_qubits = (unsigned)log2(n_unitaries) + 1;
255  unsigned mid = (unsigned)(n_unitaries / 2);
256  // We generalise eq(3) for n controls, demultiplex the multiplexor
257  // by demultiplexing all pairs {unitaries[i], unitaries[mid+i]} 0<=i<mid.
258  // i.e. I tensor diag(u) = I tensor diag(u_list)
259  // I tensor diag(v) = I tensor diag(v_list)
260  // D = ZZPhase(-0.5)
261  // R = UCRz(rz_list, [q_{n-1}, q_{1}, q_{2}, ..., q_{n-2}, q_0])
262  std::vector<Eigen::Matrix2cd> u_list;
263  std::vector<Eigen::Matrix2cd> v_list;
264  std::vector<double> rz_list(n_unitaries);
265 
266  // merge previous UCRz gate into the multiplexor
267  std::vector<Eigen::Matrix2cd> ucrz_diag =
268  ucrz_angles_to_diagonal(ucrzs[n_qubits - 2]);
269  for (unsigned i = 0; i < unitaries.size(); i++) {
270  unitaries[i] = unitaries[i] * ucrz_diag[i];
271  }
272  // demultiplex pairs (unitaries[i], unitaries[mid+i])
273  for (unsigned i = 0; i < mid; i++) {
274  auto [u, v, a0, a1] =
275  constant_demultiplex(unitaries[i], unitaries[mid + i]);
276  u_list.push_back(u);
277  v_list.push_back(v);
278  rz_list[i] = a0;
279  rz_list[i + mid] = a1;
280  }
281 
282  // update the ucrzs with the 1.5 angle resulted from decomposing ZZPhase(-0.5)
283  std::for_each(rz_list.begin(), rz_list.end(), [](double &f) { f += 1.5; });
284  ucrzs[n_qubits - 2] = rz_list;
285 
286  // adding gates to the circuit
287  // add v
288  if (v_list.size() == 1) {
289  Eigen::Matrix2cd v_prime = V_MULT * v_list[0] * left_compose;
290  circ.add_box(Unitary1qBox(v_prime), {total_qubits - 1});
291  } else {
292  recursive_demultiplex_u2(
293  v_list, total_qubits, circ, ucrzs, left_compose, V_MULT);
294  }
295  // add CX
296  circ.add_op<unsigned>(
297  OpType::CX, {total_qubits - n_qubits, total_qubits - 1});
298  circ.add_phase(1.75);
299  // add u
300  if (u_list.size() == 1) {
301  Eigen::Matrix2cd u_prime = right_compose * u_list[0] * U_MULT;
302  circ.add_box(Unitary1qBox(u_prime), {total_qubits - 1});
303  } else {
304  recursive_demultiplex_u2(
305  u_list, total_qubits, circ, ucrzs, U_MULT, right_compose);
306  }
307  return;
308 }

310 static void op_map_validate(const ctrl_op_map_t &op_map) {
311  unsigned n_controls = 0;
312  unsigned n_targets = 0;
313  for (auto it = op_map.begin(); it != op_map.end(); it++) {
314  op_signature_t op_sig = it->second->get_signature();
315  if ((unsigned long)std::count(
316  op_sig.begin(), op_sig.end(), EdgeType::Quantum) != op_sig.size()) {
317  throw BadOpType(
318  "Quantum control of classical wires not supported",
319  it->second->get_type());
320  }
321  if (it == op_map.begin()) {
322  n_controls = (unsigned)it->first.size();
323  n_targets = (unsigned)op_sig.size();
324  } else {
325  if (it->first.size() != n_controls) {
326  throw std::invalid_argument("Bitstrings must have the same width.");
327  }
328  if (op_sig.size() != n_targets) {
329  throw std::invalid_argument("Ops must have the same width.");
330  }
331  }
332  }

428 MultiplexedRotationBox::MultiplexedRotationBox(const ctrl_op_map_t &op_map)
429  : Box(OpType::MultiplexedRotationBox), op_map_(op_map) {
430  auto it = op_map.begin();
431  if (it == op_map.end()) {
432  throw std::invalid_argument("No Ops provided.");
433  }
434  for (; it != op_map.end(); it++) {
435  if (it == op_map.begin()) {
436  n_controls_ = (unsigned)it->first.size();
437  if (n_controls_ > MAX_N_CONTROLS) {
438  throw std::invalid_argument(
439  "Bitstrings longer than " + std::to_string(MAX_N_CONTROLS) +
440  " are not supported.");
441  }
442  axis_ = it->second->get_type();
443  if (axis_ != OpType::Rx && axis_ != OpType::Ry && axis_ != OpType::Rz) {
444  throw BadOpType("Ops must be either Rx, Ry, or Rz.", axis_);
445  }
446  } else {
447  if (it->second->get_type() != axis_) {
448  throw std::invalid_argument("Ops must have the same rotation type.");
449  }
450  }
451  }
452  op_map_validate(op_map);
453 }

500 void MultiplexedRotationBox::generate_circuit() const {
501  Circuit circ(n_controls_ + 1);
502  if (n_controls_ == 0) {
503  auto it = op_map_.begin();
504  circ.add_op<unsigned>(it->second, {0});
505  circ_ = std::make_shared<Circuit>(circ);
506  return;
507  }
508  unsigned n_rotations = 1 << n_controls_;
509  std::vector<Expr> rotations(n_rotations);
510  // convert op_map to a vector of 2^n_controls_ angles
511  for (unsigned i = 0; i < n_rotations; i++) {
512  auto it = op_map_.find(dec_to_bin(i, n_controls_));
513  if (it == op_map_.end()) {
514  rotations[i] = 0;
515  } else {
516  rotations[i] = it->second->get_params()[0];
517  }
518  }
519  OpType axis = axis_;
520  if (axis_ == OpType::Rx) {
521  circ.add_op<unsigned>(OpType::H, {n_controls_});
522  axis = OpType::Rz;
523  }
524  recursive_demultiplex_rotation(
525  rotations, axis, n_controls_ + 1, circ, RecursionNodeType::Root);
526  if (axis_ == OpType::Rx) {
527  circ.add_op<unsigned>(OpType::H, {n_controls_});
528  }
529  circ_ = std::make_shared<Circuit>(circ);
530 }

602 void MultiplexedU2Box::generate_circuit() const {
603  Circuit circ(n_controls_ + 1);
604  if (n_controls_ == 0) {
605  auto it = op_map_.begin();
606  circ.add_op<unsigned>(it->second, {0});
607  circ_ = std::make_shared<Circuit>(circ);
608  return;
609  }
610  unsigned n_unitaries = 1 << n_controls_;
611  std::vector<Eigen::Matrix2cd> unitaries(n_unitaries);
612  // convert op_map to a vector of 2^n_controls_ unitaries
613  for (unsigned i = 0; i < n_unitaries; i++) {
614  auto it = op_map_.find(dec_to_bin(i, n_controls_));
615  if (it == op_map_.end()) {
616  unitaries[i] = Eigen::Matrix2cd::Identity();
617  } else {
618  if (it->second->get_type() == OpType::Unitary1qBox) {
619  std::shared_ptr<const Unitary1qBox> u1box =
620  std::dynamic_pointer_cast<const Unitary1qBox>(it->second);
621  unitaries[i] = u1box->get_matrix();
622  } else {
623  if (!it->second->free_symbols().empty()) {
624  throw Unsupported("Can't decompose symbolic MultiplexedU2Box.");
625  }
626  unitaries[i] = GateUnitaryMatrix::get_unitary(*as_gate_ptr(it->second));
627  }
628  }
629  }
630  // initialise the ucrz list
631  std::vector<std::vector<double>> ucrzs(n_controls_);
632  for (unsigned i = 0; i < n_controls_; i++) {
633  ucrzs[i] = std::vector<double>(1 << (i + 1), 0.0);
634  }
635  recursive_demultiplex_u2(
636  unitaries, n_controls_ + 1, circ, ucrzs, Eigen::Matrix2cd::Identity(),
637  Eigen::Matrix2cd::Identity());
638  if (impl_diag_) {
639  // add the final ucrzs
640  for (unsigned i = 0; i < n_controls_; i++) {
641  // ith ucrzs acts on i+2 qubits
642  // get the args for the UCRz gate
643  std::vector<unsigned> args(i + 2);
644  std::iota(std::begin(args), std::end(args), n_controls_ - i - 1);
645  // swap the first and the last args
646  std::swap(args[0], args[i + 1]);
647  // create MultiplexedRotationBox
648  ctrl_op_map_t rz_map;
649  for (unsigned k = 0; k < ucrzs[i].size(); k++) {
650  if (std::abs(ucrzs[i][k]) > EPS) {
651  rz_map.insert(
652  {dec_to_bin(k, i + 1), get_op_ptr(OpType::Rz, ucrzs[i][k])});
653  }
654  }
655  circ.add_box(MultiplexedRotationBox(rz_map), args);
656  }
657  }
658  circ_ = std::make_shared<Circuit>(circ);
659 }
