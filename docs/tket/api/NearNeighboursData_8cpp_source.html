30 static void fill_neighbours_of_neighbours(
31  VertexWSM root_vertex,
32  const std::vector<std::pair<VertexWSM, WeightWSM>>& neighbours_and_weights,
33  const NeighboursData& ndata, std::set<VertexWSM>& vertices_workset,
34  std::vector<VertexWSM>& vertices) {
35  vertices_workset.clear();
36  for (const std::pair<VertexWSM, WeightWSM>& entry : neighbours_and_weights) {
37  const VertexWSM& vv1 = entry.first;
38  for (const std::pair<VertexWSM, WeightWSM>& other_entry :
39  ndata.get_neighbours_and_weights(vv1)) {
40  const VertexWSM& vv2 = other_entry.first;
41  if (vv2 == root_vertex ||
42  NeighboursData::binary_search(vv2, neighbours_and_weights)) {
43  continue;
44  }
45  vertices_workset.insert(vv2);
46  }
47  }
48  vertices = {vertices_workset.cbegin(), vertices_workset.cend()};
49 }

51 // result[i] is for distance i+2.
52 // We've already filled in result[0] at least, and resized to the desired size.
53 // Now fill in all the result[i].
54 static void fill_more_distant_vertices(
55  std::vector<std::vector<VertexWSM>>& result, unsigned old_size,
56  const std::vector<std::pair<VertexWSM, WeightWSM>>& neighbours_and_weights,
57  std::set<VertexWSM>& vertices_workset, const NeighboursData& ndata) {
58  TKET_ASSERT(old_size > 0);
59  TKET_ASSERT(old_size <= result.size());
60 
61  for (unsigned index = old_size; index < result.size(); ++index) {
62  vertices_workset.clear();
63  for (VertexWSM vv_prev : result[index - 1]) {
64  TKET_ASSERT(result[index].empty());
65  for (const std::pair<VertexWSM, WeightWSM>& entry :
66  ndata.get_neighbours_and_weights(vv_prev)) {
67  const VertexWSM& vv_new = entry.first;
68  // vv_prev is at distance d from the root vertex;
69  // thus neighbours of vv_prev are at distance d-1, d, d+1.
70  // So we only need to check two vectors (for d-1,d).
71  if (std::binary_search(
72  result[index - 1].cbegin(), result[index - 1].cend(), vv_new)) {
73  continue;
74  }
75  if (index == 1) {
76  if (NeighboursData::binary_search(vv_new, neighbours_and_weights)) {
77  continue;
78  }
79  } else {
80  if (std::binary_search(
81  result[index - 2].cbegin(), result[index - 2].cend(),
82  vv_new)) {
83  continue;
84  }
85  }
86  vertices_workset.insert(vv_new);
87  }
88  }
89  if (vertices_workset.empty()) {
90  // no point in filling in more empty vertex lists;
91  // they're already empty.
92  break;
93  }
94  result[index] = {vertices_workset.cbegin(), vertices_workset.cend()};
95  }
96 }

98 const std::vector<VertexWSM>& NearNeighboursData::get_vertices_at_distance(
99  VertexWSM vv, unsigned distance) {
100  TKET_ASSERT(distance >= 2);
101  const unsigned index = distance - 2;
102  std::vector<std::vector<VertexWSM>>& results_for_this_vertex =
103  m_data[vv].vertices_at_distance;
104  unsigned old_size = results_for_this_vertex.size();
105 
106  if (index >= old_size) {
107  // "results_for_this_vertex" is too small, so we must fill in more entries.
108  // Give it the correct size, prior to filling.
109  results_for_this_vertex.resize(index + 1);
110  const std::vector<std::pair<VertexWSM, WeightWSM>>& neighbours_and_weights =
111  m_ndata.get_neighbours_and_weights(vv);
112 
113  if (old_size == 0) {
114  // A special case. We must fill in distance 2 data.
115  fill_neighbours_of_neighbours(
116  vv, neighbours_and_weights, m_ndata, m_vertices_workset,
117  results_for_this_vertex[0]);
118  old_size = 1;
119  }
120  fill_more_distant_vertices(
121  results_for_this_vertex, old_size, neighbours_and_weights,
122  m_vertices_workset, m_ndata);
123  }
124  return results_for_this_vertex.at(index);
125 }

127 std::size_t NearNeighboursData::get_n_vertices_at_max_distance(
128  VertexWSM vv, unsigned max_distance) {
129  switch (max_distance) {
130  case 0:
131  return 0;
132  case 1:
133  return m_ndata.get_degree(vv);
134  default:
135  break;
136  }
137  std::vector<std::size_t>& sizes_list = m_data[vv].n_vertices_at_max_distance;
138  const unsigned index = max_distance - 2;
139  if (index < sizes_list.size()) {
140  return sizes_list[index];
141  }
142  unsigned old_size = sizes_list.size();
143  sizes_list.resize(index + 1);
144  if (old_size == 0) {
145  sizes_list[0] =
146  m_ndata.get_degree(vv) + get_vertices_at_distance(vv, 2).size();
147  ++old_size;
148  }
149  for (unsigned ii = old_size; ii <= index; ++ii) {
150  sizes_list[ii] =
151  sizes_list[ii - 1] + get_vertices_at_distance(vv, ii + 2).size();
152  }
153  return sizes_list[index];
154 }

156 const FilterUtils::DegreeCounts& NearNeighboursData::get_degree_counts(
157  VertexWSM vv, unsigned distance) {
158  TKET_ASSERT(distance >= 2);
159  std::vector<FilterUtils::DegreeCounts>& counts_list =
160  m_data[vv].degree_counts_for_distance;
161  const unsigned index = distance - 2;
162  if (index < counts_list.size()) {
163  return counts_list[index];
164  }
165  // We don't have data up to this distance, for this vertex.
166  // Ensure first that we HAVE calculated the vertices up to this distance.
167  get_vertices_at_distance(vv, distance);
168 
169  // Notice that the vertices, and degree counts here, are stored separately,
170  // but on the SAAME value m_data[vv],
171  // so references aren't invalidated.
172  const std::vector<std::vector<VertexWSM>>& vertices_lists =
173  m_data[vv].vertices_at_distance;
174  TKET_ASSERT(index < vertices_lists.size());
175 
176  unsigned old_size = counts_list.size();
177  counts_list.resize(index + 1);
178  m_work_map.clear();
179  if (old_size == 0) {
180  // We have to fill in distance 2 data as the initial "seed".
181  for (VertexWSM distance_2_vertex : vertices_lists[0]) {
182  // C++ standard guarantees it will be initialised with 0.
183  m_work_map[m_ndata.get_degree(distance_2_vertex)] += 1;
184  }
185  if (m_type == Type::TARGET_GRAPH) {
186  // We ALSO must fill in distance 1 data, i.e. neighbours.
187  for (const std::pair<VertexWSM, WeightWSM>& entry :
188  m_ndata.get_neighbours_and_weights(vv)) {
189  m_work_map[m_ndata.get_degree(entry.first)] += 1;
190  }
191  }
192  counts_list[0].reserve(m_work_map.size());
193  for (const std::pair<const std::size_t, std::size_t>& entry : m_work_map) {
194  counts_list[0].emplace_back(entry);
195  }
196  // Simply pretend that we'd already filled it in.
197  ++old_size;
198  }
199 
200  for (unsigned index_to_fill = old_size; index_to_fill <= index;
201  ++index_to_fill) {
202  switch (m_type) {
203  case Type::PATTERN_GRAPH:
204  m_work_map.clear();
205  break;
206  case Type::TARGET_GRAPH:
207  // We'll maintain the data within m_work_map and add to it
208  // as we increase the distances.
209  if (m_work_map.empty()) {
210  // Need the initial fill of the previous data.
211  const FilterUtils::DegreeCounts& previous_counts =
212  counts_list[index_to_fill - 1];
213  for (const std::pair<std::size_t, std::size_t>& entry :
214  previous_counts) {
215  m_work_map.emplace(entry);
216  }
217  }
218  break;
219  }
220  const std::vector<VertexWSM>& vertices = vertices_lists[index_to_fill];
221  for (VertexWSM other_vertex : vertices) {
222  m_work_map[m_ndata.get_degree(other_vertex)] += 1;
223  }
224  // Now dump it out from the map back into the vector, to be stored.
225  counts_list[index_to_fill].reserve(m_work_map.size());
226  for (const std::pair<const std::size_t, std::size_t>& entry : m_work_map) {
227  counts_list[index_to_fill].emplace_back(entry);
228  }
229  TKET_ASSERT(counts_list[index_to_fill].size() == m_work_map.size());
230  }
231  return counts_list[index];
232 }
