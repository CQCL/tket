33NeighbourPlacements::ResultVec NeighbourPlacements::get(
34 unsigned dist, unsigned n, bool optimise, unsigned seed,
35 unsigned max_tries) {
36 rng_.set_seed(seed);
37
38 // define a comparison function for placements
39 std::vector<Qubit> keys;
40 for (auto [k, v] : init_map_) {
41 keys.push_back(k);
42 }
43 auto map_compare =
44 [&keys](const std::map<Qubit, Node>& a, const std::map<Qubit, Node>& b) {
45 for (auto k : keys) {
46 if (a.at(k) < b.at(k)) {
47 return true;
48 } else if (a.at(k) > b.at(k)) {
49 return false;
50 }
51 }
52 return false;
53 };
54 // set of all generated placement maps
55 std::set<std::map<Qubit, Node>, decltype(map_compare)> placements(
56 map_compare);
57
58 ResultVec resvec;
59 for (unsigned i = 0; i < n; ++i) {
60 unsigned n_unsuccessful = 0;
61 while (n_unsuccessful < max_tries) {
62 Result res = gen_result(dist, optimise, max_tries);
63 if (!placements.contains(res.map)) {
64 resvec.push_back(res);
65 placements.insert(res.map);
66 break;
67 }
68 ++n_unsuccessful;
69 }
70 if (n_unsuccessful == max_tries) {
71 std::stringstream ss;
72 ss << "Could not generate " << n << " distinct placements";
73 tket_log()->warn(ss.str());
74 }
75 }
76 return resvec;
77}

79NeighbourPlacements::Result NeighbourPlacements::gen_result(
80 unsigned dist, bool optimise, unsigned max_tries) {
81 SwapList swaps;
82 tsa_internal::SwapListOptimiser optimiser;
83
84 // it might be impossible to find `dist` non-trivial swaps
85 unsigned n_unsuccessful = 0;
86
87 while (swaps.size() < dist && n_unsuccessful < max_tries) {
88 Swap new_swap = gen_swap();
89
90 if (optimise) {
91 SwapList swaps_candidate = swaps;
92 swaps_candidate.push_back(new_swap);
93 optimiser.full_optimise(swaps_candidate);
94 if (swaps_candidate.size() > swaps.size()) {
95 swaps = std::move(swaps_candidate);
96 n_unsuccessful = 0;
97 } else {
98 ++n_unsuccessful;
99 }
100 } else {
101 swaps.push_back(new_swap);
102 }
103 }
104
105 if (n_unsuccessful == max_tries) {
106 std::stringstream ss;
107 ss << "Unable to generate " << dist << " swaps for given architecture";
108 tket_log()->warn(ss.str());
109 }
110
111 return convert_to_res(swaps.to_vector());
112}

114Swap NeighbourPlacements::gen_swap() {
115 auto edges = arc_.get_all_edges_vec();
116 unsigned m = edges.size();
117 auto [n1, n2] = edges[rng_.get_size_t(m - 1)];
118 Swap new_swap{u_to_node_.right.at(n1), u_to_node_.right.at(n2)};
119 return new_swap;
120}

122NeighbourPlacements::Result NeighbourPlacements::convert_to_res(
123 const SwapVec& swaps) {
124 NodeSwapVec node_swaps;
125 for (auto [u1, u2] : swaps) {
126 node_swaps.push_back({u_to_node_.left.at(u1), u_to_node_.left.at(u2)});
127 }
128
129 boost::bimap<Qubit, Node> qubit_to_node;
130 qubit_to_node.left.insert(init_map_.begin(), init_map_.end());
131 for (auto [n1, n2] : node_swaps) {
132 const Qubit q1 = qubit_to_node.right.at(n1);
133 const Qubit q2 = qubit_to_node.right.at(n2);
134 qubit_to_node.left.erase(q1);
135 qubit_to_node.left.erase(q2);
136 qubit_to_node.left.insert({q1, n2});
137 qubit_to_node.left.insert({q2, n1});
138 }
139 std::map<Qubit, Node> map;
140 for (auto [k, v] : qubit_to_node.left) {
141 map.insert({k, v});
142 }
143 return {map, node_swaps};
144}
