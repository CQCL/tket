34 NeighbourPlacements::ResultVec NeighbourPlacements::get(
35  unsigned dist, unsigned n, bool optimise, unsigned seed,
36  unsigned max_tries) {
37  rng_.set_seed(seed);
38 
39  // define a comparison function for placements
40  std::vector<Qubit> keys;
41  for (auto [k, v] : init_map_) {
42  keys.push_back(k);
43  }
44  auto map_compare =
45  [&keys](const std::map<Qubit, Node>& a, const std::map<Qubit, Node>& b) {
46  for (auto k : keys) {
47  if (a.at(k) < b.at(k)) {
48  return true;
49  } else if (a.at(k) > b.at(k)) {
50  return false;
51  }
52  }
53  return false;
54  };
55  // set of all generated placement maps
56  std::set<std::map<Qubit, Node>, decltype(map_compare)> placements(
57  map_compare);
58 
59  ResultVec resvec;
60  for (unsigned i = 0; i < n; ++i) {
61  unsigned n_unsuccessful = 0;
62  while (n_unsuccessful < max_tries) {
63  Result res = gen_result(dist, optimise, max_tries);
64  if (!placements.contains(res.map)) {
65  resvec.push_back(res);
66  placements.insert(res.map);
67  break;
68  }
69  ++n_unsuccessful;
70  }
71  if (n_unsuccessful == max_tries) {
72  std::stringstream ss;
73  ss << "Could not generate " << n << " distinct placements";
74  tket_log()->warn(ss.str());
75  }
76  }
77  return resvec;
78 }

80 NeighbourPlacements::Result NeighbourPlacements::gen_result(
81  unsigned dist, bool optimise, unsigned max_tries) {
82  SwapList swaps;
83  tsa_internal::SwapListOptimiser optimiser;
84 
85  // it might be impossible to find `dist` non-trivial swaps
86  unsigned n_unsuccessful = 0;
87 
88  while (swaps.size() < dist && n_unsuccessful < max_tries) {
89  Swap new_swap = gen_swap();
90 
91  if (optimise) {
92  SwapList swaps_candidate = swaps;
93  swaps_candidate.push_back(new_swap);
94  optimiser.full_optimise(swaps_candidate);
95  if (swaps_candidate.size() > swaps.size()) {
96  swaps = std::move(swaps_candidate);
97  n_unsuccessful = 0;
98  } else {
99  ++n_unsuccessful;
100  }
101  } else {
102  swaps.push_back(new_swap);
103  }
104  }
105 
106  if (n_unsuccessful == max_tries) {
107  std::stringstream ss;
108  ss << "Unable to generate " << dist << " swaps for given architecture";
109  tket_log()->warn(ss.str());
110  }
111 
112  return convert_to_res(swaps.to_vector());
113 }

115 Swap NeighbourPlacements::gen_swap() {
116  auto edges = arc_.get_all_edges_vec();
117  unsigned m = edges.size();
118  auto [n1, n2] = edges[rng_.get_size_t(m - 1)];
119  Swap new_swap{u_to_node_.right.at(n1), u_to_node_.right.at(n2)};
120  return new_swap;
121 }

123 NeighbourPlacements::Result NeighbourPlacements::convert_to_res(
124  const SwapVec& swaps) {
125  NodeSwapVec node_swaps;
126  for (auto [u1, u2] : swaps) {
127  node_swaps.push_back({u_to_node_.left.at(u1), u_to_node_.left.at(u2)});
128  }
129 
130  boost::bimap<Qubit, Node> qubit_to_node;
131  qubit_to_node.left.insert(init_map_.begin(), init_map_.end());
132  for (auto [n1, n2] : node_swaps) {
133  const Qubit q1 = qubit_to_node.right.at(n1);
134  const Qubit q2 = qubit_to_node.right.at(n2);
135  qubit_to_node.left.erase(q1);
136  qubit_to_node.left.erase(q2);
137  qubit_to_node.left.insert({q1, n2});
138  qubit_to_node.left.insert({q2, n1});
139  }
140  std::map<Qubit, Node> map;
141  for (auto [k, v] : qubit_to_node.left) {
142  map.insert({k, v});
143  }
144  return {map, node_swaps};
145 }
