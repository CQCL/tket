47 NeighboursData::NeighboursData(const GraphEdgeWeights& edges_and_weights) {
48  // Only the edges (v1,v2) with v1<v2.
49  std::set<std::pair<VertexWSM, VertexWSM>> ordered_edges_seen;
50  std::set<VertexWSM> vertices_seen;
51  for (const std::pair<const EdgeWSM, WeightWSM>& entry : edges_and_weights) {
52  const VertexWSM& v1 = entry.first.first;
53  const VertexWSM& v2 = entry.first.second;
54  vertices_seen.insert(v1);
55  vertices_seen.insert(v2);
56  if (v1 == v2) {
57  throw std::runtime_error("Loop found in graph; not allowed");
58  }
59  ordered_edges_seen.insert(get_edge(v1, v2));
60  }
61  if (vertices_seen.empty()) {
62  throw std::runtime_error("No edges passed to NeighboursData");
63  }
64  if (*vertices_seen.cbegin() != 0 ||
65  *vertices_seen.crbegin() != vertices_seen.size() - 1) {
66  throw std::runtime_error("Vertices should be [0,1,2,...,N].");
67  }
68  for (const EdgeWSM& edge : ordered_edges_seen) {
69  const std::optional<WeightWSM> weight_opt =
70  get_optional_value(edges_and_weights, edge);
71  if (weight_opt) {
72  const std::optional<WeightWSM> other_weight_opt = get_optional_value(
73  edges_and_weights, std::make_pair(edge.second, edge.first));
74  if (other_weight_opt && weight_opt.value() != other_weight_opt.value()) {
75  throw std::runtime_error("Edge weights mismatch");
76  }
77  }
78  }
79  m_number_of_edges = ordered_edges_seen.size();
80  m_neighbours_and_weights.resize(vertices_seen.size());
81 
82  for (const EdgeWSM& edge : ordered_edges_seen) {
83  const VertexWSM& v1 = edge.first;
84  const VertexWSM& v2 = edge.second;
85  TKET_ASSERT(v1 < v2);
86  TKET_ASSERT(v2 < vertices_seen.size());
87  WeightWSM weight;
88  const std::optional<WeightWSM> weight_opt =
89  get_optional_value(edges_and_weights, edge);
90  if (weight_opt) {
91  weight = weight_opt.value();
92  } else {
93  weight = edges_and_weights.at(std::make_pair(v2, v1));
94  }
95  m_neighbours_and_weights[v1].emplace_back(v2, weight);
96  m_neighbours_and_weights[v2].emplace_back(v1, weight);
97  }
98  for (std::vector<std::pair<VertexWSM, WeightWSM>>& neigh_data :
99  m_neighbours_and_weights) {
100  // Automatically sort by vertex first; lexicographic.
101  std::sort(neigh_data.begin(), neigh_data.end());
102  TKET_ASSERT(is_sorted_and_unique(neigh_data));
103  }
104 }

106 std::optional<WeightWSM> NeighboursData::get_edge_weight_opt(
107  VertexWSM v1, VertexWSM v2) const {
108  if (v1 >= m_neighbours_and_weights.size()) {
109  return {};
110  }
111  const std::vector<std::pair<VertexWSM, WeightWSM>>& v1_data =
112  m_neighbours_and_weights[v1];
113  // (x,0) = (x,min) <= (x,w) <= (x+1, y) in lexicographic order
114  std::pair<VertexWSM, WeightWSM> key;
115  key.first = v2;
116  key.second = 0;
117  const auto v2_citer = std::lower_bound(v1_data.cbegin(), v1_data.cend(), key);
118  if (v2_citer != v1_data.cend() && v2_citer->first == v2) {
119  return v2_citer->second;
120  }
121  return {};
122 }

124 bool NeighboursData::binary_search(
125  VertexWSM v, const std::vector<std::pair<VertexWSM, WeightWSM>>& list) {
126  const auto citer = std::lower_bound(
127  list.cbegin(), list.cend(), std::make_pair(v, WeightWSM(0)));
128  if (citer == list.cend()) {
129  return false;
130  }
131  return citer->first == v;

163 std::vector<WeightWSM> NeighboursData::get_weights_expensive() const {
164  std::vector<WeightWSM> weights;
165  weights.reserve(m_number_of_edges);
166  for (unsigned v1 = 0; v1 < m_neighbours_and_weights.size(); ++v1) {
167  // Every edge is implicitly stored twice, for (v1, v2) and (v2, v1).
168  // To avoid duplicates, only write the weight when v1>v2.
169  // The neighbour edges are stored with increasing v, as always.
170  for (const std::pair<VertexWSM, WeightWSM>& inner_entry :
171  m_neighbours_and_weights[v1]) {
172  const VertexWSM& v2 = inner_entry.first;
173  if (v2 > v1) {
174  break;
175  }
176  weights.emplace_back(inner_entry.second);
177  }
178  }
179  TKET_ASSERT(weights.size() == m_number_of_edges);
180  return weights;
181 }
