25const std::vector<std::size_t>& NeighboursFromArchitecture::operator()(
26 std::size_t vertex) {
27 const auto num_vertices = m_arch_mapping.number_of_vertices();
28 // GCOVR_EXCL_START
29 TKET_ASSERT(
30 vertex < num_vertices ||
31 AssertMessage() << "get_neighbours: invalid vertex " << vertex
32 << " (only have " << num_vertices << " vertices)");
33 // GCOVR_EXCL_STOP
34 auto& neighbours = m_cached_neighbours[vertex];
35 if (!neighbours.empty()) {
36 // Already cached.
37 return neighbours;
38 }
39
40 // OK, if a vertex is isolated (has no neighbours) then this is wasteful;
41 // however this case should almost never occur in practice.
42
43 const auto& source_node = m_arch_mapping.get_node(vertex);
44 const auto neighbour_nodes =
45 m_arch_mapping.get_architecture().get_neighbour_nodes(source_node);
46
47 neighbours.reserve(neighbour_nodes.size());
48
49 for (const Node& node : neighbour_nodes) {
50 const auto neighbour_vertex = m_arch_mapping.get_vertex(node);
51 // GCOVR_EXCL_START
52 TKET_ASSERT(
53 neighbour_vertex != vertex ||
54 AssertMessage()
55 << "get_neighbours: vertex " << vertex << " for node "
56 << node.repr() << " has " << neighbour_nodes.size()
57 << " neighbours, and lists itself as a neighbour (loops not "
58 "allowed)");
59 // GCOVR_EXCL_STOP
60 neighbours.push_back(neighbour_vertex);
61 }
62 std::sort(neighbours.begin(), neighbours.end());
63 return neighbours;
64}
