36 ReductionResult NeighboursReducer::reduce(
37  std::pair<VertexWSM, VertexWSM> assignment, DomainsAccessor& accessor,
38  std::set<VertexWSM>& work_set) {
39  const std::vector<std::pair<VertexWSM, WeightWSM>>&
40  target_neighbours_and_weights =
41  m_target_ndata.get_neighbours_and_weights(assignment.second);
42  auto result = ReductionResult::SUCCESS;
43 
44  for (const std::pair<VertexWSM, WeightWSM>& p_neighbour_entry :
45  m_pattern_ndata.get_neighbours_and_weights(assignment.first)) {
46  const VertexWSM& p_neighbour = p_neighbour_entry.first;
47  const std::set<VertexWSM>& domain = accessor.get_domain(p_neighbour);
48  if (other_vertex_reduction_can_be_skipped_by_symmetry(
49  domain, accessor, assignment.first, p_neighbour)) {
50  continue;
51  }
52  fill_intersection_ignoring_second_elements(
53  domain, target_neighbours_and_weights, work_set);
54 
55  switch (accessor.overwrite_domain_with_set_swap(p_neighbour, work_set)) {
56  case ReductionResult::SUCCESS:
57  break;
58  case ReductionResult::NOGOOD:
59  return ReductionResult::NOGOOD;
60  case ReductionResult::NEW_ASSIGNMENTS:
61  // We'll still continue, just to finish this assignment off.
62  result = ReductionResult::NEW_ASSIGNMENTS;
63  }
64  }
65  return result;
66 }
