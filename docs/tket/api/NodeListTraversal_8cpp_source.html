29 // Given the index of a valid entry in domain_data_for_pv.entries,
30 // find the last node index which we need to check,
31 // for all the nodes that share this domain.
32 static unsigned get_final_node_index_for_shared_domain(
33  const NodesRawData::DomainData& domain_data_for_pv, unsigned entries_index,
34  const NodesRawData& raw_data) {
35  TKET_ASSERT(entries_index < domain_data_for_pv.entries.size());
36  if (entries_index + 1 == domain_data_for_pv.entries.size()) {
37  // It's the final domain in the list.
38  return raw_data.current_node_index();
39  }
40  return domain_data_for_pv.entries[entries_index + 1].node_index - 1;
41 }

43 std::set<VertexWSM> NodeListTraversal::get_used_target_vertices() const {
44  std::set<VertexWSM> target_vertices;
45 
46  // Examine all PV.
47  for (unsigned pv = 0; pv < m_raw_data.domains_data.size(); ++pv) {
48  const NodesRawData::DomainData& domain_data = m_raw_data.domains_data[pv];
49 
50  // Add Dom(PV) at all valid nodes.
51  const unsigned domain_data_entries_size = domain_data.entries.size();
52  for (unsigned entries_index = 0; entries_index < domain_data_entries_size;
53  ++entries_index) {
54  const unsigned& node_index =
55  domain_data.entries[entries_index].node_index;
56 
57  // Nogood nodes are ignored.
58  // So, check that at least one node sharing this domain
59  // is not a nogood; otherwise skip.
60  if (m_raw_data.nodes_data[node_index].nogood) {
61  bool some_good_node_shares_this_domain = false;
62  const unsigned final_node_index =
63  get_final_node_index_for_shared_domain(
64  domain_data, entries_index, m_raw_data);
65  for (unsigned ii = node_index + 1; ii <= final_node_index; ++ii) {
66  if (!m_raw_data.nodes_data[ii].nogood) {
67  some_good_node_shares_this_domain = true;
68  break;
69  }
70  }
71  if (!some_good_node_shares_this_domain) {
72  // Ignore this domain: all nodes sharing it are nogood.
73  continue;
74  }
75  }
76  const std::set<VertexWSM>& domain =
77  domain_data.entries[entries_index].domain;
78  if (target_vertices.empty()) {
79  target_vertices = domain;
80  } else {
81  for (VertexWSM tv : domain) {
82  target_vertices.insert(tv);
83  }
84  }
85  }
86  }
87  return target_vertices;
88 }

90 bool NodeListTraversal::move_up() {
91  if (m_raw_data.nodes_data.size() <= 1) {
92  // We will run out of nodes when we pop.
93  // So, this is the end of the search!
94  return false;
95  }
96  m_raw_data.nodes_data.pop();
97 
98  // Keep popping more nodes off, until we reach a node which is NOT a nogood
99  // (i.e., a "good" node; although not standard terminology!)
100  while (m_raw_data.nodes_data.top().nogood) {
101  m_raw_data.nodes_data.pop();
102  if (m_raw_data.nodes_data.empty()) {
103  return false;
104  }
105  }
106 
107  // Now we've reached a "good" (not nogood) node.
108  const unsigned current_node_index = m_raw_data.current_node_index();
109 
110  for (unsigned pv = 0; pv < m_raw_data.domains_data.size(); ++pv) {
111  NodesRawData::DomainData& domain_data = m_raw_data.domains_data[pv];
112  while (domain_data.entries.top().node_index > current_node_index) {
113  // Data for any higher node_index is simply junk.
114  domain_data.entries.pop();
115  TKET_ASSERT(!domain_data.entries.empty());
116  }
117  }
118  return true;
119 }

121 // Returns false if the current node becomes invalid.
122 // Also handles the new assignment PV->y, if necessary,
123 // for the CURRENT node (not new node).
124 static bool when_moving_down_check_current_domain_size(
125  const std::set<VertexWSM>& existing_domain, NodesRawData& raw_data,
126  VertexWSM p_vertex, VertexWSM t_vertex) {
127  switch (existing_domain.size()) {
128  case 0:
129  TKET_ASSERT(false);
130  return false;
131  case 1:
132  // Already Dom(PV) = {TV}, so it will become a nogood.
133  raw_data.get_current_node_nonconst().nogood = true;
134  return false;
135  case 2: {
136  // Dom(PV) = {TV, y}, so it will become Dom(PV) = {y}
137  // at this node, so we need a new assignment PV->y.
138  // The elements of a set of size 2 can be found using
139  // cbegin and crbegin.
140  VertexWSM tv_other = *existing_domain.cbegin();
141  if (tv_other == t_vertex) {
142  tv_other = *existing_domain.crbegin();
143  }
144  raw_data.get_current_node_nonconst().new_assignments.emplace_back(
145  p_vertex, tv_other);
146  break;
147  }
148  default:
149  break;
150  }
151  return true;
152 }

154 // We've taken care of erasing the TV from the existing domain,
155 // now we must create the new node with the new assignment PV->TV.
156 static void complete_move_down(
157  NodesRawData& raw_data, NodesRawData::DomainData& data_for_this_pv,
158  VertexWSM p_vertex, VertexWSM t_vertex) {
159  // The newly created node is "junk".
160  raw_data.nodes_data.push();
161 
162  // Recall that "empty" actually means "check a previous vector".
163  raw_data.get_current_node_nonconst().unassigned_vertices_superset.clear();
164 
165  auto& new_node_data = raw_data.nodes_data.top();
166  new_node_data.new_assignments.resize(1);
167  new_node_data.new_assignments[0] = {p_vertex, t_vertex};
168  new_node_data.nogood = false;
169 
170  new_node_data.scalar_product =
171  raw_data.nodes_data.one_below_top().scalar_product;
172 
173  new_node_data.total_p_edge_weights =
174  raw_data.nodes_data.one_below_top().total_p_edge_weights;
175 
176  // We must also create the domain Dom(PV) = {TV}.
177  data_for_this_pv.entries.push();
178  data_for_this_pv.entries.top().node_index = raw_data.current_node_index();
179 
180  // The domain data for the current node is now "junk".
181  std::set<VertexWSM>& new_domain = data_for_this_pv.entries.top().domain;
182  new_domain.clear();
183  new_domain.insert(t_vertex);
184 }

186 void NodeListTraversal::move_down(VertexWSM p_vertex, VertexWSM t_vertex) {
187  // We can only move down from a valid, fully reduced node.
188  bool existing_node_valid = !m_raw_data.get_current_node().nogood;
189  TKET_ASSERT(existing_node_valid);
190  TKET_ASSERT(m_raw_data.get_current_node().new_assignments.empty());
191 
192  NodesRawData::DomainData& data_for_this_pv =
193  m_raw_data.domains_data.at(p_vertex);
194  std::set<VertexWSM>& existing_domain = data_for_this_pv.entries.top().domain;
195  auto iter = existing_domain.find(t_vertex);
196 
197  // TV must be present!
198  TKET_ASSERT(iter != existing_domain.end());
199  existing_node_valid = when_moving_down_check_current_domain_size(
200  existing_domain, m_raw_data, p_vertex, t_vertex);
201 
202  // Next, we need to erase TV from the existing domain.
203  if (existing_node_valid) {
204  if (data_for_this_pv.entries.top().node_index ==
205  m_raw_data.current_node_index()) {
206  // The data is not shared by previous nodes, so we can just overwrite it
207  // in place.
208  existing_domain.erase(iter);
209  } else {
210  data_for_this_pv.entries.push();
211  data_for_this_pv.entries.top().node_index =
212  m_raw_data.current_node_index();
213  data_for_this_pv.entries.top().domain =
214  data_for_this_pv.entries.one_below_top().domain;
215  TKET_ASSERT(data_for_this_pv.entries.top().domain.erase(t_vertex) == 1);
216  }
217  }
218 
219  complete_move_down(m_raw_data, data_for_this_pv, p_vertex, t_vertex);
220 }

222 static void fill_nogood_or_new_assignment_in_all_shared_nodes(
223  VertexWSM pv, const std::set<VertexWSM>& new_domain,
224  unsigned data_for_this_pv_entry_index,
225  const NodesRawData::DomainData& data_for_this_pv, NodesRawData& raw_data) {
226  // Which nodes share the domain?
227  const unsigned final_node_index = get_final_node_index_for_shared_domain(
228  data_for_this_pv, data_for_this_pv_entry_index, raw_data);
229 
230  if (new_domain.empty()) {
231  // The nodes are all nogoods.
232  for (unsigned jj =
233  data_for_this_pv.entries[data_for_this_pv_entry_index].node_index;
234  jj <= final_node_index; ++jj) {
235  raw_data.nodes_data[jj].nogood = true;
236  }
237  return;
238  }
239 
240  // It's the same new assignment PV->TV to be added to all nodes.
241  const auto new_assignment = std::make_pair(pv, *new_domain.cbegin());
242 
243  for (unsigned jj =
244  data_for_this_pv.entries[data_for_this_pv_entry_index].node_index;
245  jj <= final_node_index; ++jj) {
246  // Of course, some shared nodes might be nogoods for some OTHER reason,
247  // related to other domains we don't know about.
248  if (!raw_data.nodes_data[jj].nogood) {
249  raw_data.nodes_data[jj].new_assignments.emplace_back(new_assignment);
250  }
251  }
252 }

254 void NodeListTraversal::erase_impossible_assignment(
255  std::pair<VertexWSM, VertexWSM> impossible_assignment) {
256  auto& data_for_this_pv =
257  m_raw_data.domains_data.at(impossible_assignment.first);
258 
259  const unsigned size = data_for_this_pv.entries.size();
260  for (unsigned ii = 0; ii < size; ++ii) {
261  TKET_ASSERT(
262  data_for_this_pv.entries[ii].node_index <=
263  m_raw_data.current_node_index());
264 
265  if (m_raw_data.nodes_data[data_for_this_pv.entries[ii].node_index].nogood) {
266  // Don't waste time with nogood nodes.
267  continue;
268  }
269  std::set<VertexWSM>& domain = data_for_this_pv.entries[ii].domain;
270  if (domain.erase(impossible_assignment.second) == 0 || domain.size() >= 2) {
271  // EITHER nothing changed, OR the change had no significant effect.
272  continue;
273  }
274  // Now, it's EITHER a nogood, OR a new assignment is created.
275  // Anyway, we go through all the nodes which share this domain and
276  // process them (the same Domain(pv) may be shared across several nodes).
277  fill_nogood_or_new_assignment_in_all_shared_nodes(
278  impossible_assignment.first, domain, ii, data_for_this_pv, m_raw_data);
279  }
