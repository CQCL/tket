26 NodesRawData::NodesRawData(
27  const DomainInitialiser::InitialDomains& initial_domains) {
28  nodes_data.push();
29  auto& node = nodes_data.top();
30  node.nogood = false;
31  node.scalar_product = 0;
32  node.total_p_edge_weights = 0;
33 
34  TKET_ASSERT(initial_domains.size() >= 2);
35  domains_data.resize(initial_domains.size());
36 
37  for (unsigned pv = 0; pv < initial_domains.size(); ++pv) {
38  const std::set<VertexWSM>& domain = initial_domains[pv];
39  auto& domain_data = domains_data[pv];
40  domain_data.entries.push();
41  domain_data.entries[0].domain = domain;
42  domain_data.entries[0].node_index = 0;
43 
44  switch (domain.size()) {
45  case 0: {
46  std::stringstream ss;
47  ss << "NodesRawData: Domain(" << pv << ") is empty!";
48  throw std::runtime_error(ss.str());
49  }
50  case 1:
51  node.new_assignments.emplace_back(pv, *domain.cbegin());
52  break;
53  default:
54  node.unassigned_vertices_superset.push_back(pv);
55  }
56  }
57 }

71 std::string NodesRawData::NodeData::str() const {
72  std::stringstream ss;
73  if (nogood) {
74  ss << "##NOGOOD!## ";
75  }
76  ss << "Has " << new_assignments.size() << " ass.: [ ";
77  for (const std::pair<VertexWSM, VertexWSM>& entry : new_assignments) {
78  ss << entry.first << ":" << entry.second << " ";

80  ss << "]; sc.prod " << scalar_product << "; p-edge weight "
