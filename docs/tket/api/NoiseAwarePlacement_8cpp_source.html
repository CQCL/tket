40double NoiseAwarePlacement::cost_placement(
41 const boost::bimap<Qubit, Node>& map, const Circuit& circ_,
42 const QubitGraph& q_graph) const {
43 double cost = 0.0;
44 if (circ_.n_gates() == 0 || circ_.n_qubits() == 0) {
45 return cost;
46 }
47 const int approx_depth = circ_.n_gates() / circ_.n_qubits() + 1;
48 // constants for scaling single qubit error
49 constexpr double c1 = 0.5;
50 constexpr double d1 = 1 - 1 / c1;
51 for (auto [qb, node] : map) {
52 node_set_t neighbours = this->architecture_.get_neighbour_nodes(node);
53 double edge_sum = 1.0;
54 for (const Node& neighbour : neighbours) {
55 double fwd_edge_weighting = 1.0, bck_edge_weighting = 1.0;
56 // check if neighbour node is mapped
57 auto nei_qb_it = map.right.find(neighbour);
58 if (nei_qb_it == map.right.end()) continue;
59
60 auto place_interactions_boost = [&](unsigned edge_v) {
61 return this->maximum_pattern_depth_ - edge_v + 1;
62 };
63
64 // check if either directed interaction exists
65 // if edge is used by interaction in mapping, weight edge higher
66 unsigned edge_val = q_graph.get_connection_weight(qb, nei_qb_it->second);
67 if (edge_val) {
68 fwd_edge_weighting += place_interactions_boost(edge_val);
69 } else {
70 edge_val = q_graph.get_connection_weight(nei_qb_it->second, qb);
71 if (edge_val) {
72 bck_edge_weighting += place_interactions_boost(edge_val);
73 }
74 }
75 gate_error_t fwd_error =
76 this->characterisation_.get_error({node, neighbour});
77 gate_error_t bck_error =
78 this->characterisation_.get_error({neighbour, node});
79
80 if (fwd_error < 1.0 && bck_error < 1.0) {
81 edge_sum += fwd_edge_weighting * (1.0 - fwd_error);
82 edge_sum += bck_edge_weighting * (1.0 - bck_error);
83 }
84 }
85 // bigger edge sum -> smaller cost
86 cost += 1.0 / (edge_sum);
87 // add error rate of node
88 gate_error_t single_error = this->characterisation_.get_error(node);
89 cost += d1 + 1.0 / ((1.0 - single_error) + c1);
90 readout_error_t readout_error =
91 this->characterisation_.get_readout_error(node);
92 if (readout_error) {
93 cost += (d1 + 1.0 / ((1.0 - readout_error) + c1)) / (approx_depth * 20);
94 }
95 }
96 return cost;
97}

99std::vector<boost::bimap<Qubit, Node>> NoiseAwarePlacement::rank_maps(
100 const std::vector<boost::bimap<Qubit, Node>>& placement_maps,
101 const Circuit& circ_,
102 const std::vector<GraphPlacement::WeightedEdge>& pattern_edges) const {
103 // each map is costed, sorted and returned
104 // only equal best costed maps are returned
105 std::vector<boost::bimap<Qubit, Node>> return_placement_maps;
106 double best_cost = 0;
107 QubitGraph q_graph =
108 this->construct_pattern_graph(pattern_edges, circ_.n_qubits());
109 for (const boost::bimap<Qubit, Node>& map : placement_maps) {
110 double cost = this->cost_placement(map, circ_, q_graph);
111 if (return_placement_maps.empty() || cost < best_cost) {
112 best_cost = cost;
113 return_placement_maps = {map};
114 } else if (cost == best_cost) {
115 return_placement_maps.push_back(map);
116 }
117 }
118 return return_placement_maps;
119}

121std::vector<std::map<Qubit, Node>> NoiseAwarePlacement::get_all_placement_maps(
122 const Circuit& circ_, unsigned matches) const {
123 std::vector<WeightedEdge> weighted_pattern_edges =
124 this->default_pattern_weighting(circ_);
125
126 if (weighted_pattern_edges.empty()) {
127 // => there are no two-qubit gates in the input circuit
128 // in this case, as this method is "noise-aware", assign
129 // qubits in the circuit to the lowest-error architecture nodes
130 std::vector<std::pair<gate_error_t, Node>> all_node_errors;
131 for (const Node& node : this->architecture_.nodes()) {
132 all_node_errors.push_back(
133 {this->characterisation_.get_error(node), node});
134 }
135 // make sure all_node_errors goes from best->worst error rates
136 // N.B. `get_error` returns "0", i.e. no error, if the `Node` passed
137 // is not held in the characterisation
138 std::sort(
139 all_node_errors.begin(), all_node_errors.end(),
140 [](const auto& lhs, const auto& rhs) {
141 // Compare based on the float values in descending order
142 return lhs.first < rhs.first;
143 });
144 std::vector<Qubit> circ_qubits = circ_.all_qubits();
145 TKET_ASSERT(all_node_errors.size() >= circ_qubits.size());
146 std::map<Qubit, Node> placement_map;
147 // assign circuit qubits to Node with best error rate
148 for (std::size_t i = 0; i < circ_qubits.size(); i++) {
149 placement_map.insert({circ_qubits[i], all_node_errors[i].second});
150 }
151 return {placement_map};
152 }
153
154 std::vector<boost::bimap<Qubit, Node>> placement_maps =
155 this->get_all_weighted_subgraph_monomorphisms(
156 circ_, weighted_pattern_edges, true);
157 std::vector<boost::bimap<Qubit, Node>> ranked_placement_maps =
158 this->rank_maps(placement_maps, circ_, weighted_pattern_edges);
159 std::vector<std::map<Qubit, Node>> all_qmaps;
160 QubitGraph::UndirectedConnGraph pattern_graph =
161 this->construct_pattern_graph(weighted_pattern_edges, circ_.n_qubits())
162 .get_undirected_connectivity();
163 for (unsigned i = 0; i < ranked_placement_maps.size() && i < matches; i++) {
164 all_qmaps.push_back(convert_bimap(ranked_placement_maps[i], pattern_graph));
165 }
166 return all_qmaps;
167}
