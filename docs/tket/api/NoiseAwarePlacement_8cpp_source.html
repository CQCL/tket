41 double NoiseAwarePlacement::cost_placement(
42  const boost::bimap<Qubit, Node>& map, const Circuit& circ_,
43  const QubitGraph& q_graph) const {
44  double cost = 0.0;
45  if (circ_.n_gates() == 0 || circ_.n_qubits() == 0) {
46  return cost;
47  }
48  const int approx_depth = circ_.n_gates() / circ_.n_qubits() + 1;
49  // constants for scaling single qubit error
50  constexpr double c1 = 0.5;
51  constexpr double d1 = 1 - 1 / c1;
52  for (auto [qb, node] : map) {
53  node_set_t neighbours = this->architecture_.get_neighbour_nodes(node);
54  double edge_sum = 1.0;
55  for (const Node& neighbour : neighbours) {
56  double fwd_edge_weighting = 1.0, bck_edge_weighting = 1.0;
57  // check if neighbour node is mapped
58  auto nei_qb_it = map.right.find(neighbour);
59  if (nei_qb_it == map.right.end()) continue;
60 
61  auto place_interactions_boost = [&](unsigned edge_v) {
62  return this->maximum_pattern_depth_ - edge_v + 1;
63  };
64 
65  // check if either directed interaction exists
66  // if edge is used by interaction in mapping, weight edge higher
67  unsigned edge_val = q_graph.get_connection_weight(qb, nei_qb_it->second);
68  if (edge_val) {
69  fwd_edge_weighting += place_interactions_boost(edge_val);
70  } else {
71  edge_val = q_graph.get_connection_weight(nei_qb_it->second, qb);
72  if (edge_val) {
73  bck_edge_weighting += place_interactions_boost(edge_val);
74  }
75  }
76  gate_error_t fwd_error =
77  this->characterisation_.get_error({node, neighbour});
78  gate_error_t bck_error =
79  this->characterisation_.get_error({neighbour, node});
80  edge_sum += fwd_edge_weighting * (1.0 - fwd_error);
81  edge_sum += bck_edge_weighting * (1.0 - bck_error);
82  }
83  // bigger edge sum -> smaller cost
84  cost += 1.0 / (edge_sum);
85  // add error rate of node
86  gate_error_t single_error = this->characterisation_.get_error(node);
87  cost += d1 + 1.0 / ((1.0 - single_error) + c1);
88  readout_error_t readout_error =
89  this->characterisation_.get_readout_error(node);
90  if (readout_error) {
91  cost += (d1 + 1.0 / ((1.0 - readout_error) + c1)) / (approx_depth * 20);
92  }
93  }
94  return cost;
95 }

97 std::vector<boost::bimap<Qubit, Node>> NoiseAwarePlacement::rank_maps(
98  const std::vector<boost::bimap<Qubit, Node>>& placement_maps,
99  const Circuit& circ_,
100  const std::vector<GraphPlacement::WeightedEdge>& pattern_edges) const {
101  // each map is costed, sorted and returned
102  // only equal best costed maps are returned
103  std::vector<boost::bimap<Qubit, Node>> return_placement_maps;
104  double best_cost = 0;
105  QubitGraph q_graph =
106  this->construct_pattern_graph(pattern_edges, circ_.n_qubits());
107  for (const boost::bimap<Qubit, Node>& map : placement_maps) {
108  double cost = this->cost_placement(map, circ_, q_graph);
109  if (return_placement_maps.empty() || cost < best_cost) {
110  best_cost = cost;
111  return_placement_maps = {map};
112  } else if (cost == best_cost) {
113  return_placement_maps.push_back(map);
114  }
115  }
116  return return_placement_maps;
117 }

119 std::vector<std::map<Qubit, Node>> NoiseAwarePlacement::get_all_placement_maps(
120  const Circuit& circ_, unsigned matches) const {
121  std::vector<WeightedEdge> weighted_pattern_edges =
122  this->default_pattern_weighting(circ_);
123  std::vector<boost::bimap<Qubit, Node>> placement_maps =
124  this->get_all_weighted_subgraph_monomorphisms(
125  circ_, weighted_pattern_edges, true);
126  std::vector<boost::bimap<Qubit, Node>> ranked_placement_maps =
127  this->rank_maps(placement_maps, circ_, weighted_pattern_edges);
128  std::vector<std::map<Qubit, Node>> all_qmaps;
129  QubitGraph::UndirectedConnGraph pattern_graph =
130  this->construct_pattern_graph(weighted_pattern_edges, circ_.n_qubits())
131  .get_undirected_connectivity();
132  for (unsigned i = 0; i < ranked_placement_maps.size() && i < matches; i++) {
133  all_qmaps.push_back(convert_bimap(ranked_placement_maps[i], pattern_graph));
134  }
135  return all_qmaps;
136 }
