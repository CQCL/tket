34bool OpJsonFactory::register_method(
35 const OpType& type, JsonConstruct create_method,
36 JsonProduce produce_method) {
37 if (auto it = c_methods_().find(type); it == c_methods_().end()) {
38 c_methods_()[type] = create_method;
39 p_methods_()[type] = produce_method;
40 return true;
41 }
42 return false;
43}

45Op_ptr OpJsonFactory::from_json(const nlohmann::json& j) {
46 const auto type = j.at("type").get<OpType>();
47 if (auto it = c_methods_().find(type); it != c_methods_().end()) {
48 return it->second(j);
49 }
50 throw JsonError(
51 "No from_json conversion for type " + optypeinfo().at(type).name);
52}

54nlohmann::json OpJsonFactory::to_json(const Op_ptr& op) {
55 const OpType& type = op->get_type();
56 if (auto it = p_methods_().find(type); it != p_methods_().end()) {
57 return it->second(op);
58 }
59 throw JsonError(
60 "No to_json conversion registered for type: " +
61 optypeinfo().at(type).name);
