28void from_json(const nlohmann::json& j, Op_ptr& op) {
29 OpType optype = j.at("type").get<OpType>();
30 if (is_barrier_type(optype)) {
31 op = BarrierOp::deserialize(j);
32 } else if (is_box_type(optype)) {
33 op = Box::deserialize(j);
34 } else if (optype == OpType::Conditional) {
35 op = Conditional::deserialize(j);
36 } else if (optype == OpType::WASM) {
37 op = WASMOp::deserialize(j);
38 } else if (is_opaque_classical_type(optype)) {
39 op = OpaqueClassicalOp::deserialize(j);
40 } else if (optype == OpType::ClExpr) {
41 op = ClExprOp::deserialize(j);
42 } else if (is_classical_type(optype)) {
43 op = ClassicalOp::deserialize(j);
44 } else if (is_gate_type(optype)) {
45 op = Gate::deserialize(j);
46 } else {
47 throw JsonError(
48 "Deserialization not yet implemented for " +
49 optypeinfo().at(optype).name);
50 }
51}
