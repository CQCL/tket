27 void from_json(const nlohmann::json& j, Op_ptr& op) {
28  OpType optype = j.at("type").get<OpType>();
29  if (is_metaop_type(optype)) {
30  op = MetaOp::deserialize(j);
31  } else if (is_box_type(optype)) {
32  op = Box::deserialize(j);
33  } else if (optype == OpType::Conditional) {
34  op = Conditional::deserialize(j);
35  } else if (optype == OpType::WASM) {
36  op = WASMOp::deserialize(j);
37  } else if (is_classical_type(optype)) {
38  op = ClassicalOp::deserialize(j);
39  } else if (is_gate_type(optype)) {
40  op = Gate::deserialize(j);
41  } else {
42  throw JsonError(
43  "Deserialization not yet implemented for " +
44  optypeinfo().at(optype).name);
45  }
46 }
