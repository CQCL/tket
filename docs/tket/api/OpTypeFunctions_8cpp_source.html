255bool is_oneway_type(OpType optype) {
256 // This set should contain only gates for which an dagger is nonsensical
257 // or we do not yet have the dagger gate as an OpType.
258 // If the gate can have an dagger, define it in the dagger() method.
259 static const OpTypeSet no_defined_inverse = {
260 OpType::Input, OpType::Output, OpType::Measure, OpType::ClInput,
261 OpType::ClOutput, OpType::Barrier, OpType::Reset, OpType::Collapse,
262 OpType::PhasePolyBox, OpType::Create, OpType::Discard};
263 return find_in_set(optype, no_defined_inverse);
