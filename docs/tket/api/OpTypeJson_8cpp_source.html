23 // map from OpType name to OpType
24 // Relies on unique OpType names.
25 const std::map<std::string, OpType>& name_to_optype() {
26  auto mapfiller = []() {
27  std::map<std::string, OpType> fill_map;
28  for (const auto& info : optypeinfo()) {
29  fill_map.insert({info.second.name, info.first});
30  }
31  return fill_map;
32  };
33  static auto final_map =
34  std::make_unique<const std::map<std::string, OpType>>(mapfiller());
35  return *final_map;
36 }
