23// map from OpType name to OpType
24// Relies on unique OpType names.
25const std::map<std::string, OpType>& name_to_optype() {
26 auto mapfiller = []() {
27 std::map<std::string, OpType> fill_map;
28 for (const auto& info : optypeinfo()) {
29 fill_map.insert({info.second.name, info.first});
30 }
31 return fill_map;
32 };
33 static auto final_map =
34 std::make_unique<const std::map<std::string, OpType>>(mapfiller());
35 return *final_map;
36}

38void to_json(nlohmann::json& j, const OpType& type) {
39 j = optypeinfo().at(type).name;

42void from_json(const nlohmann::json& j, OpType& type) {
43 const std::string name = j.get<std::string>();
44 const auto result = name_to_optype().find(name);
45 if (result == name_to_optype().end()) {
46 throw JsonError("No OpType with name " + name);
47 }
48
49 type = result->second;
50}
