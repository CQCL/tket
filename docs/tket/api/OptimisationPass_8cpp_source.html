37Transform peephole_optimise_2q(bool allow_swaps) {
38 return (
39 synthesise_tket() >> two_qubit_squash(allow_swaps) >>
40 decompose_multi_qubits_CX() >> clifford_simp(allow_swaps) >>
41 synthesise_tket());
42}

44Transform full_peephole_optimise(bool allow_swaps, OpType target_2qb_gate) {
45 switch (target_2qb_gate) {
46 case OpType::CX:
47 return (
48 synthesise_tket() >> two_qubit_squash(false) >>
49 clifford_simp(allow_swaps) >> synthesise_tket() >>
50 two_qubit_squash(allow_swaps) >> three_qubit_squash() >>
51 clifford_simp(allow_swaps) >> synthesise_tket());
52 case OpType::TK2:
53 return (
54 synthesise_tk() >> two_qubit_squash(OpType::TK2, 1., allow_swaps) >>
55 clifford_simp(allow_swaps, OpType::TK2) >>
56 two_qubit_squash(OpType::TK2, 1., allow_swaps) >> synthesise_tk() >>
57 three_qubit_squash(OpType::TK2) >>
58 clifford_simp(allow_swaps, OpType::TK2) >>
59 two_qubit_squash(OpType::TK2, 1., allow_swaps) >> synthesise_tk());
60 default:
61 throw std::invalid_argument("Invalid target 2-qubit gate");
62 }
63}

65Transform clifford_simp(bool allow_swaps, OpType target_2qb_gate) {
66 if (target_2qb_gate != OpType::CX && target_2qb_gate != OpType::TK2) {
67 throw std::invalid_argument("Invalid target 2-qubit gate");
68 }
69 return decompose_cliffords_std(target_2qb_gate == OpType::CX) >>
70 clifford_reduction(allow_swaps) >>
71 (target_2qb_gate == OpType::CX ? decompose_multi_qubits_CX()
72 : decompose_multi_qubits_TK2()) >>
73 singleq_clifford_sweep() >> squash_1qb_to_tk1();
74}

76Transform synthesise_tk() {
77 Transform seq = commute_through_multis() >> remove_redundancies();
78 Transform rep = repeat(seq);
79 Transform synth = decompose_multi_qubits_TK2() >> remove_redundancies() >>
80 rep >> squash_1qb_to_tk1();
81 Transform small_part = remove_redundancies() >> rep >> squash_1qb_to_tk1();
82 Transform repeat_synth = repeat_with_metric(
83 small_part, [](const Circuit &circ) { return circ.n_vertices(); });
84 return synth >> repeat_synth >> rebase_TK() >> remove_redundancies();
85}

98static Transform CXs_from_phase_gadgets(CXConfigType cx_config) {
99 return Transform([=](Circuit &circ) {
100 bool success = false;
101 VertexList bin;
102 auto [i, end] = boost::vertices(circ.dag);
103 for (auto next = i; i != end; i = next) {
104 ++next;
105 Vertex a = *i;
106 Op_ptr op = circ.get_Op_ptr_from_Vertex(a);
107 if (op->get_type() == OpType::PhaseGadget) {
108 unsigned n_qubits = op->n_qubits();
109 Circuit replacement =
110 phase_gadget(n_qubits, op->get_params()[0], cx_config);
111 Subcircuit sub = circ.singleton_subcircuit(a);
112 circ.substitute(replacement, sub, Circuit::VertexDeletion::Yes);
113 success = true;
114 }
115 }
116 circ.decompose_boxes_recursively();
117 return success;
118 });
119}

121Transform optimise_via_PhaseGadget(CXConfigType cx_config) {

128Transform synthesise_UMD() {
129 return Transform([](Circuit &circ) {
130 bool success = (synthesise_tket() >> decompose_ZX() >>
131 decompose_MolmerSorensen() >> squash_1qb_to_tk1())
132 .apply(circ);
133 VertexList bin;
134 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
135 const Op_ptr op_ptr = circ.get_Op_ptr_from_Vertex(v);
136 OpType type = op_ptr->get_type();
137 if (type == OpType::TK1) {
138 std::vector<Expr> tk1_angles =
139 as_gate_ptr(op_ptr)->get_tk1_angles();
140 Circuit in_circ = CircPool::tk1_to_PhasedXRz(
141 tk1_angles[0], tk1_angles[1], tk1_angles[2]);
142 remove_redundancies().apply(in_circ);
143 Subcircuit sub = circ.singleton_subcircuit(v);
144 bin.push_back(v);
145 circ.substitute(in_circ, sub, Circuit::VertexDeletion::No);
146 circ.add_phase(tk1_angles[3]);
147 success = true;
148 }
149 }
150 circ.remove_vertices(
151 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
152 return success;
153 }) >>
154 rebase_UMD() >> remove_redundancies();
155}
