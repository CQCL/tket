43 Transform full_peephole_optimise(bool allow_swaps, OpType target_2qb_gate) {
44  switch (target_2qb_gate) {
45  case OpType::CX:
46  return (
47  synthesise_tket() >> two_qubit_squash(false) >>
48  clifford_simp(allow_swaps) >> synthesise_tket() >>
49  two_qubit_squash(allow_swaps) >> three_qubit_squash() >>
50  clifford_simp(allow_swaps) >> synthesise_tket());
51  case OpType::TK2:
52  return (
53  synthesise_tk() >> two_qubit_squash(OpType::TK2, 1., allow_swaps) >>
54  clifford_simp(allow_swaps) >>
55  two_qubit_squash(OpType::TK2, 1., allow_swaps) >> synthesise_tk() >>
56  three_qubit_squash(OpType::TK2) >> clifford_simp(allow_swaps) >>
57  two_qubit_squash(OpType::TK2, 1., allow_swaps) >> synthesise_tk());
58  default:
59  throw std::invalid_argument("Invalid target 2-qubit gate");
60  }
61 }

78 Transform synthesise_tk() {
79  Transform seq = commute_through_multis() >> remove_redundancies();
80  Transform rep = repeat(seq);
81  Transform synth = decompose_multi_qubits_TK2() >> remove_redundancies() >>
82  rep >> squash_1qb_to_tk1();
83  Transform small_part = remove_redundancies() >> rep >> squash_1qb_to_tk1();
84  Transform repeat_synth = repeat_with_metric(
85  small_part, [](const Circuit &circ) { return circ.n_vertices(); });
86  return synth >> repeat_synth >> rebase_TK() >> remove_redundancies();

100 static Transform CXs_from_phase_gadgets(CXConfigType cx_config) {
101  return Transform([=](Circuit &circ) {
102  bool success = false;
103  VertexList bin;
104  auto [i, end] = boost::vertices(circ.dag);
105  for (auto next = i; i != end; i = next) {
106  ++next;
107  Vertex a = *i;
108  Op_ptr op = circ.get_Op_ptr_from_Vertex(a);
109  if (op->get_type() == OpType::PhaseGadget) {
110  unsigned n_qubits = op->n_qubits();
111  Circuit replacement =
112  phase_gadget(n_qubits, op->get_params()[0], cx_config);
113  Subcircuit sub = {
114  {circ.get_in_edges(a)}, {circ.get_all_out_edges(a)}, {a}};
115  circ.substitute(replacement, sub, Circuit::VertexDeletion::Yes);
116  success = true;
117  }
118  }
119  return success;
120  });
121 }

129 Transform synthesise_OQC() {

131  Transform rep_zx = squash_1qb_to_pqp(OpType::Rx, OpType::Rz) >>
132  commute_through_multis() >> remove_redundancies();

136  return seq.apply(circ);

140 /* Returns a Circuit with only HQS allowed Ops (Rz, PhasedX, ZZMax) */
141 Transform synthesise_HQS() {
142  return Transform([](Circuit &circ) {
143  Transform single_loop =
144  remove_redundancies() >> commute_through_multis() >> reduce_XZ_chains();
145  Transform hqs_loop = remove_redundancies() >> commute_and_combine_HQS2() >>
146  reduce_XZ_chains();
147  Transform main_seq =
148  decompose_multi_qubits_CX() >> clifford_simp() >> decompose_ZX() >>
149  repeat(single_loop) >> decompose_CX_to_HQS2() >> repeat(hqs_loop) >>
150  decompose_ZX_to_HQS1() >> rebase_HQS() >> remove_redundancies();
151  return main_seq.apply(circ);
152  });
153 }

156 Transform synthesise_UMD() {

159  decompose_MolmerSorensen() >> squash_1qb_to_tk1())

161  VertexList bin;
162  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
163  const Op_ptr op_ptr = circ.get_Op_ptr_from_Vertex(v);
164  OpType type = op_ptr->get_type();
165  if (type == OpType::TK1) {
166  std::vector<Expr> tk1_angles =
167  as_gate_ptr(op_ptr)->get_tk1_angles();
168  Circuit in_circ = CircPool::tk1_to_PhasedXRz(
169  tk1_angles[0], tk1_angles[1], tk1_angles[2]);
170  remove_redundancies().apply(in_circ);
171  Subcircuit sub = {
172  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
173  bin.push_back(v);
174  circ.substitute(in_circ, sub, Circuit::VertexDeletion::No);
175  circ.add_phase(tk1_angles[3]);
176  success = true;
177  }
178  }
179  circ.remove_vertices(
180  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
181  return success;
182  }) >>
183  rebase_UMD() >> remove_redundancies();
184 }
