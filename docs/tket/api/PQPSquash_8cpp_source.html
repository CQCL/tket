35PQPSquasher::PQPSquasher(OpType p, OpType q, bool smart_squash, bool reversed)
36 : p_(p),
37 q_(q),
38 smart_squash_(smart_squash),
39 reversed_(reversed),
40 rotation_chain() {
41 if (!(p == OpType::Rx || p == OpType::Ry || p == OpType::Rz) ||
42 !(q == OpType::Rx || q == OpType::Ry || q == OpType::Rz)) {
43 throw std::logic_error("Can only reduce chains of single qubit rotations");
44 }
45 if (p == q) {
46 throw std::logic_error(
47 "Requires two different bases to perform single qubit "
48 "rotations");
49 }
50}

64std::pair<Circuit, Gate_ptr> PQPSquasher::flush(
65 std::optional<Pauli> commutation_colour) const {
66 bool commute_through = false;
67 OpType p = p_, q = q_;
68
69 if (smart_squash_ && commutation_colour.has_value()) {
70 // Using an arbitrary non-zero angle to obtain the commutation for p_/q_.
71 Gate P(p_, {0.123}, 1);
72 Gate Q(q_, {0.123}, 1);
73 if (P.commutes_with_basis(commutation_colour, 0)) {
74 commute_through = true;
75 } else if (Q.commutes_with_basis(commutation_colour, 0)) {
76 commute_through = true;
77 p = q_;
78 q = p_;
79 }
80 }
81
82 // Construct list of merged rotations
83 std::list<Rotation> rots;
84 auto iter = rotation_chain.cbegin();
85 while (iter != rotation_chain.cend()) {
86 // Merge next q rotations
87 rots.push_back(merge_rotations(q, rotation_chain, iter));
88 // Merge next p rotations
89 rots.push_back(merge_rotations(p, rotation_chain, iter));
90 }
91
92 // Perform any cancellations
93 std::list<Rotation>::iterator r = rots.begin();
94 while (r != rots.end()) {
95 if (r->is_id()) {
96 r = rots.erase(r);
97 if (r != rots.begin() && r != rots.end()) {
98 std::prev(r)->apply(*r);
99 r = rots.erase(r);
100 r--;
101 }
102 } else
103 r++;
104 }
105
106 // Extract any P rotations from the beginning and end of the list
107 Expr p1 = 0, p2 = 0;
108 std::list<Rotation>::iterator i1 = rots.begin();
109 if (i1 != rots.end()) {
110 std::optional<Expr> a = i1->angle(p);
111 if (a) {
112 p1 = a.value();
113 rots.pop_front();
114 }
115 }
116 std::list<Rotation>::reverse_iterator i2 = rots.rbegin();
117 if (i2 != rots.rend()) {
118 std::optional<Expr> a = i2->angle(p);
119 if (a) {
120 p2 = a.value();
121 rots.pop_back();
122 }
123 }
124
125 // Finish up:
126 Rotation R = {};
127 for (auto rot : rots) {
128 R.apply(rot);
129 }
130 std::tuple<Expr, Expr, Expr> angles = R.to_pqp(p, q);
131
132 Expr angle_p1 = std::get<0>(angles) + p1;
133 Expr angle_q = std::get<1>(angles);
134 Expr angle_p2 = std::get<2>(angles) + p2;
135 fixup_angles(angle_p1, angle_q, angle_p2, reversed_);
136
137 Circuit replacement(1);
138 Gate_ptr left_over_gate = nullptr;
139 if (!equiv_0(angle_p1, 4)) {
140 if (equiv_0(angle_q, 4) && equiv_0(angle_p2, 4) && commute_through) {
141 left_over_gate =
142 std::make_shared<Gate>(p, std::vector<Expr>{angle_p1}, 1);
143 } else {
144 replacement.add_op<unsigned>(p, angle_p1, {0});
145 }
146 }
147 if (!equiv_0(angle_q, 4)) {
148 replacement.add_op<unsigned>(q, angle_q, {0});
149 }
150 if (!equiv_0(angle_p2, 4)) {
151 if (commute_through) {
152 left_over_gate =
153 std::make_shared<Gate>(p, std::vector<Expr>{angle_p2}, 1);
154 } else {
155 replacement.add_op<unsigned>(p, angle_p2, {0});
156 }
157 }
158 redundancy_removal(replacement);
159 return {replacement, left_over_gate};
160}

168static Rotation merge_rotations(
169 OpType r, const std::vector<Gate_ptr> &chain,
170 std::vector<Gate_ptr>::const_iterator &iter) {
171 Expr total_angle(0);
172 while (iter != chain.end()) {
173 const Gate_ptr rot_op = *iter;
174 if (rot_op->get_type() != r) {
175 break;
176 }
177 total_angle += rot_op->get_params()[0];
178 iter++;
179 }
180 return Rotation(r, total_angle);
181}

183static bool squash_to_pqp(
184 Circuit &circ, OpType q, OpType p, bool strict = false) {
185 bool reverse = true;
186 auto squasher = std::make_unique<PQPSquasher>(p, q, !strict, reverse);
187 return SingleQubitSquash(std::move(squasher), circ, reverse).squash();
188}

190Transform squash_1qb_to_pqp(const OpType &q, const OpType &p, bool strict) {
191 return Transform(
192 [=](Circuit &circ) { return squash_to_pqp(circ, q, p, strict); });
193}

195// To squash to TK1:
196// - we first decompose to ZYZ. Doing this was found to reduce the size of
197// symbolic expressions
198// - we then redecompose to ZXZ, so that we can commute Rz or Rx rotation past
199// multi-qubit gates (most usual multi-qb gates commute with X or Z)
200// - Rz and Rx rotations can then be straight-forwardly combined into TK1s.
201Transform squash_1qb_to_tk1() {
202 return Transforms::decompose_ZY() >>
203 squash_1qb_to_pqp(OpType::Ry, OpType::Rz, true) >>
204 Transforms::decompose_ZX() >>
205 squash_1qb_to_pqp(OpType::Rx, OpType::Rz, true) >>
206 Transforms::decompose_ZXZ_to_TK1();
207}

209static bool fixup_angles(
210 Expr &angle_p1, Expr &angle_q, Expr &angle_p2, bool reversed) {
211 bool success = false;
212 if (reversed) {
213 std::swap(angle_p1, angle_p2);
214 angle_p1 *= -1;
215 angle_q *= -1;
216 angle_p2 *= -1;
217 }
218 if (equiv_val(angle_q, 1., 2) && !equiv_0(angle_p2, 4)) {
219 // Prefer --P(p1-p2)--Q(...)--P(0)--
220 // Only occurs if angle_q is pi or 3pi and angle_p2 is non-zero
221 angle_p1 -= angle_p2;
222 angle_p2 = 0;
223 success = true;
224 } else if (equiv_val(angle_p2, 1., 4)) {
225 // Then prefer --P(p1+p2)--Q(-q)--P(0)--
226 // Only occurs if angle_p2 is pi
227 angle_p1 += 1;
228 angle_q *= -1;
229 angle_p2 = 0;
230 success = true;
231 } else if (equiv_val(angle_p2, 3., 4)) {
232 // Then prefer --P(p1+p2)--Q(-q)--P(0)--
233 // Only occurs if angle_p2 is 3pi
234 angle_p1 += 3;
235 angle_q *= -1;
236 angle_p2 = 0;
237 success = true;
238 } else if (equiv_val(angle_p1, 1., 4) && !equiv_0(angle_p2, 4)) {
239 // Then prefer --P(0)--Q(-q)--P(p1+p2)--
240 // Only occurs if angle_p1 is pi and angle_p2 is non-zero
241 angle_q *= -1;
242 angle_p2 += 1;
243 angle_p1 = 0;
244 success = true;
245 } else if (equiv_val(angle_p1, 3., 4) && !equiv_0(angle_p2, 4)) {
246 // Then prefer --P(0)--Q(-q)--P(p1+p2)--
247 // Only occurs if angle_p1 is 3pi and angle_p2 is non-zero
248 angle_q *= -1;
249 angle_p2 += 3;
250 angle_p1 = 0;
251 success = true;
252 }
253 if (reversed) {
254 std::swap(angle_p1, angle_p2);
255 angle_p1 *= -1;
256 angle_q *= -1;
257 angle_p2 *= -1;
258 }
259 return success;
260}
