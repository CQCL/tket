46 PassPtr gen_rebase_pass(
47  const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
48  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
49  tk1_replacement) {
50  Transform t = Transforms::rebase_factory(
51  allowed_gates, cx_replacement, tk1_replacement);
52 
53  PredicatePtrMap precons;
54  OpTypeSet all_types(allowed_gates);
55  all_types.insert(OpType::Measure);
56  all_types.insert(OpType::Collapse);
57  all_types.insert(OpType::Reset);
58  PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
59  PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
60  std::pair<const std::type_index, PredicatePtr> pair2 =
61  CompilationUnit::make_type_pair(postcon1);
62  PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
63  PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
64  PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
65  // record pass config
66  nlohmann::json j;
67  j["name"] = "RebaseCustom";
68  j["basis_allowed"] = allowed_gates;
69  j["basis_cx_replacement"] = cx_replacement;
70  j["basis_tk1_replacement"] =
71  "SERIALIZATION OF FUNCTIONS IS NOT YET SUPPORTED";
72  return std::make_shared<StandardPass>(precons, t, pc, j);
73 }

181 PassPtr gen_placement_pass(const Placement::Ptr& placement_ptr) {
182  Transform::Transformation trans = [=](Circuit& circ,
183  std::shared_ptr<unit_bimaps_t> maps) {
184  // Fall back to line placement if graph placement fails
185  bool changed;
186  try {
187  changed = placement_ptr->place(circ, maps);
188  } catch (const std::runtime_error& e) {
189  std::stringstream ss;
190  ss << "PlacementPass failed with message: " << e.what()
191  << " Fall back to LinePlacement.";
192  tket_log()->warn(ss.str());
193  Placement::Ptr line_placement_ptr = std::make_shared<LinePlacement>(
194  placement_ptr->get_architecture_ref());
195  changed = line_placement_ptr->place(circ, maps);
196  }
197  return changed;
198  };
199  Transform t = Transform(trans);
200  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
201  PredicatePtr n_qubit_pred = std::make_shared<MaxNQubitsPredicate>(
202  placement_ptr->get_architecture_ref().n_nodes());
203 
204  PredicatePtrMap precons{
205  CompilationUnit::make_type_pair(twoqbpred),
206  CompilationUnit::make_type_pair(n_qubit_pred)};
207  PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(
208  placement_ptr->get_architecture_ref());
209  PredicatePtrMap s_postcons{CompilationUnit::make_type_pair(placement_pred)};
210  PostConditions pc{s_postcons, {}, Guarantee::Preserve};
211  // record pass config
212  nlohmann::json j;
213  j["name"] = "PlacementPass";
214  j["placement"] = placement_ptr;
215  return std::make_shared<StandardPass>(precons, t, pc, j);
216 }

275 PassPtr gen_routing_pass(
276  const Architecture& arc, const std::vector<RoutingMethodPtr>& config) {
277  Transform::Transformation trans = [=](Circuit& circ,
278  std::shared_ptr<unit_bimaps_t> maps) {
279  MappingManager mm(std::make_shared<Architecture>(arc));
280  return mm.route_circuit_with_maps(circ, config, maps);
281  };
282  Transform t = Transform(trans);
283 
284  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
285  PredicatePtr n_qubit_pred =
286  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
287  PredicatePtrMap precons{
288  CompilationUnit::make_type_pair(twoqbpred),
289  CompilationUnit::make_type_pair(n_qubit_pred)};
290 
291  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
292  std::pair<const std::type_index, PredicatePtr> pair1 =
293  CompilationUnit::make_type_pair(postcon1);
294  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
295  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
296  std::type_index gateset_ti = typeid(GateSetPredicate);
297  // clears all Routing predicates apart from the ones which are specified
298  // clears all GateSet predicates (inserts SWAPs and BRIDGEs)
299  PredicateClassGuarantees g_postcons{
300  {pair1.first, Guarantee::Clear},
301  {gateset_ti, Guarantee::Clear},
302  {typeid(MaxTwoQubitGatesPredicate), Guarantee::Clear}};
303  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
304 
305  // record pass config
306  nlohmann::json j;
307  j["name"] = "RoutingPass";
308  j["routing_config"] = config;
309  j["architecture"] = arc;
310 
311  return std::make_shared<StandardPass>(precons, t, pc, j);
312 }

314 PassPtr gen_placement_pass_phase_poly(
315  const Architecture& arc, unsigned maximum_matches, unsigned timeout,
316  unsigned maximum_pattern_gates, unsigned maximum_pattern_depth) {
317  Transform::Transformation trans = [=](Circuit& circ,
318  std::shared_ptr<unit_bimaps_t> maps) {
319  if (arc.n_nodes() < circ.n_qubits()) {
320  throw CircuitInvalidity(
321  "Circuit has more qubits than the architecture has nodes.");
322  }
323 
324  qubit_vector_t q_vec = circ.all_qubits();
325  std::map<Qubit, Node> qubit_to_nodes;
326  unsigned counter = 0;
327 
328  for (Node x : arc.nodes()) {
329  if (counter < circ.n_qubits()) {
330  qubit_to_nodes.insert({q_vec[counter], x});
331  ++counter;
332  }
333  }
334 
335  circ.rename_units(qubit_to_nodes);
336  update_maps(maps, qubit_to_nodes, qubit_to_nodes);
337 
338  return true;
339  };
340  Transform t = Transform(trans);
341 
342  PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
343  PredicatePtrMap precons{CompilationUnit::make_type_pair(no_wire_swap)};
344 
345  PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(arc);
346  PredicatePtr n_qubit_pred =
347  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
348 
349  PredicatePtrMap s_postcons{
350  CompilationUnit::make_type_pair(placement_pred),
351  CompilationUnit::make_type_pair(n_qubit_pred),
352  CompilationUnit::make_type_pair(no_wire_swap)};
353 
354  PostConditions pc{s_postcons, {}, Guarantee::Preserve};
355  // record pass config
356  nlohmann::json j;
357  j["name"] = "PlacementPass";
358  Placement::Ptr pp = std::make_shared<GraphPlacement>(
359  arc, maximum_matches, timeout, maximum_pattern_gates,
360  maximum_pattern_depth);
361  j["params"]["placement"] = pp;
362  return std::make_shared<StandardPass>(precons, t, pc, j);
363 }

365 PassPtr aas_routing_pass(
366  const Architecture& arc, const unsigned lookahead,
367  const aas::CNotSynthType cnotsynthtype) {
368  Transform::SimpleTransformation trans = [=](Circuit& circ) {
369  // check input:
370  if (lookahead == 0) {
371  throw std::logic_error("lookahead must be > 0");
372  }
373  if (arc.n_nodes() < circ.n_qubits()) {
374  throw CircuitInvalidity(
375  "Circuit has more qubits than the architecture has nodes.");
376  }
377 
378  // this pass is not able to handle implicit wire swaps
379  // this is additionally assured by a precondition of this pass
380  TKET_ASSERT(!circ.has_implicit_wireswaps());
381 
382  qubit_vector_t all_qu = circ.all_qubits();
383 
384  Circuit input_circ = circ;
385 
386  VertexList bin;
387 
388  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
389  if (!circ.detect_boundary_Op(v)) {
390  bin.push_back(v);
391  }
392  }
393 
394  circ.remove_vertices(
395  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
396 
397  for (const Command& com : input_circ) {
398  OpType ot = com.get_op_ptr()->get_type();
399  unit_vector_t qbs = com.get_args();
400  switch (ot) {
401  case OpType::PhasePolyBox: {
402  Op_ptr op = com.get_op_ptr();
403  const PhasePolyBox& b = static_cast<const PhasePolyBox&>(*op);
404 
405  Circuit b_circ(*b.to_circuit());
406 
407  PhasePolyBox ppb(b_circ);
408 
409  Circuit result =
410  aas::phase_poly_synthesis(arc, ppb, lookahead, cnotsynthtype);
411 
412  for (const Command& res_com : result) {
413  OpType ot = res_com.get_op_ptr()->get_type();
414  unit_vector_t res_qbs = res_com.get_args();
415 
416  switch (ot) {
417  case OpType::CX: {
418  circ.add_op<Node>(ot, {Node(res_qbs[0]), Node(res_qbs[1])});
419  break;
420  }
421  case OpType::Rz: {
422  auto angle = res_com.get_op_ptr()->get_params().at(0);
423  circ.add_op<Node>(ot, angle, {Node(res_qbs[0])});
424  break;
425  }
426  default: {
427  TKET_ASSERT(!"Invalid gate type in phase poly box");
428  }
429  }
430  }
431  break;
432  }
433  case OpType::H: {
434  circ.add_op<Node>(ot, {Node(qbs[0])});
435  break;
436  }
437  case OpType::Collapse: {
438  circ.add_op<Node>(ot, {Node(qbs[0])});
439  break;
440  }
441  case OpType::Reset: {
442  circ.add_op<Node>(ot, {Node(qbs[0])});
443  break;
444  }
445  case OpType::Measure: {
446  circ.add_measure(Node(qbs[0]), Bit(qbs[1]));
447  break;
448  }
449  default: {
450  throw BadOpType("Invalid gate in input of AAS routing", ot);
451  }
452  }
453  }
454 
455  return true;
456  };
457  Transform t = Transform(trans);
458 
459  PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
460  PredicatePtr n_qubit_pred =
461  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
462  PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
463 
464  PredicatePtrMap precons{
465  CompilationUnit::make_type_pair(placedpred),
466  CompilationUnit::make_type_pair(n_qubit_pred),
467  CompilationUnit::make_type_pair(no_wire_swap)};
468 
469  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
470  std::pair<const std::type_index, PredicatePtr> pair1 =
471  CompilationUnit::make_type_pair(postcon1);
472  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
473  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
474  std::type_index gateset_ti = typeid(GateSetPredicate);
475  PredicateClassGuarantees g_postcons{
476  {pair1.first, Guarantee::Clear}, {gateset_ti, Guarantee::Clear}};
477  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
478 
479  nlohmann::json j;
480  j["name"] = "AASRoutingPass";
481  j["architecture"] = arc;
482 
483  return std::make_shared<StandardPass>(precons, t, pc, j);
484 }

511 PassPtr gen_decompose_routing_gates_to_cxs_pass(
512  const Architecture& arc, bool directed) {
513  PredicateClassGuarantees g_postcons{
514  {typeid(GateSetPredicate), Guarantee::Clear}};
515  PredicatePtrMap precons;
516  PredicatePtrMap s_postcons;
517  Transform t = Transforms::decompose_SWAP_to_CX(arc) >>
518  Transforms::decompose_BRIDGE_to_CX() >>
519  Transforms::remove_redundancies();
520  if (directed) {
521  OpTypeSet out_optypes{all_single_qubit_types()};
522  out_optypes.insert(OpType::CX);
523  OpTypeSet in_optypes = out_optypes;
524  in_optypes.insert(OpType::SWAP);
525  in_optypes.insert(OpType::BRIDGE);
526  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
527  PredicatePtr connected = std::make_shared<ConnectivityPredicate>(arc);
528  PredicatePtr wireswaps = std::make_shared<NoWireSwapsPredicate>();
529  PredicatePtr directedpred = std::make_shared<DirectednessPredicate>(arc);
530  PredicatePtr ingates = std::make_shared<GateSetPredicate>(in_optypes);
531  PredicatePtr outgates = std::make_shared<GateSetPredicate>(out_optypes);
532  precons = {
533  CompilationUnit::make_type_pair(connected),
534  CompilationUnit::make_type_pair(wireswaps),
535  CompilationUnit::make_type_pair(ingates)};
536  s_postcons = {
537  CompilationUnit::make_type_pair(directedpred),
538  CompilationUnit::make_type_pair(outgates),
539  CompilationUnit::make_type_pair(twoqbpred)};
540  t = t >> Transforms::decompose_CX_directed(arc) >>
541  Transforms::remove_redundancies();
542  }
543  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
544  // record pass config
545  nlohmann::json j;
546  j["name"] = "DecomposeSwapsToCXs";
547  j["directed"] = directed;
548  j["architecture"] = arc;
549  return std::make_shared<StandardPass>(precons, t, pc, j);
550 }

552 PassPtr gen_user_defined_swap_decomp_pass(const Circuit& replacement_circ) {
553  Transform t = Transforms::decompose_SWAP(replacement_circ);
554  PredicateClassGuarantees g_postcons{
