51PassPtr gen_rebase_pass(
52 const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
53 const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
54 tk1_replacement) {
55 Transform t = Transforms::rebase_factory(
56 allowed_gates, cx_replacement, tk1_replacement);
57
58 PredicatePtrMap precons;
59 OpTypeSet all_types(allowed_gates);
60 all_types.insert(OpType::Measure);
61 all_types.insert(OpType::Collapse);
62 all_types.insert(OpType::Reset);
63 all_types.insert(OpType::Phase);
64 PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
65 PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
66 std::pair<const std::type_index, PredicatePtr> pair2 =
67 CompilationUnit::make_type_pair(postcon1);
68 PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
69 PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
70 PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
71 // record pass config
72 nlohmann::json j;
73 j["name"] = "RebaseCustom";
74 j["basis_allowed"] = allowed_gates;
75 j["basis_cx_replacement"] = cx_replacement;
76 j["basis_tk1_replacement"] =
77 "SERIALIZATION OF FUNCTIONS IS NOT YET SUPPORTED";
78 return std::make_shared<StandardPass>(precons, t, pc, j);
79}

133static std::function<Circuit(const Expr&, const Expr&, const Expr&)>
134find_tk1_replacement(const OpTypeSet& gateset) {
135 if (gateset.contains(OpType::TK1)) {
136 return CircPool::tk1_to_tk1;
137 }
138 if (gateset.contains(OpType::U3)) {
139 return CircPool::tk1_to_u3;
140 }
141 if (gateset.contains(OpType::Rz) && gateset.contains(OpType::X) &&
142 gateset.contains(OpType::SX)) {
143 return CircPool::tk1_to_rzxsx;
144 }
145 if (gateset.contains(OpType::PhasedX) && gateset.contains(OpType::Rz)) {
146 return CircPool::tk1_to_PhasedXRz;
147 }
148 if (gateset.contains(OpType::Rz) && gateset.contains(OpType::Rx)) {
149 return CircPool::tk1_to_rzrx;
150 }
151 if (gateset.contains(OpType::Rx) && gateset.contains(OpType::Ry)) {
152 return CircPool::tk1_to_rxry;
153 }
154 if (gateset.contains(OpType::Rz) && gateset.contains(OpType::H)) {
155 return CircPool::tk1_to_rzh;
156 }
157 if (gateset.contains(OpType::Rz) && gateset.contains(OpType::SX)) {
158 return CircPool::tk1_to_rzsx;
159 }
160 if (gateset.contains(OpType::GPI) && gateset.contains(OpType::GPI2)) {
161 return CircPool::TK1_using_GPI;
162 }
163 throw Unsupported("No known decomposition from TK1 to available gateset.");
164}

166static Circuit find_cx_replacement(const OpTypeSet& gateset, bool allow_swaps) {
167 if (gateset.contains(OpType::CX)) {
168 return CircPool::CX();
169 }
170 if (gateset.contains(OpType::ZZMax)) {
171 return CircPool::CX_using_ZZMax();
172 }
173 if (gateset.contains(OpType::ISWAPMax)) {
174 return allow_swaps ? CircPool::CX_using_ISWAPMax_and_swap()
175 : CircPool::CX_using_ISWAPMax();
176 }
177 if (gateset.contains(OpType::XXPhase)) {
178 return CircPool::CX_using_XXPhase_0();
179 }
180 if (gateset.contains(OpType::ECR)) {
181 return CircPool::CX_using_ECR();
182 }
183 if (gateset.contains(OpType::CZ)) {
184 return CircPool::H_CZ_H();
185 }
186 if (gateset.contains(OpType::AAMS)) {
187 return CircPool::CX_using_AAMS();
188 }
189 throw Unsupported("No known decomposition from CX to available gateset.");
190}

192static std::function<Circuit(const Expr&, const Expr&, const Expr&)>
193find_tk2_replacement(const OpTypeSet& gateset, bool allow_swaps) {
194 if (gateset.contains(OpType::TK2)) {
195 return allow_swaps ? CircPool::TK2_using_TK2_or_swap
196 : CircPool::TK2_using_TK2;
197 }
198 if (gateset.contains(OpType::ZZPhase)) {
199 return allow_swaps ? CircPool::TK2_using_ZZPhase_and_swap
200 : CircPool::TK2_using_ZZPhase;
201 }
202 if (gateset.contains(OpType::CX)) {
203 return allow_swaps ? CircPool::TK2_using_CX_and_swap
204 : CircPool::TK2_using_CX;
205 }
206 if (gateset.contains(OpType::ZZMax)) {
207 return allow_swaps ? CircPool::TK2_using_ZZMax_and_swap
208 : CircPool::TK2_using_ZZMax;
209 }
210 if (gateset.contains(OpType::ISWAPMax)) {
211 return allow_swaps ? CircPool::TK2_using_ISWAPMax_and_swap
212 : CircPool::TK2_using_ISWAPMax;
213 }
214 if (gateset.contains(OpType::AAMS)) {
215 return CircPool::TK2_using_AAMS;
216 }
217 throw Unsupported("No known decomposition from TK2 to available gateset.");
218}

220PassPtr gen_auto_rebase_pass(const OpTypeSet& allowed_gates, bool allow_swaps) {
221 auto find_rebase = [allowed_gates, allow_swaps]() {
222 auto tk1_replacement = find_tk1_replacement(allowed_gates);
223 if (allowed_gates.contains(OpType::CX) &&
224 !allowed_gates.contains(OpType::TK2) && !allow_swaps) {
225 return Transforms::rebase_factory(
226 allowed_gates, CircPool::CX(), tk1_replacement);
227 }
228 try {
229 return Transforms::rebase_factory_via_tk2(
230 allowed_gates, tk1_replacement,
231 find_tk2_replacement(allowed_gates, allow_swaps));
232 } catch (const Unsupported&) {
233 }
234 try {
235 return Transforms::rebase_factory(
236 allowed_gates, find_cx_replacement(allowed_gates, allow_swaps),
237 tk1_replacement);
238 } catch (const Unsupported&) {
239 throw Unsupported(
240 "No known decomposition from CX or TK2 to available gateset.");
241 }
242 };
243 Transform t = find_rebase();
244 PredicatePtrMap precons;
245 OpTypeSet all_types(allowed_gates);
246 all_types.insert(OpType::Measure);
247 all_types.insert(OpType::Reset);
248 all_types.insert(OpType::Phase);
249 PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
250 PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
251 std::pair<const std::type_index, PredicatePtr> pair2 =
252 CompilationUnit::make_type_pair(postcon1);
253 PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
254 PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
255 PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
256 // record pass config
257 // sort the gateset
258 std::vector<OpType> sorted_gates(allowed_gates.begin(), allowed_gates.end());
259 std::sort(sorted_gates.begin(), sorted_gates.end());
260 nlohmann::json j;
261 j["name"] = "AutoRebase";
262 j["basis_allowed"] = sorted_gates;
263 j["allow_swaps"] = allow_swaps;
264 return std::make_shared<StandardPass>(precons, t, pc, j);
265}

267PassPtr gen_auto_squash_pass(const OpTypeSet& singleqs) {
268 auto find_squash = [singleqs]() {
269 if (singleqs.contains(OpType::Rz) && singleqs.contains(OpType::PhasedX)) {
270 return Transforms::squash_1qb_to_Rz_PhasedX(false);
271 } else {
272 auto tk1_replacement = find_tk1_replacement(singleqs);
273 return Transforms::squash_factory(singleqs, tk1_replacement, false);
274 }
275 };
276 Transform t = find_squash();
277 PredicateClassGuarantees g_postcons = {
278 {typeid(GateSetPredicate), Guarantee::Clear}};
279 PostConditions postcon = {{}, g_postcons, Guarantee::Preserve};
280 PredicatePtrMap precons;
281 // record pass config
282 // sort the gateset
283 std::vector<OpType> sorted_gates(singleqs.begin(), singleqs.end());
284 std::sort(sorted_gates.begin(), sorted_gates.end());
285 nlohmann::json j;
286 j["name"] = "AutoSquash";
287 j["basis_singleqs"] = sorted_gates;
288 return std::make_shared<StandardPass>(precons, t, postcon, j);
289}

356PassPtr gen_flatten_relabel_registers_pass(const std::string& label) {
357 Transform t =
358 Transform([=](Circuit& circuit, std::shared_ptr<unit_bimaps_t> maps) {
359 unsigned n_qubits = circuit.n_qubits();
360 circuit.remove_blank_wires(true);
361 bool changed = circuit.n_qubits() < n_qubits;
362 std::map<Qubit, Qubit> relabelling_map;
363 std::vector<Qubit> all_qubits = circuit.all_qubits();
364 for (unsigned i = 0; i < all_qubits.size(); i++) {
365 relabelling_map.insert({all_qubits[i], Qubit(label, i)});
366 }
367
368 circuit.rename_units(relabelling_map);
369 changed |= update_maps(maps, relabelling_map, relabelling_map);
370 return changed;
371 });
372 PredicatePtrMap precons = {};
373 PostConditions postcons = {};
374
375 nlohmann::json j;
376 j["name"] = "FlattenRelabelRegistersPass";
377 j["label"] = label;
378 return std::make_shared<StandardPass>(precons, t, postcons, j);
379}

401PassPtr gen_placement_pass(const Placement::Ptr& placement_ptr) {
402 Transform::Transformation trans = [=](Circuit& circ,
403 std::shared_ptr<unit_bimaps_t> maps) {
404 // Fall back to line placement if graph placement fails
405 bool changed;
406 try {
407 changed = placement_ptr->place(circ, maps);
408 } catch (const std::runtime_error& e) {
409 std::stringstream ss;
410 ss << "PlacementPass failed with message: " << e.what()
411 << " Fall back to LinePlacement.";
412 tket_log()->warn(ss.str());
413 Placement::Ptr line_placement_ptr = std::make_shared<LinePlacement>(
414 placement_ptr->get_architecture_ref());
415 changed = line_placement_ptr->place(circ, maps);
416 }
417 return changed;
418 };
419 Transform t = Transform(trans);
420 PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
421 PredicatePtr n_qubit_pred = std::make_shared<MaxNQubitsPredicate>(
422 placement_ptr->get_architecture_ref().n_nodes());
423
424 PredicatePtrMap precons{
425 CompilationUnit::make_type_pair(twoqbpred),
426 CompilationUnit::make_type_pair(n_qubit_pred)};
427 PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(
428 placement_ptr->get_architecture_ref());
429 PredicatePtrMap s_postcons{CompilationUnit::make_type_pair(placement_pred)};
430 PostConditions pc{s_postcons, {}, Guarantee::Preserve};
431 // record pass config
432 nlohmann::json j;
433 j["name"] = "PlacementPass";
434 j["placement"] = placement_ptr;
435 return std::make_shared<StandardPass>(precons, t, pc, j);
436}

479PassPtr gen_cx_mapping_pass(
480 const Architecture& arc, const Placement::Ptr& placement_ptr,
481 const std::vector<RoutingMethodPtr>& config, bool directed_cx,
482 bool delay_measures) {
483 OpTypeSet gate_set = all_single_qubit_types();
484 gate_set.insert(OpType::CX);
485 PassPtr rebase_pass = gen_auto_rebase_pass(gate_set);
486 PassPtr return_pass =
487 rebase_pass >> gen_full_mapping_pass(arc, placement_ptr, config);
488 if (delay_measures) return_pass = return_pass >> DelayMeasures();
489 return_pass = return_pass >> rebase_pass >>
490 gen_decompose_routing_gates_to_cxs_pass(arc, directed_cx);
491 Transform t{[=](Circuit& circ, std::shared_ptr<unit_bimaps_t> maps) {
492 // Relabel all two-qubit gates
493 CompilationUnit cu(circ);
494 bool changed = rebase_pass->apply(cu);
495 circ = cu.get_circ_ref();
496 // Now try placement, falling back on "LinePlacement" if "GraphPlacement"
497 // fails
498 try {
499 changed |= placement_ptr->place(circ, maps);
500 } catch (const std::runtime_error& e) {
501 std::stringstream ss;
502 ss << "PlacementPass failed with message: " << e.what()
503 << " Fall back to LinePlacement.";
504 tket_log()->warn(ss.str());
505 Placement::Ptr line_placement_ptr = std::make_shared<LinePlacement>(
506 placement_ptr->get_architecture_ref());
507 changed |= line_placement_ptr->place(circ, maps);
508 }
509 // Now route
510 MappingManager mm(std::make_shared<Architecture>(arc));
511 changed |= mm.route_circuit_with_maps(circ, config, maps);
512
513 // Now decompose routing gates to cx gates (this won't change maps)
514 CompilationUnit cu1(circ);
515 if (delay_measures) {
516 changed |= DelayMeasures()->apply(cu1);
517 }
518 changed |=
519 gen_decompose_routing_gates_to_cxs_pass(arc, directed_cx)->apply(cu1);
520 circ = cu1.get_circ_ref();
521 return changed;
522 }};
523 PassConditions conditions = return_pass->get_conditions();
524 nlohmann::json j;
525 j["name"] = "CXMappingPass";
526 j["architecture"] = arc;
527 j["placement"] = placement_ptr;
528 j["routing_config"] = config;
529 j["directed"] = directed_cx;
530 j["delay_measures"] = delay_measures;
531 return std::make_shared<StandardPass>(
532 conditions.first, t, conditions.second, j);
533}

535PassPtr gen_routing_pass(
536 const Architecture& arc, const std::vector<RoutingMethodPtr>& config) {
537 Transform::Transformation trans = [=](Circuit& circ,
538 std::shared_ptr<unit_bimaps_t> maps) {
539 MappingManager mm(std::make_shared<Architecture>(arc));
540 return mm.route_circuit_with_maps(circ, config, maps);
541 };
542 Transform t = Transform(trans);
543
544 PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
545 PredicatePtr n_qubit_pred =
546 std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
547 PredicatePtrMap precons{
548 CompilationUnit::make_type_pair(twoqbpred),
549 CompilationUnit::make_type_pair(n_qubit_pred)};
550
551 PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
552 std::pair<const std::type_index, PredicatePtr> pair1 =
553 CompilationUnit::make_type_pair(postcon1);
554 PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
555 PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
556 std::type_index gateset_ti = typeid(GateSetPredicate);
557 // clears all Routing predicates apart from the ones which are specified
558 // clears all GateSet predicates (inserts SWAPs and BRIDGEs)
559 PredicateClassGuarantees g_postcons{

625PassPtr aas_routing_pass(
626 const Architecture& arc, const unsigned lookahead,
627 const aas::CNotSynthType cnotsynthtype) {
628 Transform::SimpleTransformation trans = [=](Circuit& circ) {
629 // check input:
630 if (lookahead == 0) {
631 throw std::logic_error("lookahead must be > 0");
632 }
633 if (arc.n_nodes() < circ.n_qubits()) {
634 throw CircuitInvalidity(
635 "Circuit has more qubits than the architecture has nodes.");
636 }
637
638 // this pass is not able to handle implicit wire swaps
639 // this is additionally assured by a precondition of this pass
640 TKET_ASSERT(!circ.has_implicit_wireswaps());
641
642 qubit_vector_t all_qu = circ.all_qubits();
643
644 Circuit input_circ = circ;
645
646 VertexList bin;
647
648 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
649 if (!circ.detect_boundary_Op(v)) {
650 bin.push_back(v);
651 }
652 }
653
654 circ.remove_vertices(
655 bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
656
657 for (const Command& com : input_circ) {
658 OpType ot = com.get_op_ptr()->get_type();
659 unit_vector_t qbs = com.get_args();
660 switch (ot) {
661 case OpType::PhasePolyBox: {
662 Op_ptr op = com.get_op_ptr();
663 const auto& ppb = dynamic_cast<const PhasePolyBox&>(*op);
664
665 Circuit result =
666 aas::phase_poly_synthesis(arc, ppb, lookahead, cnotsynthtype);
667
668 for (const Command& res_com : result) {
669 OpType ot = res_com.get_op_ptr()->get_type();
670 unit_vector_t res_qbs = res_com.get_args();
671
672 switch (ot) {
673 case OpType::CX: {
674 circ.add_op<Node>(ot, {Node(res_qbs[0]), Node(res_qbs[1])});
675 break;
676 }
677 case OpType::Rz: {
678 auto angle = res_com.get_op_ptr()->get_params().at(0);
679 circ.add_op<Node>(ot, angle, {Node(res_qbs[0])});
680 break;
681 }
682 default: {
683 TKET_ASSERT(!"Invalid gate type in phase poly box");
684 }
685 }
686 }
687 break;
688 }
689 case OpType::H: {
690 circ.add_op<Node>(ot, {Node(qbs[0])});
691 break;
692 }
693 case OpType::Collapse: {
694 circ.add_op<Node>(ot, {Node(qbs[0])});
695 break;
696 }
697 case OpType::Reset: {
698 circ.add_op<Node>(ot, {Node(qbs[0])});
699 break;
700 }
701 case OpType::Measure: {
702 circ.add_measure(Node(qbs[0]), Bit(qbs[1]));
703 break;
704 }
705 default: {
706 throw BadOpType("Invalid gate in input of AAS routing", ot);
707 }
708 }
709 }
710
711 return true;
712 };
713 Transform t = Transform(trans);
714
715 PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
716 PredicatePtr n_qubit_pred =
717 std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
718 PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
719
720 PredicatePtrMap precons{
721 CompilationUnit::make_type_pair(placedpred),
722 CompilationUnit::make_type_pair(n_qubit_pred),
723 CompilationUnit::make_type_pair(no_wire_swap)};
724
725 PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
726 std::pair<const std::type_index, PredicatePtr> pair1 =
727 CompilationUnit::make_type_pair(postcon1);
728 PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
729 PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
730 std::type_index gateset_ti = typeid(GateSetPredicate);
731 PredicateClassGuarantees g_postcons{
732 {pair1.first, Guarantee::Clear}, {gateset_ti, Guarantee::Clear}};
733 PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
734
735 nlohmann::json j;
736 j["name"] = "AASRoutingPass";
737 j["architecture"] = arc;
738
739 return std::make_shared<StandardPass>(precons, t, pc, j);
740}
