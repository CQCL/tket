46 PassPtr gen_rebase_pass(
47  const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
48  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
49  tk1_replacement) {
50  Transform t = Transforms::rebase_factory(
51  allowed_gates, cx_replacement, tk1_replacement);
52 
53  PredicatePtrMap precons;
54  OpTypeSet all_types(allowed_gates);
55  all_types.insert(OpType::Measure);
56  all_types.insert(OpType::Collapse);
57  all_types.insert(OpType::Reset);
58  PredicatePtr postcon1 = std::make_shared<GateSetPredicate>(all_types);
59  PredicatePtr postcon2 = std::make_shared<MaxTwoQubitGatesPredicate>();
60  std::pair<const std::type_index, PredicatePtr> pair2 =
61  CompilationUnit::make_type_pair(postcon1);
62  PredicatePtrMap s_postcons{pair2, CompilationUnit::make_type_pair(postcon2)};
63  PredicateClassGuarantees g_postcons{{pair2.first, Guarantee::Clear}};
64  PostConditions pc = {s_postcons, g_postcons, Guarantee::Preserve};
65  // record pass config
66  nlohmann::json j;
67  j["name"] = "RebaseCustom";
68  j["basis_allowed"] = allowed_gates;
69  j["basis_cx_replacement"] = cx_replacement;
70  j["basis_tk1_replacement"] =
71  "SERIALIZATION OF FUNCTIONS IS NOT YET SUPPORTED";
72  return std::make_shared<StandardPass>(precons, t, pc, j);
73 }

161 PassPtr gen_flatten_relabel_registers_pass(const std::string& label) {
162  Transform t =
163  Transform([=](Circuit& circuit, std::shared_ptr<unit_bimaps_t> maps) {
164  unsigned n_qubits = circuit.n_qubits();
165  circuit.remove_blank_wires();
166  bool changed = circuit.n_qubits() < n_qubits;
167  std::map<Qubit, Qubit> relabelling_map;
168  std::vector<Qubit> all_qubits = circuit.all_qubits();
169  for (unsigned i = 0; i < all_qubits.size(); i++) {
170  relabelling_map.insert({all_qubits[i], Qubit(label, i)});
171  }
172 
173  circuit.rename_units(relabelling_map);
174  changed |= update_maps(maps, relabelling_map, relabelling_map);
175  return changed;
176  });
177  PredicatePtrMap precons = {};
178  PostConditions postcons = {};
179 
180  nlohmann::json j;
181  j["name"] = "FlattenRelabelRegistersPass";
182  j["label"] = label;
183  return std::make_shared<StandardPass>(precons, t, postcons, j);

206 PassPtr gen_placement_pass(const Placement::Ptr& placement_ptr) {
207  Transform::Transformation trans = [=](Circuit& circ,
208  std::shared_ptr<unit_bimaps_t> maps) {
209  // Fall back to line placement if graph placement fails
210  bool changed;
211  try {
212  changed = placement_ptr->place(circ, maps);
213  } catch (const std::runtime_error& e) {
214  std::stringstream ss;
215  ss << "PlacementPass failed with message: " << e.what()
216  << " Fall back to LinePlacement.";
217  tket_log()->warn(ss.str());
218  Placement::Ptr line_placement_ptr = std::make_shared<LinePlacement>(
219  placement_ptr->get_architecture_ref());
220  changed = line_placement_ptr->place(circ, maps);
221  }
222  return changed;
223  };
224  Transform t = Transform(trans);
225  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
226  PredicatePtr n_qubit_pred = std::make_shared<MaxNQubitsPredicate>(
227  placement_ptr->get_architecture_ref().n_nodes());
228 
229  PredicatePtrMap precons{
230  CompilationUnit::make_type_pair(twoqbpred),
231  CompilationUnit::make_type_pair(n_qubit_pred)};
232  PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(
233  placement_ptr->get_architecture_ref());
234  PredicatePtrMap s_postcons{CompilationUnit::make_type_pair(placement_pred)};
235  PostConditions pc{s_postcons, {}, Guarantee::Preserve};
236  // record pass config
237  nlohmann::json j;
238  j["name"] = "PlacementPass";
239  j["placement"] = placement_ptr;
240  return std::make_shared<StandardPass>(precons, t, pc, j);
241 }

300 PassPtr gen_routing_pass(
301  const Architecture& arc, const std::vector<RoutingMethodPtr>& config) {
302  Transform::Transformation trans = [=](Circuit& circ,
303  std::shared_ptr<unit_bimaps_t> maps) {
304  MappingManager mm(std::make_shared<Architecture>(arc));
305  return mm.route_circuit_with_maps(circ, config, maps);
306  };
307  Transform t = Transform(trans);
308 
309  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
310  PredicatePtr n_qubit_pred =
311  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
312  PredicatePtrMap precons{
313  CompilationUnit::make_type_pair(twoqbpred),
314  CompilationUnit::make_type_pair(n_qubit_pred)};
315 
316  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
317  std::pair<const std::type_index, PredicatePtr> pair1 =
318  CompilationUnit::make_type_pair(postcon1);
319  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
320  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
321  std::type_index gateset_ti = typeid(GateSetPredicate);
322  // clears all Routing predicates apart from the ones which are specified
323  // clears all GateSet predicates (inserts SWAPs and BRIDGEs)
324  PredicateClassGuarantees g_postcons{
325  {pair1.first, Guarantee::Clear},
326  {gateset_ti, Guarantee::Clear},
327  {typeid(MaxTwoQubitGatesPredicate), Guarantee::Clear}};
328  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
329 
330  // record pass config
331  nlohmann::json j;
332  j["name"] = "RoutingPass";
333  j["routing_config"] = config;
334  j["architecture"] = arc;
335 
336  return std::make_shared<StandardPass>(precons, t, pc, j);
337 }

339 PassPtr gen_placement_pass_phase_poly(
340  const Architecture& arc, unsigned maximum_matches, unsigned timeout,
341  unsigned maximum_pattern_gates, unsigned maximum_pattern_depth) {
342  Transform::Transformation trans = [=](Circuit& circ,
343  std::shared_ptr<unit_bimaps_t> maps) {
344  if (arc.n_nodes() < circ.n_qubits()) {
345  throw CircuitInvalidity(
346  "Circuit has more qubits than the architecture has nodes.");
347  }
348 
349  qubit_vector_t q_vec = circ.all_qubits();
350  std::map<Qubit, Node> qubit_to_nodes;
351  unsigned counter = 0;
352 
353  for (Node x : arc.nodes()) {
354  if (counter < circ.n_qubits()) {
355  qubit_to_nodes.insert({q_vec[counter], x});
356  ++counter;
357  }
358  }
359 
360  circ.rename_units(qubit_to_nodes);
361  update_maps(maps, qubit_to_nodes, qubit_to_nodes);
362 
363  return true;
364  };
365  Transform t = Transform(trans);
366 
367  PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
368  PredicatePtrMap precons{CompilationUnit::make_type_pair(no_wire_swap)};
369 
370  PredicatePtr placement_pred = std::make_shared<PlacementPredicate>(arc);
371  PredicatePtr n_qubit_pred =
372  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
373 
374  PredicatePtrMap s_postcons{
375  CompilationUnit::make_type_pair(placement_pred),
376  CompilationUnit::make_type_pair(n_qubit_pred),
377  CompilationUnit::make_type_pair(no_wire_swap)};
378 
379  PostConditions pc{s_postcons, {}, Guarantee::Preserve};
380  // record pass config
381  nlohmann::json j;
382  j["name"] = "PlacementPass";
383  Placement::Ptr pp = std::make_shared<GraphPlacement>(
384  arc, maximum_matches, timeout, maximum_pattern_gates,
385  maximum_pattern_depth);
386  j["params"]["placement"] = pp;
387  return std::make_shared<StandardPass>(precons, t, pc, j);
388 }

390 PassPtr aas_routing_pass(
391  const Architecture& arc, const unsigned lookahead,
392  const aas::CNotSynthType cnotsynthtype) {
393  Transform::SimpleTransformation trans = [=](Circuit& circ) {
394  // check input:
395  if (lookahead == 0) {
396  throw std::logic_error("lookahead must be > 0");
397  }
398  if (arc.n_nodes() < circ.n_qubits()) {
399  throw CircuitInvalidity(
400  "Circuit has more qubits than the architecture has nodes.");
401  }
402 
403  // this pass is not able to handle implicit wire swaps
404  // this is additionally assured by a precondition of this pass
405  TKET_ASSERT(!circ.has_implicit_wireswaps());
406 
407  qubit_vector_t all_qu = circ.all_qubits();
408 
409  Circuit input_circ = circ;
410 
411  VertexList bin;
412 
413  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
414  if (!circ.detect_boundary_Op(v)) {
415  bin.push_back(v);
416  }
417  }
418 
419  circ.remove_vertices(
420  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
421 
422  for (const Command& com : input_circ) {
423  OpType ot = com.get_op_ptr()->get_type();
424  unit_vector_t qbs = com.get_args();
425  switch (ot) {
426  case OpType::PhasePolyBox: {
427  Op_ptr op = com.get_op_ptr();
428  const PhasePolyBox& b = static_cast<const PhasePolyBox&>(*op);
429 
430  Circuit b_circ(*b.to_circuit());
431 
432  PhasePolyBox ppb(b_circ);
433 
434  Circuit result =
435  aas::phase_poly_synthesis(arc, ppb, lookahead, cnotsynthtype);
436 
437  for (const Command& res_com : result) {
438  OpType ot = res_com.get_op_ptr()->get_type();
439  unit_vector_t res_qbs = res_com.get_args();
440 
441  switch (ot) {
442  case OpType::CX: {
443  circ.add_op<Node>(ot, {Node(res_qbs[0]), Node(res_qbs[1])});
444  break;
445  }
446  case OpType::Rz: {
447  auto angle = res_com.get_op_ptr()->get_params().at(0);
448  circ.add_op<Node>(ot, angle, {Node(res_qbs[0])});
449  break;
450  }
451  default: {
452  TKET_ASSERT(!"Invalid gate type in phase poly box");
453  }
454  }
455  }
456  break;
457  }
458  case OpType::H: {
459  circ.add_op<Node>(ot, {Node(qbs[0])});
460  break;
461  }
462  case OpType::Collapse: {
463  circ.add_op<Node>(ot, {Node(qbs[0])});
464  break;
465  }
466  case OpType::Reset: {
467  circ.add_op<Node>(ot, {Node(qbs[0])});
468  break;
469  }
470  case OpType::Measure: {
471  circ.add_measure(Node(qbs[0]), Bit(qbs[1]));
472  break;
473  }
474  default: {
475  throw BadOpType("Invalid gate in input of AAS routing", ot);
476  }
477  }
478  }
479 
480  return true;
481  };
482  Transform t = Transform(trans);
483 
484  PredicatePtr placedpred = std::make_shared<PlacementPredicate>(arc);
485  PredicatePtr n_qubit_pred =
486  std::make_shared<MaxNQubitsPredicate>(arc.n_nodes());
487  PredicatePtr no_wire_swap = std::make_shared<NoWireSwapsPredicate>();
488 
489  PredicatePtrMap precons{
490  CompilationUnit::make_type_pair(placedpred),
491  CompilationUnit::make_type_pair(n_qubit_pred),
492  CompilationUnit::make_type_pair(no_wire_swap)};
493 
494  PredicatePtr postcon1 = std::make_shared<ConnectivityPredicate>(arc);
495  std::pair<const std::type_index, PredicatePtr> pair1 =
496  CompilationUnit::make_type_pair(postcon1);
497  PredicatePtr postcon2 = std::make_shared<NoWireSwapsPredicate>();
498  PredicatePtrMap s_postcons{pair1, CompilationUnit::make_type_pair(postcon2)};
499  std::type_index gateset_ti = typeid(GateSetPredicate);
500  PredicateClassGuarantees g_postcons{
501  {pair1.first, Guarantee::Clear}, {gateset_ti, Guarantee::Clear}};
502  PostConditions pc{s_postcons, g_postcons, Guarantee::Preserve};
503 
504  nlohmann::json j;
505  j["name"] = "AASRoutingPass";
506  j["architecture"] = arc;
507 
508  return std::make_shared<StandardPass>(precons, t, pc, j);
509 }

536 PassPtr gen_decompose_routing_gates_to_cxs_pass(
537  const Architecture& arc, bool directed) {
538  PredicateClassGuarantees g_postcons{
539  {typeid(GateSetPredicate), Guarantee::Clear}};
540  PredicatePtrMap precons;
541  PredicatePtrMap s_postcons;
542  Transform t = Transforms::decompose_SWAP_to_CX(arc) >>
543  Transforms::decompose_BRIDGE_to_CX() >>
544  Transforms::remove_redundancies();
545  if (directed) {
546  OpTypeSet out_optypes{all_single_qubit_types()};
547  out_optypes.insert(OpType::CX);
548  OpTypeSet in_optypes = out_optypes;
549  in_optypes.insert(OpType::SWAP);
550  in_optypes.insert(OpType::BRIDGE);
551  PredicatePtr twoqbpred = std::make_shared<MaxTwoQubitGatesPredicate>();
552  PredicatePtr connected = std::make_shared<ConnectivityPredicate>(arc);
553  PredicatePtr wireswaps = std::make_shared<NoWireSwapsPredicate>();
554  PredicatePtr directedpred = std::make_shared<DirectednessPredicate>(arc);
555  PredicatePtr ingates = std::make_shared<GateSetPredicate>(in_optypes);
