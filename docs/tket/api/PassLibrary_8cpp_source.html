34 template <typename T>
35 static PassPtr gate_translation_pass(
36  const T &transform, OpTypeSet after_set, bool respect_connectivity,
37  const std::string &name) {
38  after_set.insert(OpType::Measure);
39  after_set.insert(OpType::Collapse);
40  after_set.insert(OpType::Reset);
41  PredicatePtrMap precons;
42  std::type_index ti = typeid(ConnectivityPredicate);
43  PredicatePtr out_gateset = std::make_shared<GateSetPredicate>(after_set);
44  PredicatePtr max2qb = std::make_shared<MaxTwoQubitGatesPredicate>();
45  PredicatePtrMap postcon_spec = {
46  CompilationUnit::make_type_pair(out_gateset),
47  CompilationUnit::make_type_pair(max2qb)};
48  PredicateClassGuarantees g_postcons;
49  if (!respect_connectivity)
50  g_postcons.insert({ti, Guarantee::Clear}); // synthesis passes do not in
51  // general preserve connectivity
52  PostConditions postcon{postcon_spec, g_postcons, Guarantee::Preserve};
53  // record pass config
54  nlohmann::json j;
55  j["name"] = name;
56  PassPtr ptr = std::make_shared<StandardPass>(precons, transform, postcon, j);
57  return ptr;
58 }
