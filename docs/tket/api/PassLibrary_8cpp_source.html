36 template <typename T>
37 static PassPtr gate_translation_pass(
38  const T &transform, OpTypeSet after_set, bool respect_connectivity,
39  const std::string &name) {
40  after_set.insert(OpType::Measure);
41  after_set.insert(OpType::Collapse);
42  after_set.insert(OpType::Reset);
43  PredicatePtrMap precons;
44  std::type_index ti = typeid(ConnectivityPredicate);
45  PredicatePtr out_gateset = std::make_shared<GateSetPredicate>(after_set);
46  PredicatePtr max2qb = std::make_shared<MaxTwoQubitGatesPredicate>();
47  PredicatePtrMap postcon_spec = {
48  CompilationUnit::make_type_pair(out_gateset),
49  CompilationUnit::make_type_pair(max2qb)};
50  PredicateClassGuarantees g_postcons;
51  if (!respect_connectivity)
52  g_postcons.insert({ti, Guarantee::Clear}); // synthesis passes do not in
53  // general preserve connectivity
54  PostConditions postcon{postcon_spec, g_postcons, Guarantee::Preserve};
55  // record pass config
56  nlohmann::json j;
57  j["name"] = name;
58  PassPtr ptr = std::make_shared<StandardPass>(precons, transform, postcon, j);
59  return ptr;
60 }

487 const PassPtr &ZXGraphlikeOptimisation() {
488  static const PassPtr pp([]() {
489  Transform t = Transform([](Circuit &circ) {
490  zx::ZXDiagram diag = circuit_to_zx(circ).first;
491  zx::Rewrite::to_graphlike_form().apply(diag);
492  zx::Rewrite::reduce_graphlike_form().apply(diag);
493  zx::Rewrite::to_MBQC_diag().apply(diag);
494  Circuit c = zx_to_circuit(diag);
495  qubit_vector_t orig_qs = circ.all_qubits();
496  qubit_vector_t c_qs = c.all_qubits();
497  qubit_map_t qmap;
498  for (unsigned i = 0; i < orig_qs.size(); ++i)
499  qmap.insert({c_qs.at(i), orig_qs.at(i)});
500  c.rename_units<Qubit, Qubit>(qmap);
501  circ = c;
502  return true;
503  });
504  OpTypeSet in_optypes = {OpType::Input, OpType::Output, OpType::noop,
505  OpType::SWAP, OpType::H, OpType::Rz,
506  OpType::Rx, OpType::X, OpType::Z,
