30 // breaks for devices with n_qubits >= UINT_MAX/2
31 // For internal use.
32 PathHandler::PathHandler(const MatrixXb &connectivity) {
33  unsigned n = connectivity.rows();
34  size = n;
35 
36  unsigned approx_infinity = UINT_MAX >> 1;
37  if (n >= approx_infinity) throw std::out_of_range("Qubit number too large");
38  distance_matrix_ = MatrixXu::Constant(n, n, approx_infinity);
39  path_matrix_ = MatrixXu::Constant(n, n, n); // set all unreachable nodes to n
40  connectivity_matrix_ = connectivity;
41 
42  // Floyd-Warshall with path reconstruction, see:
43  // https://en.wikipedia.org/wiki/Floyd–Warshall_algorithm#Pseudocode_[11]
44  for (unsigned i = 0; i != n; ++i) {
45  distance_matrix_(i, i) = 0;
46  path_matrix_(i, i) = i;
47  for (unsigned j = 0; j != n; ++j) {
48  if (i == j) continue;
49  if (connectivity_matrix_(i, j)) {
50  distance_matrix_(i, j) = 1;
51  path_matrix_(i, j) = j;
52  }
53  }
54  }
55  for (unsigned k = 0; k != n; ++k) {
56  for (unsigned i = 0; i != n; ++i) {
57  for (unsigned j = 0; j != n; ++j) {
58  unsigned threshold = distance_matrix_(i, k) + distance_matrix_(k, j);
59  if (distance_matrix_(i, j) > threshold) {
60  distance_matrix_(i, j) = threshold;
61  path_matrix_(i, j) = path_matrix_(i, k);
62  }
63  }
64  }
65  }
66 }

68 PathHandler PathHandler::construct_acyclic_handler() const {
69  PathHandler acyclic_handler;
70  unsigned n = distance_matrix_.rows();
71  MatrixXb acyclic_connectivity(n, n);
72  std::vector<unsigned> num_neighbours(n, 0);
73 
74  for (unsigned i = 0; i != n; ++i) {
75  for (unsigned j = 0; j != n; ++j) {
76  if (connectivity_matrix_(i, j)) ++num_neighbours[i];
77  }
78  }
79 
80  for (unsigned i = 0; i != n; ++i) {
81  for (unsigned j = 0; j != n; ++j) {
82  acyclic_connectivity(i, j) = 0;
83  }
84  }
85 
86  // find the centre
87  unsigned threshold_max_dist = n, max_dist_row = 0, centre_node = 0;
88 
89  for (unsigned i = 0; i != n; ++i) {
90  for (unsigned j = 0; j != n; ++j) {
91  if (distance_matrix_(i, j) > max_dist_row)
92  max_dist_row = distance_matrix_(i, j);
93  }
94  if (max_dist_row < threshold_max_dist) {
95  centre_node = i;
96  threshold_max_dist = max_dist_row;
97  }
98  max_dist_row = 0;
99  }
100 
101  // build the acyclic graph outwards from the centre
102  std::list<unsigned> current_layer_vertices{centre_node};
103  std::list<unsigned> next_layer_vertices;
104  std::vector<std::pair<unsigned, unsigned>> parents_neighbours(
105  n); // pair(num_neighbours, parent vertex)
106  std::vector<bool> vertices_in_tree(
107  n, 0); // track which vertices are in the acyclic graph
108  vertices_in_tree[centre_node] = 1;
109  std::pair<unsigned, unsigned> empty_pair{};
110  while (!current_layer_vertices.empty()) {
111  for (unsigned vert : current_layer_vertices) {
112  for (unsigned j = 0; j != n; ++j) {
113  if ((distance_matrix_(vert, j) == 1) && (!vertices_in_tree[j])) {
114  // if first encounter of vertex, add to next layer
115  if (parents_neighbours[j] == empty_pair) {
116  next_layer_vertices.push_back(j);
117  parents_neighbours[j] = {num_neighbours[vert], vert};
118 
119  }
120  // choose parent with most neighbours
121  else if (parents_neighbours[j].first < num_neighbours[vert]) {
122  parents_neighbours[j] = {num_neighbours[vert], vert};
123  }
124  }
125  }
126  }
127  current_layer_vertices.clear();
128  // add in edges
129  for (unsigned vert : next_layer_vertices) {
130  unsigned parent_vertex = parents_neighbours[vert].second;
131  acyclic_connectivity(vert, parent_vertex) = 1;
132  acyclic_connectivity(parent_vertex, vert) = 1;
133 
134  current_layer_vertices.push_back(vert);
135  vertices_in_tree[vert] = 1;
136  parents_neighbours[vert] = {};
137  }
138  next_layer_vertices.clear();
139  }
140 
141  return PathHandler(acyclic_connectivity);
142 }

153 std::vector<Node> find_hampath(const Architecture &arch, long timeout) {
154  unsigned n_nodes = arch.n_nodes();
155  std::vector<Qubit> arch_qubits;
156  for (unsigned i = 0; i < n_nodes; i++) {
157  arch_qubits.push_back(Qubit(i));
158  }
159  QubitGraph q_graph(arch_qubits);
160  for (unsigned i = 0; i != n_nodes - 1; ++i) {
161  q_graph.add_connection(Qubit(i), Qubit(i + 1));
162  }
163 
164  Architecture::UndirectedConnGraph undirected_target =
165  arch.get_undirected_connectivity();
166  QubitGraph::UndirectedConnGraph undirected_pattern =
167  q_graph.get_undirected_connectivity();
168  std::vector<boost::bimap<Qubit, Node>> all_maps =
169  get_weighted_subgraph_monomorphisms(
170  undirected_pattern, undirected_target, 1, timeout, false);
171 
172  /* Architecture has no hampath, sad. */
173  if (all_maps.empty()) {
174  throw NoHamiltonPath(
175  "[AAS]: no Hamilton path found in the given architecture, CNOT "
176  "synthesis stopped. Please try an alternative CNotSynthType.");
177  }
178 
179  /* Left: line, Right: input architecture. */
180  const boost::bimap<Qubit, Node> &qmap = all_maps[0];
181  std::vector<Node> hampath;
182  for (auto it = qmap.left.begin(); it != qmap.left.end(); ++it) {
183  hampath.push_back(it->second);
184  }
185  return hampath;
186 }

188 IterationOrder::IterationOrder(const Architecture &arch) {
189  std::set<Node> visited_nodes;
190 
191  Node no = *arch.nodes().begin();
192 
193  iterationorder.push_back(no);
194  visited_nodes.insert(no);
195 
196  unsigned whilecount = 0;
197  while ((visited_nodes.size() < arch.n_nodes()) &&
198  (whilecount < arch.n_nodes())) {
199  for (auto edge : arch.get_all_edges_vec()) {
200  if (visited_nodes.count(edge.first) &&
201  (!visited_nodes.count(edge.second))) {
202  iterationorder.push_back(edge.second);
203  visited_nodes.insert(edge.second);
204  edgelist.push_back(edge);
205  } else if (
206  (!visited_nodes.count(edge.first)) &&
207  (visited_nodes.count(edge.second))) {
208  iterationorder.push_back(edge.first);
209  visited_nodes.insert(edge.first);
210  edgelist.push_back(edge);
211  }
212  }
213  ++whilecount;
214  }
215 
216  if (visited_nodes.size() != arch.n_nodes()) {
217  throw std::logic_error("Unconnected architecture");
218  }
219 
220  std::reverse(iterationorder.begin(), iterationorder.end());
221 }
