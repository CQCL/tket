26// The idiomatic way to initialise a PathHandler, and assumes the architecture
27// is symmetric. The way using a MatrixXb is for internal use. We initialise
28// without using the distance matrix from Architecture, as we generate distances
29// using Floyd-Warshall anyway.
30PathHandler::PathHandler(const Architecture &arch)
31 : PathHandler(arch.get_connectivity()) {}
32
33// breaks for devices with n_qubits >= UINT_MAX/2
34// For internal use.
35PathHandler::PathHandler(const MatrixXb &connectivity) {
36 unsigned n = connectivity.rows();
37 size = n;
38
39 unsigned approx_infinity = UINT_MAX >> 1;
40 if (n >= approx_infinity) throw std::out_of_range("Qubit number too large");
41 distance_matrix_ = MatrixXu::Constant(n, n, approx_infinity);
42 path_matrix_ = MatrixXu::Constant(n, n, n); // set all unreachable nodes to n
43 connectivity_matrix_ = connectivity;
44
45 // Floyd-Warshall with path reconstruction, see:
46 // https://en.wikipedia.org/wiki/Floyd–Warshall_algorithm#Pseudocode_[11]
47 for (unsigned i = 0; i != n; ++i) {
48 distance_matrix_(i, i) = 0;
49 path_matrix_(i, i) = i;
50 for (unsigned j = 0; j != n; ++j) {
51 if (i == j) continue;
52 if (connectivity_matrix_(i, j)) {
53 distance_matrix_(i, j) = 1;
54 path_matrix_(i, j) = j;
55 }
56 }
57 }
58 for (unsigned k = 0; k != n; ++k) {
59 for (unsigned i = 0; i != n; ++i) {
60 for (unsigned j = 0; j != n; ++j) {
61 unsigned threshold = distance_matrix_(i, k) + distance_matrix_(k, j);
62 if (distance_matrix_(i, j) > threshold) {
63 distance_matrix_(i, j) = threshold;
64 path_matrix_(i, j) = path_matrix_(i, k);
65 }
66 }
67 }
68 }
69}

71PathHandler PathHandler::construct_acyclic_handler() const {
72 PathHandler acyclic_handler;
73 unsigned n = distance_matrix_.rows();
74 MatrixXb acyclic_connectivity(n, n);
75 std::vector<unsigned> num_neighbours(n, 0);
76
77 for (unsigned i = 0; i != n; ++i) {
78 for (unsigned j = 0; j != n; ++j) {
79 if (connectivity_matrix_(i, j)) ++num_neighbours[i];
80 }
81 }
82
83 for (unsigned i = 0; i != n; ++i) {
84 for (unsigned j = 0; j != n; ++j) {
85 acyclic_connectivity(i, j) = 0;
86 }
87 }
88
89 // find the centre
90 unsigned threshold_max_dist = n, max_dist_row = 0, centre_node = 0;
91
92 for (unsigned i = 0; i != n; ++i) {
93 for (unsigned j = 0; j != n; ++j) {
94 if (distance_matrix_(i, j) > max_dist_row)
95 max_dist_row = distance_matrix_(i, j);
96 }
97 if (max_dist_row < threshold_max_dist) {
98 centre_node = i;
99 threshold_max_dist = max_dist_row;
100 }
101 max_dist_row = 0;
102 }
103
104 // build the acyclic graph outwards from the centre
105 std::list<unsigned> current_layer_vertices{centre_node};
106 std::list<unsigned> next_layer_vertices;
107 std::vector<std::pair<unsigned, unsigned>> parents_neighbours(
108 n); // pair(num_neighbours, parent vertex)
109 std::vector<bool> vertices_in_tree(
110 n, 0); // track which vertices are in the acyclic graph
111 vertices_in_tree[centre_node] = 1;
112 std::pair<unsigned, unsigned> empty_pair{};
113 while (!current_layer_vertices.empty()) {
114 for (unsigned vert : current_layer_vertices) {
115 for (unsigned j = 0; j != n; ++j) {
116 if ((distance_matrix_(vert, j) == 1) && (!vertices_in_tree[j])) {
117 // if first encounter of vertex, add to next layer
118 if (parents_neighbours[j] == empty_pair) {
119 next_layer_vertices.push_back(j);
120 parents_neighbours[j] = {num_neighbours[vert], vert};
121
122 }
123 // choose parent with most neighbours
124 else if (parents_neighbours[j].first < num_neighbours[vert]) {
125 parents_neighbours[j] = {num_neighbours[vert], vert};
126 }
127 }
128 }
129 }
130 current_layer_vertices.clear();
131 // add in edges
132 for (unsigned vert : next_layer_vertices) {
133 unsigned parent_vertex = parents_neighbours[vert].second;
134 acyclic_connectivity(vert, parent_vertex) = 1;
135 acyclic_connectivity(parent_vertex, vert) = 1;
136
137 current_layer_vertices.push_back(vert);
138 vertices_in_tree[vert] = 1;
139 parents_neighbours[vert] = {};
140 }
141 next_layer_vertices.clear();
142 }
143
144 return PathHandler(acyclic_connectivity);
145}

147std::list<unsigned> PathHandler::find_path(unsigned i, unsigned j) {
148 std::list<unsigned> path{i};
149 while (i != j) {
150 i = path_matrix_(i, j);
151 path.push_back(i);
152 }
153 return path;

156std::vector<Node> find_hampath(const Architecture &arch, long timeout) {
157 unsigned n_nodes = arch.n_nodes();
158 std::vector<Qubit> arch_qubits;
159 for (unsigned i = 0; i < n_nodes; i++) {
160 arch_qubits.push_back(Qubit(i));
161 }
162 QubitGraph q_graph(arch_qubits);
163 for (unsigned i = 0; i != n_nodes - 1; ++i) {
164 q_graph.add_connection(Qubit(i), Qubit(i + 1));
165 }
166
167 Architecture::UndirectedConnGraph undirected_target =
168 arch.get_undirected_connectivity();
169 QubitGraph::UndirectedConnGraph undirected_pattern =
170 q_graph.get_undirected_connectivity();
171 std::vector<boost::bimap<Qubit, Node>> all_maps =
172 get_weighted_subgraph_monomorphisms(
173 undirected_pattern, undirected_target, 1, timeout, false);
174
175 /* Architecture has no hampath, sad. */
176 if (all_maps.empty()) {
177 throw NoHamiltonPath(
178 "[AAS]: no Hamilton path found in the given architecture, CNOT "
179 "synthesis stopped. Please try an alternative CNotSynthType.");
180 }
181
182 /* Left: line, Right: input architecture. */
183 const boost::bimap<Qubit, Node> &qmap = all_maps[0];
184 std::vector<Node> hampath;
185 for (auto it = qmap.left.begin(); it != qmap.left.end(); ++it) {
186 hampath.push_back(it->second);
187 }
188 return hampath;
189}

191IterationOrder::IterationOrder(const Architecture &arch) {
192 std::set<Node> visited_nodes;
193
194 Node no = *arch.nodes().begin();
195
196 iterationorder.push_back(no);
197 visited_nodes.insert(no);
198
199 unsigned whilecount = 0;
200 while ((visited_nodes.size() < arch.n_nodes()) &&
201 (whilecount < arch.n_nodes())) {
202 for (auto edge : arch.get_all_edges_vec()) {
203 if (visited_nodes.count(edge.first) &&
204 (!visited_nodes.count(edge.second))) {
205 iterationorder.push_back(edge.second);
206 visited_nodes.insert(edge.second);
207 edgelist.push_back(edge);
208 } else if (
209 (!visited_nodes.count(edge.first)) &&
210 (visited_nodes.count(edge.second))) {
211 iterationorder.push_back(edge.first);
212 visited_nodes.insert(edge.first);
213 edgelist.push_back(edge);
214 }
215 }
216 ++whilecount;
217 }
218
219 if (visited_nodes.size() != arch.n_nodes()) {
220 throw std::logic_error("Unconnected architecture");
221 }
222
223 std::reverse(iterationorder.begin(), iterationorder.end());
224}
