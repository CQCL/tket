26// A nonzero entry in the matrix containing only 1,0,-1 values,
27// which we will build up for the tensor product
28// (taking out a factor of i from Y).
29typedef Eigen::Triplet<int> Entry;
30
31static std::map<Pauli, std::array<Entry, 2>> get_pauli_map() {
32 std::map<Pauli, std::array<Entry, 2>> result;
33 {
34 auto& i_matr = result[Pauli::I];
35 i_matr[0] = Entry(0, 0, 1);
36 i_matr[1] = Entry(1, 1, 1);
37 }
38 {
39 auto& x_matr = result[Pauli::X];
40 x_matr[0] = Entry(0, 1, 1);
41 x_matr[1] = Entry(1, 0, 1);
42 }
43 {
44 // Remove a factor of i.
45 auto& y_matr = result[Pauli::Y];
46 y_matr[0] = Entry(0, 1, -1);
47 y_matr[1] = Entry(1, 0, 1);
48 }
49 {
50 auto& z_matr = result[Pauli::Z];
51 z_matr[0] = Entry(0, 0, 1);
52 z_matr[1] = Entry(1, 1, -1);
53 }
54 return result;
55}

57// The tensor product of the left and right entries.
58static Entry get_combined_entry(const Entry& left, const Entry& right) {
59 return Entry(
60 2 * left.row() + right.row(), 2 * left.col() + right.col(),
61 left.value() * right.value());
62}

65// To save repeated reallocation, build the data up inside here.
66struct PauliExpBoxUnitaryCalculator {
67 // The two nonzero entries of the 2x2 matrix.
68 const std::map<Pauli, std::array<Entry, 2>> pauli_map;
69
70 // The tensor product matrix, all factors of i removed.
71 // Although PauliTensor could probably be made to do the work for us, it is
72 // more complicated than we need.
73 std::vector<Entry> sparse_matrix;
74
75 // The number of Y which occurred.
76 unsigned power_of_i;
77
78 // A work vector to avoid repeated reallocation.
79 std::vector<TripletCd> triplets;
80
81 // We need to remember which diagonal entries were filled,
82 // to construct the final triplets.
83 std::vector<unsigned> set_diagonals;
84
85 // In the tensor product of the given entry in "sparse_matrix"
86 // with the new 2x2 Pauli matrix on the right,
87 // add the new entries to "sparse_matrix"
88 // (including changing the existing entry).
89 // I.e., each entry in "sparse_matrix" is a 1 or -1,
90 // which adds two more +/-1 values to the list of nonzero entries
91 // (whilst being removed itself),
92 // since we take out a factor of i from Y.
93 void add_entries(unsigned sparse_matrix_index, Pauli pauli);
94
95 // Call this to begin entering a new Pauli string "manually".
96 void clear();
97
98 // Add a single Pauli to the RIGHT of the current tensor product.
99 void append(Pauli pauli);
100
101 void fill_triplets(double phase);
102
103 PauliExpBoxUnitaryCalculator();
104
105 static PauliExpBoxUnitaryCalculator& get();
106};

116void PauliExpBoxUnitaryCalculator::clear() {

123void PauliExpBoxUnitaryCalculator::add_entries(
124 unsigned sparse_matrix_index, Pauli pauli) {
125 TKET_ASSERT(sparse_matrix_index < sparse_matrix.size());
126 const auto& single_pauli = pauli_map.at(pauli);
127 sparse_matrix.push_back(
128 get_combined_entry(sparse_matrix[sparse_matrix_index], single_pauli[0]));
129 // Overwrite existing entry.
130 // The reference is valid until after the new entry returns!
131 auto& existing_entry = sparse_matrix[sparse_matrix_index];
132 existing_entry = get_combined_entry(existing_entry, single_pauli[1]);
133}

135void PauliExpBoxUnitaryCalculator::append(Pauli pauli) {
136 const auto current_size = sparse_matrix.size();
137 if (pauli == Pauli::Y) {
138 ++power_of_i;
139 }
140 for (unsigned ii = 0; ii < current_size; ++ii) {
141 add_entries(ii, pauli);
142 }
143}

145void PauliExpBoxUnitaryCalculator::fill_triplets(double phase) {
146 triplets.reserve(sparse_matrix.size());
147 triplets.clear();
148
149 // If M^2 = I then exp(itM) = cos(t)I + i.sin(t)M.
150 const double angle = -0.5 * PI * phase;
151 const double cc = std::cos(angle);
152 const double ss = std::sin(angle);
153
154 // TODO: efficiency: (i) check if cos(t) or sin(t) is nearly zero;
155 // (ii): fill the I coefficients FIRST, so we don't have to bother with
156 // "set_diagonals" (as we can overwrite diagonal entries directly).
157 const std::complex<double> identity_coefficient(cc);
158
159 // We took out factors of i from M (coming from Pauli Y),
160 // so have to put them back in.
161 power_of_i = power_of_i % 4;
162 const auto matrix_coefficient =
163 std::polar(1.0, 0.5 * (power_of_i + 1) * PI) * ss;
164
165 // When we form the sparse representation of the matrix exp(itM),
166 // we must check which diagonal entries were missed out,
167 // coming from the cos(t)I term.
168 set_diagonals.clear();
169
170 for (const auto& entry : sparse_matrix) {
171 std::complex<double> value =
172 matrix_coefficient * std::complex<double>(entry.value());
173
174 if (entry.row() == entry.col()) {
175 set_diagonals.push_back(entry.row());
176 value += identity_coefficient;
177 }
178 triplets.emplace_back(entry.row(), entry.col(), value);
179 }
180 // Complexity: let there be k diagonals set, out of N, and you want
181 // the other N-k (remembering that N = 2^n can be large).
182 // Here, we storing the diagonals we DON'T want in a std::vector, sort them,
183 // then select the OTHER elements. This takes time
184 // O(k.log k) + O(N.log k) time (with pretty small O constants,
185 // since sorting a std::vector is fast).
186 //
187 // If instead we use std::set to store all diagonals,
188 // remove each as it is set, and finally iterate through the
189 // REMAINING elements, it would be
190 // O(N.log N) + O((N-k).log N) + O(k.log k) time.
191 // This is probably a bit worse.
192 //
193 // If we really wanted to speed it up, we'd use bit operations.
194 std::sort(set_diagonals.begin(), set_diagonals.end());
195 for (unsigned ii = 0; ii < sparse_matrix.size(); ++ii) {
196 if (!std::binary_search(set_diagonals.cbegin(), set_diagonals.cend(), ii)) {
197 triplets.emplace_back(ii, ii, identity_coefficient);
198 }
199 }
200}

203std::vector<TripletCd> get_triplets(const PauliExpBox& box) {
204 const auto phase_optional = eval_expr(box.get_phase());
205 if (!phase_optional) {
206 throw SymbolsNotSupported(
207 "PauliExpBoxUnitaryCalculator called "
208 "with symbolic phase parameter");
209 }
210 const auto pauli_string = box.get_paulis();
211 auto& calculator = PauliExpBoxUnitaryCalculator::get();
212 calculator.clear();
213 for (auto pauli : pauli_string) {
214 calculator.append(pauli);
215 }
216 calculator.fill_triplets(phase_optional.value());
217 return calculator.triplets;
218}
