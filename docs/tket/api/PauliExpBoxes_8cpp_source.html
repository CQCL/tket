205 PauliExpCommutingSetBox::PauliExpCommutingSetBox(
206  const std::vector<SymPauliTensor> &pauli_gadgets,
207  CXConfigType cx_config_type)
208  : Box(OpType::PauliExpCommutingSetBox),
209  pauli_gadgets_(pauli_gadgets),
210  cx_config_(cx_config_type) {
211  // check at least one gadget
212  if (pauli_gadgets.empty()) {
213  throw PauliExpBoxInvalidity(
214  "PauliExpCommutingSetBox requires at least one Pauli string");
215  }
216  // check all gadgets have same Pauli string length
217  auto n_qubits = pauli_gadgets[0].size();
218  for (const auto &gadget : pauli_gadgets) {
219  if (gadget.size() != n_qubits) {
220  throw PauliExpBoxInvalidity(
221  "the Pauli strings within PauliExpCommutingSetBox must all be the "
222  "same length");
223  }
224  }
225  if (!this->paulis_commute()) {
226  throw PauliExpBoxInvalidity(
227  "Pauli strings used to define PauliExpCommutingSetBox must all "
228  "commute");
229  }
230  signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
231 }

250 bool PauliExpCommutingSetBox::paulis_commute() const {
251  for (auto string0 = pauli_gadgets_.begin(); string0 != pauli_gadgets_.end();
252  ++string0) {
253  for (auto string1 = std::next(string0); string1 != pauli_gadgets_.end();
254  ++string1) {
255  if (!string0->commutes_with(*string1)) {
256  return false;
257  }
258  }
259  }
260  return true;
261 }

300 void PauliExpCommutingSetBox::generate_circuit() const {
301  unsigned n_qubits = pauli_gadgets_[0].size();
302  Circuit circ(n_qubits);
303 
304  std::list<SpSymPauliTensor> gadgets;
305  for (const auto &pauli_gadget : pauli_gadgets_) {
306  gadgets.push_back((SpSymPauliTensor)pauli_gadget);
307  }
308  std::set<Qubit> qubits;
309  for (unsigned i = 0; i < n_qubits; i++) qubits.insert(Qubit(i));
310 
311  Circuit cliff_circ = mutual_diagonalise(gadgets, qubits, cx_config_);
312 
313  Circuit phase_poly_circ(n_qubits);
314 
315  for (const SpSymPauliTensor &pgp : gadgets) {
316  phase_poly_circ.append(pauli_gadget(pgp, CXConfigType::Snake));
317  }
318  phase_poly_circ.decompose_boxes_recursively();
319  PhasePolyBox ppbox(phase_poly_circ);
320  Circuit after_synth_circ = *ppbox.to_circuit();
321 
322  ConjugationBox box(
323  std::make_shared<CircBox>(cliff_circ),
324  std::make_shared<CircBox>(after_synth_circ));
325 
326  circ.add_box(box, circ.all_qubits());
327 
328  circ_ = std::make_shared<Circuit>(circ);
329 }

385 void append_pauli_gadget_pair_as_box(
386  Circuit &circ, const SpSymPauliTensor &pauli0,
387  const SpSymPauliTensor &pauli1, CXConfigType cx_config) {
388  std::vector<Qubit> mapping;
389  std::vector<Pauli> paulis0;
390  std::vector<Pauli> paulis1;
391  QubitPauliMap p1map = pauli1.string;
392  // add paulis for qubits in pauli0_string
393  for (const std::pair<const Qubit, Pauli> &term : pauli0.string) {
394  mapping.push_back(term.first);
395  paulis0.push_back(term.second);
396  auto found = p1map.find(term.first);
397  if (found == p1map.end()) {
398  paulis1.push_back(Pauli::I);
399  } else {
400  paulis1.push_back(found->second);
401  p1map.erase(found);
402  }
403  }
404  // add paulis for qubits in pauli1_string that weren't in pauli0_string
405  for (const std::pair<const Qubit, Pauli> &term : p1map) {
406  mapping.push_back(term.first);
407  paulis1.push_back(term.second);
408  paulis0.push_back(Pauli::I); // If pauli0_string contained qubit, would
409  // have been handled above
410  }
411  PauliExpPairBox box(
412  SymPauliTensor(paulis0, pauli0.coeff),
413  SymPauliTensor(paulis1, pauli1.coeff), cx_config);
414  circ.add_box(box, mapping);
415 }

417 void append_commuting_pauli_gadget_set_as_box(
418  Circuit &circ, const std::list<SpSymPauliTensor> &gadgets,
419  CXConfigType cx_config) {
420  // Translate from QubitPauliTensors to vectors of Paulis of same length
421  // Preserves ordering of qubits
422 
423  std::set<Qubit> all_qubits;
424  for (const SpSymPauliTensor &gadget : gadgets) {
425  for (const std::pair<const Qubit, Pauli> &qubit_pauli : gadget.string) {
426  all_qubits.insert(qubit_pauli.first);
427  }
428  }
429 
430  std::vector<Qubit> mapping;
431  for (const auto &qubit : all_qubits) {
432  mapping.push_back(qubit);
433  }
434 
435  std::vector<SymPauliTensor> pauli_gadgets;
436  for (const SpSymPauliTensor &gadget : gadgets) {
437  SymPauliTensor &new_gadget =
438  pauli_gadgets.emplace_back(DensePauliMap{}, gadget.coeff);
439  for (const Qubit &qubit : mapping) {
440  new_gadget.string.push_back(gadget.get(qubit));
441  }
442  }
443 
444  PauliExpCommutingSetBox box(pauli_gadgets, cx_config);
445  circ.add_box(box, mapping);
446 }
