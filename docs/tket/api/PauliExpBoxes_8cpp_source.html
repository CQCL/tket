205PauliExpCommutingSetBox::PauliExpCommutingSetBox(
206 const std::vector<SymPauliTensor> &pauli_gadgets,
207 CXConfigType cx_config_type)
208 : Box(OpType::PauliExpCommutingSetBox),
209 pauli_gadgets_(pauli_gadgets),
210 cx_config_(cx_config_type) {
211 // check at least one gadget
212 if (pauli_gadgets.empty()) {
213 throw PauliExpBoxInvalidity(
214 "PauliExpCommutingSetBox requires at least one Pauli string");
215 }
216 // check all gadgets have same Pauli string length
217 auto n_qubits = pauli_gadgets[0].size();
218 for (const auto &gadget : pauli_gadgets) {
219 if (gadget.size() != n_qubits) {
220 throw PauliExpBoxInvalidity(
221 "the Pauli strings within PauliExpCommutingSetBox must all be the "
222 "same length");
223 }
224 }
225 if (!this->paulis_commute()) {
226 throw PauliExpBoxInvalidity(
227 "Pauli strings used to define PauliExpCommutingSetBox must all "
228 "commute");
229 }
230 signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
231}

246 return equiv_0(4 * pauli_exp.coeff) || pauli_exp.string.empty();

250bool PauliExpCommutingSetBox::paulis_commute() const {
251 for (auto string0 = pauli_gadgets_.begin(); string0 != pauli_gadgets_.end();
252 ++string0) {
253 for (auto string1 = std::next(string0); string1 != pauli_gadgets_.end();
254 ++string1) {
255 if (!string0->commutes_with(*string1)) {
256 return false;
257 }
258 }
259 }
260 return true;
261}

300void PauliExpCommutingSetBox::generate_circuit() const {
301 unsigned n_qubits = pauli_gadgets_[0].size();
302 Circuit circ(n_qubits);
303
304 std::list<SpSymPauliTensor> gadgets;
305 for (const auto &pauli_gadget : pauli_gadgets_) {
306 gadgets.push_back((SpSymPauliTensor)pauli_gadget);
307 }
308 std::set<Qubit> qubits;
309 for (unsigned i = 0; i < n_qubits; i++) qubits.insert(Qubit(i));
310
311 Circuit cliff_circ = mutual_diagonalise(gadgets, qubits, cx_config_);
312
313 Circuit phase_poly_circ(n_qubits);
314
315 for (const SpSymPauliTensor &pgp : gadgets) {
316 phase_poly_circ.append(pauli_gadget(pgp, CXConfigType::Snake));
317 }
318 phase_poly_circ.decompose_boxes_recursively();
319 PhasePolyBox ppbox(phase_poly_circ);
320 Circuit after_synth_circ = *ppbox.to_circuit();
321
322 ConjugationBox box(
323 std::make_shared<CircBox>(cliff_circ),
324 std::make_shared<CircBox>(after_synth_circ));
325
326 circ.add_box(box, circ.all_qubits());
327
328 circ_ = std::make_shared<Circuit>(circ);
329}

537DensePauliMap pad_sparse_pauli_map(
538 const QubitPauliMap &sparse_string, size_t size) {
539 // n.b. as a helper method this can assume that all qubits in sparse string
540 // are labelled with the default qubit register and no index is larger than or
541 // equal to the size
542 std::vector<Pauli> dense_string(size, Pauli::I);
543 for (const auto &letter : sparse_string) {
544 TKET_ASSERT(letter.first.reg_name() == q_default_reg());
545 std::vector<unsigned> index = letter.first.index();
546 TKET_ASSERT(index.size() == 1);
547 TKET_ASSERT(index[0] < size);
548 dense_string[index[0]] = letter.second;
549 }
550 return dense_string;
551}

553void TermSequenceBox::generate_circuit() const {
554 // constructor guarantees all gadgets are the same size
555 unsigned n_qubits = pauli_gadgets_[0].size();
556 Circuit circ(n_qubits);
557
558 // Saves later handling
559 if (this->pauli_gadgets_.empty()) {
560 circ_ = std::make_shared<Circuit>(circ);
561 return;
562 }
563
564 // First combine any PauliGadgets with the same dense string into the same
565 // term
566 // n.b. the map assigns an ordering to the Pauli terms, meaning it
567 // is an algorithm-affecting step
568 std::map<DensePauliMap, Expr> reduced_pauli_gadgets;
569 for (const SymPauliTensor &pauli_gadget : this->pauli_gadgets_) {
570 auto it = reduced_pauli_gadgets.find(pauli_gadget.string);
571 if (it != reduced_pauli_gadgets.end()) {
572 it->second += pauli_gadget.coeff;
573 } else {
574 reduced_pauli_gadgets.insert(
575 std::make_pair(pauli_gadget.string, pauli_gadget.coeff));
576 }
577 }
578
579 // Construct circuit depending on specified strategy, using other Pauli
580 // Exponential Boxes
581 switch (this->synth_strategy_) {
582 case Transforms::PauliSynthStrat::Individual: {
583 for (auto it = reduced_pauli_gadgets.begin();
584 it != reduced_pauli_gadgets.end(); ++it) {
585 circ.add_box(
586 PauliExpBox(
587 SymPauliTensor(it->first, it->second), this->cx_configuration_),
588 circ.all_qubits());
589 }
590 break;
591 }
592 case Transforms::PauliSynthStrat::Pairwise: {
593 auto it = reduced_pauli_gadgets.begin();
594 // if there is an odd number of gadgets, add first gadget as a single
595 if (reduced_pauli_gadgets.size() % 2 == 1) {
596 circ.add_box(
597 PauliExpBox(
598 SymPauliTensor(it->first, it->second), this->cx_configuration_),
599 circ.all_qubits());
600 ++it;
601 }
602 while (it != reduced_pauli_gadgets.end()) {
603 auto next_it = std::next(it);
604 TKET_ASSERT(next_it != reduced_pauli_gadgets.end());
605 circ.add_box(
606 PauliExpPairBox(
607 SymPauliTensor(it->first, it->second),
608 SymPauliTensor(next_it->first, next_it->second),
609 this->cx_configuration_),
610 circ.all_qubits());
611 std::advance(it, 2);
612 }
613 break;
614 }
615 case Transforms::PauliSynthStrat::Sets: {
616 // term_sequence expects std::list<SpPauliString>, i.e. QubitPauliMap
617 // instead of DensePauliMap first convert the keys of
618 // reduced_pauli_gadgets to a std::list<SpPauliString> object
619 std::list<SpPauliString> all_terms;
620 for (const auto &gadget : reduced_pauli_gadgets) {
621 all_terms.push_back(SpPauliString(gadget.first));
622 }
623 // then get the term sequence into commuting sets
624 std::list<std::list<SpPauliString>> commuting_sets = term_sequence(
625 all_terms, this->partition_strategy_, this->graph_colouring_);
626 // construct each commuting set using PauliExpCommutingSetBox:
627 // 1) pad each QubitPauliMap into a DensePauliMap by adding Pauli.I terms
628 // for missing keys 2) retrieve the exponential coefficient from
629 // reduced_pauli_gadgets 3) construct a SymPauliTensor object as the
630 // argument to PauliExpCommutingSetBox
631 for (const std::list<SpPauliString> &strings : commuting_sets) {
632 std::vector<SymPauliTensor> commuting_gadgets;
633 for (const SpPauliString &string : strings) {
634 DensePauliMap padded_string =
635 pad_sparse_pauli_map(string.string, n_qubits);
636 commuting_gadgets.push_back(SymPauliTensor(
637 padded_string, reduced_pauli_gadgets[padded_string]));
638 }
639 circ.add_box(
640 PauliExpCommutingSetBox(commuting_gadgets, this->cx_configuration_),
641 circ.all_qubits());
642 }
643 break;
644 }
645 case Transforms::PauliSynthStrat::Greedy: {
646 std::vector<SymPauliTensor> unordered_gadgets;
647 for (const auto &pair : reduced_pauli_gadgets) {
648 unordered_gadgets.push_back(SymPauliTensor(pair.first, pair.second));
649 }
650 circ = Transforms::GreedyPauliSimp::greedy_pauli_set_synthesis(
651 unordered_gadgets, this->depth_weight_);
652 Transforms::singleq_clifford_sweep().apply(circ);
653 break;
654 }
655 default: {
656 throw std::logic_error(
657 "TermSequenceBox passed an unsupported PauliSynthStrat");
658 }
659 }
660 circ_ = std::make_shared<Circuit>(circ);
661}

677void append_pauli_gadget_pair_as_box(
678 Circuit &circ, const SpSymPauliTensor &pauli0,
679 const SpSymPauliTensor &pauli1, CXConfigType cx_config) {
680 std::vector<Qubit> mapping;
681 std::vector<Pauli> paulis0;
682 std::vector<Pauli> paulis1;
683 QubitPauliMap p1map = pauli1.string;
684 // add paulis for qubits in pauli0_string
685 for (const std::pair<const Qubit, Pauli> &term : pauli0.string) {
686 mapping.push_back(term.first);
687 paulis0.push_back(term.second);
688 auto found = p1map.find(term.first);
689 if (found == p1map.end()) {
690 paulis1.push_back(Pauli::I);
691 } else {
692 paulis1.push_back(found->second);
693 p1map.erase(found);
694 }
695 }
696 // add paulis for qubits in pauli1_string that weren't in pauli0_string
697 for (const std::pair<const Qubit, Pauli> &term : p1map) {
698 mapping.push_back(term.first);
699 paulis1.push_back(term.second);
700 paulis0.push_back(Pauli::I); // If pauli0_string contained qubit, would
701 // have been handled above
702 }
703 PauliExpPairBox box(
704 SymPauliTensor(paulis0, pauli0.coeff),
705 SymPauliTensor(paulis1, pauli1.coeff), cx_config);
706 circ.add_box(box, mapping);
707}

709void append_commuting_pauli_gadget_set_as_box(
710 Circuit &circ, const std::list<SpSymPauliTensor> &gadgets,
711 CXConfigType cx_config) {
712 // Translate from QubitPauliTensors to vectors of Paulis of same length
713 // Preserves ordering of qubits
714
715 std::set<Qubit> all_qubits;
716 for (const SpSymPauliTensor &gadget : gadgets) {
717 for (const std::pair<const Qubit, Pauli> &qubit_pauli : gadget.string) {
718 all_qubits.insert(qubit_pauli.first);
719 }
720 }
721
722 std::vector<Qubit> mapping;
723 for (const auto &qubit : all_qubits) {
724 mapping.push_back(qubit);
725 }
726
727 std::vector<SymPauliTensor> pauli_gadgets;
728 for (const SpSymPauliTensor &gadget : gadgets) {
729 SymPauliTensor &new_gadget =
730 pauli_gadgets.emplace_back(DensePauliMap{}, gadget.coeff);
731 for (const Qubit &qubit : mapping) {
732 new_gadget.string.push_back(gadget.get(qubit));
733 }
734 }
735
736 PauliExpCommutingSetBox box(pauli_gadgets, cx_config);
737 circ.add_box(box, mapping);
738}
