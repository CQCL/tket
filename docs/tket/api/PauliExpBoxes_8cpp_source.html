205PauliExpCommutingSetBox::PauliExpCommutingSetBox(
206 const std::vector<SymPauliTensor> &pauli_gadgets,
207 CXConfigType cx_config_type)
208 : Box(OpType::PauliExpCommutingSetBox),
209 pauli_gadgets_(pauli_gadgets),
210 cx_config_(cx_config_type) {
211 // check at least one gadget
212 if (pauli_gadgets.empty()) {
213 throw PauliExpBoxInvalidity(
214 "PauliExpCommutingSetBox requires at least one Pauli string");
215 }
216 // check all gadgets have same Pauli string length
217 auto n_qubits = pauli_gadgets[0].size();
218 for (const auto &gadget : pauli_gadgets) {
219 if (gadget.size() != n_qubits) {
220 throw PauliExpBoxInvalidity(
221 "the Pauli strings within PauliExpCommutingSetBox must all be the "
222 "same length");
223 }
224 }
225 if (!this->paulis_commute()) {
226 throw PauliExpBoxInvalidity(
227 "Pauli strings used to define PauliExpCommutingSetBox must all "
228 "commute");
229 }
230 signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
231}

242bool PauliExpCommutingSetBox::is_clifford() const {

246 return equiv_0(4 * pauli_exp.coeff) || pauli_exp.string.empty();

250bool PauliExpCommutingSetBox::paulis_commute() const {
251 for (auto string0 = pauli_gadgets_.begin(); string0 != pauli_gadgets_.end();
252 ++string0) {
253 for (auto string1 = std::next(string0); string1 != pauli_gadgets_.end();
254 ++string1) {
255 if (!string0->commutes_with(*string1)) {
256 return false;
257 }
258 }
259 }
260 return true;
261}

300void PauliExpCommutingSetBox::generate_circuit() const {
301 unsigned n_qubits = pauli_gadgets_[0].size();
302 Circuit circ(n_qubits);
303
304 std::list<SpSymPauliTensor> gadgets;
305 for (const auto &pauli_gadget : pauli_gadgets_) {
306 gadgets.push_back((SpSymPauliTensor)pauli_gadget);
307 }
308 std::set<Qubit> qubits;
309 for (unsigned i = 0; i < n_qubits; i++) qubits.insert(Qubit(i));
310
311 Circuit cliff_circ = mutual_diagonalise(gadgets, qubits, cx_config_);
312
313 Circuit phase_poly_circ(n_qubits);
314
315 for (const SpSymPauliTensor &pgp : gadgets) {
316 phase_poly_circ.append(pauli_gadget(pgp, CXConfigType::Snake));
317 }
318 phase_poly_circ.decompose_boxes_recursively();
319 PhasePolyBox ppbox(phase_poly_circ);
320 Circuit after_synth_circ = *ppbox.to_circuit();
321
322 ConjugationBox box(
323 std::make_shared<CircBox>(cliff_circ),
324 std::make_shared<CircBox>(after_synth_circ));
325
326 circ.add_box(box, circ.all_qubits());
327
328 circ_ = std::make_shared<Circuit>(circ);
329}

373TermSequenceBox::TermSequenceBox(
374 const std::vector<SymPauliTensor> &pauli_gadgets,
375 Transforms::PauliSynthStrat synth_strategy,
376 PauliPartitionStrat partition_strategy, GraphColourMethod graph_colouring,
377 CXConfigType cx_configuration, double depth_weight)
378 : Box(OpType::TermSequenceBox),
379 pauli_gadgets_(pauli_gadgets),
380 synth_strategy_(synth_strategy),
381 partition_strategy_(partition_strategy),
382 graph_colouring_(graph_colouring),
383 cx_configuration_(cx_configuration),
384 depth_weight_(depth_weight) {
385 // check at least one gadget
386 if (pauli_gadgets.empty()) {
387 signature_ = op_signature_t(0, EdgeType::Quantum);
388 } else {
389 // check all gadgets have same Pauli string length
390 auto n_qubits = pauli_gadgets[0].size();
391 for (const auto &gadget : pauli_gadgets) {
392 if (gadget.size() != n_qubits) {
393 throw PauliExpBoxInvalidity(
394 "the Pauli strings within TermSequenceBox must all be the "
395 "same length - add Pauli.I to pad strings to required length.");
396 }
397 if (synth_strategy == Transforms::PauliSynthStrat::Greedy &&
398 gadget.size() ==
399 std::count(
400 gadget.string.begin(), gadget.string.end(), Pauli::I)) {
401 throw PauliExpBoxInvalidity(
402 "A TermSequenceBox using the Greedy strategy does not support "
403 "rotations with an identity Pauli string. "
404 "Note: global phase is not preserved with the Greedy synthesis.");
405 }
406 }
407 signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
408 }
409}

546DensePauliMap pad_sparse_pauli_map(
547 const QubitPauliMap &sparse_string, size_t size) {
548 // n.b. as a helper method this can assume that all qubits in sparse string
549 // are labelled with the default qubit register and no index is larger than or
550 // equal to the size
551 std::vector<Pauli> dense_string(size, Pauli::I);
552 for (const auto &letter : sparse_string) {
553 TKET_ASSERT(letter.first.reg_name() == q_default_reg());
554 std::vector<unsigned> index = letter.first.index();
555 TKET_ASSERT(index.size() == 1);
556 TKET_ASSERT(index[0] < size);
557 dense_string[index[0]] = letter.second;
558 }
559 return dense_string;
560}

562void TermSequenceBox::generate_circuit() const {
563 // constructor guarantees all gadgets are the same size
564 unsigned n_qubits = pauli_gadgets_[0].size();
565 Circuit circ(n_qubits);
566
567 // Saves later handling
568 if (this->pauli_gadgets_.empty()) {
569 circ_ = std::make_shared<Circuit>(circ);
570 return;
571 }
572
573 // First combine any PauliGadgets with the same dense string into the same
574 // term
575 // n.b. the map assigns an ordering to the Pauli terms, meaning it
576 // is an algorithm-affecting step
577 std::map<DensePauliMap, Expr> reduced_pauli_gadgets;
578 for (const SymPauliTensor &pauli_gadget : this->pauli_gadgets_) {
579 auto it = reduced_pauli_gadgets.find(pauli_gadget.string);
580 if (it != reduced_pauli_gadgets.end()) {
581 it->second += pauli_gadget.coeff;
582 } else {
583 reduced_pauli_gadgets.insert(
584 std::make_pair(pauli_gadget.string, pauli_gadget.coeff));
585 }
586 }
587
588 // Construct circuit depending on specified strategy, using other Pauli
589 // Exponential Boxes
590 switch (this->synth_strategy_) {
591 case Transforms::PauliSynthStrat::Individual: {
592 for (auto it = reduced_pauli_gadgets.begin();
593 it != reduced_pauli_gadgets.end(); ++it) {
594 circ.add_box(
595 PauliExpBox(
596 SymPauliTensor(it->first, it->second), this->cx_configuration_),
597 circ.all_qubits());
598 }
599 break;
600 }
601 case Transforms::PauliSynthStrat::Pairwise: {
602 auto it = reduced_pauli_gadgets.begin();
603 // if there is an odd number of gadgets, add first gadget as a single
604 if (reduced_pauli_gadgets.size() % 2 == 1) {
605 circ.add_box(
606 PauliExpBox(
607 SymPauliTensor(it->first, it->second), this->cx_configuration_),
608 circ.all_qubits());
609 ++it;
610 }
611 while (it != reduced_pauli_gadgets.end()) {
612 auto next_it = std::next(it);
613 TKET_ASSERT(next_it != reduced_pauli_gadgets.end());
614 circ.add_box(
615 PauliExpPairBox(
616 SymPauliTensor(it->first, it->second),
617 SymPauliTensor(next_it->first, next_it->second),
618 this->cx_configuration_),
619 circ.all_qubits());
620 std::advance(it, 2);
621 }
622 break;
623 }
624 case Transforms::PauliSynthStrat::Sets: {
625 // term_sequence expects std::list<SpPauliString>, i.e. QubitPauliMap
626 // instead of DensePauliMap first convert the keys of
627 // reduced_pauli_gadgets to a std::list<SpPauliString> object
628 std::list<SpPauliString> all_terms;
629 for (const auto &gadget : reduced_pauli_gadgets) {
630 all_terms.push_back(SpPauliString(gadget.first));
631 }
632 // then get the term sequence into commuting sets
633 std::list<std::list<SpPauliString>> commuting_sets = term_sequence(
634 all_terms, this->partition_strategy_, this->graph_colouring_);
635 // construct each commuting set using PauliExpCommutingSetBox:
636 // 1) pad each QubitPauliMap into a DensePauliMap by adding Pauli.I terms
637 // for missing keys 2) retrieve the exponential coefficient from
638 // reduced_pauli_gadgets 3) construct a SymPauliTensor object as the
639 // argument to PauliExpCommutingSetBox
640 for (const std::list<SpPauliString> &strings : commuting_sets) {
641 std::vector<SymPauliTensor> commuting_gadgets;
642 for (const SpPauliString &string : strings) {
643 DensePauliMap padded_string =
644 pad_sparse_pauli_map(string.string, n_qubits);
645 commuting_gadgets.push_back(SymPauliTensor(
646 padded_string, reduced_pauli_gadgets[padded_string]));
647 }
648 circ.add_box(
649 PauliExpCommutingSetBox(commuting_gadgets, this->cx_configuration_),
650 circ.all_qubits());
651 }
652 break;
653 }
654 case Transforms::PauliSynthStrat::Greedy: {
655 std::vector<SymPauliTensor> unordered_gadgets;
656 for (const auto &pair : reduced_pauli_gadgets) {
657 unordered_gadgets.push_back(SymPauliTensor(pair.first, pair.second));
658 }
659 circ = Transforms::GreedyPauliSimp::greedy_pauli_set_synthesis(
660 unordered_gadgets, this->depth_weight_);
661 Transforms::singleq_clifford_sweep().apply(circ);
662 break;
663 }
664 default: {
665 throw std::logic_error(
666 "TermSequenceBox passed an unsupported PauliSynthStrat");
667 }
668 }
669 circ_ = std::make_shared<Circuit>(circ);
670}

686void append_pauli_gadget_pair_as_box(
687 Circuit &circ, const SpSymPauliTensor &pauli0,
688 const SpSymPauliTensor &pauli1, CXConfigType cx_config) {
689 std::vector<Qubit> mapping;
690 std::vector<Pauli> paulis0;
691 std::vector<Pauli> paulis1;
692 QubitPauliMap p1map = pauli1.string;
693 // add paulis for qubits in pauli0_string
694 for (const std::pair<const Qubit, Pauli> &term : pauli0.string) {
695 mapping.push_back(term.first);
696 paulis0.push_back(term.second);
697 auto found = p1map.find(term.first);
698 if (found == p1map.end()) {
699 paulis1.push_back(Pauli::I);
700 } else {
701 paulis1.push_back(found->second);
702 p1map.erase(found);
703 }
704 }
705 // add paulis for qubits in pauli1_string that weren't in pauli0_string
706 for (const std::pair<const Qubit, Pauli> &term : p1map) {
707 mapping.push_back(term.first);
708 paulis1.push_back(term.second);
709 paulis0.push_back(Pauli::I); // If pauli0_string contained qubit, would
710 // have been handled above
711 }
712 PauliExpPairBox box(
713 SymPauliTensor(paulis0, pauli0.coeff),
714 SymPauliTensor(paulis1, pauli1.coeff), cx_config);
715 circ.add_box(box, mapping);
716}

718void append_commuting_pauli_gadget_set_as_box(
719 Circuit &circ, const std::list<SpSymPauliTensor> &gadgets,
720 CXConfigType cx_config) {
721 // Translate from QubitPauliTensors to vectors of Paulis of same length
722 // Preserves ordering of qubits
723
724 std::set<Qubit> all_qubits;
725 for (const SpSymPauliTensor &gadget : gadgets) {
726 for (const std::pair<const Qubit, Pauli> &qubit_pauli : gadget.string) {
727 all_qubits.insert(qubit_pauli.first);
728 }
729 }
730
731 std::vector<Qubit> mapping;
732 for (const auto &qubit : all_qubits) {
733 mapping.push_back(qubit);
734 }
735
736 std::vector<SymPauliTensor> pauli_gadgets;
737 for (const SpSymPauliTensor &gadget : gadgets) {
738 SymPauliTensor &new_gadget =
739 pauli_gadgets.emplace_back(DensePauliMap{}, gadget.coeff);
740 for (const Qubit &qubit : mapping) {
741 new_gadget.string.push_back(gadget.get(qubit));
742 }
743 }
744
745 PauliExpCommutingSetBox box(pauli_gadgets, cx_config);
746 circ.add_box(box, mapping);
747}
