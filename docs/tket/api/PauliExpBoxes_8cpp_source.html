55 Op_ptr PauliExpBox::transpose() const {
56  std::vector<Pauli> paulis = get_paulis();
57  int number_y_pauli_mod2 =
58  std::count(paulis.begin(), paulis.end(), Pauli::Y) % 2;
59  // Negate the parameter if odd nr of Paulis (mult with 1 if mod2=0, -1 if
60  // mod2=1)
61  int t_fac = -number_y_pauli_mod2 * 2 + 1;
62  return std::make_shared<PauliExpBox>(paulis_, t_fac * t_, cx_config_);
63 }

146 // Get the multiplicative change factor in Pauli angle during transpose ( -1 for
147 // odd nr of Y, 1 otherwise)
148 int transpose_angle_factor(const std::vector<Pauli> &paulis) {
149  int pauli_odd_number_y = std::count_if(
150  paulis.begin(), paulis.end(),
151  [](auto pauli) { return pauli == Pauli::Y; }) %
152  2;
153  // transform 0 -> 1, 1 -> -1
154  return -pauli_odd_number_y * 2 + 1;
155 }

219 PauliExpCommutingSetBox::PauliExpCommutingSetBox(
220  const std::vector<std::pair<std::vector<Pauli>, Expr>> &pauli_gadgets,
221  CXConfigType cx_config_type)
222  : Box(OpType::PauliExpCommutingSetBox),
223  pauli_gadgets_(pauli_gadgets),
224  cx_config_(cx_config_type) {
225  // check at least one gadget
226  if (pauli_gadgets.empty()) {
227  throw PauliExpBoxInvalidity(
228  "PauliExpCommutingSetBox requires at least one Pauli string");
229  }
230  // check all gadgets have same Pauli string length
231  auto n_qubits = pauli_gadgets[0].first.size();
232  for (const auto &gadget : pauli_gadgets) {
233  if (gadget.first.size() != n_qubits) {
234  throw PauliExpBoxInvalidity(
235  "the Pauli strings within PauliExpCommutingSetBox must all be the "
236  "same length");
237  }
238  }
239  if (!this->paulis_commute()) {
240  throw PauliExpBoxInvalidity(
241  "Pauli strings used to define PauliExpCommutingSetBox must all "
242  "commute");
243  }
244  signature_ = op_signature_t(n_qubits, EdgeType::Quantum);
245 }

264 bool PauliExpCommutingSetBox::paulis_commute() const {
265  std::vector<QubitPauliString> pauli_strings;
266  pauli_strings.reserve(pauli_gadgets_.size());
267  for (const auto &pauli_gadget : pauli_gadgets_) {
268  pauli_strings.emplace_back(pauli_gadget.first);
269  }
270  for (auto string0 = pauli_strings.begin(); string0 != pauli_strings.end();
271  string0++) {
272  for (auto string1 = string0 + 1; string1 != pauli_strings.end();
273  string1++) {
274  if (!string0->commutes_with(*string1)) {
275  return false;
276  }
277  }
278  }
279  return true;
280 }

320 void PauliExpCommutingSetBox::generate_circuit() const {
321  unsigned n_qubits = pauli_gadgets_[0].first.size();
322  Circuit circ(n_qubits);
323 
324  std::list<std::pair<QubitPauliTensor, Expr>> gadgets;
325  for (const auto &pauli_gadget : pauli_gadgets_) {
326  gadgets.emplace_back(
327  QubitPauliTensor(pauli_gadget.first), pauli_gadget.second);
328  }
329  std::set<Qubit> qubits;
330  for (unsigned i = 0; i < n_qubits; i++) qubits.insert(Qubit(i));
331 
332  Circuit cliff_circ = mutual_diagonalise(gadgets, qubits, cx_config_);
333 
334  Circuit phase_poly_circ(n_qubits);
335 
336  for (const std::pair<QubitPauliTensor, Expr> &pgp : gadgets) {
337  append_single_pauli_gadget(phase_poly_circ, pgp.first, pgp.second);
338  }
339  phase_poly_circ.decompose_boxes_recursively();
340  PhasePolyBox ppbox(phase_poly_circ);
341  Circuit after_synth_circ = *ppbox.to_circuit();
342 
343  ConjugationBox box(
344  std::make_shared<CircBox>(cliff_circ),
345  std::make_shared<CircBox>(after_synth_circ));
346 
347  circ.add_box(box, circ.all_qubits());
348 
349  circ_ = std::make_shared<Circuit>(circ);
350 }

352 // check two gadges are semantically equal
353 static bool gadget_compare(
354  const std::vector<std::pair<std::vector<Pauli>, Expr>> &g1,
355  const std::vector<std::pair<std::vector<Pauli>, Expr>> &g2) {
356  return std::equal(
357  g1.begin(), g1.end(), g2.begin(), g2.end(),
358  [](const std::pair<std::vector<Pauli>, Expr> &a,
359  const std::pair<std::vector<Pauli>, Expr> &b) {
360  return a.first == b.first && equiv_expr(a.second, b.second, 4);
361  });
362 }
