21 void append_single_pauli_gadget(
22  Circuit &circ, const QubitPauliTensor &pauli, Expr angle,
23  CXConfigType cx_config) {
24  if (pauli.coeff == -1.) {
25  angle *= -1;
26  } else if (pauli.coeff != 1.) {
27  throw CircuitInvalidity("Pauli coefficient must be +/- 1");
28  }
29  std::vector<Pauli> string;
30  unit_map_t mapping;
31  unsigned i = 0;
32  for (const std::pair<const Qubit, Pauli> &term : pauli.string.map) {
33  string.push_back(term.second);
34  mapping.insert({Qubit(q_default_reg(), i), term.first});
35  i++;
36  }
37  Circuit gadget = pauli_gadget(string, angle, cx_config);
38  circ.append_with_map(gadget, mapping);
39 }

41 static void reduce_shared_qs_by_CX_snake(
42  Circuit &circ, std::set<Qubit> &match, QubitPauliTensor &pauli0,
43  QubitPauliTensor &pauli1) {
44  unsigned match_size = match.size();
45  while (match_size > 1) { // We allow one match left over
46  auto it = --match.end();
47  Qubit to_eliminate = *it;
48  match.erase(it);
49  Qubit helper = *match.rbegin();
50  // extend CX snake
51  circ.add_op<Qubit>(OpType::CX, {to_eliminate, helper});
52  pauli0.string.map.erase(to_eliminate);
53  pauli1.string.map.erase(to_eliminate);
54  match_size--;
55  }
56 }

58 static void reduce_shared_qs_by_CX_star(
59  Circuit &circ, std::set<Qubit> &match, QubitPauliTensor &pauli0,
60  QubitPauliTensor &pauli1) {
61  std::set<Qubit>::iterator iter = match.begin();
62  for (std::set<Qubit>::iterator next = match.begin(); match.size() > 1;
63  iter = next) {
64  ++next;
65  Qubit to_eliminate = *iter;
66  circ.add_op<Qubit>(OpType::CX, {to_eliminate, *match.rbegin()});
67  pauli0.string.map.erase(to_eliminate);
68  pauli1.string.map.erase(to_eliminate);
69  match.erase(iter);
70  }
71 }

73 static void reduce_shared_qs_by_CX_tree(
74  Circuit &circ, std::set<Qubit> &match, QubitPauliTensor &pauli0,
75  QubitPauliTensor &pauli1) {
76  while (match.size() > 1) {
77  std::set<Qubit> remaining;
78  std::set<Qubit>::iterator it = match.begin();
79  while (it != match.end()) {
80  Qubit maintained = *it;
81  it++;
82  remaining.insert(maintained);
83  if (it != match.end()) {
84  Qubit to_eliminate = *it;
85  it++;
86  circ.add_op<Qubit>(OpType::CX, {to_eliminate, maintained});
87  pauli0.string.map.erase(to_eliminate);
88  pauli1.string.map.erase(to_eliminate);
89  }
90  }
91  match = remaining;
92  }
93 }

95 static void reduce_shared_qs_by_CX_multiqgate(
96  Circuit &circ, std::set<Qubit> &match, QubitPauliTensor &pauli0,
97  QubitPauliTensor &pauli1) {
98  if (match.size() <= 1) {
99  return;
100  }
101  // last qubit is target
102  Qubit target = *match.rbegin();
103  while (match.size() > 1) {
104  std::set<Qubit>::iterator iter = match.begin();
105  if (match.size() == 2) {
106  // use CX
107  Qubit to_eliminate = *iter;
108  match.erase(iter);
109  pauli0.string.map.erase(to_eliminate);
110  pauli1.string.map.erase(to_eliminate);
111 
112  circ.add_op<Qubit>(OpType::CX, {to_eliminate, target});
113  } else {
114  // use XXPhase3
115  Qubit to_eliminate1 = *iter;
116  match.erase(iter++);
117  pauli0.string.map.erase(to_eliminate1);
118  pauli1.string.map.erase(to_eliminate1);
119 
120  Qubit to_eliminate2 = *iter;
121  match.erase(iter);
122  pauli0.string.map.erase(to_eliminate2);
123  pauli1.string.map.erase(to_eliminate2);
124 
125  circ.add_op<Qubit>(OpType::H, {to_eliminate1});
126  circ.add_op<Qubit>(OpType::H, {to_eliminate2});
127  circ.add_op<Qubit>(
128  OpType::XXPhase3, 0.5, {to_eliminate1, to_eliminate2, target});
129  circ.add_op<Qubit>(OpType::X, {target});
130  }
131  }
132 }

134 void append_pauli_gadget_pair(
135  Circuit &circ, QubitPauliTensor pauli0, Expr angle0,
136  QubitPauliTensor pauli1, Expr angle1, CXConfigType cx_config) {
137  /*
138  * Cowtan, Dilkes, Duncan, Simmons, Sivarajah: Phase Gadget Synthesis for
139  * Shallow Circuits, Lemma 4.9
140  * Let s and t be Pauli strings; then there exists a Clifford unitary U such
141  * that
142  * P(a, s) . P(b, t) = U . P(a, s') . P(b, t') . U^\dagger
143  * where s' and t' are Pauli strings with intersection at most 1.
144  *
145  * Follows the procedure to reduce the intersection of the gadgets and then
146  * synthesises the remainder individually.
147  */
148  pauli0.compress();
149  pauli1.compress();
150 
151  /*
152  * Step 1: Partition qubits into those just affected by pauli0 (just0) and
153  * pauli1 (just1), and those in both which either match or don't
154  */
155  std::set<Qubit> just0 = pauli0.own_qubits(pauli1);
156  std::set<Qubit> just1 = pauli1.own_qubits(pauli0);
157  std::set<Qubit> match = pauli0.common_qubits(pauli1);
158  std::set<Qubit> mismatch = pauli0.conflicting_qubits(pauli1);
159 
160  /*
161  * Step 2: Build the unitary U that minimises the intersection of the gadgets.
162  */
163  Circuit u;
164  for (const Qubit &qb : just0) u.add_qubit(qb);
165  for (const Qubit &qb : just1) u.add_qubit(qb);
166  for (const Qubit &qb : match) u.add_qubit(qb);
167  for (const Qubit &qb : mismatch) u.add_qubit(qb);
168 
169  /*
170  * Step 2.i: Remove (almost) all matches by converting to Z basis and applying
171  * CXs
172  */
173  for (const Qubit &qb : match) {
174  switch (pauli0.string.map.at(qb)) {
175  case Pauli::X:
176  u.add_op<Qubit>(OpType::H, {qb});
177  pauli0.string.map.at(qb) = Pauli::Z;
178  pauli1.string.map.at(qb) = Pauli::Z;
179  break;
180  case Pauli::Y:
181  u.add_op<Qubit>(OpType::V, {qb});
182  pauli0.string.map.at(qb) = Pauli::Z;
183  pauli1.string.map.at(qb) = Pauli::Z;
184  break;
185  default:
186  break;
187  }
188  }
189  switch (cx_config) {
190  case CXConfigType::Snake: {
191  reduce_shared_qs_by_CX_snake(u, match, pauli0, pauli1);
192  break;
193  }
194  case CXConfigType::Star: {
195  reduce_shared_qs_by_CX_star(u, match, pauli0, pauli1);
196  break;
197  }
198  case CXConfigType::Tree: {
199  reduce_shared_qs_by_CX_tree(u, match, pauli0, pauli1);
200  break;
201  }
202  case CXConfigType::MultiQGate: {
203  reduce_shared_qs_by_CX_multiqgate(u, match, pauli0, pauli1);
204  break;
205  }
206  default:
207  throw UnknownCXConfigType();
208  }
209  /*
210  * Step 2.ii: Convert mismatches to Z in pauli0 and X in pauli1
211  */
212  for (const Qubit &qb : mismatch) {
213  switch (pauli0.string.map.at(qb)) {
214  case Pauli::X: {
215  switch (pauli1.string.map.at(qb)) {
216  case Pauli::Y:
217  u.add_op<Qubit>(OpType::Sdg, {qb});
218  u.add_op<Qubit>(OpType::Vdg, {qb});
219  break;
220  case Pauli::Z:
221  u.add_op<Qubit>(OpType::H, {qb});
222  break;
223  default:
224  break; // Cannot hit this case
225  }
226  break;
227  }
228  case Pauli::Y: {
229  switch (pauli1.string.map.at(qb)) {
230  case Pauli::X:
231  u.add_op<Qubit>(OpType::V, {qb});
232  break;
233  case Pauli::Z:
234  u.add_op<Qubit>(OpType::V, {qb});
235  u.add_op<Qubit>(OpType::S, {qb});
236  break;
237  default:
238  break; // Cannot hit this case
239  }
240  break;
241  }
242  default: { // Necessarily Z
243  if (pauli1.string.map.at(qb) == Pauli::Y)
244  u.add_op<Qubit>(OpType::Sdg, {qb});
245  // No need to act if already X
246  }
247  }
248  pauli0.string.map.at(qb) = Pauli::Z;
249  pauli1.string.map.at(qb) = Pauli::X;
250  }
251 
252  /*
253  * Step 2.iii: Remove the final matching qubit against a mismatch if one
254  * exists, otherwise allow both gadgets to build it
255  */
256  if (!match.empty()) {
257  Qubit last_match = *match.begin();
258  match.erase(last_match);
259  if (!mismatch.empty()) {
260  Qubit mismatch_used =
261  *mismatch.rbegin(); // Prefer to use the one that may be left over
262  // after reducing pairs
263  u.add_op<Qubit>(OpType::S, {mismatch_used});
264  u.add_op<Qubit>(OpType::CX, {last_match, mismatch_used});
265  u.add_op<Qubit>(OpType::Sdg, {mismatch_used});
266  pauli0.string.map.erase(last_match);
267  pauli1.string.map.erase(last_match);
268  } else {
269  just0.insert(last_match);
270  just1.insert(last_match);
271  }
272  }
273 
274  /*
275  * Step 2.iv: Reduce pairs of mismatches to different qubits.
276  * Allow both gadgets to build a remaining qubit if it exists.
277  */
278  std::set<Qubit>::iterator mis_it = mismatch.begin();
279  while (mis_it != mismatch.end()) {
280  Qubit z_in_0 = *mis_it;
281  just0.insert(z_in_0);
282  mis_it++;
283  if (mis_it == mismatch.end()) {
284  just1.insert(z_in_0);
285  } else {
286  Qubit x_in_1 = *mis_it;
287  u.add_op<Qubit>(OpType::CX, {x_in_1, z_in_0});
288  pauli0.string.map.erase(x_in_1);
289  pauli1.string.map.erase(z_in_0);
290  just1.insert(x_in_1);
291  mis_it++;
292  }
293  }
294 
295  /*
296  * Step 3: Combine circuits to give final result
297  */
298  circ.append(u);
299  append_single_pauli_gadget(circ, pauli0, angle0);
300  append_single_pauli_gadget(circ, pauli1, angle1);
301  Circuit udg = u.dagger();
302  circ.append(udg);
303 }
