27 PauliGraph circuit_to_pauli_graph(const Circuit &circ) {
28  PauliGraph pg(circ.all_qubits(), circ.all_bits());
29  for (const Command &com : circ) {
30  const Op &op = *com.get_op_ptr();
31  unit_vector_t args = com.get_args();
32  OpDesc od = op.get_desc();
33  if (od.is_gate()) {
34  pg.apply_gate_at_end(static_cast<const Gate &>(op), args);
35  } else if (od.type() == OpType::PauliExpBox) {
36  const PauliExpBox &peb = static_cast<const PauliExpBox &>(op);
37  std::vector<Pauli> paulis = peb.get_paulis();
38  Expr phase = peb.get_phase();
39  if (args.size() != paulis.size())
40  throw std::logic_error("Incorrect Pauli tensor size for qubit count");
41  QubitPauliTensor qpt;
42  for (unsigned i = 0; i != args.size(); ++i) {
43  switch (paulis[i]) {
44  case Pauli::I:
45  break;
46  case Pauli::X:
47  qpt = qpt * pg.cliff_.get_xpauli(Qubit(args[i]));
48  break;
49  case Pauli::Y:
50  qpt = qpt * pg.cliff_.get_xpauli(Qubit(args[i]));
51  qpt = qpt * pg.cliff_.get_zpauli(Qubit(args[i]));
52  qpt = i_ * qpt;
53  break;
54  case Pauli::Z:
55  qpt = qpt * pg.cliff_.get_zpauli(Qubit(args[i]));
56  break;
57  }
58  }
59  pg.apply_pauli_gadget_at_end(qpt, phase);
60  } else
61  throw BadOpType(
62  "Can only make a PauliGraph from a circuit of basic gates "
63  "and Paulis",
64  od.type());
65  }
66  return pg;
67 }

69 Circuit pauli_graph_to_circuit_individually(
70  const PauliGraph &pg, CXConfigType cx_config) {
71  Circuit circ;
72  for (const Qubit &qb : pg.cliff_.get_qubits()) {
73  circ.add_qubit(qb);
74  }
75  for (const Bit &b : pg.bits_) {
76  circ.add_bit(b);
77  }
78  for (const PauliVert &vert : pg.vertices_in_order()) {
79  const QubitPauliTensor &pauli = pg.graph_[vert].tensor_;
80  const Expr &angle = pg.graph_[vert].angle_;
81  append_single_pauli_gadget(circ, pauli, angle, cx_config);
82  }
83  Circuit cliff_circuit = tableau_to_circuit(pg.cliff_);
84  circ.append(cliff_circuit);
85  for (auto it = pg.measures_.begin(); it != pg.measures_.end(); ++it) {
86  circ.add_measure(it->left, it->right);
87  }
88  return circ;
89 }

91 Circuit pauli_graph_to_circuit_pairwise(
92  const PauliGraph &pg, CXConfigType cx_config) {
93  Circuit circ;
94  for (const Qubit &qb : pg.cliff_.get_qubits()) {
95  circ.add_qubit(qb);
96  }
97  for (const Bit &b : pg.bits_) {
98  circ.add_bit(b);
99  }
100  std::vector<PauliVert> vertices = pg.vertices_in_order();
101  auto it = vertices.begin();
102  while (it != vertices.end()) {
103  PauliVert vert0 = *it;
104  const QubitPauliTensor &pauli0 = pg.graph_[vert0].tensor_;
105  const Expr &angle0 = pg.graph_[vert0].angle_;
106  ++it;
107  if (it == vertices.end()) {
108  append_single_pauli_gadget(circ, pauli0, angle0, cx_config);
109  } else {
110  PauliVert vert1 = *it;
111  const QubitPauliTensor &pauli1 = pg.graph_[vert1].tensor_;
112  const Expr &angle1 = pg.graph_[vert1].angle_;
113  ++it;
114  append_pauli_gadget_pair(circ, pauli0, angle0, pauli1, angle1, cx_config);
115  }
116  }
117  Circuit cliff_circuit = tableau_to_circuit(pg.cliff_);
118  circ.append(cliff_circuit);
119  for (auto it = pg.measures_.begin(); it != pg.measures_.end(); ++it) {
120  circ.add_measure(it->left, it->right);
121  }
122  return circ;
123 }

126 Circuit pauli_graph_to_circuit_sets(
127  const PauliGraph &pg, CXConfigType cx_config) {
128  Circuit circ;
129  const std::set<Qubit> qbs = pg.cliff_.get_qubits();
130  Circuit spare_circ;
131  for (const Qubit &qb : qbs) {
132  circ.add_qubit(qb);
133  spare_circ.add_qubit(qb);
134  }
135  for (const Bit &b : pg.bits_) {
136  circ.add_bit(b);
137  }
138  std::vector<PauliVert> vertices = pg.vertices_in_order();
139  auto it = vertices.begin();
140  while (it != vertices.end()) {
141  const PauliGadgetProperties &pgp = pg.graph_[*it];
142  QubitOperator gadget_map;
143  gadget_map[pgp.tensor_] = pgp.angle_;
144  ++it;
145  while (it != vertices.end()) {
146  const PauliGadgetProperties &pauli_gadget = pg.graph_[*it];
147  QubitOperator::iterator pgs_iter = gadget_map.find(pauli_gadget.tensor_);
148  if (pgs_iter != gadget_map.end()) {
149  insert_into_gadget_map(gadget_map, pauli_gadget);
150  } else {
151  bool commutes_with_all = true;
152  for (const std::pair<const QubitPauliTensor, Expr> &pv : gadget_map) {
153  if (!pauli_gadget.tensor_.commutes_with(pv.first)) {
154  commutes_with_all = false;
155  break;
156  }
157  }
158  if (!commutes_with_all) break;
159  insert_into_gadget_map(gadget_map, pauli_gadget);
160  }
161  ++it;
162  }
163  if (gadget_map.size() == 1) {
164  const std::pair<const QubitPauliTensor, Expr> &pgp0 = *gadget_map.begin();
165  append_single_pauli_gadget(circ, pgp0.first, pgp0.second, cx_config);
166  } else if (gadget_map.size() == 2) {
167  const std::pair<const QubitPauliTensor, Expr> &pgp0 = *gadget_map.begin();
168  const std::pair<const QubitPauliTensor, Expr> &pgp1 =
169  *(++gadget_map.begin());
170  append_pauli_gadget_pair(
171  circ, pgp0.first, pgp0.second, pgp1.first, pgp1.second, cx_config);
172  } else {
173  std::list<std::pair<QubitPauliTensor, Expr>> gadgets;
174  for (const std::pair<const QubitPauliTensor, Expr> &qps_pair :
175  gadget_map) {
176  gadgets.push_back(qps_pair);
177  }
178  Circuit cliff_circ = mutual_diagonalise(gadgets, qbs, cx_config);
179  circ.append(cliff_circ);
180  Circuit phase_poly_circ(spare_circ);
181  for (const std::pair<QubitPauliTensor, Expr> &pgp : gadgets) {
182  append_single_pauli_gadget(phase_poly_circ, pgp.first, pgp.second);
183  }
184  PhasePolyBox ppbox(phase_poly_circ);
185  Circuit after_synth_circ = *ppbox.to_circuit();
186  circ.append(after_synth_circ);
187  circ.append(cliff_circ.dagger());
188  }
189  }
190  Circuit cliff_circuit = tableau_to_circuit(pg.cliff_);
191  circ.append(cliff_circuit);
192  for (auto it1 = pg.measures_.begin(); it1 != pg.measures_.end(); ++it1) {
193  circ.add_measure(it1->left, it1->right);
194  }
195  return circ;
196 }
