23PauliGraph circuit_to_pauli_graph(const Circuit &circ) {
24 PauliGraph pg(circ.all_qubits(), circ.all_bits());
25 for (const Command &com : circ) {
26 const Op &op = *com.get_op_ptr();
27 unit_vector_t args = com.get_args();
28 OpDesc od = op.get_desc();
29 if (od.is_gate()) {
30 pg.apply_gate_at_end(static_cast<const Gate &>(op), args);
31 } else if (od.type() == OpType::PauliExpBox) {
32 const PauliExpBox &peb = static_cast<const PauliExpBox &>(op);
33 std::vector<Pauli> paulis = peb.get_paulis();
34 Expr phase = peb.get_phase();
35 if (args.size() != paulis.size())
36 throw std::logic_error("Incorrect Pauli tensor size for qubit count");
37 QubitPauliMap qpm;
38 for (unsigned i = 0; i != args.size(); ++i)
39 qpm.insert({Qubit(args[i]), paulis[i]});
40 SpPauliStabiliser qpt = pg.cliff_.get_row_product(SpPauliStabiliser(qpm));
41 pg.apply_pauli_gadget_at_end(qpt, phase);
42 } else
43 throw BadOpType(
44 "Can only make a PauliGraph from a circuit of basic gates "
45 "and Paulis",
46 od.type());
47 }
48 return pg;
49}

51Circuit pauli_graph_to_pauli_exp_box_circuit_individually(
52 const PauliGraph &pg, CXConfigType cx_config) {
53 Circuit circ;
54 for (const Qubit &qb : pg.cliff_.get_qubits()) {
55 circ.add_qubit(qb);
56 }
57 for (const Bit &b : pg.bits_) {
58 circ.add_bit(b);
59 }
60 for (const PauliVert &vert : pg.vertices_in_order()) {
61 const SpPauliStabiliser &pauli = pg.graph_[vert].tensor_;
62 const Expr &angle = pg.graph_[vert].angle_;
63 append_single_pauli_gadget_as_pauli_exp_box(
64 circ, SpSymPauliTensor(pauli) * SpSymPauliTensor({}, angle), cx_config);
65 }
66 Circuit cliff_circuit = unitary_rev_tableau_to_circuit(pg.cliff_);
67 circ.append(cliff_circuit);
68 for (auto it = pg.measures_.begin(); it != pg.measures_.end(); ++it) {
69 circ.add_measure(it->left, it->right);
70 }
71 return circ;
72}

74Circuit pauli_graph_to_pauli_exp_box_circuit_pairwise(
75 const PauliGraph &pg, CXConfigType cx_config) {
76 Circuit circ;
77 for (const Qubit &qb : pg.cliff_.get_qubits()) {
78 circ.add_qubit(qb);
79 }
80 for (const Bit &b : pg.bits_) {
81 circ.add_bit(b);
82 }
83 std::vector<PauliVert> vertices = pg.vertices_in_order();
84 auto it = vertices.begin();
85 while (it != vertices.end()) {
86 PauliVert vert0 = *it;
87 const SpPauliStabiliser &pauli0 = pg.graph_[vert0].tensor_;
88 const Expr &angle0 = pg.graph_[vert0].angle_;
89 ++it;
90 if (it == vertices.end()) {
91 append_single_pauli_gadget_as_pauli_exp_box(
92 circ, SpSymPauliTensor(pauli0) * SpSymPauliTensor({}, angle0),
93 cx_config);
94 } else {
95 PauliVert vert1 = *it;
96 const SpPauliStabiliser &pauli1 = pg.graph_[vert1].tensor_;
97 const Expr &angle1 = pg.graph_[vert1].angle_;
98 ++it;
99 append_pauli_gadget_pair_as_box(
100 circ, SpSymPauliTensor(pauli0) * SpSymPauliTensor({}, angle0),
101 SpSymPauliTensor(pauli1) * SpSymPauliTensor({}, angle1), cx_config);
102 }
103 }
104 Circuit cliff_circuit = unitary_rev_tableau_to_circuit(pg.cliff_);
105 circ.append(cliff_circuit);
106 for (auto it = pg.measures_.begin(); it != pg.measures_.end(); ++it) {
107 circ.add_measure(it->left, it->right);
108 }
109 return circ;
110}

113Circuit pauli_graph_to_pauli_exp_box_circuit_sets(
114 const PauliGraph &pg, CXConfigType cx_config) {
115 Circuit circ;
116 const std::set<Qubit> qbs = pg.cliff_.get_qubits();
117 Circuit spare_circ;
118 for (const Qubit &qb : qbs) {
119 circ.add_qubit(qb);
120 spare_circ.add_qubit(qb);
121 }
122 for (const Bit &b : pg.bits_) {
123 circ.add_bit(b);
124 }
125 std::vector<PauliVert> vertices = pg.vertices_in_order();
126 auto it = vertices.begin();
127 while (it != vertices.end()) {
128 const PauliGadgetProperties &pgp = pg.graph_[*it];
129 QubitOperator gadget_map;
130 insert_into_gadget_map(gadget_map, pgp);
131 ++it;
132 while (it != vertices.end()) {
133 const PauliGadgetProperties &pauli_gadget = pg.graph_[*it];
134 bool commutes_with_all = true;
135 for (const std::pair<const SpPauliString, Expr> &pv : gadget_map) {
136 if (!pauli_gadget.tensor_.commutes_with(pv.first)) {
137 commutes_with_all = false;
138 break;
139 }
140 }
141 if (!commutes_with_all) break;
142 insert_into_gadget_map(gadget_map, pauli_gadget);
143 ++it;
144 }
145 if (gadget_map.size() == 1) {
146 const std::pair<const SpPauliString, Expr> &pgp0 = *gadget_map.begin();
147 append_single_pauli_gadget_as_pauli_exp_box(
148 circ, SpSymPauliTensor(pgp0.first.string, pgp0.second), cx_config);
149 } else if (gadget_map.size() == 2) {
150 const std::pair<const SpPauliString, Expr> &pgp0 = *gadget_map.begin();
151 const std::pair<const SpPauliString, Expr> &pgp1 =
152 *(++gadget_map.begin());
153 append_pauli_gadget_pair_as_box(
154 circ, SpSymPauliTensor(pgp0.first.string, pgp0.second),
155 SpSymPauliTensor(pgp1.first.string, pgp1.second), cx_config);
156 } else {
157 std::list<SpSymPauliTensor> gadgets;
158 for (const std::pair<const SpPauliString, Expr> &qps_pair : gadget_map) {
159 gadgets.push_back(
160 SpSymPauliTensor(qps_pair.first.string, qps_pair.second));
161 }
162 append_commuting_pauli_gadget_set_as_box(circ, gadgets, cx_config);
163 }
164 }
165 Circuit cliff_circuit = unitary_rev_tableau_to_circuit(pg.cliff_);
166 circ.append(cliff_circuit);
167 for (auto it1 = pg.measures_.begin(); it1 != pg.measures_.end(); ++it1) {
168 circ.add_measure(it1->left, it1->right);
169 }
170 return circ;
171}
