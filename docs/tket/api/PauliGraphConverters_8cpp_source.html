25 PauliGraph circuit_to_pauli_graph(const Circuit &circ) {
26  PauliGraph pg(circ.all_qubits(), circ.all_bits());
27  for (const Command &com : circ) {
28  const Op &op = *com.get_op_ptr();
29  unit_vector_t args = com.get_args();
30  OpDesc od = op.get_desc();
31  if (od.is_gate()) {
32  pg.apply_gate_at_end(static_cast<const Gate &>(op), args);
33  } else if (od.type() == OpType::PauliExpBox) {
34  const PauliExpBox &peb = static_cast<const PauliExpBox &>(op);
35  std::vector<Pauli> paulis = peb.get_paulis();
36  Expr phase = peb.get_phase();
37  if (args.size() != paulis.size())
38  throw std::logic_error("Incorrect Pauli tensor size for qubit count");
39  QubitPauliMap qpm;
40  for (unsigned i = 0; i != args.size(); ++i)
41  qpm.insert({Qubit(args[i]), paulis[i]});
42  QubitPauliTensor qpt = pg.cliff_.get_row_product(QubitPauliTensor(qpm));
43  pg.apply_pauli_gadget_at_end(qpt, phase);
44  } else
45  throw BadOpType(
46  "Can only make a PauliGraph from a circuit of basic gates "
47  "and Paulis",
48  od.type());
49  }
50  return pg;
51 }

53 Circuit pauli_graph_to_pauli_exp_box_circuit_individually(
54  const PauliGraph &pg, CXConfigType cx_config) {
55  Circuit circ;
56  for (const Qubit &qb : pg.cliff_.get_qubits()) {
57  circ.add_qubit(qb);
58  }
59  for (const Bit &b : pg.bits_) {
60  circ.add_bit(b);
61  }
62  for (const PauliVert &vert : pg.vertices_in_order()) {
63  const QubitPauliTensor &pauli = pg.graph_[vert].tensor_;
64  const Expr &angle = pg.graph_[vert].angle_;
65  append_single_pauli_gadget_as_pauli_exp_box(circ, pauli, angle, cx_config);
66  }
67  Circuit cliff_circuit = unitary_rev_tableau_to_circuit(pg.cliff_);
68  circ.append(cliff_circuit);
69  for (auto it = pg.measures_.begin(); it != pg.measures_.end(); ++it) {
70  circ.add_measure(it->left, it->right);
71  }
72  return circ;
73 }

115 Circuit pauli_graph_to_pauli_exp_box_circuit_sets(
116  const PauliGraph &pg, CXConfigType cx_config) {
117  Circuit circ;
118  const std::set<Qubit> qbs = pg.cliff_.get_qubits();
119  Circuit spare_circ;
120  for (const Qubit &qb : qbs) {
121  circ.add_qubit(qb);
122  spare_circ.add_qubit(qb);
123  }
124  for (const Bit &b : pg.bits_) {
125  circ.add_bit(b);
126  }
127  std::vector<PauliVert> vertices = pg.vertices_in_order();
128  auto it = vertices.begin();
129  while (it != vertices.end()) {
130  const PauliGadgetProperties &pgp = pg.graph_[*it];
131  QubitOperator gadget_map;
132  gadget_map[pgp.tensor_] = pgp.angle_;
133  ++it;
134  while (it != vertices.end()) {
135  const PauliGadgetProperties &pauli_gadget = pg.graph_[*it];
136  QubitOperator::iterator pgs_iter = gadget_map.find(pauli_gadget.tensor_);
137  if (pgs_iter != gadget_map.end()) {
138  insert_into_gadget_map(gadget_map, pauli_gadget);
139  } else {
140  bool commutes_with_all = true;
141  for (const std::pair<const QubitPauliTensor, Expr> &pv : gadget_map) {
142  if (!pauli_gadget.tensor_.commutes_with(pv.first)) {
143  commutes_with_all = false;
144  break;
145  }
146  }
147  if (!commutes_with_all) break;
148  insert_into_gadget_map(gadget_map, pauli_gadget);
149  }
150  ++it;
151  }
152  if (gadget_map.size() == 1) {
153  const std::pair<const QubitPauliTensor, Expr> &pgp0 = *gadget_map.begin();
154  append_single_pauli_gadget_as_pauli_exp_box(
155  circ, pgp0.first, pgp0.second, cx_config);
156  } else if (gadget_map.size() == 2) {
157  const std::pair<const QubitPauliTensor, Expr> &pgp0 = *gadget_map.begin();
158  const std::pair<const QubitPauliTensor, Expr> &pgp1 =
159  *(++gadget_map.begin());
160  append_pauli_gadget_pair_as_box(
161  circ, pgp0.first, pgp0.second, pgp1.first, pgp1.second, cx_config);
162  } else {
163  std::list<std::pair<QubitPauliTensor, Expr>> gadgets;
164  for (const std::pair<const QubitPauliTensor, Expr> &qps_pair :
165  gadget_map) {
166  gadgets.push_back(qps_pair);
167  }
168  append_commuting_pauli_gadget_set_as_box(circ, gadgets, cx_config);
169  }
170  }
171  Circuit cliff_circuit = unitary_rev_tableau_to_circuit(pg.cliff_);
172  circ.append(cliff_circuit);
173  for (auto it1 = pg.measures_.begin(); it1 != pg.measures_.end(); ++it1) {
174  circ.add_measure(it1->left, it1->right);
175  }
176  return circ;
177 }
