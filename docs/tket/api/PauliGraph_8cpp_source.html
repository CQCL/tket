80 void PauliGraph::apply_gate_at_end(
81  const Gate &gate, const unit_vector_t &args) {
82  for (const UnitID &arg : args) {
83  if (arg.type() == UnitType::Qubit) {
84  if (measures_.left.find(Qubit(arg)) != measures_.left.end()) {
85  throw MidCircuitMeasurementNotAllowed(
86  "PauliGraph does not support mid-circuit measurements "
87  "- cannot add gate after measure on qubit " +
88  arg.repr());
89  }
90  } else if (measures_.right.find(Bit(arg)) != measures_.right.end()) {
91  throw MidCircuitMeasurementNotAllowed(
92  "PauliGraph does not support mid-circuit measurements - "
93  "cannot add gate after measure to bit " +
94  arg.repr());
95  }
96  }
97  OpType type = gate.get_type();
98  if (type == OpType::Measure) {
99  measures_.insert({Qubit(args.at(0)), Bit(args.at(1))});
100  return;
101  }
102  qubit_vector_t qbs = {args.begin(), args.end()};
103  switch (type) {
104  case OpType::Z:
105  case OpType::X:
106  case OpType::Y:
107  case OpType::S:
108  case OpType::Sdg:
109  case OpType::V:
110  case OpType::Vdg:
111  case OpType::H:
112  case OpType::CX:
113  case OpType::CY:
114  case OpType::CZ:
115  case OpType::SWAP:
116  case OpType::noop:
117  case OpType::Phase: {
118  cliff_.apply_gate_at_end(type, qbs);
119  break;
120  }
121  case OpType::Rz: {
122  QubitPauliTensor pauli = cliff_.get_zrow(qbs.at(0));
123  Expr angle = gate.get_params().at(0);
124  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
125  if (cliff_angle) {
126  for (unsigned i = 0; i < cliff_angle.value(); i++) {
127  cliff_.apply_gate_at_end(OpType::S, qbs);
128  }
129  } else
130  apply_pauli_gadget_at_end(pauli, angle);
131  break;
132  }
133  case OpType::Rx: {
134  QubitPauliTensor pauli = cliff_.get_xrow(qbs.at(0));
135  Expr angle = gate.get_params().at(0);
136  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
137  if (cliff_angle) {
138  for (unsigned i = 0; i < cliff_angle.value(); i++) {
139  cliff_.apply_gate_at_end(OpType::V, qbs);
140  }
141  } else
142  apply_pauli_gadget_at_end(pauli, angle);
143  break;
144  }
145  case OpType::Ry: {
146  Expr angle = gate.get_params().at(0);
147  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
148  if (cliff_angle) {
149  if (cliff_angle.value() != 0) {
150  cliff_.apply_gate_at_end(OpType::V, qbs);
151  for (unsigned i = 0; i < cliff_angle.value(); i++) {
152  cliff_.apply_gate_at_end(OpType::S, qbs);
153  }
154  cliff_.apply_gate_at_end(OpType::Vdg, qbs);
155  }
156  } else {
157  QubitPauliTensor ypauli =
158  cliff_.get_row_product(QubitPauliTensor(qbs.at(0), Pauli::Y));
159  apply_pauli_gadget_at_end(ypauli, angle);
160  }
161  break;
162  }
163  case OpType::PhasedX: {
164  Expr alpha = gate.get_params().at(0);
165  Expr beta = gate.get_params().at(1);
166  QubitPauliTensor zpauli = cliff_.get_zrow(qbs.at(0));
167  QubitPauliTensor xpauli = cliff_.get_xrow(qbs.at(0));
168  std::optional<unsigned> cliff_alpha = equiv_Clifford(alpha);
169  std::optional<unsigned> cliff_beta = equiv_Clifford(beta);
170  // Rz(-b)
171  if (cliff_beta) {
172  for (unsigned i = 0; i < cliff_beta.value(); i++) {
173  cliff_.apply_gate_at_end(OpType::Sdg, qbs);
174  }
175  } else {
176  apply_pauli_gadget_at_end(zpauli, -beta);
177  }
178  // Rx(a)
179  if (cliff_alpha) {
180  for (unsigned i = 0; i < cliff_alpha.value(); i++) {
181  cliff_.apply_gate_at_end(OpType::V, qbs);
182  }
183  } else {
184  apply_pauli_gadget_at_end(xpauli, alpha);
185  }
186  // Rz(b)
187  if (cliff_beta) {
188  for (unsigned i = 0; i < cliff_beta.value(); i++) {
189  cliff_.apply_gate_at_end(OpType::S, qbs);
190  }
191  } else {
192  apply_pauli_gadget_at_end(zpauli, beta);
193  }
194  break;
195  }
196  case OpType::T: {
197  QubitPauliTensor pauli = cliff_.get_zrow(qbs.at(0));
198  apply_pauli_gadget_at_end(pauli, 0.25);
199  break;
200  }
201  case OpType::Tdg: {
202  QubitPauliTensor pauli = cliff_.get_zrow(qbs.at(0));
203  apply_pauli_gadget_at_end(pauli, -0.25);
204  break;
205  }
206  case OpType::ZZMax: {
207  cliff_.apply_gate_at_end(OpType::S, {qbs.at(0)});
208  cliff_.apply_gate_at_end(OpType::Z, {qbs.at(1)});
209  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
210  cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
211  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
212  cliff_.apply_gate_at_end(OpType::CX, qbs);
213  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
214  cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
215  break;
216  }
217  case OpType::PhaseGadget:
218  case OpType::ZZPhase: {
219  Expr angle = gate.get_params().at(0);
220  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
221  if (cliff_angle) {
222  if (cliff_angle.value() != 0) {
223  for (unsigned i = 1; i < qbs.size(); i++) {
224  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(i - 1), qbs.at(i)});
225  }
226  for (unsigned i = 0; i < cliff_angle.value(); i++) {
227  cliff_.apply_gate_at_end(OpType::S, {qbs.back()});
228  }
229  for (unsigned i = qbs.size() - 1; i > 0; i--) {
230  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(i - 1), qbs.at(i)});
231  }
232  }
233  } else {
234  QubitPauliTensor pauli =
235  cliff_.get_row_product(QubitPauliTensor(QubitPauliString(
236  std::list<Qubit>{qbs.begin(), qbs.end()},
237  std::list<Pauli>{qbs.size(), Pauli::Z})));
238  apply_pauli_gadget_at_end(pauli, angle);
239  }
240  break;
241  }
242  case OpType::XXPhase: {
243  Expr angle = gate.get_params().at(0);
244  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
245  if (cliff_angle) {
246  if (cliff_angle.value() != 0) {
247  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(1), qbs.at(0)});
248  for (unsigned i = 0; i < cliff_angle.value(); i++) {
249  cliff_.apply_gate_at_end(OpType::V, {qbs.back()});
250  }
251  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(1), qbs.at(0)});
252  }
253  } else {
254  QubitPauliTensor pauli = cliff_.get_row_product(QubitPauliTensor(
255  QubitPauliString({qbs.at(0), qbs.at(1)}, {Pauli::X, Pauli::X})));
256  apply_pauli_gadget_at_end(pauli, angle);
257  }
258  break;
259  }
260  case OpType::YYPhase: {
261  Expr angle = gate.get_params().at(0);
262  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
263  if (cliff_angle) {
264  if (cliff_angle.value() != 0) {
265  const Qubit &arg0 = qbs.at(0);
266  const Qubit &arg1 = qbs.at(1);
267  cliff_.apply_gate_at_end(OpType::S, {arg0});
268  cliff_.apply_gate_at_end(OpType::S, {arg1});
269  cliff_.apply_gate_at_end(OpType::CX, {arg1, arg0});
270  for (unsigned i = 0; i < cliff_angle.value(); i++) {
271  cliff_.apply_gate_at_end(OpType::V, {arg1});
272  }
273  cliff_.apply_gate_at_end(OpType::CX, {arg1, arg0});
274  cliff_.apply_gate_at_end(OpType::Sdg, {arg0});
275  cliff_.apply_gate_at_end(OpType::Sdg, {arg1});
276  }
277  } else {
278  QubitPauliTensor pauli = cliff_.get_row_product(QubitPauliTensor(
279  QubitPauliString({qbs.at(0), qbs.at(1)}, {Pauli::Y, Pauli::Y})));
280  apply_pauli_gadget_at_end(pauli, angle);
281  }
282  break;
283  }
284  default: {
285  throw BadOpType("Cannot add gate to PauliGraph", type);
286  }
287  }

290 void PauliGraph::apply_pauli_gadget_at_end(
291  const QubitPauliTensor &pauli, const Expr &angle) {
292  PauliVertSet to_search = end_line_;
293  PauliVertSet commuted;
294  PauliVert new_vert = boost::add_vertex(graph_);
295  graph_[new_vert] = {pauli, angle};
296  while (!to_search.empty()) {
297  // Get next candidate parent
298  PauliVert to_compare = *to_search.begin();
299  to_search.erase(to_search.begin());
300 
301  // Check that we have already commuted past all of its children
302  bool ready = true;
303  for (const PauliVert &child : get_successors(to_compare)) {
304  if (commuted.get<TagKey>().find(child) == commuted.get<TagKey>().end()) {
305  ready = false;
306  break;
307  }
308  }
309  if (!ready) continue;
310 
311  // Check if we can commute past it
312  QubitPauliTensor compare_pauli = graph_[to_compare].tensor_;
313  if (pauli.commutes_with(compare_pauli)) {
314  if (pauli.string == compare_pauli.string) {
315  // Identical strings - we can merge vertices
316  if (pauli.coeff == compare_pauli.coeff) {
317  graph_[to_compare].angle_ += angle;
318  } else {
319  graph_[to_compare].angle_ -= angle;
320  }
321  boost::clear_vertex(new_vert, graph_);
322  boost::remove_vertex(new_vert, graph_);
323 
324  std::optional<unsigned> cl_ang =
325  equiv_Clifford(graph_[to_compare].angle_);
326  if (cl_ang) {
327  cliff_.apply_pauli_at_front(graph_[to_compare].tensor_, *cl_ang);
328  start_line_.erase(to_compare);
329  for (const PauliVert &v : get_predecessors(to_compare)) {
330  if (boost::out_degree(v, graph_) == 1) {
331  end_line_.insert(v);
332  }
333  }
334  end_line_.erase(to_compare);
335  boost::clear_vertex(to_compare, graph_);
336  boost::remove_vertex(to_compare, graph_);
337  }
338  return;
339  } else {
340  // Commute and continue searching
341  PauliVertSet preds = get_predecessors(to_compare);
342  to_search.insert(preds.begin(), preds.end());
343  commuted.insert(to_compare);
344  }
345  } else {
346  // Does not commute - add dependency edge
347  boost::add_edge(to_compare, new_vert, graph_);
348  end_line_.erase(to_compare);
349  }
350  }
351  end_line_.insert(new_vert);
352  if (get_predecessors(new_vert).empty()) start_line_.insert(new_vert);
353 }

360 void PauliGraph::to_graphviz(std::ostream &out) const {
361  out << "digraph G {\n";
362 
363  std::map<PauliVert, unsigned> index_map;
364  unsigned i = 0;

366  index_map.insert({v, i});
367  out << i << " [label = \"" << graph_[v].tensor_.to_str() << ", "

369  ++i;
370  }
371 
372  BGL_FORALL_EDGES(e, graph_, PauliDAG) {
373  PauliVert v_so = source(e);
374  PauliVert v_ta = target(e);
375  out << index_map.at(v_so) << " -> " << index_map.at(v_ta) << ";\n";
376  }

378  out << "}";

391 void PauliGraph::sanity_check() const {
392  for (const PauliVert &vert : vertices_in_order()) {
393  PauliVertSet succs;
394  boost::graph_traits<PauliDAG>::adjacency_iterator ai, a_end;
395  boost::tie(ai, a_end) = boost::adjacent_vertices(vert, graph_);
396  for (; ai != a_end; ai++) {
397  TKET_ASSERT(!succs.contains(*ai));
398  succs.insert(*ai);
399  }
400 
401  PauliVertSet preds;
402  PauliDAG::inv_adjacency_iterator iai, ia_end;
403  boost::tie(iai, ia_end) = boost::inv_adjacent_vertices(vert, graph_);
404  for (; iai != ia_end; iai++) {
405  TKET_ASSERT(!preds.contains(*iai));
406  preds.insert(*iai);
407  }
408  }
