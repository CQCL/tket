80 void PauliGraph::apply_gate_at_end(
81  const Gate &gate, const unit_vector_t &args) {
82  for (const UnitID &arg : args) {
83  if (arg.type() == UnitType::Qubit) {
84  if (measures_.left.find(Qubit(arg)) != measures_.left.end()) {
85  throw MidCircuitMeasurementNotAllowed(
86  "PauliGraph does not support mid-circuit measurements "
87  "- cannot add gate after measure on qubit " +
88  arg.repr());
89  }
90  } else if (measures_.right.find(Bit(arg)) != measures_.right.end()) {
91  throw MidCircuitMeasurementNotAllowed(
92  "PauliGraph does not support mid-circuit measurements - "
93  "cannot add gate after measure to bit " +
94  arg.repr());
95  }
96  }
97  OpType type = gate.get_type();
98  if (type == OpType::Measure) {
99  measures_.insert({Qubit(args.at(0)), Bit(args.at(1))});
100  return;
101  }
102  qubit_vector_t qbs = {args.begin(), args.end()};
103  switch (type) {
104  case OpType::Z:
105  case OpType::X:
106  case OpType::Y:
107  case OpType::S:
108  case OpType::Sdg:
109  case OpType::V:
110  case OpType::Vdg:
111  case OpType::H:
112  case OpType::CX:
113  case OpType::CY:
114  case OpType::CZ:
115  case OpType::SWAP:
116  case OpType::noop:
117  case OpType::Phase: {
118  cliff_.apply_gate_at_end(type, qbs);
119  break;
120  }
121  case OpType::Rz: {
122  QubitPauliTensor pauli = cliff_.get_zpauli(qbs.at(0));
123  Expr angle = gate.get_params().at(0);
124  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
125  if (cliff_angle) {
126  for (unsigned i = 0; i < cliff_angle.value(); i++) {
127  cliff_.apply_gate_at_end(OpType::S, qbs);
128  }
129  } else
130  apply_pauli_gadget_at_end(pauli, angle);
131  break;
132  }
133  case OpType::Rx: {
134  QubitPauliTensor pauli = cliff_.get_xpauli(qbs.at(0));
135  Expr angle = gate.get_params().at(0);
136  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
137  if (cliff_angle) {
138  for (unsigned i = 0; i < cliff_angle.value(); i++) {
139  cliff_.apply_gate_at_end(OpType::V, qbs);
140  }
141  } else
142  apply_pauli_gadget_at_end(pauli, angle);
143  break;
144  }
145  case OpType::Ry: {
146  Expr angle = gate.get_params().at(0);
147  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
148  if (cliff_angle) {
149  if (cliff_angle.value() != 0) {
150  cliff_.apply_gate_at_end(OpType::V, qbs);
151  for (unsigned i = 0; i < cliff_angle.value(); i++) {
152  cliff_.apply_gate_at_end(OpType::S, qbs);
153  }
154  cliff_.apply_gate_at_end(OpType::Vdg, qbs);
155  }
156  } else {
157  QubitPauliTensor zpauli = cliff_.get_zpauli(qbs.at(0));
158  QubitPauliTensor xpauli = cliff_.get_xpauli(qbs.at(0));
159  QubitPauliTensor ypauli = i_ * xpauli * zpauli;
160  apply_pauli_gadget_at_end(ypauli, angle);
161  }
162  break;
163  }
164  case OpType::PhasedX: {
165  Expr alpha = gate.get_params().at(0);
166  Expr beta = gate.get_params().at(1);
167  QubitPauliTensor zpauli = cliff_.get_zpauli(qbs.at(0));
168  QubitPauliTensor xpauli = cliff_.get_xpauli(qbs.at(0));
169  std::optional<unsigned> cliff_alpha = equiv_Clifford(alpha);
170  std::optional<unsigned> cliff_beta = equiv_Clifford(beta);
171  // Rz(-b)
172  if (cliff_beta) {
173  for (unsigned i = 0; i < cliff_beta.value(); i++) {
174  cliff_.apply_gate_at_end(OpType::Sdg, qbs);
175  }
176  } else {
177  apply_pauli_gadget_at_end(zpauli, -beta);
178  }
179  // Rx(a)
180  if (cliff_alpha) {
181  for (unsigned i = 0; i < cliff_alpha.value(); i++) {
182  cliff_.apply_gate_at_end(OpType::V, qbs);
183  }
184  } else {
185  apply_pauli_gadget_at_end(xpauli, alpha);
186  }
187  // Rz(b)
188  if (cliff_beta) {
189  for (unsigned i = 0; i < cliff_beta.value(); i++) {
190  cliff_.apply_gate_at_end(OpType::S, qbs);
191  }
192  } else {
193  apply_pauli_gadget_at_end(zpauli, beta);
194  }
195  break;
196  }
197  case OpType::T: {
198  QubitPauliTensor pauli = cliff_.get_zpauli(qbs.at(0));
199  apply_pauli_gadget_at_end(pauli, 0.25);
200  break;
201  }
202  case OpType::Tdg: {
203  QubitPauliTensor pauli = cliff_.get_zpauli(qbs.at(0));
204  apply_pauli_gadget_at_end(pauli, -0.25);
205  break;
206  }
207  case OpType::ZZMax: {
208  cliff_.apply_gate_at_end(OpType::S, {qbs.at(0)});
209  cliff_.apply_gate_at_end(OpType::Z, {qbs.at(1)});
210  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
211  cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
212  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
213  cliff_.apply_gate_at_end(OpType::CX, qbs);
214  cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
215  cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
216  break;
217  }
218  case OpType::PhaseGadget:
219  case OpType::ZZPhase: {
220  Expr angle = gate.get_params().at(0);
221  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
222  if (cliff_angle) {
223  if (cliff_angle.value() != 0) {
224  for (unsigned i = 1; i < qbs.size(); i++) {
225  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(i - 1), qbs.at(i)});
226  }
227  for (unsigned i = 0; i < cliff_angle.value(); i++) {
228  cliff_.apply_gate_at_end(OpType::S, {qbs.back()});
229  }
230  for (unsigned i = qbs.size() - 1; i > 0; i--) {
231  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(i - 1), qbs.at(i)});
232  }
233  }
234  } else {
235  QubitPauliTensor pauli;
236  for (const Qubit &qb : qbs) {
237  pauli = pauli * cliff_.get_zpauli(qb);
238  }
239  apply_pauli_gadget_at_end(pauli, angle);
240  }
241  break;
242  }
243  case OpType::XXPhase: {
244  Expr angle = gate.get_params().at(0);
245  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
246  if (cliff_angle) {
247  if (cliff_angle.value() != 0) {
248  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(1), qbs.at(0)});
249  for (unsigned i = 0; i < cliff_angle.value(); i++) {
250  cliff_.apply_gate_at_end(OpType::V, {qbs.back()});
251  }
252  cliff_.apply_gate_at_end(OpType::CX, {qbs.at(1), qbs.at(0)});
253  }
254  } else {
255  QubitPauliTensor pauli =
256  cliff_.get_xpauli(qbs.at(0)) * cliff_.get_xpauli(qbs.at(1));
257  apply_pauli_gadget_at_end(pauli, angle);
258  }
259  break;
260  }
261  case OpType::YYPhase: {
262  Expr angle = gate.get_params().at(0);
263  std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
264  if (cliff_angle) {
265  if (cliff_angle.value() != 0) {
266  const Qubit &arg0 = qbs.at(0);
267  const Qubit &arg1 = qbs.at(1);
268  cliff_.apply_gate_at_end(OpType::S, {arg0});
269  cliff_.apply_gate_at_end(OpType::S, {arg1});
270  cliff_.apply_gate_at_end(OpType::CX, {arg1, arg0});
271  for (unsigned i = 0; i < cliff_angle.value(); i++) {
272  cliff_.apply_gate_at_end(OpType::V, {arg1});
273  }
274  cliff_.apply_gate_at_end(OpType::CX, {arg1, arg0});
275  cliff_.apply_gate_at_end(OpType::Sdg, {arg0});
276  cliff_.apply_gate_at_end(OpType::Sdg, {arg1});
277  }
278  } else {
279  QubitPauliTensor pauli =
280  -1. * cliff_.get_xpauli(qbs.at(0)) * cliff_.get_zpauli(qbs.at(0)) *
281  cliff_.get_xpauli(qbs.at(1)) * cliff_.get_zpauli(qbs.at(1));
282  apply_pauli_gadget_at_end(pauli, angle);
283  }
284  break;
285  }
286  default: {
287  throw BadOpType("Cannot add gate to PauliGraph", type);
288  }
289  }

292 void PauliGraph::apply_pauli_gadget_at_end(
293  const QubitPauliTensor &pauli, const Expr &angle) {
294  PauliVertSet to_search = end_line_;
295  PauliVertSet commuted;
296  PauliVert new_vert = boost::add_vertex(graph_);
297  graph_[new_vert] = {pauli, angle};
298  while (!to_search.empty()) {
299  // Get next candidate parent
300  PauliVert to_compare = *to_search.begin();
301  to_search.erase(to_search.begin());
302 
303  // Check that we have already commuted past all of its children
304  bool ready = true;
305  for (const PauliVert &child : get_successors(to_compare)) {
306  if (commuted.get<TagKey>().find(child) == commuted.get<TagKey>().end()) {
307  ready = false;
308  break;
309  }
310  }
311  if (!ready) continue;
312 
313  // Check if we can commute past it
314  QubitPauliTensor compare_pauli = graph_[to_compare].tensor_;
315  if (pauli.commutes_with(compare_pauli)) {
316  if (pauli.string == compare_pauli.string) {
317  // Identical strings - we can merge vertices
318  if (pauli.coeff == compare_pauli.coeff) {
319  graph_[to_compare].angle_ += angle;
320  } else {
321  graph_[to_compare].angle_ -= angle;
322  }
323  boost::clear_vertex(new_vert, graph_);
324  boost::remove_vertex(new_vert, graph_);
325 
326  std::optional<unsigned> cl_ang =
327  equiv_Clifford(graph_[to_compare].angle_);
328  if (cl_ang) {
329  cliff_.apply_pauli_at_front(graph_[to_compare].tensor_, *cl_ang);
330  start_line_.erase(to_compare);
331  for (const PauliVert &v : get_predecessors(to_compare)) {
332  if (boost::out_degree(v, graph_) == 1) {
333  end_line_.insert(v);
334  }
335  }
336  end_line_.erase(to_compare);
337  boost::clear_vertex(to_compare, graph_);
338  boost::remove_vertex(to_compare, graph_);
339  }
340  return;
341  } else {
342  // Commute and continue searching
343  PauliVertSet preds = get_predecessors(to_compare);
344  to_search.insert(preds.begin(), preds.end());
345  commuted.insert(to_compare);
346  }
347  } else {
348  // Does not commute - add dependency edge
349  boost::add_edge(to_compare, new_vert, graph_);
350  end_line_.erase(to_compare);
351  }
352  }
353  end_line_.insert(new_vert);
354  if (get_predecessors(new_vert).empty()) start_line_.insert(new_vert);
355 }

362 void PauliGraph::to_graphviz(std::ostream &out) const {
363  out << "digraph G {\n";
364 
365  std::map<PauliVert, unsigned> index_map;
366  unsigned i = 0;
367  BGL_FORALL_VERTICES(v, graph_, PauliDAG) {
368  index_map.insert({v, i});
369  out << i << " [label = \"" << graph_[v].tensor_.to_str() << ", "
370  << graph_[v].angle_ << "\"];\n";
371  ++i;
372  }
373 
374  BGL_FORALL_EDGES(e, graph_, PauliDAG) {
375  PauliVert v_so = source(e);
376  PauliVert v_ta = target(e);
377  out << index_map.at(v_so) << " -> " << index_map.at(v_ta) << ";\n";
378  }
379 
380  out << "}";

383 std::vector<PauliVert> PauliGraph::vertices_in_order() const {
384  PauliVIndex index = boost::get(boost::vertex_index, graph_);

386  BGL_FORALL_VERTICES(v, graph_, PauliDAG) { boost::put(index, v, i++); }

389  std::reverse(vertices.begin(), vertices.end());

393 void PauliGraph::sanity_check() const {
394  for (const PauliVert &vert : vertices_in_order()) {
395  PauliVertSet succs;
396  boost::graph_traits<PauliDAG>::adjacency_iterator ai, a_end;
397  boost::tie(ai, a_end) = boost::adjacent_vertices(vert, graph_);
398  for (; ai != a_end; ai++) {
399  TKET_ASSERT(!succs.contains(*ai));
400  succs.insert(*ai);
401  }
402 
403  PauliVertSet preds;
404  PauliDAG::inv_adjacency_iterator iai, ia_end;
405  boost::tie(iai, ia_end) = boost::inv_adjacent_vertices(vert, graph_);
406  for (; iai != ia_end; iai++) {
407  TKET_ASSERT(!preds.contains(*iai));
408  preds.insert(*iai);
409  }
410  }
