79void PauliGraph::apply_gate_at_end(
80 const Gate &gate, const unit_vector_t &args) {
81 for (const UnitID &arg : args) {
82 if (arg.type() == UnitType::Qubit) {
83 if (measures_.left.find(Qubit(arg)) != measures_.left.end()) {
84 throw MidCircuitMeasurementNotAllowed(
85 "PauliGraph does not support mid-circuit measurements "
86 "- cannot add gate after measure on qubit " +
87 arg.repr());
88 }
89 } else if (measures_.right.find(Bit(arg)) != measures_.right.end()) {
90 throw MidCircuitMeasurementNotAllowed(
91 "PauliGraph does not support mid-circuit measurements - "
92 "cannot add gate after measure to bit " +
93 arg.repr());
94 }
95 }
96 OpType type = gate.get_type();
97 if (type == OpType::Measure) {
98 measures_.insert({Qubit(args.at(0)), Bit(args.at(1))});
99 return;
100 }
101 qubit_vector_t qbs = {args.begin(), args.end()};
102 switch (type) {
103 case OpType::Z:
104 case OpType::X:
105 case OpType::Y:
106 case OpType::S:
107 case OpType::Sdg:
108 case OpType::V:
109 case OpType::Vdg:
110 case OpType::H:
111 case OpType::SX:
112 case OpType::SXdg:
113 case OpType::CX:
114 case OpType::CY:
115 case OpType::CZ:
116 case OpType::SWAP:
117 case OpType::noop:
118 case OpType::Phase: {
119 cliff_.apply_gate_at_end(type, qbs);
120 break;
121 }
122 case OpType::Rz: {
123 SpPauliStabiliser pauli = cliff_.get_zrow(qbs.at(0));
124 Expr angle = gate.get_params().at(0);
125 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
126 if (cliff_angle) {
127 for (unsigned i = 0; i < cliff_angle.value(); i++) {
128 cliff_.apply_gate_at_end(OpType::S, qbs);
129 }
130 } else
131 apply_pauli_gadget_at_end(pauli, angle);
132 break;
133 }
134 case OpType::Rx: {
135 SpPauliStabiliser pauli = cliff_.get_xrow(qbs.at(0));
136 Expr angle = gate.get_params().at(0);
137 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
138 if (cliff_angle) {
139 for (unsigned i = 0; i < cliff_angle.value(); i++) {
140 cliff_.apply_gate_at_end(OpType::V, qbs);
141 }
142 } else
143 apply_pauli_gadget_at_end(pauli, angle);
144 break;
145 }
146 case OpType::Ry: {
147 Expr angle = gate.get_params().at(0);
148 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
149 if (cliff_angle) {
150 if (cliff_angle.value() != 0) {
151 cliff_.apply_gate_at_end(OpType::V, qbs);
152 for (unsigned i = 0; i < cliff_angle.value(); i++) {
153 cliff_.apply_gate_at_end(OpType::S, qbs);
154 }
155 cliff_.apply_gate_at_end(OpType::Vdg, qbs);
156 }
157 } else {
158 SpPauliStabiliser ypauli =
159 cliff_.get_row_product(SpPauliStabiliser(qbs.at(0), Pauli::Y));
160 apply_pauli_gadget_at_end(ypauli, angle);
161 }
162 break;
163 }
164 case OpType::PhasedX: {
165 Expr alpha = gate.get_params().at(0);
166 Expr beta = gate.get_params().at(1);
167 std::optional<unsigned> cliff_alpha = equiv_Clifford(alpha);
168 std::optional<unsigned> cliff_beta = equiv_Clifford(beta);
169 // Rz(-b)
170 if (cliff_beta) {
171 for (unsigned i = 0; i < cliff_beta.value(); i++) {
172 cliff_.apply_gate_at_end(OpType::Sdg, qbs);
173 }
174 } else {
175 SpPauliStabiliser zpauli = cliff_.get_zrow(qbs.at(0));
176 apply_pauli_gadget_at_end(zpauli, -beta);
177 }
178 // Rx(a)
179 if (cliff_alpha) {
180 for (unsigned i = 0; i < cliff_alpha.value(); i++) {
181 cliff_.apply_gate_at_end(OpType::V, qbs);
182 }
183 } else {
184 SpPauliStabiliser xpauli = cliff_.get_xrow(qbs.at(0));
185 apply_pauli_gadget_at_end(xpauli, alpha);
186 }
187 // Rz(b)
188 if (cliff_beta) {
189 for (unsigned i = 0; i < cliff_beta.value(); i++) {
190 cliff_.apply_gate_at_end(OpType::S, qbs);
191 }
192 } else {
193 SpPauliStabiliser zpauli = cliff_.get_zrow(qbs.at(0));
194 apply_pauli_gadget_at_end(zpauli, beta);
195 }
196 break;
197 }
198 case OpType::T: {
199 SpPauliStabiliser pauli = cliff_.get_zrow(qbs.at(0));
200 apply_pauli_gadget_at_end(pauli, 0.25);
201 break;
202 }
203 case OpType::Tdg: {
204 SpPauliStabiliser pauli = cliff_.get_zrow(qbs.at(0));
205 apply_pauli_gadget_at_end(pauli, -0.25);
206 break;
207 }
208 case OpType::ZZMax: {
209 cliff_.apply_gate_at_end(OpType::S, {qbs.at(0)});
210 cliff_.apply_gate_at_end(OpType::Z, {qbs.at(1)});
211 cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
212 cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
213 cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
214 cliff_.apply_gate_at_end(OpType::CX, qbs);
215 cliff_.apply_gate_at_end(OpType::S, {qbs.at(1)});
216 cliff_.apply_gate_at_end(OpType::V, {qbs.at(1)});
217 break;
218 }
219 case OpType::PhaseGadget:
220 case OpType::ZZPhase: {
221 Expr angle = gate.get_params().at(0);
222 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
223 if (cliff_angle) {
224 if (cliff_angle.value() != 0) {
225 QubitPauliMap qpm;
226 for (const Qubit &q : qbs) qpm.insert({q, Pauli::Z});
227 cliff_.apply_pauli_at_end(SpPauliStabiliser(qpm), *cliff_angle);
228 }
229 } else {
230 QubitPauliMap qpm;
231 for (const Qubit &q : qbs) qpm.insert({q, Pauli::Z});
232 SpPauliStabiliser pauli =
233 cliff_.get_row_product(SpPauliStabiliser(qpm));
234 apply_pauli_gadget_at_end(pauli, angle);
235 }
236 break;
237 }
238 case OpType::XXPhase: {
239 Expr angle = gate.get_params().at(0);
240 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
241 if (cliff_angle) {
242 if (cliff_angle.value() != 0) {
243 cliff_.apply_pauli_at_end(
244 SpPauliStabiliser({{qbs.at(0), Pauli::X}, {qbs.at(1), Pauli::X}}),
245 *cliff_angle);
246 }
247 } else {
248 SpPauliStabiliser pauli = cliff_.get_row_product(
249 SpPauliStabiliser({{qbs.at(0), Pauli::X}, {qbs.at(1), Pauli::X}}));
250 apply_pauli_gadget_at_end(pauli, angle);
251 }
252 break;
253 }
254 case OpType::YYPhase: {
255 Expr angle = gate.get_params().at(0);
256 std::optional<unsigned> cliff_angle = equiv_Clifford(angle);
257 if (cliff_angle) {
258 if (cliff_angle.value() != 0) {
259 cliff_.apply_pauli_at_end(
260 SpPauliStabiliser({{qbs.at(0), Pauli::Y}, {qbs.at(1), Pauli::Y}}),
261 *cliff_angle);
262 }
263 } else {
264 SpPauliStabiliser pauli = cliff_.get_row_product(
265 SpPauliStabiliser({{qbs.at(0), Pauli::Y}, {qbs.at(1), Pauli::Y}}));
266 apply_pauli_gadget_at_end(pauli, angle);
267 }
268 break;
269 }
270 default: {
271 throw BadOpType("Cannot add gate to PauliGraph", type);
272 }
273 }
274}

276void PauliGraph::apply_pauli_gadget_at_end(
277 const SpPauliStabiliser &pauli, const Expr &angle) {
278 PauliVertSet to_search = end_line_;
279 PauliVertSet commuted;
280 PauliVert new_vert = boost::add_vertex(graph_);
281 graph_[new_vert] = {pauli, angle};
282 while (!to_search.empty()) {
283 // Get next candidate parent
284 PauliVert to_compare = *to_search.begin();
285 to_search.erase(to_search.begin());
286
287 // Check that we have already commuted past all of its children
288 bool ready = true;
289 for (const PauliVert &child : get_successors(to_compare)) {
290 if (commuted.get<TagKey>().find(child) == commuted.get<TagKey>().end()) {
291 ready = false;
292 break;
293 }
294 }
295 if (!ready) continue;
296
297 // Check if we can commute past it
298 SpPauliStabiliser compare_pauli = graph_[to_compare].tensor_;
299 if (pauli.commutes_with(compare_pauli)) {
300 if (pauli.string == compare_pauli.string) {
301 // Identical strings - we can merge vertices
302 if (pauli.is_real_negative() == compare_pauli.is_real_negative()) {
303 graph_[to_compare].angle_ += angle;
304 } else {
305 graph_[to_compare].angle_ -= angle;
306 }
307 boost::clear_vertex(new_vert, graph_);
308 boost::remove_vertex(new_vert, graph_);
309
310 std::optional<unsigned> cl_ang =
311 equiv_Clifford(graph_[to_compare].angle_);
312 if (cl_ang) {
313 cliff_.apply_pauli_at_front(graph_[to_compare].tensor_, *cl_ang);
314 start_line_.erase(to_compare);
315 for (const PauliVert &v : get_predecessors(to_compare)) {
316 if (boost::out_degree(v, graph_) == 1) {
317 end_line_.insert(v);
318 }
319 }
320 end_line_.erase(to_compare);
321 boost::clear_vertex(to_compare, graph_);
322 boost::remove_vertex(to_compare, graph_);
323 }
324 return;
325 } else {
326 // Commute and continue searching
327 PauliVertSet preds = get_predecessors(to_compare);
328 to_search.insert(preds.begin(), preds.end());
329 commuted.insert(to_compare);
330 }
331 } else {
332 // Does not commute - add dependency edge
333 boost::add_edge(to_compare, new_vert, graph_);
334 end_line_.erase(to_compare);
335 }
336 }
337 end_line_.insert(new_vert);
338 if (get_predecessors(new_vert).empty()) start_line_.insert(new_vert);
339}

346void PauliGraph::to_graphviz(std::ostream &out) const {
347 out << "digraph G {\n";
348
349 std::map<PauliVert, unsigned> index_map;
350 unsigned i = 0;
351 BGL_FORALL_VERTICES(v, graph_, PauliDAG) {
352 index_map.insert({v, i});
353 out << i << " [label = \"" << graph_[v].tensor_.to_str() << ", "
354 << graph_[v].angle_ << "\"];\n";
355 ++i;
356 }
357
358 BGL_FORALL_EDGES(e, graph_, PauliDAG) {
359 PauliVert v_so = source(e);
360 PauliVert v_ta = target(e);
361 out << index_map.at(v_so) << " -> " << index_map.at(v_ta) << ";\n";
362 }
363
364 out << "}";
365}

367std::vector<PauliVert> PauliGraph::vertices_in_order() const {
368 PauliVIndex index = boost::get(boost::vertex_index, graph_);
369 int i = 0;
370 BGL_FORALL_VERTICES(v, graph_, PauliDAG) { boost::put(index, v, i++); }
371 std::vector<PauliVert> vertices;
372 boost::topological_sort(graph_, std::back_inserter(vertices));
373 std::reverse(vertices.begin(), vertices.end());
374 return vertices;
375}

377void PauliGraph::sanity_check() const {
378 for (const PauliVert &vert : vertices_in_order()) {
379 PauliVertSet succs;
380 boost::graph_traits<PauliDAG>::adjacency_iterator ai, a_end;
381 boost::tie(ai, a_end) = boost::adjacent_vertices(vert, graph_);
382 for (; ai != a_end; ai++) {
383 TKET_ASSERT(!succs.contains(*ai));
384 succs.insert(*ai);
385 }
386
387 PauliVertSet preds;
388 PauliDAG::inv_adjacency_iterator iai, ia_end;
389 boost::tie(iai, ia_end) = boost::inv_adjacent_vertices(vert, graph_);
390 for (; iai != ia_end; iai++) {
391 TKET_ASSERT(!preds.contains(*iai));
392 preds.insert(*iai);
393 }
394 }
395}
