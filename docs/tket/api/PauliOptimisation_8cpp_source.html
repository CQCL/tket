31 Transform pairwise_pauli_gadgets(CXConfigType cx_config) {
32  return Transform([=](Circuit &circ) {
33  Expr t = circ.get_phase();
34  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
35  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
36  OpType optype = op->get_type();
37  if (is_boundary_c_type(optype)) continue;
38  if (optype == OpType::Conditional)
39  throw CircuitInvalidity(
40  "Cannot currently do `pauli_gadgets` optimisation on a "
41  "circuit with conditional gates");
42  if (op->get_desc().is_box())
43  throw CircuitInvalidity(
44  "Cannot currently do `pauli_gadgets` optimisation on a "
45  "circuit with boxes");
46  if (optype == OpType::Measure || optype == OpType::Collapse) {
47  VertexVec q_suc = circ.get_successors_of_type(v, EdgeType::Quantum);
48  if (q_suc.size() != 1 ||
49  !is_final_q_type(circ.get_OpType_from_Vertex(q_suc[0])))
50  throw CircuitInvalidity(
51  "Cannot currently do `pauli_gadgets` optimisation "
52  "on a circuit with a measure in the middle of the "
53  "circuit");
54  }
55  }
56  Transform setup = decompose_multi_qubits_CX() >> decompose_ZX() >>
57  decompose_ZX_to_cliffords();
58  setup.apply(circ);
59  // We effectively commute non-Clifford rotations to the front of the circuit
60  // This gives a sequence of just Pauli gadgets (gadget_circ), followed by
61  // all of the Clifford operations (clifford_circ)
62  std::vector<std::pair<QubitPauliTensor, Expr>> pauli_gadgets;
63  // rx_pauli[i] specifies which Pauli gadget would be built by applying an Rx
64  // rotation on qubit i and then pushing it through the Cliffords to the
65  // front of the circuit. Likewise for rz_pauli with Rz rotations. Clifford
66  // operations will update these and non-Clifford rotations will introduce
67  // Pauli gadgets accordingly
68  Circuit gadget_circ;
69  Circuit clifford_circ;
70  std::map<Qubit, QubitPauliTensor> rx_pauli;
71  std::map<Qubit, QubitPauliTensor> rz_pauli;
72  for (const Qubit &qb : circ.all_qubits()) {
73  gadget_circ.add_qubit(qb);
74  clifford_circ.add_qubit(qb);
75  rx_pauli.insert({qb, QubitPauliTensor(qb, Pauli::X)});
76  rz_pauli.insert({qb, QubitPauliTensor(qb, Pauli::Z)});
77  }
78  for (const Bit &cb : circ.all_bits()) {
79  gadget_circ.add_bit(cb);
80  clifford_circ.add_bit(cb);
81  }
82  // Identify Pauli Gadgets and build Clifford circuit
83  for (const Command &c : circ) {
84  const Op_ptr op_ptr = c.get_op_ptr();
85  unit_vector_t args = c.get_args();
86  OpType type = op_ptr->get_type();
87  switch (type) {
88  // Update rx_pauli and rz_pauli
89  case OpType::S: {
90  Qubit q(args[0]);
91  rx_pauli[q] = i_ * rz_pauli[q] * rx_pauli[q];
92  break;
93  }
94  case OpType::V: {
95  Qubit q(args[0]);
96  rz_pauli[q] = i_ * rx_pauli[q] * rz_pauli[q];
97  break;
98  }
99  case OpType::Z: {
100  Qubit q(args[0]);
101  rx_pauli[q] = -1. * rx_pauli[q];
102  break;
103  }
104  case OpType::X: {
105  Qubit q(args[0]);
106  rz_pauli[q] = -1. * rz_pauli[q];
107  break;
108  }
109  case OpType::Sdg: {
110  Qubit q(args[0]);
111  rx_pauli[q] = -i_ * rz_pauli[q] * rx_pauli[q];
112  break;
113  }
114  case OpType::Vdg: {
115  Qubit q(args[0]);
116  rz_pauli[q] = -i_ * rx_pauli[q] * rz_pauli[q];
117  break;
118  }
119  case OpType::CX: {
120  Qubit q_ctrl(args[0]);
121  Qubit q_trgt(args[1]);
122  rx_pauli[q_ctrl] = rx_pauli[q_ctrl] * rx_pauli[q_trgt];
123  rz_pauli[q_trgt] = rz_pauli[q_ctrl] * rz_pauli[q_trgt];
124  break;
125  }
126  // Introduce a Pauli gadget
127  case OpType::Rz: {
128  Qubit q(args[0]);
129  Expr angle = (op_ptr)->get_params()[0];
130  pauli_gadgets.push_back({rz_pauli[q], angle});
131  break;
132  }
133  case OpType::Rx: {
134  Qubit q(args[0]);
135  Expr angle = (op_ptr)->get_params()[0];
136  pauli_gadgets.push_back({rx_pauli[q], angle});
137  break;
138  }
139  case OpType::noop:
140  case OpType::Phase:
141  case OpType::Measure:
142  case OpType::Collapse:
143  case OpType::Reset:
144  break;
145  default: {
146  std::string error_gate =
147  "Cannot perform pairwise Pauli gadget optimisation using: " +
148  op_ptr->get_name();
149  throw BadOpType(error_gate, type);
150  }
151  }
152  // Add Clifford gates to the back of the circuit to recreate the final
153  // combination at the outputs
154  switch (type) {
155  case OpType::Rz:
156  case OpType::Rx: {
157  break;
158  }
159  default: {
160  clifford_circ.add_op(op_ptr, args);
161  }
162  }
163  }
164  // Synthesise pairs of Pauli Gadgets
165  unsigned g = 0;
166  while (g + 1 < pauli_gadgets.size()) {
167  auto [pauli0, angle0] = pauli_gadgets[g];
168  auto [pauli1, angle1] = pauli_gadgets[g + 1];
169  append_pauli_gadget_pair(
170  gadget_circ, pauli0, angle0, pauli1, angle1, cx_config);
171  g += 2;
172  }
173  // As we synthesised Pauli gadgets 2 at a time, if there were an odd
174  // number, we will have one left over, so add that one on its own
175  if (g < pauli_gadgets.size()) {
176  auto [pauli, angle] = pauli_gadgets[g];
177  append_single_pauli_gadget(gadget_circ, pauli, angle, cx_config);
178  }
179  // Stitch gadget circuit and Clifford circuit together
180  circ = gadget_circ >> clifford_circ;
181  circ.add_phase(t);
182  clifford_simp().apply(circ);
183  return true;
184  });
185 }

187 Transform synthesise_pauli_graph(
188  PauliSynthStrat strat, CXConfigType cx_config) {
189  return Transform([=](Circuit &circ) {
190  Expr t = circ.get_phase();
191  std::optional<std::string> name = circ.get_name();
192  circ.replace_all_implicit_wire_swaps();
193  PauliGraph pg = circuit_to_pauli_graph(circ);
194  switch (strat) {
195  case PauliSynthStrat::Individual: {
196  circ = pauli_graph_to_circuit_individually(pg, cx_config);
197  break;
198  }
199  case PauliSynthStrat::Pairwise: {
200  circ = pauli_graph_to_circuit_pairwise(pg, cx_config);
201  break;
202  }
203  case PauliSynthStrat::Sets: {
204  circ = pauli_graph_to_circuit_sets(pg, cx_config);
205  break;
206  }
207  default:
208  TKET_ASSERT(!"Unknown Pauli Synthesis Strategy");
209  }
210  circ.add_phase(t);
211  if (name) {
212  circ.set_name(*name);
213  }
214  // always turn circuit into PauliGraph and back, so always return true
215  return true;
216  });
217 }

219 Transform special_UCC_synthesis(PauliSynthStrat strat, CXConfigType cx_config) {
220  return Transform([=](Circuit &circ) {
221  Transform synther = synthesise_pauli_graph(strat, cx_config);
222  // make list so we don't run into unboxing vertex issues
223  std::list<Vertex> circbox_verts;
224  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
225  if (circ.get_OpType_from_Vertex(v) == OpType::CircBox)
226  circbox_verts.push_back(v);
227  }
228  for (Vertex v : circbox_verts) {
229  const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
230  std::shared_ptr<const CircBox> box_ptr =
231  std::dynamic_pointer_cast<const CircBox>(op);
232  Circuit inner_circ = *(box_ptr->to_circuit());
233  synther.apply(inner_circ);
234  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
235  circ.substitute(inner_circ, sub);
236  }
237  return !circbox_verts
238  .empty(); // always true if we have left Circuit formalism
239  });
