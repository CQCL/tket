26 PauliPartitionerGraph::PauliPartitionerGraph(
27  const std::list<QubitPauliString>& strings, PauliPartitionStrat strat) {
28  pac_graph = {};
29  for (const QubitPauliString& tensor : strings) {
30  PauliACVertex new_vert = boost::add_vertex(tensor, pac_graph);
31  BGL_FORALL_VERTICES(v, pac_graph, PauliACGraph) {
32  if (v != new_vert) {
33  switch (strat) {
34  case (PauliPartitionStrat::NonConflictingSets): {
35  bool conflict = !tensor.conflicting_qubits(pac_graph[v]).empty();
36  if (conflict) boost::add_edge(new_vert, v, pac_graph);
37  break;
38  }
39  case (PauliPartitionStrat::CommutingSets): {
40  if (!tensor.commutes_with(pac_graph[v])) {
41  boost::add_edge(new_vert, v, pac_graph);
42  }
43  break;
44  }
45  default: {
46  throw UnknownPauliPartitionStrat();
47  }
48  }
49  }
50  }
51  }
52 }

105 AbstractGraphData::AbstractGraphData(const PauliACGraph& pac_graph)
106  : m_adjacency_data(boost::num_vertices(pac_graph)) {
107  const auto vertex_iterators = boost::vertices(pac_graph);
108  for (auto v_iter = vertex_iterators.first; v_iter != vertex_iterators.second;
109  ++v_iter) {
110  // NOTE: on Windows, *v_iter appears to be of type uint64,
111  // but presumably this is an internal Boost implementation detail
112  // and we cannot rely on this.
113  // (It might even be that the Boost vertex IDs are contiguous,
114  // just like ours, in which case this conversion
115  // is unnecessary - but again, we cannot rely on this).
116  const auto this_vertex_id = get_vertex_id(pac_graph[*v_iter]);
117  const auto neighbour_iterators =
118  boost::adjacent_vertices(*v_iter, pac_graph);
119 
120  for (auto n_iter = neighbour_iterators.first;
121  n_iter != neighbour_iterators.second; ++n_iter) {
122  m_adjacency_data.add_edge(
123  this_vertex_id, get_vertex_id(pac_graph[*n_iter]));
124  }
125  }
126 }

128 static std::map<unsigned, std::list<QubitPauliString>>
129 get_partitioned_paulis_for_exhaustive_method(const PauliACGraph& pac_graph) {
130  const AbstractGraphData data(pac_graph);
131  const graphs::GraphColouringResult colouring =
132  graphs::GraphColouringRoutines ::get_colouring(data.get_adjacency_data());
133 
134  TKET_ASSERT(data.get_vertex_map().size() == colouring.colours.size());
135 
136  std::map<unsigned, std::list<QubitPauliString>> colour_map;
137 
138  for (const auto& entry : data.get_vertex_map()) {
139  const QubitPauliString& vertex = entry.first;
140  const size_t id = entry.second;
141  TKET_ASSERT(id < colouring.colours.size());
142 
143  // "id" is the index of this vertex.
144  const size_t colour = colouring.colours[id];
145  TKET_ASSERT(colour < colouring.number_of_colours);
146 
147  colour_map[colour].push_back(vertex);
148  }
149  if (!colour_map.empty()) {
150  TKET_ASSERT(colour_map.size() == 1 + colour_map.crbegin()->first);
151  TKET_ASSERT(colour_map.cbegin()->first == 0);
152  }
153  return colour_map;
154 }

156 static std::map<unsigned, std::list<QubitPauliString>>
157 get_partitioned_paulis_for_largest_first_method(const PauliACGraph& pac_graph) {
158  std::vector<unsigned> order_vec(boost::num_vertices(pac_graph));
159  std::iota(order_vec.begin(), order_vec.end(), 0);
160 
161  std::sort(order_vec.begin(), order_vec.end(), [&](unsigned i, unsigned j) {
162  return (boost::out_degree(i, pac_graph) > boost::out_degree(j, pac_graph));
163  });
164 
165  /* Some boost machinery */
166  std::vector<unsigned> colour_vec(boost::num_vertices(pac_graph));
167  boost::iterator_property_map<
168  unsigned*,
169  boost::property_map<PauliACGraph, boost::vertex_index_t>::const_type>
170  colour_prop_map(
171  &colour_vec.front(), boost::get(boost::vertex_index, pac_graph));
172  boost::sequential_vertex_coloring(
173  pac_graph,
174  boost::make_iterator_property_map(
175  order_vec.begin(), boost::identity_property_map()),
176  colour_prop_map);
177 
178  std::map<unsigned, std::list<QubitPauliString>> colour_map;
179  BGL_FORALL_VERTICES(v, pac_graph, PauliACGraph) {
180  unsigned v_colour = colour_prop_map[v];
181  colour_map[v_colour].push_back(pac_graph[v]);
182  }
183  return colour_map;
184 }

187 PauliPartitionerGraph::partition_paulis(GraphColourMethod method) const {
188  switch (method) {

205 static std::list<std::list<QubitPauliString>>
206 get_term_sequence_for_lazy_colouring_method(
207  const std::list<QubitPauliString>& strings, PauliPartitionStrat strat) {
208  std::list<std::list<QubitPauliString>> terms;
209  for (const QubitPauliString& qpt : strings) {
210  if (terms.empty()) {
211  terms.push_back({qpt});
212  continue;
213  }
214 
215  bool found_bin = false;
216  for (std::list<std::list<QubitPauliString>>::iterator term_iter =
217  terms.begin();
218  term_iter != terms.end(); ++term_iter) {
219  const std::list<QubitPauliString>& qpt_list = *term_iter;
220  bool viable_bin = true;
221  for (const QubitPauliString& qpt2 : qpt_list) {
222  switch (strat) {
223  case (PauliPartitionStrat::NonConflictingSets): {
224  bool conflict = !qpt.conflicting_qubits(qpt2).empty();
225  if (conflict) viable_bin = false;
226  break;
227  }
228  case (PauliPartitionStrat::CommutingSets): {
229  if (!qpt.commutes_with(qpt2)) viable_bin = false;
230  break;
231  }
232  default: {
233  throw UnknownPauliPartitionStrat();
234  }
235  }
236 
237  if (viable_bin == false) break;
238  }
239  if (viable_bin) {
240  term_iter->push_back(qpt);
241  found_bin = true;
242  break;
243  }
244  }
245 
246  if (found_bin == false) {
247  terms.push_back({qpt});
248  }
249  }
250  return terms;
251 }

253 static std::list<std::list<QubitPauliString>>
254 get_term_sequence_with_constructed_dependency_graph(
255  const std::list<QubitPauliString>& strings, PauliPartitionStrat strat,
256  GraphColourMethod method) {

258  PauliPartitionerGraph pp(strings, strat);
259  std::map<unsigned, std::list<QubitPauliString>> colour_map =
260  pp.partition_paulis(method);
261
