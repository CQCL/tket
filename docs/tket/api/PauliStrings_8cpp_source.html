31 static const CmplxSpMat const_2x2_matrix(
32  Complex tl, Complex tr, Complex bl, Complex br) {

34  if (tl != czero) {

49 static const CmplxSpMat &pauli_sparse_mat(Pauli p) {
50  static const CmplxSpMat I_mat = const_2x2_matrix(1, 0, 0, 1);
51  static const CmplxSpMat X_mat = const_2x2_matrix(0, 1, 1, 0);
52  static const CmplxSpMat Y_mat = const_2x2_matrix(0, -i_, i_, 0);
53  static const CmplxSpMat Z_mat = const_2x2_matrix(1, 0, 0, -1);
54  switch (p) {
55  case Pauli::X:
56  return X_mat;
57  case Pauli::Y:
58  return Y_mat;
59  case Pauli::Z:
60  return Z_mat;
61  default:
62  TKET_ASSERT(p == Pauli::I);
63  return I_mat;
64  }
65 }

73 unsigned get_n_qb_from_statevector(const Eigen::VectorXcd &state) {
74  // allowing room for big states
75  unsigned long long n = state.size();
76  if (!(n && (!(n & (n - 1))))) {
77  throw StateNotPowerTwo();
78  }
79 
80  unsigned count = 0;
81  while (n) {
82  n >>= 1;
83  ++count;
84  }
85  return count - 1;
86 }

106 CmplxSpMat QubitPauliString::to_sparse_matrix(
107  const qubit_vector_t &qubits) const {
108  std::vector<Pauli> paulis(qubits.size(), Pauli::I);
109  std::map<Qubit, unsigned> index_map;
110  unsigned index = 0;
111  for (const Qubit &q : qubits) {
112  index_map.insert({q, index});
113  ++index;
114  }
115  if (index_map.size() != qubits.size())
116  throw std::logic_error(
117  "Qubit list given to to_sparse_matrix contains repeats");
118  for (const std::pair<const Qubit, Pauli> &pair : map) {
119  std::map<Qubit, unsigned>::iterator found = index_map.find(pair.first);
120  if (found == index_map.end())
121  throw std::logic_error(
122  "Qubit list given to to_sparse_matrix doesn't contain " +
123  pair.first.repr());
124  paulis.at(found->second) = pair.second;
125  }
126  CmplxSpMat result(1, 1);
127  result.insert(0, 0) = 1;
128  for (Pauli p : paulis) {
129  const CmplxSpMat pauli_mat = pauli_sparse_mat(p);
130  result = Eigen::KroneckerProductSparse(result, pauli_mat).eval();
131  }
132  return result;
133 }

244 int QubitPauliString::compare(const QubitPauliString &other) const {
245  QubitPauliMap::const_iterator p1_it = this->map.begin();
246  QubitPauliMap::const_iterator p2_it = other.map.begin();
247  while (p1_it != this->map.end()) {
248  if (p1_it->second == Pauli::I) {
249  ++p1_it;
250  continue;
251  }
252  while (p2_it != other.map.end() && p2_it->second == Pauli::I) {
253  ++p2_it;
254  }
255  if (p2_it == other.map.end()) return 1;
256  // QubitPauliString order should reflect ILO
257  // i.e. IZ < ZI (Zq1 < Zq0)
258  // Hence we first order by reverse of leading qubit
259  if (p1_it->first < p2_it->first) return 1;
260  if (p2_it->first < p1_it->first) return -1;
261  // and then by increasing order of Pauli letter on the same qubit
262  if (p1_it->second < p2_it->second) return -1;
263  if (p1_it->second > p2_it->second) return 1;
264  ++p1_it;
265  ++p2_it;
266  }
267  while (p2_it != other.map.end() && p2_it->second == Pauli::I) {
268  ++p2_it;
269  }
270  return (p2_it == other.map.end()) ? 0 : -1;
271 }

424 QubitPauliTensor QubitPauliTensor::operator*(
425  const QubitPauliTensor &other) const {
426  QubitPauliTensor result(this->coeff * other.coeff);
427  QubitPauliMap::const_iterator p1i = this->string.map.begin();
428  QubitPauliMap::const_iterator p2i = other.string.map.begin();
429  while (p1i != this->string.map.end()) {
430  while (p2i != other.string.map.end() && p2i->first < p1i->first) {
431  result.string.map.insert(*p2i);
432  p2i++;
433  }
434  if (p2i != other.string.map.end() && p2i->first == p1i->first) {
435  // Pauli in the same position, so need to multiply
436  const std::pair<Complex, Pauli> &prod =
437  QubitPauliTensor::get_mult_matrix().at({p1i->second, p2i->second});
438  result.coeff *= prod.first;
439  if (prod.second != Pauli::I) {
440  result.string.map.insert({p1i->first, prod.second});
441  }
442  p2i++;
443  } else {
444  result.string.map.insert(*p1i);
445  }
446  p1i++;
447  }
448  while (p2i != other.string.map.end()) {
449  result.string.map.insert(*p2i);
450  p2i++;
451  }
452  return result;
453 }
