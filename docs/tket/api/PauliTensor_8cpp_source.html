58 template <>
59 DensePauliMap cast_container<QubitPauliMap, DensePauliMap>(
60  const QubitPauliMap &cont) {
61  if (cont.empty()) return {};
62  unsigned max_index = 0;
63  for (const std::pair<const Qubit, Pauli> &pair : cont) {
64  if (pair.first.reg_info() != register_info_t{UnitType::Qubit, 1} ||
65  pair.first.reg_name() != q_default_reg())
66  throw std::logic_error(
67  "Cannot cast a QubitPauliMap with non-default register qubits to a "
68  "DensePauliMap");
69  unsigned i = pair.first.index().front();
70  if (i > max_index) max_index = i;
71  }
72  DensePauliMap res(max_index + 1, Pauli::I);
73  for (const std::pair<const Qubit, Pauli> &pair : cont) {
74  res.at(pair.first.index().front()) = pair.second;
75  }
76  return res;
77 }

152  if (std::abs(coeff - 1.) < EPS)
153  return 0;
154  else if (std::abs(coeff - i_) < EPS)

158  else if (std::abs(coeff + i_) < EPS)

201 template <>
202 int compare_containers<QubitPauliMap>(
203  const QubitPauliMap &first, const QubitPauliMap &second) {
204  QubitPauliMap::const_iterator p1_it = first.begin();
205  QubitPauliMap::const_iterator p2_it = second.begin();
206  while (p1_it != first.end()) {
207  if (p1_it->second == Pauli::I) {
208  ++p1_it;
209  continue;
210  }
211  while (p2_it != second.end() && p2_it->second == Pauli::I) {
212  ++p2_it;
213  }
214  if (p2_it == second.end()) return 1;
215  // QubitPauliMap order should reflect ILO
216  // i.e. IZ < ZI (Zq1 < Zq0)
217  // Hence we first order by reverse of leading qubit
218  if (p1_it->first < p2_it->first) return 1;
219  if (p2_it->first < p1_it->first) return -1;
220  // and then by increasing order of Pauli letter on the same qubit
221  if (p1_it->second < p2_it->second) return -1;
222  if (p1_it->second > p2_it->second) return 1;
223  ++p1_it;
224  ++p2_it;
225  }
226  while (p2_it != second.end() && p2_it->second == Pauli::I) {
227  ++p2_it;
228  }
229  return (p2_it == second.end()) ? 0 : -1;
230 }

232 template <>
233 int compare_containers<DensePauliMap>(
234  const DensePauliMap &first, const DensePauliMap &second) {
235  DensePauliMap::const_iterator p1_it = first.begin();
236  DensePauliMap::const_iterator p2_it = second.begin();
237  while (p1_it != first.end() && p2_it != second.end()) {
238  if (*p1_it == Pauli::I) {
239  if (*p2_it != Pauli::I) return -1;
240  } else if (*p2_it == Pauli::I)
241  return 1;
242  else if (*p1_it < *p2_it)
243  return -1;
244  else if (*p1_it > *p2_it)
245  return 1;
246  ++p1_it;
247  ++p2_it;
248  }
249  while (p1_it != first.end() && *p1_it == Pauli::I) ++p1_it;
250  if (p1_it != first.end()) return 1;
251  while (p2_it != second.end() && *p2_it == Pauli::I) ++p2_it;
252  return (p2_it == second.end()) ? 0 : -1;
253 }

272 template <>
273 int compare_coeffs<Expr>(const Expr &first, const Expr &second) {
274  // Comparison of SymEngine expressions will distinguish between e.g. integer
275  // 1, double 1., and complex 1., so only use for actual symbolic expressions
276  std::optional<Complex> reduced_first = eval_expr_c(first);
277  std::optional<Complex> reduced_second = eval_expr_c(second);
278  if (reduced_first && reduced_second)
279  return compare_coeffs<Complex>(*reduced_first, *reduced_second);
280  else
281  return first.get_basic()->compare(second);
282 }

554 template <>
555 std::pair<quarter_turns_t, QubitPauliMap> multiply_strings<QubitPauliMap>(
556  const QubitPauliMap &first, const QubitPauliMap &second) {
557  quarter_turns_t total_turns = 0;
558  QubitPauliMap result;
559  QubitPauliMap::const_iterator fi = first.begin();
560  QubitPauliMap::const_iterator si = second.begin();
561  while (fi != first.end()) {
562  while (si != second.end() && si->first < fi->first) {
563  result.insert(*si);
564  ++si;
565  }
566  if (si != second.end() && si->first == fi->first) {
567  // Pauli in the same position, so need to multiply
568  const std::pair<quarter_turns_t, Pauli> &prod =
569  get_mult_matrix().at({fi->second, si->second});
570  total_turns += prod.first;
571  if (prod.second != Pauli::I) {
572  result.insert({fi->first, prod.second});
573  }
574  ++si;
575  } else {
576  result.insert(*fi);
577  }
578  ++fi;
579  }
580  while (si != second.end()) {
581  result.insert(*si);
582  ++si;
583  }
584  return {total_turns, result};
585 }

707 template <>
708 CmplxSpMat to_sparse_matrix<QubitPauliMap>(
709  const QubitPauliMap &paulis, const qubit_vector_t &qubits) {
710  DensePauliMap matrix_paulis(qubits.size(), Pauli::I);
711  std::map<Qubit, unsigned> index_map;
712  for (const Qubit &q : qubits)
713  index_map.insert({q, (unsigned)index_map.size()});
714  if (index_map.size() != qubits.size())
715  throw std::logic_error(
716  "Qubit list given to to_sparse_matrix contains repeats.");
717  for (const std::pair<const Qubit, Pauli> &pair : paulis) {
718  std::map<Qubit, unsigned>::iterator found = index_map.find(pair.first);
719  if (found == index_map.end())
720  throw std::logic_error(
721  "Qubit list given to to_sparse_matrix doesn't contain " +
722  pair.first.repr());
723  matrix_paulis.at(found->second) = pair.second;
724  }
725  return to_sparse_matrix<DensePauliMap>(matrix_paulis);
726 }
