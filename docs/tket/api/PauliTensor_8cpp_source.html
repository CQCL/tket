58template <>
59DensePauliMap cast_container<QubitPauliMap, DensePauliMap>(
60 const QubitPauliMap &cont) {
61 if (cont.empty()) return {};
62 unsigned max_index = 0;
63 for (const std::pair<const Qubit, Pauli> &pair : cont) {
64 if (pair.first.reg_info() != register_info_t{UnitType::Qubit, 1} ||
65 pair.first.reg_name() != q_default_reg())
66 throw std::logic_error(
67 "Cannot cast a QubitPauliMap with non-default register qubits to a "
68 "DensePauliMap");
69 unsigned i = pair.first.index().front();
70 if (i > max_index) max_index = i;
71 }
72 DensePauliMap res(max_index + 1, Pauli::I);
73 for (const std::pair<const Qubit, Pauli> &pair : cont) {
74 res.at(pair.first.index().front()) = pair.second;
75 }
76 return res;
77}

150template <>
151quarter_turns_t cast_coeff<Complex, quarter_turns_t>(const Complex &coeff) {
152 if (std::abs(coeff - 1.) < EPS)
153 return 0;
154 else if (std::abs(coeff - i_) < EPS)
155 return 1;
156 else if (std::abs(coeff + 1.) < EPS)
157 return 2;
158 else if (std::abs(coeff + i_) < EPS)
159 return 3;
160 else
161 throw std::logic_error(
162 "Could not cast PauliTensor coefficient to quarter turns: not a power "
163 "of i.");
164}

201template <>
202int compare_containers<QubitPauliMap>(
203 const QubitPauliMap &first, const QubitPauliMap &second) {
204 QubitPauliMap::const_iterator p1_it = first.begin();
205 QubitPauliMap::const_iterator p2_it = second.begin();
206 while (p1_it != first.end()) {
207 if (p1_it->second == Pauli::I) {
208 ++p1_it;
209 continue;
210 }
211 while (p2_it != second.end() && p2_it->second == Pauli::I) {
212 ++p2_it;
213 }
214 if (p2_it == second.end()) return 1;
215 // QubitPauliMap order should reflect ILO
216 // i.e. IZ < ZI (Zq1 < Zq0)
217 // Hence we first order by reverse of leading qubit
218 if (p1_it->first < p2_it->first) return 1;
219 if (p2_it->first < p1_it->first) return -1;
220 // and then by increasing order of Pauli letter on the same qubit
221 if (p1_it->second < p2_it->second) return -1;
222 if (p1_it->second > p2_it->second) return 1;
223 ++p1_it;
224 ++p2_it;
225 }
226 while (p2_it != second.end() && p2_it->second == Pauli::I) {
227 ++p2_it;
228 }
229 return (p2_it == second.end()) ? 0 : -1;
230}

232template <>
233int compare_containers<DensePauliMap>(
234 const DensePauliMap &first, const DensePauliMap &second) {
235 DensePauliMap::const_iterator p1_it = first.begin();
236 DensePauliMap::const_iterator p2_it = second.begin();
237 while (p1_it != first.end() && p2_it != second.end()) {
238 if (*p1_it == Pauli::I) {
239 if (*p2_it != Pauli::I) return -1;
240 } else if (*p2_it == Pauli::I)
241 return 1;
242 else if (*p1_it < *p2_it)
243 return -1;
244 else if (*p1_it > *p2_it)
245 return 1;
246 ++p1_it;
247 ++p2_it;
248 }
249 while (p1_it != first.end() && *p1_it == Pauli::I) ++p1_it;
250 if (p1_it != first.end()) return 1;
251 while (p2_it != second.end() && *p2_it == Pauli::I) ++p2_it;
252 return (p2_it == second.end()) ? 0 : -1;
253}

265template <>
266int compare_coeffs<Complex>(const Complex &first, const Complex &second) {
267 if (first.real() < second.real()) return -1;
268 if (first.real() > second.real()) return 1;
269 if (first.imag() < second.imag()) return -1;
270 return (first.imag() == second.imag()) ? 0 : 1;
271}
272template <>
273int compare_coeffs<Expr>(const Expr &first, const Expr &second) {
274 // Comparison of SymEngine expressions will distinguish between e.g. integer
275 // 1, double 1., and complex 1., so only use for actual symbolic expressions
276 std::optional<Complex> reduced_first = eval_expr_c(first);
277 std::optional<Complex> reduced_second = eval_expr_c(second);
278 if (reduced_first && reduced_second)
279 return compare_coeffs<Complex>(*reduced_first, *reduced_second);
280 else
281 return first.get_basic()->compare(second);
282}

284std::set<Qubit> common_qubits(
285 const QubitPauliMap &first, const QubitPauliMap &second) {
286 std::set<Qubit> common;
287 for (const std::pair<const Qubit, Pauli> &p : first) {
288 if (p.second == Pauli::I) continue;
289 QubitPauliMap::const_iterator found = second.find(p.first);
290 if (found != second.end() && found->second == p.second)
291 common.insert(p.first);
292 }
293 return common;
294}

296std::set<unsigned> common_indices(
297 const DensePauliMap &first, const DensePauliMap &second) {
298 std::set<unsigned> common;
299 unsigned min_size = std::min(first.size(), second.size());
300 for (unsigned i = 0; i < min_size; ++i) {
301 Pauli p = first.at(i);
302 if (p != Pauli::I && p == second.at(i)) common.insert(i);
303 }
304 return common;
305}

307std::set<Qubit> own_qubits(
308 const QubitPauliMap &first, const QubitPauliMap &second) {
309 std::set<Qubit> own;
310 for (const std::pair<const Qubit, Pauli> &p : first) {
311 if (p.second == Pauli::I) continue;
312 QubitPauliMap::const_iterator found = second.find(p.first);
313 if (found == second.end() || found->second == Pauli::I) own.insert(p.first);
314 }
315 return own;
316}

318std::set<unsigned> own_indices(
319 const DensePauliMap &first, const DensePauliMap &second) {
320 std::set<unsigned> own;
321 unsigned min_size = std::min(first.size(), second.size());
322 for (unsigned i = 0; i < min_size; ++i) {
323 if (first.at(i) != Pauli::I && second.at(i) == Pauli::I) own.insert(i);
324 }
325 for (unsigned i = min_size; i < first.size(); ++i) {
326 if (first.at(i) != Pauli::I) own.insert(i);
327 }
328 return own;
329}

331std::set<Qubit> conflicting_qubits(
332 const QubitPauliMap &first, const QubitPauliMap &second) {
333 std::set<Qubit> conflicts;
334 for (const std::pair<const Qubit, Pauli> &p : first) {
335 if (p.second == Pauli::I) continue;
336 QubitPauliMap::const_iterator found = second.find(p.first);
337 if (found != second.end() && found->second != Pauli::I &&
338 found->second != p.second)
339 conflicts.insert(p.first);
340 }
341 return conflicts;
342}

344std::set<unsigned> conflicting_indices(
345 const DensePauliMap &first, const DensePauliMap &second) {
346 std::set<unsigned> conflicts;
347 unsigned min_size = std::min(first.size(), second.size());
348 for (unsigned i = 0; i < min_size; ++i) {
349 Pauli p = first.at(i);
350 Pauli p2 = second.at(i);
351 if (p != Pauli::I && p2 != Pauli::I && p != p2) conflicts.insert(i);
352 }
353 return conflicts;
354}

554template <>
555std::pair<quarter_turns_t, QubitPauliMap> multiply_strings<QubitPauliMap>(
556 const QubitPauliMap &first, const QubitPauliMap &second) {
557 quarter_turns_t total_turns = 0;
558 QubitPauliMap result;
559 QubitPauliMap::const_iterator fi = first.begin();
560 QubitPauliMap::const_iterator si = second.begin();
561 while (fi != first.end()) {
562 while (si != second.end() && si->first < fi->first) {
563 result.insert(*si);
564 ++si;
565 }
566 if (si != second.end() && si->first == fi->first) {
567 // Pauli in the same position, so need to multiply
568 const std::pair<quarter_turns_t, Pauli> &prod =
569 get_mult_matrix().at({fi->second, si->second});
570 total_turns += prod.first;
571 if (prod.second != Pauli::I) {
572 result.insert({fi->first, prod.second});
573 }
574 ++si;
575 } else {
576 result.insert(*fi);
577 }
578 ++fi;
579 }
580 while (si != second.end()) {
581 result.insert(*si);
582 ++si;
583 }
584 return {total_turns, result};
585}

587template <>
588std::pair<quarter_turns_t, DensePauliMap> multiply_strings<DensePauliMap>(
589 const DensePauliMap &first, const DensePauliMap &second) {
590 quarter_turns_t total_turns = 0;
591 DensePauliMap result;
592 DensePauliMap::const_iterator fi = first.begin();
593 DensePauliMap::const_iterator si = second.begin();
594 while (fi != first.end() && si != second.end()) {
595 const std::pair<quarter_turns_t, Pauli> &prod =
596 get_mult_matrix().at({*fi, *si});
597 total_turns += prod.first;
598 result.push_back(prod.second);
599 ++fi;
600 ++si;
601 }
602 while (fi != first.end()) {
603 result.push_back(*fi);
604 ++fi;
605 }
606 while (si != second.end()) {
607 result.push_back(*si);
608 ++si;
609 }
610 return {total_turns, result};
611}

634static const CmplxSpMat const_2x2_matrix(
635 Complex tl, Complex tr, Complex bl, Complex br) {
636 CmplxSpMat m(2, 2);
637 if (tl != czero) {
638 m.insert(0, 0) = tl;
639 }
640 if (tr != czero) {
641 m.insert(0, 1) = tr;
642 }
643 if (bl != czero) {
644 m.insert(1, 0) = bl;
645 }
646 if (br != czero) {
647 m.insert(1, 1) = br;
648 }
649 return m;
650}

652static const CmplxSpMat &pauli_sparse_mat(Pauli p) {
653 static const CmplxSpMat I_mat = const_2x2_matrix(1, 0, 0, 1);
654 static const CmplxSpMat X_mat = const_2x2_matrix(0, 1, 1, 0);
655 static const CmplxSpMat Y_mat = const_2x2_matrix(0, -i_, i_, 0);
656 static const CmplxSpMat Z_mat = const_2x2_matrix(1, 0, 0, -1);
657 switch (p) {
658 case Pauli::X:
659 return X_mat;
660 case Pauli::Y:
661 return Y_mat;
662 case Pauli::Z:
663 return Z_mat;
664 default:
665 TKET_ASSERT(p == Pauli::I);
666 return I_mat;
667 }
668}

670template <>
671CmplxSpMat to_sparse_matrix<QubitPauliMap>(const QubitPauliMap &paulis) {
672 DensePauliMap matrix_paulis;
673 for (const std::pair<const Qubit, Pauli> &pair : paulis)
674 matrix_paulis.push_back(pair.second);
675 return to_sparse_matrix<DensePauliMap>(matrix_paulis);
676}
677template <>
678CmplxSpMat to_sparse_matrix<DensePauliMap>(const DensePauliMap &paulis) {
679 CmplxSpMat result = CmplxSpMat(1, 1);
680 result.insert(0, 0) = 1.;
681 for (Pauli p : paulis) {
682 const CmplxSpMat pauli_mat = pauli_sparse_mat(p);
683 result = Eigen::KroneckerProductSparse(result, pauli_mat).eval();
684 }
685 return result;
686}

697 const DensePauliMap &paulis, unsigned n_qubits) {
698 if (n_qubits < paulis.size())
699 throw std::logic_error(
700 "Called to_sparse_matrix for fewer qubits than in the Pauli string.");
701 DensePauliMap matrix_paulis = paulis;
702 for (unsigned i = paulis.size(); i < n_qubits; ++i)
703 matrix_paulis.push_back(Pauli::I);
704 return to_sparse_matrix<DensePauliMap>(matrix_paulis);
705}

707template <>
708CmplxSpMat to_sparse_matrix<QubitPauliMap>(
709 const QubitPauliMap &paulis, const qubit_vector_t &qubits) {
710 DensePauliMap matrix_paulis(qubits.size(), Pauli::I);
711 std::map<Qubit, unsigned> index_map;
712 for (const Qubit &q : qubits)
713 index_map.insert({q, (unsigned)index_map.size()});
714 if (index_map.size() != qubits.size())
715 throw std::logic_error(
716 "Qubit list given to to_sparse_matrix contains repeats.");
717 for (const std::pair<const Qubit, Pauli> &pair : paulis) {
718 std::map<Qubit, unsigned>::iterator found = index_map.find(pair.first);
719 if (found == index_map.end())
720 throw std::logic_error(
721 "Qubit list given to to_sparse_matrix doesn't contain " +
722 pair.first.repr());
723 matrix_paulis.at(found->second) = pair.second;
724 }
725 return to_sparse_matrix<DensePauliMap>(matrix_paulis);
726}
727template <>
728CmplxSpMat to_sparse_matrix<DensePauliMap>(
729 const DensePauliMap &paulis, const qubit_vector_t &qubits) {
730 return to_sparse_matrix<QubitPauliMap>(
731 cast_container<DensePauliMap, QubitPauliMap>(paulis), qubits);
732}
