521template <typename PauliContainer, typename CoeffType>
522class PauliTensor {
523 static_assert(
524 std::is_same<PauliContainer, QubitPauliMap>::value ||
525 std::is_same<PauliContainer, DensePauliMap>::value,
526 "PauliTensor must be either dense or qubit-indexed.");
527 static_assert(
528 std::is_same<CoeffType, no_coeff_t>::value ||
529 std::is_same<CoeffType, quarter_turns_t>::value ||
530 std::is_same<CoeffType, Complex>::value ||
531 std::is_same<CoeffType, Expr>::value,
532 "PauliTensor must either support no coefficient, quarter turns, or "
533 "arbitrary complex (floating point or symbolic) coefficients.");
534
535 static const CoeffType default_coeff;
536
537 public:
538 PauliContainer string;
539 CoeffType coeff;
540
544 PauliTensor() : string(), coeff(default_coeff) {}
545
549 PauliTensor(
550 const PauliContainer &_string, const CoeffType &_coeff = default_coeff)
551 : string(_string), coeff(_coeff) {}

556 template <typename PC = PauliContainer>
557 PauliTensor(
558 const Qubit &q, Pauli p, const CoeffType &_coeff = default_coeff,
559 typename std::enable_if<std::is_same<PC, QubitPauliMap>::value>::type * =
560 0)
561 : string({{q, p}}), coeff(_coeff) {}
562
567 template <typename PC = PauliContainer>
568 PauliTensor(
569 const DensePauliMap &_string, const CoeffType &_coeff = default_coeff,
570 typename std::enable_if<std::is_same<PC, QubitPauliMap>::value>::type * =
571 0)
572 : string(cast_container<DensePauliMap, QubitPauliMap>(_string)),
573 coeff(_coeff) {}

579 template <typename PC = PauliContainer>
580 PauliTensor(
581 const std::list<Qubit> &qubits, const std::list<Pauli> &paulis,
582 const CoeffType &_coeff = default_coeff,
583 typename std::enable_if<std::is_same<PC, QubitPauliMap>::value>::type * =
584 0)
585 : string(), coeff(_coeff) {
586 if (qubits.size() != paulis.size()) {
587 throw std::logic_error(
588 "Mismatch of Qubits and Paulis upon QubitPauliString "
589 "construction");
590 }
591 std::list<Pauli>::const_iterator p_it = paulis.begin();
592 for (const Qubit &qb : qubits) {
593 Pauli p = *p_it;
594 if (string.find(qb) != string.end()) {
595 throw std::logic_error(
596 "Non-unique Qubit inserted into QubitPauliString map");
597 }
598 string[qb] = p;
599 ++p_it;
600 }
601 }

626 int compare(const PauliTensor<PauliContainer, CoeffType> &other) const {
627 int cont_comp =
628 compare_containers<PauliContainer>(this->string, other.string);
629 if (cont_comp != 0) return cont_comp;
630 return compare_coeffs<CoeffType>(this->coeff, other.coeff);
631 }
632 bool operator==(const PauliTensor<PauliContainer, CoeffType> &other) const {
633 return (compare(other) == 0);
634 }
635 bool operator!=(const PauliTensor<PauliContainer, CoeffType> &other) const {
636 return !(*this == other);
637 }
638 bool operator<(const PauliTensor<PauliContainer, CoeffType> &other) const {
639 return compare(other) < 0;
640 }

964 Eigen::VectorXcd dot_state(const Eigen::VectorXcd &state) const {
965 // allowing room for big states
966 unsigned long long n = state.size();
967 if (!(n && (!(n & (n - 1)))))
968 throw std::logic_error("Statevector size is not a power of two.");
969 unsigned n_qubits = 0;
970 while (n) {
971 n >>= 1;
972 ++n_qubits;
973 }
974 --n_qubits;
975 return (to_sparse_matrix(n_qubits) * state);
976 }
984 Eigen::VectorXcd dot_state(
985 const Eigen::VectorXcd &state, const qubit_vector_t &qubits) const {
986 if (state.size() != 1 << qubits.size())
987 throw std::logic_error(
988 "Size of statevector does not match number of qubits passed to "
989 "dot_state");
990 return (to_sparse_matrix(qubits) * state);
991 }
992
1000 Complex state_expectation(const Eigen::VectorXcd &state) const {
1001 return state.dot(dot_state(state));
1002 }
1011 Complex state_expectation(
1012 const Eigen::VectorXcd &state, const qubit_vector_t &qubits) const {
1013 return state.dot(dot_state(state, qubits));
1014 }
