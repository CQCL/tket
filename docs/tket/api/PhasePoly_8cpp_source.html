32/* Used only for gray_synth */
33struct SynthStruct {
34 std::list<phase_term_t> terms;
35 std::set<unsigned> remaining_indices;
36 std::optional<unsigned> target;
37};
38
39// update the phase gadgets when adding a CNOT
40static void adjust_vectors(
41 unsigned ctrl, unsigned tgt, std::list<SynthStruct>& Q) {
42 for (SynthStruct& S : Q) {
43 for (std::pair<std::vector<bool>, Expr>& term : S.terms) {
44 std::vector<bool>& vec = term.first;
45 vec[ctrl] = vec[ctrl] ^ vec[tgt];
46 }
47 }
48}

50// see https://arxiv.org/pdf/1712.01859.pdf p12, line 18
51// get qubit with either greatest or least hamming weight
52static unsigned find_best_split(
53 const std::list<phase_term_t>& terms, const std::set<unsigned>& indices) {
54 int max = -1;
55 int max_i = -1;
56 for (unsigned i : indices) {
57 int num_ones = std::count_if(
58 terms.begin(), terms.end(),
59 [=](const std::pair<std::vector<bool>, Expr>& term) {
60 return term.first[i];
61 });
62 int num_zeros = terms.size() - num_ones;
63
64 if (num_zeros > max || num_ones > max) {
65 max = num_zeros > num_ones ? num_zeros : num_ones;
66 max_i = i;
67 }
68 }
69
70 return (unsigned)max_i;
71}

73// divide into S0 and S1 based on value at qubit j
74static std::pair<std::list<phase_term_t>, std::list<phase_term_t>> split(
75 std::list<phase_term_t>& terms, int j) {
76 std::list<phase_term_t> zeros;
77 std::list<phase_term_t> ones;
78
79 while (!terms.empty()) {
80 if (terms.front().first[j])
81 ones.splice(ones.end(), terms, terms.begin());
82 else
83 zeros.splice(zeros.end(), terms, terms.begin());
84 }
85
86 return std::make_pair(zeros, ones);
87}

89/* see: arXiv:1712.01859 */
90Circuit gray_synth(
91 unsigned n_qubits, const std::list<phase_term_t>& parities,
92 const MatrixXb& linear_transformation) {
93 MatrixXb A = linear_transformation;
94 Circuit circ(n_qubits);
95 std::list<SynthStruct> Q; // used to recur over
96 std::set<unsigned>
97 indices; // correspond to qubits which have to be recurred over
98 for (unsigned i = 0; i < n_qubits; ++i) indices.insert(i);
99 Q.push_front({parities, indices, std::nullopt});
100
101 while (!Q.empty()) {
102 SynthStruct S = Q.front();
103 Q.pop_front();
104
105 if (S.terms.size() == 0) continue;
106 // only one column in GraySynth matrix being recurred over
107 else if (S.terms.size() == 1 && S.target) {
108 // special case to avoid doing extra recursion
109 unsigned tgt = *(S.target);
110 auto& [vec, angle] = S.terms.front();
111 for (unsigned ctrl = 0; ctrl < vec.size(); ++ctrl) {
112 if (ctrl != tgt && vec[ctrl]) {
113 circ.add_op<unsigned>(OpType::CX, {ctrl, tgt});
114 adjust_vectors(ctrl, tgt, Q);
115 for (unsigned i = 0; i < A.rows(); i++) {
116 // do column operation on linear
117 // transformation
118 // this will allow us to correct for the CXs
119 // we produce here using Gaussian elim
120 A(i, ctrl) ^= A(i, tgt);
121 }
122 }
123 }
124 circ.add_op<unsigned>(OpType::Rz, angle, {tgt});
125 } else if (!S.remaining_indices.empty()) {
126 unsigned i = find_best_split(S.terms, S.remaining_indices);
127 auto [S0, S1] = split(S.terms, i);
128
129 S.remaining_indices.erase(i);
130
131 if (S.target) {
132 Q.push_front({S1, S.remaining_indices, S.target});
133 } else {
134 Q.push_front({S1, S.remaining_indices, i});
135 }
136 Q.push_front({S0, S.remaining_indices, S.target});
137 }
138 }
139 DiagMatrix m(A);
140 CXMaker cxmaker(n_qubits, false);
141 m.gauss(cxmaker);
142 circ.append(cxmaker._circ.dagger());
143 return circ;
144}

146PhasePolyBox::PhasePolyBox(const Circuit& circ)
147 : Box(OpType::PhasePolyBox), n_qubits_(circ.n_qubits()) {
148 Circuit newcirc(circ);
149
150 // check for classical bits
151 if (newcirc.n_bits() != 0)
152 throw std::invalid_argument(
153 "Cannot construct phase polynomial from classical controlled "
154 "gates");
155
156 // check the gateset of the circuit
157 for (const Command& com : newcirc) {
158 OpType ot = com.get_op_ptr()->get_type();
159
160 switch (ot) {
161 case OpType::CX: {
162 break;
163 }
164 case OpType::Rz: {
165 break;
166 }
167 default: {
168 throw BadOpType("Only CXs and Rzs allowed in Phase Polynomials", ot);
169 }
170 }
171 }
172
173 // replace wireswaps with three CX
174 while (newcirc.has_implicit_wireswaps()) {
175 bool foundswap = false;
176
177 qubit_map_t perm = newcirc.implicit_qubit_permutation();
178 for (const std::pair<const Qubit, Qubit>& pair : perm) {
179 if (pair.first != pair.second) {
180 if (!foundswap) {
181 newcirc.replace_implicit_wire_swap(pair.first, pair.second);
182 foundswap = true;
183 }
184 }
185 }
186 }
187
188 // generate box
189 signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
190 unsigned i = 0;
191 for (const Qubit& qb : newcirc.all_qubits()) {
192 qubit_indices_.insert({qb, i});
193 ++i;
194 }
195 linear_transformation_ = MatrixXb::Identity(n_qubits_, n_qubits_);
196 for (const Command& com : newcirc) {
197 OpType ot = com.get_op_ptr()->get_type();
198 unit_vector_t qbs = com.get_args();
199 if (ot == OpType::CX) {
200 unsigned ctrl = qubit_indices_.left.at(Qubit(qbs[0]));
201 unsigned target = qubit_indices_.left.at(Qubit(qbs[1]));
202 for (unsigned j = 0; j < n_qubits_; ++j) {
203 linear_transformation_.row(target)[j] ^=
204 linear_transformation_.row(ctrl)[j];
205 }
206 } else if (ot == OpType::Rz) {
207 unsigned qb = qubit_indices_.left.at(Qubit(qbs[0]));
208 std::vector<bool> boolvec(n_qubits_);
209 for (unsigned j = 0; j < n_qubits_; ++j) {
210 boolvec[j] = linear_transformation_.row(qb)[j];
211 }
212 PhasePolynomial::iterator pp_it = phase_polynomial_.find(boolvec);
213 if (pp_it == phase_polynomial_.end())
214 phase_polynomial_[boolvec] = com.get_op_ptr()->get_params().at(0);
215 else
216 pp_it->second += com.get_op_ptr()->get_params().at(0);
217 } else
218 TKET_ASSERT(!"Only CXs and Rzs allowed in Phase Polynomials");
219 }
220}

222PhasePolyBox::PhasePolyBox(
223 unsigned n_qubits, const boost::bimap<Qubit, unsigned>& qubit_indices,
224 const PhasePolynomial& phase_polynomial,
225 const MatrixXb& linear_transformation)
226 : Box(OpType::PhasePolyBox),
227 n_qubits_(n_qubits),
228 qubit_indices_(qubit_indices),
229 phase_polynomial_(phase_polynomial),
230 linear_transformation_(linear_transformation) {
231 for (const auto& pair : qubit_indices_) {
232 if (pair.right >= n_qubits) {
233 throw std::invalid_argument(
234 "The creation of a phasepolybox failed: index in qubit "
235 "list is out of range");
236 }
237 }
238
239 for (auto const& ps : phase_polynomial_) {
240 if (ps.first.size() != n_qubits_) {
241 throw std::invalid_argument(
242 "The creation of a phasepolybox failed: PhasePolynomial "
243 "does not match the given number of qubits");
244 }
245
246 if (std::none_of(
247 ps.first.begin(), ps.first.end(), [](bool x) { return x; })) {
248 throw std::invalid_argument(
249 "The creation of a phasepolybox failed: PhasePolynomial "
250 "contains invalid element");
251 }
252 }
253
254 if (linear_transformation_.rows() != n_qubits) {
255 throw std::invalid_argument(
256 "The creation of a phasepolybox failed: row size of the "
257 "linear transformation does not match the number of qubits");
258 }
259
260 if (linear_transformation_.cols() != n_qubits) {
261 throw std::invalid_argument(
262 "The creation of a phasepolybox failed: cols size of the "
263 "linear transformation does not match the number of qubits");
264 }
265
266 signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
267}

269/*
270 * Note: The generate circuit method no longer uses the input qubits to rename
271 * the units of the generated circuit. This is because doing this contradicts
272 * the expectations of the decompose boxes method and was causing errors in some
273 * cases when using custom registers.
274 *
275 * Usually boxes don't know about external
276 * qubit names, because this info is stored in the way the box is wired into a
277 * circuit. For some reason the PhasePolyBox box was written so that it does
278 * know about the names of qubits it was wired to at generation time (either
279 * deduced from the circuit it was generated from or provided by a map). This
280 * internal information is used in some of the AAS compilation strategies.
281 * Although it appears to me that this could be handled in a better way I don't
282 * have time to refactor the AAS code at the moment. So I added a method
283 * generate_circuit_with_original_placement (see below) that uses the logic that
284 * the AAS code expects (the generate_circuit logic), and this is used in the
285 * AAS code.
286 */
287
288void PhasePolyBox::generate_circuit() const {
289 std::list<phase_term_t> phases;
290 for (phase_term_t phase : phase_polynomial_) phases.push_back(phase);
291 Circuit circ = gray_synth(n_qubits_, phases, linear_transformation_);
292 circ_ = std::make_shared<Circuit>(circ);
293}
294
295Circuit PhasePolyBox::generate_circuit_with_original_placement() const {
296 std::list<phase_term_t> phases;
297 for (phase_term_t phase : phase_polynomial_) phases.push_back(phase);
298 Circuit circ = gray_synth(n_qubits_, phases, linear_transformation_);
299 unit_map_t qmap;
300 for (const auto& pair : qubit_indices_.right) {
301 qmap.insert({Qubit(q_default_reg(), pair.first), pair.second});
302 }
303 circ.rename_units(qmap);
304 return circ;
305}

325// Dynamic Eigen matrix requires special treatment to load, to allocate memory
326template <typename T>
327Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> load_dynamic_matrix(
328 const nlohmann::json& j, std::size_t rows, std::size_t cols) {
329 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> mat(rows, cols);
330
331 for (std::size_t row = 0; row < j.size(); ++row) {
332 const auto& j_row = j.at(row);
333 for (std::size_t col = 0; col < j_row.size(); ++col) {
334 mat(row, col) = j_row.at(col).get<T>();
335 }
336 }
337 return mat;
338}

340nlohmann::json PhasePolyBox::to_json(const Op_ptr& op) {
341 const auto& box = static_cast<const PhasePolyBox&>(*op);
342 nlohmann::json j = core_box_json(box);
343 j["n_qubits"] = box.get_n_qubits();
344 j["qubit_indices"] = nlohmann::json::array();
345 for (const auto& pair : box.get_qubit_indices()) {
346 nlohmann::json ind_j;
347 ind_j.push_back(pair.left);
348 ind_j.push_back(pair.right);
349
350 j["qubit_indices"].push_back(ind_j);
351 }
352
353 j["phase_polynomial"] = box.get_phase_polynomial();
354 j["linear_transformation"] = box.get_linear_transformation();
355 return j;
356}
357
358Op_ptr PhasePolyBox::from_json(const nlohmann::json& j) {
359 boost::bimap<Qubit, unsigned> q_ind;
360 for (const auto& j_ar : j.at("qubit_indices")) {
361 q_ind.insert({j_ar.at(0), j_ar.at(1)});
362 }
363 const unsigned n_qb = j.at("n_qubits").get<unsigned>();
364 const MatrixXb& lin_trans =
365 load_dynamic_matrix<bool>(j.at("linear_transformation"), n_qb, n_qb);
366
367 PhasePolyBox box = PhasePolyBox(
368 n_qb, q_ind, j.at("phase_polynomial").get<PhasePolynomial>(), lin_trans);
369 return set_box_id(
370 box,
371 boost::lexical_cast<boost::uuids::uuid>(j.at("id").get<std::string>()));
372}

376CircToPhasePolyConversion::CircToPhasePolyConversion(
377 const Circuit& circ, unsigned min_size) {
378 min_size_ = min_size;
379 circ_ = circ;
380 box_size_ = 0;
381
382 nq_ = circ_.n_qubits();
383 nb_ = circ_.n_bits();
384
385 qubit_types_ = std::vector<QubitType>(nq_, QubitType::pre);
386
387 unsigned i = 0;
388 for (const Qubit& qb : circ_.all_qubits()) {
389 qubit_indices_.insert({qb, i});
390 ++i;
391 }
392
393 i = 0;
394 for (const Bit& b : circ_.all_bits()) {
395 bit_indices_.insert({b, i});
396 ++i;
397 }
398
399 empty_circ_ = Circuit(circ_);
400
401 VertexList bin;
402
403 BGL_FORALL_VERTICES(v, empty_circ_.dag, DAG) {
404 if (!empty_circ_.detect_boundary_Op(v)) {
405 bin.push_back(v);
406 }
407 }
408
409 empty_circ_.remove_vertices(
410 bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
411
412 input_circ_ = Circuit(empty_circ_);
413 box_circ_ = Circuit(nq_);
414 post_circ_ = Circuit(empty_circ_);
415
416 all_qu_ = circ_.all_qubits();
417}

419void CircToPhasePolyConversion::add_phase_poly_box() {
420 qubit_types_.assign(nq_, QubitType::pre);
421
422 if (box_size_ >= min_size_) {
423 PhasePolyBox ppbox(box_circ_);
424 circ_.add_box(ppbox, all_qu_);
425 } else {
426 for (const Command& com : box_circ_) {
427 OpType ot = com.get_op_ptr()->get_type();
428 unit_vector_t qbs = com.get_args();
429 switch (ot) {
430 case OpType::CX: {
431 unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
432 unsigned target = qubit_indices_.at(Qubit(qbs[1]));
433 circ_.add_op<unsigned>(ot, {ctrl, target});
434 break;
435 }
436 case OpType::Rz: {
437 unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
438 auto angle = com.get_op_ptr()->get_params().at(0);
439 circ_.add_op<unsigned>(ot, angle, {qb});
440 break;
441 }
442 default: {
443 // no other types should be in this circuit
444 TKET_ASSERT(!"invalid op type in phase poly box construction");
445 }
446 }
447 }
448 }
449
450 for (const Command& post_com : post_circ_) {
451 OpType post_ot = post_com.get_op_ptr()->get_type();
452 // no other type should be in this list
453 bool expected = (post_ot == OpType::H) || (post_ot == OpType::Measure) ||
454 (post_ot == OpType::Collapse) || (post_ot == OpType::Reset);
455 TKET_ASSERT(expected);
456 unit_vector_t qbs = post_com.get_args();
457 unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
458 if (post_ot == OpType::Measure) {
459 unsigned b = bit_indices_.at(Bit(qbs[1]));
460 circ_.add_op<unsigned>(post_ot, {qb, b});
461 } else {
462 circ_.add_op<unsigned>(post_ot, {qb});
463 }
464 }
465
466 post_circ_ = Circuit(empty_circ_);
467 box_circ_ = Circuit(nq_);
468 box_size_ = 0;
469}

471void CircToPhasePolyConversion::convert() {
472 for (const Command& com : circ_) {
473 OpType ot = com.get_op_ptr()->get_type();
474 unit_vector_t qbs = com.get_args();
475 switch (ot) {
476 case OpType::CX: {
477 input_circ_.add_op<UnitID>(ot, {qbs[0], qbs[1]});
478 break;
479 }
480 case OpType::Rz: {
481 auto angle = com.get_op_ptr()->get_params().at(0);
482 input_circ_.add_op<UnitID>(ot, angle, {qbs[0]});
483 break;
484 }
485 case OpType::H:
486 case OpType::Collapse:
487 case OpType::Reset: {
488 input_circ_.add_op<UnitID>(ot, {qbs[0]});
489 break;
490 }
491 case OpType::Measure: {
492 input_circ_.add_op<UnitID>(ot, {qbs[0], qbs[1]});
493 break;
494 }
495 case OpType::Barrier: {
496 input_circ_.add_barrier(qbs);
497 break;
498 }
499 case OpType::Phase: {
500 input_circ_.add_op<UnitID>(
501 ot, com.get_op_ptr()->get_params().at(0), qbs);
502 break;
503 }
504 default: {
505 throw BadOpType(
506 "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
507 "measure, reset, collapse, barrier gates. Found gate of different "
508 "type",
509 ot);
510 }
511 }
512 }
513
514 VertexList bin;
515
516 BGL_FORALL_VERTICES(v, circ_.dag, DAG) {
517 if (!circ_.detect_boundary_Op(v)) {
518 bin.push_back(v);
519 }
520 }
521
522 circ_.remove_vertices(
523 bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
524
525 /*
526this for loop checks all gates in the circuits and try to
527find the biggest possible sub circuits which contains only
528CX+Rz Gates. This is done in a way that all qubits get
529marked if they are currently outside before (pre), in (in)
530or outside after (post) of a currently constructed box. If
531a qubit is in the pre state and a gate, which is no valid
532type in a box should be added, the type is not changed and
533the gate is added. If a CX or Rz gate should be added to
534this qubit the type is changes to in and the gate is added
535to the box. If another gate should be added to a qubit in
536state, the state stays the same for CX and Rz and they are
537added to the box. If the added gate has a different type
538the type of the qubit is changed to post and the gate is
539added after the box. If a CX or Rz gate should be added to
540a qubit in the post state, the current box is added and a
541new box is started to which the current gate is added. All
542qubits states are reset to pre.
543*/
544 for (const Command& com : input_circ_) {
545 OpType ot = com.get_op_ptr()->get_type();
546 unit_vector_t qbs = com.get_args();
547 switch (ot) {
548 case OpType::Barrier: {
549 add_phase_poly_box();
550
551 circ_.add_barrier(qbs);
552 break;
553 }
554 case OpType::CX: {
555 unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
556 unsigned target = qubit_indices_.at(Qubit(qbs[1]));
557 if ((qubit_types_[ctrl] == QubitType::in) &&
558 (qubit_types_[target] == QubitType::in)) {
559 box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
560 } else if (
561 (qubit_types_[ctrl] == QubitType::pre) &&
562 (qubit_types_[target] == QubitType::in)) {
563 qubit_types_[ctrl] = QubitType::in;
564 box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
565 } else if (
566 (qubit_types_[ctrl] == QubitType::in) &&
567 (qubit_types_[target] == QubitType::pre)) {
568 qubit_types_[target] = QubitType::in;
569 box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
570 } else if (
571 (qubit_types_[ctrl] == QubitType::pre) &&
572 (qubit_types_[target] == QubitType::pre)) {
573 qubit_types_[ctrl] = QubitType::in;
574 qubit_types_[target] = QubitType::in;
575 box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
576 } else if (
577 (qubit_types_[ctrl] == QubitType::post) ||
578 (qubit_types_[target] == QubitType::post)) {
579 add_phase_poly_box();
580
581 qubit_types_[ctrl] = QubitType::in;
582 qubit_types_[target] = QubitType::in;
583 box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
584 } else {
585 // no other types should be in this list
586 TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
587 }
588 ++box_size_;
589 break;
590 }
591 case OpType::Rz: {
592 unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
593 auto angle = com.get_op_ptr()->get_params().at(0);
594 switch (qubit_types_[qb]) {
595 case QubitType::pre: {
596 box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
597 qubit_types_[qb] = QubitType::in;
598 break;
599 }
600 case QubitType::in: {
601 box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
602 break;
603 }
604 case QubitType::post: {
605 add_phase_poly_box();
606
607 qubit_types_[qb] = QubitType::in;
608 box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
609 break;
610 }
611 default: {
612 // no other types should be in this list
613 TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
614 }
615 }
616 break;
617 }
618 case OpType::H:
619 case OpType::Collapse:
620 case OpType::Reset: {
621 unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
622 switch (qubit_types_[qb]) {
623 case QubitType::pre: {
624 circ_.add_op<unsigned>(ot, {qb});
625 break;
626 }
627 case QubitType::in: {
628 post_circ_.add_op<unsigned>(ot, {qb});
629 qubit_types_[qb] = QubitType::post;
630 break;
631 }
632 case QubitType::post: {
633 post_circ_.add_op<unsigned>(ot, {qb});
634 break;
635 }
636 default: {
637 // no other types should be in this list
638 TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
639 }
640 }
641 break;
642 }
643 case OpType::Measure: {
644 unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
645 unsigned b = bit_indices_.at(Bit(qbs[1]));
646
647 switch (qubit_types_[qb]) {
648 case QubitType::pre: {
649 circ_.add_op<unsigned>(ot, {qb, b});
650 break;
651 }
652 case QubitType::in: {
653 post_circ_.add_op<unsigned>(ot, {qb, b});
654 qubit_types_[qb] = QubitType::post;
655 break;
656 }
657 case QubitType::post: {
658 post_circ_.add_op<unsigned>(ot, {qb, b});
659 break;
660 }
661 default: {
662 // no other types should be in this list
663 TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
664 }
665 }

667 }
668 case OpType::Phase: {

670 break;
671 }

673 throw BadOpType(
674 "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
675 "measure, reset, collapse, barrier gates. Found gate of different "
676 "type",
677 ot);
678 }
679 }
680 }

683 add_phase_poly_box();
