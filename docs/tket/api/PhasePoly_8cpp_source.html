41 // update the phase gadgets when adding a CNOT
42 static void adjust_vectors(
43  unsigned ctrl, unsigned tgt, std::list<SynthStruct>& Q) {
44  for (SynthStruct& S : Q) {
45  for (std::pair<std::vector<bool>, Expr>& term : S.terms) {
46  std::vector<bool>& vec = term.first;
47  vec[ctrl] = vec[ctrl] ^ vec[tgt];
48  }
49  }
50 }

52 // see https://arxiv.org/pdf/1712.01859.pdf p12, line 18
53 // get qubit with either greatest or least hamming weight
54 static unsigned find_best_split(
55  const std::list<phase_term_t>& terms, const std::set<unsigned>& indices) {
56  int max = -1;
57  int max_i = -1;
58  for (unsigned i : indices) {
59  int num_ones = std::count_if(
60  terms.begin(), terms.end(),
61  [=](const std::pair<std::vector<bool>, Expr>& term) {
62  return term.first[i];
63  });
64  int num_zeros = terms.size() - num_ones;
65 
66  if (num_zeros > max || num_ones > max) {
67  max = num_zeros > num_ones ? num_zeros : num_ones;
68  max_i = i;
69  }
70  }
71 
72  return (unsigned)max_i;
73 }

75 // divide into S0 and S1 based on value at qubit j
76 static std::pair<std::list<phase_term_t>, std::list<phase_term_t>> split(
77  std::list<phase_term_t>& terms, int j) {
78  std::list<phase_term_t> zeros;
79  std::list<phase_term_t> ones;
80 
81  while (!terms.empty()) {
82  if (terms.front().first[j])
83  ones.splice(ones.end(), terms, terms.begin());
84  else
85  zeros.splice(zeros.end(), terms, terms.begin());
86  }
87 
88  return std::make_pair(zeros, ones);
89 }

91 /* see: arXiv:1712.01859 */
92 Circuit gray_synth(
93  unsigned n_qubits, const std::list<phase_term_t>& parities,
94  const MatrixXb& linear_transformation) {
95  MatrixXb A = linear_transformation;
96  Circuit circ(n_qubits);
97  std::list<SynthStruct> Q; // used to recur over
98  std::set<unsigned>
99  indices; // correspond to qubits which have to be recurred over
100  for (unsigned i = 0; i < n_qubits; ++i) indices.insert(i);
101  Q.push_front({parities, indices, std::nullopt});
102 
103  while (!Q.empty()) {
104  SynthStruct S = Q.front();
105  Q.pop_front();
106 
107  if (S.terms.size() == 0) continue;
108  // only one column in GraySynth matrix being recurred over
109  else if (S.terms.size() == 1 && S.target) {
110  // special case to avoid doing extra recursion
111  unsigned tgt = *(S.target);
112  auto& [vec, angle] = S.terms.front();
113  for (unsigned ctrl = 0; ctrl < vec.size(); ++ctrl) {
114  if (ctrl != tgt && vec[ctrl]) {
115  circ.add_op<unsigned>(OpType::CX, {ctrl, tgt});
116  adjust_vectors(ctrl, tgt, Q);
117  for (unsigned i = 0; i < A.rows(); i++) {
118  // do column operation on linear
119  // transformation
120  // this will allow us to correct for the CXs
121  // we produce here using Gaussian elim
122  A(i, ctrl) ^= A(i, tgt);
123  }
124  }
125  }
126  circ.add_op<unsigned>(OpType::Rz, angle, {tgt});
127  } else if (!S.remaining_indices.empty()) {
128  unsigned i = find_best_split(S.terms, S.remaining_indices);
129  auto [S0, S1] = split(S.terms, i);
130 
131  S.remaining_indices.erase(i);
132 
133  if (S.target) {
134  Q.push_front({S1, S.remaining_indices, S.target});
135  } else {
136  Q.push_front({S1, S.remaining_indices, i});
137  }
138  Q.push_front({S0, S.remaining_indices, S.target});
139  }
140  }
141  DiagMatrix m(A);
142  CXMaker cxmaker(n_qubits, false);
143  m.gauss(cxmaker);
144  circ.append(cxmaker._circ.dagger());
145  return circ;
146 }

148 PhasePolyBox::PhasePolyBox(const Circuit& circ)
149  : Box(OpType::PhasePolyBox), n_qubits_(circ.n_qubits()) {
150  Circuit newcirc(circ);
151 
152  // check for classical bits
153  if (newcirc.n_bits() != 0)
154  throw std::invalid_argument(
155  "Cannot construct phase polynomial from classical controlled "
156  "gates");
157 
158  // check the gateset of the circuit
159  for (const Command& com : newcirc) {
160  OpType ot = com.get_op_ptr()->get_type();
161 
162  switch (ot) {
163  case OpType::CX: {
164  break;
165  }
166  case OpType::Rz: {
167  break;
168  }
169  default: {
170  throw BadOpType("Only CXs and Rzs allowed in Phase Polynomials", ot);
171  }
172  }
173  }
174 
175  // replace wireswaps with three CX
176  while (newcirc.has_implicit_wireswaps()) {
177  bool foundswap = false;
178 
179  qubit_map_t perm = newcirc.implicit_qubit_permutation();
180  for (const std::pair<const Qubit, Qubit>& pair : perm) {
181  if (pair.first != pair.second) {
182  if (!foundswap) {
183  newcirc.replace_implicit_wire_swap(pair.first, pair.second);
184  foundswap = true;
185  }
186  }
187  }
188  }
189 
190  // generate box
191  signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
192  unsigned i = 0;
193  for (const Qubit& qb : newcirc.all_qubits()) {
194  qubit_indices_.insert({qb, i});
195  ++i;
196  }
197  linear_transformation_ = MatrixXb::Identity(n_qubits_, n_qubits_);
198  for (const Command& com : newcirc) {
199  OpType ot = com.get_op_ptr()->get_type();
200  unit_vector_t qbs = com.get_args();
201  if (ot == OpType::CX) {
202  unsigned ctrl = qubit_indices_.left.at(Qubit(qbs[0]));
203  unsigned target = qubit_indices_.left.at(Qubit(qbs[1]));
204  for (unsigned j = 0; j < n_qubits_; ++j) {
205  linear_transformation_.row(target)[j] ^=
206  linear_transformation_.row(ctrl)[j];
207  }
208  } else if (ot == OpType::Rz) {
209  unsigned qb = qubit_indices_.left.at(Qubit(qbs[0]));
210  std::vector<bool> boolvec(n_qubits_);
211  for (unsigned j = 0; j < n_qubits_; ++j) {
212  boolvec[j] = linear_transformation_.row(qb)[j];
213  }
214  PhasePolynomial::iterator pp_it = phase_polynomial_.find(boolvec);
215  if (pp_it == phase_polynomial_.end())
216  phase_polynomial_[boolvec] = com.get_op_ptr()->get_params().at(0);
217  else
218  pp_it->second += com.get_op_ptr()->get_params().at(0);
219  } else
220  TKET_ASSERT(!"Only CXs and Rzs allowed in Phase Polynomials");
221  }
222 }

224 PhasePolyBox::PhasePolyBox(
225  unsigned n_qubits, const boost::bimap<Qubit, unsigned>& qubit_indices,
226  const PhasePolynomial& phase_polynomial,
227  const MatrixXb& linear_transformation)
228  : Box(OpType::PhasePolyBox),
229  n_qubits_(n_qubits),
230  qubit_indices_(qubit_indices),
231  phase_polynomial_(phase_polynomial),
232  linear_transformation_(linear_transformation) {
233  for (const auto& pair : qubit_indices_) {
234  if (pair.right >= n_qubits) {
235  throw std::invalid_argument(
236  "The creation of a phasepolybox failed: index in qubit "
237  "list is out of range");
238  }
239  }
240 
241  for (auto const& ps : phase_polynomial_) {
242  if (ps.first.size() != n_qubits_) {
243  throw std::invalid_argument(
244  "The creation of a phasepolybox failed: PhasePolynomial "
245  "does not match the given number of qubits");
246  }
247 
248  if (std::none_of(
249  ps.first.begin(), ps.first.end(), [](bool x) { return x; })) {
250  throw std::invalid_argument(
251  "The creation of a phasepolybox failed: PhasePolynomial "
252  "contains invalid element");
253  }
254  }
255 
256  if (linear_transformation_.rows() != n_qubits) {
257  throw std::invalid_argument(
258  "The creation of a phasepolybox failed: row size of the "
259  "linear transformation does not match the number of qubits");
260  }
261 
262  if (linear_transformation_.cols() != n_qubits) {
263  throw std::invalid_argument(
264  "The creation of a phasepolybox failed: cols size of the "
265  "linear transformation does not match the number of qubits");
266  }
267 
268  signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
269 }

271 /*
272  * Note: The generate circuit method no longer uses the input qubits to rename
273  * the units of the generated circuit. This is because doing this contradicts
274  * the expectations of the decompose boxes method and was causing errors in some
275  * cases when using custom registers.
276  *
277  * Usually boxes don't know about external
278  * qubit names, because this info is stored in the way the box is wired into a
279  * circuit. For some reason the PhasePolyBox box was written so that it does
280  * know about the names of qubits it was wired to at generation time (either
281  * deduced from the circuit it was generated from or provided by a map). This
282  * internal information is used in some of the AAS compilation strategies.
283  * Although it appears to me that this could be handled in a better way I don't
284  * have time to refactor the AAS code at the moment. So I added a method
285  * generate_circuit_with_original_placement (see below) that uses the logic that
286  * the AAS code expects (the generate_circuit logic), and this is used in the
287  * AAS code.
288  */

290 void PhasePolyBox::generate_circuit() const {
291  std::list<phase_term_t> phases;
292  for (phase_term_t phase : phase_polynomial_) phases.push_back(phase);
293  Circuit circ = gray_synth(n_qubits_, phases, linear_transformation_);
294  circ_ = std::make_shared<Circuit>(circ);
295 }

297 Circuit PhasePolyBox::generate_circuit_with_original_placement() const {
298  std::list<phase_term_t> phases;
299  for (phase_term_t phase : phase_polynomial_) phases.push_back(phase);
300  Circuit circ = gray_synth(n_qubits_, phases, linear_transformation_);
301  unit_map_t qmap;
302  for (const auto& pair : qubit_indices_.right) {
303  qmap.insert({Qubit(q_default_reg(), pair.first), pair.second});
304  }
305  circ.rename_units(qmap);
306  return circ;
307 }

327 // Dynamic Eigen matrix requires special treatment to load, to allocate memory
328 template <typename T>
329 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> load_dynamic_matrix(
330  const nlohmann::json& j, std::size_t rows, std::size_t cols) {
331  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> mat(rows, cols);
332 
333  for (std::size_t row = 0; row < j.size(); ++row) {
334  const auto& j_row = j.at(row);
335  for (std::size_t col = 0; col < j_row.size(); ++col) {
336  mat(row, col) = j_row.at(col).get<T>();
337  }
338  }
339  return mat;
340 }

360 Op_ptr PhasePolyBox::from_json(const nlohmann::json& j) {
361  boost::bimap<Qubit, unsigned> q_ind;
362  for (const auto& j_ar : j.at("qubit_indices")) {
363  q_ind.insert({j_ar.at(0), j_ar.at(1)});
364  }
365  const unsigned n_qb = j.at("n_qubits").get<unsigned>();
366  const MatrixXb& lin_trans =
367  load_dynamic_matrix<bool>(j.at("linear_transformation"), n_qb, n_qb);
368 
369  PhasePolyBox box = PhasePolyBox(
370  n_qb, q_ind, j.at("phase_polynomial").get<PhasePolynomial>(), lin_trans);
371  return set_box_id(
372  box,
373  boost::lexical_cast<boost::uuids::uuid>(j.at("id").get<std::string>()));
374 }

378 CircToPhasePolyConversion::CircToPhasePolyConversion(
379  const Circuit& circ, unsigned min_size) {
380  min_size_ = min_size;
381  circ_ = circ;
382  box_size_ = 0;
383 
384  nq_ = circ_.n_qubits();
385  nb_ = circ_.n_bits();
386 
387  qubit_types_ = std::vector<QubitType>(nq_, QubitType::pre);
388 
389  unsigned i = 0;
390  for (const Qubit& qb : circ_.all_qubits()) {
391  qubit_indices_.insert({qb, i});
392  ++i;
393  }
394 
395  i = 0;
396  for (const Bit& b : circ_.all_bits()) {
397  bit_indices_.insert({b, i});
398  ++i;
399  }
400 
401  empty_circ_ = Circuit(circ_);
402 
403  VertexList bin;
404 
405  BGL_FORALL_VERTICES(v, empty_circ_.dag, DAG) {
406  if (!empty_circ_.detect_boundary_Op(v)) {
407  bin.push_back(v);
408  }
409  }
410 
411  empty_circ_.remove_vertices(
412  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
413 
414  input_circ_ = Circuit(empty_circ_);
415  box_circ_ = Circuit(nq_);
416  post_circ_ = Circuit(empty_circ_);
417 
418  all_qu_ = circ_.all_qubits();
419 }

421 void CircToPhasePolyConversion::add_phase_poly_box() {
422  qubit_types_.assign(nq_, QubitType::pre);
423 
424  if (box_size_ >= min_size_) {
425  PhasePolyBox ppbox(box_circ_);
426  circ_.add_box(ppbox, all_qu_);
427  } else {
428  for (const Command& com : box_circ_) {
429  OpType ot = com.get_op_ptr()->get_type();
430  unit_vector_t qbs = com.get_args();
431  switch (ot) {
432  case OpType::CX: {
433  unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
434  unsigned target = qubit_indices_.at(Qubit(qbs[1]));
435  circ_.add_op<unsigned>(ot, {ctrl, target});
436  break;
437  }
438  case OpType::Rz: {
439  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
440  auto angle = com.get_op_ptr()->get_params().at(0);
441  circ_.add_op<unsigned>(ot, angle, {qb});
442  break;
443  }
444  default: {
445  // no other types should be in this circuit
446  TKET_ASSERT(!"invalid op type in phase poly box construction");
447  }
448  }
449  }
450  }
451 
452  for (const Command& post_com : post_circ_) {
453  OpType post_ot = post_com.get_op_ptr()->get_type();
454  // no other type should be in this list
455  bool expected = (post_ot == OpType::H) || (post_ot == OpType::Measure) ||
456  (post_ot == OpType::Collapse) || (post_ot == OpType::Reset);
457  TKET_ASSERT(expected);
458  unit_vector_t qbs = post_com.get_args();
459  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
460  if (post_ot == OpType::Measure) {
461  unsigned b = bit_indices_.at(Bit(qbs[1]));
462  circ_.add_op<unsigned>(post_ot, {qb, b});
463  } else {
464  circ_.add_op<unsigned>(post_ot, {qb});
465  }
466  }
467 
468  post_circ_ = Circuit(empty_circ_);
469  box_circ_ = Circuit(nq_);
470  box_size_ = 0;
471 }

473 void CircToPhasePolyConversion::convert() {
474  for (const Command& com : circ_) {
475  OpType ot = com.get_op_ptr()->get_type();
476  unit_vector_t qbs = com.get_args();
477  switch (ot) {
478  case OpType::CX: {
479  input_circ_.add_op<UnitID>(ot, {qbs[0], qbs[1]});
480  break;
481  }
482  case OpType::Rz: {
483  auto angle = com.get_op_ptr()->get_params().at(0);
484  input_circ_.add_op<UnitID>(ot, angle, {qbs[0]});
485  break;
486  }
487  case OpType::H:
488  case OpType::Collapse:
489  case OpType::Reset: {
490  input_circ_.add_op<UnitID>(ot, {qbs[0]});
491  break;
492  }
493  case OpType::Measure: {
494  input_circ_.add_op<UnitID>(ot, {qbs[0], qbs[1]});
495  break;
496  }
497  case OpType::Barrier: {
498  input_circ_.add_barrier(qbs);
499  break;
500  }
501  default: {
502  throw BadOpType(
503  "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
504  "measure, reset, collapse, barrier gates. Found gate of different "
505  "type",
506  ot);
507  }
508  }
509  }
510 
511  VertexList bin;
512 
513  BGL_FORALL_VERTICES(v, circ_.dag, DAG) {
514  if (!circ_.detect_boundary_Op(v)) {
515  bin.push_back(v);
516  }
517  }
518 
519  circ_.remove_vertices(
520  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
521 
522  /*
523 this for loop checks all gates in the circuits and try to
524 find the biggest possible sub circuits which contains only
525 CX+Rz Gates. This is done in a way that all qubits get
526 marked if they are currently outside before (pre), in (in)
527 or outside after (post) of a currently constructed box. If
528 a qubit is in the pre state and a gate, which is no valid
529 type in a box should be added, the type is not changed and
530 the gate is added. If a CX or Rz gate should be added to
531 this qubit the type is changes to in and the gate is added
532 to the box. If another gate should be added to a qubit in
533 state, the state stays the same for CX and Rz and they are
534 added to the box. If the added gate has a different type
535 the type of the qubit is changed to post and the gate is
536 added after the box. If a CX or Rz gate should be added to
537 a qubit in the post state, the current box is added and a
538 new box is started to which the current gate is added. All
539 qubits states are reset to pre.
540 */
541  for (const Command& com : input_circ_) {
542  OpType ot = com.get_op_ptr()->get_type();
543  unit_vector_t qbs = com.get_args();
544  switch (ot) {
545  case OpType::Barrier: {
546  add_phase_poly_box();
547 
548  circ_.add_barrier(qbs);
549  break;
550  }
551  case OpType::CX: {
552  unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
553  unsigned target = qubit_indices_.at(Qubit(qbs[1]));
554  if ((qubit_types_[ctrl] == QubitType::in) &&
555  (qubit_types_[target] == QubitType::in)) {
556  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
557  } else if (
558  (qubit_types_[ctrl] == QubitType::pre) &&
559  (qubit_types_[target] == QubitType::in)) {
560  qubit_types_[ctrl] = QubitType::in;
561  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
562  } else if (
563  (qubit_types_[ctrl] == QubitType::in) &&
564  (qubit_types_[target] == QubitType::pre)) {
565  qubit_types_[target] = QubitType::in;
566  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
567  } else if (
568  (qubit_types_[ctrl] == QubitType::pre) &&
569  (qubit_types_[target] == QubitType::pre)) {
570  qubit_types_[ctrl] = QubitType::in;
571  qubit_types_[target] = QubitType::in;
572  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
573  } else if (
574  (qubit_types_[ctrl] == QubitType::post) ||
575  (qubit_types_[target] == QubitType::post)) {
576  add_phase_poly_box();
577 
578  qubit_types_[ctrl] = QubitType::in;
579  qubit_types_[target] = QubitType::in;
580  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
581  } else {
582  // no other types should be in this list
583  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
584  }
585  ++box_size_;
586  break;
587  }
588  case OpType::Rz: {
589  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
590  auto angle = com.get_op_ptr()->get_params().at(0);
591  switch (qubit_types_[qb]) {
592  case QubitType::pre: {
593  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
594  qubit_types_[qb] = QubitType::in;
595  break;
596  }
597  case QubitType::in: {
598  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
599  break;
600  }
601  case QubitType::post: {
602  add_phase_poly_box();
603 
604  qubit_types_[qb] = QubitType::in;
605  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
606  break;
607  }
608  default: {
609  // no other types should be in this list
610  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
611  }
612  }
613  break;
614  }
615  case OpType::H:
616  case OpType::Collapse:
617  case OpType::Reset: {
618  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
619  switch (qubit_types_[qb]) {
620  case QubitType::pre: {
621  circ_.add_op<unsigned>(ot, {qb});
622  break;
623  }
624  case QubitType::in: {
625  post_circ_.add_op<unsigned>(ot, {qb});
626  qubit_types_[qb] = QubitType::post;
627  break;
628  }
629  case QubitType::post: {
630  post_circ_.add_op<unsigned>(ot, {qb});
631  break;
632  }
633  default: {
634  // no other types should be in this list
635  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
636  }
637  }
638  break;
639  }
640  case OpType::Measure: {
641  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
642  unsigned b = bit_indices_.at(Bit(qbs[1]));
643 
644  switch (qubit_types_[qb]) {
645  case QubitType::pre: {
646  circ_.add_op<unsigned>(ot, {qb, b});
647  break;
648  }
649  case QubitType::in: {
650  post_circ_.add_op<unsigned>(ot, {qb, b});
651  qubit_types_[qb] = QubitType::post;
652  break;
653  }
654  case QubitType::post: {
655  post_circ_.add_op<unsigned>(ot, {qb, b});
656  break;
657  }
658  default: {
659  // no other types should be in this list
660  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
661  }
662  }
663  break;
664  }
665  default: {
666  throw BadOpType(
667  "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
668  "measure, reset, collapse, barrier gates. Found gate of different "
669  "type",
670  ot);
671  }
672  }
673  }
674 
675  // add the last box to the circuit
676  add_phase_poly_box();
