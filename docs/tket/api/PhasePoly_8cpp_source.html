42 // update the phase gadgets when adding a CNOT
43 static void adjust_vectors(
44  unsigned ctrl, unsigned tgt, std::list<SynthStruct>& Q) {
45  for (SynthStruct& S : Q) {
46  for (std::pair<std::vector<bool>, Expr>& term : S.terms) {
47  std::vector<bool>& vec = term.first;
48  vec[ctrl] = vec[ctrl] ^ vec[tgt];
49  }
50  }
51 }

53 // see https://arxiv.org/pdf/1712.01859.pdf p12, line 18
54 // get qubit with either greatest or least hamming weight
55 static unsigned find_best_split(
56  const std::list<phase_term_t>& terms, const std::set<unsigned>& indices) {
57  int max = -1;
58  int max_i = -1;
59  for (unsigned i : indices) {
60  int num_ones = std::count_if(
61  terms.begin(), terms.end(),
62  [=](const std::pair<std::vector<bool>, Expr>& term) {
63  return term.first[i];
64  });
65  int num_zeros = terms.size() - num_ones;
66 
67  if (num_zeros > max || num_ones > max) {
68  max = num_zeros > num_ones ? num_zeros : num_ones;
69  max_i = i;
70  }
71  }
72 
73  return (unsigned)max_i;
74 }

76 // divide into S0 and S1 based on value at qubit j
77 static std::pair<std::list<phase_term_t>, std::list<phase_term_t>> split(
78  std::list<phase_term_t>& terms, int j) {
79  std::list<phase_term_t> zeros;
80  std::list<phase_term_t> ones;
81 
82  while (!terms.empty()) {
83  if (terms.front().first[j])
84  ones.splice(ones.end(), terms, terms.begin());
85  else
86  zeros.splice(zeros.end(), terms, terms.begin());
87  }
88 
89  return std::make_pair(zeros, ones);
90 }

92 /* see: arXiv:1712.01859 */
93 Circuit gray_synth(
94  unsigned n_qubits, const std::list<phase_term_t>& parities,
95  const MatrixXb& linear_transformation) {
96  MatrixXb A = linear_transformation;
97  Circuit circ(n_qubits);
98  std::list<SynthStruct> Q; // used to recur over
99  std::set<unsigned>
100  indices; // correspond to qubits which have to be recurred over
101  for (unsigned i = 0; i < n_qubits; ++i) indices.insert(i);
102  Q.push_front({parities, indices, std::nullopt});
103 
104  while (!Q.empty()) {
105  SynthStruct S = Q.front();
106  Q.pop_front();
107 
108  if (S.terms.size() == 0) continue;
109  // only one column in GraySynth matrix being recurred over
110  else if (S.terms.size() == 1 && S.target) {
111  // special case to avoid doing extra recursion
112  unsigned tgt = *(S.target);
113  auto& [vec, angle] = S.terms.front();
114  for (unsigned ctrl = 0; ctrl < vec.size(); ++ctrl) {
115  if (ctrl != tgt && vec[ctrl]) {
116  circ.add_op<unsigned>(OpType::CX, {ctrl, tgt});
117  adjust_vectors(ctrl, tgt, Q);
118  for (unsigned i = 0; i < A.rows(); i++) {
119  // do column operation on linear
120  // transformation
121  // this will allow us to correct for the CXs
122  // we produce here using Gaussian elim
123  A(i, ctrl) ^= A(i, tgt);
124  }
125  }
126  }
127  circ.add_op<unsigned>(OpType::Rz, angle, {tgt});
128  } else if (!S.remaining_indices.empty()) {
129  unsigned i = find_best_split(S.terms, S.remaining_indices);
130  auto [S0, S1] = split(S.terms, i);
131 
132  S.remaining_indices.erase(i);
133 
134  if (S.target) {
135  Q.push_front({S1, S.remaining_indices, S.target});
136  } else {
137  Q.push_front({S1, S.remaining_indices, i});
138  }
139  Q.push_front({S0, S.remaining_indices, S.target});
140  }
141  }
142  DiagMatrix m(A);
143  CXMaker cxmaker(n_qubits, false);
144  m.gauss(cxmaker);
145  circ.append(cxmaker._circ.dagger());
146  return circ;
147 }

149 PhasePolyBox::PhasePolyBox(const Circuit& circ)
150  : Box(OpType::PhasePolyBox), n_qubits_(circ.n_qubits()) {
151  Circuit newcirc(circ);
152 
153  // check for classical bits
154  if (newcirc.n_bits() != 0)
155  throw std::invalid_argument(
156  "Cannot construct phase polynomial from classical controlled "
157  "gates");
158 
159  // check the gateset of the circuit
160  for (const Command& com : newcirc) {
161  OpType ot = com.get_op_ptr()->get_type();
162 
163  switch (ot) {
164  case OpType::CX: {
165  break;
166  }
167  case OpType::Rz: {
168  break;
169  }
170  default: {
171  throw BadOpType("Only CXs and Rzs allowed in Phase Polynomials", ot);
172  }
173  }
174  }
175 
176  // replace wireswaps with three CX
177  while (newcirc.has_implicit_wireswaps()) {
178  bool foundswap = false;
179 
180  qubit_map_t perm = newcirc.implicit_qubit_permutation();
181  for (const std::pair<const Qubit, Qubit>& pair : perm) {
182  if (pair.first != pair.second) {
183  if (!foundswap) {
184  newcirc.replace_implicit_wire_swap(pair.first, pair.second);
185  foundswap = true;
186  }
187  }
188  }
189  }
190 
191  // generate box
192  signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
193  unsigned i = 0;
194  for (const Qubit& qb : newcirc.all_qubits()) {
195  qubit_indices_.insert({qb, i});
196  ++i;
197  }
198  linear_transformation_ = MatrixXb::Identity(n_qubits_, n_qubits_);
199  for (const Command& com : newcirc) {
200  OpType ot = com.get_op_ptr()->get_type();
201  unit_vector_t qbs = com.get_args();
202  if (ot == OpType::CX) {
203  unsigned ctrl = qubit_indices_.left.at(Qubit(qbs[0]));
204  unsigned target = qubit_indices_.left.at(Qubit(qbs[1]));
205  for (unsigned j = 0; j < n_qubits_; ++j) {
206  linear_transformation_.row(target)[j] ^=
207  linear_transformation_.row(ctrl)[j];
208  }
209  } else if (ot == OpType::Rz) {
210  unsigned qb = qubit_indices_.left.at(Qubit(qbs[0]));
211  std::vector<bool> boolvec(n_qubits_);
212  for (unsigned j = 0; j < n_qubits_; ++j) {
213  boolvec[j] = linear_transformation_.row(qb)[j];
214  }
215  PhasePolynomial::iterator pp_it = phase_polynomial_.find(boolvec);
216  if (pp_it == phase_polynomial_.end())
217  phase_polynomial_[boolvec] = com.get_op_ptr()->get_params().at(0);
218  else
219  pp_it->second += com.get_op_ptr()->get_params().at(0);
220  } else
221  TKET_ASSERT(!"Only CXs and Rzs allowed in Phase Polynomials");
222  }
223 }

225 PhasePolyBox::PhasePolyBox(
226  unsigned n_qubits, const boost::bimap<Qubit, unsigned>& qubit_indices,
227  const PhasePolynomial& phase_polynomial,
228  const MatrixXb& linear_transformation)
229  : Box(OpType::PhasePolyBox),
230  n_qubits_(n_qubits),
231  qubit_indices_(qubit_indices),
232  phase_polynomial_(phase_polynomial),
233  linear_transformation_(linear_transformation) {
234  for (const auto& pair : qubit_indices_) {
235  if (pair.right >= n_qubits) {
236  throw std::invalid_argument(
237  "The creation of a phasepolybox failed: index in qubit "
238  "list is out of range");
239  }
240  }
241 
242  for (auto const& ps : phase_polynomial_) {
243  if (ps.first.size() != n_qubits_) {
244  throw std::invalid_argument(
245  "The creation of a phasepolybox failed: PhasePolynomial "
246  "does not match the given number of qubits");
247  }
248 
249  if (std::none_of(
250  ps.first.begin(), ps.first.end(), [](bool x) { return x; })) {
251  throw std::invalid_argument(
252  "The creation of a phasepolybox failed: PhasePolynomial "
253  "contains invalid element");
254  }
255  }
256 
257  if (linear_transformation_.rows() != n_qubits) {
258  throw std::invalid_argument(
259  "The creation of a phasepolybox failed: row size of the "
260  "linear transformation does not match the number of qubits");
261  }
262 
263  if (linear_transformation_.cols() != n_qubits) {
264  throw std::invalid_argument(
265  "The creation of a phasepolybox failed: cols size of the "
266  "linear transformation does not match the number of qubits");
267  }
268 
269  signature_ = op_signature_t(n_qubits_, EdgeType::Quantum);
270 }

272 void PhasePolyBox::generate_circuit() const {
273  std::list<phase_term_t> phases;
274  for (phase_term_t phase : phase_polynomial_) phases.push_back(phase);
275  Circuit circ = gray_synth(n_qubits_, phases, linear_transformation_);
276  unit_map_t qmap;
277  for (const auto& pair : qubit_indices_.right) {
278  qmap.insert({Qubit(q_default_reg(), pair.first), pair.second});
279  }
280  circ.rename_units(qmap);
281  circ_ = std::make_shared<Circuit>(circ);
282 }

302 // Dynamic Eigen matrix requires special treatment to load, to allocate memory
303 template <typename T>
304 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> load_dynamic_matrix(
305  const nlohmann::json& j, size_t rows, size_t cols) {
306  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> mat(rows, cols);
307 
308  for (size_t row = 0; row < j.size(); ++row) {
309  const auto& j_row = j.at(row);
310  for (size_t col = 0; col < j_row.size(); ++col) {
311  mat(row, col) = j_row.at(col).get<T>();
312  }
313  }
314  return mat;
315 }

335 Op_ptr PhasePolyBox::from_json(const nlohmann::json& j) {
336  boost::bimap<Qubit, unsigned> q_ind;
337  for (const auto& j_ar : j.at("qubit_indices")) {
338  q_ind.insert({j_ar.at(0), j_ar.at(1)});

344  PhasePolyBox box = PhasePolyBox(

346  return set_box_id(

353 CircToPhasePolyConversion::CircToPhasePolyConversion(
354  const Circuit& circ, unsigned min_size) {
355  min_size_ = min_size;
356  circ_ = circ;
357  box_size_ = 0;
358 
359  nq_ = circ_.n_qubits();
360  nb_ = circ_.n_bits();
361 
362  qubit_types_ = std::vector<QubitType>(nq_, QubitType::pre);
363 
364  unsigned i = 0;
365  for (const Qubit& qb : circ_.all_qubits()) {
366  qubit_indices_.insert({qb, i});
367  ++i;
368  }
369 
370  i = 0;
371  for (const Bit& b : circ_.all_bits()) {
372  bit_indices_.insert({b, i});
373  ++i;
374  }
375 
376  empty_circ_ = Circuit(circ_);
377 
378  VertexList bin;
379 
380  BGL_FORALL_VERTICES(v, empty_circ_.dag, DAG) {
381  if (!empty_circ_.detect_boundary_Op(v)) {
382  bin.push_back(v);
383  }
384  }
385 
386  empty_circ_.remove_vertices(
387  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
388 
389  input_circ_ = Circuit(empty_circ_);
390  box_circ_ = Circuit(nq_);
391  post_circ_ = Circuit(empty_circ_);
392 
393  all_qu_ = circ_.all_qubits();
394 }

396 void CircToPhasePolyConversion::add_phase_poly_box() {
397  qubit_types_.assign(nq_, QubitType::pre);
398 
399  if (box_size_ >= min_size_) {
400  PhasePolyBox ppbox(box_circ_);
401  circ_.add_box(ppbox, all_qu_);
402  } else {
403  for (const Command& com : box_circ_) {
404  OpType ot = com.get_op_ptr()->get_type();
405  unit_vector_t qbs = com.get_args();
406  switch (ot) {
407  case OpType::CX: {
408  unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
409  unsigned target = qubit_indices_.at(Qubit(qbs[1]));
410  circ_.add_op<unsigned>(ot, {ctrl, target});
411  break;
412  }
413  case OpType::Rz: {
414  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
415  auto angle = com.get_op_ptr()->get_params().at(0);
416  circ_.add_op<unsigned>(ot, angle, {qb});
417  break;
418  }
419  default: {
420  // no other types should be in this circuit
421  TKET_ASSERT(!"invalid op type in phase poly box construction");
422  }
423  }
424  }
425  }
426 
427  for (const Command& post_com : post_circ_) {
428  OpType post_ot = post_com.get_op_ptr()->get_type();
429  // no other type should be in this list
430  bool expected = (post_ot == OpType::H) || (post_ot == OpType::Measure) ||
431  (post_ot == OpType::Collapse) || (post_ot == OpType::Reset);
432  TKET_ASSERT(expected);
433  unit_vector_t qbs = post_com.get_args();
434  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
435  if (post_ot == OpType::Measure) {
436  unsigned b = bit_indices_.at(Bit(qbs[1]));
437  circ_.add_op<unsigned>(post_ot, {qb, b});
438  } else {
439  circ_.add_op<unsigned>(post_ot, {qb});
440  }
441  }
442 
443  post_circ_ = Circuit(empty_circ_);
444  box_circ_ = Circuit(nq_);
445  box_size_ = 0;
446 }

448 void CircToPhasePolyConversion::convert() {
449  for (const Command& com : circ_) {
450  OpType ot = com.get_op_ptr()->get_type();
451  unit_vector_t qbs = com.get_args();
452  switch (ot) {
453  case OpType::CX: {
454  unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
455  unsigned target = qubit_indices_.at(Qubit(qbs[1]));
456  input_circ_.add_op<unsigned>(ot, {ctrl, target});
457  break;
458  }
459  case OpType::Rz: {
460  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
461  auto angle = com.get_op_ptr()->get_params().at(0);
462  input_circ_.add_op<unsigned>(ot, angle, {qb});
463  break;
464  }
465  case OpType::H:
466  case OpType::Collapse:
467  case OpType::Reset: {
468  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
469  input_circ_.add_op<unsigned>(ot, {qb});
470  break;
471  }
472  case OpType::Measure: {
473  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
474  unsigned b = bit_indices_.at(Bit(qbs[1]));
475  input_circ_.add_op<unsigned>(ot, {qb, b});
476  break;
477  }
478  case OpType::Barrier: {
479  input_circ_.add_barrier(qbs);
480  break;
481  }
482  default: {
483  throw BadOpType(
484  "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
485  "measure, reset, collapse, barrier gates. Found gate of different "
486  "type",
487  ot);
488  }
489  }
490  }
491 
492  VertexList bin;
493 
494  BGL_FORALL_VERTICES(v, circ_.dag, DAG) {
495  if (!circ_.detect_boundary_Op(v)) {
496  bin.push_back(v);
497  }
498  }
499 
500  circ_.remove_vertices(
501  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
502 
503  /*
504 this for loop checks all gates in the circuits and try to
505 find the biggest possible sub circuits which contains only
506 CX+Rz Gates. This is done in a way that all qubits get
507 marked if they are currently outside before (pre), in (in)
508 or outside after (post) of a currently constructed box. If
509 a qubit is in the pre state and a gate, which is no valid
510 type in a box should be added, the type is not changed and
511 the gate is added. If a CX or Rz gate should be added to
512 this qubit the type is changes to in and the gate is added
513 to the box. If another gate should be added to a qubit in
514 state, the state stays the same for CX and Rz and they are
515 added to the box. If the added gate has a different type
516 the type of the qubit is changed to post and the gate is
517 added after the box. If a CX or Rz gate should be added to
518 a qubit in the post state, the current box is added and a
519 new box is started to which the current gate is added. All
520 qubits states are reset to pre.
521 */
522  for (const Command& com : input_circ_) {
523  OpType ot = com.get_op_ptr()->get_type();
524  unit_vector_t qbs = com.get_args();
525  switch (ot) {
526  case OpType::Barrier: {
527  add_phase_poly_box();
528 
529  circ_.add_barrier(qbs);
530  break;
531  }
532  case OpType::CX: {
533  unsigned ctrl = qubit_indices_.at(Qubit(qbs[0]));
534  unsigned target = qubit_indices_.at(Qubit(qbs[1]));
535  if ((qubit_types_[ctrl] == QubitType::in) &&
536  (qubit_types_[target] == QubitType::in)) {
537  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
538  } else if (
539  (qubit_types_[ctrl] == QubitType::pre) &&
540  (qubit_types_[target] == QubitType::in)) {
541  qubit_types_[ctrl] = QubitType::in;
542  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
543  } else if (
544  (qubit_types_[ctrl] == QubitType::in) &&
545  (qubit_types_[target] == QubitType::pre)) {
546  qubit_types_[target] = QubitType::in;
547  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
548  } else if (
549  (qubit_types_[ctrl] == QubitType::pre) &&
550  (qubit_types_[target] == QubitType::pre)) {
551  qubit_types_[ctrl] = QubitType::in;
552  qubit_types_[target] = QubitType::in;
553  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
554  } else if (
555  (qubit_types_[ctrl] == QubitType::post) ||
556  (qubit_types_[target] == QubitType::post)) {
557  add_phase_poly_box();
558 
559  qubit_types_[ctrl] = QubitType::in;
560  qubit_types_[target] = QubitType::in;
561  box_circ_.add_op<unsigned>(OpType::CX, {ctrl, target});
562  } else {
563  // no other types should be in this list
564  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
565  }
566  ++box_size_;
567  break;
568  }
569  case OpType::Rz: {
570  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
571  auto angle = com.get_op_ptr()->get_params().at(0);
572  switch (qubit_types_[qb]) {
573  case QubitType::pre: {
574  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
575  qubit_types_[qb] = QubitType::in;
576  break;
577  }
578  case QubitType::in: {
579  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
580  break;
581  }
582  case QubitType::post: {
583  add_phase_poly_box();
584 
585  qubit_types_[qb] = QubitType::in;
586  box_circ_.add_op<unsigned>(OpType::Rz, angle, {qb});
587  break;
588  }
589  default: {
590  // no other types should be in this list
591  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
592  }
593  }
594  break;
595  }
596  case OpType::H:
597  case OpType::Collapse:
598  case OpType::Reset: {
599  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
600  switch (qubit_types_[qb]) {
601  case QubitType::pre: {
602  circ_.add_op<unsigned>(ot, {qb});
603  break;
604  }
605  case QubitType::in: {
606  post_circ_.add_op<unsigned>(ot, {qb});
607  qubit_types_[qb] = QubitType::post;
608  break;
609  }
610  case QubitType::post: {
611  post_circ_.add_op<unsigned>(ot, {qb});
612  break;
613  }
614  default: {
615  // no other types should be in this list
616  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
617  }
618  }
619  break;
620  }
621  case OpType::Measure: {
622  unsigned qb = qubit_indices_.at(Qubit(qbs[0]));
623  unsigned b = bit_indices_.at(Bit(qbs[1]));
624 
625  switch (qubit_types_[qb]) {
626  case QubitType::pre: {
627  circ_.add_op<unsigned>(ot, {qb, b});
628  break;
629  }
630  case QubitType::in: {
631  post_circ_.add_op<unsigned>(ot, {qb, b});
632  qubit_types_[qb] = QubitType::post;
633  break;
634  }
635  case QubitType::post: {
636  post_circ_.add_op<unsigned>(ot, {qb, b});
637  break;
638  }
639  default: {
640  // no other types should be in this list
641  TKET_ASSERT(!"Invalid Qubit Type in Phase Poly Box creation");
642  }
643  }
644  break;
645  }
646  default: {
647  throw BadOpType(
648  "Please rebase with the compiler pass RebaseUFR to only CX, Rz, H, "
649  "measure, reset, collapse, barrier gates. Found gate of different "
650  "type",
651  ot);
652  }
653  }
654  }

657  add_phase_poly_box();
