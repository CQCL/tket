32 static std::map<T, unsigned> count(std::vector<T> xs) {
33  std::map<T, unsigned> cnts;
34  for (const auto& x : xs) {
35  auto it = cnts.find(x);
36  if (it != cnts.end()) {
37  ++(it->second);
38  } else {
39  cnts[x] = 1;
40  }
41  }
42  return cnts;

71 OptEdge PhasedXFrontier::get_beta_edge(unsigned i) const {
72  const auto& [start, end] = intervals_[i];
73  Edge e = start;
74  while (e != end) {
75  Vertex v = circ_.target(e);
76  OpType type = circ_.get_OpType_from_Vertex(v);
77  if (type == OpType::PhasedX || type == OpType::NPhasedX) {
78  return e;
79  }
80  e = circ_.get_next_edge(v, e);
81  }
82  return std::nullopt;
83 }

85 OptEdgeVec PhasedXFrontier::get_all_beta_edges() const {
86  OptEdgeVec beta_edges;
87  std::vector<Vertex> beta_vertices;
88  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
89  OptEdge beta_e = get_beta_edge(i);
90  beta_edges.push_back(beta_e);
91  if (beta_e) {
92  beta_vertices.push_back(circ_.target(*beta_e));
93  }
94  }
95 
96  // for OpType::NPhasedX, we need to reset some betas to zero if they
97  // are shadowed (i.e. if one NPhasedX is behind another one)
98  std::map<Vertex, unsigned> arities = count(beta_vertices);
99  for (auto& e : beta_edges) {
100  if (e) {
101  Vertex v = circ_.target(*e);
102  Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
103  if (arities[v] != op->n_qubits()) {
104  e = std::nullopt;
105  }
106  }
107  }
108  return beta_edges;
109 }

154 Edge PhasedXFrontier::get_interval_end(Edge e) const {
155  Vertex v = circ_.target(e);
156  while (!circ_.detect_final_Op(v) && !is_interval_boundary(v)) {
157  auto p = circ_.get_next_pair(v, e);
158  v = p.first;

200 PhasedXFrontier::PhasedXFrontier(Circuit& circ)
201  : intervals_(),
202  circ_(circ),
203  squasher_(std::make_unique<PhasedXSquasher>(), circ, false) {
204  const unsigned n = circ_.n_qubits();
205  intervals_.resize(n);
206 
207  // initialise intervals
208  const qubit_vector_t all_qbs = circ_.all_qubits();
209  for (unsigned i = 0; i < n; ++i) {
210  Qubit q = all_qbs[i];
211  Vertex v_in = circ_.get_in(q);
212  EdgeVec e_vec = circ_.get_all_out_edges(v_in);
213  TKET_ASSERT(e_vec.size() == 1);
214 
215  // interval [start, end]
216  Edge start = e_vec.front();
217  Edge end = get_interval_end(start);
218  intervals_[i] = {start, end};
219  }
220 }

222 void PhasedXFrontier::squash_interval(unsigned i) {
223  auto& [start_e, end_e] = intervals_[i];
224 
225  // backup edges
226  VertPort start{circ_.source(start_e), circ_.get_source_port(start_e)};
227  VertPort end{circ_.target(end_e), circ_.get_target_port(end_e)};
228 
229  squasher_.squash_between(start_e, end_e);
230 
231  // restore interval edges
232  start_e = circ_.get_nth_out_edge(start.first, start.second);
233  end_e = circ_.get_nth_in_edge(end.first, end.second);
234 }

255 void PhasedXFrontier::skip_global_gates(unsigned n) {
256  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
257  unsigned count = 0;
258  auto& [e, end] = intervals_[i];
259  while (e != end) {
260  Vertex v = circ_.target(e);
261  OpType type = circ_.get_OpType_from_Vertex(v);
262  e = circ_.get_next_edge(v, e);
263  if (type == OpType::NPhasedX ||
264  (circ_.n_qubits() == 1 && type == OpType::PhasedX)) {
265  unsigned in_edges = circ_.n_in_edges_of_type(v, EdgeType::Quantum);
266  unsigned out_edges = circ_.n_out_edges_of_type(v, EdgeType::Quantum);
267  if (in_edges != circ_.n_qubits() || out_edges != circ_.n_qubits()) {
268  throw CircuitInvalidity("Found non-global NPhasedX gate");
269  }
270  ++count;
271  if (count == n) {
272  break;
273  }
274  }
275  }
276  if (count < n) {
277  throw CircuitInvalidity("Did not find expected global gates");
278  }
279  }

293 void PhasedXFrontier::insert_1_phasedx(unsigned i) {
294  std::vector<Expr> betas = get_all_betas();
295  OptEdgeVec edges = get_all_beta_edges();
296  OptVertexVec vertices = get_all_beta_vertices();
297 
298  if (!vertices[i]) {
299  throw CircuitInvalidity("No PhasedX found on qubit " + std::to_string(i));
300  }
301  Expr beta = betas[i];
302 
303  VertexSet bin;
304  EdgeVec in_hole, out_hole;
305  Circuit sub1(circ_.n_qubits());
306  Circuit sub2(circ_.n_qubits());
307 
308  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
309  if (vertices[i]) {
310  Vertex v = vertices[i].value();
311  Edge e = edges[i].value();
312  Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
313  OpType type = op->get_type();
314  Expr new_beta = betas[i] - beta;
315 
316  in_hole.push_back(e);
317  out_hole.push_back(circ_.get_next_edge(v, e));
318 
319  if (!bin.contains(v)) {
320  bin.insert(v);
321 
322  // v's qubits
323  std::vector<unsigned> qubits;
324  for (unsigned j = 0; j < circ_.n_qubits(); ++j) {
325  if (vertices[j] == v) {
326  qubits.push_back(j);
327  }
328  }
329 
330  if (type == OpType::NPhasedX || type == OpType::PhasedX) {
331  Expr alpha = op->get_params()[1];
332  if (!equiv_0(new_beta, 4)) {
333  if (equiv_0(new_beta, 2)) {
334  if (qubits.size() % 2) {
335  sub1.add_phase(-1);
336  }
337  } else {
338  sub2.add_op(type, {new_beta, 0.}, qubits);
339  }
340  }
341  if (!equiv_0(alpha, 2)) {
342  for (unsigned q : qubits) {
343  sub1.add_op<unsigned>(OpType::Rz, {-alpha}, {q});
344  sub2.add_op<unsigned>(OpType::Rz, {alpha}, {q});
345  }
346  }
347  } else {
348  throw BadOpType("Encountered invalid beta angle OpType", type);
349  }
350  }
351  } else {
352  Edge interval_begin = intervals_[i].first;
353  in_hole.push_back(interval_begin);
354  out_hole.push_back(interval_begin);
355  if (!equiv_0(beta, 2)) {
356  sub2.add_op<unsigned>(OpType::PhasedX, {-beta, 0}, {i});
357  } else if (!equiv_0(beta, 4)) {
358  sub2.add_phase(-1);
359  }
360  }
361  }
362 
363  Circuit sub(circ_.n_qubits());
364  sub.append(sub1);
365  sub.add_op<Qubit>(OpType::NPhasedX, {beta, 0}, sub.all_qubits());
366  sub.append(sub2);
367 
368  Subcircuit hole{in_hole, out_hole, bin};
369 
370  // perform substitution
371  BackupIntervals backup = backup_intervals();
372  circ_.substitute(sub, hole, Circuit::VertexDeletion::Yes);
373  restore_intervals(backup);
374 
375  skip_global_gates(1);
376 }

378 void PhasedXFrontier::insert_2_phasedx() {
379  EdgeVec hole_in, hole_out;
380  Circuit sub1(circ_.n_qubits());
381  Circuit sub2(circ_.n_qubits());
382  Circuit sub3(circ_.n_qubits());
383  VertexSet bin;
384 
385  std::vector<Expr> betas = get_all_betas();
386  OptEdgeVec edges = get_all_beta_edges();
387  OptVertexVec vertices = get_all_beta_vertices();
388 
389  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
390  if (vertices[i]) {
391  Vertex v = vertices[i].value();
392  Edge e = edges[i].value();
393  Op_ptr op = circ_.get_Op_ptr_from_Vertex(v);
394  OpType type = op->get_type();
395  hole_in.push_back(e);
396  hole_out.push_back(circ_.get_next_edge(v, e));
397  bin.insert(v);
398  Expr beta = betas[i];
399  if (type == OpType::NPhasedX || type == OpType::PhasedX) {
400  Expr alpha = op->get_params()[1];
401  if (!equiv_0(alpha, 2)) {
402  sub1.add_op<unsigned>(OpType::Rz, {-alpha}, {i});
403  sub3.add_op<unsigned>(OpType::Rz, {alpha}, {i});
404  }
405  }
406  if (!equiv_0(beta, 2)) {
407  sub2.add_op<unsigned>(OpType::Rz, {beta}, {i});
408  } else if (!equiv_0(beta, 4)) {
409  sub2.add_phase(-1);
410  }
411  } else {
412  Edge interval_begin = intervals_[i].first;
413  hole_in.push_back(interval_begin);
414  hole_out.push_back(interval_begin);
415  }
416  }
417 
418  Circuit sub(circ_.n_qubits());
419  sub.append(sub1);
420  sub.add_op(OpType::NPhasedX, {-0.5, 0.5}, sub.all_qubits());
421  sub.append(sub2);
422  sub.add_op(OpType::NPhasedX, {0.5, 0.5}, sub.all_qubits());
423  sub.append(sub3);
424  Subcircuit hole{hole_in, hole_out, bin};
425 
426  BackupIntervals backup = backup_intervals();
427  circ_.substitute(sub, hole, Circuit::VertexDeletion::Yes);
428  restore_intervals(backup);
429 
430  skip_global_gates(2);
431 }
