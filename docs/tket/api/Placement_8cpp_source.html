27 void fill_partial_mapping(
28  const qubit_vector_t& current_qubits,
29  std::map<Qubit, Node>& partial_mapping) {
30  unsigned up_nu = 0;
31  for (Qubit q : current_qubits) {
32  if (partial_mapping.find(q) == partial_mapping.end()) {
33  partial_mapping.insert({q, Node(Placement::unplaced_reg(), up_nu)});
34  up_nu++;
35  }
36  }
37 
38  if (std::any_of(
39  partial_mapping.begin(), partial_mapping.end(),
40  [&current_qubits](auto x) {
41  return std::find(
42  current_qubits.begin(), current_qubits.end(), x.first) ==
43  current_qubits.end();
44  })) {
45  tket_log()->warn(
46  "Placement map has some Qubit not present in the Circuit.");
47  }
48 }

53 bool Placement::place(
54  Circuit& circ_, std::shared_ptr<unit_bimaps_t> compilation_map) const {
55  if (circ_.n_qubits() > this->architecture_.n_nodes()) {
56  throw std::invalid_argument(
57  "Circuit has more qubits than Architecture has nodes.");
58  }
59  std::map<Qubit, Node> map_ = this->get_placement_map(circ_);
60  return this->place_with_map(circ_, map_, compilation_map);
61 }

71 std::map<Qubit, Node> Placement::get_placement_map(const Circuit& circ_) const {
72  std::vector<std::map<Qubit, Node>> all_maps =
73  this->get_all_placement_maps(circ_, 1);
74 
75  // basic handling to avoid segmentation faults, as placement method may not
76  // return any valid map
77  auto it = all_maps.begin();
78  if (it != all_maps.end()) {
79  std::map<Qubit, Node> map = *it;
80  fill_partial_mapping(circ_.all_qubits(), map);
81  return map;
82  } else {
83  return {};
84  }

87 std::vector<std::map<Qubit, Node>> Placement::get_all_placement_maps(
88  const Circuit& circ_, unsigned /*matches*/) const {
89  std::map<Qubit, Node> placement;
90  qubit_vector_t to_place;
91  std::vector<Node> placed;
92 
93  // Find which/if any qubits need placing
94  for (const Qubit& q : circ_.all_qubits()) {
95  Node n(q);
96  if (!this->architecture_.node_exists(n)) {
97  to_place.push_back(n);
98  } else {
99  placed.push_back(n);
100  // if already placed, make sure qubit retains placement
101  placement.insert({n, n});
102  }
103  }
104  // avoid doing std::set_difference unless qubits need to be placed
105  unsigned n_placed = to_place.size();
106  if (n_placed > 0) {
107  std::vector<Node> difference,
108  architecture_nodes = this->architecture_.get_all_nodes_vec();
109  std::set_difference(
110  architecture_nodes.begin(), architecture_nodes.end(), placed.begin(),
111  placed.end(), std::inserter(difference, difference.begin()));
112  // should always be enough remaining qubits to assign unplaced qubits to
113  if (difference.size() < n_placed) {
114  throw std::invalid_argument(
115  "There are more unplaced Qubit in Circuit then there are free Nodes "
116  "in Architecture.");
117  }
118  for (unsigned i = 0; i < n_placed; i++) {
119  // naively assign each qubit to some free node
120  placement.insert({to_place[i], difference[i]});
121  }
122  }
123  return {placement};
124 }

141  j["maximum_pattern_gates"] = cast_placer->get_maximum_pattern_gates();
142  j["maximum_pattern_depth"] = cast_placer->get_maximum_pattern_depth();
143  j["characterisation"] = cast_placer->get_characterisation();
144  } else if (
145  std::shared_ptr<GraphPlacement> cast_placer =
146  std::dynamic_pointer_cast<GraphPlacement>(placement_ptr)) {
147  j["type"] = "GraphPlacement";

157 void from_json(const nlohmann::json& j, Placement::Ptr& placement_ptr) {
158  std::string classname = j.at("type").get<std::string>();
159  Architecture arc = j.at("architecture").get<Architecture>();
160  if (classname == "GraphPlacement") {
161  unsigned matches = j.at("matches").get<unsigned>();
162  unsigned timeout = j.at("timeout").get<unsigned>();
163  unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
164  unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
165  placement_ptr = std::make_shared<GraphPlacement>(
166  arc, matches, timeout, max_pattern_gates, max_pattern_depth);
167  } else if (classname == "LinePlacement") {
168  unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
169  unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
170  std::shared_ptr<LinePlacement> lp = std::make_shared<LinePlacement>(
171  arc, max_pattern_gates, max_pattern_depth);
172  placement_ptr = lp;
173  } else if (classname == "NoiseAwarePlacement") {
174  unsigned matches = j.at("matches").get<unsigned>();
175  unsigned timeout = j.at("timeout").get<unsigned>();
176  unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
177  unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
178  DeviceCharacterisation characterisation =
179  j.at("characterisation").get<DeviceCharacterisation>();
180  avg_node_errors_t empty_node_errors = {};
181  avg_readout_errors_t empty_readout_errors = {};
182  avg_link_errors_t empty_link_errors = {};
183  std::shared_ptr<NoiseAwarePlacement> nap =
184  std::make_shared<NoiseAwarePlacement>(
185  arc, empty_node_errors, empty_link_errors, empty_readout_errors,
186  matches, timeout, max_pattern_gates, max_pattern_depth);
187  nap->set_characterisation(characterisation);
188  placement_ptr = nap;
189  } else {
190  placement_ptr = std::make_shared<Placement>(arc);
191  }
192 }
