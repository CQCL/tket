25void fill_partial_mapping(
26 const qubit_vector_t& current_qubits,
27 std::map<Qubit, Node>& partial_mapping) {
28 unsigned up_nu = 0;
29 for (Qubit q : current_qubits) {
30 if (partial_mapping.find(q) == partial_mapping.end()) {
31 partial_mapping.insert({q, Node(Placement::unplaced_reg(), up_nu)});
32 up_nu++;
33 }
34 }
35
36 if (std::any_of(
37 partial_mapping.begin(), partial_mapping.end(),
38 [&current_qubits](auto x) {
39 return std::find(
40 current_qubits.begin(), current_qubits.end(), x.first) ==
41 current_qubits.end();
42 })) {
43 tket_log()->warn(
44 "Placement map has some Qubit not present in the Circuit.");
45 }
46}

51bool Placement::place(
52 Circuit& circ_, std::shared_ptr<unit_bimaps_t> compilation_map) const {
53 if (circ_.n_qubits() > this->architecture_.n_nodes()) {
54 throw std::invalid_argument(
55 "Circuit has more qubits than Architecture has nodes.");
56 }
57 std::map<Qubit, Node> map_ = this->get_placement_map(circ_);
58 return this->place_with_map(circ_, map_, compilation_map);
59}

61bool Placement::place_with_map(
62 Circuit& circ_, std::map<Qubit, Node>& map_,
63 std::shared_ptr<unit_bimaps_t> compilation_map) {
64 bool changed = circ_.rename_units(map_);
65 changed |= update_maps(compilation_map, map_, map_);
66 return changed;
67}

69std::map<Qubit, Node> Placement::get_placement_map(const Circuit& circ_) const {
70 std::vector<std::map<Qubit, Node>> all_maps =
71 this->get_all_placement_maps(circ_, 1);
72
73 // basic handling to avoid segmentation faults, as placement method may not
74 // return any valid map
75 auto it = all_maps.begin();
76 if (it != all_maps.end()) {
77 std::map<Qubit, Node> map = *it;
78 fill_partial_mapping(circ_.all_qubits(), map);
79 return map;
80 } else {
81 return {};
82 }
83}

85std::vector<std::map<Qubit, Node>> Placement::get_all_placement_maps(
86 const Circuit& circ_, unsigned /*matches*/) const {
87 std::map<Qubit, Node> placement;
88 qubit_vector_t to_place;
89 std::vector<Node> placed;
90
91 // Find which/if any qubits need placing
92 for (const Qubit& q : circ_.all_qubits()) {
93 Node n(q);
94 if (!this->architecture_.node_exists(n)) {
95 to_place.push_back(n);
96 } else {
97 placed.push_back(n);
98 // if already placed, make sure qubit retains placement
99 placement.insert({n, n});
100 }
101 }
102 // avoid doing std::set_difference unless qubits need to be placed
103 unsigned n_placed = to_place.size();
104 if (n_placed > 0) {
105 std::vector<Node> difference,
106 architecture_nodes = this->architecture_.get_all_nodes_vec();
107 std::set_difference(
108 architecture_nodes.begin(), architecture_nodes.end(), placed.begin(),
109 placed.end(), std::inserter(difference, difference.begin()));
110 // should always be enough remaining qubits to assign unplaced qubits to
111 if (difference.size() < n_placed) {
112 throw std::invalid_argument(
113 "There are more unplaced Qubit in Circuit then there are free Nodes "
114 "in Architecture.");
115 }
116 for (unsigned i = 0; i < n_placed; i++) {
117 // naively assign each qubit to some free node
118 placement.insert({to_place[i], difference[i]});
119 }
120 }
121 return {placement};
122}

124void to_json(nlohmann::json& j, const Placement::Ptr& placement_ptr) {
125 // n.b. due to inheritance NoiseAwarePlacement and LinePlacement
126 // need to be cast before GraphPlacement
127 j["architecture"] = placement_ptr->get_architecture_ref();
128 if (std::shared_ptr<LinePlacement> cast_placer =
129 std::dynamic_pointer_cast<LinePlacement>(placement_ptr)) {
130 j["type"] = "LinePlacement";
131 j["maximum_pattern_gates"] = cast_placer->get_maximum_pattern_gates();
132 j["maximum_pattern_depth"] = cast_placer->get_maximum_pattern_depth();
133 } else if (
134 std::shared_ptr<NoiseAwarePlacement> cast_placer =
135 std::dynamic_pointer_cast<NoiseAwarePlacement>(placement_ptr)) {
136 j["type"] = "NoiseAwarePlacement";
137 j["matches"] = cast_placer->get_maximum_matches();
138 j["timeout"] = cast_placer->get_timeout();
139 j["maximum_pattern_gates"] = cast_placer->get_maximum_pattern_gates();
140 j["maximum_pattern_depth"] = cast_placer->get_maximum_pattern_depth();
141 j["characterisation"] = cast_placer->get_characterisation();
142 } else if (
143 std::shared_ptr<GraphPlacement> cast_placer =
144 std::dynamic_pointer_cast<GraphPlacement>(placement_ptr)) {
145 j["type"] = "GraphPlacement";
146 j["matches"] = cast_placer->get_maximum_matches();
147 j["timeout"] = cast_placer->get_timeout();
148 j["maximum_pattern_gates"] = cast_placer->get_maximum_pattern_gates();
149 j["maximum_pattern_depth"] = cast_placer->get_maximum_pattern_depth();
150 } else {
151 j["type"] = "Placement";
152 }
153}

155void from_json(const nlohmann::json& j, Placement::Ptr& placement_ptr) {
156 std::string classname = j.at("type").get<std::string>();
157 Architecture arc = j.at("architecture").get<Architecture>();
158 if (classname == "GraphPlacement") {
159 unsigned matches = j.at("matches").get<unsigned>();
160 unsigned timeout = j.at("timeout").get<unsigned>();
161 unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
162 unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
163 placement_ptr = std::make_shared<GraphPlacement>(
164 arc, matches, timeout, max_pattern_gates, max_pattern_depth);
165 } else if (classname == "LinePlacement") {
166 unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
167 unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
168 std::shared_ptr<LinePlacement> lp = std::make_shared<LinePlacement>(
169 arc, max_pattern_gates, max_pattern_depth);
170 placement_ptr = lp;
171 } else if (classname == "NoiseAwarePlacement") {
172 unsigned matches = j.at("matches").get<unsigned>();
173 unsigned timeout = j.at("timeout").get<unsigned>();
174 unsigned max_pattern_gates = j.at("maximum_pattern_gates").get<unsigned>();
175 unsigned max_pattern_depth = j.at("maximum_pattern_depth").get<unsigned>();
176 DeviceCharacterisation characterisation =
177 j.at("characterisation").get<DeviceCharacterisation>();
178 avg_node_errors_t empty_node_errors = {};
179 avg_readout_errors_t empty_readout_errors = {};
180 avg_link_errors_t empty_link_errors = {};
181 std::shared_ptr<NoiseAwarePlacement> nap =
182 std::make_shared<NoiseAwarePlacement>(
183 arc, empty_node_errors, empty_link_errors, empty_readout_errors,
184 matches, timeout, max_pattern_gates, max_pattern_depth);
185 nap->set_characterisation(characterisation);
186 placement_ptr = nap;
187 } else {
188 placement_ptr = std::make_shared<Placement>(arc);
189 }
190}
