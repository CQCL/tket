27template <typename T>
28static bool auto_implication(const T&, const Predicate& other) {
29 try {
30 (void)dynamic_cast<const T&>(other);
31 } catch (const std::bad_cast&) {
32 throw IncorrectPredicate(
33 "Cannot compare predicates of different subclasses");
34 }
35 return true;
36}
37
38template <typename T>
39static PredicatePtr auto_meet(const T&, const Predicate& other) {
40 try {
41 (void)dynamic_cast<const T&>(other);
42 PredicatePtr pp = std::make_shared<T>();
43 return pp;
44 } catch (const std::bad_cast&) {
45 throw IncorrectPredicate(
46 "Cannot compare predicates of different subclasses");
47 }
48}

87bool GateSetPredicate::verify(const Circuit& circ) const {
88 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
89 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
90 OpDesc desc = op->get_desc();
91 if (desc.is_meta()) continue;
92 OpType type = op->get_type();
93 while (type == OpType::Conditional) {
94 op = static_cast<const Conditional&>(*op).get_op();
95 type = op->get_type();
96 }
97 if (type == OpType::Phase) continue;
98 if (!find_in_set(type, allowed_types_)) return false;
99 }
100 return true;
101}

171static bool fast_feed_forward_helper(
172 const Command& com, unit_set_t& unset_bits) {
173 // Allows conditionals from unset_bits
174 // Encountering a measurement removed the bits from unset_bits
175 // Returns whether or not a feed-forward conditional is found
176 // Applies recursively for CircBoxes
177 if (com.get_op_ptr()->get_type() == OpType::Conditional) {
178 const Conditional& cond =
179 static_cast<const Conditional&>(*com.get_op_ptr());
180 unit_vector_t all_args = com.get_args();
181 unit_vector_t::iterator arg_it = all_args.begin();
182 for (unsigned i = 0; i < cond.get_width(); ++i) {
183 if (unset_bits.find(*arg_it) == unset_bits.end()) return false;
184 ++arg_it;
185 }
186 unit_vector_t new_args = {arg_it, all_args.end()};
187 Command new_com = {cond.get_op(), new_args};
188 return fast_feed_forward_helper(new_com, unset_bits);
189 } else if (
190 com.get_op_ptr()->get_type() == OpType::CircBox ||
191 com.get_op_ptr()->get_type() == OpType::CustomGate) {
192 const Box& box = static_cast<const Box&>(*com.get_op_ptr());
193 unit_map_t interface;
194 unit_set_t inner_set;
195 unsigned i = 0;
196 for (const Bit& b : com.get_bits()) {
197 Bit inner_bit(i);
198 interface.insert({Bit(i), b});
199 if (unset_bits.find(b) != unset_bits.end()) {
200 inner_set.insert(inner_bit);
201 }
202 ++i;
203 }
204 for (const Command& c : *box.to_circuit()) {
205 if (!fast_feed_forward_helper(c, inner_set)) return false;
206 }
207 for (const std::pair<const UnitID, UnitID>& pair : interface) {
208 if (inner_set.find(pair.first) == inner_set.end())
209 unset_bits.erase(pair.second);
210 }
211 } else if (com.get_op_ptr()->get_type() == OpType::Measure) {
212 unset_bits.erase(com.get_args().at(1));
213 }
214 return true;
215}

217bool NoFastFeedforwardPredicate::verify(const Circuit& circ) const {
218 if (circ.n_bits() == 0) return true;
219 bit_vector_t all_bits = circ.all_bits();
220 unit_set_t unset_bits = {all_bits.begin(), all_bits.end()};
221 for (const Command& com : circ) {
222 if (!fast_feed_forward_helper(com, unset_bits)) return false;
223 }
224 return true;
225}

338bool ConnectivityPredicate::implies(const Predicate& other) const {
339 try {
340 const ConnectivityPredicate& other_c =
341 dynamic_cast<const ConnectivityPredicate&>(other);
342 const Architecture& arc1 = arch_;
343 const Architecture& arc2 = other_c.arch_;
344 // Check that all nodes in arc1 are in arc2:
345 for (const Node& n : arc1.get_all_nodes_vec()) {
346 if (!arc2.node_exists(n)) {
347 return false;
348 }
349 }
350 // Collect all edges in arc1
351 for (auto [n1, n2] : arc1.get_all_edges_vec()) {
352 if (!arc2.edge_exists(n1, n2) && !arc2.edge_exists(n2, n1)) {
353 return false; // if not in second architecture, return false
354 }
355 }
356 return true;
357 } catch (const std::bad_cast&) {
358 throw IncorrectPredicate(
359 "Cannot compare predicates of different subclasses");
360 }
361}

363PredicatePtr ConnectivityPredicate::meet(const Predicate& other) const {
364 try {
365 const ConnectivityPredicate& other_c =
366 dynamic_cast<const ConnectivityPredicate&>(other);
367 const Architecture& arc1 = arch_;
368 const Architecture& arc2 = other_c.arch_;
369 std::vector<std::pair<Node, Node>> new_edges;
370 // Collect all edges in arc1 which are also in arc2
371 for (auto [n1, n2] : arc1.get_all_edges_vec()) {
372 if (arc2.edge_exists(n1, n2)) {
373 new_edges.push_back({n1, n2});
374 new_edges.push_back({n2, n1});
375 }
376 }
377 Architecture arc3(new_edges);
378 PredicatePtr pp = std::make_shared<ConnectivityPredicate>(arc3);
379 return pp;
380 } catch (const std::bad_cast&) {
381 throw IncorrectPredicate(
382 "Cannot compare predicates of different subclasses");
383 }
384}

651bool NormalisedTK2Predicate::verify(const Circuit& circ) const {
652 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
653 Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
654 while (op->get_type() == OpType::Conditional) {
655 op = static_cast<const Conditional&>(*op).get_op();
656 }
657 if (op->get_type() == OpType::TK2) {
658 auto params = op->get_params();
659 TKET_ASSERT(params.size() == 3);
660 if (!in_weyl_chamber({params[0], params[1], params[2]})) {
661 return false;
662 }
663 }
664 }
665 return true;
666}
