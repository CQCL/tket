27 template <typename T>
28 static bool auto_implication(const T&, const Predicate& other) {
29  try {
30  (void)dynamic_cast<const T&>(other);
31  } catch (const std::bad_cast&) {
32  throw IncorrectPredicate(
33  "Cannot compare predicates of different subclasses");
34  }
35  return true;
36 }

86 bool GateSetPredicate::verify(const Circuit& circ) const {
87  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
88  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
89  OpDesc desc = op->get_desc();
90  if (desc.is_meta()) continue;
91  OpType type = op->get_type();
92  if (type == OpType::Conditional) {
93  const Conditional& cond = static_cast<const Conditional&>(*op);
94  type = cond.get_op()->get_type();
95  }
96  if (type == OpType::Phase) continue;
97  if (!find_in_set(type, allowed_types_)) return false;
98  }
99  return true;
100 }

170 static bool fast_feed_forward_helper(
171  const Command& com, unit_set_t& unset_bits) {
172  // Allows conditionals from unset_bits
173  // Encountering a measurement removed the bits from unset_bits
174  // Returns whether or not a feed-forward conditional is found
175  // Applies recursively for CircBoxes
176  if (com.get_op_ptr()->get_type() == OpType::Conditional) {
177  const Conditional& cond =
178  static_cast<const Conditional&>(*com.get_op_ptr());
179  unit_vector_t all_args = com.get_args();
180  unit_vector_t::iterator arg_it = all_args.begin();
181  for (unsigned i = 0; i < cond.get_width(); ++i) {
182  if (unset_bits.find(*arg_it) == unset_bits.end()) return false;
183  ++arg_it;
184  }
185  unit_vector_t new_args = {arg_it, all_args.end()};
186  Command new_com = {cond.get_op(), new_args};
187  return fast_feed_forward_helper(new_com, unset_bits);
188  } else if (
189  com.get_op_ptr()->get_type() == OpType::CircBox ||
190  com.get_op_ptr()->get_type() == OpType::CustomGate) {
191  const Box& box = static_cast<const Box&>(*com.get_op_ptr());
192  unit_map_t interface;
193  unit_set_t inner_set;
194  unsigned i = 0;
195  for (const Bit& b : com.get_bits()) {
196  Bit inner_bit(i);
197  interface.insert({Bit(i), b});
198  if (unset_bits.find(b) != unset_bits.end()) {
199  inner_set.insert(inner_bit);
200  }
201  ++i;
202  }
203  for (const Command& c : *box.to_circuit()) {
204  if (!fast_feed_forward_helper(c, inner_set)) return false;
205  }
206  for (const std::pair<const UnitID, UnitID>& pair : interface) {
207  if (inner_set.find(pair.first) == inner_set.end())
208  unset_bits.erase(pair.second);
209  }
210  } else if (com.get_op_ptr()->get_type() == OpType::Measure) {
211  unset_bits.erase(com.get_args().at(1));
212  }
213  return true;
214 }

216 bool NoFastFeedforwardPredicate::verify(const Circuit& circ) const {
217  if (circ.n_bits() == 0) return true;
218  bit_vector_t all_bits = circ.all_bits();
219  unit_set_t unset_bits = {all_bits.begin(), all_bits.end()};
220  for (const Command& com : circ) {
221  if (!fast_feed_forward_helper(com, unset_bits)) return false;
222  }
223  return true;
224 }

337 bool ConnectivityPredicate::implies(const Predicate& other) const {
338  try {
339  const ConnectivityPredicate& other_c =
340  dynamic_cast<const ConnectivityPredicate&>(other);
341  const Architecture& arc1 = arch_;
342  const Architecture& arc2 = other_c.arch_;
343  // Check that all nodes in arc1 are in arc2:
344  for (const Node& n : arc1.get_all_nodes_vec()) {
345  if (!arc2.node_exists(n)) {
346  return false;
347  }
348  }
349  // Collect all edges in arc1
350  for (auto [n1, n2] : arc1.get_all_edges_vec()) {
351  if (!arc2.edge_exists(n1, n2) && !arc2.edge_exists(n2, n1)) {
352  return false; // if not in second architecture, return false
353  }
354  }
355  return true;
356  } catch (const std::bad_cast&) {
357  throw IncorrectPredicate(
358  "Cannot compare predicates of different subclasses");
359  }
360 }

362 PredicatePtr ConnectivityPredicate::meet(const Predicate& other) const {
363  try {
364  const ConnectivityPredicate& other_c =
365  dynamic_cast<const ConnectivityPredicate&>(other);
366  const Architecture& arc1 = arch_;
367  const Architecture& arc2 = other_c.arch_;
368  std::vector<std::pair<Node, Node>> new_edges;
369  // Collect all edges in arc1 which are also in arc2
370  for (auto [n1, n2] : arc1.get_all_edges_vec()) {
371  if (arc2.edge_exists(n1, n2)) {
372  new_edges.push_back({n1, n2});
373  new_edges.push_back({n2, n1});
374  }
375  }
376  Architecture arc3(new_edges);
377  PredicatePtr pp = std::make_shared<ConnectivityPredicate>(arc3);
378  return pp;
379  } catch (const std::bad_cast&) {
