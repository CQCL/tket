27 template <typename T>
28 static bool auto_implication(const T&, const Predicate& other) {
29  try {
30  (void)dynamic_cast<const T&>(other);
31  } catch (const std::bad_cast&) {
32  throw IncorrectPredicate(
33  "Cannot compare predicates of different subclasses");
34  }
35  return true;
36 }

85 bool GateSetPredicate::verify(const Circuit& circ) const {
86  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
87  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
88  OpDesc desc = op->get_desc();
89  if (desc.is_meta()) continue;
90  OpType type = op->get_type();
91  if (type == OpType::Conditional) {
92  const Conditional& cond = static_cast<const Conditional&>(*op);
93  type = cond.get_op()->get_type();
94  }
95  if (type == OpType::Phase) continue;
96  if (!find_in_set(type, allowed_types_)) return false;
97  }
98  return true;
99 }

169 static bool fast_feed_forward_helper(
170  const Command& com, unit_set_t& unset_bits) {
171  // Allows conditionals from unset_bits
172  // Encountering a measurement removed the bits from unset_bits
173  // Returns whether or not a feed-forward conditional is found
174  // Applies recursively for CircBoxes
175  if (com.get_op_ptr()->get_type() == OpType::Conditional) {
176  const Conditional& cond =
177  static_cast<const Conditional&>(*com.get_op_ptr());
178  unit_vector_t all_args = com.get_args();
179  unit_vector_t::iterator arg_it = all_args.begin();
180  for (unsigned i = 0; i < cond.get_width(); ++i) {
181  if (unset_bits.find(*arg_it) == unset_bits.end()) return false;
182  ++arg_it;
183  }
184  unit_vector_t new_args = {arg_it, all_args.end()};
185  Command new_com = {cond.get_op(), new_args};
186  return fast_feed_forward_helper(new_com, unset_bits);
187  } else if (
188  com.get_op_ptr()->get_type() == OpType::CircBox ||
189  com.get_op_ptr()->get_type() == OpType::CustomGate) {
190  const Box& box = static_cast<const Box&>(*com.get_op_ptr());
191  unit_map_t interface;
192  unit_set_t inner_set;
193  unsigned i = 0;
194  for (const Bit& b : com.get_bits()) {
195  Bit inner_bit(i);
196  interface.insert({Bit(i), b});
197  if (unset_bits.find(b) != unset_bits.end()) {
198  inner_set.insert(inner_bit);
199  }
200  ++i;
201  }
202  for (const Command& c : *box.to_circuit()) {
203  if (!fast_feed_forward_helper(c, inner_set)) return false;
204  }
205  for (const std::pair<const UnitID, UnitID>& pair : interface) {
206  if (inner_set.find(pair.first) == inner_set.end())
207  unset_bits.erase(pair.second);
208  }
209  } else if (com.get_op_ptr()->get_type() == OpType::Measure) {
210  unset_bits.erase(com.get_args().at(1));
211  }
212  return true;
213 }

215 bool NoFastFeedforwardPredicate::verify(const Circuit& circ) const {
216  if (circ.n_bits() == 0) return true;
217  bit_vector_t all_bits = circ.all_bits();
218  unit_set_t unset_bits = {all_bits.begin(), all_bits.end()};
219  for (const Command& com : circ) {
220  if (!fast_feed_forward_helper(com, unset_bits)) return false;
221  }
222  return true;
223 }

336 bool ConnectivityPredicate::implies(const Predicate& other) const {
337  try {
338  const ConnectivityPredicate& other_c =
339  dynamic_cast<const ConnectivityPredicate&>(other);
340  const Architecture& arc1 = arch_;
341  const Architecture& arc2 = other_c.arch_;
342  // Check that all nodes in arc1 are in arc2:
343  for (const Node& n : arc1.get_all_nodes_vec()) {
344  if (!arc2.node_exists(n)) {
345  return false;
346  }
347  }
348  // Collect all edges in arc1
349  for (auto [n1, n2] : arc1.get_all_edges_vec()) {
350  if (!arc2.edge_exists(n1, n2) && !arc2.edge_exists(n2, n1)) {
351  return false; // if not in second architecture, return false
352  }
353  }
354  return true;
355  } catch (const std::bad_cast&) {
356  throw IncorrectPredicate(
357  "Cannot compare predicates of different subclasses");
358  }
359 }

361 PredicatePtr ConnectivityPredicate::meet(const Predicate& other) const {
362  try {
363  const ConnectivityPredicate& other_c =
364  dynamic_cast<const ConnectivityPredicate&>(other);
365  const Architecture& arc1 = arch_;
366  const Architecture& arc2 = other_c.arch_;
367  std::vector<std::pair<Node, Node>> new_edges;
368  // Collect all edges in arc1 which are also in arc2
369  for (auto [n1, n2] : arc1.get_all_edges_vec()) {
370  if (arc2.edge_exists(n1, n2)) {
371  new_edges.push_back({n1, n2});
372  new_edges.push_back({n2, n1});
373  }
374  }
375  Architecture arc3(new_edges);
376  PredicatePtr pp = std::make_shared<ConnectivityPredicate>(arc3);
377  return pp;
378  } catch (const std::bad_cast&) {
379  throw IncorrectPredicate(
380  "Cannot compare predicates of different subclasses");

632 bool NormalisedTK2Predicate::verify(const Circuit& circ) const {
633  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
634  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
635  bool conditional = op->get_type() == OpType::Conditional;
636  if (conditional) {
