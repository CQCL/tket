35 // Rebase a 0- or 1-qubit unitary operation
36 static Circuit rebase_op(
37  Gate_ptr op,
38  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
39  tk1_replacement) {
40  OpType type = op->get_type();
41  if (type == OpType::Phase) {
42  Circuit replacement(0);
43  replacement.add_phase(op->get_params()[0]);
44  return replacement;
45  } else {
46  std::vector<Expr> tk1_angles = op->get_tk1_angles();
47  Circuit replacement =
48  tk1_replacement(tk1_angles[0], tk1_angles[1], tk1_angles[2]);
49  remove_redundancies().apply(replacement);
50  replacement.add_phase(tk1_angles[3]);
51  return replacement;
52  }
53 }

55 static bool standard_rebase(
56  Circuit& circ, const OpTypeSet& allowed_gates,
57  const Circuit& cx_replacement,
58  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
59  tk1_replacement) {
60  bool success = false;
61  VertexSet bin;
62  for (const Vertex& v : circ.all_vertices()) {
63  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
64  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
65  if (n_qubits <= 1) continue;
66  bool conditional = op->get_type() == OpType::Conditional;
67  if (conditional) {
68  const Conditional& cond = static_cast<const Conditional&>(*op);
69  op = cond.get_op();
70  }
71  OpType type = op->get_type();
72  if (allowed_gates.find(type) != allowed_gates.end() || type == OpType::CX ||
73  type == OpType::Barrier)
74  continue;
75  // need to convert
76  Circuit replacement = CX_circ_from_multiq(op);
77  if (conditional) {
78  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
79  } else {
80  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
81  }
82  bin.insert(v);
83  success = true;
84  }
85  if (allowed_gates.find(OpType::CX) == allowed_gates.end()) {
86  const Op_ptr cx_op = get_op_ptr(OpType::CX);
87  success = circ.substitute_all(cx_replacement, cx_op) | success;
88  }
89  for (const Vertex& v : circ.all_vertices()) {
90  if (bin.contains(v)) continue;
91  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) > 1) continue;
92  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
93  bool conditional = op->get_type() == OpType::Conditional;
94  if (conditional) {
95  const Conditional& cond = static_cast<const Conditional&>(*op);
96  op = cond.get_op();
97  }
98  OpType type = op->get_type();
99  if (!is_gate_type(type) || is_projective_type(type) ||
100  allowed_gates.find(type) != allowed_gates.end())
101  continue;
102  // need to convert
103  Circuit replacement = rebase_op(as_gate_ptr(op), tk1_replacement);
104  if (conditional) {
105  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
106  } else {
107  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
108  }
109  bin.insert(v);
110  success = true;
111  }
112  circ.remove_vertices(
113  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
114  return success;
115 }

117 static bool standard_rebase_via_tk2(
118  Circuit& circ, const OpTypeSet& allowed_gates,
119  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
120  tk1_replacement,
121  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
122  tk2_replacement) {
123  bool success = false;
124  VertexSet bin;
125  // 1. Replace all multi-qubit gates outside the target gateset to TK2.
126  for (const Vertex& v : circ.all_vertices()) {
127  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
128  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
129  if (n_qubits <= 1) continue;
130  bool conditional = op->get_type() == OpType::Conditional;
131  if (conditional) {
132  const Conditional& cond = static_cast<const Conditional&>(*op);
133  op = cond.get_op();
134  }
135  OpType type = op->get_type();
136  if (allowed_gates.contains(type) || type == OpType::TK2 ||
137  type == OpType::Barrier)
138  continue;
139  // need to convert
140  Circuit replacement = TK2_circ_from_multiq(op);
141  // Find replacement Circuit for all TK2 gates
142  VertexSet TK2_bin;
143  for (const Vertex& u : replacement.all_vertices()) {
144  Op_ptr op = circ.get_Op_ptr_from_Vertex(u);
145  TKET_ASSERT(op->get_type() != OpType::Conditional);
146  if (op->get_type() == OpType::TK2) {
147  std::vector<Expr> params = op->get_params();
148  TKET_ASSERT(params.size() == 3);
149  Circuit u_replacement =
150  tk2_replacement(params[0], params[1], params[2]);
151  replacement.substitute(u_replacement, u, Circuit::VertexDeletion::No);
152  TK2_bin.insert(u);
153  }
154  }
155  Transforms::squash_1qb_to_tk1().apply(replacement);
156  remove_redundancies().apply(replacement);
157  replacement.remove_vertices(
158  TK2_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
159  if (conditional) {
160  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
161  } else {
162  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
163  }
164  bin.insert(v);
165  success = true;
166  }
167 
168  // 2. Replace 0- and 1-qubit gates by converting to TK1 and replacing.
169  for (const Vertex& v : circ.all_vertices()) {
170  if (bin.contains(v)) continue;
171  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) > 1) continue;
172  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
173  bool conditional = op->get_type() == OpType::Conditional;
174  if (conditional) {
175  const Conditional& cond = static_cast<const Conditional&>(*op);
176  op = cond.get_op();
177  }
178  OpType type = op->get_type();
179  if (!is_gate_type(type) || is_projective_type(type) ||
180  allowed_gates.contains(type))
181  continue;
182  // need to convert
183  Circuit replacement = rebase_op(as_gate_ptr(op), tk1_replacement);
184  if (conditional) {
185  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
186  } else {
187  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
188  }
189  bin.insert(v);
190  success = true;
191  }
192  circ.remove_vertices(
193  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
194  return success;
195 }

197 Transform rebase_factory(
198  const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
199  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
200  tk1_replacement) {

219 Transform rebase_tket() {
220  std::function<Circuit(const Expr&, const Expr&, const Expr&)> tk1_to_tk1 =
221  [](const Expr& alpha, const Expr& beta, const Expr& gamma) {
222  Circuit c(1);
223  c.add_op<unsigned>(OpType::TK1, {alpha, beta, gamma}, {0});
224  return c;
225  };
226  return rebase_factory({OpType::CX, OpType::TK1}, CircPool::CX(), tk1_to_tk1);
227 }
