35 // Rebase a 0- or 1-qubit unitary operation
36 static Circuit rebase_op(
37  Gate_ptr op,
38  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
39  tk1_replacement) {
40  OpType type = op->get_type();
41  if (type == OpType::Phase) {
42  Circuit replacement(0);
43  replacement.add_phase(op->get_params()[0]);
44  return replacement;
45  } else {
46  std::vector<Expr> tk1_angles = op->get_tk1_angles();
47  Circuit replacement =
48  tk1_replacement(tk1_angles[0], tk1_angles[1], tk1_angles[2]);
49  remove_redundancies().apply(replacement);
50  replacement.add_phase(tk1_angles[3]);
51  return replacement;
52  }
53 }

55 static bool standard_rebase(
56  Circuit& circ, const OpTypeSet& allowed_gates,
57  const Circuit& cx_replacement,
58  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
59  tk1_replacement) {
60  bool success = false;
61  VertexSet bin;
62  for (const Vertex& v : circ.all_vertices()) {
63  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
64  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
65  if (n_qubits <= 1) continue;
66  bool conditional = op->get_type() == OpType::Conditional;
67  if (conditional) {
68  const Conditional& cond = static_cast<const Conditional&>(*op);
69  op = cond.get_op();
70  }
71  OpType type = op->get_type();
72  if (allowed_gates.find(type) != allowed_gates.end() || type == OpType::CX ||
73  type == OpType::Barrier)
74  continue;
75  // need to convert
76  Circuit replacement = CX_circ_from_multiq(op);
77  if (conditional) {
78  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
79  } else {
80  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
81  }
82  bin.insert(v);
83  success = true;
84  }
85  if (allowed_gates.find(OpType::CX) == allowed_gates.end()) {
86  const Op_ptr cx_op = get_op_ptr(OpType::CX);
87  success = circ.substitute_all(cx_replacement, cx_op) | success;
88  }
89  for (const Vertex& v : circ.all_vertices()) {
90  if (bin.contains(v)) continue;
91  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) > 1) continue;
92  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
93  bool conditional = op->get_type() == OpType::Conditional;
94  if (conditional) {
95  const Conditional& cond = static_cast<const Conditional&>(*op);
96  op = cond.get_op();
97  }
98  OpType type = op->get_type();
99  if (!is_gate_type(type) || is_projective_type(type) ||
100  allowed_gates.find(type) != allowed_gates.end())
101  continue;
102  // need to convert
103  Circuit replacement = rebase_op(as_gate_ptr(op), tk1_replacement);
104  if (conditional) {
105  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
106  } else {
107  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
108  }
109  bin.insert(v);
110  success = true;
111  }
112  circ.remove_vertices(
113  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
114  return success;
115 }

117 static bool standard_rebase_via_tk2(
118  Circuit& circ, const OpTypeSet& allowed_gates,
119  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
120  tk1_replacement,
121  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
122  tk2_replacement) {
123  bool success = false;
124  VertexSet bin;
125 
126  // 1. Replace all multi-qubit gates outside the target gateset to TK2.
127  for (const Vertex& v : circ.all_vertices()) {
128  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
129  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
130  if (n_qubits <= 1) continue;
131  bool conditional = op->get_type() == OpType::Conditional;
132  if (conditional) {
133  const Conditional& cond = static_cast<const Conditional&>(*op);
134  op = cond.get_op();
135  }
136  OpType type = op->get_type();
137  if (allowed_gates.contains(type) || type == OpType::TK2 ||
138  type == OpType::Barrier)
139  continue;
140  // need to convert
141  Circuit replacement = TK2_circ_from_multiq(op);
142  if (conditional) {
143  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
144  } else {
145  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
146  }
147  bin.insert(v);
148  success = true;
149  }
150 
151  // 2. If TK2 is not in the target gateset, decompose TK2 gates.
152  if (!allowed_gates.contains(OpType::TK2)) {
153  for (const Vertex& v : circ.all_vertices()) {
154  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
155  bool conditional = op->get_type() == OpType::Conditional;
156  if (conditional) {
157  const Conditional& cond = static_cast<const Conditional&>(*op);
158  op = cond.get_op();
159  }
160  if (op->get_type() == OpType::TK2) {
161  std::vector<Expr> params = op->get_params();
162  TKET_ASSERT(params.size() == 3);
163  Circuit replacement = tk2_replacement(params[0], params[1], params[2]);
164  remove_redundancies().apply(replacement);
165  if (conditional) {
166  circ.substitute_conditional(
167  replacement, v, Circuit::VertexDeletion::No);
168  } else {
169  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
170  }
171  bin.insert(v);
172  success = true;
173  }
174  }
175  }
176 
177  // 3. Replace 0- and 1-qubit gates by converting to TK1 and replacing.
178  for (const Vertex& v : circ.all_vertices()) {
179  if (bin.contains(v)) continue;
180  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) > 1) continue;
181  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
182  bool conditional = op->get_type() == OpType::Conditional;
183  if (conditional) {
184  const Conditional& cond = static_cast<const Conditional&>(*op);
185  op = cond.get_op();
186  }
187  OpType type = op->get_type();
188  if (!is_gate_type(type) || is_projective_type(type) ||
189  allowed_gates.contains(type))
190  continue;
191  // need to convert
192  Circuit replacement = rebase_op(as_gate_ptr(op), tk1_replacement);
193  if (conditional) {
194  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
195  } else {
196  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
197  }
198  bin.insert(v);
199  success = true;
200  }
201 
202  circ.remove_vertices(
203  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
204  return success;
205 }

207 Transform rebase_factory(
208  const OpTypeSet& allowed_gates, const Circuit& cx_replacement,
209  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
210  tk1_replacement) {
211  return Transform([=](Circuit& circ) {
212  return standard_rebase(

229 Transform rebase_tket() {
230  std::function<Circuit(const Expr&, const Expr&, const Expr&)> tk1_to_tk1 =
231  [](const Expr& alpha, const Expr& beta, const Expr& gamma) {
232  Circuit c(1);
233  c.add_op<unsigned>(OpType::TK1, {alpha, beta, gamma}, {0});
234  return c;
235  };
236  return rebase_factory({OpType::CX, OpType::TK1}, CircPool::CX(), tk1_to_tk1);
237 }
