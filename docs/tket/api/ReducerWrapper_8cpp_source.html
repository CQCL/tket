32 bool ReducerInterface::other_vertex_reduction_can_be_skipped_by_symmetry(
33  const std::set<VertexWSM>& other_domain, const DomainsAccessor& accessor,
34  VertexWSM this_vertex, VertexWSM other_vertex) {
35  // If this other pv was already assigned in a previous node
36  // (i.e., its domain was the same as now), then the reducer
37  // already reduced this domain when that node was reduced
38  // (and, we could only have reached this current node by moving down,
39  // so our current domain is a subset of that one).
40  //
41  // Otherwise, if pv1, pv2 both had domains reduced to size 1 in
42  // the current node, we only need to perform one reduction.
43  // We could decide by knowing which vertex was assigned earlier,
44  // but that information is not available (although it could be deduced
45  // using some labour).
46  // Instead, we decide by using vertex numbers.
47  return other_domain.size() == 1 &&
48  (!accessor.domain_created_in_current_node(other_vertex) ||
49  other_vertex < this_vertex);
50 }

61 ReductionResult ReducerWrapper::reduce(
62  DomainsAccessor& accessor, std::set<VertexWSM>& work_set) {
63  auto result = ReductionResult::SUCCESS;
64  for (const std::vector<std::pair<VertexWSM, VertexWSM>>& new_assignments =
65  accessor.get_new_assignments();
66  m_number_of_processed_assignments < new_assignments.size();) {
67  result = m_reducer.reduce(
68  new_assignments[m_number_of_processed_assignments], accessor, work_set);
69 
70  // We should increment this BEFORE breaking out of the loop.
71  // We've finished processing this particular assignment,
72  // so don't want to waste time reprocessing it
73  // next time "reduce" is called.
74  ++m_number_of_processed_assignments;
75  if (result != ReductionResult::SUCCESS) {
76  break;
77  }
78  }
79  return result;
80 }
