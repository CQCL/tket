31 static void detach_vertex(
32  Circuit &circuit, const Vertex &vertex,
33  VertexDetachmentInfo &detachmentInfo) {
34  detachmentInfo.detachedVertices.emplace_back(vertex);
35  auto &vertexPredecessors = detachmentInfo.detachedVertexPredecessors;
36  const auto &newVertexPredecessors = circuit.get_predecessors(vertex);
37  vertexPredecessors.insert(
38  vertexPredecessors.cend(), newVertexPredecessors.cbegin(),
39  newVertexPredecessors.cend());
40  circuit.remove_vertex(
41  vertex, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
42 }

59 static bool try_detach_identity(
60  Circuit &circuit, const Vertex &vertex,
61  VertexDetachmentInfo &detachmentInfo) {
62  auto vertex_operator = circuit.get_Op_ptr_from_Vertex(vertex);
63  if (auto phase = vertex_operator->is_identity()) {
64  circuit.add_phase(phase.value());
65  detach_vertex(circuit, vertex, detachmentInfo);
66  return true;
67  }
68  return false;
69 }

76 static bool
77 vertex_is_succeeded_only_by_z_basis_measurements_with_which_it_commutes(
78  const Circuit &circuit, const Vertex &vertex) {
79  // If vertex has classical out edges, no need to continue
80  if (circuit.n_out_edges_of_type(vertex, EdgeType::Classical) != 0)
81  return false;
82  auto successors = circuit.get_successors(vertex);
83  std::vector<port_t> ports(successors.size());
84  std::iota(ports.begin(), ports.end(), 0);
85 
86  return std::all_of(ports.cbegin(), ports.cend(), [&](port_t port) {
87  return vertex_is_a_measurement(circuit, successors[port]) &&
88  circuit.commutes_with_basis(
89  vertex, Pauli::Z, PortType::Source, port);
90  });
91 }

104 static bool port_ordering_is_compatible(
105  Circuit &circuit, const Vertex &vertex, const Vertex &successor) {
106  const Op_ptr vertex_op = circuit.get_Op_ptr_from_Vertex(vertex);
107  // Vertex port must be symmetrically equivalent to successor port for any edge
108  // between them Examples:
109  // CX[0,1]==CX[0,1] passes test because ports line up
110  // CX[0,1]==CX[1,0] fails test because ports don't line up and CX is not
111  // invariant with respect to exchange of qubits CZ[0,1]==CZ[1,0] passes test
112  // because CZ is invariant with respect to exchange of qubits
113  for (const Edge &in : circuit.get_in_edges(successor)) {
114  auto source_port = circuit.get_source_port(in);
115  auto target_port = circuit.get_target_port(in);
116  if (not vertex_op->has_symmetry(source_port, target_port)) {
117  return false;
118  }
119  }
120  return true;
121 }

123 static bool preliminary_vertex_successor_checks_pass(
124  Circuit &circuit, const Vertex &vertex) {
125  // check that the classical edges match up correctly
126  if (circuit.n_in_edges_of_type(vertex, EdgeType::Boolean) != 0) return false;
127 
128  // check that both the vertex and its successor have each other and only each
129  // other
130  auto successors = circuit.get_successors(vertex);
131  if (successors.size() != 1) return false;
132  auto successor = successors[0];
133  if (circuit.get_predecessors(successor).size() != 1) return false;
134 
135  // check that successor has adjoint
136  if (circuit.get_Op_ptr_from_Vertex(successor)->get_desc().is_oneway()) {
137  return false;
138  }
139 
140  return port_ordering_is_compatible(circuit, vertex, successor);
141 }

143 static bool try_detach_both_because_successor_is_adjoint(
144  Circuit &circuit, Vertex const &vertex, Vertex const &successor,
145  VertexDetachmentInfo &detachmentInfo) {
146  const Op_ptr successor_op = circuit.get_Op_ptr_from_Vertex(successor);
147  const Op_ptr vertex_op = circuit.get_Op_ptr_from_Vertex(vertex);
148  if (*vertex_op == *successor_op->dagger()) {
149  detach_vertex_and_successor(circuit, vertex, successor, detachmentInfo);
150  return true;
151  }
152  return false;
153 }

155 static bool try_join_rotations_and_detach_successor(
156  Circuit &circuit, Vertex const &vertex, Vertex const &successor,
157  VertexDetachmentInfo &detachmentInfo) {
158  const Op_ptr successor_op = circuit.get_Op_ptr_from_Vertex(successor);
159  const OpDesc successor_op_descriptor = successor_op->get_desc();
160  const Op_ptr vertex_op = circuit.get_Op_ptr_from_Vertex(vertex);
161  const OpDesc vertex_op_descriptor = vertex_op->get_desc();
162 
163  // check vertex and successor are same rotation type
164  if (not(vertex_op_descriptor.is_rotation() &&
165  vertex_op_descriptor.type() == successor_op_descriptor.type())) {
166  return false;
167  }
168 
169  // replace vertex with combined rotation
170  auto expr1 = vertex_op->get_params()[0];
171  auto expr2 = successor_op->get_params()[0];
172  Op_ptr op_new = get_op_ptr(
173  vertex_op_descriptor.type(), {expr1 + expr2},
174  circuit.get_in_edges(successor).size());
175  circuit.dag[vertex].op = op_new;
176 
177  // detach successor only (this adds vertex to list of vertices to check again,
178  // so will be removed later if is identity)
179  detach_vertex(circuit, successor, detachmentInfo);
180  return true;
181 }

195 static bool try_detach_vertex_and_successor(
196  Circuit &circuit, const Vertex &vertex,
197  VertexDetachmentInfo &detachmentInfo) {
198  if (not preliminary_vertex_successor_checks_pass(circuit, vertex))
199  return false;
200  auto successor = circuit.get_successors(vertex)[0];
201  if (try_detach_both_because_successor_is_adjoint(
202  circuit, vertex, successor, detachmentInfo)) {
203  return true;
204  }
205  if (try_join_rotations_and_detach_successor(
206  circuit, vertex, successor, detachmentInfo)) {
207  return true;
208  }
209  return false;
210 }

212 static bool is_apriori_not_detachable(
213  const Circuit &circuit, Vertex const &vertex) {
214  const OpDesc op_descriptor =
215  circuit.get_Op_ptr_from_Vertex(vertex)->get_desc();
216 
217  return (not op_descriptor.is_gate()) or // not a gate
218  circuit.n_out_edges(vertex) ==
219  0 or // vertex is boundary or already detached
220  circuit.n_in_edges(vertex) == 0; // vertex is boundary
221 }

223 static bool try_detach_vertex(
224  Circuit &circuit, const Vertex &vertex,
225  VertexDetachmentInfo &detachmentInfo) {
226  if (is_apriori_not_detachable(circuit, vertex)) {
227  return false;
228  }
229  if (try_detach_single_vertex(circuit, vertex, detachmentInfo)) {
230  return true;
231  }
232  if (try_detach_vertex_and_successor(circuit, vertex, detachmentInfo)) {
233  return true;
234  }
235  return false;

238 static VertexDetachmentInfo detach_vertices_if_redundant(
239  Circuit &circuit, const VertexVec &vertices) {
240  VertexDetachmentInfo detachmentInfo;
241  for (const auto &vertex : vertices) {
242  try_detach_vertex(circuit, vertex, detachmentInfo);
243  }
244  return detachmentInfo;

247 static VertexSet detach_any_redundant_vertices(Circuit &circuit) {
248  VertexSet bin;
249  VertexVec verticesToCheckForRemoval = circuit.vertices_in_order();
250  while (!verticesToCheckForRemoval.empty()) {
251  auto detachmentInfo =
252  detach_vertices_if_redundant(circuit, verticesToCheckForRemoval);
253  bin.insert(
254  detachmentInfo.detachedVertices.cbegin(),
255  detachmentInfo.detachedVertices.cend());
256  std::swap(
257  verticesToCheckForRemoval, detachmentInfo.detachedVertexPredecessors);
258  }
259  return bin;
260 }

262 // this method annihilates all primitives next to each other (accounting for
263 // previous annihilation)
264 // also removes redundant non-classically controlled Z basis gates before a z
265 // basis measurement so that eg. -H-X-X-H- always annihilates to -----
266 bool redundancy_removal(Circuit &circuit) {
267  VertexSet bin = detach_any_redundant_vertices(circuit);
268  circuit.remove_vertices(
269  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
270  return !bin.empty();
271 }
