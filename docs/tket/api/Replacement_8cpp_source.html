31 Circuit multi_controlled_to_2q(
32  const Op_ptr op, const std::optional<OpType>& two_q_type) {
33  unsigned n_qubits = op->n_qubits();
34  OpType optype = op->get_type();
35  Circuit c(n_qubits);
36  switch (optype) {
37  case OpType::CnRy:
38  c = CircPool::CnRy_normal_decomp(op, n_qubits);
39  break;
40  case OpType::CnX:
41  case OpType::CnZ:
42  case OpType::CnY:
43  if (n_qubits >= 6 && n_qubits <= 50) {
44  // CnU_linear_depth_decomp performs better in this case
45  OpType target_type =
46  (optype == OpType::CnX)
47  ? OpType::X
48  : ((optype == OpType::CnZ) ? OpType::Z : OpType::Y);
49  Eigen::Matrix2cd target_u =
50  GateUnitaryMatrix::get_unitary(target_type, 1, {});
51  c = CircPool::CnU_linear_depth_decomp(n_qubits - 1, target_u);
52  } else {
53  if (optype == OpType::CnZ) {
54  c.add_op<unsigned>(OpType::H, {n_qubits - 1});
55  } else if (optype == OpType::CnY) {
56  c.add_op<unsigned>(OpType::Sdg, {n_qubits - 1});
57  }
58  c.append(CircPool::CnX_normal_decomp(n_qubits - 1));
59  if (optype == OpType::CnZ) {
60  c.add_op<unsigned>(OpType::H, {n_qubits - 1});
61  } else if (optype == OpType::CnY) {
62  c.add_op<unsigned>(OpType::S, {n_qubits - 1});
63  }
64  }
65  break;
66  default:
67  throw BadOpType("The operation is not multi-controlled", optype);
68  }
69 
70  if (two_q_type == std::nullopt) {
71  return c;
72  } else if (two_q_type.value() == OpType::CX) {
73  decompose_multi_qubits_CX().apply(c);
74  } else if (two_q_type.value() == OpType::TK2) {
75  decompose_multi_qubits_TK2().apply(c);
76  } else {
77  throw BadOpType("The target 2-q gate can only be CX or TK2", optype);
78  }
79  return c;
80 }

82 Circuit TK2_circ_from_multiq(const Op_ptr op) {
83  OpDesc desc = op->get_desc();
84  if (!desc.is_gate())
85  throw BadOpType(
86  "Can only build replacement circuits for basic gates", desc.type());
87  switch (desc.type()) {
88  case OpType::CnRy:
89  case OpType::CnX:
90  case OpType::CnZ:
91  case OpType::CnY:
92  // TODO We should be able to do better than this.
93  return multi_controlled_to_2q(op, OpType::TK2);
94  default:
95  return with_TK2(as_gate_ptr(op));
96  }
97 }

115 Circuit CX_ZX_circ_from_op(const Op_ptr op) {
116  OpDesc desc = op->get_desc();
117  if (!desc.is_gate())
118  throw BadOpType(
119  "Can only build replacement circuits for basic gates", desc.type());
120  switch (desc.type()) {
121  case OpType::Phase: {
122  Circuit replacement(0);
123  replacement.add_phase(op->get_params()[0]);
124  return replacement;
125  }
126  case OpType::Z: {
127  Circuit replacement(1);
128  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
129  replacement.add_phase(0.5);
130  return replacement;
131  }
132  case OpType::X: {
133  Circuit replacement(1);
134  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
135  replacement.add_phase(0.5);
136  return replacement;
137  }
138  case OpType::Y: {
139  Circuit replacement(1);
140  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
141  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
142  replacement.add_phase(-0.5);
143  return replacement;
144  }
145  case OpType::S: {
146  Circuit replacement(1);
147  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
148  replacement.add_phase(0.25);
149  return replacement;
150  }
151  case OpType::Sdg: {
152  Circuit replacement(1);
153  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
154  replacement.add_phase(-0.25);
155  return replacement;
156  }
157  case OpType::T: {
158  Circuit replacement(1);
159  replacement.add_op<unsigned>(OpType::Rz, 0.25, {0});
160  replacement.add_phase(0.125);
161  return replacement;
162  }
163  case OpType::Tdg: {
164  Circuit replacement(1);
165  replacement.add_op<unsigned>(OpType::Rz, -0.25, {0});
166  replacement.add_phase(-0.125);
167  return replacement;
168  }
169  case OpType::V: {
170  Circuit replacement(1);
171  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
172  return replacement;
173  }
174  case OpType::Vdg: {
175  Circuit replacement(1);
176  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
177  return replacement;
178  }
179  case OpType::SX: {
180  Circuit replacement(1);
181  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
182  replacement.add_phase(0.25);
183  return replacement;
184  }
185  case OpType::SXdg: {
186  Circuit replacement(1);
187  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
188  replacement.add_phase(-0.25);
189  return replacement;
190  }
191  case OpType::H: {
192  Circuit replacement(1);
193  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
194  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
195  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
196  replacement.add_phase(0.5);
197  return replacement;
198  }
199  case OpType::Ry: {
200  Expr angle = op->get_params()[0];
201  Circuit replacement(1);
202  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
203  replacement.add_op<unsigned>(OpType::Rx, angle, {0});
204  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
205  return replacement;
206  }
207  case OpType::Rx:
208  case OpType::Rz:
209  case OpType::Measure:
210  case OpType::Collapse: {
211  Circuit replacement(1);
212  replacement.add_op<unsigned>(op, {0});
213  return replacement;
214  }
215  case OpType::U3: {
216  Circuit replacement(1);
217  Expr angle_z1 = op->get_params()[2];
218  Expr angle_y = op->get_params()[0];
219  Expr angle_z2 = op->get_params()[1];
220  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
221  replacement.add_op<unsigned>(OpType::Rx, angle_y, {0});
222  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
223  replacement.add_phase((angle_z1 + angle_z2) / 2);
224  return replacement;
225  }
226  case OpType::U2: {
227  Circuit replacement(1);
228  Expr angle_z1 = op->get_params()[1];
229  Expr angle_z2 = op->get_params()[0];
230  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
231  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
232  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
233  replacement.add_phase((angle_z1 + angle_z2) / 2);
234  return replacement;
235  }
236  case OpType::U1: {
237  Circuit replacement(1);
238  Expr angle = op->get_params()[0];
239  replacement.add_op<unsigned>(OpType::Rz, angle, {0});
240  replacement.add_phase(angle / 2);
241  return replacement;
242  }
243  case OpType::PhasedX: {
244  Circuit replacement(1);
245  Expr theta = op->get_params()[0];
246  Expr phi = op->get_params()[1];
247  replacement.add_op<unsigned>(OpType::Rz, -phi, {0});
248  replacement.add_op<unsigned>(OpType::Rx, theta, {0});
249  replacement.add_op<unsigned>(OpType::Rz, phi, {0});
250  return replacement;
251  }
252  case OpType::CX: {
253  Circuit replacement(2);
254  replacement.add_op<unsigned>(op, {0, 1});
255  return replacement;
256  }
257  case OpType::TK2:
258  case OpType::CY:
259  case OpType::CZ:
260  case OpType::CH:
261  case OpType::CV:
262  case OpType::CVdg:
263  case OpType::CSX:
264  case OpType::CSXdg:
265  case OpType::CRz:
266  case OpType::CRx:
267  case OpType::CRy:
268  case OpType::CU1:
269  case OpType::CU3:
270  case OpType::PhaseGadget:
271  case OpType::CCX:
272  case OpType::SWAP:
273  case OpType::CSWAP:
274  case OpType::ECR:
275  case OpType::ISWAP:
276  case OpType::XXPhase:
277  case OpType::XXPhase3:
278  case OpType::ZZMax:
279  case OpType::ZZPhase:
280  case OpType::YYPhase:
281  case OpType::CnRy:
282  case OpType::CnX:
283  case OpType::ESWAP:
284  case OpType::FSim:
285  case OpType::Sycamore:
286  case OpType::ISWAPMax:
287  case OpType::BRIDGE: {
288  Circuit replacement = CX_circ_from_multiq(op);
289  decompose_ZX().apply(replacement);
290  return replacement;
291  }
292  case OpType::TK1: {
293  Circuit replacement(1);
294  std::vector<Expr> params = op->get_params();
295  replacement.add_op<unsigned>(OpType::Rz, params[2], {0});
296  replacement.add_op<unsigned>(OpType::Rx, params[1], {0});
297  replacement.add_op<unsigned>(OpType::Rz, params[0], {0});
298  return replacement;
299  }
300  default:
301  throw BadOpType("Cannot find replacement circuit", desc.type());
302  }
