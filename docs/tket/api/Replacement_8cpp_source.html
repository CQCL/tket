31 Circuit multi_controlled_to_2q(
32  const Op_ptr op, const std::optional<OpType>& two_q_type) {
33  unsigned n_qubits = op->n_qubits();
34  OpType optype = op->get_type();
35  Circuit c(n_qubits);
36  switch (optype) {
37  case OpType::CnRy:
38  c = CircPool::CnRy_normal_decomp(op, n_qubits);
39  break;
40  case OpType::CnX:
41  case OpType::CnZ:
42  case OpType::CnY:
43  if (n_qubits >= 6 && n_qubits <= 50) {
44  // CnU_linear_depth_decomp performs better in this case
45  OpType target_type =
46  (optype == OpType::CnX)
47  ? OpType::X
48  : ((optype == OpType::CnZ) ? OpType::Z : OpType::Y);
49  Eigen::Matrix2cd target_u =
50  GateUnitaryMatrix::get_unitary(target_type, 1, {});
51  c = CircPool::CnU_linear_depth_decomp(n_qubits - 1, target_u);
52  } else {
53  if (optype == OpType::CnZ) {
54  c.add_op<unsigned>(OpType::H, {n_qubits - 1});
55  } else if (optype == OpType::CnY) {
56  c.add_op<unsigned>(OpType::Sdg, {n_qubits - 1});
57  }
58  c.append(CircPool::CnX_normal_decomp(n_qubits - 1));
59  if (optype == OpType::CnZ) {
60  c.add_op<unsigned>(OpType::H, {n_qubits - 1});
61  } else if (optype == OpType::CnY) {
62  c.add_op<unsigned>(OpType::S, {n_qubits - 1});
63  }
64  }
65  break;
66  default:
67  throw BadOpType("The operation is not multi-controlled", optype);
68  }
69 
70  if (two_q_type == std::nullopt) {
71  return c;
72  } else if (two_q_type.value() == OpType::CX) {
73  decompose_multi_qubits_CX().apply(c);
74  } else if (two_q_type.value() == OpType::TK2) {
75  decompose_multi_qubits_TK2().apply(c);
76  } else {
77  throw BadOpType("The target 2-q gate can only be CX or TK2", optype);
78  }
79  return c;
80 }

82 Circuit TK2_circ_from_multiq(const Op_ptr op) {
83  OpDesc desc = op->get_desc();
84  std::vector<Expr> params = op->get_params();
85  if (!desc.is_gate())
86  throw BadOpType(
87  "Can only build replacement circuits for basic gates", desc.type());
88  switch (desc.type()) {
89  case OpType::CnRy:
90  case OpType::CnX:
91  case OpType::CnZ:
92  case OpType::CnY:
93  // TODO We should be able to do better than this.
94  return multi_controlled_to_2q(op, OpType::TK2);
95  case OpType::XXPhase:
96  return CircPool::XXPhase_using_TK2(params[0]);
97  case OpType::YYPhase:
98  return CircPool::YYPhase_using_TK2(params[0]);
99  case OpType::ZZPhase:
100  return CircPool::ZZPhase_using_TK2(params[0]);
101  default:
102  return with_TK2(as_gate_ptr(op));
103  }
104 }

122 Circuit CX_ZX_circ_from_op(const Op_ptr op) {
123  OpDesc desc = op->get_desc();
124  if (!desc.is_gate())
125  throw BadOpType(
126  "Can only build replacement circuits for basic gates", desc.type());
127  switch (desc.type()) {
128  case OpType::Phase: {
129  Circuit replacement(0);
130  replacement.add_phase(op->get_params()[0]);
131  return replacement;
132  }
133  case OpType::Z: {
134  Circuit replacement(1);
135  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
136  replacement.add_phase(0.5);
137  return replacement;
138  }
139  case OpType::X: {
140  Circuit replacement(1);
141  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
142  replacement.add_phase(0.5);
143  return replacement;
144  }
145  case OpType::Y: {
146  Circuit replacement(1);
147  replacement.add_op<unsigned>(OpType::Rz, 1., {0});
148  replacement.add_op<unsigned>(OpType::Rx, 1., {0});
149  replacement.add_phase(-0.5);
150  return replacement;
151  }
152  case OpType::S: {
153  Circuit replacement(1);
154  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
155  replacement.add_phase(0.25);
156  return replacement;
157  }
158  case OpType::Sdg: {
159  Circuit replacement(1);
160  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
161  replacement.add_phase(-0.25);
162  return replacement;
163  }
164  case OpType::T: {
165  Circuit replacement(1);
166  replacement.add_op<unsigned>(OpType::Rz, 0.25, {0});
167  replacement.add_phase(0.125);
168  return replacement;
169  }
170  case OpType::Tdg: {
171  Circuit replacement(1);
172  replacement.add_op<unsigned>(OpType::Rz, -0.25, {0});
173  replacement.add_phase(-0.125);
174  return replacement;
175  }
176  case OpType::V: {
177  Circuit replacement(1);
178  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
179  return replacement;
180  }
181  case OpType::Vdg: {
182  Circuit replacement(1);
183  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
184  return replacement;
185  }
186  case OpType::SX: {
187  Circuit replacement(1);
188  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
189  replacement.add_phase(0.25);
190  return replacement;
191  }
192  case OpType::SXdg: {
193  Circuit replacement(1);
194  replacement.add_op<unsigned>(OpType::Rx, -0.5, {0});
195  replacement.add_phase(-0.25);
196  return replacement;
197  }
198  case OpType::H: {
199  Circuit replacement(1);
200  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
201  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
202  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
203  replacement.add_phase(0.5);
204  return replacement;
205  }
206  case OpType::Ry: {
207  Expr angle = op->get_params()[0];
208  Circuit replacement(1);
209  replacement.add_op<unsigned>(OpType::Rz, -0.5, {0});
210  replacement.add_op<unsigned>(OpType::Rx, angle, {0});
211  replacement.add_op<unsigned>(OpType::Rz, 0.5, {0});
212  return replacement;
213  }
214  case OpType::Rx:
215  case OpType::Rz:
216  case OpType::Measure:
217  case OpType::Collapse: {
218  Circuit replacement(1);
219  replacement.add_op<unsigned>(op, {0});
220  return replacement;
221  }
222  case OpType::U3: {
223  Circuit replacement(1);
224  Expr angle_z1 = op->get_params()[2];
225  Expr angle_y = op->get_params()[0];
226  Expr angle_z2 = op->get_params()[1];
227  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
228  replacement.add_op<unsigned>(OpType::Rx, angle_y, {0});
229  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
230  replacement.add_phase((angle_z1 + angle_z2) / 2);
231  return replacement;
232  }
233  case OpType::U2: {
234  Circuit replacement(1);
235  Expr angle_z1 = op->get_params()[1];
236  Expr angle_z2 = op->get_params()[0];
237  replacement.add_op<unsigned>(OpType::Rz, angle_z1 - 0.5, {0});
238  replacement.add_op<unsigned>(OpType::Rx, 0.5, {0});
239  replacement.add_op<unsigned>(OpType::Rz, angle_z2 + 0.5, {0});
240  replacement.add_phase((angle_z1 + angle_z2) / 2);
241  return replacement;
242  }
243  case OpType::U1: {
244  Circuit replacement(1);
245  Expr angle = op->get_params()[0];
246  replacement.add_op<unsigned>(OpType::Rz, angle, {0});
247  replacement.add_phase(angle / 2);
248  return replacement;
249  }
250  case OpType::PhasedX: {
251  Circuit replacement(1);
252  Expr theta = op->get_params()[0];
253  Expr phi = op->get_params()[1];
254  replacement.add_op<unsigned>(OpType::Rz, -phi, {0});
255  replacement.add_op<unsigned>(OpType::Rx, theta, {0});
256  replacement.add_op<unsigned>(OpType::Rz, phi, {0});
257  return replacement;
258  }
259  case OpType::CX: {
260  Circuit replacement(2);
261  replacement.add_op<unsigned>(op, {0, 1});
262  return replacement;
263  }
264  case OpType::TK2:
265  case OpType::CY:
266  case OpType::CZ:
267  case OpType::CH:
268  case OpType::CV:
269  case OpType::CVdg:
270  case OpType::CSX:
271  case OpType::CSXdg:
272  case OpType::CRz:
273  case OpType::CRx:
274  case OpType::CRy:
275  case OpType::CU1:
276  case OpType::CU3:
277  case OpType::PhaseGadget:
278  case OpType::CCX:
279  case OpType::SWAP:
280  case OpType::CSWAP:
281  case OpType::ECR:
282  case OpType::ISWAP:
283  case OpType::XXPhase:
284  case OpType::XXPhase3:
285  case OpType::ZZMax:
286  case OpType::ZZPhase:
287  case OpType::YYPhase:
288  case OpType::CnRy:
289  case OpType::CnX:
290  case OpType::ESWAP:
291  case OpType::FSim:
292  case OpType::Sycamore:
293  case OpType::ISWAPMax:
294  case OpType::BRIDGE: {
295  Circuit replacement = CX_circ_from_multiq(op);
296  decompose_ZX().apply(replacement);
297  return replacement;
298  }
299  case OpType::TK1: {
300  Circuit replacement(1);
301  std::vector<Expr> params = op->get_params();
302  replacement.add_op<unsigned>(OpType::Rz, params[2], {0});
303  replacement.add_op<unsigned>(OpType::Rx, params[1], {0});
304  replacement.add_op<unsigned>(OpType::Rz, params[0], {0});
305  return replacement;
306  }
307  default:
308  throw BadOpType("Cannot find replacement circuit", desc.type());
309  }
