25 static Expr atan2_bypi(const Expr &a, const Expr &b) {
26  std::optional<double> va = eval_expr(a);
27  std::optional<double> vb = eval_expr(b);
28  if (va && vb) {
29  double vva = va.value();
30  double vvb = vb.value();
31  if (std::abs(vva) < EPS && std::abs(vvb) < EPS) return Expr(0.);
32  return atan2(vva, vvb) / PI;
33  } else {
34  // Convert symbolic zero to 0. This is a workaround for
35  // https://github.com/symengine/symengine/issues/1875 .
36  Expr a1 = a, b1 = b;
37  if (a1 == SymEngine::zero) a1 = 0.;
38  if (b1 == SymEngine::zero) b1 = 0.;
39  return SymEngine::div(SymEngine::atan2(a1, b1), SymEngine::pi);
40  }
41 }

43 static Expr acos_bypi(const Expr &a) {
44  std::optional<double> va = eval_expr(a);
45  if (va) {
46  double vva = va.value();
47  // avoid undefined values due to rounding
48  if (vva >= 1.) return 0.;
49  if (vva <= -1.) return 1.;
50  return acos(vva) / PI;
51  } else {
52  return SymEngine::div(SymEngine::acos(a), SymEngine::pi);
53  }
54 }

56 // SymEngine::div does not always spot when the numerator is a scalar multiple
57 // of the denominator, for example in expressions like (-a + b) / (a - b) where
58 // a and b are symbolic. This function picks out the common cases.
59 static Expr expr_div(const Expr &num, const Expr &den) {
60  if (approx_0(SymEngine::expand(num - den))) return 1;
61  if (approx_0(SymEngine::expand(num + den))) return -1;
62  return SymEngine::div(num, den);
63 }

65 static std::tuple<Expr, Expr, Expr> xyx_angles_from_coeffs(
66  const Expr &s, const Expr &i, const Expr &j, const Expr &k) {
67  // Handle exceptional cases first.
68  bool s_zero = approx_0(s);
69  bool s_one = approx_0(s - 1);
70  bool i_zero = approx_0(i);
71  bool i_one = approx_0(i - 1);
72  bool j_zero = approx_0(j);
73  bool j_one = approx_0(j - 1);
74  bool k_zero = approx_0(k);
75  bool k_one = approx_0(k - 1);
76  if (i_zero && j_zero && k_zero) {
77  if (s_one)
78  return {0, 0, 0};
79  else
80  return {2, 0, 0};
81  }
82  if (s_zero && j_zero && k_zero) {
83  if (i_one)
84  return {1, 0, 0};
85  else
86  return {3, 0, 0};
87  }
88  if (s_zero && i_zero && k_zero) {
89  if (j_one)
90  return {0, 1, 0};
91  else
92  return {0, 3, 0};
93  }
94  if (s_zero && i_zero && j_zero) {
95  if (k_one)
96  return {3, 1, 0};
97  else
98  return {1, 1, 0};
99  }
100  if (s_zero && i_zero) return {-2 * atan2_bypi(k, j), 1, 0};
101  if (s_zero && j_zero) return {0, 2 * atan2_bypi(k, i), 1};
102  if (s_zero && k_zero) return {0.5, 2 * atan2_bypi(j, i), 0.5};
103  if (i_zero && j_zero) return {-0.5, 2 * atan2_bypi(k, s), 0.5};
104  if (i_zero && k_zero) return {0, 2 * atan2_bypi(j, s), 0};
105  if (j_zero && k_zero) return {2 * atan2_bypi(i, s), 0, 0};
106 
107  // This is a (partial) workaround for
108  // https://github.com/symengine/symengine/issues/1806
109  // (since it avoids the use of atan2 with proportional symbolics).
110  // Explanation: When the quaternion is of the form
111  // A + uA i + B j -/+ uB k
112  // where u is a pure number (with no free symbols) but A and B are symbolic,
113  // symengine wrongly simplifies atan2(uA, A) and atan2(uB, B), ignoring the
114  // symbols (whose sign ought to affect the result). So the general formula
115  // below does not work. We therefore handle these cases differently, noting
116  // that the quaternion factorizes as either
117  // (A + Bj) (1 + ui) or (1 + ui) (A + Bj)
118  // -- that is, an Rx followed by an Ry or vice versa.
119  // We will analyse the first case; the second is similar.
120  // Let alpha = atan(u). Note that -pi/2 < alpha < pi/2, so cos(alpha) > 0.
121  // The quaternion is then
122  // (A/cos(alpha) + B/cos(alpha) j) (cos(alpha) + sin(alpha) i)
123  // So we can take the angle of the Rx as 2*alpha, and the angle of the Ry as
124  // 2 * atan2(B, A).
125  // Finally, note that u must be well-defined because we have already dealt
126  // with all cases where s = 0.
127  if (approx_0(SymEngine::expand(i * j + s * k))) {
128  Expr u = expr_div(i, s);
129  if (SymEngine::free_symbols(u).empty()) {
130  Expr a = SymEngine::atan(u);
131  Expr q = 2 * atan2_bypi(j, s);
132  Expr two_a_by_pi = SymEngine::div(2 * a, SymEngine::pi);
133  return std::tuple<Expr, Expr, Expr>(two_a_by_pi, q, 0);
134  }
135  } else if (approx_0(SymEngine::expand(i * j - s * k))) {
136  Expr u = expr_div(i, s);
137  if (SymEngine::free_symbols(u).empty()) {
138  Expr a = SymEngine::atan(u);
139  Expr q = 2 * atan2_bypi(j, s);
140  Expr two_a_by_pi = SymEngine::div(2 * a, SymEngine::pi);
141  return std::tuple<Expr, Expr, Expr>(0, q, two_a_by_pi);
142  }
143  }
144 
145  // Now the general case.
146  Expr a = atan2_bypi(i, s);
147  Expr b = atan2_bypi(k, j);
148  Expr q = acos_bypi(SymEngine::expand(s * s + i * i - j * j - k * k));
149  return std::tuple<Expr, Expr, Expr>(a - b, q, a + b);
150 }

152 Rotation::Rotation(OpType optype, Expr a)
153  : i_(0), j_(0), k_(0), optype_(optype), a_(a) {
154  if (equiv_0(a, 4)) {
155  rep_ = Rep::id;
156  s_ = 1;
157  i_ = j_ = k_ = 0;
158  } else if (equiv_0(a - 2, 4)) {
159  rep_ = Rep::minus_id;
160  s_ = -1;
161  i_ = j_ = k_ = 0;
162  } else {
163  rep_ = Rep::orth_rot;
164  s_ = cos_halfpi_times(a);
165  Expr t = sin_halfpi_times(a);
166  switch (optype) {
167  case OpType::Rx:
168  i_ = t;
169  break;
170  case OpType::Ry:
171  j_ = t;
172  break;
173  case OpType::Rz:
174  k_ = t;
175  break;
176  default:
177  throw std::logic_error(
178  "Quaternions can only be constructed "
179  "from Rx, Ry or Rz rotations");
180  }
181  }
182 }

196 std::tuple<Expr, Expr, Expr> Rotation::to_pqp(OpType p, OpType q) const {
197  if (rep_ == Rep::id) {
198  return {Expr(0), Expr(0), Expr(0)};
199  } else if (rep_ == Rep::minus_id) {
200  return {Expr(2), Expr(0), Expr(0)};
201  } else if (rep_ == Rep::orth_rot) {
202  if (optype_ == p) {
203  return {a_, Expr(0), Expr(0)};
204  } else if (optype_ == q) {
205  return {Expr(0), a_, Expr(0)};
206  }
207  }
208  if (p == OpType::Rx && q == OpType::Ry) {
209  return xyx_angles_from_coeffs(s_, i_, j_, k_);
210  } else if (p == OpType::Ry && q == OpType::Rx) {
211  return xyx_angles_from_coeffs(s_, j_, i_, -k_);
212  } else if (p == OpType::Ry && q == OpType::Rz) {
213  return xyx_angles_from_coeffs(s_, j_, k_, i_);
214  } else if (p == OpType::Rz && q == OpType::Ry) {
215  return xyx_angles_from_coeffs(s_, k_, j_, -i_);
216  } else if (p == OpType::Rz && q == OpType::Rx) {
217  return xyx_angles_from_coeffs(s_, k_, i_, j_);
218  } else if (p == OpType::Rx && q == OpType::Rz) {
219  return xyx_angles_from_coeffs(s_, i_, k_, -j_);
220  } else {
221  throw std::logic_error("Axes must be a pair of X, Y, Z.");
222  }
223 }

255 void Rotation::apply(const Rotation &other) {
256  if (other.rep_ == Rep::id) return;
257 
258  if (rep_ == Rep::id) {
259  rep_ = other.rep_;
260  s_ = other.s_;
261  i_ = other.i_;
262  j_ = other.j_;
263  k_ = other.k_;
264  optype_ = other.optype_;
265  a_ = other.a_;
266  return;
267  }
268 
269  if (rep_ == Rep::minus_id) {
270  if (other.rep_ == Rep::minus_id) {
271  rep_ = Rep::id;
272  s_ = 1;
273  i_ = j_ = k_ = 0;
274  } else if (other.rep_ == Rep::orth_rot) {
275  rep_ = Rep::orth_rot;
276  optype_ = other.optype_;
277  a_ = other.a_ + 2;
278  s_ = -other.s_;
279  i_ = -other.i_;
280  j_ = -other.j_, k_ = -other.k_;
281  } else {
282  rep_ = Rep::quat;
283  s_ = -other.s_;
284  i_ = -other.i_;
285  j_ = -other.j_, k_ = -other.k_;
286  }
287  return;
288  }
289 
290  if (rep_ == Rep::orth_rot && other.rep_ == Rep::orth_rot) {
291  if (optype_ == other.optype_) {
292  a_ += other.a_;
293  if (equiv_0(a_, 4)) {
294  rep_ = Rep::id;
295  } else if (equiv_0(a_, 2)) {
296  rep_ = Rep::minus_id;
297  }
298  } else if (
299  (equiv_val(a_, 1., 4) || equiv_val(a_, -1., 4)) &&
300  (equiv_val(other.a_, 1., 4) || equiv_val(other.a_, -1., 4))) {
301  // We are in a subgroup of order 8
302  int m0 = equiv_val(a_, 1., 4) ? 1 : -1;
303  int m1 = equiv_val(other.a_, 1., 4) ? 1 : -1;
304  std::tie(optype_, a_) = product.at({optype_, m0, other.optype_, m1});
305  } else
306  rep_ = Rep::quat;
307  } else
308  rep_ = Rep::quat;
309 
310  Expr s1 = other.s_ * s_ - other.i_ * i_ - other.j_ * j_ - other.k_ * k_;
311  Expr i1 = other.s_ * i_ + other.i_ * s_ + other.j_ * k_ - other.k_ * j_;
312  Expr j1 = other.s_ * j_ - other.i_ * k_ + other.j_ * s_ + other.k_ * i_;
313  Expr k1 = other.s_ * k_ + other.i_ * j_ - other.j_ * i_ + other.k_ * s_;
314  s_ = SymEngine::expand(s1);
315  i_ = SymEngine::expand(i1);
316  j_ = SymEngine::expand(j1);
317  k_ = SymEngine::expand(k1);
318 
319  if (rep_ == Rep::quat) {
320  // See if we can simplify the representation.
321  bool i_zero = approx_0(i_);
322  bool j_zero = approx_0(j_);
323  bool k_zero = approx_0(k_);
324  if (i_zero && j_zero && k_zero) {
325  if (approx_0(s_ - 1)) {
326  rep_ = Rep::id;
327  s_ = 1;
328  i_ = j_ = k_ = 0;
329  } else {
330  rep_ = Rep::minus_id;
331  s_ = -1;
332  i_ = j_ = k_ = 0;
333  }
334  } else if (j_zero && k_zero) {
335  rep_ = Rep::orth_rot;
336  optype_ = OpType::Rx;
337  a_ = 2 * atan2_bypi(i_, s_);
338  j_ = k_ = 0;
339  } else if (k_zero && i_zero) {
340  rep_ = Rep::orth_rot;
341  optype_ = OpType::Ry;
342  a_ = 2 * atan2_bypi(j_, s_);
343  k_ = i_ = 0;
344  } else if (i_zero && j_zero) {
345  rep_ = Rep::orth_rot;
346  optype_ = OpType::Rz;
347  a_ = 2 * atan2_bypi(k_, s_);
348  i_ = j_ = 0;
349  }
350  }
351 }

366 std::vector<double> tk1_angles_from_unitary(const Eigen::Matrix2cd &U) {
367  // clang-format off
368  //
369  // Assume U = e^{i pi p} TK1(a,b,c)
370  // = e^{i pi p} Rz(a) Rx(b) Rz(c)
371  // | e^{-i pi (a+c)/2} cos(pi b/2) -i e^{-i pi (a-c)/2} sin(pi b/2) |
372  // = e^{i pi p} | |
373  // | -i e^{ i pi (a-c)/2} sin(pi b/2) e^{ i pi (a+c)/2} cos(pi b/2) |
374  //
375  // clang-format on
376 
377  double a, b, c, p; // to be found satisfying the above equation
378 
379  std::complex s = 0.5 * (U(0, 0) + U(1, 1));
380  std::complex x = 0.5 * i_ * (U(1, 0) + U(0, 1));
381  std::complex y = 0.5 * (U(1, 0) - U(0, 1));
382  std::complex z = 0.5 * i_ * (U(0, 0) - U(1, 1));
383 
384  // s = e^{i pi p} cos(pi b/2) cos(pi (a+c)/2)
385  // x = e^{i pi p} sin(pi b/2) cos(pi (a-c)/2)
386  // y = e^{i pi p} sin(pi b/2) sin(pi (a-c)/2)
387  // z = e^{i pi p} cos(pi b/2) sin(pi (a+c)/2)
388 
389  // s, x, y and z all have phase e^{i pi p}. Extract it from the one with
390  // largest absolute value, to minimize numerical instability. Note that
391  // there are two possible values (p and p+1), but the choice is w.l.o.g.
392  // because (p,b) --> (p+1,b+2) does not change the value of the unitary.
393 
394  std::complex w = s;
395  if (std::abs(x) > std::abs(w)) w = x;
396  if (std::abs(y) > std::abs(w)) w = y;
397  if (std::abs(z) > std::abs(w)) w = z;
398  std::complex eip = w / std::abs(w);
399 
400  // eip = e^{i pi p}
401  p = std::arg(eip) / PI;
402 
403  // Now we've got the phase, factor it out.
404 
405  std::complex emip = std::conj(eip);
406  double s0 = std::real(emip * s);
407  double x0 = std::real(emip * x);
408  double y0 = std::real(emip * y);
409  double z0 = std::real(emip * z);
410 
411  // s0 = cos(pi b/2) cos(pi (a+c)/2)
412  // x0 = sin(pi b/2) cos(pi (a-c)/2)
413  // y0 = sin(pi b/2) sin(pi (a-c)/2)
414  // z0 = cos(pi b/2) sin(pi (a+c)/2)
415 
416  std::complex u = {s0, z0}, v = {x0, y0};
417 
418  // u = cos(pi b/2) e^{i pi (a+c)/2}
419  // v = sin(pi b/2) e^{i pi (a-c)/2}
420  // |u|^2 + |v|^2 = 1
421 
422  // Note that (a,b) --> (a+2, b+2) does not change the value of the unitary, so
423  // we are free to choose either solution.
424  //
425  // We treat the two special cases u=0 and v=0 separately, then the general
426  // case.
427 
428  if (std::abs(u) < EPS) { // special case, b = 1 or 3
429  // v = +/- e^{i pi (a-c)/2}
430 
431  // We may as well choose c = 0.
432  c = 0;
433 
434  // Assume b=1 and compute a.
435  // (b'=3, a'=a+2) is the other possibility but the unitary is the same.
436  b = 1;
437  a = 2 * std::arg(v) / PI;
438  } else if (std::abs(v) < EPS) { // special case, b = 0 or 2
439  // u = e^{i pi (a+c)/2}
440 
441  // We may as well choose c = 0.
442  c = 0;
443 
444  // Assume b=0 and compute a.
445  // (b'=2, a'=a+2) is the other possibility but the unitary is the same.
446  b = 0;
447  a = 2 * std::arg(u) / PI;
448  } else { // general case
449  // s0^2 + z0^2 - x0^2 - y0^2 = cos(pi b)
450  double t = s0 * s0 + z0 * z0 - x0 * x0 - y0 * y0;
451  // Rounding errors may mean t is outside the domain of acos. Fix this.
452  if (t > +1.) t = +1.;
453  if (t < -1.) t = -1.;
454  b = std::acos(t) / PI;
455 
456  // w.l.o.g. b is in the range (-1,+1).
457  double ac0 = std::arg(u), ac1 = std::arg(v);
458  a = (ac0 + ac1) / PI;
459  c = (ac0 - ac1) / PI;
460  }
461 
462  return {a, b, c, p};
463 }

465 std::tuple<double, double, double> get_bloch_coordinate_from_state(
466  const Complex &a, const Complex &b) {
467  // We require the state to be normalised
468  double norm = a.real() * a.real() + a.imag() * a.imag() +
469  b.real() * b.real() + b.imag() * b.imag();
470  if (std::abs(norm - 1) > EPS) {
471  throw std::invalid_argument(
472  "Attempt to find the Bloch sphere coordinate for an unnormalised "
473  "state. Only unit length vectors have coordinates on the Bloch "
474  "sphere.");
475  }
476  double r_a = std::abs(a);
477  double theta_a = std::arg(a) / PI;
478  double theta_b = std::arg(b) / PI;
479  double theta = (std::acos(r_a) * 2) / PI;
480  double phi = theta_b - theta_a;
481  double phase = theta_a;
482  return {theta, phi, phase};
483 }

485 Eigen::Matrix2cd get_matrix_from_tk1_angles(std::vector<Expr> params) {
486  double alpha = eval_expr(params[0]).value();
487  double beta = eval_expr(params[1]).value();
488  double gamma = eval_expr(params[2]).value();
489  double t = eval_expr(params[3]).value();
490  Eigen::Matrix2cd m;
491  alpha *= PI;
492  beta *= PI;
493  gamma *= PI;
494  t *= PI;
495  double c = cos(0.5 * beta);
496  double s = sin(0.5 * beta);
497  m << exp(-0.5 * i_ * (alpha + gamma)) * c,
498  -i_ * exp(0.5 * i_ * (gamma - alpha)) * s,
499  -i_ * exp(0.5 * i_ * (alpha - gamma)) * s,
500  exp(0.5 * i_ * (alpha + gamma)) * c;
501  return exp(i_ * t) * m;
502 }
