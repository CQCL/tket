28 static Expr atan2_bypi(const Expr &a, const Expr &b) {
29  std::optional<double> va = eval_expr(a);
30  std::optional<double> vb = eval_expr(b);
31  if (va && vb) {
32  double vva = va.value();
33  double vvb = vb.value();
34  if (std::abs(vva) < EPS && std::abs(vvb) < EPS) return Expr(0.);
35  return atan2(vva, vvb) / PI;
36  } else {
37  // Convert symbolic zero to 0. This is a workaround for
38  // https://github.com/symengine/symengine/issues/1875 .
39  Expr a1 = a, b1 = b;
40  if (a1 == SymEngine::zero) a1 = 0.;
41  if (b1 == SymEngine::zero) b1 = 0.;
42  return SymEngine::div(SymEngine::atan2(a1, b1), SymEngine::pi);
43  }
44 }

46 static Expr acos_bypi(const Expr &a) {
47  std::optional<double> va = eval_expr(a);
48  if (va) {
49  double vva = va.value();
50  // avoid undefined values due to rounding
51  if (vva >= 1.) return 0.;
52  if (vva <= -1.) return 1.;
53  return acos(vva) / PI;
54  } else {
55  return SymEngine::div(SymEngine::acos(a), SymEngine::pi);
56  }
57 }

59 // SymEngine::div does not always spot when the numerator is a scalar multiple
60 // of the denominator, for example in expressions like (-a + b) / (a - b) where
61 // a and b are symbolic. This function picks out the common cases.
62 static Expr expr_div(const Expr &num, const Expr &den) {
63  if (approx_0(SymEngine::expand(num - den))) return 1;
64  if (approx_0(SymEngine::expand(num + den))) return -1;
65  return SymEngine::div(num, den);
66 }

68 static std::tuple<Expr, Expr, Expr> xyx_angles_from_coeffs(
69  const Expr &s, const Expr &i, const Expr &j, const Expr &k) {
70  // Handle exceptional cases first.
71  bool s_zero = approx_0(s);
72  bool s_one = approx_0(s - 1);
73  bool i_zero = approx_0(i);
74  bool i_one = approx_0(i - 1);
75  bool j_zero = approx_0(j);
76  bool j_one = approx_0(j - 1);
77  bool k_zero = approx_0(k);
78  bool k_one = approx_0(k - 1);
79  if (i_zero && j_zero && k_zero) {
80  if (s_one)
81  return {0, 0, 0};
82  else
83  return {2, 0, 0};
84  }
85  if (s_zero && j_zero && k_zero) {
86  if (i_one)
87  return {1, 0, 0};
88  else
89  return {3, 0, 0};
90  }
91  if (s_zero && i_zero && k_zero) {
92  if (j_one)
93  return {0, 1, 0};
94  else
95  return {0, 3, 0};
96  }
97  if (s_zero && i_zero && j_zero) {
98  if (k_one)
99  return {3, 1, 0};
100  else
101  return {1, 1, 0};
102  }
103  if (s_zero && i_zero) return {-2 * atan2_bypi(k, j), 1, 0};
104  if (s_zero && j_zero) return {0, 2 * atan2_bypi(k, i), 1};
105  if (s_zero && k_zero) return {0.5, 2 * atan2_bypi(j, i), 0.5};
106  if (i_zero && j_zero) return {-0.5, 2 * atan2_bypi(k, s), 0.5};
107  if (i_zero && k_zero) return {0, 2 * atan2_bypi(j, s), 0};
108  if (j_zero && k_zero) return {2 * atan2_bypi(i, s), 0, 0};
109  if (s_zero) {
110  // We want to return (a, b, c) s.t.
111  // Rx(c) Ry(b) Rx(a) = i*i_ + j*j_ + k*k_ (where i_, j_, k_ are the
112  // quaternionic roots of -1. Multiplying out the LHS gives
113  // (Cc + i_ Sc)(Cb + j_ Sb)(Ca + i_ Sa)
114  // where for brevity Sa = sin(pi*a/2), Ca = cos(pi*a/2) etc.
115  // Expanding and using the sin/cos addition identities gives
116  // Cb cos(pi*l) = 0, Cb sin(pi*l) = i, Sb cos(pi*m) = j, Sb sin(pi*m) = k
117  // where l = (c+a)/2 and m = (c-a)/2.
118  // Since we have already handled the case s=i=0 we know that Cb != 0, so
119  // cos(pi*l) = 0. A solution to these equations is given by
120  // l = 1/2, m = atan2(k,j)/pi, b = (2/pi)*acos(i)
121  // and finally substituting a=l-m, b=l+m gives the following solution:
122  Expr m = atan2_bypi(k, j);
123  return {0.5 - m, 2 * acos_bypi(i), 0.5 + m};
124  }
125 
126  // This is a (partial) workaround for
127  // https://github.com/symengine/symengine/issues/1806
128  // (since it avoids the use of atan2 with proportional symbolics).
129  // Explanation: When the quaternion is of the form
130  // A + uA i + B j -/+ uB k
131  // where u is a pure number (with no free symbols) but A and B are symbolic,
132  // symengine wrongly simplifies atan2(uA, A) and atan2(uB, B), ignoring the
133  // symbols (whose sign ought to affect the result). So the general formula
134  // below does not work. We therefore handle these cases differently, noting
135  // that the quaternion factorizes as either
136  // (A + Bj) (1 + ui) or (1 + ui) (A + Bj)
137  // -- that is, an Rx followed by an Ry or vice versa.
138  // We will analyse the first case; the second is similar.
139  // Let alpha = atan(u). Note that -pi/2 < alpha < pi/2, so cos(alpha) > 0.
140  // The quaternion is then
141  // (A/cos(alpha) + B/cos(alpha) j) (cos(alpha) + sin(alpha) i)
142  // So we can take the angle of the Rx as 2*alpha, and the angle of the Ry as
143  // 2 * atan2(B, A).
144  // Finally, note that u must be well-defined because we have already dealt
145  // with all cases where s = 0.
146  TKET_ASSERT(!s_zero);
147  if (approx_0(SymEngine::expand(i * j + s * k))) {
148  Expr u = expr_div(i, s);
149  if (SymEngine::free_symbols(u).empty()) {
150  Expr a = SymEngine::atan(u);
151  Expr q = 2 * atan2_bypi(j, s);
152  Expr two_a_by_pi = SymEngine::div(2 * a, SymEngine::pi);
153  return std::tuple<Expr, Expr, Expr>(two_a_by_pi, q, 0);
154  }
155  } else if (approx_0(SymEngine::expand(i * j - s * k))) {
156  Expr u = expr_div(i, s);
157  if (SymEngine::free_symbols(u).empty()) {
158  Expr a = SymEngine::atan(u);
159  Expr q = 2 * atan2_bypi(j, s);
160  Expr two_a_by_pi = SymEngine::div(2 * a, SymEngine::pi);
161  return std::tuple<Expr, Expr, Expr>(0, q, two_a_by_pi);
162  }
163  }
164 
165  // Now the general case.
166  Expr a = atan2_bypi(i, s);
167  Expr b = atan2_bypi(k, j);
168  Expr q = acos_bypi(SymEngine::expand(s * s + i * i - j * j - k * k));
169  return std::tuple<Expr, Expr, Expr>(a - b, q, a + b);
170 }

172 Rotation::Rotation(OpType optype, Expr a)
173  : i_(0), j_(0), k_(0), optype_(optype), a_(a) {
174  if (equiv_0(a, 4)) {
175  rep_ = Rep::id;
176  s_ = 1;
177  i_ = j_ = k_ = 0;
178  } else if (equiv_0(a - 2, 4)) {
179  rep_ = Rep::minus_id;
180  s_ = -1;
181  i_ = j_ = k_ = 0;
182  } else {
183  rep_ = Rep::orth_rot;
184  s_ = cos_halfpi_times(a);
185  Expr t = sin_halfpi_times(a);
186  switch (optype) {
187  case OpType::Rx:
188  i_ = t;
189  break;
190  case OpType::Ry:
191  j_ = t;
192  break;
193  case OpType::Rz:
194  k_ = t;
195  break;
196  default:
197  throw std::logic_error(
198  "Quaternions can only be constructed "
199  "from Rx, Ry or Rz rotations");
200  }
201  }

216 std::tuple<Expr, Expr, Expr> Rotation::to_pqp(OpType p, OpType q) const {
217  if (rep_ == Rep::id) {
218  return {Expr(0), Expr(0), Expr(0)};
219  } else if (rep_ == Rep::minus_id) {
220  return {Expr(2), Expr(0), Expr(0)};
221  } else if (rep_ == Rep::orth_rot) {
222  if (optype_ == p) {
223  return {a_, Expr(0), Expr(0)};
224  } else if (optype_ == q) {
225  return {Expr(0), a_, Expr(0)};
226  }
227  }
228  if (p == OpType::Rx && q == OpType::Ry) {
229  return xyx_angles_from_coeffs(s_, i_, j_, k_);
230  } else if (p == OpType::Ry && q == OpType::Rx) {
231  return xyx_angles_from_coeffs(s_, j_, i_, -k_);
232  } else if (p == OpType::Ry && q == OpType::Rz) {
233  return xyx_angles_from_coeffs(s_, j_, k_, i_);
234  } else if (p == OpType::Rz && q == OpType::Ry) {
235  return xyx_angles_from_coeffs(s_, k_, j_, -i_);
236  } else if (p == OpType::Rz && q == OpType::Rx) {
237  return xyx_angles_from_coeffs(s_, k_, i_, j_);
238  } else if (p == OpType::Rx && q == OpType::Rz) {
239  return xyx_angles_from_coeffs(s_, i_, k_, -j_);
240  } else {
241  throw std::logic_error("Axes must be a pair of X, Y, Z.");
242  }
243 }

275 void Rotation::apply(const Rotation &other) {
276  if (other.rep_ == Rep::id) return;
277 
278  if (rep_ == Rep::id) {
279  rep_ = other.rep_;
280  s_ = other.s_;
281  i_ = other.i_;
282  j_ = other.j_;
283  k_ = other.k_;
284  optype_ = other.optype_;
285  a_ = other.a_;
286  return;
287  }
288 
289  if (rep_ == Rep::minus_id) {
290  if (other.rep_ == Rep::minus_id) {
291  rep_ = Rep::id;
292  s_ = 1;
293  i_ = j_ = k_ = 0;
294  } else if (other.rep_ == Rep::orth_rot) {
295  rep_ = Rep::orth_rot;
296  optype_ = other.optype_;
297  a_ = other.a_ + 2;
298  s_ = -other.s_;
299  i_ = -other.i_;
300  j_ = -other.j_, k_ = -other.k_;
301  } else {
302  rep_ = Rep::quat;
303  s_ = -other.s_;
304  i_ = -other.i_;
305  j_ = -other.j_, k_ = -other.k_;
306  }
307  return;
308  }
309 
310  if (rep_ == Rep::orth_rot && other.rep_ == Rep::orth_rot) {
311  if (optype_ == other.optype_) {
312  a_ += other.a_;
313  if (equiv_0(a_, 4)) {
314  rep_ = Rep::id;
315  } else if (equiv_0(a_, 2)) {
316  rep_ = Rep::minus_id;
317  }
318  } else if (
319  (equiv_val(a_, 1., 4) || equiv_val(a_, -1., 4)) &&
320  (equiv_val(other.a_, 1., 4) || equiv_val(other.a_, -1., 4))) {
321  // We are in a subgroup of order 8
322  int m0 = equiv_val(a_, 1., 4) ? 1 : -1;
323  int m1 = equiv_val(other.a_, 1., 4) ? 1 : -1;
324  std::tie(optype_, a_) = product.at({optype_, m0, other.optype_, m1});
325  } else
326  rep_ = Rep::quat;
327  } else
328  rep_ = Rep::quat;
329 
330  Expr s1 = other.s_ * s_ - other.i_ * i_ - other.j_ * j_ - other.k_ * k_;
331  Expr i1 = other.s_ * i_ + other.i_ * s_ + other.j_ * k_ - other.k_ * j_;
332  Expr j1 = other.s_ * j_ - other.i_ * k_ + other.j_ * s_ + other.k_ * i_;
333  Expr k1 = other.s_ * k_ + other.i_ * j_ - other.j_ * i_ + other.k_ * s_;
334  s_ = SymEngine::expand(s1);
335  i_ = SymEngine::expand(i1);
336  j_ = SymEngine::expand(j1);
337  k_ = SymEngine::expand(k1);
338 
339  if (rep_ == Rep::quat) {
340  // See if we can simplify the representation.
341  bool i_zero = approx_0(i_);
342  bool j_zero = approx_0(j_);
343  bool k_zero = approx_0(k_);
344  if (i_zero && j_zero && k_zero) {
345  if (approx_0(s_ - 1)) {
346  rep_ = Rep::id;
347  s_ = 1;
348  i_ = j_ = k_ = 0;
349  } else {
350  rep_ = Rep::minus_id;
351  s_ = -1;
352  i_ = j_ = k_ = 0;
353  }
354  } else if (j_zero && k_zero) {
355  rep_ = Rep::orth_rot;
356  optype_ = OpType::Rx;
357  a_ = 2 * atan2_bypi(i_, s_);
358  j_ = k_ = 0;
359  } else if (k_zero && i_zero) {
360  rep_ = Rep::orth_rot;
361  optype_ = OpType::Ry;
362  a_ = 2 * atan2_bypi(j_, s_);
363  k_ = i_ = 0;
364  } else if (i_zero && j_zero) {
365  rep_ = Rep::orth_rot;
366  optype_ = OpType::Rz;
367  a_ = 2 * atan2_bypi(k_, s_);
368  i_ = j_ = 0;
369  }
370  }
371 }

386 std::vector<double> tk1_angles_from_unitary(const Eigen::Matrix2cd &U) {
387  // clang-format off
388  //
389  // Assume U = e^{i pi p} TK1(a,b,c)
390  // = e^{i pi p} Rz(a) Rx(b) Rz(c)
391  // | e^{-i pi (a+c)/2} cos(pi b/2) -i e^{-i pi (a-c)/2} sin(pi b/2) |
392  // = e^{i pi p} | |
393  // | -i e^{ i pi (a-c)/2} sin(pi b/2) e^{ i pi (a+c)/2} cos(pi b/2) |
394  //
395  // clang-format on
396 
397  double a, b, c, p; // to be found satisfying the above equation
398 
399  std::complex s = 0.5 * (U(0, 0) + U(1, 1));
400  std::complex x = 0.5 * i_ * (U(1, 0) + U(0, 1));
401  std::complex y = 0.5 * (U(1, 0) - U(0, 1));
402  std::complex z = 0.5 * i_ * (U(0, 0) - U(1, 1));
403 
404  // s = e^{i pi p} cos(pi b/2) cos(pi (a+c)/2)
405  // x = e^{i pi p} sin(pi b/2) cos(pi (a-c)/2)
406  // y = e^{i pi p} sin(pi b/2) sin(pi (a-c)/2)
407  // z = e^{i pi p} cos(pi b/2) sin(pi (a+c)/2)
408 
409  // s, x, y and z all have phase e^{i pi p}. Extract it from the one with
410  // largest absolute value, to minimize numerical instability. Note that
411  // there are two possible values (p and p+1), but the choice is w.l.o.g.
412  // because (p,b) --> (p+1,b+2) does not change the value of the unitary.
413 
414  std::complex w = s;
415  if (std::abs(x) > std::abs(w)) w = x;
416  if (std::abs(y) > std::abs(w)) w = y;
417  if (std::abs(z) > std::abs(w)) w = z;
418  std::complex eip = w / std::abs(w);
419 
420  // eip = e^{i pi p}
421  p = std::arg(eip) / PI;
422 
423  // Now we've got the phase, factor it out.
424 
425  std::complex emip = std::conj(eip);
426  double s0 = std::real(emip * s);
427  double x0 = std::real(emip * x);
428  double y0 = std::real(emip * y);
429  double z0 = std::real(emip * z);
430 
431  // s0 = cos(pi b/2) cos(pi (a+c)/2)
432  // x0 = sin(pi b/2) cos(pi (a-c)/2)
433  // y0 = sin(pi b/2) sin(pi (a-c)/2)
434  // z0 = cos(pi b/2) sin(pi (a+c)/2)
435 
436  std::complex u = {s0, z0}, v = {x0, y0};
437 
438  // u = cos(pi b/2) e^{i pi (a+c)/2}
439  // v = sin(pi b/2) e^{i pi (a-c)/2}
440  // |u|^2 + |v|^2 = 1
441 
442  // Note that (a,b) --> (a+2, b+2) does not change the value of the unitary, so
443  // we are free to choose either solution.
444  //
445  // We treat the two special cases u=0 and v=0 separately, then the general
446  // case.
447 
448  if (std::abs(u) < EPS) { // special case, b = 1 or 3
449  // v = +/- e^{i pi (a-c)/2}
450 
451  // We may as well choose c = 0.
452  c = 0;
453 
454  // Assume b=1 and compute a.
455  // (b'=3, a'=a+2) is the other possibility but the unitary is the same.
456  b = 1;
457  a = 2 * std::arg(v) / PI;
458  } else if (std::abs(v) < EPS) { // special case, b = 0 or 2
459  // u = e^{i pi (a+c)/2}
460 
461  // We may as well choose c = 0.
462  c = 0;
463 
464  // Assume b=0 and compute a.
465  // (b'=2, a'=a+2) is the other possibility but the unitary is the same.
466  b = 0;
467  a = 2 * std::arg(u) / PI;
468  } else { // general case
469  // s0^2 + z0^2 - x0^2 - y0^2 = cos(pi b)
470  double t = s0 * s0 + z0 * z0 - x0 * x0 - y0 * y0;
471  // Rounding errors may mean t is outside the domain of acos. Fix this.
472  if (t > +1.) t = +1.;
473  if (t < -1.) t = -1.;
474  b = std::acos(t) / PI;
475 
476  // w.l.o.g. b is in the range (-1,+1).
477  double ac0 = std::arg(u), ac1 = std::arg(v);
478  a = (ac0 + ac1) / PI;
479  c = (ac0 - ac1) / PI;
480  }
481 
482  return {a, b, c, p};
483 }

485 std::tuple<double, double, double> get_bloch_coordinate_from_state(
486  const Complex &a, const Complex &b) {
487  // We require the state to be normalised
488  double norm = a.real() * a.real() + a.imag() * a.imag() +
489  b.real() * b.real() + b.imag() * b.imag();
490  if (std::abs(norm - 1) > EPS) {
491  throw std::invalid_argument(
492  "Attempt to find the Bloch sphere coordinate for an unnormalised "
493  "state. Only unit length vectors have coordinates on the Bloch "
494  "sphere.");
495  }
496  double r_a = std::abs(a);
497  double theta_a = std::arg(a) / PI;
498  double theta_b = std::arg(b) / PI;
