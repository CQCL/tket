28std::pair<bool, unit_map_t> RoutingMethodCircuit::routing_method(
29 MappingFrontier_ptr& mapping_frontier,
30 const ArchitecturePtr& architecture) const {
31 // Produce subcircuit and circuit
32 Subcircuit frontier_subcircuit = mapping_frontier->get_frontier_subcircuit(
33 this->max_depth_, this->max_size_);
34 Circuit frontier_circuit =
35 mapping_frontier->circuit_.subcircuit(frontier_subcircuit);
36 frontier_circuit.rename_units(
37 mapping_frontier->get_default_to_linear_boundary_unit_map());
38
39 // get routed subcircuit
40 std::tuple<bool, Circuit, unit_map_t, unit_map_t> routed_subcircuit =
41 this->route_subcircuit_(frontier_circuit, architecture);
42
43 if (!std::get<0>(routed_subcircuit)) {
44 return {false, {}};
45 }
46
47 // update unit id at boundary in case of relabelling
48 // The route_subcircuit_ method populates its initial map
49 // with unit ids from the circuit. e.g. Initial map from frontier ==
50 // q[0]:unplaced[0], circuit.all_qubits() == unplaced[0]. Then the produced
51 // initial map == unplaced[0]:node[0] We have to update the initial map to
52 // q[0]:node[0]
53 mapping_frontier->update_linear_boundary_uids(std::get<2>(routed_subcircuit));
54 for (const auto& pair : std::get<2>(routed_subcircuit)) {
55 mapping_frontier->update_bimaps(
56 mapping_frontier->get_qubit_from_circuit_uid(pair.first), pair.second);
57 }
58
59 unit_map_t swap_permutation;
60 for (const auto& pair : std::get<2>(routed_subcircuit)) {
61 if (pair.first != pair.second &&
62 architecture->node_exists(Node(pair.first))) {
63 swap_permutation.insert(pair);
64 }
65 }
66 // permute edges held by unitid at out boundary due to swaps
67 mapping_frontier->permute_subcircuit_q_out_hole(
68 std::get<3>(routed_subcircuit), frontier_subcircuit);
69
70 // substitute old boundary with new cirucit
71 std::get<1>(routed_subcircuit).flatten_registers();
72 mapping_frontier->circuit_.substitute(
73 std::get<1>(routed_subcircuit), frontier_subcircuit);
74 // return initial unit_map_t incase swap network required
75 return {true, swap_permutation};
76}
