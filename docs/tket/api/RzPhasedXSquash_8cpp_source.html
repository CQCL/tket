32std::pair<Circuit, Gate_ptr> RzPhasedXSquasher::flush(
33 std::optional<Pauli> commutation_colour) const {
34 // For Quantinuum gate set, commutation_colour can only be Z or nullopt (e.g.
35 // Measure)
36 std::pair<Circuit, Gate_ptr> pair = PQPSquasher::flush(commutation_colour);
37
38 // Setting smart_squash to false should guarantee that only Rzs can commute
39 // through.
40 TKET_ASSERT(pair.second == nullptr || pair.second->get_type() == OpType::Rz);
41 Circuit replacement(1);
42 replacement.add_phase(pair.first.get_phase());
43
44 // 1. Recover the angles of RzRxRz and the leftover Rz from PQPSquasher
45 std::vector<Command> coms = pair.first.get_commands();
46 // pqp
47 Expr rz1_angle = 0;
48 Expr rx_angle = 0;
49 Expr rz2_angle = 0;
50 // left over Rz
51 Expr rz3_angle = 0;
52 if (coms.size() > 0) {
53 if (coms[0].get_op_ptr()->get_type() == OpType::Rz) {
54 rz1_angle = coms[0].get_op_ptr()->get_params()[0];
55 } else {
56 rx_angle = coms[0].get_op_ptr()->get_params()[0];
57 }
58 }
59 if (coms.size() > 1) {
60 if (coms[1].get_op_ptr()->get_type() == OpType::Rz) {
61 rz2_angle = coms[1].get_op_ptr()->get_params()[0];
62 } else {
63 rx_angle = coms[1].get_op_ptr()->get_params()[0];
64 }
65 }
66 if (coms.size() > 2) {
67 rz2_angle = coms[2].get_op_ptr()->get_params()[0];
68 }
69 if (pair.second != nullptr) {
70 rz3_angle = pair.second->get_params()[0];
71 }
72
73 // 2. Rebase (Rz Rx Rz | Rz) into (PhasedX Rz | Rz)
74 if (commutation_colour == Pauli::Z || commutation_colour == Pauli::I) {
75 if (!equiv_0(rx_angle, 4)) {
76 replacement.add_op<unsigned>(
77 OpType::PhasedX, {rx_angle, -rz1_angle}, {0});
78 }
79 rz3_angle += rz1_angle + rz2_angle;
80 } else {
81 if (!equiv_0(rx_angle, 4)) {
82 replacement.add_op<unsigned>(
83 OpType::PhasedX, {rx_angle, -rz1_angle}, {0});
84 }
85 if (!equiv_0(rz1_angle + rz2_angle, 4)) {
86 replacement.add_op<unsigned>(OpType::Rz, {rz1_angle + rz2_angle}, {0});
87 }
88 }
89 Gate_ptr rz3_gate =
90 equiv_0(rz3_angle, 4)
91 ? nullptr
92 : std::make_shared<Gate>(OpType::Rz, std::vector<Expr>{rz3_angle}, 1);
93 return {replacement, rz3_gate};
94}

96Transform squash_1qb_to_Rz_PhasedX(bool always_squash_symbols) {
97 return Transform([always_squash_symbols](Circuit &circ) {
98 bool reverse = false;
99 bool success = decompose_ZX().apply(circ);
100 success = remove_redundancies().apply(circ) || success;
101 auto squasher = std::make_unique<RzPhasedXSquasher>(reverse);
102 return SingleQubitSquash(
103 std::move(squasher), circ, reverse, always_squash_symbols)
104 .squash() ||
105 success;
106 });
107}
