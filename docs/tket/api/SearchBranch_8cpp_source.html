29 SearchBranch::SearchBranch(
30  const DomainInitialiser::InitialDomains& initial_domains,
31  const NeighboursData& pattern_ndata, NearNeighboursData& pattern_near_ndata,
32  const NeighboursData& target_ndata, NearNeighboursData& target_near_ndata,
33  unsigned max_distance_reduction_value, ExtraStatistics& extra_statistics)
34  : m_pattern_ndata(pattern_ndata),
35  m_target_ndata(target_ndata),
36  m_extra_statistics(extra_statistics),
37  m_derived_graphs_reducer(m_pattern_ndata, m_target_ndata),
38  m_neighbours_reducer(m_pattern_ndata, m_target_ndata),
39  m_nodes_raw_data_wrapper(initial_domains),
40  m_domains_accessor(m_nodes_raw_data_wrapper),
41  m_node_list_traversal(m_nodes_raw_data_wrapper) {
42  m_extra_statistics.number_of_pattern_vertices =
43  m_pattern_ndata.get_number_of_nonisolated_vertices();
44  m_extra_statistics.number_of_target_vertices =
45  m_target_ndata.get_number_of_nonisolated_vertices();
46  m_extra_statistics.initial_number_of_possible_assignments = 0;
47  for (const std::set<VertexWSM>& entry : initial_domains) {
48  m_extra_statistics.initial_number_of_possible_assignments += entry.size();
49  }
50 
51  // In what order should we do reduction/checks?
52  // The simplest/cheapest first? Most powerful?
53  // Seems a difficult question...
54  m_reducer_wrappers.emplace_back(m_neighbours_reducer);
55  m_reducer_wrappers.emplace_back(m_derived_graphs_reducer);
56 
57  if (max_distance_reduction_value >= 2) {
58  m_distance_reducers.reserve(max_distance_reduction_value - 1);
59  m_reducer_wrappers.reserve(max_distance_reduction_value - 1);
60  for (unsigned distance = 2; distance <= max_distance_reduction_value;
61  ++distance) {
62  m_distance_reducers.emplace_back(
63  pattern_near_ndata, m_target_ndata, target_near_ndata, distance);
64  }
65  // Now that all the reducer objects are stored
66  // (the vector will not be resized),
67  // we can safely take references to them.
68  for (DistancesReducer& reducer : m_distance_reducers) {
69  m_reducer_wrappers.emplace_back(reducer);
70  }
71  }
72 }

110 bool SearchBranch::perform_single_assignment_checks_in_reduce_loop(
111  std::size_t num_assignments_alldiff_processed) {
112  const std::vector<std::pair<VertexWSM, VertexWSM>>& new_assignments =
113  m_domains_accessor.get_new_assignments();
114  for (auto ii = num_assignments_alldiff_processed; ii < new_assignments.size();
115  ++ii) {
116  if (m_checked_assignments[new_assignments[ii].first].count(
117  new_assignments[ii].second) != 0) {
118  continue;
119  }
120  for (auto& reducer_wrapper : m_reducer_wrappers) {
121  if (!reducer_wrapper.check(new_assignments[ii])) {
122  // The whole point is, the check for PV->TV does NOT depend on
123  // the other domains; since it's failed the check, it was ALWAYS
124  // invalid, so we remove it completely from ALL data.
125  m_node_list_traversal.erase_impossible_assignment(new_assignments[ii]);
126  // The current node is a nogood, so we'll move up from it shortly;
127  // no point in further checks or reductions, this node is doomed!
128  ++m_extra_statistics.total_number_of_impossible_assignments;
129  return false;
130  }
131  }
132  }
133  return true;
134 }

136 bool SearchBranch::check_and_update_scalar_product_in_reduce_loop(
137  const ReductionParameters& parameters,
138  std::size_t num_assignments_alldiff_processed) {
139  const auto weight_result_opt = m_weight_calculator(
140  m_pattern_ndata, m_target_ndata, m_domains_accessor,
141  num_assignments_alldiff_processed, parameters.max_weight);
142

144  return false;
145  }
146  m_domains_accessor.set_scalar_product(weight_result_opt->scalar_product)
147  .set_total_p_edge_weights(
148  weight_result_opt->total_extra_p_edge_weights +
149  m_domains_accessor.get_total_p_edge_weights());
150  return true;

153 bool SearchBranch::perform_weight_nogood_check_in_reduce_loop(
154  const ReductionParameters& parameters) {
155  if (is_maximum(parameters.max_weight)) {
156  // No scalar product constraint, so nothing to check.
157  return true;
158  }
159  const auto scalar_product = m_domains_accessor.get_scalar_product();
160  if (scalar_product > parameters.max_weight) {
161  return false;
162  }
163 
164  if (m_weight_checker_ptr) {
165  m_impossible_target_vertices.clear();
166  const WeightWSM max_extra_scalar_product =
167  parameters.max_weight - scalar_product;
168 
169  bool node_is_valid = m_weight_checker_ptr->check(
170  m_domains_accessor, max_extra_scalar_product);
171 
172  const auto number_of_pv =
173  m_domains_accessor.get_number_of_pattern_vertices();
174  for (VertexWSM tv : m_impossible_target_vertices) {
175  // This is rare. Crudely treat as a list of assignments.
176  // We want to pass them all in now, though, even if we're at a nogood;
177  // they might not be detected again.
178  m_extra_statistics.impossible_target_vertices.emplace_back(tv);
179  for (unsigned pv = 0; pv < number_of_pv; ++pv) {
180  m_node_list_traversal.erase_impossible_assignment(
181  std::make_pair(pv, tv));
182  }
183  // Detecting an invalid TV is separate from whether
184  // the current node is a nogood.
185  node_is_valid &= m_domains_accessor.current_node_is_valid();
186  }
187  return node_is_valid;
188  }
189  return true;
190 }

192 ReductionResult SearchBranch::perform_reducers_in_reduce_loop() {
193  for (auto& reducer_wrapper : m_reducer_wrappers) {
194  const auto reduction_result =
195  reducer_wrapper.reduce(m_domains_accessor, m_work_set_for_reducers);
196  if (reduction_result != ReductionResult::SUCCESS) {
197  return reduction_result;
198  }
199  }
200  return ReductionResult::SUCCESS;

203 bool SearchBranch::perform_main_reduce_loop(
204  const ReductionParameters& parameters) {
205  // Within the loop, we reduce; we break out when it stops changing,
206  // so is valid and fully reduced.
207 
208  // At the start of each new loop, this tells us how many initial elements
209  // of new_assignments have been processed by the alldiff reducer.
210  std::size_t num_assignments_alldiff_processed = 0;
211 
212  // We will not change this directly. The reference will remain valid,
213  // but it will change indirectly due to other actions (reducing, etc.).
214  const std::vector<std::pair<VertexWSM, VertexWSM>>& new_assignments =
215  m_domains_accessor.get_new_assignments();
216 
217  for (;;) {
218  if (!(m_domains_accessor.alldiff_reduce_current_node(
219  num_assignments_alldiff_processed) &&
220  // Checks/updates which don't alter domains follow.
221  perform_single_assignment_checks_in_reduce_loop(
222  num_assignments_alldiff_processed) &&
223  check_and_update_scalar_product_in_reduce_loop(
224  parameters, num_assignments_alldiff_processed))) {
225  return false;
226  }
227 
228  num_assignments_alldiff_processed = new_assignments.size();
229 
230  if (!perform_weight_nogood_check_in_reduce_loop(parameters)) {
231  return false;
232  }
233  // It's rare, but possible, that the weight nogood check found an impossible
234  // TV, in which case it was erased, and the domains DID change...
235  if (num_assignments_alldiff_processed != new_assignments.size()) {
236  // Jump back to start!
237  continue;
238  }
239 
240  // Now do REDUCTIONS, which do alter domains.
241  // The scalar product updates etc. have all been completed.
242  // If we detect that a new assignment has occurred,
243  // we jump back to the start of the loop. (Since the node reductions
244  // and assignment checking should be faster
245  // than these more complicated reductions).
246  {
247  const auto reducers_result = perform_reducers_in_reduce_loop();
248  if (reducers_result == ReductionResult::NOGOOD) {
249  return false;
250  }
251  if (reducers_result == ReductionResult::NEW_ASSIGNMENTS) {
252  // Jump back to start of loop.
253  TKET_ASSERT(num_assignments_alldiff_processed < new_assignments.size());
254  continue;
255  }
256  TKET_ASSERT(num_assignments_alldiff_processed == new_assignments.size());
257  }
258 
259  // It is reasonable that the final reduction is the Hall set reducer,
260  // because it alone benefits from smaller domains
261  // (i.e., is more likely to detect a possible reduction).
262  // The standard reducers ONLY work with actual ASSIGNMENTS;
263  // it makes no difference if a domain has size 2 or 100.
264  const auto hall_set_result =
265  m_hall_set_reduction.reduce(m_domains_accessor);
266  if (hall_set_result == ReductionResult::NOGOOD) {
267  return false;
268  }
269  TKET_ASSERT(num_assignments_alldiff_processed <= new_assignments.size());
270  if (num_assignments_alldiff_processed == new_assignments.size()) {
271  TKET_ASSERT(hall_set_result == ReductionResult::SUCCESS);
272  // No further assignments from reductions.
273  break;
274  }
275  // Will jump back to start of loop.
276  TKET_ASSERT(hall_set_result == ReductionResult::NEW_ASSIGNMENTS);
277  TKET_ASSERT(num_assignments_alldiff_processed < new_assignments.size());
278  }
279  return true;
280 }

282 bool SearchBranch::reduce_current_node(const ReductionParameters& parameters) {
283  for (auto& reducer_wrapper : m_reducer_wrappers) {
284  reducer_wrapper.clear();
285  }
286  m_hall_set_reduction.clear();
287 
288  if (!perform_main_reduce_loop(parameters)) {
289  return false;
290  }
291  // Now that we've processed all the assignments,
292  // we don't need them.
293  m_domains_accessor.clear_new_assignments();
294  return true;
295 }

297 bool SearchBranch::backtrack(const ReductionParameters& parameters) {
298  do {
299  if (!m_node_list_traversal.move_up()) {
300  return false;
301  }
302  } while (!reduce_current_node(parameters));
303  return true;
