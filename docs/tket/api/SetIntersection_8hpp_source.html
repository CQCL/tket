97 template <
98  class T, class ResultInserter, class ExtraData, class GetTFromExtraData,
99  class GetExtraDataFromT>
100 void fill_intersection_using_inserter(
101  const std::set<T>& set, const std::vector<ExtraData>& sorted_vect,
102  ResultInserter& inserter, GetTFromExtraData get_t,
103  GetExtraDataFromT get_extra_data) {
104  if (set.empty() || sorted_vect.empty()) {
105  return;
106  }
107  auto vect_citer = sorted_vect.cbegin();
108 
109  for (;;) {
110  const T vect_t_value = get_t(*vect_citer);
111  auto set_citer = set.lower_bound(vect_t_value);
112  if (set_citer == set.cend()) {
113  break;
114  }
115  // We have s >= v, where s in S, v in V.
116  if (vect_t_value == *set_citer) {
117  // s=v is a common element.
118  inserter.insert(vect_t_value);
119  // Advance in S.
120  ++set_citer;
121  if (set_citer == set.cend()) {
122  break;
123  }
124  // Now, t>s with t in S.
125  }
126  // If we had s=v above, then now t>v.
127  // Otherwise, s>v.
128  // In both cases, set_citer now refers to z>v.
129  // Get y >= z, with y in V.
130  // We can advance v [vector iterator advancement is very fast O(1),
131  // unlike O(log N) for std::set].
132  ++vect_citer;
133 
134  // Note: we're searching over a vector subinterval,
135  // so slightly faster than the whole vector.
136  vect_citer = std::lower_bound(
137  vect_citer, sorted_vect.cend(), get_extra_data(*set_citer));
138  if (vect_citer == sorted_vect.cend()) {
139  break;
140  }
141  }
142  return;
143 }

145 // Slightly streamlined, because we return as soon as
146 // a common element is found.
147 template <class T>
148 bool disjoint(const std::set<T>& set1, const std::set<T>& set2) {
149  if (set1.empty() || set2.empty()) {
150  return true;
151  }
152 
153  // We always have x1 in S1.
154  auto citer1 = set1.cbegin();
155 
156  // Break when we run out of elements.
157  for (;;) {
158  // Get x2 >= x1, where x(j) in S(j).
159  auto citer2 = set2.lower_bound(*citer1);
160  if (citer2 == set2.cend()) {
161  break;
162  }
163  if (*citer1 == *citer2) {
164  // x1=x2; a common element.
165  return false;
166  }
167  // We now know that x2 > x1.
168  // Get x1' >= x2.
169  citer1 = set1.lower_bound(*citer2);
170  if (citer1 == set1.cend()) {
171  break;
172  }
173  if (*citer1 == *citer2) {
174  return false;
175  }
176  }
177  return true;
178 }
