48 bool SingleQubitSquash::squash() {
49  bool success = false;
50 
51  VertexVec inputs = circ_.q_inputs();
52  VertexVec outputs = circ_.q_outputs();
53  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
54  Edge in = circ_.get_nth_out_edge(inputs[i], 0);
55  Edge out = circ_.get_nth_in_edge(outputs[i], 0);
56  if (reversed_) {
57  success |= squash_between(out, in);
58  } else {
59  success |= squash_between(in, out);
60  }
61  }
62 
63  return success;
64 }

66 bool SingleQubitSquash::squash_between(const Edge &in, const Edge &out) {
67  squasher_->clear();
68  Edge e = in;
69  Vertex v = next_vertex(e);
70  std::vector<Gate_ptr> single_chain;
71  VertexVec bin;
72  bool success = false;
73  Condition condition = std::nullopt;
74  while (true) {
75  Op_ptr v_op = circ_.get_Op_ptr_from_Vertex(v);
76  OpType v_type = v_op->get_type();
77  bool move_to_next_vertex = false;
78  bool reset_search = false;
79  Condition this_condition = std::nullopt;
80 
81  if (v_type == OpType::Conditional) {
82  // => deal with conditional case
83  this_condition = get_condition(v);
84  v_op = static_cast<const Conditional &>(*v_op).get_op();
85  v_type = v_op->get_type();
86 
87  if (single_chain.empty()) {
88  condition = this_condition;
89  }
90  }
91 
92  bool is_squashable = circ_.n_in_edges_of_type(v, EdgeType::Quantum) == 1 &&
93  is_gate_type(v_type) &&
94  squasher_->accepts(as_gate_ptr(v_op));
95 
96  if (e != out && condition == this_condition && is_squashable) {
97  // => add gate to current squash
98  squasher_->append(as_gate_ptr(reversed_ ? v_op->dagger() : v_op));
99  move_to_next_vertex = true;
100  } else {
101  // => squash and reset
102  reset_search = true;
103  if (single_chain.empty()) {
104  // => nothing to do, move on
105  move_to_next_vertex = true;
106  } else {
107  Circuit sub;
108  std::optional<Pauli> commutation_colour = std::nullopt;
109  if (is_gate_type(v_type) && v_op->n_qubits() > 1) {
110  commutation_colour =
111  circ_.commuting_basis(v, PortType::Target, next_port(e));
112  move_to_next_vertex = true;
113  }
114  auto pair = squasher_->flush(commutation_colour);
115  sub = pair.first;
116  Gate_ptr left_over_gate = pair.second;
117  if (left_over_gate != nullptr) {
118  // => commute leftover through before squashing
119  insert_left_over_gate(left_over_gate, next_edge(v, e), condition);
120  left_over_gate = nullptr;
121  }
122  if (reversed_) {
123  sub = sub.dagger();
124  }
125 
126  // we squash if the replacement is at least as good as the original
127  // (and it's not a no-op)
128  if (sub_is_better(sub, single_chain)) {
129  substitute(sub, bin, e, condition);
130  success = true;
131  }
132  }
133  }
134  if (e == out || is_last_optype(v_type)) {
135  squasher_->clear();
136  break;
137  }
138  if (move_to_next_vertex) {
139  if (is_gate_type(v_type)) {
140  bin.push_back(v);
141  single_chain.push_back(as_gate_ptr(v_op));
142  }
143  e = next_edge(v, e);
144  v = next_vertex(e);
145  }
146  if (reset_search) {
147  bin.clear();
148  single_chain.clear();
149  squasher_->clear();
150  condition = std::nullopt;
151  }
152  }
153  return success;
154 }

156 void SingleQubitSquash::substitute(
157  const Circuit &sub, const VertexVec &single_chain, Edge &e,
158  const Condition &condition) {
159  // backup edge
160  VertPort backup = {next_vertex(e), next_port(e)};
161 
162  if (condition) {
163  circ_.substitute_conditional(
164  sub, single_chain.front(), Circuit::VertexDeletion::No);
165  } else {
166  circ_.substitute(sub, single_chain.front(), Circuit::VertexDeletion::No);
167  }
168  circ_.remove_vertices(
169  VertexSet{single_chain.begin(), single_chain.end()},
170  Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
171 
172  // restore backup
173  e = prev_edge(backup);
174 }

176 void SingleQubitSquash::insert_left_over_gate(
177  Op_ptr left_over, const Edge &e, const Condition &condition) {
178  if (reversed_) {
179  left_over = left_over->dagger();
180  }
181  EdgeVec preds;
182  op_signature_t sigs;
183  if (condition) {
184  left_over = std::make_shared<Conditional>(
185  left_over, (unsigned)condition->first.size(), condition->second);
186  }
187  Vertex new_v = circ_.add_vertex(left_over);
188  if (condition) {
189  for (const VertPort &vp : condition->first) {
190  preds.push_back(circ_.get_nth_out_edge(vp.first, vp.second));
191  sigs.push_back(EdgeType::Boolean);
192  }
193  }
194  preds.push_back(e);
195  sigs.push_back(EdgeType::Quantum);
196  circ_.rewire(new_v, preds, sigs);
197 }

199 bool SingleQubitSquash::sub_is_better(
200  const Circuit &sub, const std::vector<Gate_ptr> chain) const {
201  const unsigned n_gates = sub.n_gates();
202  return n_gates < chain.size() ||
203  (n_gates == chain.size() && !is_equal(sub, chain, reversed_));
204 }

206 // returns a description of the condition of current vertex
207 SingleQubitSquash::Condition SingleQubitSquash::get_condition(Vertex v) const {
208  Op_ptr v_op = circ_.get_Op_ptr_from_Vertex(v);
209  OpType v_type = v_op->get_type();
210  if (v_type != OpType::Conditional) {
211  throw BadOpType("Cannot get condition from non-conditional OpType", v_type);
212  }
213  const Conditional &cond_op = static_cast<const Conditional &>(*v_op);
214  EdgeVec ins = circ_.get_in_edges(v);
215  Condition cond = std::pair<std::list<VertPort>, unsigned>();
216  for (port_t p = 0; p < cond_op.get_width(); ++p) {
217  Edge in_p = ins.at(p);
218  VertPort vp = {circ_.source(in_p), circ_.get_source_port(in_p)};
219  cond->first.push_back(vp);
220  }
221  cond->second = cond_op.get_value();
222  return cond;
223 }

248 bool SingleQubitSquash::is_equal(
249  const Circuit &circ, const std::vector<Gate_ptr> &gates, bool reversed) {
250  if (reversed) {
251  return is_equal(circ, {gates.rbegin(), gates.rend()});
252  }
253  if (circ.n_qubits() != 1) {
254  throw CircuitInvalidity("Only circuits with one qubit are supported");
255  }
256 
257  auto it1 = circ.begin();
258  auto it2 = gates.cbegin();
259 
260  while (it1 != circ.end() && it2 != gates.end()) {
261  const Gate_ptr op1 = as_gate_ptr(it1->get_op_ptr());
262  const Gate_ptr op2 = as_gate_ptr(*it2);
263  if (!(*op1 == *op2)) {
264  return false;
265  }
266  ++it1;
267  ++it2;
268  }
269 
270  if (it1 != circ.end() || it2 != gates.cend()) {
271  return false;
272  }
273  return true;
274 }
