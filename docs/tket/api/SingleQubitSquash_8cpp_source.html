41 bool SingleQubitSquash::squash() {
42  bool success = false;
43 
44  VertexVec inputs = circ_.q_inputs();
45  VertexVec outputs = circ_.q_outputs();
46  for (unsigned i = 0; i < circ_.n_qubits(); ++i) {
47  Edge in = circ_.get_nth_out_edge(inputs[i], 0);
48  Edge out = circ_.get_nth_in_edge(outputs[i], 0);
49  if (reversed_) {
50  success |= squash_between(out, in);
51  } else {
52  success |= squash_between(in, out);
53  }
54  }
55 
56  return success;
57 }

59 bool SingleQubitSquash::squash_between(const Edge &in, const Edge &out) {
60  squasher_->clear();
61  Edge e = in;
62  Vertex v = next_vertex(e);
63  std::vector<Gate_ptr> single_chain;
64  VertexVec bin;
65  bool success = false;
66  Condition condition = std::nullopt;
67  while (true) {
68  Op_ptr v_op = circ_.get_Op_ptr_from_Vertex(v);
69  OpType v_type = v_op->get_type();
70  bool move_to_next_vertex = false;
71  bool reset_search = false;
72  Condition this_condition = std::nullopt;
73 
74  if (v_type == OpType::Conditional) {
75  // => deal with conditional case
76  this_condition = get_condition(v);
77  v_op = static_cast<const Conditional &>(*v_op).get_op();
78  v_type = v_op->get_type();
79 
80  if (single_chain.empty()) {
81  condition = this_condition;
82  }
83  }
84 
85  bool is_squashable = circ_.n_in_edges_of_type(v, EdgeType::Quantum) == 1 &&
86  is_gate_type(v_type) &&
87  squasher_->accepts(as_gate_ptr(v_op));
88 
89  if (e != out && condition == this_condition && is_squashable) {
90  // => add gate to current squash
91  squasher_->append(as_gate_ptr(reversed_ ? v_op->dagger() : v_op));
92  move_to_next_vertex = true;
93  } else {
94  // => squash and reset
95  reset_search = true;
96  if (single_chain.empty()) {
97  // => nothing to do, move on
98  move_to_next_vertex = true;
99  } else {
100  Circuit sub;
101  std::optional<Pauli> commutation_colour = std::nullopt;
102  if (is_gate_type(v_type) && v_op->n_qubits() > 1) {
103  commutation_colour =
104  circ_.commuting_basis(v, PortType::Target, next_port(e));
105  move_to_next_vertex = true;
106  }
107  auto pair = squasher_->flush(commutation_colour);
108  sub = pair.first;
109  Gate_ptr left_over_gate = pair.second;
110  if (left_over_gate != nullptr) {
111  // => commute leftover through before squashing
112  insert_left_over_gate(left_over_gate, next_edge(v, e), condition);
113  left_over_gate = nullptr;
114  }
115  if (reversed_) {
116  sub = sub.dagger();
117  }
118 
119  // we squash if the replacement is at least as good as the original
120  // (and it's not a no-op)
121  if (sub_is_better(sub, single_chain)) {
122  substitute(sub, bin, e, condition);
123  success = true;
124  }
125  }
126  }
127  if (e == out || is_last_optype(v_type)) {
128  squasher_->clear();
129  break;
130  }
131  if (move_to_next_vertex) {
132  if (is_gate_type(v_type)) {
133  bin.push_back(v);
134  single_chain.push_back(as_gate_ptr(v_op));
135  }
136  e = next_edge(v, e);
137  v = next_vertex(e);
138  }
139  if (reset_search) {
140  bin.clear();
141  single_chain.clear();
142  squasher_->clear();
143  condition = std::nullopt;
144  }
145  }
146  return success;
147 }

149 void SingleQubitSquash::substitute(
150  const Circuit &sub, const VertexVec &single_chain, Edge &e,
151  const Condition &condition) {
152  // backup edge
153  VertPort backup = {next_vertex(e), next_port(e)};
154 
155  if (condition) {
156  circ_.substitute_conditional(
157  sub, single_chain.front(), Circuit::VertexDeletion::No);
158  } else {
159  circ_.substitute(sub, single_chain.front(), Circuit::VertexDeletion::No);
160  }
161  circ_.remove_vertices(
162  VertexSet{single_chain.begin(), single_chain.end()},
163  Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
164 
165  // restore backup
166  e = prev_edge(backup);
167 }

169 void SingleQubitSquash::insert_left_over_gate(
170  Op_ptr left_over, const Edge &e, const Condition &condition) {
171  if (reversed_) {
172  left_over = left_over->dagger();
173  }
174  EdgeVec preds;
175  op_signature_t sigs;
176  if (condition) {
177  left_over = std::make_shared<Conditional>(
178  left_over, (unsigned)condition->first.size(), condition->second);
179  }
180  Vertex new_v = circ_.add_vertex(left_over);
181  if (condition) {
182  for (const VertPort &vp : condition->first) {
183  preds.push_back(circ_.get_nth_out_edge(vp.first, vp.second));
184  sigs.push_back(EdgeType::Boolean);
185  }
186  }
187  preds.push_back(e);
188  sigs.push_back(EdgeType::Quantum);
189  circ_.rewire(new_v, preds, sigs);
190 }

192 bool SingleQubitSquash::sub_is_better(
193  const Circuit &sub, const std::vector<Gate_ptr> chain) const {
194  const unsigned n_gates = sub.n_gates();
195  if (n_gates > chain.size()) {
196  return false;
197  }
198  if (!sub.is_symbolic() || always_squash_symbols_) {
199  return n_gates < chain.size() || !is_equal(sub, chain, reversed_);
200  }
201  // For symbolic circuits, we don't want to squash gates if it blows up the
202  // complexity of the expressions. As a crude but adequate measure, we compare
203  // the total size of the string representations.
204  return std::accumulate(
205  sub.begin(), sub.end(), std::size_t{0},
206  [](std::size_t a, const Command &cmd) {
207  return a + cmd.get_op_ptr()->get_name().size();
208  }) <
209  std::accumulate(
210  chain.begin(), chain.end(), std::size_t{0},
211  [](std::size_t a, const Gate_ptr &gpr) {
212  return a + gpr->get_name().size();
213  });
214 }

216 // returns a description of the condition of current vertex
217 SingleQubitSquash::Condition SingleQubitSquash::get_condition(Vertex v) const {
218  Op_ptr v_op = circ_.get_Op_ptr_from_Vertex(v);
219  OpType v_type = v_op->get_type();
220  if (v_type != OpType::Conditional) {
221  throw BadOpType("Cannot get condition from non-conditional OpType", v_type);
222  }
223  const Conditional &cond_op = static_cast<const Conditional &>(*v_op);
224  EdgeVec ins = circ_.get_in_edges(v);
225  Condition cond = std::pair<std::list<VertPort>, unsigned>();
226  for (port_t p = 0; p < cond_op.get_width(); ++p) {
227  Edge in_p = ins.at(p);
228  VertPort vp = {circ_.source(in_p), circ_.get_source_port(in_p)};
229  cond->first.push_back(vp);
230  }
231  cond->second = cond_op.get_value();
232  return cond;
233 }

258 bool SingleQubitSquash::is_equal(
259  const Circuit &circ, const std::vector<Gate_ptr> &gates, bool reversed) {
260  if (reversed) {
261  return is_equal(circ, {gates.rbegin(), gates.rend()});
262  }
263  if (circ.n_qubits() != 1) {
264  throw CircuitInvalidity("Only circuits with one qubit are supported");
265  }
266 
267  auto it1 = circ.begin();
268  auto it2 = gates.cbegin();
269 
270  while (it1 != circ.end() && it2 != gates.end()) {
271  const Gate_ptr op1 = as_gate_ptr(it1->get_op_ptr());
272  const Gate_ptr op2 = as_gate_ptr(*it2);
273  if (!(*op1 == *op2)) {
274  return false;
275  }
276  ++it1;
277  ++it2;
278  }
279 
280  if (it1 != circ.end() || it2 != gates.cend()) {
281  return false;
282  }
283  return true;
284 }
