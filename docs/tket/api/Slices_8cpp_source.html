22SliceIterator::SliceIterator(const Circuit& circ) : cut_(), circ_(&circ) {
23 cut_.init();
24 // add qubits to u_frontier
25 for (const Qubit& q : circ.all_qubits()) {
26 Vertex in = circ.get_in(q);
27 cut_.slice->push_back(in);
28 cut_.u_frontier->insert({q, circ.get_nth_out_edge(in, 0)});
29 }
30
31 // add bits to u_frontier and b_frontier
32 for (const Bit& b : circ.all_bits()) {
33 Vertex in = circ.get_in(b);
34 cut_.slice->push_back(in);
35 cut_.b_frontier->insert({b, circ.get_nth_b_out_bundle(in, 0)});
36 cut_.u_frontier->insert({b, circ.get_nth_out_edge(in, 0)});
37 }
38
39 // add WasmState to u_frontier
40 for (unsigned i = 0; i < circ._number_of_wasm_wires; ++i) {
41 Vertex in = circ.get_in(circ.wasmwire[i]);
42 cut_.slice->push_back(in);
43 cut_.u_frontier->insert({circ.wasmwire[i], circ.get_nth_out_edge(in, 0)});
44 }
45
46 prev_b_frontier_ = cut_.b_frontier;
47 cut_ = circ.next_cut(cut_.u_frontier, cut_.b_frontier);
48
49 // Add all vertices that have no Quantum or Classical edges (e.g. Phase) and
50 // no Boolean inputs:
51 BGL_FORALL_VERTICES(v, circ.dag, DAG) {
52 if (circ.n_in_edges(v) == 0 &&
53 circ.n_out_edges_of_type(v, EdgeType::Quantum) == 0 &&
54 circ.n_out_edges_of_type(v, EdgeType::Classical) == 0 &&
55 circ.n_out_edges_of_type(v, EdgeType::WASM) == 0) {
56 cut_.slice->push_back(v);
57 }
58 }
59}

110bool SliceIterator::finished() const {
111 for (const std::pair<UnitID, Edge>& pair :
112 this->cut_.u_frontier->get<TagKey>()) {
113 if (!circ_->detect_final_Op(circ_->target(pair.second))) return false;
114 }
115 for (const std::pair<Bit, EdgeVec>& pair :
116 this->cut_.b_frontier->get<TagKey>()) {
117 if (!pair.second.empty()) return false;
118 }
119 return true;
120}

122static std::shared_ptr<unit_frontier_t> get_next_u_frontier(
123 const Circuit& circ, std::shared_ptr<const unit_frontier_t> u_frontier,
124 const VertexSet& next_slice_lookup) {
125 std::shared_ptr<unit_frontier_t> next_frontier =
126 std::make_shared<unit_frontier_t>();
127 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
128 Vertex next_v = circ.target(pair.second);
129 if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) {
130 next_frontier->insert(pair);
131 } else {
132 next_frontier->insert(
133 {pair.first, circ.get_next_edge(next_v, pair.second)});
134 }
135 }
136 return next_frontier;
137}

139static std::shared_ptr<b_frontier_t> get_next_b_frontier(
140 const Circuit& circ, std::shared_ptr<const b_frontier_t> b_frontier,
141 std::shared_ptr<const unit_frontier_t> u_frontier,
142 const VertexSet& next_slice_lookup) {
143 std::shared_ptr<b_frontier_t> next_b_frontier =
144 std::make_shared<b_frontier_t>();
145 // Copy any remaining edges
146 for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
147 EdgeVec remaining;
148 for (const Edge& e : pair.second) {
149 Vertex targ = circ.target(e);
150 if (next_slice_lookup.find(targ) == next_slice_lookup.end()) {
151 remaining.push_back(e);
152 }
153 }
154 if (!remaining.empty()) {
155 next_b_frontier->insert({pair.first, remaining});
156 }
157 }
158 // Add any new bits introduced in this slice
159 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
160 switch (circ.get_edgetype(pair.second)) {
161 case EdgeType::Quantum:
162 case EdgeType::WASM: {
163 break;
164 }
165 case EdgeType::Classical: {
166 Vertex next_v = circ.target(pair.second);
167 if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) continue;
168 if (next_b_frontier->get<TagKey>().find(Bit(pair.first)) !=
169 next_b_frontier->end()) {
170 TKET_ASSERT(!"RAW hazard created in slicing");
171 }
172 port_t p = circ.get_target_port(pair.second);
173 EdgeVec reads = circ.get_nth_b_out_bundle(next_v, p);
174 if (!reads.empty()) next_b_frontier->insert({Bit(pair.first), reads});
175 break;
176 }
177 case EdgeType::Boolean: {
178 throw CircuitInvalidity("Boolen edge not allowed in b_frontier_t");
179 }
180 default: {
181 TKET_ASSERT(!"get_next_b_frontier found invalid edge type in sig");
182 }
183 }
184 }
185 return next_b_frontier;
186}

188CutFrontier Circuit::next_cut(
189 std::shared_ptr<const unit_frontier_t> u_frontier,
190 std::shared_ptr<const b_frontier_t> b_frontier,
191 const std::function<bool(Op_ptr)>& skip_func) const {
192 auto next_slice = std::make_shared<Slice>();
193 VertexSet next_slice_lookup;
194 VertexSet bad_vertices;
195 std::list<Edge> all_edges;
196 EdgeSet edge_lookup;
197 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
198 const UnitID& unit = pair.first;
199 const Edge& e = pair.second;
200 if (unit.type() == UnitType::Bit) {
201 Vertex targ = target(e);
202 b_frontier_t::const_iterator found =
203 b_frontier->get<TagKey>().find(Bit(unit));
204 if (found != b_frontier->get<TagKey>().end()) {
205 bool still_live = false;
206 for (const Edge& e : found->second) {
207 if (target(e) != targ) {
208 still_live = true;
209 break;
210 }
211 }
212 if (still_live) continue;
213 }
214 }
215 all_edges.push_back(e);
216 edge_lookup.insert(e);
217 }
218 for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
219 for (const Edge& e : pair.second) {
220 all_edges.push_back(e);
221 edge_lookup.insert(e);
222 }
223 }
224 if (skip_func) {
225 // advance through skippable
226 bool can_skip;
227 do {
228 can_skip = false;
229 VertexSet skip_slice_lookup;
230 for (const Edge& e : all_edges) {
231 Vertex try_v = target(e);
232 if (detect_final_Op(try_v) ||
233 (!skip_func(get_Op_ptr_from_Vertex(try_v))))
234 continue;
235 if (skip_slice_lookup.find(try_v) != skip_slice_lookup.end()) continue;
236 bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
237 if (!good_vertex) continue;
238 const EdgeVec ins = get_in_edges(try_v);
239 for (const Edge& in : ins) {
240 if (edge_lookup.find(in) == edge_lookup.end()) {
241 good_vertex = false;
242 break;
243 }
244 }
245 if (!good_vertex) {
246 bad_vertices.insert(try_v);
247 continue;
248 }
249 skip_slice_lookup.insert(try_v);
250 }
251 if (!skip_slice_lookup.empty()) {
252 b_frontier = get_next_b_frontier(
253 *this, b_frontier, u_frontier, skip_slice_lookup);
254 u_frontier = get_next_u_frontier(*this, u_frontier, skip_slice_lookup);
255 bad_vertices = {};
256 all_edges = {};
257 edge_lookup = {};
258
259 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
260 Edge e = pair.second;
261 all_edges.push_back(e);
262 edge_lookup.insert(e);
263 }
264 for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
265 for (const Edge& edge : pair.second) {
266 Edge e = edge;
267 all_edges.push_back(e);
268 edge_lookup.insert(e);
269 }
270 }
271 can_skip = true;
272 }
273 } while (can_skip);
274 }
275 // find the next slice first
276 for (const Edge& e : all_edges) {
277 Vertex try_v = target(e);
278 if (detect_final_Op(try_v)) continue;
279 if (next_slice_lookup.find(try_v) != next_slice_lookup.end())
280 continue; // already going to be in next slice
281 bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
282 if (!good_vertex) continue;
283 const EdgeVec ins = get_in_edges(try_v);
284 for (const Edge& in : ins) {
285 if (edge_lookup.find(in) == edge_lookup.end()) {
286 good_vertex = false;
287 bad_vertices.insert(try_v);
288 break;
289 }
290 }
291 if (good_vertex) {
292 next_slice_lookup.insert(try_v);
293 next_slice->push_back(try_v);
294 }
295 }
296 return {
297 next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
298 get_next_b_frontier(*this, b_frontier, u_frontier, next_slice_lookup)};
299}

301CutFrontier Circuit::next_q_cut(
302 std::shared_ptr<const unit_frontier_t> u_frontier) const {
303 auto next_slice = std::make_shared<Slice>();
304 VertexSet next_slice_lookup;
305 VertexSet bad_vertices;
306 EdgeSet edge_lookup;
307 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
308 edge_lookup.insert(pair.second);
309 }
310
311 // find the next slice first
312 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
313 Vertex try_v = target(pair.second);
314 if (detect_final_Op(try_v)) continue;
315 if (next_slice_lookup.contains(try_v))
316 continue; // already going to be in next slice
317 bool good_vertex = !bad_vertices.contains(try_v);
318 if (!good_vertex) continue;
319 EdgeVec ins = get_in_edges(try_v);
320 for (const Edge& in : ins) {
321 if (!edge_lookup.contains(in) && (get_edgetype(in) == EdgeType::Quantum ||
322 get_edgetype(in) == EdgeType::WASM)) {
323 good_vertex = false;
324 bad_vertices.insert(try_v);
325 break;
326 }
327 }
328 if (good_vertex) {
329 next_slice_lookup.insert(try_v);
330 next_slice->push_back(try_v);
331 }
332 }
333
334 return {
335 next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
336 std::make_shared<b_frontier_t>()};
337}
