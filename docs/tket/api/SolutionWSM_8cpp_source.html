24 // Fills "assignments_map" with the actual assignments made;
25 // fills "values" with the TV values seen.
26 static void check_assignments_for_value_clashes(
27  Assignments& assignments_map, const SolutionWSM& solution,
28  std::set<VertexWSM>& values, std::stringstream& ss) {
29  values.clear();
30  for (const std::pair<VertexWSM, VertexWSM>& entry : solution.assignments) {
31  const auto value_opt = get_optional_value(assignments_map, entry.first);
32  if (value_opt) {
33  ss << "\nRepeated PV: seen " << entry.first << "->" << entry.second
34  << " and " << entry.first << "->" << value_opt.value();
35  }
36  assignments_map[entry.first] = entry.second;
37 
38  if (values.count(entry.second) != 0) {
39  ss << "\nDuplicate value " << entry.second << " seen, when adding "
40  << entry.first << "->" << entry.second;
41  }
42  values.insert(entry.second);
43  }
44  if (assignments_map.size() != solution.assignments.size() ||
45  (values.size() != solution.assignments.size())) {
46  ss << "\nSizes mismatch: " << assignments_map.size() << ","
47  << solution.assignments.size() << "," << values.size();
48  }
49 }

51 std::string SolutionWSM::get_errors(
52  const GraphEdgeWeights& pattern_edges_and_weights,
53  const GraphEdgeWeights& target_edges_and_weights) const {
54  std::stringstream ss;
55  if (assignments.empty()) {
56  if (scalar_product != 0 || total_p_edges_weight != 0) {
57  ss << "empty assignments, but sc.prod=" << scalar_product
58  << ", total p.edge.weights=" << total_p_edges_weight;
59  }
60  return ss.str();
61  }
62  Assignments assignments_map;
63  std::set<VertexWSM> work_set;
64 
65  check_assignments_for_value_clashes(assignments_map, *this, work_set, ss);
66 
67  // Now, recalculate the weights, checking carefully for overflow.
68  WeightWSM total_expected_p_edge_weight = 0;
69  WeightWSM expected_scalar_product = 0;
70  std::set<VertexWSM>& p_vertices_used = work_set;
71  p_vertices_used.clear();
72 
73  try {
74  for (const auto& p_edge_data : pattern_edges_and_weights) {
75  const WeightWSM& p_edge_weight = p_edge_data.second;
76  total_expected_p_edge_weight =
77  get_sum_or_throw(total_expected_p_edge_weight, p_edge_weight);
78 
79  const EdgeWSM& p_edge = p_edge_data.first;
80  if (p_edge.first == p_edge.second) {
81  ss << "\nInvalid loop at PV=" << p_edge.first;
82  }
83  if (pattern_edges_and_weights.count(
84  std::make_pair(p_edge.second, p_edge.first)) != 0) {
85  ss << "\nRepeated pattern edge: (" << p_edge.first << ","
86  << p_edge.second << ") and reversed edge both present.";
87  }
88  p_vertices_used.insert(p_edge.first);
89  p_vertices_used.insert(p_edge.second);
90  const auto tv1_opt = get_optional_value(assignments_map, p_edge.first);
91  const auto tv2_opt = get_optional_value(assignments_map, p_edge.second);
92  if (!tv1_opt || !tv2_opt) {
93  ss << "\nP-edge (" << p_edge.first << "," << p_edge.second
94  << ") has unassigned vertices";
95  break;
96  }
97  const auto tv1 = tv1_opt.value();
98  const auto tv2 = tv2_opt.value();
99  if (tv1 == tv2) {
100  ss << "\nP vertices " << p_edge.first << "," << p_edge.second
101  << " both map to " << tv1;
102  break;
103  }
104  const auto t_edge = get_edge(tv1, tv2);
105  const auto t_weight_opt =
106  get_optional_value(target_edges_and_weights, t_edge);
107  if (!t_weight_opt) {
108  ss << "\nP-edge [" << p_edge.first << "," << p_edge.second
109  << "] maps to nonexistent target edge [" << tv1 << "," << tv2 << "]";
110  break;
111  }
112  const auto extra_sc_product =
113  get_product_or_throw(p_edge_weight, t_weight_opt.value());
114  expected_scalar_product =
115  get_sum_or_throw(expected_scalar_product, extra_sc_product);
116  }
117  } catch (const IntegerOverflow& overflow) {
118  ss << "\nInteger overflow occurred " << overflow.what();
119  }
120  if (expected_scalar_product != scalar_product ||
121  total_expected_p_edge_weight != total_p_edges_weight) {
122  ss << "\nWeights mismatch: scalar products " << expected_scalar_product
123  << "," << scalar_product << "; total p-edge weights "
124  << total_expected_p_edge_weight << "," << total_p_edges_weight;
125  }
126  if (p_vertices_used.size() != assignments.size()) {
127  ss << "\nNumber of used p vertices mismatch: " << p_vertices_used.size()
128  << "," << assignments.size();
129  }
130  return ss.str();
131 }
