30StandardSquasher::StandardSquasher(
31 const OpTypeSet &singleqs, const Func &tk1_replacement)
32 : singleqs_(singleqs),
33 squash_fn_(tk1_replacement),
34 combined_(),
35 phase_(0.) {
36 for (OpType ot : singleqs_) {
37 if (!is_single_qubit_type(ot))
38 throw BadOpType(
39 "OpType given to standard_squash is not a single qubit gate", ot);
40 }
41}

43bool StandardSquasher::accepts(OpType optype) const {
44 return (singleqs_.find(optype) != singleqs_.end()) &&
45 !is_projective_type(optype);
46}

48void StandardSquasher::append(Gate_ptr gp) {
49 std::vector<Expr> angs = gp->get_tk1_angles();
50 combined_.apply(Rotation(OpType::Rz, angs.at(2)));
51 combined_.apply(Rotation(OpType::Rx, angs.at(1)));
52 combined_.apply(Rotation(OpType::Rz, angs.at(0)));
53 phase_ += angs.at(3);
54}

56std::pair<Circuit, Gate_ptr> StandardSquasher::flush(
57 std::optional<Pauli>) const {
58 auto [a, b, c] = combined_.to_pqp(OpType::Rz, OpType::Rx);
59 Circuit replacement = squash_fn_(c, b, a);
60 BGL_FORALL_VERTICES(rv, replacement.dag, DAG) {
61 OpType v_type = replacement.get_OpType_from_Vertex(rv);
62 if (!is_boundary_q_type(v_type) &&
63 singleqs_.find(v_type) == singleqs_.end()) {
64 throw BadOpType(
65 "tk1_replacement given to standard_squash "
66 "does not preserve gate set",
67 v_type);
68 }
69 }
70 replacement.add_phase(phase_);
71 return {replacement, nullptr};
72}

83static bool standard_squash(
84 Circuit &circ, const OpTypeSet &singleqs,
85 const std::function<Circuit(const Expr &, const Expr &, const Expr &)>
86 &tk1_replacement,
87 bool always_squash_symbols) {
88 auto squasher = std::make_unique<StandardSquasher>(singleqs, tk1_replacement);
89 return SingleQubitSquash(
90 std::move(squasher), circ, false, always_squash_symbols)
91 .squash();
92}

94Transform squash_factory(
95 const OpTypeSet &singleqs,
96 const std::function<Circuit(const Expr &, const Expr &, const Expr &)>
97 &tk1_replacement,
98 bool always_squash_symbols) {
99 return Transform([=](Circuit &circ) {
100 return standard_squash(
101 circ, singleqs, tk1_replacement, always_squash_symbols);
102 });
