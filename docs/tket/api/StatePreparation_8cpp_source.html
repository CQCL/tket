29 static unsigned compute_log2(std::size_t length) {
30  if (length < 2 || (length & (length - 1)) != 0) {
31  throw std::invalid_argument(
32  "The length of the statevector is not a power of 2.");
33  }
34  return (unsigned)log2(length);
35 }

91 static Circuit state_prep_circ(
92  const Eigen::VectorXcd &statevector, bool is_inverse,
93  bool with_initial_reset) {
94  unsigned n_qubits = compute_log2(statevector.size());
95  Circuit circ(n_qubits);
96  if (with_initial_reset) {
97  circ.qubit_create_all();
98  }
99  std::vector<std::optional<MultiplexedRotationBox>> mutip_ry_vec, mutip_rz_vec;
100  Eigen::VectorXcd psi = statevector;
101  for (unsigned step = 0; step < n_qubits; step++) {
102  unsigned half_length = (unsigned)psi.size() / 2;
103  // in each step, we disentangle one qubit from |psi> so that
104  // multip_ry*multip_rz*|psi> = rs*exp(ts*i*pi)|0>
105  // multip_ry and multip_rz are multiplexed Ry and Rz respectively
106  // after this loop, we apply the multiplexors in reverse order to obtain the
107  // state prep circuit (if is_inverse is false)
108  ctrl_op_map_t multip_ry, multip_rz;
109  Eigen::VectorXd rs = Eigen::VectorXd::Zero(half_length);
110  Eigen::VectorXd ts = Eigen::VectorXd::Zero(half_length);
111  for (unsigned c = 0; c < half_length; c++) {
112  Complex a = psi[2 * c];
113  Complex b = psi[2 * c + 1];
114  if (std::abs(a) < EPS && std::abs(b) < EPS) {
115  // if both a and b are zeros
116  continue;
117  }
118  // normalise [a,b]
119  double r = 1;
120  double norm = a.real() * a.real() + a.imag() * a.imag() +
121  b.real() * b.real() + b.imag() * b.imag();
122  if (std::abs(norm - 1) > EPS) {
123  r = std::sqrt(norm);
124  a = a / r;
125  b = b / r;
126  }
127  double theta, phi, phase;
128  std::tie(theta, phi, phase) = get_bloch_coordinate_from_state(a, b);
129  ts[c] = phase + 0.5 * phi;
130  rs[c] = r;
131  unsigned n_controls = n_qubits - step - 1;
132  std::vector<bool> bitstr;
133  if (n_controls > 0) {
134  bitstr = dec_to_bin(c, n_controls);
135  }
136 
137  double y_angle = is_inverse ? -theta : theta;
138  double z_angle = is_inverse ? -phi : phi;
139  if (std::abs(y_angle) > EPS) {
140  multip_ry.insert({bitstr, get_op_ptr(OpType::Ry, y_angle)});
141  }
142  if (std::abs(z_angle) > EPS) {
143  multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, z_angle)});
144  }
145  }
146  if (!multip_ry.empty()) {
147  mutip_ry_vec.push_back(MultiplexedRotationBox(multip_ry));
148  } else {
149  mutip_ry_vec.push_back(std::nullopt);
150  }
151  if (!multip_rz.empty()) {
152  mutip_rz_vec.push_back(MultiplexedRotationBox(multip_rz));
153  } else {
154  mutip_rz_vec.push_back(std::nullopt);
155  }
156  if (step == n_qubits - 1) {
157  TKET_ASSERT(ts.size() == 1);
158  double circ_phase = is_inverse ? -ts[0] : ts[0];
159  circ.add_phase(circ_phase);
160  break;
161  }
162  psi = rs.array() * (ts.array() * i_ * PI).exp();
163  }
164  TKET_ASSERT(mutip_rz_vec.size() == n_qubits);
165  // add multiplexors;
166  if (!is_inverse) {
167  for (unsigned i = n_qubits; i-- > 0;) {
168  std::vector<unsigned> args(n_qubits - i);
169  std::iota(std::begin(args), std::end(args), 0);
170  if (mutip_ry_vec[i] != std::nullopt) {
171  circ.add_box(mutip_ry_vec[i].value(), args);
172  }
173  if (mutip_rz_vec[i] != std::nullopt) {
174  circ.add_box(mutip_rz_vec[i].value(), args);
175  }
176  }
177  } else {
178  for (unsigned i = 0; i < n_qubits; i++) {
179  std::vector<unsigned> args(n_qubits - i);
180  std::iota(std::begin(args), std::end(args), 0);
181  if (mutip_rz_vec[i] != std::nullopt) {
182  circ.add_box(mutip_rz_vec[i].value(), args);
183  }
184  if (mutip_ry_vec[i] != std::nullopt) {
185  circ.add_box(mutip_ry_vec[i].value(), args);
186  }
187  }
188  }
189  return circ;
190 };

197 bool StatePreparationBox::is_equal(const Op &op_other) const {
198  const StatePreparationBox &other =
199  dynamic_cast<const StatePreparationBox &>(op_other);
200  if (id_ == other.get_id()) return true;
201  return is_inverse_ == other.is_inverse_ &&
202  with_initial_reset_ == other.with_initial_reset_ &&
203  statevector_.isApprox(other.statevector_);
