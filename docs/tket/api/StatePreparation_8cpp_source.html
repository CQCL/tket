28static unsigned compute_log2(std::size_t length) {
29 if (length < 2 || (length & (length - 1)) != 0) {
30 throw std::invalid_argument(
31 "The length of the statevector is not a power of 2.");
32 }
33 return (unsigned)log2(length);
34}

90static Circuit state_prep_circ(
91 const Eigen::VectorXcd &statevector, bool is_inverse,
92 bool with_initial_reset) {
93 unsigned n_qubits = compute_log2(statevector.size());
94 Circuit circ(n_qubits);
95 if (with_initial_reset) {
96 circ.qubit_create_all();
97 }
98 std::vector<std::optional<MultiplexedRotationBox>> mutip_ry_vec, mutip_rz_vec;
99 Eigen::VectorXcd psi = statevector;
100 for (unsigned step = 0; step < n_qubits; step++) {
101 unsigned half_length = (unsigned)psi.size() / 2;
102 // in each step, we disentangle one qubit from |psi> so that
103 // multip_ry*multip_rz*|psi> = rs*exp(ts*i*pi)|0>
104 // multip_ry and multip_rz are multiplexed Ry and Rz respectively
105 // after this loop, we apply the multiplexors in reverse order to obtain the
106 // state prep circuit (if is_inverse is false)
107 ctrl_op_map_t multip_ry, multip_rz;
108 Eigen::VectorXd rs = Eigen::VectorXd::Zero(half_length);
109 Eigen::VectorXd ts = Eigen::VectorXd::Zero(half_length);
110 for (unsigned c = 0; c < half_length; c++) {
111 Complex a = psi[2 * c];
112 Complex b = psi[2 * c + 1];
113 if (std::abs(a) < EPS && std::abs(b) < EPS) {
114 // if both a and b are zeros
115 continue;
116 }
117 // normalise [a,b]
118 double r = 1;
119 double norm = a.real() * a.real() + a.imag() * a.imag() +
120 b.real() * b.real() + b.imag() * b.imag();
121 if (std::abs(norm - 1) > EPS) {
122 r = std::sqrt(norm);
123 a = a / r;
124 b = b / r;
125 }
126 double theta, phi, phase;
127 std::tie(theta, phi, phase) = get_bloch_coordinate_from_state(a, b);
128 ts[c] = phase + 0.5 * phi;
129 rs[c] = r;
130 unsigned n_controls = n_qubits - step - 1;
131 std::vector<bool> bitstr;
132 if (n_controls > 0) {
133 bitstr = dec_to_bin(c, n_controls);
134 }
135
136 double y_angle = is_inverse ? -theta : theta;
137 double z_angle = is_inverse ? -phi : phi;
138 if (std::abs(y_angle) > EPS) {
139 multip_ry.insert({bitstr, get_op_ptr(OpType::Ry, y_angle)});
140 }
141 if (std::abs(z_angle) > EPS) {
142 multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, z_angle)});
143 }
144 }
145 if (!multip_ry.empty()) {
146 mutip_ry_vec.push_back(MultiplexedRotationBox(multip_ry));
147 } else {
148 mutip_ry_vec.push_back(std::nullopt);
149 }
150 if (!multip_rz.empty()) {
151 mutip_rz_vec.push_back(MultiplexedRotationBox(multip_rz));
152 } else {
153 mutip_rz_vec.push_back(std::nullopt);
154 }
155 if (step == n_qubits - 1) {
156 TKET_ASSERT(ts.size() == 1);
157 double circ_phase = is_inverse ? -ts[0] : ts[0];
158 circ.add_phase(circ_phase);
159 break;
160 }
161 psi = rs.array() * (ts.array() * i_ * PI).exp();
162 }
163 TKET_ASSERT(mutip_rz_vec.size() == n_qubits);
164 // add multiplexors;
165 if (!is_inverse) {
166 for (unsigned i = n_qubits; i-- > 0;) {
167 std::vector<unsigned> args(n_qubits - i);
168 std::iota(std::begin(args), std::end(args), 0);
169 if (mutip_ry_vec[i] != std::nullopt) {
170 circ.add_box(mutip_ry_vec[i].value(), args);
171 }
172 if (mutip_rz_vec[i] != std::nullopt) {
173 circ.add_box(mutip_rz_vec[i].value(), args);
174 }
175 }
176 } else {
177 for (unsigned i = 0; i < n_qubits; i++) {
178 std::vector<unsigned> args(n_qubits - i);
179 std::iota(std::begin(args), std::end(args), 0);
180 if (mutip_rz_vec[i] != std::nullopt) {
181 circ.add_box(mutip_rz_vec[i].value(), args);
182 }
183 if (mutip_ry_vec[i] != std::nullopt) {
184 circ.add_box(mutip_ry_vec[i].value(), args);
185 }
186 }
187 }
188 return circ;
189};

191void StatePreparationBox::generate_circuit() const {
192 circ_ = std::make_shared<Circuit>(
193 state_prep_circ(statevector_, is_inverse_, with_initial_reset_));

196bool StatePreparationBox::is_equal(const Op &op_other) const {
197 const StatePreparationBox &other =
198 dynamic_cast<const StatePreparationBox &>(op_other);
199 if (id_ == other.get_id()) return true;
200 return is_inverse_ == other.is_inverse_ &&
201 with_initial_reset_ == other.with_initial_reset_ &&
202 statevector_.isApprox(other.statevector_);
203}
