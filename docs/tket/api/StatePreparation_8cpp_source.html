28 StatePreparationBox::StatePreparationBox(

30  : Box(OpType::StatePreparationBox),

40  throw std::invalid_argument("The input statevector is not normalised.");

74 static Circuit state_prep_circ(
75  const Eigen::VectorXcd &statevector, bool is_inverse) {
76  unsigned n_qubits = (unsigned)log2(statevector.size());
77  Circuit circ(n_qubits);
78  std::vector<std::optional<MultiplexedRotationBox>> mutip_ry_vec, mutip_rz_vec;
79  Eigen::VectorXcd psi = statevector;
80  for (unsigned step = 0; step < n_qubits; step++) {
81  unsigned half_length = (unsigned)psi.size() / 2;
82  // in each step, we disentangle one qubit from |psi> so that
83  // multip_ry*multip_rz*|psi> = rs*exp(ts*i*pi)|0>
84  // multip_ry and multip_rz are multiplexed Ry and Rz respectively
85  // after this loop, we apply the multiplexors in reverse order to obtain the
86  // state prep circuit (if is_inverse is false)
87  ctrl_op_map_t multip_ry, multip_rz;
88  Eigen::VectorXd rs = Eigen::VectorXd::Zero(half_length);
89  Eigen::VectorXd ts = Eigen::VectorXd::Zero(half_length);
90  for (unsigned c = 0; c < half_length; c++) {
91  Complex a = psi[2 * c];
92  Complex b = psi[2 * c + 1];
93  if (std::abs(a) < EPS && std::abs(b) < EPS) {
94  // if both a and b are zeros
95  continue;
96  }
97  // normalise [a,b]
98  double r = 1;
99  double norm = a.real() * a.real() + a.imag() * a.imag() +
100  b.real() * b.real() + b.imag() * b.imag();
101  if (std::abs(norm - 1) > EPS) {
102  r = std::sqrt(norm);
103  a = a / r;
104  b = b / r;
105  }
106  double theta, phi, phase;
107  std::tie(theta, phi, phase) = get_bloch_coordinate_from_state(a, b);
108  ts[c] = phase + 0.5 * phi;
109  rs[c] = r;
110  unsigned n_controls = n_qubits - step - 1;
111  std::vector<bool> bitstr;
112  if (n_controls > 0) {
113  bitstr = dec_to_bin(c, n_controls);
114  }
115 
116  double y_angle = is_inverse ? -theta : theta;
117  double z_angle = is_inverse ? -phi : phi;
118  if (std::abs(y_angle) > EPS) {
119  multip_ry.insert({bitstr, get_op_ptr(OpType::Ry, y_angle)});
120  }
121  if (std::abs(z_angle) > EPS) {
122  multip_rz.insert({bitstr, get_op_ptr(OpType::Rz, z_angle)});
123  }
124  }
125  if (!multip_ry.empty()) {
126  mutip_ry_vec.push_back(MultiplexedRotationBox(multip_ry));
127  } else {
128  mutip_ry_vec.push_back(std::nullopt);
129  }
130  if (!multip_rz.empty()) {
131  mutip_rz_vec.push_back(MultiplexedRotationBox(multip_rz));
132  } else {
133  mutip_rz_vec.push_back(std::nullopt);
134  }
135  if (step == n_qubits - 1) {
136  TKET_ASSERT(ts.size() == 1);
137  double circ_phase = is_inverse ? -ts[0] : ts[0];
138  circ.add_phase(circ_phase);
139  break;
140  }
141  psi = rs.array() * (ts.array() * i_ * PI).exp();
142  }
143  TKET_ASSERT(mutip_rz_vec.size() == n_qubits);
144  // add multiplexors;
145  if (!is_inverse) {
146  for (unsigned i = n_qubits; i-- > 0;) {
147  std::vector<unsigned> args(n_qubits - i);
148  std::iota(std::begin(args), std::end(args), 0);
149  if (mutip_ry_vec[i] != std::nullopt) {
150  circ.add_box(mutip_ry_vec[i].value(), args);
151  }
152  if (mutip_rz_vec[i] != std::nullopt) {
153  circ.add_box(mutip_rz_vec[i].value(), args);
154  }
155  }
156  } else {
157  for (unsigned i = 0; i < n_qubits; i++) {
158  std::vector<unsigned> args(n_qubits - i);
159  std::iota(std::begin(args), std::end(args), 0);
160  if (mutip_rz_vec[i] != std::nullopt) {
161  circ.add_box(mutip_rz_vec[i].value(), args);
162  }
163  if (mutip_ry_vec[i] != std::nullopt) {
164  circ.add_box(mutip_ry_vec[i].value(), args);
165  }
166  }
167  }
168  return circ;
169 };
