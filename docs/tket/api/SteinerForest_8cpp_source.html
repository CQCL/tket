36 SteinerForest::SteinerForest(
37  const PathHandler &paths, const PhasePolyBox &phasepolybox) {
38  global_cost = 0;
39  unsigned id_counter = 0;
40  const PhasePolynomial &phasepoly = phasepolybox.get_phase_polynomial();
41  const MatrixXb &linear_fn_matrix_ppb =
42  phasepolybox.get_linear_transformation();
43  unsigned n_qubits = phasepolybox.get_n_qubits();
44  synth_circuit = Circuit(n_qubits);
45  linear_function = DiagMatrix(linear_fn_matrix_ppb);
46 
47  for (const auto &parity : phasepoly) {
48  ParityList parity_list = parity_column_to_list(parity.first);
49  SteinerTree tree(paths, parity_list, *parity_list.begin());
50  global_cost += tree.tree_cost;
51  std::pair<SteinerTree, Expr> parity_on_graph{tree, parity.second};
52 
53  CostedTrees::iterator iter = current_trees.find(tree.tree_cost);
54  if (iter == current_trees.end())
55  current_trees.insert({tree.tree_cost, {parity_on_graph}});
56  else
57  iter->second.push_back(parity_on_graph);
58 
59  ++id_counter;
60  }
61  tree_count = id_counter;
62 
63  CostedTrees new_trees;
64  for (const auto &cost_trees : current_trees) {
65  for (auto tree_expr : cost_trees.second) {
66  /* If we come across a tree which has only 1 node, check that its
67  properties are correct
68  and remove it from the forest, leaving an Rz in the circuit. */
69  if (tree_expr.first.fully_reduced()) {
70  unsigned qubit_id = UINT_MAX;
71  for (SteinerNodeType q : tree_expr.first.node_types) {
72  if (q == SteinerNodeType::Leaf) {
73  TKET_ASSERT(qubit_id == UINT_MAX); // Two nodes found to add the Rz
74  qubit_id = tree_expr.first.root;
75  }
76  }
77 
78  TKET_ASSERT(qubit_id != UINT_MAX); // No node found to add the Rz
79 
80  std::vector<unsigned> qubit{qubit_id};
81 
82  synth_circuit.add_op(OpType::Rz, tree_expr.second, qubit);
83  --tree_count;
84  /* Otherwise, the tree stays in. */
85  } else {
86  CostedTrees::iterator iter = new_trees.find(tree_expr.first.tree_cost);
87  if (iter == new_trees.end())
88  new_trees.insert({tree_expr.first.tree_cost, {tree_expr}});
89  else
90  iter->second.push_back(tree_expr);
91  unsigned cost_change = tree_expr.first.last_operation_cost;
92  global_cost += cost_change;
93  }
94  }
95  }
96  current_trees = new_trees;
97 }

99 void SteinerForest::add_row_globally(unsigned i, unsigned j) {
100  /* CNOT with control j and target i. Which way round the indices are is a
101  * wee bit fiddly. */
102  std::vector<unsigned> qbs = {j, i};
103  synth_circuit.add_op(OpType::CX, qbs);
104  linear_function.col_add(
105  i, j); // prepend a CNOT to the linear reversible function
106 
107  CostedTrees new_trees;
108 
109  for (const auto &cost_trees : current_trees) {
110  for (auto tree_expr : cost_trees.second) {
111  tree_expr.first.add_row(i, j);
112  /* If the tree has been fully reduced, remove it from the
113  forest */
114  if (tree_expr.first.fully_reduced()) {
115  std::vector<unsigned> qubit{i};
116  synth_circuit.add_op(OpType::Rz, tree_expr.second, qubit);
117  --tree_count;
118  }
119  /* Otherwise, update the forest */
120  else {
121  CostedTrees::iterator iter = new_trees.find(tree_expr.first.tree_cost);
122  if (iter == new_trees.end())
123  new_trees.insert({tree_expr.first.tree_cost, {tree_expr}});
124  else
125  iter->second.push_back(tree_expr);
126  unsigned cost_change = tree_expr.first.last_operation_cost;
127  global_cost += cost_change;
128  }
129  }
130  }
131  current_trees = new_trees;
132 }

181 CostedOperations best_operations_lookahead(
182  const PathHandler &path, const SteinerForest &forest, unsigned lookahead) {
183  if (lookahead == 0) {
184  throw std::logic_error("Must look ahead at least one step");
185  }
186  CostedTrees::const_reverse_iterator r_iter = forest.current_trees.rbegin();
187  if (r_iter == forest.current_trees.rend()) {
188  throw std::logic_error("Forest is empty");
189  }
190 
191  OperationList operations_available =
192  forest.operations_available_at_min_costs(path);
193 
194  TKET_ASSERT(!operations_available.empty()); // Cannot find any operations
195 
196  // set up base case
197  OperationList ops{operations_available.front()};
198  CostedOperations costed_operations =
199  recursive_operation_search(path, forest, lookahead - 1, ops);
200  operations_available.pop_front();
201 
202  for (OperationList::iterator op_iter = operations_available.begin();
203  op_iter != operations_available.end(); ++op_iter) {
204  ops = {*op_iter};
205  CostedOperations candidate_operations =
206  recursive_operation_search(path, forest, lookahead - 1, ops);
207 
208  if ((candidate_operations.first < costed_operations.first) ||
209  ((candidate_operations.first == costed_operations.first) &&
210  (candidate_operations.second.size() <
211  costed_operations.second.size()))) {
212  costed_operations = std::move(candidate_operations);
213  }
214  }
215 
216  return costed_operations;
217 }

219 CostedOperations recursive_operation_search(
220  const PathHandler &path, SteinerForest forest, unsigned lookahead,
221  OperationList row_operations) {
222  CostedOperations costed_operations;
223  CostedOperations candidate_operations;
224 
225  forest.add_row_globally(
226  row_operations.back().first, row_operations.back().second);
227 
228  if ((lookahead == 0) || (forest.current_trees.empty())) {
229  return {forest.global_cost, row_operations};
230  }
231  CostedTrees::const_reverse_iterator r_iter = forest.current_trees.rbegin();
232  unsigned index = r_iter->first;
233  OperationList operations_available =
234  forest.operations_available_under_the_index(path, index);
235  if (operations_available.empty()) {
236  return {forest.global_cost, row_operations};
237  } else {
238  row_operations.push_back(operations_available.front());
239  costed_operations =
240  recursive_operation_search(path, forest, lookahead - 1, row_operations);
241  row_operations.pop_back();
242  operations_available.pop_front();
243 
244  for (OperationList::iterator op_iter = operations_available.begin();
245  op_iter != operations_available.end(); ++op_iter) {
246  row_operations.push_back(*op_iter);
247  candidate_operations = recursive_operation_search(
248  path, forest, lookahead - 1, row_operations);
249 
250  row_operations.pop_back();
251 
252  if ((candidate_operations.first < costed_operations.first) ||
253  ((candidate_operations.first == costed_operations.first) &&
254  (candidate_operations.second.size() <
255  costed_operations.second.size()))) {
256  costed_operations = std::move(candidate_operations);
257  }
258  }
259  return costed_operations;
260  }

263 Circuit phase_poly_synthesis_int(
264  const Architecture &arch, const PhasePolyBox &phasepolybox,
265  unsigned lookahead, CNotSynthType cnottype) {
266  if (lookahead == 0)
267  throw std::logic_error(
268  "[AAS] the lookahead of the phase polynominal synthesis has to be "
269  "greater than 0");
270 
271  CostedOperations best_operations;
272 
273  PathHandler path(arch);
274 
275  PathHandler acyclic_path = path.construct_acyclic_handler();
276 
277  SteinerForest forest(acyclic_path, phasepolybox);
278 
279  while (!forest.current_trees.empty()) {
280  best_operations =
281  best_operations_lookahead(acyclic_path, forest, lookahead);
282  forest.add_operation_list(best_operations.second);
283  }
284  Circuit cnot_circ(path.get_size());
285  switch (cnottype) {
286  case CNotSynthType::HamPath: {
287  cnot_circ =
288  aas_CNOT_synth(forest.linear_function, path, CNotSynthType::HamPath);
289 
290  // check if matrix is id, if not the result will be wrong
291  TKET_ASSERT(forest.linear_function.is_id());
292  break;
293  }
294  case CNotSynthType::Rec: {
295  Circuit resultofstep =
296  aas_CNOT_synth(forest.linear_function, path, CNotSynthType::Rec);
297  cnot_circ = cnot_circ >> resultofstep;
298 
299  // check if matrix is id, if not the result will be wrong
300  TKET_ASSERT(forest.linear_function.is_id());
301  break;
302  }
303  case CNotSynthType::SWAP: {
304  cnot_circ = aas_CNOT_synth_SWAP(forest.linear_function, path);
305  // the check if the forest.linear_function is id is executed in the
306  // aas_CNOT_synth_SWAP function
307  break;
308  }
309  default: {
310  TKET_ASSERT(!"[AAS]: unknown type of cnot synth");
311  }
312  }
313 
314  return forest.synth_circuit >> cnot_circ.dagger();
315 }

317 static PhasePolyBox make_placed_ppb(
318  const Architecture &arch, const PhasePolyBox &phasepolybox) {
319  Circuit circuit_ppb_place =
320  phasepolybox.generate_circuit_with_original_placement();
321 
322  const std::string register_name = "surplus";
323 
324  unsigned qb_counter = circuit_ppb_place.n_qubits();
325  while (arch.n_nodes() > circuit_ppb_place.n_qubits()) {
326  Qubit qb = Qubit(register_name, qb_counter);
327  circuit_ppb_place.add_qubit(qb);
328  ++qb_counter;
329  }
330 
331  TKET_ASSERT(circuit_ppb_place.n_qubits() == arch.n_nodes());
332 
333  qubit_vector_t q_vec_place = circuit_ppb_place.all_qubits();
334  std::map<Qubit, Node> qubit_to_nodes_place;
335  unsigned counter_place = 0;
336 
337  for (Node no_place : arch.nodes()) {
338  if (counter_place < circuit_ppb_place.n_qubits()) {
339  qubit_to_nodes_place.insert({q_vec_place[counter_place], no_place});
340  ++counter_place;
341  }
342  }
343 
344  circuit_ppb_place.rename_units(qubit_to_nodes_place);
345 
346  return PhasePolyBox(circuit_ppb_place);
347 }

349 class PhasePolySynthesizer {
350  public:
351  PhasePolySynthesizer(
352  const Architecture &arch, const PhasePolyBox &phasepolybox,
353  unsigned lookahead, CNotSynthType cnottype)
354  : arch(arch),
355  placed_ppb(make_placed_ppb(arch, phasepolybox)),
356  lookahead(lookahead),
357  cnottype(cnottype) {}
358 
359  Circuit get_result() {
360  return (cnottype == CNotSynthType::HamPath) ? get_result_using_hampath()
361  : get_result_standard();
362  }
363 
364  private:
365  Circuit get_result_from_hampath(const std::vector<Node> &hampath) {
366  // maps from qubits/node to int
367  std::map<UnitID, UnitID> forward_contiguous_uids_q;
368  std::map<UnitID, UnitID> backward_contiguous_uids_n;
369  // extra map with node type needed for the creation of the architecture
370  std::map<UnitID, Node> unitid_to_int_nodes;
371  for (unsigned i = 0; i < hampath.size(); i++) {
372  UnitID qu_no = UnitID(hampath[i]);
373  Qubit q = Qubit(i);
374  Node n = Node(i);
375  unitid_to_int_nodes.insert({qu_no, n});
376  forward_contiguous_uids_q.insert({q, qu_no});
377  backward_contiguous_uids_n.insert({qu_no, n});
378  }
379  // define new arcitecture
380  std::vector<Architecture::Connection> new_con;
381  for (auto pair : arch.get_all_edges_vec()) {
382  new_con.push_back(
383  {unitid_to_int_nodes[UnitID(pair.first)],
384  unitid_to_int_nodes[UnitID(pair.second)]});
385  }
386  Architecture con_arch(new_con);
387  return make_result_from_con_arch(
388  con_arch, forward_contiguous_uids_q, backward_contiguous_uids_n);
389  }

391  Circuit get_result_using_hampath() {
392  std::vector<Node> hampath = find_hampath(arch); // using default timeout
393  Circuit c0 = get_result_from_hampath(hampath);
394  // Sometimes the reversed path gives a better circuit. Try both!
395  std::reverse(hampath.begin(), hampath.end());
396  Circuit c1 = get_result_from_hampath(hampath);
397  return (c0.depth() < c1.depth()) ? c0 : c1;
398  }

400  Circuit get_result_standard() {
401  // calculate iteration order
402  IterationOrder iter_order(arch);
403  std::vector<Node> node_order = iter_order.get_iterationorder();
404  // maps from qubits/node to int
405  std::map<UnitID, UnitID> forward_contiguous_uids_q;
406  std::map<UnitID, UnitID> backward_contiguous_uids_n;
407  // extra map with node type needed for the creation of the architecture
408  std::map<UnitID, Node> unitid_to_int_nodes;
409  for (unsigned i = 0; i < node_order.size(); i++) {
410  // convert node to superclass type of qubit/node
411  UnitID qu_no = UnitID(node_order[i]);
412  Qubit q = Qubit(i);
413  Node n = Node(i);
414  forward_contiguous_uids_q.insert({q, qu_no});
415  backward_contiguous_uids_n.insert({qu_no, n});
416  unitid_to_int_nodes.insert({qu_no, n});
417  }
418  // define new arcitecture: include only the tree edges
419  std::vector<Architecture::Connection> new_con;
420  for (auto pair : iter_order.get_edgelist()) {
421  new_con.push_back(
422  {unitid_to_int_nodes[UnitID(pair.first)],
423  unitid_to_int_nodes[UnitID(pair.second)]});
424  }
425  Architecture con_arch(new_con);
426  return make_result_from_con_arch(
427  con_arch, forward_contiguous_uids_q, backward_contiguous_uids_n);
428  }

430  Circuit make_result_from_con_arch(
431  const Architecture &con_arch,
432  const std::map<UnitID, UnitID> &forward_contiguous_uids_q,
433  const std::map<UnitID, UnitID> &backward_contiguous_uids_n) {
434  // define new phase poly box
435  Circuit circuit_ppb = placed_ppb.generate_circuit_with_original_placement();
436  // the aas code is implemented under the assumption that all qubits in the
437  // circuit are named from 0 to n. The same assumption was made for the nodes
438  // of the architecture. To make sure that this condition is fulfilled the
439  // qubits in the circuit and the architecture are renamed. The new names are
440  // reverted at the end of the aas procedure. The qubits and the nodes have
441  // the same name in the input.
442  circuit_ppb.rename_units(backward_contiguous_uids_n);
443  PhasePolyBox new_ppb(circuit_ppb);
444  Circuit result =
445  phase_poly_synthesis_int(con_arch, new_ppb, lookahead, cnottype);
446  result.rename_units(forward_contiguous_uids_q);
447  return result;
448  }
449 
450  Architecture arch;
451  PhasePolyBox placed_ppb;
452  unsigned lookahead;
453  CNotSynthType cnottype;
454 };
