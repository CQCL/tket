36 SteinerForest::SteinerForest(
37  const PathHandler &paths, const PhasePolyBox &phasepolybox) {
38  global_cost = 0;
39  unsigned id_counter = 0;
40  const PhasePolynomial &phasepoly = phasepolybox.get_phase_polynomial();
41  const MatrixXb &linear_fn_matrix_ppb =
42  phasepolybox.get_linear_transformation();
43  unsigned n_qubits = phasepolybox.get_n_qubits();
44  synth_circuit = Circuit(n_qubits);
45  linear_function = DiagMatrix(linear_fn_matrix_ppb);
46 
47  for (const auto &parity : phasepoly) {
48  ParityList parity_list = parity_column_to_list(parity.first);
49  SteinerTree tree(paths, parity_list, *parity_list.begin());
50  global_cost += tree.tree_cost;
51  std::pair<SteinerTree, Expr> parity_on_graph{tree, parity.second};
52 
53  CostedTrees::iterator iter = current_trees.find(tree.tree_cost);
54  if (iter == current_trees.end())
55  current_trees.insert({tree.tree_cost, {parity_on_graph}});
56  else
57  iter->second.push_back(parity_on_graph);
58 
59  ++id_counter;
60  }
61  tree_count = id_counter;
62 
63  CostedTrees new_trees;
64  for (const auto &cost_trees : current_trees) {
65  for (auto tree_expr : cost_trees.second) {
66  /* If we come across a tree which has only 1 node, check that its
67  properties are correct
68  and remove it from the forest, leaving an Rz in the circuit. */
69  if (tree_expr.first.fully_reduced()) {
70  unsigned qubit_id = UINT_MAX;
71  for (SteinerNodeType q : tree_expr.first.node_types) {
72  if (q == SteinerNodeType::Leaf) {
73  if (qubit_id != UINT_MAX) {
74  throw std::logic_error("Two nodes found to add the Rz");
75  } else {
76  qubit_id = tree_expr.first.root;
77  }
78  }
79  }
80 
81  if (qubit_id == UINT_MAX) {
82  throw std::logic_error("No node found to add the Rz");
83  }
84 
85  std::vector<unsigned> qubit{qubit_id};
86 
87  synth_circuit.add_op(OpType::Rz, tree_expr.second, qubit);
88  --tree_count;
89  /* Otherwise, the tree stays in. */
90  } else {
91  CostedTrees::iterator iter = new_trees.find(tree_expr.first.tree_cost);
92  if (iter == new_trees.end())
93  new_trees.insert({tree_expr.first.tree_cost, {tree_expr}});
94  else
95  iter->second.push_back(tree_expr);
96  unsigned cost_change = tree_expr.first.last_operation_cost;
97  global_cost += cost_change;
98  }
99  }
100  }
101  current_trees = new_trees;
102 }

104 void SteinerForest::add_row_globally(unsigned i, unsigned j) {
105  /* CNOT with control j and target i. Which way round the indices are is a wee
106  * bit fiddly. */
107  std::vector<unsigned> qbs = {j, i};
108  synth_circuit.add_op(OpType::CX, qbs);
109  linear_function.col_add(
110  i, j); // prepend a CNOT to the linear reversible function
111 
112  CostedTrees new_trees;
113 
114  for (const auto &cost_trees : current_trees) {
115  for (auto tree_expr : cost_trees.second) {
116  tree_expr.first.add_row(i, j);
117  /* If the tree has been fully reduced, remove it from the
118  forest */
119  if (tree_expr.first.fully_reduced()) {
120  std::vector<unsigned> qubit{i};
121  synth_circuit.add_op(OpType::Rz, tree_expr.second, qubit);
122  --tree_count;
123  }
124  /* Otherwise, update the forest */
125  else {
126  CostedTrees::iterator iter = new_trees.find(tree_expr.first.tree_cost);
127  if (iter == new_trees.end())
128  new_trees.insert({tree_expr.first.tree_cost, {tree_expr}});
129  else
130  iter->second.push_back(tree_expr);
131  unsigned cost_change = tree_expr.first.last_operation_cost;
132  global_cost += cost_change;
133  }
134  }
135  }
136  current_trees = new_trees;
137 }

186 CostedOperations best_operations_lookahead(
187  const PathHandler &path, const SteinerForest &forest, unsigned lookahead) {
188  if (lookahead == 0) {
189  throw std::logic_error("Must look ahead at least one step");
190  }
191  CostedTrees::const_reverse_iterator r_iter = forest.current_trees.rbegin();
192  if (r_iter == forest.current_trees.rend()) {
193  throw std::logic_error("Forest is empty");
194  }
195 
196  OperationList operations_available =
197  forest.operations_available_at_min_costs(path);
198 
199  if (operations_available.empty()) {
200  throw std::logic_error("Cannot find any operations");
201  }
202 
203  // set up base case
204  OperationList ops{operations_available.front()};
205  CostedOperations costed_operations =
206  recursive_operation_search(path, forest, lookahead - 1, ops);
207  operations_available.pop_front();
208 
209  for (OperationList::iterator op_iter = operations_available.begin();
210  op_iter != operations_available.end(); ++op_iter) {
211  ops = {*op_iter};
212  CostedOperations candidate_operations =
213  recursive_operation_search(path, forest, lookahead - 1, ops);
214 
215  if ((candidate_operations.first < costed_operations.first) ||
216  ((candidate_operations.first == costed_operations.first) &&
217  (candidate_operations.second.size() <
218  costed_operations.second.size()))) {
219  costed_operations = std::move(candidate_operations);
220  }
221  }
222 
223  return costed_operations;
224 }

226 CostedOperations recursive_operation_search(
227  const PathHandler &path, SteinerForest forest, unsigned lookahead,
228  OperationList row_operations) {
229  CostedOperations costed_operations;
230  CostedOperations candidate_operations;
231 
232  forest.add_row_globally(
233  row_operations.back().first, row_operations.back().second);
234 
235  if ((lookahead == 0) || (forest.current_trees.empty())) {
236  return {forest.global_cost, row_operations};
237  }
238  CostedTrees::const_reverse_iterator r_iter = forest.current_trees.rbegin();
239  unsigned index = r_iter->first;
240  OperationList operations_available =
241  forest.operations_available_under_the_index(path, index);
242  if (operations_available.empty()) {
243  return {forest.global_cost, row_operations};
244  } else {
245  row_operations.push_back(operations_available.front());
246  costed_operations =
247  recursive_operation_search(path, forest, lookahead - 1, row_operations);
248  row_operations.pop_back();
249  operations_available.pop_front();
250 
251  for (OperationList::iterator op_iter = operations_available.begin();
252  op_iter != operations_available.end(); ++op_iter) {
253  row_operations.push_back(*op_iter);
254  candidate_operations = recursive_operation_search(
255  path, forest, lookahead - 1, row_operations);
256 
257  row_operations.pop_back();
258 
259  if ((candidate_operations.first < costed_operations.first) ||
260  ((candidate_operations.first == costed_operations.first) &&
261  (candidate_operations.second.size() <
262  costed_operations.second.size()))) {
263  costed_operations = std::move(candidate_operations);
264  }
265  }
266  return costed_operations;
267  }

270 Circuit phase_poly_synthesis_int(
271  const Architecture &arch, const PhasePolyBox &phasepolybox,
272  unsigned lookahead, CNotSynthType cnottype) {
273  if (lookahead == 0)
274  throw std::logic_error(
275  "[AAS] the lookahead of the phase polynominal synthesis has to be "
276  "greater than 0");
277 
278  CostedOperations best_operations;
279 
280  PathHandler path(arch);
281 
282  PathHandler acyclic_path = path.construct_acyclic_handler();
283 
284  SteinerForest forest(acyclic_path, phasepolybox);
285 
286  while (!forest.current_trees.empty()) {
287  best_operations =
288  best_operations_lookahead(acyclic_path, forest, lookahead);
289  forest.add_operation_list(best_operations.second);
290  }
291  Circuit cnot_circ(path.get_size());
292  switch (cnottype) {
293  case CNotSynthType::HamPath: {
294  cnot_circ =
295  aas_CNOT_synth(forest.linear_function, path, CNotSynthType::HamPath);
296 
297  // check if matrix is id, if not the result will be wrong
298  TKET_ASSERT(forest.linear_function.is_id());
299  break;
300  }
301  case CNotSynthType::Rec: {
302  Circuit resultofstep =
303  aas_CNOT_synth(forest.linear_function, path, CNotSynthType::Rec);
304  cnot_circ = cnot_circ >> resultofstep;
305 
306  // check if matrix is id, if not the result will be wrong
307  TKET_ASSERT(forest.linear_function.is_id());
308  break;
309  }
310  case CNotSynthType::SWAP: {
311  cnot_circ = aas_CNOT_synth_SWAP(forest.linear_function, path);
312  // the check if the forest.linear_function is id is executed in the
313  // aas_CNOT_synth_SWAP function
314  break;
315  }
316  default: {
317  TKET_ASSERT(!"[AAS]: unknown type of cnot synth");
318  }
319  }
320 
321  return forest.synth_circuit >> cnot_circ.dagger();
322 }

324 static PhasePolyBox make_placed_ppb(
325  const Architecture &arch, const PhasePolyBox &phasepolybox) {
326  Circuit circuit_ppb_place(*phasepolybox.to_circuit());
327 
328  const std::string register_name = "surplus";
329 
330  unsigned qb_counter = circuit_ppb_place.n_qubits();
331  while (arch.n_nodes() > circuit_ppb_place.n_qubits()) {
332  Qubit qb = Qubit(register_name, qb_counter);
333  circuit_ppb_place.add_qubit(qb);
334  ++qb_counter;
335  }
336 
337  TKET_ASSERT(circuit_ppb_place.n_qubits() == arch.n_nodes());
338 
339  qubit_vector_t q_vec_place = circuit_ppb_place.all_qubits();
340  std::map<Qubit, Node> qubit_to_nodes_place;
341  unsigned counter_place = 0;
342 
343  for (Node no_place : arch.nodes()) {
344  if (counter_place < circuit_ppb_place.n_qubits()) {
345  qubit_to_nodes_place.insert({q_vec_place[counter_place], no_place});
346  ++counter_place;
347  }
348  }
349 
350  circuit_ppb_place.rename_units(qubit_to_nodes_place);
351 
352  return PhasePolyBox(circuit_ppb_place);
353 }

355 class PhasePolySynthesizer {
356  public:
357  PhasePolySynthesizer(
358  const Architecture &arch, const PhasePolyBox &phasepolybox,
359  unsigned lookahead, CNotSynthType cnottype)
360  : arch(arch),
361  placed_ppb(make_placed_ppb(arch, phasepolybox)),
362  lookahead(lookahead),
363  cnottype(cnottype) {}
364 
365  Circuit get_result() {
366  return (cnottype == CNotSynthType::HamPath) ? get_result_using_hampath()
367  : get_result_standard();
368  }
369 
370  private:
371  Circuit get_result_from_hampath(const std::vector<Node> &hampath) {
372  // maps from qubits/node to int
373  std::map<UnitID, UnitID> forward_contiguous_uids_q;
374  std::map<UnitID, UnitID> backward_contiguous_uids_n;
375  // extra map with node type needed for the creation of the architecture
376  std::map<UnitID, Node> unitid_to_int_nodes;
377  for (unsigned i = 0; i < hampath.size(); i++) {
378  UnitID qu_no = UnitID(hampath[i]);
379  Qubit q = Qubit(i);
380  Node n = Node(i);
381  unitid_to_int_nodes.insert({qu_no, n});
382  forward_contiguous_uids_q.insert({q, qu_no});
383  backward_contiguous_uids_n.insert({qu_no, n});
384  }
385  // define new arcitecture
386  std::vector<Architecture::Connection> new_con;
387  for (auto pair : arch.get_all_edges_vec()) {
388  new_con.push_back(
389  {unitid_to_int_nodes[UnitID(pair.first)],
390  unitid_to_int_nodes[UnitID(pair.second)]});
391  }
392  Architecture con_arch(new_con);
393  return make_result_from_con_arch(
394  con_arch, forward_contiguous_uids_q, backward_contiguous_uids_n);
395  }

397  Circuit get_result_using_hampath() {
398  std::vector<Node> hampath = find_hampath(arch); // using default timeout
399  Circuit c0 = get_result_from_hampath(hampath);
400  // Sometimes the reversed path gives a better circuit. Try both!
401  std::reverse(hampath.begin(), hampath.end());
402  Circuit c1 = get_result_from_hampath(hampath);
403  return (c0.depth() < c1.depth()) ? c0 : c1;
404  }

406  Circuit get_result_standard() {
407  // calculate iteration order
408  IterationOrder iter_order(arch);
409  std::vector<Node> node_order = iter_order.get_iterationorder();
410  // maps from qubits/node to int
411  std::map<UnitID, UnitID> forward_contiguous_uids_q;
412  std::map<UnitID, UnitID> backward_contiguous_uids_n;
413  // extra map with node type needed for the creation of the architecture
414  std::map<UnitID, Node> unitid_to_int_nodes;
415  for (unsigned i = 0; i < node_order.size(); i++) {
416  // convert node to superclass type of qubit/node
417  UnitID qu_no = UnitID(node_order[i]);
418  Qubit q = Qubit(i);
419  Node n = Node(i);
420  forward_contiguous_uids_q.insert({q, qu_no});
421  backward_contiguous_uids_n.insert({qu_no, n});
422  unitid_to_int_nodes.insert({qu_no, n});
423  }
424  // define new arcitecture: include only the tree edges
425  std::vector<Architecture::Connection> new_con;
426  for (auto pair : iter_order.get_edgelist()) {
427  new_con.push_back(
428  {unitid_to_int_nodes[UnitID(pair.first)],
429  unitid_to_int_nodes[UnitID(pair.second)]});
430  }
431  Architecture con_arch(new_con);
432  return make_result_from_con_arch(
433  con_arch, forward_contiguous_uids_q, backward_contiguous_uids_n);
434  }

436  Circuit make_result_from_con_arch(
437  const Architecture &con_arch,
438  const std::map<UnitID, UnitID> &forward_contiguous_uids_q,
439  const std::map<UnitID, UnitID> &backward_contiguous_uids_n) {
440  // define new phase poly box
441  Circuit circuit_ppb(*placed_ppb.to_circuit());
442  // the aas code is implemented under the assumption that all qubits in the
443  // circuit are named from 0 to n. The same assumption was made for the nodes
444  // of the architecture. To make sure that this condition is fulfilled the
445  // qubits in the circuit and the architecture are renamed. The new names are
446  // reverted at the end of the aas procedure. The qubits and the nodes have
447  // the same name in the input.
448  circuit_ppb.rename_units(backward_contiguous_uids_n);
449  PhasePolyBox new_ppb(circuit_ppb);
450  Circuit result =
451  phase_poly_synthesis_int(con_arch, new_ppb, lookahead, cnottype);
452  result.rename_units(forward_contiguous_uids_q);
453  return result;
454  }
455 
456  Architecture arch;
457  PhasePolyBox placed_ppb;
458  unsigned lookahead;
459  CNotSynthType cnottype;
460 };
