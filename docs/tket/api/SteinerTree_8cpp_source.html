20 // Steiner Tree generation follows the method of Takahashi and
21 // Matsuyama et al. see
22 // https://en.wikipedia.org/wiki/Steiner_tree_problem#Approximating_the_Steiner_tree
23 SteinerTree::SteinerTree(
24  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add,
25  unsigned root_node) {
26  root = root_node;
27  last_operation_cost = 0;
28  init_tree(pathhandler, nodes_to_add);
29  while (!nodes_to_add.empty()) {
30  add_closest_node_to_tree(pathhandler, nodes_to_add);
31  }
32  tree_cost = calculate_cost();
33 }

35 unsigned SteinerTree::calculate_cost() const {
36  unsigned cost = 0;
37  for (SteinerNodeType nt : node_types) {
38  switch (nt) {
39  case SteinerNodeType::ZeroInTree: {
40  cost += 2;
41  break;
42  }
43  case SteinerNodeType::OneInTree: {
44  ++cost;
45  break;
46  }
47  case SteinerNodeType::Leaf: {
48  ++cost;
49  break;
50  }
51  default: {
52  // no action occurs
53  break;
54  }
55  }
56  }
57  if (cost > 0) --cost;
58  return cost;
59 }

79 void SteinerTree::init_tree(
80  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add) {
81  if (nodes_to_add.empty())
82  throw std::logic_error("Cannot initialise empty Steiner Tree.");
83 
84  unsigned n = pathhandler.get_connectivity_matrix().rows();
85  node_types = std::vector<SteinerNodeType>(n, SteinerNodeType::OutOfTree);
86  num_neighbours = std::vector<unsigned>(n, 0);
87  if (nodes_to_add.size() == 1) {
88  node_types[nodes_to_add.front()] =
89  SteinerNodeType::Leaf; // A single root is considered a leaf
90  tree_nodes = nodes_to_add;
91  nodes_to_add.clear();
92  } else if (nodes_to_add.size() > 1) {
93  // determine the shortest path
94  unsigned node1 = nodes_to_add.front(), node2 = nodes_to_add.back();
95  unsigned min_distance = pathhandler.get_distance_matrix()(node1, node2);
96  for (unsigned no1 : nodes_to_add) {
97  for (unsigned no2 : nodes_to_add) {
98  if (no1 != no2) {
99  unsigned distance = pathhandler.get_distance_matrix()(no1, no2);
100  if (distance < min_distance) {
101  node1 = no1;
102  node2 = no2;
103  min_distance = distance;
104  }
105  }
106  }
107  }
108  // neighbours must both be leaves
109  if (pathhandler.get_distance_matrix()(node1, node2) == 1) {
110  node_types[node1] = SteinerNodeType::Leaf;
111  node_types[node2] = SteinerNodeType::Leaf;
112  num_neighbours[node1] = 1;
113  num_neighbours[node2] = 1;
114  tree_nodes.push_back(node1);
115  tree_nodes.push_back(node2);
116 
117  }
118  // otherwise, 0 nodes exist between.
119  else {
120  node_types[node1] = SteinerNodeType::Leaf;
121  num_neighbours[node1] = 1;
122  tree_nodes.push_back(node1);
123  add_path_to_tree(pathhandler, node1, node2);
124  }
125  nodes_to_add.remove(node1);
126  nodes_to_add.remove(node2);
127  }
128 }

130 void SteinerTree::add_path_to_tree(
131  const PathHandler& pathhandler, unsigned node_in_tree,
132  unsigned node_to_add) {
133  // last node is a leaf, with one neighbor
134  node_types[node_to_add] = SteinerNodeType::Leaf;
135  num_neighbours[node_to_add] = 1;
136  tree_nodes.push_back(node_to_add);
137 
138  if ((node_in_tree == pathhandler.get_size()) ||
139  (node_to_add == pathhandler.get_size())) {
140  throw std::logic_error("searching for a node which is not in the tree");
141  }
142 
143  if (pathhandler.get_distance_matrix()(node_in_tree, node_to_add) <
144  pathhandler.get_distance_matrix()(node_to_add, node_in_tree)) {
145  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
146  pathhandler.get_size()) {
147  node_in_tree = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
148  } else {
149  node_in_tree = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
150  }
151 
152  if ((node_in_tree == pathhandler.get_size()) ||
153  (node_to_add == pathhandler.get_size())) {
154  throw std::logic_error("searching for a node which is not in the tree");
155  }
156 
157  while (node_in_tree != node_to_add) {
158  // node in a path = zero node, it has two neighbbors
159  node_types[node_in_tree] = SteinerNodeType::ZeroInTree;
160  tree_nodes.push_back(node_in_tree);
161  num_neighbours[node_in_tree] = 2;
162 
163  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
164  pathhandler.get_size()) {
165  node_in_tree = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
166  } else {
167  node_in_tree = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
168  }
169 
170  if ((node_in_tree == pathhandler.get_size()) ||
171  (node_to_add == pathhandler.get_size())) {
172  throw std::logic_error("searching for a node which is not in the tree");
173  }
174  }
175  } else {
176  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
177  pathhandler.get_size()) {
178  node_to_add = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
179  } else {
180  node_to_add = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
181  }
182 
183  if ((node_in_tree == pathhandler.get_size()) ||
184  (node_to_add == pathhandler.get_size())) {
185  throw std::logic_error("searching for a node which is not in the tree");
186  }
187 
188  while (node_in_tree != node_to_add) {
189  node_types[node_to_add] =
190  SteinerNodeType::ZeroInTree; // node in a path = zero node
191 
192  tree_nodes.push_back(node_to_add);
193 
194  num_neighbours[node_to_add] = 2; // it has two neighbbors
195 
196  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
197  pathhandler.get_size()) {
198  node_to_add = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
199  } else {
200  node_to_add = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
201  }
202 
203  if ((node_in_tree == pathhandler.get_size()) ||
204  (node_to_add == pathhandler.get_size())) {
205  throw std::logic_error("searching for a node which is not in the tree");
206  }
207  }
208  }

211 void SteinerTree::add_closest_node_to_tree(
212  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add) {
213  unsigned closest_node = 0;
214  unsigned distant_node = UINT_MAX; // initialise to "infinity"
215  unsigned closest_tree_node(
216  tree_nodes.front()); // closest steiner point to the nodes we want to add
217 
218  std::list<unsigned>::iterator itr1;
219  std::list<unsigned>::iterator itr2;
220 
221  // take the two closest points (one of the tree, the other among the
222  // nodes_to_add)
223  for (unsigned node_to_add : nodes_to_add) {
224  for (unsigned tree_node : tree_nodes) {
225  if (pathhandler.get_distance_matrix()(tree_node, node_to_add) <
226  distant_node) // graph is oriented, we want distance
227  // tree->nodes_to_add
228  {
229  closest_node = node_to_add;
230  closest_tree_node = tree_node;
231  distant_node =
232  pathhandler.get_distance_matrix()(tree_node, node_to_add);
233  }
234  }
235  }
236  nodes_to_add.remove(closest_node);
237 
238  // if the node of the tree is a leaf, then its a one node after adding the
239  // new node
240  if (node_types[closest_tree_node] == SteinerNodeType::Leaf) {
241  node_types[closest_tree_node] = SteinerNodeType::OneInTree;
242  }
243  ++num_neighbours[closest_tree_node];
244  add_path_to_tree(pathhandler, closest_tree_node, closest_node);
245 }

247 // assumes i & j are neighbouring vertices
248 int SteinerTree::cost_of_operation(unsigned i, unsigned j) const {
249  SteinerNodeType i_type = node_types[i];
250  SteinerNodeType j_type = node_types[j];
251 
252  // tedious case analysis
253  switch (i_type) {
254  case SteinerNodeType::ZeroInTree: {
255  switch (j_type) {
256  case SteinerNodeType::ZeroInTree:
257  case SteinerNodeType::OneInTree:
258  case SteinerNodeType::Leaf:
259  case SteinerNodeType::OutOfTree: {
260  return 0;
261  }
262  default: {
263  throw InvalidCostCalculation(
264  "[AAS]: Invalid cost calculation, wrong SteinerNodeType");
265  }
266  }
267  }
268  case SteinerNodeType::OneInTree: {
269  switch (j_type) {
270  case SteinerNodeType::ZeroInTree:
271  case SteinerNodeType::Leaf: {
272  return -1;
273  }
274  case SteinerNodeType::OneInTree:
275  case SteinerNodeType::OutOfTree: {
276  return 1;
277  }
278  default: {
279  throw InvalidCostCalculation(
280  "[AAS]: Invalid cost calculation, wrong SteinerNodeType");
281  }
282  }
283  }
284  case SteinerNodeType::Leaf: {
285  switch (j_type) {
286  case SteinerNodeType::ZeroInTree:
287  case SteinerNodeType::Leaf: {
288  return -1;
289  }
290  case SteinerNodeType::OneInTree:
291  case SteinerNodeType::OutOfTree: {
292  return 1;
293  }
294  default: {
295  throw InvalidCostCalculation(
296  "[AAS]: Invalid cost calculation, wrong SteinerNodeType");
297  }
298  }
299  }
300  case SteinerNodeType::OutOfTree: {
301  switch (j_type) {
302  case SteinerNodeType::ZeroInTree:
303  case SteinerNodeType::OneInTree:
304  case SteinerNodeType::Leaf:
305  case SteinerNodeType::OutOfTree: {
306  return 0;
307  }
308  default: {
309  throw InvalidCostCalculation(
310  "[AAS]: Invalid cost calculation, wrong SteinerNodeType");
311  }
312  }
313  }
314  default: {
315  throw InvalidCostCalculation(
316  "[AAS]: Invalid cost calculation, wrong SteinerNodeType");
317  }
318  }
319 }

321 OperationList SteinerTree::operations_available(
322  const PathHandler& pathhandler) const {
323  OperationList operations;
324  for (unsigned i = 0; i != node_types.size(); ++i) {
325  for (unsigned j = 0; j != node_types.size(); ++j) {
326  if (i == j) continue;
327  if (!pathhandler.get_connectivity_matrix()(i, j)) continue;
328 
329  if (node_types[i] == SteinerNodeType::OneInTree ||
330  node_types[i] == SteinerNodeType::Leaf) {
331  if (node_types[j] == SteinerNodeType::ZeroInTree ||
332  node_types[j] == SteinerNodeType::Leaf) {
333  operations.push_back({i, j});
334  }
335  }
336  }
337  }
338  return operations;
339 }

341 void SteinerTree::add_row(unsigned i, unsigned j) {
342  /* CNOT with control j and target i */
343  SteinerNodeType i_type = node_types[i];
344  SteinerNodeType j_type = node_types[j];
345 
346  last_operation_cost = cost_of_operation(i, j);
347  tree_cost += last_operation_cost;
348 
349  switch (i_type) {
350  case SteinerNodeType::ZeroInTree: {
351  // no action occurs
352  break;
353  }
354  case SteinerNodeType::OneInTree: {
355  switch (j_type) {
356  case SteinerNodeType::ZeroInTree: {
357  node_types[j] = SteinerNodeType::OneInTree;
358  break;
359  }
360  case SteinerNodeType::OneInTree: {
361  node_types[j] = SteinerNodeType::ZeroInTree;
362  break;
363  }
364  case SteinerNodeType::Leaf: {
365  TKET_ASSERT(num_neighbours[i] != 0);
366  TKET_ASSERT(num_neighbours[j] != 0);
367  node_types[j] = SteinerNodeType::OutOfTree;
368  num_neighbours[i] -= 1;
369  num_neighbours[j] -= 1;
370  if (num_neighbours[i] == 1) {
371  node_types[i] = SteinerNodeType::Leaf;
372  }
373  break;
374  }
375  case SteinerNodeType::OutOfTree: {
376  node_types[j] = SteinerNodeType::Leaf;
377  node_types[i] = SteinerNodeType::OneInTree;
378  num_neighbours[i] += 1;
379  num_neighbours[j] += 1;
380  break;
381  }
382  default: {
383  throw InvalidRowOperation(
384  "[AAS]: Invalid row operation, invalid combination "
385  "SteinerNodeType in add_row");
386  }
387  }
388  break;
389  }
390  case SteinerNodeType::Leaf: {
391  switch (j_type) {
392  case SteinerNodeType::ZeroInTree: {
393  node_types[j] = SteinerNodeType::OneInTree;
394  break;
395  }
396  case SteinerNodeType::OneInTree: {
397  node_types[j] = SteinerNodeType::ZeroInTree;
398  break;
399  }
400  // only happens when there are 2 vertices lefts
401  case SteinerNodeType::Leaf: {
402  TKET_ASSERT(num_neighbours[i] != 0);
403  TKET_ASSERT(num_neighbours[j] != 0);
404  node_types[j] = SteinerNodeType::OutOfTree;
405  node_types[i] = SteinerNodeType::OutOfTree;
406  num_neighbours[i] -= 1;
407  num_neighbours[j] -= 1;
408  break;
409  }
410  case SteinerNodeType::OutOfTree: {
411  node_types[j] = SteinerNodeType::Leaf;
412  node_types[i] = SteinerNodeType::OneInTree;
413  num_neighbours[i] += 1;
414  num_neighbours[j] += 1;
415  break;
416  }
417  default: {
418  // Invalid combination of nodes types in add row operation
419  TKET_ASSERT(false);
420  }
421  }
422  }
423  case SteinerNodeType::OutOfTree: {
424  // no action occurs
425  break;
426  }
427  default: {
428  TKET_ASSERT(!"Invalid combination of nodes types in add row operation");
429  }
430  }
431 }

435 static std::pair<unsigned, std::vector<unsigned>> steiner_reduce(
436  Circuit& circ, DiagMatrix& CNOT_matrix, const PathHandler& paths,
437  unsigned col, unsigned root, std::list<unsigned>& nodes, bool upper,
438  CNotSynthType cnottype) {
439  std::pair<unsigned, std::vector<unsigned>> result;
440 
441  PathHandler directed_paths;
442 
443  std::list<unsigned> fresh_node_list(nodes);
444 
445  if (upper) {
446  MatrixXb directed_connectivity = paths.get_connectivity_matrix();
447  for (unsigned i = 0; i < directed_connectivity.rows(); ++i) {
448  for (unsigned j = 0; j < directed_connectivity.cols(); ++j) {
449  if (i < root) directed_connectivity(i, j) = 0;
450  if (j < root) directed_connectivity(i, j) = 0;
451  }
452  }
453 
454  directed_paths = PathHandler(directed_connectivity);
455  } else {
456  MatrixXb directed_connectivity = paths.get_connectivity_matrix();
457  if (cnottype == CNotSynthType::HamPath) {
458  for (unsigned i = 0; i < directed_connectivity.rows(); ++i) {
459  for (unsigned j = 0; j < directed_connectivity.cols(); ++j) {
460  if (j > i) directed_connectivity(i, j) = 0;
461  if ((j + 1 != i) && (j != i + 1)) directed_connectivity(i, j) = 0;
462  }
463  }
464  }
465 
466  directed_paths = PathHandler(directed_connectivity);
467  }
468  SteinerTree cnot_tree = SteinerTree(directed_paths, fresh_node_list, root);
469 
470  // make list of edges, starting from root to leaves of tree
471  std::list<std::pair<unsigned, unsigned>> parent_child_list;
472  std::set<unsigned> possible_parents{root};
473  unsigned n_edges = cnot_tree.tree_nodes.size();
474  if (n_edges > 0) --n_edges;
475  std::set<unsigned> visited_parents{root};
476  unsigned counterwhile = 0;
477  while ((parent_child_list.size() < n_edges) &&
478  (counterwhile < n_edges * n_edges)) {
479  ++counterwhile;
480  std::set<unsigned> new_parents;
481 
482  for (unsigned node : cnot_tree.tree_nodes) {
483  for (unsigned parent : possible_parents) {
484  if (directed_paths.get_connectivity_matrix()(parent, node)) {
485  if (visited_parents.find(node) == visited_parents.end()) {
486  new_parents.insert(node);
487  visited_parents.insert(node);
488  parent_child_list.push_back({parent, node});
489  }
490  }
491  }
492  }
493 
494  possible_parents = new_parents;
495  }
496 
497  /* Remove zeros */
498  if (upper) {
499  std::list<std::pair<unsigned, unsigned>> zeros;
500  for (auto& [s0, s1] : parent_child_list) {
501  if (!CNOT_matrix._matrix(s0, col)) {
502  zeros.push_back({s0, s1});
503  }
504  }
505  while (!zeros.empty()) {
506  std::pair<unsigned, unsigned> s_pair = zeros.back();
507  unsigned s0 = s_pair.first, s1 = s_pair.second;
508  zeros.pop_back();
509  if (!CNOT_matrix._matrix(s0, col)) {
510  CNOT_matrix.row_add(s1, s0);
511  circ.add_op<unsigned>(OpType::CX, {s1, s0});
512  }
513  }
514  parent_child_list.reverse();
515  for (auto& [s0, s1] : parent_child_list) {
516  CNOT_matrix.row_add(s0, s1);
517  circ.add_op<unsigned>(OpType::CX, {s0, s1});
518  }
519  } else {
520  for (auto& [s0, s1] : parent_child_list) {
521  if (!CNOT_matrix._matrix(s1, col)) {
522  CNOT_matrix.row_add(s0, s1);
523  circ.add_op<unsigned>(OpType::CX, {s0, s1});
524  }
525  }
526  parent_child_list.reverse();
527  for (auto& [s0, s1] : parent_child_list) {
528  CNOT_matrix.row_add(s0, s1);
529  circ.add_op<unsigned>(OpType::CX, {s0, s1});
530  }
531  }
532  result.first = cnot_tree.get_max_element();
533  result.second = cnot_tree.nodes();
534  return result;
535 }

537 static std::pair<unsigned, std::vector<unsigned>> steiner_reduce_rec(
538  Circuit& circ, DiagMatrix& CNOT_matrix, const PathHandler& paths,
539  unsigned col, unsigned root, std::list<unsigned>& nodes) {
540  std::pair<unsigned, std::vector<unsigned>> result;
541 
542  std::list<unsigned> fresh_node_list(nodes);
543 
544  SteinerTree cnot_tree = SteinerTree(paths, fresh_node_list, root);
545 
546  // make list of edges, starting from root to leaves of tree
547  std::list<std::pair<unsigned, unsigned>> parent_child_list;
548  std::set<unsigned> possible_parents{root};
549  unsigned n_edges = cnot_tree.tree_nodes.size();
550  if (n_edges > 0) --n_edges;
551  std::set<unsigned> visited_parents{root};
552  unsigned counterwhile = 0;
553  while ((parent_child_list.size() < n_edges) &&
554  (counterwhile < n_edges * n_edges)) {
555  ++counterwhile;
556  std::set<unsigned> new_parents;
557 
558  for (unsigned node : cnot_tree.tree_nodes) {
559  for (unsigned parent : possible_parents) {
560  if (paths.get_connectivity_matrix()(parent, node)) {
561  if (visited_parents.find(node) == visited_parents.end()) {
562  new_parents.insert(node);
563  visited_parents.insert(node);
564  parent_child_list.push_back({parent, node});
565  }
566  }
567  }
568  }
569 
570  possible_parents = new_parents;
571  }
572 
573  /* Remove zeros */
574  for (auto& [s0, s1] : parent_child_list) {
575  if (!CNOT_matrix._matrix(s1, col)) {
576  CNOT_matrix.row_add(s0, s1);
577  circ.add_op<unsigned>(OpType::CX, {s0, s1});
578  }
579  }
580  parent_child_list.reverse();
581  for (auto& [s0, s1] : parent_child_list) {
582  CNOT_matrix.row_add(s0, s1);
583  circ.add_op<unsigned>(OpType::CX, {s0, s1});
584  }
585  result.first = cnot_tree.get_max_element();
586  result.second = cnot_tree.nodes();
587  return result;
588 }

590 void aas_cnot_synth_rec(
591  DiagMatrix& CNOT_matrix, const PathHandler& paths,
592  std::vector<unsigned>& pivot_cols, Circuit& cnot_circuit,
593  std::vector<unsigned> usablenodes = {}) {
594  // order usable nodes from highest to lowest element, the list should already
595  // be close to the opposite of this order anyway.
596  std::sort(usablenodes.begin(), usablenodes.end());
597  std::reverse(usablenodes.begin(), usablenodes.end());
598  for (unsigned current_row : usablenodes) {
599  unsigned pivot = pivot_cols[current_row];
600 
601  std::list<unsigned> nodes;
602 
603  for (unsigned r = 0; r != current_row; ++r) {
604  if (CNOT_matrix._matrix(r, pivot)) {
605  nodes.push_back(r);
606  }
607  }
608  unsigned max_node_in_tree = 0;
609  std::vector<unsigned> new_usable_nodes;
610  if (!nodes.empty()) {
611  nodes.push_back(current_row);
612  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce_rec(
613  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes);
614  max_node_in_tree = res.first;
615  new_usable_nodes = res.second;
616  }
617 
618  if (max_node_in_tree > current_row) {
619  aas_cnot_synth_rec(
620  CNOT_matrix, paths, pivot_cols, cnot_circuit, new_usable_nodes);
621  }
622  }
623 }

625 // see https://arxiv.org/abs/2004.06052 and https://github.com/Quantomatic/pyzx
626 // for more information.
627 Circuit aas_CNOT_synth(
628  DiagMatrix& CNOT_matrix, const PathHandler& paths, CNotSynthType cnottype) {
629  unsigned pivot = 0;
630  unsigned max_node_in_tree = 0;
631  std::vector<unsigned> usable_nodes(paths.get_size());
632  std::iota(usable_nodes.begin(), usable_nodes.end(), 0);
633 
634  std::vector<unsigned> pivot_cols;
635  Circuit cnot_circuit(paths.get_size());
636  for (unsigned current_row = 0; current_row != CNOT_matrix.n_rows();
637  ++current_row) {
638  bool found_pivot = false;
639  std::list<unsigned> nodes;
640  while ((!found_pivot) && pivot < CNOT_matrix.n_cols()) {
641  for (unsigned r = current_row; r != CNOT_matrix.n_rows(); ++r) {
642  if (CNOT_matrix._matrix(r, pivot)) {
643  nodes.push_back(r);
644  }
645  }
646  if (!nodes.empty()) {
647  pivot_cols.push_back(pivot);
648  found_pivot = true;
649  } else {
650  ++pivot;
651  }
652  }
653 
654  // can't try any more pivots
655  if (!found_pivot)
656  throw std::logic_error("Could not find pivot node in CNOT synthesis.");
657 
658  bool row_not_in_notes = std::none_of(
659  nodes.cbegin(), nodes.cend(),
660  [current_row](unsigned no) { return no == current_row; });
661 
662  if (row_not_in_notes) nodes.push_front(current_row);
663  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce(
664  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes, true,
665  cnottype);
666 
667  max_node_in_tree = res.first;
668  usable_nodes = res.second;
669  ++pivot;
670  }
671 
672  unsigned current_row = pivot_cols.size() - 1;
673  while (current_row > 0) {
674  std::list<unsigned> nodes;
675  if (CNOT_matrix.is_id_until_columns(current_row)) {
676  pivot = pivot_cols[current_row];
677 
678  for (unsigned r = 0; r != current_row; ++r) {
679  if (CNOT_matrix._matrix(r, pivot)) {
680  nodes.push_back(r);
681  }
682  }
683  if (!nodes.empty()) {
684  nodes.push_back(current_row);
685 
686  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce(
687  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes, false,
688  cnottype);
689 
690  max_node_in_tree = res.first;
691  usable_nodes = res.second;
692  } else {
693  max_node_in_tree = 0;
694  std::iota(usable_nodes.begin(), usable_nodes.end(), 0);
695  }
696  }
697 
698  if ((max_node_in_tree > current_row) && (cnottype == CNotSynthType::Rec)) {
699  aas_cnot_synth_rec(
700  CNOT_matrix, paths, pivot_cols, cnot_circuit, usable_nodes);
701  }
702 
703  --current_row;
704 
705  TKET_ASSERT(CNOT_matrix.is_id_until_columns(current_row));
706  }
707 
708  // when constructing circuit, we want to
709  // prepend gates, but it is easier to append and then take the dagger.
710  return cnot_circuit;
711 }

730 CNotSwapSynth::CNotSwapSynth(
731  const PathHandler& pathhandler, const DiagMatrix& CNOT_mat)
732  : paths(pathhandler),
733  CNOT_matrix(CNOT_mat),
734  circ(Circuit(paths.get_size())) {
735  for (unsigned current_row = 0; current_row != CNOT_matrix.n_rows();
736  ++current_row) {
737  if (!CNOT_matrix._matrix(current_row, current_row)) {
738  // try to find element equal to 1
739  // and set diagonal element to 1
740  unsigned one = current_row;
741  while (!CNOT_matrix._matrix(one, current_row)) {
742  ++one;
743  }
744  unsigned current_node = swap_to_root(one, current_row);
745 
746  // remove the 1 with the use of the root
747  CNOT_matrix.row_add(current_node, current_row);
748  circ.add_op<unsigned>(OpType::CX, {current_node, current_row});
749  cleanup_swaps();
750  }
751 
752  if (!CNOT_matrix._matrix(current_row, current_row)) {
753  throw std::logic_error(
754  "The given matrix is not invertible, the input was not created by a "
755  "cnot circuit");
756  }
757 
758  for (unsigned r = current_row + 1; r != CNOT_matrix.n_rows(); ++r) {
759  if (CNOT_matrix._matrix(r, current_row)) {
760  unsigned current_node = swap_to_root(r, current_row);
761 
762  CNOT_matrix.row_add(current_row, current_node);
763  circ.add_op<unsigned>(OpType::CX, {current_row, current_node});
764  cleanup_swaps();
765  }
766  }
767  }
768  // end of upper creation
769 
770  for (unsigned current_row = CNOT_matrix.n_rows() - 1; current_row > 0;
771  --current_row) {
772  for (unsigned r = 0; r < current_row; ++r) {
773  if (CNOT_matrix._matrix(r, current_row)) {
774  unsigned current_node = swap_to_root(r, current_row);
775 
776  CNOT_matrix.row_add(current_row, current_node);
777  circ.add_op<unsigned>(OpType::CX, {current_row, current_node});
778 
779  cleanup_swaps();
780  }
781  }
782  }

793 unsigned CNotSwapSynth::swap_to_root(
794  unsigned start_node, unsigned current_row) {
795  unsigned current_node = start_node;
796 
797  while (paths.get_path_matrix()(current_node, current_row) != current_row) {
