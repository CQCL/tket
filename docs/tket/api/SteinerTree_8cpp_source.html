20 // Steiner Tree generation follows the method of Takahashi and
21 // Matsuyama et al. see
22 // https://en.wikipedia.org/wiki/Steiner_tree_problem#Approximating_the_Steiner_tree
23 SteinerTree::SteinerTree(
24  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add,
25  unsigned root_node) {
26  root = root_node;
27  last_operation_cost = 0;
28  init_tree(pathhandler, nodes_to_add);
29  while (!nodes_to_add.empty()) {
30  add_closest_node_to_tree(pathhandler, nodes_to_add);
31  }
32  tree_cost = calculate_cost();
33 }

35 unsigned SteinerTree::calculate_cost() const {
36  unsigned cost = 0;
37  for (SteinerNodeType nt : node_types) {
38  switch (nt) {
39  case SteinerNodeType::ZeroInTree: {
40  cost += 2;
41  break;
42  }
43  case SteinerNodeType::OneInTree: {
44  ++cost;
45  break;
46  }
47  case SteinerNodeType::Leaf: {
48  ++cost;
49  break;
50  }
51  default: {
52  // no action occurs
53  break;
54  }
55  }
56  }
57  if (cost > 0) --cost;
58  return cost;
59 }

79 void SteinerTree::init_tree(
80  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add) {
81  if (nodes_to_add.empty())
82  throw std::logic_error("Cannot initialise empty Steiner Tree.");
83 
84  unsigned n = pathhandler.get_connectivity_matrix().rows();
85  node_types = std::vector<SteinerNodeType>(n, SteinerNodeType::OutOfTree);
86  num_neighbours = std::vector<unsigned>(n, 0);
87  if (nodes_to_add.size() == 1) {
88  node_types[nodes_to_add.front()] =
89  SteinerNodeType::Leaf; // A single root is considered a leaf
90  tree_nodes = nodes_to_add;
91  nodes_to_add.clear();
92  } else if (nodes_to_add.size() > 1) {
93  // determine the shortest path
94  unsigned node1 = nodes_to_add.front(), node2 = nodes_to_add.back();
95  unsigned min_distance = pathhandler.get_distance_matrix()(node1, node2);
96  for (unsigned no1 : nodes_to_add) {
97  for (unsigned no2 : nodes_to_add) {
98  if (no1 != no2) {
99  unsigned distance = pathhandler.get_distance_matrix()(no1, no2);
100  if (distance < min_distance) {
101  node1 = no1;
102  node2 = no2;
103  min_distance = distance;
104  }
105  }
106  }
107  }
108  // neighbours must both be leaves
109  if (pathhandler.get_distance_matrix()(node1, node2) == 1) {
110  node_types[node1] = SteinerNodeType::Leaf;
111  node_types[node2] = SteinerNodeType::Leaf;
112  num_neighbours[node1] = 1;
113  num_neighbours[node2] = 1;
114  tree_nodes.push_back(node1);
115  tree_nodes.push_back(node2);
116 
117  }
118  // otherwise, 0 nodes exist between.
119  else {
120  node_types[node1] = SteinerNodeType::Leaf;
121  num_neighbours[node1] = 1;
122  tree_nodes.push_back(node1);
123  add_path_to_tree(pathhandler, node1, node2);
124  }
125  nodes_to_add.remove(node1);
126  nodes_to_add.remove(node2);
127  }
128 }

130 void SteinerTree::add_path_to_tree(
131  const PathHandler& pathhandler, unsigned node_in_tree,
132  unsigned node_to_add) {
133  // last node is a leaf, with one neighbor
134  node_types[node_to_add] = SteinerNodeType::Leaf;
135  num_neighbours[node_to_add] = 1;
136  tree_nodes.push_back(node_to_add);
137 
138  if ((node_in_tree == pathhandler.get_size()) ||
139  (node_to_add == pathhandler.get_size())) {
140  throw std::logic_error("searching for a node which is not in the tree");
141  }
142 
143  if (pathhandler.get_distance_matrix()(node_in_tree, node_to_add) <
144  pathhandler.get_distance_matrix()(node_to_add, node_in_tree)) {
145  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
146  pathhandler.get_size()) {
147  node_in_tree = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
148  } else {
149  node_in_tree = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
150  }
151 
152  if ((node_in_tree == pathhandler.get_size()) ||
153  (node_to_add == pathhandler.get_size())) {
154  throw std::logic_error("searching for a node which is not in the tree");
155  }
156 
157  while (node_in_tree != node_to_add) {
158  // node in a path = zero node, it has two neighbbors
159  node_types[node_in_tree] = SteinerNodeType::ZeroInTree;
160  tree_nodes.push_back(node_in_tree);
161  num_neighbours[node_in_tree] = 2;
162 
163  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
164  pathhandler.get_size()) {
165  node_in_tree = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
166  } else {
167  node_in_tree = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
168  }
169 
170  if ((node_in_tree == pathhandler.get_size()) ||
171  (node_to_add == pathhandler.get_size())) {
172  throw std::logic_error("searching for a node which is not in the tree");
173  }
174  }
175  } else {
176  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
177  pathhandler.get_size()) {
178  node_to_add = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
179  } else {
180  node_to_add = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
181  }
182 
183  if ((node_in_tree == pathhandler.get_size()) ||
184  (node_to_add == pathhandler.get_size())) {
185  throw std::logic_error("searching for a node which is not in the tree");
186  }
187 
188  while (node_in_tree != node_to_add) {
189  node_types[node_to_add] =
190  SteinerNodeType::ZeroInTree; // node in a path = zero node
191 
192  tree_nodes.push_back(node_to_add);
193 
194  num_neighbours[node_to_add] = 2; // it has two neighbbors
195 
196  if (pathhandler.get_path_matrix()(node_to_add, node_in_tree) ==
197  pathhandler.get_size()) {
198  node_to_add = pathhandler.get_path_matrix()(node_in_tree, node_to_add);
199  } else {
200  node_to_add = pathhandler.get_path_matrix()(node_to_add, node_in_tree);
201  }
202 
203  if ((node_in_tree == pathhandler.get_size()) ||
204  (node_to_add == pathhandler.get_size())) {
205  throw std::logic_error("searching for a node which is not in the tree");
206  }
207  }
208  }

211 void SteinerTree::add_closest_node_to_tree(
212  const PathHandler& pathhandler, std::list<unsigned>& nodes_to_add) {
213  unsigned closest_node = 0;
214  unsigned distant_node = UINT_MAX; // initialise to "infinity"
215  unsigned closest_tree_node(
216  tree_nodes.front()); // closest steiner point to the nodes we want to add
217 
218  std::list<unsigned>::iterator itr1;
219  std::list<unsigned>::iterator itr2;
220 
221  // take the two closest points (one of the tree, the other among the
222  // nodes_to_add)
223  for (unsigned node_to_add : nodes_to_add) {
224  for (unsigned tree_node : tree_nodes) {
225  if (pathhandler.get_distance_matrix()(tree_node, node_to_add) <
226  distant_node) // graph is oriented, we want distance
227  // tree->nodes_to_add
228  {
229  closest_node = node_to_add;
230  closest_tree_node = tree_node;
231  distant_node =
232  pathhandler.get_distance_matrix()(tree_node, node_to_add);
233  }
234  }
235  }
236  nodes_to_add.remove(closest_node);
237 
238  // if the node of the tree is a leaf, then its a one node after adding the
239  // new node
240  if (node_types[closest_tree_node] == SteinerNodeType::Leaf) {
241  node_types[closest_tree_node] = SteinerNodeType::OneInTree;
242  }
243  ++num_neighbours[closest_tree_node];
244  add_path_to_tree(pathhandler, closest_tree_node, closest_node);
245 }

247 // assumes i & j are neighbouring vertices
248 int SteinerTree::cost_of_operation(unsigned i, unsigned j) const {
249  SteinerNodeType i_type = node_types[i];
250  SteinerNodeType j_type = node_types[j];
251 
252  // tedious case analysis
253  switch (i_type) {
254  case SteinerNodeType::ZeroInTree: {
255  switch (j_type) {
256  case SteinerNodeType::ZeroInTree:
257  case SteinerNodeType::OneInTree:
258  case SteinerNodeType::Leaf:
259  case SteinerNodeType::OutOfTree: {
260  return 0;
261  }
262  default: {
263  TKET_ASSERT(!"[AAS]: Invalid cost, wrong SteinerNodeType");
264  }
265  }
266  break;
267  }
268  case SteinerNodeType::OneInTree: {
269  switch (j_type) {
270  case SteinerNodeType::ZeroInTree:
271  case SteinerNodeType::Leaf: {
272  return -1;
273  }
274  case SteinerNodeType::OneInTree:
275  case SteinerNodeType::OutOfTree: {
276  return 1;
277  }
278  default: {
279  TKET_ASSERT(!"[AAS]: Invalid cost, wrong SteinerNodeType");
280  }
281  }
282  break;
283  }
284  case SteinerNodeType::Leaf: {
285  switch (j_type) {
286  case SteinerNodeType::ZeroInTree:
287  case SteinerNodeType::Leaf: {
288  return -1;
289  }
290  case SteinerNodeType::OneInTree:
291  case SteinerNodeType::OutOfTree: {
292  return 1;
293  }
294  default: {
295  TKET_ASSERT(!"[AAS]: Invalid cost, wrong SteinerNodeType");
296  }
297  }
298  break;
299  }
300  case SteinerNodeType::OutOfTree: {
301  switch (j_type) {
302  case SteinerNodeType::ZeroInTree:
303  case SteinerNodeType::OneInTree:
304  case SteinerNodeType::Leaf:
305  case SteinerNodeType::OutOfTree: {
306  return 0;
307  }
308  default: {
309  TKET_ASSERT(!"[AAS]: Invalid cost, wrong SteinerNodeType");
310  }
311  }
312  break;
313  }
314  default: {
315  TKET_ASSERT(!"[AAS]: Invalid cost, wrong SteinerNodeType");
316  }
317  }
318  return 0;
319 }

321 OperationList SteinerTree::operations_available(
322  const PathHandler& pathhandler) const {
323  OperationList operations;
324  for (unsigned i = 0; i != node_types.size(); ++i) {
325  for (unsigned j = 0; j != node_types.size(); ++j) {
326  if (i == j) continue;
327  if (!pathhandler.get_connectivity_matrix()(i, j)) continue;
328 
329  if (node_types[i] == SteinerNodeType::OneInTree ||
330  node_types[i] == SteinerNodeType::Leaf) {
331  if (node_types[j] == SteinerNodeType::ZeroInTree ||
332  node_types[j] == SteinerNodeType::Leaf) {
333  operations.push_back({i, j});
334  }
335  }
336  }
337  }
338  return operations;
339 }

341 void SteinerTree::add_row(unsigned i, unsigned j) {
342  /* CNOT with control j and target i */
343  SteinerNodeType i_type = node_types[i];
344  SteinerNodeType j_type = node_types[j];
345 
346  last_operation_cost = cost_of_operation(i, j);
347  tree_cost += last_operation_cost;
348 
349  switch (i_type) {
350  case SteinerNodeType::ZeroInTree: {
351  // no action occurs
352  break;
353  }
354  case SteinerNodeType::OneInTree: {
355  switch (j_type) {
356  case SteinerNodeType::ZeroInTree: {
357  node_types[j] = SteinerNodeType::OneInTree;
358  break;
359  }
360  case SteinerNodeType::OneInTree: {
361  node_types[j] = SteinerNodeType::ZeroInTree;
362  break;
363  }
364  case SteinerNodeType::Leaf: {
365  TKET_ASSERT(num_neighbours[i] != 0);
366  TKET_ASSERT(num_neighbours[j] != 0);
367  node_types[j] = SteinerNodeType::OutOfTree;
368  num_neighbours[i] -= 1;
369  num_neighbours[j] -= 1;
370  if (num_neighbours[i] == 1) {
371  node_types[i] = SteinerNodeType::Leaf;
372  }
373  break;
374  }
375  case SteinerNodeType::OutOfTree: {
376  node_types[j] = SteinerNodeType::Leaf;
377  node_types[i] = SteinerNodeType::OneInTree;
378  num_neighbours[i] += 1;
379  num_neighbours[j] += 1;
380  break;
381  }
382  default: {
383  TKET_ASSERT(!"[AAS]: Invalid row op, wrong SteinerNodeType");
384  }
385  }
386  break;
387  }
388  case SteinerNodeType::Leaf: {
389  switch (j_type) {
390  case SteinerNodeType::ZeroInTree: {
391  node_types[j] = SteinerNodeType::OneInTree;
392  break;
393  }
394  case SteinerNodeType::OneInTree: {
395  node_types[j] = SteinerNodeType::ZeroInTree;
396  break;
397  }
398  // only happens when there are 2 vertices lefts
399  case SteinerNodeType::Leaf: {
400  TKET_ASSERT(num_neighbours[i] != 0);
401  TKET_ASSERT(num_neighbours[j] != 0);
402  node_types[j] = SteinerNodeType::OutOfTree;
403  node_types[i] = SteinerNodeType::OutOfTree;
404  num_neighbours[i] -= 1;
405  num_neighbours[j] -= 1;
406  break;
407  }
408  case SteinerNodeType::OutOfTree: {
409  node_types[j] = SteinerNodeType::Leaf;
410  node_types[i] = SteinerNodeType::OneInTree;
411  num_neighbours[i] += 1;
412  num_neighbours[j] += 1;
413  break;
414  }
415  default: {
416  // Invalid combination of nodes types in add row operation
417  TKET_ASSERT(false);
418  }
419  }
420  break;
421  }
422  case SteinerNodeType::OutOfTree: {
423  // no action occurs
424  break;
425  }
426  default: {
427  TKET_ASSERT(!"Invalid combination of nodes types in add row operation");
428  }
429  }
430 }

434 static std::pair<unsigned, std::vector<unsigned>> steiner_reduce(
435  Circuit& circ, DiagMatrix& CNOT_matrix, const PathHandler& paths,
436  unsigned col, unsigned root, std::list<unsigned>& nodes, bool upper,
437  CNotSynthType cnottype) {
438  std::pair<unsigned, std::vector<unsigned>> result;
439 
440  PathHandler directed_paths;
441 
442  std::list<unsigned> fresh_node_list(nodes);
443 
444  if (upper) {
445  MatrixXb directed_connectivity = paths.get_connectivity_matrix();
446  for (unsigned i = 0; i < directed_connectivity.rows(); ++i) {
447  for (unsigned j = 0; j < directed_connectivity.cols(); ++j) {
448  if (i < root) directed_connectivity(i, j) = 0;
449  if (j < root) directed_connectivity(i, j) = 0;
450  }
451  }
452 
453  directed_paths = PathHandler(directed_connectivity);
454  } else {
455  MatrixXb directed_connectivity = paths.get_connectivity_matrix();
456  if (cnottype == CNotSynthType::HamPath) {
457  for (unsigned i = 0; i < directed_connectivity.rows(); ++i) {
458  for (unsigned j = 0; j < directed_connectivity.cols(); ++j) {
459  if (j > i) directed_connectivity(i, j) = 0;
460  if ((j + 1 != i) && (j != i + 1)) directed_connectivity(i, j) = 0;
461  }
462  }
463  }
464 
465  directed_paths = PathHandler(directed_connectivity);
466  }
467  SteinerTree cnot_tree = SteinerTree(directed_paths, fresh_node_list, root);
468 
469  // make list of edges, starting from root to leaves of tree
470  std::list<std::pair<unsigned, unsigned>> parent_child_list;
471  std::set<unsigned> possible_parents{root};
472  unsigned n_edges = cnot_tree.tree_nodes.size();
473  if (n_edges > 0) --n_edges;
474  std::set<unsigned> visited_parents{root};
475  unsigned counterwhile = 0;
476  while ((parent_child_list.size() < n_edges) &&
477  (counterwhile < n_edges * n_edges)) {
478  ++counterwhile;
479  std::set<unsigned> new_parents;
480 
481  for (unsigned node : cnot_tree.tree_nodes) {
482  for (unsigned parent : possible_parents) {
483  if (directed_paths.get_connectivity_matrix()(parent, node)) {
484  if (visited_parents.find(node) == visited_parents.end()) {
485  new_parents.insert(node);
486  visited_parents.insert(node);
487  parent_child_list.push_back({parent, node});
488  }
489  }
490  }
491  }
492 
493  possible_parents = new_parents;
494  }
495 
496  /* Remove zeros */
497  if (upper) {
498  std::list<std::pair<unsigned, unsigned>> zeros;
499  for (auto& [s0, s1] : parent_child_list) {
500  if (!CNOT_matrix._matrix(s0, col)) {
501  zeros.push_back({s0, s1});
502  }
503  }
504  while (!zeros.empty()) {
505  std::pair<unsigned, unsigned> s_pair = zeros.back();
506  unsigned s0 = s_pair.first, s1 = s_pair.second;
507  zeros.pop_back();
508  if (!CNOT_matrix._matrix(s0, col)) {
509  CNOT_matrix.row_add(s1, s0);
510  circ.add_op<unsigned>(OpType::CX, {s1, s0});
511  }
512  }
513  parent_child_list.reverse();
514  for (auto& [s0, s1] : parent_child_list) {
515  CNOT_matrix.row_add(s0, s1);
516  circ.add_op<unsigned>(OpType::CX, {s0, s1});
517  }
518  } else {
519  for (auto& [s0, s1] : parent_child_list) {
520  if (!CNOT_matrix._matrix(s1, col)) {
521  CNOT_matrix.row_add(s0, s1);
522  circ.add_op<unsigned>(OpType::CX, {s0, s1});
523  }
524  }
525  parent_child_list.reverse();
526  for (auto& [s0, s1] : parent_child_list) {
527  CNOT_matrix.row_add(s0, s1);
528  circ.add_op<unsigned>(OpType::CX, {s0, s1});
529  }
530  }
531  result.first = cnot_tree.get_max_element();
532  result.second = cnot_tree.nodes();
533  return result;
534 }

536 static std::pair<unsigned, std::vector<unsigned>> steiner_reduce_rec(
537  Circuit& circ, DiagMatrix& CNOT_matrix, const PathHandler& paths,
538  unsigned col, unsigned root, std::list<unsigned>& nodes) {
539  std::pair<unsigned, std::vector<unsigned>> result;
540 
541  std::list<unsigned> fresh_node_list(nodes);
542 
543  SteinerTree cnot_tree = SteinerTree(paths, fresh_node_list, root);
544 
545  // make list of edges, starting from root to leaves of tree
546  std::list<std::pair<unsigned, unsigned>> parent_child_list;
547  std::set<unsigned> possible_parents{root};
548  unsigned n_edges = cnot_tree.tree_nodes.size();
549  if (n_edges > 0) --n_edges;
550  std::set<unsigned> visited_parents{root};
551  unsigned counterwhile = 0;
552  while ((parent_child_list.size() < n_edges) &&
553  (counterwhile < n_edges * n_edges)) {
554  ++counterwhile;
555  std::set<unsigned> new_parents;
556 
557  for (unsigned node : cnot_tree.tree_nodes) {
558  for (unsigned parent : possible_parents) {
559  if (paths.get_connectivity_matrix()(parent, node)) {
560  if (visited_parents.find(node) == visited_parents.end()) {
561  new_parents.insert(node);
562  visited_parents.insert(node);
563  parent_child_list.push_back({parent, node});
564  }
565  }
566  }
567  }
568 
569  possible_parents = new_parents;
570  }
571 
572  /* Remove zeros */
573  for (auto& [s0, s1] : parent_child_list) {
574  if (!CNOT_matrix._matrix(s1, col)) {
575  CNOT_matrix.row_add(s0, s1);
576  circ.add_op<unsigned>(OpType::CX, {s0, s1});
577  }
578  }
579  parent_child_list.reverse();
580  for (auto& [s0, s1] : parent_child_list) {
581  CNOT_matrix.row_add(s0, s1);
582  circ.add_op<unsigned>(OpType::CX, {s0, s1});
583  }
584  result.first = cnot_tree.get_max_element();
585  result.second = cnot_tree.nodes();
586  return result;
587 }

589 void aas_cnot_synth_rec(
590  DiagMatrix& CNOT_matrix, const PathHandler& paths,
591  std::vector<unsigned>& pivot_cols, Circuit& cnot_circuit,
592  std::vector<unsigned> usablenodes = {}) {
593  // order usable nodes from highest to lowest element, the list should already
594  // be close to the opposite of this order anyway.
595  std::sort(usablenodes.begin(), usablenodes.end());
596  std::reverse(usablenodes.begin(), usablenodes.end());
597  for (unsigned current_row : usablenodes) {
598  unsigned pivot = pivot_cols[current_row];
599 
600  std::list<unsigned> nodes;
601 
602  for (unsigned r = 0; r != current_row; ++r) {
603  if (CNOT_matrix._matrix(r, pivot)) {
604  nodes.push_back(r);
605  }
606  }
607  unsigned max_node_in_tree = 0;
608  std::vector<unsigned> new_usable_nodes;
609  if (!nodes.empty()) {
610  nodes.push_back(current_row);
611  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce_rec(
612  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes);
613  max_node_in_tree = res.first;
614  new_usable_nodes = res.second;
615  }
616 
617  if (max_node_in_tree > current_row) {
618  aas_cnot_synth_rec(
619  CNOT_matrix, paths, pivot_cols, cnot_circuit, new_usable_nodes);
620  }
621  }
622 }

624 // see https://arxiv.org/abs/2004.06052 and https://github.com/Quantomatic/pyzx
625 // for more information.
626 Circuit aas_CNOT_synth(
627  DiagMatrix& CNOT_matrix, const PathHandler& paths, CNotSynthType cnottype) {
628  unsigned pivot = 0;
629  unsigned max_node_in_tree = 0;
630  std::vector<unsigned> usable_nodes(paths.get_size());
631  std::iota(usable_nodes.begin(), usable_nodes.end(), 0);
632 
633  std::vector<unsigned> pivot_cols;
634  Circuit cnot_circuit(paths.get_size());
635  for (unsigned current_row = 0; current_row != CNOT_matrix.n_rows();
636  ++current_row) {
637  bool found_pivot = false;
638  std::list<unsigned> nodes;
639  while ((!found_pivot) && pivot < CNOT_matrix.n_cols()) {
640  for (unsigned r = current_row; r != CNOT_matrix.n_rows(); ++r) {
641  if (CNOT_matrix._matrix(r, pivot)) {
642  nodes.push_back(r);
643  }
644  }
645  if (!nodes.empty()) {
646  pivot_cols.push_back(pivot);
647  found_pivot = true;
648  } else {
649  ++pivot;
650  }
651  }
652 
653  // can't try any more pivots
654  if (!found_pivot)
655  throw std::logic_error("Could not find pivot node in CNOT synthesis.");
656 
657  bool row_not_in_notes = std::none_of(
658  nodes.cbegin(), nodes.cend(),
659  [current_row](unsigned no) { return no == current_row; });
660 
661  if (row_not_in_notes) nodes.push_front(current_row);
662  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce(
663  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes, true,
664  cnottype);
665 
666  max_node_in_tree = res.first;
667  usable_nodes = res.second;
668  ++pivot;
669  }
670 
671  unsigned current_row = pivot_cols.size() - 1;
672  while (current_row > 0) {
673  std::list<unsigned> nodes;
674  if (CNOT_matrix.is_id_until_columns(current_row)) {
675  pivot = pivot_cols[current_row];
676 
677  for (unsigned r = 0; r != current_row; ++r) {
678  if (CNOT_matrix._matrix(r, pivot)) {
679  nodes.push_back(r);
680  }
681  }
682  if (!nodes.empty()) {
683  nodes.push_back(current_row);
684 
685  std::pair<unsigned, std::vector<unsigned>> res = steiner_reduce(
686  cnot_circuit, CNOT_matrix, paths, pivot, current_row, nodes, false,
687  cnottype);
688 
689  max_node_in_tree = res.first;
690  usable_nodes = res.second;
691  } else {
692  max_node_in_tree = 0;
693  std::iota(usable_nodes.begin(), usable_nodes.end(), 0);
694  }
695  }
696 
697  if ((max_node_in_tree > current_row) && (cnottype == CNotSynthType::Rec)) {
698  aas_cnot_synth_rec(
699  CNOT_matrix, paths, pivot_cols, cnot_circuit, usable_nodes);
700  }
701 
702  --current_row;
703 
704  TKET_ASSERT(CNOT_matrix.is_id_until_columns(current_row));
705  }
706 
707  // when constructing circuit, we want to
708  // prepend gates, but it is easier to append and then take the dagger.
709  return cnot_circuit;
710 }

729 CNotSwapSynth::CNotSwapSynth(
730  const PathHandler& pathhandler, const DiagMatrix& CNOT_mat)
731  : paths(pathhandler),
732  CNOT_matrix(CNOT_mat),
733  circ(Circuit(paths.get_size())) {
734  for (unsigned current_row = 0; current_row != CNOT_matrix.n_rows();
735  ++current_row) {
736  if (!CNOT_matrix._matrix(current_row, current_row)) {
737  // try to find element equal to 1
738  // and set diagonal element to 1
739  unsigned one = current_row;
740  while (!CNOT_matrix._matrix(one, current_row)) {
741  ++one;
742  }
743  unsigned current_node = swap_to_root(one, current_row);
744 
745  // remove the 1 with the use of the root
746  CNOT_matrix.row_add(current_node, current_row);
747  circ.add_op<unsigned>(OpType::CX, {current_node, current_row});
748  cleanup_swaps();
749  }
750 
751  if (!CNOT_matrix._matrix(current_row, current_row)) {
752  throw std::logic_error(
753  "The given matrix is not invertible, the input was not created by a "
754  "cnot circuit");
755  }
756 
757  for (unsigned r = current_row + 1; r != CNOT_matrix.n_rows(); ++r) {
758  if (CNOT_matrix._matrix(r, current_row)) {
759  unsigned current_node = swap_to_root(r, current_row);
760 
761  CNOT_matrix.row_add(current_row, current_node);
762  circ.add_op<unsigned>(OpType::CX, {current_row, current_node});
763  cleanup_swaps();
764  }
765  }
766  }
767  // end of upper creation
768 
769  for (unsigned current_row = CNOT_matrix.n_rows() - 1; current_row > 0;
770  --current_row) {
771  for (unsigned r = 0; r < current_row; ++r) {
772  if (CNOT_matrix._matrix(r, current_row)) {
773  unsigned current_node = swap_to_root(r, current_row);
774 
775  CNOT_matrix.row_add(current_row, current_node);
776  circ.add_op<unsigned>(OpType::CX, {current_row, current_node});
777 
778  cleanup_swaps();
779  }
780  }
781  }

792 unsigned CNotSwapSynth::swap_to_root(
793  unsigned start_node, unsigned current_row) {
794  unsigned current_node = start_node;
795 
796  while (paths.get_path_matrix()(current_node, current_row) != current_row) {

802  }
