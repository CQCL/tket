29// Representation of a connected convex subcircuit.
30typedef struct {
31 VertexSet verts; // all vertices in the subcircuit
32 VertexSet preds; // all predecessors of vertices in the subcircuit that are
33 // not themselves in it
34 VertexSet succs; // all successors of vertices in the subcircuit that are not
35 // themselves in it
36} subcircuit_info_t;
37
38// Test whether the union of two connected convex subcircuits is connected.
39static bool union_is_connected(
40 const subcircuit_info_t &subcircuit_info0,
41 const subcircuit_info_t &subcircuit_info1) {
42 const VertexSet &verts0 = subcircuit_info0.verts;
43 const VertexSet &verts1 = subcircuit_info1.verts;
44 for (const Vertex &v : subcircuit_info0.succs) {
45 if (verts1.contains(v)) {
46 return true;
47 }
48 }
49 for (const Vertex &v : subcircuit_info1.succs) {
50 if (verts0.contains(v)) {
51 return true;
52 }
53 }
54 return false;
55}

57static VertexSet set_diff(const VertexSet &A, const VertexSet &B) {
58 VertexSet C;
59 for (const Vertex &v : A) {
60 if (!B.contains(v)) {
61 C.insert(v);
62 }
63 }
64 return C;
65}

67// Return the union of two disjoint convex connected subcircuits, assuming that
68// the union is convex and connected.
69static subcircuit_info_t convex_union(
70 const subcircuit_info_t &subcircuit_info0,
71 const subcircuit_info_t &subcircuit_info1) {
72 const VertexSet &verts0 = subcircuit_info0.verts;
73 const VertexSet &preds0 = subcircuit_info0.preds;
74 const VertexSet &succs0 = subcircuit_info0.succs;
75 const VertexSet &verts1 = subcircuit_info1.verts;
76 const VertexSet &preds1 = subcircuit_info1.preds;
77 const VertexSet &succs1 = subcircuit_info1.succs;
78
79 // verts = verts0 ∪ verts1
80 VertexSet verts = verts0;
81 verts.insert(verts1.begin(), verts1.end());
82
83 // preds = (preds0 ∪ preds1) ∖ verts
84 VertexSet preds01 = preds0;
85 preds01.insert(preds1.begin(), preds1.end());
86 VertexSet preds = set_diff(preds01, verts);
87
88 // succs = (succs0 ∪ succs1) ∖ verts
89 VertexSet succs01 = succs0;
90 succs01.insert(succs1.begin(), succs1.end());
91 VertexSet succs = set_diff(succs01, verts);
92
93 return {verts, preds, succs};
94}

96static std::set<std::pair<Vertex, Vertex>> order_relations(
97 Circuit *circ /*const*/) {
98 // Put the vertices in reverse topological order:
99 VertexVec verts;
100 boost::topological_sort(circ->dag, std::back_inserter(verts));
101 // Construct a map v --> {all vertices in the future of v}:
102 std::map<Vertex, VertexSet> futures;
103 for (const Vertex &v : verts) {
104 VertexSet v_futures = {v};
105 for (const Vertex &w : circ->get_successors(v)) {
106 const VertexSet &w_futures = futures[w];
107 v_futures.insert(w_futures.begin(), w_futures.end());
108 }
109 futures[v] = v_futures;
110 }
111 // Construct the set of order relations:
112 std::set<std::pair<Vertex, Vertex>> rels;
113 for (const auto &v_ws : futures) {
114 const Vertex &v = v_ws.first;
115 for (const Vertex &w : v_ws.second) {
116 rels.insert({v, w});
117 }
118 }
119 return rels;
120}

122// Helper class for finding connected convex subcircuits.
123class SubcircuitFinder {
124 public:
125 SubcircuitFinder(Circuit *circ /*const*/)
126 : circ_(circ), order_relations_(order_relations(circ)) {}
127 std::vector<VertexSet> find_subcircuits(
128 std::function<bool(Op_ptr)> criterion) {
129 // Find a maximal partition of the vertices satisfying the criterion into
130 // connected convex subcircuits. We use a greedy algorithm, beginning with
131 // the trivial partition into singletons, and then repeatedly looking for a
132 // pair that we can merge.
133 std::vector<subcircuit_info_t> subcircuit_infos;
134 // Form a trivial partition, one node per subcircuit:
135 BGL_FORALL_VERTICES(v, circ_->dag, DAG) {
136 if (criterion(circ_->get_Op_ptr_from_Vertex(v))) {
137 VertexVec preds = circ_->get_predecessors(v);
138 VertexVec succs = circ_->get_successors(v);
139 subcircuit_infos.push_back(
140 {{v}, {preds.begin(), preds.end()}, {succs.begin(), succs.end()}});
141 }
142 }
143 while (true) {
144 // If possible, find a mergeable pair of subcircuits and merge them.
145 std::optional<std::pair<std::size_t, std::size_t>> indices =
146 find_mergeable_pair(subcircuit_infos);
147 if (!indices.has_value()) {
148 break;
149 } else {
150 std::size_t i0 = indices->first;
151 std::size_t i1 = indices->second;
152 TKET_ASSERT(i0 < i1);
153 subcircuit_info_t subcircuit_info0 = subcircuit_infos[i0];
154 subcircuit_info_t subcircuit_info1 = subcircuit_infos[i1];
155 // Must erase in this order, since i0 < i1:
156 subcircuit_infos.erase(subcircuit_infos.begin() + i1);
157 subcircuit_infos.erase(subcircuit_infos.begin() + i0);
158 subcircuit_infos.push_back(
159 convex_union(subcircuit_info0, subcircuit_info1));
160 }
161 }
162 std::vector<VertexSet> subcircuits;
163 for (const subcircuit_info_t &subcircuit_info : subcircuit_infos) {
164 subcircuits.push_back(subcircuit_info.verts);
165 }
166 return subcircuits;
167 }
168
169 private:
170 // Test whether the union of two disjoint connected convex subcircuits is
171 // convex.
172 bool union_is_convex(
173 const subcircuit_info_t &subcircuit_info0,
174 const subcircuit_info_t &subcircuit_info1) {
175 const VertexSet &preds0 = subcircuit_info0.preds;
176 const VertexSet &succs0 = subcircuit_info0.succs;
177 const VertexSet &preds1 = subcircuit_info1.preds;
178 const VertexSet &succs1 = subcircuit_info1.succs;
179 for (const Vertex &v0 : succs0) {
180 for (const Vertex &v1 : preds1) {
181 if (order_relations_.contains({v0, v1})) {
182 return false;
183 }
184 }
185 }
186 for (const Vertex &v1 : succs1) {
187 for (const Vertex &v0 : preds0) {
188 if (order_relations_.contains({v1, v0})) {
189 return false;
190 }
191 }
192 }
193 return true;
194 }

196 // Given a vector of disjoint connected convex subcircuits, look for a pair
197 // whose union is connected and convex, and return the indices of such a pair
198 // if it exists.
199 std::optional<std::pair<std::size_t, std::size_t>> find_mergeable_pair(
200 const std::vector<subcircuit_info_t> &subcircuit_infos) {
201 std::size_t n_subcircuits = subcircuit_infos.size();
202 for (std::size_t i0 = 0; i0 < n_subcircuits; i0++) {
203 for (std::size_t i1 = i0 + 1; i1 < n_subcircuits; i1++) {
204 const subcircuit_info_t &subcircuit_info0 = subcircuit_infos[i0];
205 const subcircuit_info_t &subcircuit_info1 = subcircuit_infos[i1];
206 if (union_is_connected(subcircuit_info0, subcircuit_info1) &&
207 union_is_convex(subcircuit_info0, subcircuit_info1)) {
208 return std::pair<std::size_t, std::size_t>{i0, i1};
209 }
210 }
211 }
212 return std::nullopt;
213 }
214 const Circuit *circ_;
215 std::set<std::pair<Vertex, Vertex>> order_relations_;
216};

225Subcircuit Circuit::make_subcircuit(const VertexSet &verts) const {
226 const std::map<Edge, UnitID> unitmap = edge_unit_map();
227 std::map<UnitID, Edge> out_lookup;
228 EdgeSet ins, outs, b_future;
229 for (const Vertex &v : verts) {
230 for (const Edge &e : get_in_edges(v)) {
231 if (!verts.contains(source(e))) {
232 ins.insert(e);
233 }
234 }
235 for (const Edge &e : get_all_out_edges(v)) {
236 if (!verts.contains(target(e))) {
237 if (get_edgetype(e) == EdgeType::Boolean) {
238 b_future.insert(e);
239 } else {
240 outs.insert(e);
241 out_lookup.insert({unitmap.at(e), e});
242 }
243 }
244 }
245 }
246 // Convert to vectors, sort by UnitID, making UnitIDs of corresponding ins and
247 // outs match position
248 EdgeVec ins_vec{ins.begin(), ins.end()};
249 auto compare = [&unitmap](const Edge &e0, const Edge &e1) {
250 return unitmap.at(e0) < unitmap.at(e1);
251 };
252 std::sort(ins_vec.begin(), ins_vec.end(), compare);
253 std::vector<std::optional<Edge>> outs_vec;
254 for (const Edge &e : ins_vec) {
255 if (get_edgetype(e) == EdgeType::Boolean) {
256 outs_vec.push_back(std::nullopt);
257 } else {
258 outs_vec.push_back(out_lookup.at(unitmap.at(e)));
259 }
260 }
261 // b_future can go in any order
262 EdgeVec b_future_vec{b_future.begin(), b_future.end()};
263 return {ins_vec, outs_vec, b_future_vec, verts};
264}
