25static GraphEdgeWeights get_weight_one_edges(
26 const std::vector<Swap>& edges, std::size_t number_of_vertices) {
27 GraphEdgeWeights result;
28 for (const auto& edge : edges) {
29 TKET_ASSERT(edge.first < number_of_vertices);
30 TKET_ASSERT(edge.second < number_of_vertices);
31 result[get_edge(edge.first, edge.second)] = 1;
32 }
33 TKET_ASSERT(edges.size() == result.size());
34 return result;
35}

37static void add_solver_solutions(
38 const std::vector<SolutionWSM>& solutions, std::size_t pattern_n_vertices,
39 std::size_t target_n_vertices, SubgraphMonomorphisms& monomorphisms) {
40 monomorphisms.mappings.reserve(solutions.size());
41
42 // The solver ignores isolated vertices, so we'll fill them in separately.
43 std::set<std::size_t> used_pattern_vertices;
44 std::set<std::size_t> used_pattern_vertices_copy;
45 std::set<std::size_t> used_target_vertices;
46
47 for (const auto& solution : solutions) {
48 monomorphisms.mappings.emplace_back();
49 monomorphisms.mappings.back().resize(pattern_n_vertices);
50 used_pattern_vertices.clear();
51 used_target_vertices.clear();
52 for (const auto& pair : solution.assignments) {
53 const auto& pv = pair.first;
54 const auto& tv = pair.second;
55 TKET_ASSERT(used_pattern_vertices.insert(pv).second);
56 TKET_ASSERT(used_target_vertices.insert(tv).second);
57 if (!used_pattern_vertices_copy.empty()) {
58 // It's always the same nonisolated pattern vertices.
59 TKET_ASSERT(used_pattern_vertices_copy.count(pv) != 0);
60 }
61 // The architecture mapping guarantees
62 // contiguous vertex numbers {0,1,2,...,N}
63 TKET_ASSERT(pv < pattern_n_vertices);
64 TKET_ASSERT(tv < target_n_vertices);
65 monomorphisms.mappings.back()[pv] = tv;
66 }
67 if (used_pattern_vertices_copy.empty()) {
68 used_pattern_vertices_copy = used_pattern_vertices;
69 } else {
70 TKET_ASSERT(
71 used_pattern_vertices_copy.size() == used_pattern_vertices.size());
72 }
73
74 // Now, fill in isolated p-vertex values.
75 std::size_t next_tv = 0;
76 for (std::size_t pv = 0; pv < pattern_n_vertices; ++pv) {
77 // We could save time by saving the isolated PVs,
78 // but surely not worth it
79 if (used_pattern_vertices.count(pv) != 0) {
80 continue;
81 }
82 while (used_target_vertices.count(next_tv) != 0) {
83 ++next_tv;
84 }
85 monomorphisms.mappings.back()[pv] = next_tv;
86 ++next_tv;
87 }
88 }
89}

91// Strictly speaking, we should go through all (T choose k) subsets
92// of target vertices, and all permutations of k pattern vertices,
93// but don't bother for now - just give one solution.
94static void fill_with_all_isolated_pattern_vertices(
95 SubgraphMonomorphisms& monomorphisms, std::size_t pattern_n_vertices) {
96 monomorphisms.mappings.resize(1);
97 monomorphisms.mappings[0].resize(pattern_n_vertices);
98 for (std::size_t ii = 0; ii < pattern_n_vertices; ++ii) {
99 monomorphisms.mappings[0][ii] = ii;
100 }
101}

103SubgraphMonomorphisms::SubgraphMonomorphisms(
104 const ArchitectureMapping& pattern_arch_mapping,
105 const ArchitectureMapping& target_arch_mapping,
106 const Parameters& parameters)
107 : time_taken_ms(0) {
108 if (parameters.max_number_of_mappings == 0) {
109 return;
110 }
111 const auto pattern_n_vertices = pattern_arch_mapping.number_of_vertices();
112 const auto target_n_vertices = target_arch_mapping.number_of_vertices();
113 if (pattern_n_vertices > target_n_vertices) {
114 return;
115 }
116 const auto pattern_edges = pattern_arch_mapping.get_edges();
117 const auto target_edges = target_arch_mapping.get_edges();
118 if (pattern_edges.size() > target_edges.size()) {
119 return;
120 }
121 if (pattern_edges.empty()) {
122 // A pointless special case: all pattern vertices are isolated!
123 fill_with_all_isolated_pattern_vertices(*this, pattern_n_vertices);
124 return;
125 }
126 const auto pattern_edges_and_weights =
127 get_weight_one_edges(pattern_edges, pattern_n_vertices);
128 const auto target_edges_and_weights =
129 get_weight_one_edges(target_edges, target_n_vertices);
130
131 MainSolverParameters solver_parameters;
132 solver_parameters.terminate_with_first_full_solution = false;
133 solver_parameters.for_multiple_full_solutions_the_max_number_to_obtain =
134 parameters.max_number_of_mappings;
135 solver_parameters.timeout_ms = parameters.timeout_ms;
136
137 const MainSolver main_solver(
138 pattern_edges_and_weights, target_edges_and_weights, solver_parameters);
139
140 const auto& solution_data = main_solver.get_solution_data();
141
142 time_taken_ms =
143 solution_data.initialisation_time_ms + solution_data.search_time_ms;
144
145 add_solver_solutions(
146 solution_data.solutions, pattern_n_vertices, target_n_vertices, *this);
147}
