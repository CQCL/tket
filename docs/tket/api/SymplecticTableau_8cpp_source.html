78 SymplecticTableau::SymplecticTableau(const PauliStabiliserList &rows) {
79  n_rows_ = rows.size();
80  if (n_rows_ == 0)
81  n_qubits_ = 0;
82  else
83  n_qubits_ = rows[0].string.size();
84  xmat_ = MatrixXb::Zero(n_rows_, n_qubits_);
85  zmat_ = MatrixXb::Zero(n_rows_, n_qubits_);
86  phase_ = VectorXb::Zero(n_rows_);
87  for (unsigned i = 0; i < n_rows_; ++i) {
88  const PauliStabiliser &stab = rows[i];
89  if (stab.string.size() != n_qubits_)
90  throw std::invalid_argument(
91  "Tableau must have the same number of qubits in each row.");
92  for (unsigned q = 0; q < n_qubits_; ++q) {
93  const Pauli &p = stab.string[q];
94  xmat_(i, q) = (p == Pauli::X) || (p == Pauli::Y);
95  zmat_(i, q) = (p == Pauli::Z) || (p == Pauli::Y);
96  }
97  phase_(i) = !stab.coeff;
98  }
99 }

148 void SymplecticTableau::apply_CX(unsigned qc, unsigned qt) {
149  if (qc == qt)
150  throw std::logic_error(
151  "Attempting to apply a CX with equal control and target in a tableau");
152  for (unsigned i = 0; i < n_rows_; ++i) {
153  phase_(i) = phase_(i) ^ (xmat_(i, qc) && zmat_(i, qt) &&
154  !(xmat_(i, qt) ^ zmat_(i, qc)));
155  xmat_(i, qt) = xmat_(i, qc) ^ xmat_(i, qt);
156  zmat_(i, qc) = zmat_(i, qc) ^ zmat_(i, qt);
157  }
158 }

160 void SymplecticTableau::apply_gate(
161  OpType type, const std::vector<unsigned> &qbs) {
162  switch (type) {
163  case OpType::Z: {
164  apply_S(qbs.at(0));
165  apply_S(qbs.at(0));
166  break;
167  }
168  case OpType::X: {
169  apply_V(qbs.at(0));
170  apply_V(qbs.at(0));
171  break;
172  }
173  case OpType::Y: {
174  apply_S(qbs.at(0));
175  apply_S(qbs.at(0));
176  apply_V(qbs.at(0));
177  apply_V(qbs.at(0));
178  break;
179  }
180  case OpType::S: {
181  apply_S(qbs.at(0));
182  break;
183  }
184  case OpType::Sdg: {
185  apply_S(qbs.at(0));
186  apply_S(qbs.at(0));
187  apply_S(qbs.at(0));
188  break;
189  }
190  case OpType::V: {
191  apply_V(qbs.at(0));
192  break;
193  }
194  case OpType::Vdg: {
195  apply_V(qbs.at(0));
196  apply_V(qbs.at(0));
197  apply_V(qbs.at(0));
198  break;
199  }
200  case OpType::H: {
201  apply_S(qbs.at(0));
202  apply_V(qbs.at(0));
203  apply_S(qbs.at(0));
204  break;
205  }
206  case OpType::CX: {
207  apply_CX(qbs.at(0), qbs.at(1));
208  break;
209  }
210  case OpType::CY: {
211  apply_S(qbs.at(1));
212  apply_S(qbs.at(1));
213  apply_S(qbs.at(1));
214  apply_CX(qbs.at(0), qbs.at(1));
215  apply_S(qbs.at(1));
216  break;
217  }
218  case OpType::CZ: {
219  apply_S(qbs.at(1));
220  apply_V(qbs.at(1));
221  apply_S(qbs.at(1));
222  apply_CX(qbs.at(0), qbs.at(1));
223  apply_S(qbs.at(1));
224  apply_V(qbs.at(1));
225  apply_S(qbs.at(1));
226  break;
227  }
228  case OpType::SWAP: {
229  apply_CX(qbs.at(0), qbs.at(1));
230  apply_CX(qbs.at(1), qbs.at(0));
231  apply_CX(qbs.at(0), qbs.at(1));
232  break;
233  }
234  case OpType::BRIDGE: {
235  apply_CX(qbs.at(0), qbs.at(2));
236  break;
237  }
238  case OpType::noop: {
239  break;
240  }
241  default: {
242  throw BadOpType(
243  "Cannot be applied to a SymplecticTableau: not a Clifford gate",
244  type);
245  }
246  }

249 void SymplecticTableau::apply_pauli_gadget(
250  const PauliStabiliser &pauli, unsigned half_pis) {
251  if (pauli.string.size() != n_qubits_) {
252  throw std::invalid_argument(
253  "Cannot apply pauli gadget to SymplecticTableau; string and tableau "
254  "have different numbers of qubits");
255  }
256  half_pis = half_pis % 4;
257  if (half_pis == 0) return; // Identity
258  if (half_pis == 2) { // Degenerates to product of PI rotations
259  for (unsigned i = 0; i < pauli.string.size(); ++i) {
260  switch (pauli.string.at(i)) {
261  case Pauli::I: {
262  break;
263  }
264  case Pauli::X: {
265  apply_gate(OpType::X, {i});
266  break;
267  }
268  case Pauli::Y: {
269  apply_gate(OpType::Y, {i});
270  break;
271  }
272  case Pauli::Z: {
273  apply_gate(OpType::Z, {i});
274  break;
275  }
276  }
277  }
278  return;
279  }
280 
281  // From here, half_pis == 1 or 3
282  // They act the same except for a phase flip on the product term
283  MatrixXb pauli_xrow = MatrixXb::Zero(1, n_qubits_);
284  MatrixXb pauli_zrow = MatrixXb::Zero(1, n_qubits_);
285  for (unsigned i = 0; i < n_qubits_; ++i) {
286  Pauli p = pauli.string.at(i);
287  pauli_xrow(i) = (p == Pauli::X) || (p == Pauli::Y);
288  pauli_zrow(i) = (p == Pauli::Z) || (p == Pauli::Y);
289  }
290  bool phase = (!pauli.coeff) ^ (half_pis == 3);
291 
292  for (unsigned i = 0; i < n_rows_; ++i) {
293  bool anti = false;
294  MatrixXb::RowXpr xr = xmat_.row(i);
295  MatrixXb::RowXpr zr = zmat_.row(i);
296  for (unsigned q = 0; q < n_qubits_; ++q) {
297  anti ^= (xr(q) && pauli_zrow(q));
298  anti ^= (zr(q) && pauli_xrow(q));
299  }
300  if (anti) {
301  row_mult(
302  xr, zr, phase_(i), pauli_xrow.row(0), pauli_zrow.row(0), phase, i_,
303  xr, zr, phase_(i));
304  }
305  }
306 }

308 MatrixXb SymplecticTableau::anticommuting_rows() const {
309  MatrixXb res = MatrixXb::Zero(n_rows_, n_rows_);
310  for (unsigned i = 0; i < n_rows_; ++i) {
311  for (unsigned j = 0; j < i; ++j) {
312  bool anti = false;
313  for (unsigned q = 0; q < n_qubits_; ++q) {
314  anti ^= (xmat_(i, q) && zmat_(j, q));
315  anti ^= (xmat_(j, q) && zmat_(i, q));
316  }
317  res(i, j) = anti;
318  res(j, i) = anti;
319  }
320  }
321  return res;
322 }

324 unsigned SymplecticTableau::rank() const {
325  // Create a copy in gaussian form and count the empty rows
326  SymplecticTableau copy(*this);
327  copy.gaussian_form();
328  unsigned empty_rows = 0;
329  for (unsigned i = 0; i < n_rows_; ++i) {
330  if (copy.xmat_.row(n_rows_ - 1 - i).isZero() &&
331  copy.zmat_.row(n_rows_ - 1 - i).isZero())
332  ++empty_rows;
333  else
334  break;
335  }
336  return n_rows_ - empty_rows;
337 }

339 SymplecticTableau SymplecticTableau::conjugate() const {
340  SymplecticTableau conj(*this);
341  for (unsigned i = 0; i < n_rows_; ++i) {
342  unsigned sum = 0;
343  for (unsigned j = 0; j < n_qubits_; ++j) {
344  if (xmat_(i, j) && zmat_(i, j)) ++sum;
345  }
346  if (sum % 2 == 1) conj.phase_(i) ^= true;
347  }
348  return conj;

351 void SymplecticTableau::gaussian_form() {
352  MatrixXb fullmat = MatrixXb::Zero(n_rows_, 2 * n_qubits_);
353  fullmat(Eigen::all, Eigen::seq(0, Eigen::last, 2)) = xmat_;
354  fullmat(Eigen::all, Eigen::seq(1, Eigen::last, 2)) = zmat_;
355  std::vector<std::pair<unsigned, unsigned>> row_ops =
356  gaussian_elimination_row_ops(fullmat);
357  for (const std::pair<unsigned, unsigned> &op : row_ops) {
358  row_mult(op.first, op.second);
359  }

362 void SymplecticTableau::row_mult(
363  const MatrixXb::RowXpr &xa, const MatrixXb::RowXpr &za, const bool &pa,
364  const MatrixXb::RowXpr &xb, const MatrixXb::RowXpr &zb, const bool &pb,
365  Complex phase, MatrixXb::RowXpr &xw, MatrixXb::RowXpr &zw, bool &pw) {
366  if (pa) phase *= -1;
367  if (pb) phase *= -1;
368  for (unsigned i = 0; i < n_qubits_; i++) {
369  std::pair<BoolPauli, Complex> res =
370  BoolPauli::mult_lut.at({{xa(i), za(i)}, {xb(i), zb(i)}});
371  xw(i) = res.first.x;
372  zw(i) = res.first.z;
373  phase *= res.second;
374  }
375  pw = (phase == -1.);
376 }
