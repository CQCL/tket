63SymplecticTableau::SymplecticTableau(
64 const MatrixXb &xmat, const MatrixXb &zmat, const VectorXb &phase)
65 : xmat(xmat), zmat(zmat), phase(phase) {
66 if (zmat.rows() != xmat.rows() || phase.size() != xmat.rows())
67 throw std::invalid_argument(
68 "Tableau must have the same number of rows in each component.");
69 if (zmat.cols() != xmat.cols())
70 throw std::invalid_argument(
71 "Tableau must have the same number of columns in x and z components.");
72}

74SymplecticTableau::SymplecticTableau(const PauliStabiliserVec &rows) {
75 unsigned n_rows = rows.size();
76 unsigned n_qubits = 0;
77 if (n_rows != 0) n_qubits = rows[0].string.size();
78 xmat = MatrixXb::Zero(n_rows, n_qubits);
79 zmat = MatrixXb::Zero(n_rows, n_qubits);
80 phase = VectorXb::Zero(n_rows);
81 for (unsigned i = 0; i < n_rows; ++i) {
82 const PauliStabiliser &stab = rows[i];
83 if (stab.string.size() != n_qubits)
84 throw std::invalid_argument(
85 "Tableau must have the same number of qubits in each row.");
86 for (unsigned q = 0; q < n_qubits; ++q) {
87 const Pauli &p = stab.get(q);
88 xmat(i, q) = (p == Pauli::X) || (p == Pauli::Y);
89 zmat(i, q) = (p == Pauli::Z) || (p == Pauli::Y);
90 }
91 phase(i) = stab.is_real_negative();
92 }
93}

98PauliStabiliser SymplecticTableau::get_pauli(unsigned i) const {
99 unsigned n_qubits = get_n_qubits();
100 std::vector<Pauli> str(n_qubits);
101 for (unsigned q = 0; q < n_qubits; ++q) {
102 str[q] = BoolPauli{xmat(i, q), zmat(i, q)}.to_pauli();
103 }
104 return PauliStabiliser(str, phase(i) ? 2 : 0);
105}

115bool SymplecticTableau::operator==(const SymplecticTableau &other) const {
116 // Need this to short-circuit before matrix checks as comparing matrices of

119 (this->get_n_qubits() == other.get_n_qubits()) &&

148void SymplecticTableau::apply_X(unsigned qb) {

152void SymplecticTableau::apply_H(unsigned qb) {
153 for (unsigned i = 0; i < get_n_rows(); ++i) {
154 phase(i) = phase(i) ^ (xmat(i, qb) && zmat(i, qb));
155 bool temp = xmat(i, qb);
156 xmat(i, qb) = zmat(i, qb);
157 zmat(i, qb) = temp;
158 }
159}

161void SymplecticTableau::apply_CX(unsigned qc, unsigned qt) {
162 if (qc == qt)
163 throw std::logic_error(
164 "Attempting to apply a CX with equal control and target in a tableau");
165 for (unsigned i = 0; i < get_n_rows(); ++i) {
166 phase(i) =
167 phase(i) ^ (xmat(i, qc) && zmat(i, qt) && !(xmat(i, qt) ^ zmat(i, qc)));
168 xmat(i, qt) = xmat(i, qc) ^ xmat(i, qt);
169 zmat(i, qc) = zmat(i, qc) ^ zmat(i, qt);
170 }

173void SymplecticTableau::apply_gate(
174 OpType type, const std::vector<unsigned> &qbs) {
175 switch (type) {
176 case OpType::Z: {
177 apply_Z(qbs.at(0));
178 break;
179 }
180 case OpType::X: {
181 apply_X(qbs.at(0));
182 break;
183 }
184 case OpType::Y: {
185 apply_Z(qbs.at(0));
186 apply_X(qbs.at(0));
187 break;
188 }
189 case OpType::S: {
190 apply_S(qbs.at(0));
191 break;
192 }
193 case OpType::Sdg: {
194 apply_S(qbs.at(0));
195 apply_Z(qbs.at(0));
196 break;
197 }
198 case OpType::V:
199 case OpType::SX: {
200 apply_V(qbs.at(0));
201 break;
202 }
203 case OpType::Vdg:
204 case OpType::SXdg: {
205 apply_V(qbs.at(0));
206 apply_X(qbs.at(0));
207 break;
208 }
209 case OpType::H: {
210 apply_H(qbs.at(0));
211 break;
212 }
213 case OpType::CX: {
214 apply_CX(qbs.at(0), qbs.at(1));
215 break;
216 }
217 case OpType::CY: {
218 apply_S(qbs.at(1));
219 apply_Z(qbs.at(1));
220 apply_CX(qbs.at(0), qbs.at(1));
221 apply_S(qbs.at(1));
222 break;
223 }
224 case OpType::CZ: {
225 apply_H(qbs.at(1));
226 apply_CX(qbs.at(0), qbs.at(1));
227 apply_H(qbs.at(1));
228 break;
229 }
230 case OpType::ZZMax: {
231 apply_H(qbs.at(1));
232 apply_S(qbs.at(0));
233 apply_V(qbs.at(1));
234 apply_CX(qbs.at(0), qbs.at(1));
235 apply_H(qbs.at(1));
236 break;
237 }
238 case OpType::ECR: {
239 apply_S(qbs.at(0));
240 apply_X(qbs.at(0));
241 apply_V(qbs.at(1));
242 apply_X(qbs.at(1));
243 apply_CX(qbs.at(0), qbs.at(1));
244 break;
245 }
246 case OpType::ISWAPMax: {
247 apply_V(qbs.at(0));
248 apply_V(qbs.at(1));
249 apply_CX(qbs.at(0), qbs.at(1));
250 apply_V(qbs.at(0));
251 apply_S(qbs.at(1));
252 apply_Z(qbs.at(1));
253 apply_CX(qbs.at(0), qbs.at(1));
254 apply_V(qbs.at(0));
255 apply_V(qbs.at(1));
256 break;
257 }
258 case OpType::SWAP: {
259 apply_CX(qbs.at(0), qbs.at(1));
260 apply_CX(qbs.at(1), qbs.at(0));
261 apply_CX(qbs.at(0), qbs.at(1));
262 break;
263 }
264 case OpType::BRIDGE: {
265 apply_CX(qbs.at(0), qbs.at(2));
266 break;
267 }
268 case OpType::noop:
269 case OpType::Phase: {
270 break;
271 }
272 default: {
273 throw BadOpType(
274 "Cannot be applied to a SymplecticTableau: not a Clifford gate",
275 type);
276 }
277 }

280void SymplecticTableau::apply_pauli_gadget(
281 const PauliStabiliser &pauli, unsigned half_pis) {
282 unsigned n_qubits = get_n_qubits();
283 if (pauli.string.size() != n_qubits) {
284 throw std::invalid_argument(
285 "Cannot apply pauli gadget to SymplecticTableau; string and tableau "
286 "have different numbers of qubits");
287 }
288 half_pis = half_pis % 4;
289 if (half_pis == 0) return; // Identity
290 if (half_pis == 2) { // Degenerates to product of PI rotations
291 for (unsigned i = 0; i < pauli.string.size(); ++i) {
292 switch (pauli.string.at(i)) {
293 case Pauli::I: {
294 break;
295 }
296 case Pauli::X: {
297 apply_gate(OpType::X, {i});
298 break;
299 }
300 case Pauli::Y: {
301 apply_gate(OpType::Y, {i});
302 break;
303 }
304 case Pauli::Z: {
305 apply_gate(OpType::Z, {i});
306 break;
307 }
308 }
309 }
310 return;
311 }
312
313 // From here, half_pis == 1 or 3
314 // They act the same except for a phase flip on the product term
315 MatrixXb pauli_xrow = MatrixXb::Zero(1, n_qubits);
316 MatrixXb pauli_zrow = MatrixXb::Zero(1, n_qubits);
317 for (unsigned i = 0; i < n_qubits; ++i) {
318 Pauli p = pauli.string.at(i);
319 pauli_xrow(i) = (p == Pauli::X) || (p == Pauli::Y);
320 pauli_zrow(i) = (p == Pauli::Z) || (p == Pauli::Y);
321 }
322 bool phase_flip = pauli.is_real_negative() ^ (half_pis == 3);
323
324 for (unsigned i = 0; i < get_n_rows(); ++i) {
325 bool anti = false;
326 MatrixXb::RowXpr xr = xmat.row(i);
327 MatrixXb::RowXpr zr = zmat.row(i);
328 for (unsigned q = 0; q < n_qubits; ++q) {
329 anti ^= (xr(q) && pauli_zrow(q));
330 anti ^= (zr(q) && pauli_xrow(q));
331 }
332 if (anti) {
333 row_mult(
334 xr, zr, phase(i), pauli_xrow.row(0), pauli_zrow.row(0), phase_flip,
335 i_, xr, zr, phase(i));
336 }
337 }
338}

340MatrixXb SymplecticTableau::anticommuting_rows() const {
341 unsigned n_rows = get_n_rows();
342 MatrixXb res = MatrixXb::Zero(n_rows, n_rows);
343 for (unsigned i = 0; i < n_rows; ++i) {
344 for (unsigned j = 0; j < i; ++j) {
345 bool anti = false;
346 for (unsigned q = 0; q < get_n_qubits(); ++q) {
347 anti ^= (xmat(i, q) && zmat(j, q));
348 anti ^= (xmat(j, q) && zmat(i, q));
349 }
350 res(i, j) = anti;
351 res(j, i) = anti;
352 }
353 }
354 return res;
355}

357unsigned SymplecticTableau::rank() const {
358 // Create a copy in gaussian form and count the empty rows
359 SymplecticTableau copy(*this);
360 copy.gaussian_form();
361 unsigned empty_rows = 0;
362 unsigned n_rows = get_n_rows();
363 for (unsigned i = 0; i < n_rows; ++i) {
364 if (copy.xmat.row(n_rows - 1 - i).isZero() &&
365 copy.zmat.row(n_rows - 1 - i).isZero())
366 ++empty_rows;
367 else
368 break;
369 }
370 return n_rows - empty_rows;
371}

373SymplecticTableau SymplecticTableau::conjugate() const {
374 SymplecticTableau conj(*this);
375 for (unsigned i = 0; i < get_n_rows(); ++i) {
376 unsigned sum = 0;
377 for (unsigned j = 0; j < get_n_qubits(); ++j) {
378 if (xmat(i, j) && zmat(i, j)) ++sum;
379 }
380 if (sum % 2 == 1) conj.phase(i) ^= true;
381 }
382 return conj;
383}

385void SymplecticTableau::gaussian_form() {
386 MatrixXb fullmat = MatrixXb::Zero(get_n_rows(), 2 * get_n_qubits());
387 fullmat(
388 Eigen::placeholders::all, Eigen::seq(0, Eigen::placeholders::last, 2)) =
389 xmat;
390 fullmat(
391 Eigen::placeholders::all, Eigen::seq(1, Eigen::placeholders::last, 2)) =
392 zmat;
393 std::vector<std::pair<unsigned, unsigned>> row_ops =
394 gaussian_elimination_row_ops(fullmat);
395 for (const std::pair<unsigned, unsigned> &op : row_ops) {
396 row_mult(op.first, op.second);
397 }
398}

400void SymplecticTableau::row_mult(
401 const MatrixXb::RowXpr &xa, const MatrixXb::RowXpr &za, const bool &pa,
402 const MatrixXb::RowXpr &xb, const MatrixXb::RowXpr &zb, const bool &pb,
403 Complex phase, MatrixXb::RowXpr &xw, MatrixXb::RowXpr &zw, bool &pw) {
404 if (pa) phase *= -1;
405 if (pb) phase *= -1;
406 for (unsigned i = 0; i < get_n_qubits(); i++) {
407 std::pair<BoolPauli, Complex> res =
408 BoolPauli::mult_lut.at({{xa(i), za(i)}, {xb(i), zb(i)}});
409 xw(i) = res.first.x;
410 zw(i) = res.first.z;
411 phase *= res.second;
412 }
413 pw = (phase == -1.);
414}

416void SymplecticTableau::col_mult(
417 const MatrixXb::ColXpr &a, const MatrixXb::ColXpr &b, bool flip,
418 MatrixXb::ColXpr &w, VectorXb &pw) {
419 for (unsigned i = 0; i < get_n_rows(); i++) {
420 pw(i) = pw(i) ^ (a(i) && (b(i) ^ flip));
421 w(i) = a(i) ^ b(i);
422 }
423}
