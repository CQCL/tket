37// Return a 3-qubit circuit implementing the unitary
38// [ D ]
39// [ D* ]
40// using 4 Rz and 4 CX operations, where D is a 4x4 diagonal unitary matrix.
41// The circuit consists of Rz operations on qubit 0 and CX operations with
42// target qubit 0.
43static Circuit two_qubit_diag_adjoint_plex(const Eigen::Matrix4cd &D) {
44 // Compute angles a_i such that D_ii = e^{-i pi/2 a_i}.
45 static const double f = -2 / PI;
46 double a0 = f * std::arg(D(0, 0));
47 double a1 = f * std::arg(D(1, 1));
48 double a2 = f * std::arg(D(2, 2));
49 double a3 = f * std::arg(D(3, 3));
50 double t0 = (a0 + a1 + a2 + a3) / 4;
51 double t1 = (a0 + a1 - a2 - a3) / 4;
52 double t2 = (a0 - a1 - a2 + a3) / 4;
53 double t3 = (a0 - a1 + a2 - a3) / 4;
54 Circuit circ(3);
55 circ.add_op<unsigned>(OpType::Rz, t0, {0});
56 circ.add_op<unsigned>(OpType::CX, {1, 0});
57 circ.add_op<unsigned>(OpType::Rz, t1, {0});
58 circ.add_op<unsigned>(OpType::CX, {2, 0});
59 circ.add_op<unsigned>(OpType::Rz, t2, {0});
60 circ.add_op<unsigned>(OpType::CX, {1, 0});
61 circ.add_op<unsigned>(OpType::Rz, t3, {0});
62 circ.add_op<unsigned>(OpType::CX, {2, 0});
63 return circ;
64}

66// Return a 3-qubit circuit implementing the unitary
67// [ D ]
68// [ D* ]
69// using 1-qubit and 4 TK2 operations, where D is a 4x4 diagonal unitary matrix.
70// The circuit consists of Rz operations on qubit 0 and TK2 operations.
71static Circuit two_qubit_diag_adjoint_plex_tk(const Eigen::Matrix4cd &D) {
72 // Compute angles a_i such that D_ii = e^{-i pi/2 a_i}.
73 static const double f = -2 / PI;
74 double a0 = f * std::arg(D(0, 0));
75 double a1 = f * std::arg(D(1, 1));
76 double a2 = f * std::arg(D(2, 2));
77 double a3 = f * std::arg(D(3, 3));
78 double t0 = (a0 + a1 + a2 + a3) / 4;
79 double t1 = (a0 + a1 - a2 - a3) / 4;
80 double t2 = (a0 - a1 - a2 + a3) / 4;
81 double t3 = (a0 - a1 + a2 - a3) / 4;
82 Circuit circ(3);
83 circ.add_op<unsigned>(OpType::Rz, t0, {0});
84 circ.append_with_map(
85 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
86 circ.add_op<unsigned>(OpType::Rz, t1, {0});
87 circ.append_with_map(
88 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(2)}, {Qubit(1), Qubit(0)}});
89 circ.add_op<unsigned>(OpType::Rz, t2, {0});
90 circ.append_with_map(
91 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
92 circ.add_op<unsigned>(OpType::Rz, t3, {0});
93 circ.append_with_map(
94 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(2)}, {Qubit(1), Qubit(0)}});
95 return circ;
96}

121// Return a 3-qubit circuit and a unit complex number z which together implement
122// the unitary
123// [ U0 ]
124// [ U1 ]
125// where U0 and U1 are 4x4 unitaries.
126//
127// The unitary is implemented by the circuit followed by the diagonal operator
128// diag(z, z*, z*, z) on qubits 1 and 2.
129//
130// If `extract_final_diagonal` is false, then z=1 and the circuit implements the
131// unitary exactly, using 9 CX gates.
132//
133// If `extract_final_diagonal` is true, then the circuit uses 8 CX gates.
134static std::pair<Circuit, Complex> two_qubit_plex(
135 const Eigen::Matrix4cd &U0, const Eigen::Matrix4cd &U1,
136 bool extract_final_diagonal) {
137 // 1. Decompose U0 U1* as L T L* where L and T are unitary and T is diagonal.
138 Eigen::Matrix4cd U = U0 * U1.adjoint();
139 Eigen::ComplexSchur<Eigen::Matrix4cd> schur(U);
140 Eigen::Matrix4cd L = schur.matrixU();
141 Eigen::Matrix4cd T = schur.matrixT();
142 // By construction T is unitary and upper-triangular, hence diagonal.
143 TKET_ASSERT(T.isDiagonal());
144 // 2. Let D = sqrt(T)
145 Eigen::Matrix4cd D = Eigen::Matrix4cd::Zero();
146 for (unsigned i = 0; i < 4; i++) {
147 D(i, i) = sqrt(T(i, i));
148 }
149 // 3. Compute R such that U0 = L D R and U1 = L D* R.
150 Eigen::Matrix4cd R = D * L.adjoint() * U1;
151
152 // We try conjugating the L and R circuits to see if we can reduce CX count.
153 std::optional<Circuit> best_circ;
154 std::optional<Complex> best_z0;
155 std::optional<unsigned> best_n_cx;
156 for (const Eigen::Matrix4cd &u_conj : get_conj_unitaries()) {
157 // 4. Decompose R into a 2-CX circuit followed by a diagonal.
158 auto u_conj_adj = u_conj.adjoint();
159 auto [R_circ, w0] = decompose_2cx_DV(u_conj * R);
160 Eigen::Vector4cd w = {w0, std::conj(w0), std::conj(w0), w0};
161 auto wD = w.asDiagonal();
162
163 // 5. Construct the circuit.
164 // The diagonal from R's decomposition commutes forward through the controls
165 // on qubits 1 and 2, so can be absorbed into L.
166 Circuit circ(3);
167 unit_map_t qm{{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}};
168 circ.append_with_map(R_circ, qm);
169 circ.append(two_qubit_diag_adjoint_plex(u_conj * D * u_conj_adj));
170 Complex z0;
171 Circuit L_circ;
172 if (extract_final_diagonal) {
173 std::tie(L_circ, z0) = decompose_2cx_DV(L * u_conj_adj * wD);
174 } else {
175 L_circ = two_qubit_canonical(L * u_conj_adj * wD, OpType::CX);
176 z0 = 1.;
177 }
178 circ.append_with_map(L_circ, qm);
179
180 unsigned n_cx = circ.count_gates(OpType::CX);
181 if (!best_circ || best_n_cx > n_cx) {
182 best_circ = circ;
183 best_z0 = z0;
184 best_n_cx = n_cx;
185 }
186 }
187 TKET_ASSERT(best_circ);
188 return {*best_circ, *best_z0};
189}

191// Return a 3-qubit circuit which implements the unitary
192// [ U0 ]
193// [ U1 ]
194// where U0 and U1 are 4x4 unitaries.
195static Circuit two_qubit_plex_tk(
196 const Eigen::Matrix4cd &U0, const Eigen::Matrix4cd &U1) {
197 // 1. Decompose U0 U1* as L T L* where L and T are unitary and T is diagonal.
198 Eigen::Matrix4cd U = U0 * U1.adjoint();
199 Eigen::ComplexSchur<Eigen::Matrix4cd> schur(U);
200 Eigen::Matrix4cd L = schur.matrixU();
201 Eigen::Matrix4cd T = schur.matrixT();
202 // By construction T is unitary and upper-triangular, hence diagonal.
203 TKET_ASSERT(T.isDiagonal());
204 // 2. Let D = sqrt(T)
205 Eigen::Matrix4cd D = Eigen::Matrix4cd::Zero();
206 for (unsigned i = 0; i < 4; i++) {
207 D(i, i) = sqrt(T(i, i));
208 }
209 // 3. Compute R such that U0 = L D R and U1 = L D* R.
210 Eigen::Matrix4cd R = D * L.adjoint() * U1;
211 // 4. Construct the circuit.
212 Circuit circ(3);
213 unit_map_t qm{{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}};
214 circ.append_with_map(two_qubit_canonical(R), qm);
215 circ.append(two_qubit_diag_adjoint_plex_tk(D));
216 circ.append_with_map(two_qubit_canonical(L), qm);
217 return circ;
218}

220// Return a 3-qubit circuit implementing the unitary
221// [ C -S ]
222// [ DS DC ]
223// using TK2 and 1-qubit operations, where C and S are 4x4 real diagonal
224// matrices, C^2 + S^2 = I, and D = diag(1,-1,1,-1).
225//
226// Note that
227// [ C -S ] = U [ C -S ]
228// [ DS DC ] [ S C ]
229// where U represents a CZ on qubits 0 and 2. We convert the CZ operations to CX
230// by adding Hadamards and simplifying H Ry(t) H to Ry(-t).
231static Circuit two_qubit_modified_cossin_circ(
232 const Eigen::Matrix4d &C, const Eigen::Matrix4d &S) {
233 // Compute angles a_i such that C_ii = cos(pi/2 a_i) and S_ii = sin(pi/2 a_i).
234 static const double f = 2 / PI;
235 double a0 = f * atan2(S(0, 0), C(0, 0));
236 double a1 = f * atan2(S(1, 1), C(1, 1));
237 double a2 = f * atan2(S(2, 2), C(2, 2));
238 double a3 = f * atan2(S(3, 3), C(3, 3));
239 double t0 = (a0 + a1 + a2 + a3) / 4;
240 double t1 = (a0 + a1 - a2 - a3) / 4;
241 double t2 = (a0 - a1 - a2 + a3) / 4;
242 double t3 = (a0 - a1 + a2 - a3) / 4;
243 Circuit circ(3);
244 circ.add_op<unsigned>(OpType::Ry, t0, {0});
245 circ.add_op<unsigned>(OpType::H, {0});
246 circ.add_op<unsigned>(OpType::CX, {1, 0});
247 circ.add_op<unsigned>(OpType::Ry, -t1, {0});
248 circ.add_op<unsigned>(OpType::CX, {2, 0});
249 circ.add_op<unsigned>(OpType::Ry, -t2, {0});
250 circ.add_op<unsigned>(OpType::CX, {1, 0});
251 circ.add_op<unsigned>(OpType::H, {0});
252 circ.add_op<unsigned>(OpType::Ry, t3, {0});
253 return circ;
254}

256// Return a 3-qubit circuit implementing the unitary
257// [ C -S ]
258// [ DS DC ]
259// using TK2 and 1-qubit operations, where C and S are 4x4 real diagonal
260// matrices, C^2 + S^2 = I, and D = diag(1,-1,1,-1).
261//
262// Note that
263// [ C -S ] = U [ C -S ]
264// [ DS DC ] [ S C ]
265// where U represents a CZ on qubits 0 and 2. We convert the CZ operations to CX
266// by adding Hadamards and simplifying H Ry(t) H to Ry(-t).
267static Circuit two_qubit_modified_cossin_circ_tk(
268 const Eigen::Matrix4d &C, const Eigen::Matrix4d &S) {
269 // Compute angles a_i such that C_ii = cos(pi/2 a_i) and S_ii = sin(pi/2 a_i).
270 static const double f = 2 / PI;
271 double a0 = f * atan2(S(0, 0), C(0, 0));
272 double a1 = f * atan2(S(1, 1), C(1, 1));
273 double a2 = f * atan2(S(2, 2), C(2, 2));
274 double a3 = f * atan2(S(3, 3), C(3, 3));
275 double t0 = (a0 + a1 + a2 + a3) / 4;
276 double t1 = (a0 + a1 - a2 - a3) / 4;
277 double t2 = (a0 - a1 - a2 + a3) / 4;
278 double t3 = (a0 - a1 + a2 - a3) / 4;
279 Circuit circ(3);
280 circ.add_op<unsigned>(OpType::Ry, t0, {0});
281 circ.add_op<unsigned>(OpType::H, {0});
282 circ.append_with_map(
283 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
284 circ.add_op<unsigned>(OpType::Ry, -t1, {0});
285 circ.append_with_map(
286 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(2)}, {Qubit(1), Qubit(0)}});
287 circ.add_op<unsigned>(OpType::Ry, -t2, {0});
288 circ.append_with_map(
289 CircPool::CX_using_TK2(), {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
290 circ.add_op<unsigned>(OpType::H, {0});
291 circ.add_op<unsigned>(OpType::Ry, t3, {0});
292 return circ;
293}

295// Given matrices U and V, if UV* is a scalar multiple of the identity, return
296// the scalar.
297static std::optional<Complex> id_coeff(
298 const Eigen::Matrix4cd &U, const Eigen::Matrix4cd &V) {
299 Eigen::Matrix4cd W = U * V.adjoint();
300 Complex w = W(0, 0);
301 if (W.isApprox(w * Eigen::Matrix4cd::Identity())) {
302 return w;
303 }
304 if (W.isZero()) {
305 // Eigen's isApprox compares multiplicatively, so doesn't catch this case.
306 return 0.;
307 }
308 return std::nullopt;
309}

311// If the given 8x8 unitary represents a circuit with no entanglement between
312// qubit 0 and the other qubits, return a decomposition into a circuit on qubit
313// 0 and a circuit on qubits 1 and 2. (The qubits on the second circuit are
314// indexed with 0 and 1.)
315static std::optional<std::pair<Circuit, Circuit>> separate_0_12(
316 const Eigen::MatrixXcd &U) {
317 // We want to check whether the unitary is of the form
318 // [ w_00 V w_01 V ]
319 // [ w_10 V w_11 V ]
320 // where W is a 2x2 unitary and V is a 4x4 unitary.
321 // W.l.o.g. we will assume w_00 (or w_10) is real and positive, compute the
322 // w_ij assuming the above form, and then check that the form is correct.
323 Eigen::Matrix4cd U00 = U.topLeftCorner(4, 4);
324 Eigen::Matrix4cd U01 = U.topRightCorner(4, 4);
325 Eigen::Matrix4cd U10 = U.bottomLeftCorner(4, 4);
326 Eigen::Matrix4cd U11 = U.bottomRightCorner(4, 4);
327 // If U is of the desired form, then U_ij U_kl* = w_ij W_kl* I for all
328 // i, j, k, l.
329 std::optional<Complex> w0000 = id_coeff(U00, U00); // |w_00|^2
330 if (!w0000) return std::nullopt;
331 std::optional<Complex> w0101 = id_coeff(U01, U01); // |w_01|^2
332 if (!w0101) return std::nullopt;
333 double x0000 = w0000->real(), y0000 = w0000->imag();
334 double x0101 = w0101->real(), y0101 = w0101->imag();
335 if (std::abs(y0000) > EPS || std::abs(y0101) > EPS || x0000 < -EPS ||
336 x0101 < -EPS) {
337 return std::nullopt;
338 }
339 // If negative because of rounding errors, clamp to 0:
340 if (x0000 < 0.) x0000 = 0.;
341 if (x0101 < 0.) x0101 = 0.;
342 // By unitarity of U, w0000 + w0101 = 1, and so x0000 + x0101 = 1.
343 // Choose the larger one to work with.
344 Complex w00, w01, w10, w11;
345 Eigen::Matrix4cd V;
346 if (x0000 >= x0101) {
347 w00 = sqrt(x0000);
348 V = U00 / w00;
349 std::optional<Complex> w0001 = id_coeff(U00, U01);
350 if (!w0001) return std::nullopt;
351 std::optional<Complex> w0010 = id_coeff(U00, U10);
352 if (!w0010) return std::nullopt;
353 std::optional<Complex> w0011 = id_coeff(U00, U11);
354 if (!w0011) return std::nullopt;
355 w01 = std::conj(*w0001) / w00;
356 w10 = std::conj(*w0010) / w00;
357 w11 = std::conj(*w0011) / w00;
358 } else {
359 w01 = sqrt(x0101);
360 V = U01 / w01;
361 std::optional<Complex> w0100 = id_coeff(U01, U00);
362 if (!w0100) return std::nullopt;
363 std::optional<Complex> w0110 = id_coeff(U01, U10);
364 if (!w0110) return std::nullopt;
365 std::optional<Complex> w0111 = id_coeff(U01, U11);
366 if (!w0111) return std::nullopt;
367 w00 = std::conj(*w0100) / w01;
368 w10 = std::conj(*w0110) / w01;
369 w11 = std::conj(*w0111) / w01;
370 }
371
372 Eigen::Matrix2cd W;
373 W << w00, w01, w10, w11;
374 if (U.isApprox(Eigen::KroneckerProduct(W, V))) {
375 std::vector<double> angs = tk1_angles_from_unitary(W);
376 Circuit c_1q(1);
377 c_1q.add_op<unsigned>(OpType::TK1, {angs[0], angs[1], angs[2]}, {0});
378 c_1q.add_phase(angs[3]);
379 Circuit c_2q = two_qubit_canonical(V);
380 return std::pair<Circuit, Circuit>(c_1q, c_2q);
381 }
382
383 return std::nullopt;
384}

386// Special cases worth handling. This is not necessary for correctness, but
387// allows us to obtain circuits that are more amenable to later optimization.
388static std::optional<Circuit> special_3q_synth(const Eigen::MatrixXcd &U) {
389 static const Eigen::PermutationMatrix<8> P1 = []() {
390 Eigen::VectorXi V1(8);
391 V1 << 0, 1, 4, 5, 2, 3, 6, 7;
392 return Eigen::PermutationMatrix<8>(V1);
393 }();
394 static const Eigen::PermutationMatrix<8> P2 = []() {
395 Eigen::VectorXi V2(8);
396 V2 << 0, 4, 2, 6, 1, 5, 3, 7;
397 return Eigen::PermutationMatrix<8>(V2);
398 }();
399
400 // Try separating qubit 0 from qubits 1 and 2:
401 std::optional<std::pair<Circuit, Circuit>> c0 = separate_0_12(U);
402 if (c0) {
403 Circuit c_1q = c0->first;
404 Circuit c_2q = c0->second;
405 Circuit c(3);
406 c.append(c_1q);
407 c.append_with_map(c_2q, {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}});
408 return c;
409 }
410
411 // Try separating qubit 1 from qubits 0 and 2:
412 std::optional<std::pair<Circuit, Circuit>> c1 = separate_0_12(P1 * U * P1);
413 if (c1) {
414 Circuit c_1q = c1->first;
415 Circuit c_2q = c1->second;
416 Circuit c(3);
417 c.append_with_map(c_1q, {{Qubit(0), Qubit(1)}});
418 c.append_with_map(c_2q, {{Qubit(1), Qubit(2)}});
419 return c;
420 }
421
422 // Try separating qubit 2 from qubits 0 and 1:
423 std::optional<std::pair<Circuit, Circuit>> c2 = separate_0_12(P2 * U * P2);
424 if (c2) {
425 Circuit c_1q = c2->first;
426 Circuit c_2q = c2->second;
427 Circuit c(3);
428 c.append_with_map(c_1q, {{Qubit(0), Qubit(2)}});
429 c.append_with_map(c_2q, {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
430 return c;
431 }
432
433 return std::nullopt;
434}

436Circuit three_qubit_synthesis(const Eigen::MatrixXcd &U) {
437 if (U.rows() != 8 || U.cols() != 8) {
438 throw std::invalid_argument("Wrong-size matrix for three-qubit synthesis");
439 }
440
441 std::optional<Circuit> c_special = special_3q_synth(U);
442 if (c_special) return *c_special;
443
444 auto [l0, l1, r0, r1, c, s] = CS_decomp(U);
445 auto [R_circ, z0] = two_qubit_plex(r0, r1, true);
446 Complex z1 = std::conj(z0);
447 Circuit circ(3);
448 circ.append(R_circ);
449 circ.append(two_qubit_modified_cossin_circ(c, s));
450 // We chopped off the last CZ (on qubits 0 and 2) from the circuit
451 // implementing the CS decomposition. Account for this by changing the signs
452 // of columns 1 and 3 of l1.
453 //
454 // We also carried a diagonal from the earlier subcircuit, which commutes
455 // through the controls in the middle circuit and merges with l0 and l1.
456 //
457 // Together these imply the following adjustments:
458 l0.col(0) *= z0;
459 l0.col(1) *= z1;
460 l0.col(2) *= z1;
461 l0.col(3) *= z0;
462 l1.col(0) *= z0;
463 l1.col(1) *= -z1;
464 l1.col(2) *= z1;
465 l1.col(3) *= -z0;
466 circ.append(two_qubit_plex(l0, l1, false).first);
467 return circ;
468}

470Circuit three_qubit_tk_synthesis(const Eigen::MatrixXcd &U) {
471 if (U.rows() != 8 || U.cols() != 8) {
472 throw std::invalid_argument("Wrong-size matrix for three-qubit synthesis");
473 }
474
475 std::optional<Circuit> c_special = special_3q_synth(U);
476 if (c_special) return *c_special;
477
478 auto [l0, l1, r0, r1, c, s] = CS_decomp(U);
479 Circuit circ(3);
480 circ.append(two_qubit_plex_tk(r0, r1));
481 circ.append(two_qubit_modified_cossin_circ_tk(c, s));
482 // We chopped off the last CZ (on qubits 0 and 2) from the circuit
483 // implementing the CS decomposition. Account for this by changing the signs
484 // of columns 1 and 3 of l1.
485 l1.col(1) *= -1;
486 l1.col(3) *= -1;
487 circ.append(two_qubit_plex_tk(l0, l1));
488 return circ;
489}

491Eigen::MatrixXcd get_3q_unitary(const Circuit &c) {
492 if (c.n_qubits() != 3) {
493 throw CircuitInvalidity("Circuit in get_3q_unitary must have 3 qubits");
494 }
495
496 // Construct map from qubits to indices {0,1,2}.
497 qubit_vector_t all_qbs = c.all_qubits();
498 std::map<Qubit, unsigned> idx;
499 for (unsigned i = 0; i < 3; i++) {
500 idx[all_qbs[i]] = i;
501 }
502
503 // Step through commands, building unitary as we go.
504 Eigen::MatrixXcd U = Eigen::MatrixXcd::Identity(8, 8);
505 for (const Command &cmd : c) {
506 qubit_vector_t qbs = cmd.get_qubits();
507 Op_ptr op = cmd.get_op_ptr();
508 Eigen::MatrixXcd u = op->get_unitary();
509 Eigen::MatrixXcd M = Eigen::MatrixXcd::Zero(8, 8);
510 switch (qbs.size()) {
511 case 1: {
512 // Construct the 8x8 matrix representing u.
513 unsigned i = idx[qbs[0]];
514 switch (i) {
515 case 0:
516 M(0, 0) = M(1, 1) = M(2, 2) = M(3, 3) = u(0, 0);
517 M(0, 4) = M(1, 5) = M(2, 6) = M(3, 7) = u(0, 1);
518 M(4, 0) = M(5, 1) = M(6, 2) = M(7, 3) = u(1, 0);
519 M(4, 4) = M(5, 5) = M(6, 6) = M(7, 7) = u(1, 1);
520 break;
521 case 1:
522 M(0, 0) = M(1, 1) = M(4, 4) = M(5, 5) = u(0, 0);
523 M(0, 2) = M(1, 3) = M(4, 6) = M(5, 7) = u(0, 1);
524 M(2, 0) = M(3, 1) = M(6, 4) = M(7, 5) = u(1, 0);
525 M(2, 2) = M(3, 3) = M(6, 6) = M(7, 7) = u(1, 1);
526 break;
527 case 2:
528 M(0, 0) = M(2, 2) = M(4, 4) = M(6, 6) = u(0, 0);
529 M(0, 1) = M(2, 3) = M(4, 5) = M(6, 7) = u(0, 1);
530 M(1, 0) = M(3, 2) = M(5, 4) = M(7, 6) = u(1, 0);
531 M(1, 1) = M(3, 3) = M(5, 5) = M(7, 7) = u(1, 1);
532 break;
533 default:
534 TKET_ASSERT(!"Invalid index");
535 }
536 break;
537 }
538 case 2: {
539 // Note reversal of indices here:
540 unsigned i = 2 - idx[qbs[0]];
541 unsigned j = 2 - idx[qbs[1]];
542 // Construct the 8x8 matrix representing the gate.
543 // Let k be the untouched index, so that {i,j,k} = {0,1,2}:
544 unsigned k = 3 - i - j;
545 unsigned t = 1 << k;
546 for (unsigned s0 = 0; s0 < 4; s0++) {
547 unsigned s0_ = ((s0 >> 1) << i) + ((s0 & 1) << j);
548 for (unsigned s1 = 0; s1 < 4; s1++) {
549 unsigned s1_ = ((s1 >> 1) << i) + ((s1 & 1) << j);
550 M(s0_, s1_) = M(s0_ + t, s1_ + t) = u(s0, s1);
551 }
552 }
553 break;
554 }
555 default:
556 throw CircuitInvalidity(
557 "Circuit in get_3q_unitary contains gates with more than 2 qubits");
558 }
559 U = M * U;
560 }
561
562 return std::exp(i_ * PI * eval_expr(c.get_phase()).value()) * U;
563}
