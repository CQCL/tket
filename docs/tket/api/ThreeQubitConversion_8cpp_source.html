39 // Return a 3-qubit circuit implementing the unitary
40 // [ D ]
41 // [ D* ]
42 // using 4 Rz and 4 CX operations, where D is a 4x4 diagonal unitary matrix.
43 // The circuit consists of Rz operations on qubit 0 and CX operations with
44 // target qubit 0.
45 static Circuit two_qubit_diag_adjoint_plex(const Eigen::Matrix4cd &D) {
46  // Compute angles a_i such that D_ii = e^{-i pi/2 a_i}.
47  static const double f = -2 / PI;
48  double a0 = f * std::arg(D(0, 0));
49  double a1 = f * std::arg(D(1, 1));
50  double a2 = f * std::arg(D(2, 2));
51  double a3 = f * std::arg(D(3, 3));
52  double t0 = (a0 + a1 + a2 + a3) / 4;
53  double t1 = (a0 + a1 - a2 - a3) / 4;
54  double t2 = (a0 - a1 - a2 + a3) / 4;
55  double t3 = (a0 - a1 + a2 - a3) / 4;
56  Circuit circ(3);
57  circ.add_op<unsigned>(OpType::Rz, t0, {0});
58  circ.add_op<unsigned>(OpType::CX, {1, 0});
59  circ.add_op<unsigned>(OpType::Rz, t1, {0});
60  circ.add_op<unsigned>(OpType::CX, {2, 0});
61  circ.add_op<unsigned>(OpType::Rz, t2, {0});
62  circ.add_op<unsigned>(OpType::CX, {1, 0});
63  circ.add_op<unsigned>(OpType::Rz, t3, {0});
64  circ.add_op<unsigned>(OpType::CX, {2, 0});
65  return circ;
66 }

123 // Return a 3-qubit circuit and a unit complex number z which together implement
124 // the unitary
125 // [ U0 ]
126 // [ U1 ]
127 // where U0 and U1 are 4x4 unitaries.
128 //
129 // The unitary is implemented by the circuit followed by the diagonal operator
130 // diag(z, z*, z*, z) on qubits 1 and 2.
131 //
132 // If `extract_final_diagonal` is false, then z=1 and the circuit implements the
133 // unitary exactly, using 9 CX gates.
134 //
135 // If `extract_final_diagonal` is true, then the circuit uses 8 CX gates.
136 static std::pair<Circuit, Complex> two_qubit_plex(
137  const Eigen::Matrix4cd &U0, const Eigen::Matrix4cd &U1,
138  bool extract_final_diagonal) {
139  // 1. Decompose U0 U1* as L T L* where L and T are unitary and T is diagonal.
140  Eigen::Matrix4cd U = U0 * U1.adjoint();
141  Eigen::ComplexSchur<Eigen::Matrix4cd> schur(U);
142  Eigen::Matrix4cd L = schur.matrixU();
143  Eigen::Matrix4cd T = schur.matrixT();
144  // By construction T is unitary and upper-triangular, hence diagonal.
145  TKET_ASSERT(T.isDiagonal());
146  // 2. Let D = sqrt(T)
147  Eigen::Matrix4cd D = Eigen::Matrix4cd::Zero();
148  for (unsigned i = 0; i < 4; i++) {
149  D(i, i) = sqrt(T(i, i));
150  }
151  // 3. Compute R such that U0 = L D R and U1 = L D* R.
152  Eigen::Matrix4cd R = D * L.adjoint() * U1;
153 
154  // We try conjugating the L and R circuits to see if we can reduce CX count.
155  std::optional<Circuit> best_circ;
156  std::optional<Complex> best_z0;
157  std::optional<unsigned> best_n_cx;
158  for (const Eigen::Matrix4cd &u_conj : get_conj_unitaries()) {
159  // 4. Decompose R into a 2-CX circuit followed by a diagonal.
160  auto u_conj_adj = u_conj.adjoint();
161  auto [R_circ, w0] = decompose_2cx_DV(u_conj * R);
162  Eigen::Vector4cd w = {w0, std::conj(w0), std::conj(w0), w0};
163  auto wD = w.asDiagonal();
164 
165  // 5. Construct the circuit.
166  // The diagonal from R's decomposition commutes forward through the controls
167  // on qubits 1 and 2, so can be absorbed into L.
168  Circuit circ(3);
169  unit_map_t qm{{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}};
170  circ.append_with_map(R_circ, qm);
171  circ.append(two_qubit_diag_adjoint_plex(u_conj * D * u_conj_adj));
172  Complex z0;
173  Circuit L_circ;
174  if (extract_final_diagonal) {
175  std::tie(L_circ, z0) = decompose_2cx_DV(L * u_conj_adj * wD);
176  } else {
177  L_circ = two_qubit_canonical(L * u_conj_adj * wD, OpType::CX);
178  z0 = 1.;
179  }
180  circ.append_with_map(L_circ, qm);
181 
182  unsigned n_cx = circ.count_gates(OpType::CX);
183  if (!best_circ || best_n_cx > n_cx) {
184  best_circ = circ;
185  best_z0 = z0;
186  best_n_cx = n_cx;
187  }
188  }
189  TKET_ASSERT(best_circ);
190  return {*best_circ, *best_z0};
191 }

193 // Return a 3-qubit circuit which implements the unitary
194 // [ U0 ]
195 // [ U1 ]
196 // where U0 and U1 are 4x4 unitaries.
197 static Circuit two_qubit_plex_tk(
198  const Eigen::Matrix4cd &U0, const Eigen::Matrix4cd &U1) {
199  // 1. Decompose U0 U1* as L T L* where L and T are unitary and T is diagonal.
200  Eigen::Matrix4cd U = U0 * U1.adjoint();
201  Eigen::ComplexSchur<Eigen::Matrix4cd> schur(U);
202  Eigen::Matrix4cd L = schur.matrixU();
203  Eigen::Matrix4cd T = schur.matrixT();
204  // By construction T is unitary and upper-triangular, hence diagonal.
205  TKET_ASSERT(T.isDiagonal());
206  // 2. Let D = sqrt(T)
207  Eigen::Matrix4cd D = Eigen::Matrix4cd::Zero();
208  for (unsigned i = 0; i < 4; i++) {
209  D(i, i) = sqrt(T(i, i));
210  }
211  // 3. Compute R such that U0 = L D R and U1 = L D* R.
212  Eigen::Matrix4cd R = D * L.adjoint() * U1;
213  // 4. Construct the circuit.
214  Circuit circ(3);
215  unit_map_t qm{{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}};
216  circ.append_with_map(two_qubit_canonical(R), qm);
217  circ.append(two_qubit_diag_adjoint_plex_tk(D));
218  circ.append_with_map(two_qubit_canonical(L), qm);
219  return circ;
220 }

222 // Return a 3-qubit circuit implementing the unitary
223 // [ C -S ]
224 // [ DS DC ]
225 // using TK2 and 1-qubit operations, where C and S are 4x4 real diagonal
226 // matrices, C^2 + S^2 = I, and D = diag(1,-1,1,-1).
227 //
228 // Note that
229 // [ C -S ] = U [ C -S ]
230 // [ DS DC ] [ S C ]
231 // where U represents a CZ on qubits 0 and 2. We convert the CZ operations to CX
232 // by adding Hadamards and simplifying H Ry(t) H to Ry(-t).
233 static Circuit two_qubit_modified_cossin_circ(
234  const Eigen::Matrix4d &C, const Eigen::Matrix4d &S) {
235  // Compute angles a_i such that C_ii = cos(pi/2 a_i) and S_ii = sin(pi/2 a_i).
236  static const double f = 2 / PI;
237  double a0 = f * atan2(S(0, 0), C(0, 0));
238  double a1 = f * atan2(S(1, 1), C(1, 1));
239  double a2 = f * atan2(S(2, 2), C(2, 2));
240  double a3 = f * atan2(S(3, 3), C(3, 3));
241  double t0 = (a0 + a1 + a2 + a3) / 4;
242  double t1 = (a0 + a1 - a2 - a3) / 4;
243  double t2 = (a0 - a1 - a2 + a3) / 4;
244  double t3 = (a0 - a1 + a2 - a3) / 4;
245  Circuit circ(3);
246  circ.add_op<unsigned>(OpType::Ry, t0, {0});
247  circ.add_op<unsigned>(OpType::H, {0});
248  circ.add_op<unsigned>(OpType::CX, {1, 0});
249  circ.add_op<unsigned>(OpType::Ry, -t1, {0});
250  circ.add_op<unsigned>(OpType::CX, {2, 0});
251  circ.add_op<unsigned>(OpType::Ry, -t2, {0});
252  circ.add_op<unsigned>(OpType::CX, {1, 0});
253  circ.add_op<unsigned>(OpType::H, {0});
254  circ.add_op<unsigned>(OpType::Ry, t3, {0});
255  return circ;
256 }

297 // Given matrices U and V, if UV* is a scalar multiple of the identity, return
298 // the scalar.
299 static std::optional<Complex> id_coeff(
300  const Eigen::Matrix4cd &U, const Eigen::Matrix4cd &V) {
301  Eigen::Matrix4cd W = U * V.adjoint();
302  Complex w = W(0, 0);
303  if (W.isApprox(w * Eigen::Matrix4cd::Identity())) {
304  return w;
305  }
306  if (W.isZero()) {
307  // Eigen's isApprox compares multiplicatively, so doesn't catch this case.
308  return 0.;
309  }
310  return std::nullopt;
311 }

313 // If the given 8x8 unitary represents a circuit with no entanglement between
314 // qubit 0 and the other qubits, return a decomposition into a circuit on qubit
315 // 0 and a circuit on qubits 1 and 2. (The qubits on the second circuit are
316 // indexed with 0 and 1.)
317 static std::optional<std::pair<Circuit, Circuit>> separate_0_12(
318  const Eigen::MatrixXcd &U) {
319  // We want to check whether the unitary is of the form
320  // [ w_00 V w_01 V ]
321  // [ w_10 V w_11 V ]
322  // where W is a 2x2 unitary and V is a 4x4 unitary.
323  // W.l.o.g. we will assume w_00 (or w_10) is real and positive, compute the
324  // w_ij assuming the above form, and then check that the form is correct.
325  Eigen::Matrix4cd U00 = U.topLeftCorner(4, 4);
326  Eigen::Matrix4cd U01 = U.topRightCorner(4, 4);
327  Eigen::Matrix4cd U10 = U.bottomLeftCorner(4, 4);
328  Eigen::Matrix4cd U11 = U.bottomRightCorner(4, 4);
329  // If U is of the desired form, then U_ij U_kl* = w_ij W_kl* I for all
330  // i, j, k, l.
331  std::optional<Complex> w0000 = id_coeff(U00, U00); // |w_00|^2
332  if (!w0000) return std::nullopt;
333  std::optional<Complex> w0101 = id_coeff(U01, U01); // |w_01|^2
334  if (!w0101) return std::nullopt;
335  double x0000 = w0000->real(), y0000 = w0000->imag();
336  double x0101 = w0101->real(), y0101 = w0101->imag();
337  if (std::abs(y0000) > EPS || std::abs(y0101) > EPS || x0000 < -EPS ||
338  x0101 < -EPS) {
339  return std::nullopt;
340  }
341  // If negative because of rounding errors, clamp to 0:
342  if (x0000 < 0.) x0000 = 0.;
343  if (x0101 < 0.) x0101 = 0.;
344  // By unitarity of U, w0000 + w0101 = 1, and so x0000 + x0101 = 1.
345  // Choose the larger one to work with.
346  Complex w00, w01, w10, w11;
347  Eigen::Matrix4cd V;
348  if (x0000 >= x0101) {
349  w00 = sqrt(x0000);
350  V = U00 / w00;
351  std::optional<Complex> w0001 = id_coeff(U00, U01);
352  if (!w0001) return std::nullopt;
353  std::optional<Complex> w0010 = id_coeff(U00, U10);
354  if (!w0010) return std::nullopt;
355  std::optional<Complex> w0011 = id_coeff(U00, U11);
356  if (!w0011) return std::nullopt;
357  w01 = std::conj(*w0001) / w00;
358  w10 = std::conj(*w0010) / w00;
359  w11 = std::conj(*w0011) / w00;
360  } else {
361  w01 = sqrt(x0101);
362  V = U01 / w01;
363  std::optional<Complex> w0100 = id_coeff(U01, U00);
364  if (!w0100) return std::nullopt;
365  std::optional<Complex> w0110 = id_coeff(U01, U10);
366  if (!w0110) return std::nullopt;
367  std::optional<Complex> w0111 = id_coeff(U01, U11);
368  if (!w0111) return std::nullopt;
369  w00 = std::conj(*w0100) / w01;
370  w10 = std::conj(*w0110) / w01;
371  w11 = std::conj(*w0111) / w01;
372  }
373 
374  Eigen::Matrix2cd W;
375  W << w00, w01, w10, w11;
376  if (U.isApprox(Eigen::KroneckerProduct(W, V))) {
377  std::vector<double> angs = tk1_angles_from_unitary(W);
378  Circuit c_1q(1);
379  c_1q.add_op<unsigned>(OpType::TK1, {angs[0], angs[1], angs[2]}, {0});
380  c_1q.add_phase(angs[3]);
381  Circuit c_2q = two_qubit_canonical(V);
382  return std::pair<Circuit, Circuit>(c_1q, c_2q);
383  }
384 
385  return std::nullopt;
386 }

388 // Special cases worth handling. This is not necessary for correctness, but
389 // allows us to obtain circuits that are more amenable to later optimization.
390 static std::optional<Circuit> special_3q_synth(const Eigen::MatrixXcd &U) {
391  static const Eigen::PermutationMatrix<8> P1 = []() {
392  Eigen::VectorXi V1(8);
393  V1 << 0, 1, 4, 5, 2, 3, 6, 7;
394  return Eigen::PermutationMatrix<8>(V1);
395  }();
396  static const Eigen::PermutationMatrix<8> P2 = []() {
397  Eigen::VectorXi V2(8);
398  V2 << 0, 4, 2, 6, 1, 5, 3, 7;
399  return Eigen::PermutationMatrix<8>(V2);
400  }();
401 
402  // Try separating qubit 0 from qubits 1 and 2:
403  std::optional<std::pair<Circuit, Circuit>> c0 = separate_0_12(U);
404  if (c0) {
405  Circuit c_1q = c0->first;
406  Circuit c_2q = c0->second;
407  Circuit c(3);
408  c.append(c_1q);
409  c.append_with_map(c_2q, {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(2)}});
410  return c;
411  }
412 
413  // Try separating qubit 1 from qubits 0 and 2:
414  std::optional<std::pair<Circuit, Circuit>> c1 = separate_0_12(P1 * U * P1);
415  if (c1) {
416  Circuit c_1q = c1->first;
417  Circuit c_2q = c1->second;
418  Circuit c(3);
419  c.append_with_map(c_1q, {{Qubit(0), Qubit(1)}});
420  c.append_with_map(c_2q, {{Qubit(1), Qubit(2)}});
421  return c;
422  }
423 
424  // Try separating qubit 2 from qubits 0 and 1:
425  std::optional<std::pair<Circuit, Circuit>> c2 = separate_0_12(P2 * U * P2);
426  if (c2) {
427  Circuit c_1q = c2->first;
428  Circuit c_2q = c2->second;
429  Circuit c(3);
430  c.append_with_map(c_1q, {{Qubit(0), Qubit(2)}});
431  c.append_with_map(c_2q, {{Qubit(0), Qubit(1)}, {Qubit(1), Qubit(0)}});
432  return c;
433  }
434 
435  return std::nullopt;
436 }

438 Circuit three_qubit_synthesis(const Eigen::MatrixXcd &U) {
439  if (U.rows() != 8 || U.cols() != 8) {
440  throw std::invalid_argument("Wrong-size matrix for three-qubit synthesis");
441  }
442 
443  std::optional<Circuit> c_special = special_3q_synth(U);
444  if (c_special) return *c_special;
445 
446  auto [l0, l1, r0, r1, c, s] = CS_decomp(U);
447  auto [R_circ, z0] = two_qubit_plex(r0, r1, true);
448  Complex z1 = std::conj(z0);
449  Circuit circ(3);
450  circ.append(R_circ);
451  circ.append(two_qubit_modified_cossin_circ(c, s));
452  // We chopped off the last CZ (on qubits 0 and 2) from the circuit
453  // implementing the CS decomposition. Account for this by changing the signs
454  // of columns 1 and 3 of l1.
455  //
456  // We also carried a diagonal from the earlier subcircuit, which commutes
457  // through the controls in the middle circuit and merges with l0 and l1.
458  //
459  // Together these imply the following adjustments:
460  l0.col(0) *= z0;
461  l0.col(1) *= z1;
462  l0.col(2) *= z1;
463  l0.col(3) *= z0;
464  l1.col(0) *= z0;
465  l1.col(1) *= -z1;
466  l1.col(2) *= z1;
467  l1.col(3) *= -z0;
468  circ.append(two_qubit_plex(l0, l1, false).first);
469  return circ;
470 }

472 Circuit three_qubit_tk_synthesis(const Eigen::MatrixXcd &U) {
473  if (U.rows() != 8 || U.cols() != 8) {
474  throw std::invalid_argument("Wrong-size matrix for three-qubit synthesis");
475  }
476 
477  std::optional<Circuit> c_special = special_3q_synth(U);
478  if (c_special) return *c_special;
479 
480  auto [l0, l1, r0, r1, c, s] = CS_decomp(U);
481  Circuit circ(3);
482  circ.append(two_qubit_plex_tk(r0, r1));
483  circ.append(two_qubit_modified_cossin_circ_tk(c, s));
484  // We chopped off the last CZ (on qubits 0 and 2) from the circuit
485  // implementing the CS decomposition. Account for this by changing the signs
486  // of columns 1 and 3 of l1.
487  l1.col(1) *= -1;
488  l1.col(3) *= -1;
489  circ.append(two_qubit_plex_tk(l0, l1));
490  return circ;
491 }

493 Eigen::MatrixXcd get_3q_unitary(const Circuit &c) {
494  if (c.n_qubits() != 3) {
495  throw CircuitInvalidity("Circuit in get_3q_unitary must have 3 qubits");
496  }
497 
498  // Construct map from qubits to indices {0,1,2}.
499  qubit_vector_t all_qbs = c.all_qubits();
500  std::map<Qubit, unsigned> idx;
501  for (unsigned i = 0; i < 3; i++) {
502  idx[all_qbs[i]] = i;
503  }
504 
505  // Step through commands, building unitary as we go.
506  Eigen::MatrixXcd U = Eigen::MatrixXcd::Identity(8, 8);
507  for (const Command &cmd : c) {
508  qubit_vector_t qbs = cmd.get_qubits();
509  Op_ptr op = cmd.get_op_ptr();
510  Gate_ptr gate = as_gate_ptr(op);
511  if (!gate) {
512  throw CircuitInvalidity("Circuit in get_3q_unitary not unitary");
513  }
514  Eigen::MatrixXcd M = Eigen::MatrixXcd::Zero(8, 8);
515  switch (qbs.size()) {
516  case 1: {
517  std::vector<Expr> angles = gate->get_tk1_angles();
518  Eigen::Matrix2cd u = get_matrix_from_tk1_angles(angles);
519  // Construct the 8x8 matrix representing u.
520  unsigned i = idx[qbs[0]];
521  switch (i) {
522  case 0:
523  M(0, 0) = M(1, 1) = M(2, 2) = M(3, 3) = u(0, 0);
524  M(0, 4) = M(1, 5) = M(2, 6) = M(3, 7) = u(0, 1);
525  M(4, 0) = M(5, 1) = M(6, 2) = M(7, 3) = u(1, 0);
526  M(4, 4) = M(5, 5) = M(6, 6) = M(7, 7) = u(1, 1);
527  break;
528  case 1:
529  M(0, 0) = M(1, 1) = M(4, 4) = M(5, 5) = u(0, 0);
530  M(0, 2) = M(1, 3) = M(4, 6) = M(5, 7) = u(0, 1);
531  M(2, 0) = M(3, 1) = M(6, 4) = M(7, 5) = u(1, 0);
532  M(2, 2) = M(3, 3) = M(6, 6) = M(7, 7) = u(1, 1);
533  break;
534  case 2:
535  M(0, 0) = M(2, 2) = M(4, 4) = M(6, 6) = u(0, 0);
536  M(0, 1) = M(2, 3) = M(4, 5) = M(6, 7) = u(0, 1);
537  M(1, 0) = M(3, 2) = M(5, 4) = M(7, 6) = u(1, 0);
538  M(1, 1) = M(3, 3) = M(5, 5) = M(7, 7) = u(1, 1);
539  break;
540  default:
541  TKET_ASSERT(!"Invalid index");
542  }
543  break;
544  }
545  case 2: {
546  Eigen::Matrix4cd m = gate->get_unitary();
547  // Note reversal of indices here:
548  unsigned i = 2 - idx[qbs[0]];
549  unsigned j = 2 - idx[qbs[1]];
550  // Construct the 8x8 matrix representing the gate.
551  // Let k be the untouched index, so that {i,j,k} = {0,1,2}:
552  unsigned k = 3 - i - j;
553  unsigned t = 1 << k;
554  for (unsigned s0 = 0; s0 < 4; s0++) {
555  unsigned s0_ = ((s0 >> 1) << i) + ((s0 & 1) << j);
556  for (unsigned s1 = 0; s1 < 4; s1++) {
557  unsigned s1_ = ((s1 >> 1) << i) + ((s1 & 1) << j);
558  M(s0_, s1_) = M(s0_ + t, s1_ + t) = m(s0, s1);
559  }
560  }
561  break;
562  }
563  default:
564  throw CircuitInvalidity(
565  "Circuit in get_3q_unitary contains gates with more than 2 qubits");
566  }
567  U = M * U;
568  }
569 
570  return std::exp(i_ * PI * eval_expr(c.get_phase()).value()) * U;
571 }
