43 // Helper class defining a pure-quantum subcircuit of up to 3 qubits.
44 class QInteraction {
45  public:
46  // Construct an empty subcircuit with a single edge that is both input and
47  // output, and no vertices.
48  QInteraction(const Circuit &circ, const Edge &e) : circ_(circ) {
49  in_edges_.push_back(e);
50  out_edges_.push_back(e);
51  n_wires_ = 1;
52  }
53 
54  // Combine with another subcircuit disjoint from this. Disjointness, and the
55  // fact that the combined subcircuit has at most three wires, is assumed.
56  void combine(const QInteraction &other) {
57  in_edges_.insert(
58  in_edges_.end(), other.in_edges_.begin(), other.in_edges_.end());
59  out_edges_.insert(
60  out_edges_.end(), other.out_edges_.begin(), other.out_edges_.end());
61  n_wires_ += other.n_wires_;
62  vertices_.insert(other.vertices_.begin(), other.vertices_.end());
63  }
64 
65  EdgeVec out_edges() const { return out_edges_; }
66 
67  VertexSet vertices() const { return vertices_; }
68 
69  unsigned n_wires() const { return n_wires_; }
70 
71  unsigned n_vertices() const { return vertices_.size(); }
72 
73  Subcircuit subcircuit() const { return {in_edges_, out_edges_, vertices_}; }
74 
75  // Append a vertex following the subcircuit. It is assumed that every input
76  // edge of the vertex matches exactly one output edge of the existing
77  // subcircuit.
78  void append(const Vertex &v) {
79  EdgeVec v_ins = circ_.get_in_edges_of_type(v, EdgeType::Quantum);
80  EdgeVec v_outs = circ_.get_out_edges_of_type(v, EdgeType::Quantum);
81  unsigned n = v_ins.size();
82  TKET_ASSERT(n == v_outs.size());
83  TKET_ASSERT(n <= n_wires_);
84  for (unsigned i = 0; i < n; i++) {
85  Edge e_new_0 = v_ins[i];
86  Edge e_new_1 = v_outs[i];
87  bool matched = false;
88  for (unsigned j = 0; j < n_wires_; j++) {
89  if (out_edges_[j] == e_new_0) {
90  TKET_ASSERT(!matched);
91  out_edges_[j] = e_new_1;
92  matched = true;
93  }
94  }
95  TKET_ASSERT(matched);
96  }
97  vertices_.insert(v);
98  }
99 
100  private:
101  const Circuit &circ_;
102  EdgeVec in_edges_;
103  EdgeVec out_edges_;
104  unsigned n_wires_; // number of in/out edge pairs
105  VertexSet vertices_; // all internal vertices
106 };

110 // Candidate substitution for a 2- or 3-qubit circuit.
111 static Circuit candidate_sub(const Circuit &circ, OpType target_2qb_gate) {
112  unsigned n_qb = circ.n_qubits();
113  if (n_qb == 2) {
114  Circuit repl = two_qubit_canonical(get_matrix_from_2qb_circ(circ));
115  // TODO Remove this once `clifford_simp` supports TK2.
116  if (target_2qb_gate == OpType::CX) {
117  clifford_simp(false).apply(repl);
118  }
119  return repl;
120  } else {
121  TKET_ASSERT(n_qb == 3);
122  Eigen::MatrixXcd U = get_3q_unitary(circ);
123  if (target_2qb_gate == OpType::CX) {
124  Circuit repl = three_qubit_synthesis(U);
125  normalise_TK2().apply(repl);
126  decompose_TK2().apply(repl);
127  clifford_simp(false).apply(repl);
128  squash_1qb_to_tk1().apply(repl);
129  return repl;
130  } else {
131  TKET_ASSERT(target_2qb_gate == OpType::TK2);
132  Circuit repl = three_qubit_tk_synthesis(U);
133  squash_1qb_to_tk1().apply(repl);
134  return repl;
135  }
136  }
137 }

139 // Helper class representing a system of disjoint interactions, each with at
140 // most three qubits. The interactions are represented by integer labels.
141 class QISystem {
142  public:
143  // Construct an empty system.
144  explicit QISystem(Circuit &circ, OpType target_2qb_gate)
145  : circ_(circ),
146  target_2qb_gate_(target_2qb_gate),
147  bin_(),
148  interactions_(),
149  idx_(0) {}
150 
151  // Add a new interaction to the system consisting of a single edge, and
152  // return its index.
153  int create_new_interaction_from_edge(const Edge &e) {
154  TKET_ASSERT(!interactions_.contains(idx_));
155  interactions_[idx_] = std::make_unique<QInteraction>(circ_, e);
156  return idx_++;
157  }
158 
159  // Return the set of (indices of) interactions in the system that have v as a
160  // direct successor.
161  std::vector<int> interactions_feeding_vertex(const Vertex &v) const {
162  EdgeVec edges = circ_.get_in_edges_of_type(v, EdgeType::Quantum);
163  std::vector<int> meets;
164  for (const auto &[i, I] : interactions_) {
165  EdgeVec I_outs = I->out_edges();
166  EdgeSet I_outset = {I_outs.begin(), I_outs.end()};
167  if (std::any_of(edges.begin(), edges.end(), [&I_outset](const Edge &e) {
168  return I_outset.contains(e);
169  })) {
170  meets.push_back(i);
171  }
172  }
173  return meets;
174  }
175 
176  // The total width (number of wires) of a subset of the interactions.
177  unsigned total_n_wires(const std::vector<int> &S) const {
178  unsigned total = 0;
179  for (int i : S) {
180  total += interactions_.at(i)->n_wires();
181  }
182  return total;
183  }
184 
185  // From a set of indices, choose the one indexing the largest interaction,
186  // in terms of vertex count.
187  int largest_interaction(const std::vector<int> &S) const {
188  return *std::max_element(
189  S.begin(), S.end(), [this](const int &i0, const int &i1) {
190  return interactions_.at(i0)->n_vertices() <
191  interactions_.at(i1)->n_vertices();
192  });
193  }
194 
195  // Combine a set of existing interactions into one and append the vertex v.
196  // It is assumed that the interactions are combinable and the vertex
197  // appendable.
198  void combine_and_append(const std::vector<int> &S, const Vertex &v) {
199  unsigned N = S.size();
200  TKET_ASSERT(N > 0);
201  iptr &I = interactions_.at(S[0]);
202  for (unsigned i = 1; i < N; i++) {
203  I->combine(*interactions_.at(S[i]));
204  interactions_.erase(S[i]);
205  }
206  I->append(v);
207  }
208 
209  // Close an interaction, squashing it if possible, and erasing it from the
210  // set. Return true iff any substitution was made, and the (possibly new)
211  // vector of outgoing edges from the region of the interaction.
212  std::pair<bool, EdgeVec> close_interaction(int i) {
213  iptr &I = interactions_.at(i);
214  bool changed = false;
215  EdgeVec outs = I->out_edges();
216  switch (I->n_wires()) {
217  case 1:
218  break;
219  case 2:
220  case 3: {
221  Subcircuit sub = I->subcircuit();
222  Circuit subc = circ_.subcircuit(sub);
223  Circuit replacement = candidate_sub(subc, target_2qb_gate_);
224  if (replacement.count_gates(target_2qb_gate_) <
225  subc.count_gates(target_2qb_gate_)) {
226  // 1. Collect data needed later to reconstruct the list of out-edges:
227  std::vector<std::pair<Vertex, port_t>> out_vertex_ports;
228  for (const Edge &e : outs) {
229  out_vertex_ports.push_back(
230  {circ_.target(e), circ_.get_target_port(e)});
231  }
232  // 2. Do the substitution:
233  VertexSet old_vertices = I->vertices();
234  bin_.insert(bin_.end(), old_vertices.begin(), old_vertices.end());
235  circ_.substitute(replacement, sub, Circuit::VertexDeletion::No);
236  // 3. Construct the new list of out-edges:
237  EdgeVec new_outs;
238  for (const auto &[v, p] : out_vertex_ports) {
239  new_outs.push_back(circ_.get_nth_in_edge(v, p));
240  }
241  changed = true;
242  outs = std::move(new_outs);
243  }
244  break;
245  }
246  default:
247  TKET_ASSERT(!"Interaction with invalid number of wires");
248  }
249  interactions_.erase(i);
250  return {changed, outs};
251  }
252 
253  // Close an interaction and spawn new ones on its outgoing edges. Return true
254  // iff any substitution is made.
255  bool close_interaction_and_spawn(int i) {
256  auto [changed, outs] = close_interaction(i);
257  for (const Edge &e : outs) {
258  create_new_interaction_from_edge(e);
259  }
260  return changed;
261  }
262 
263  // Close all interactions that have v as a direct successor, and start new
264  // ones following them. Return true iff any substitution is made.
265  bool close_interactions_feeding_vertex(const Vertex &v) {
266  bool changed = false;
267  std::vector<int> v_interactions = interactions_feeding_vertex(v);
268 
269  for (int i : v_interactions) {
270  auto [change, outs] = close_interaction(i);
271  changed |= change;
272  for (const Edge &e : outs) {
273  if (circ_.target(e) != v) {
274  create_new_interaction_from_edge(e);
275  }
276  }
277  }
278 
279  for (const Edge &e : circ_.get_out_edges_of_type(v, EdgeType::Quantum)) {
280  create_new_interaction_from_edge(e);
281  }
282 
283  return changed;
284  }
285 
286  // Close all interactions. Return true iff any substitution is made.
287  bool close_all_interactions() {
288  bool changed = false;
289  // Form set of keys.
290  std::set<int> indices;
291  for (const auto &pair : interactions_) {
292  indices.insert(pair.first);
293  }
294  // Close each one.
295  for (int i : indices) {
296  auto [change, outs] = close_interaction(i);
297  changed |= change;
298  }
299  return changed;
300  }
301 
302  // Delete all vertices marked for deletion.
303  void destroy_bin() {
304  circ_.remove_vertices(
305  bin_, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
306  }
307 
308  private:
309  Circuit &circ_;
310  OpType target_2qb_gate_;
311  VertexList bin_;
312  std::map<int, iptr> interactions_;
313  int idx_;
314 };

316 Transform three_qubit_squash(OpType target_2qb_gate) {
317  return Transform([target_2qb_gate](Circuit &circ) {
318  bool changed = false;
319 
320  // Step through the vertices in topological order.
321  QISystem Is(circ, target_2qb_gate); // set of "live" interactions
322  for (const Vertex &v : circ.vertices_in_order()) {
323  const EdgeVec v_q_ins = circ.get_in_edges_of_type(v, EdgeType::Quantum);
324  const EdgeVec v_q_outs = circ.get_out_edges_of_type(v, EdgeType::Quantum);
325  unsigned n_q_ins = v_q_ins.size();
326  unsigned n_q_outs = v_q_outs.size();
327 
328  // If v has no quantum wires, ignore it and move on.
329  if (n_q_ins == 0 && n_q_outs == 0) continue;
330 
331  // If v is initial, create an interaction from its out-edge, and move on.
332  if (n_q_ins == 0) {
333  TKET_ASSERT(n_q_outs == 1);
334  Is.create_new_interaction_from_edge(v_q_outs[0]);
335  continue;
336  }
337 
338  // If v is final, ignore it and move on.
339  if (n_q_outs == 0) continue;
340 
341  // It's an internal operation with >0 quantum wires.
342  TKET_ASSERT(n_q_ins == n_q_outs);
343 
344  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
345  OpType optype = op->get_type();
346 
347  // If there are any incoming classical wires, or if this is a Barrier or
348  // Reset or Collapse operation, or if the operation contains symbols,
349  // close all existing interactions meeting v and create new ones, then
350  // move on.
351  if (!circ.get_in_edges_of_type(v, EdgeType::Classical).empty() ||
352  !circ.get_in_edges_of_type(v, EdgeType::Boolean).empty() ||
353  optype == OpType::Barrier || optype == OpType::Reset ||
354  optype == OpType::Collapse || !op->free_symbols().empty()) {
355  changed |= Is.close_interactions_feeding_vertex(v);
356  continue;
357  }
358 
359  // The circuit should contain only 1-qubit and CX gates.
360  if ((n_q_ins == 2 && optype != target_2qb_gate) || (n_q_ins > 2)) {
361  throw std::invalid_argument(
362  "Three-qubit squash requires circuits with 1-qubit and target "
363  "2-qubit gates only");
364  }
365 
366  // Absorb v into existing interactions, closing or merging as necessary.
367  bool done_with_v = false;
368  while (!done_with_v) {
369  std::vector<int> v_Is = Is.interactions_feeding_vertex(v);
370  unsigned total_n_wires = Is.total_n_wires(v_Is);
371  if (total_n_wires <= 3) {
372  Is.combine_and_append(v_Is, v);
373  done_with_v = true;
374  } else {
375  // Close one of the interactions meeting v.
376  int i = Is.largest_interaction(v_Is);
377  changed |= Is.close_interaction_and_spawn(i);
378  }
379  }
380  }
381 
382  // Close all remaining interactions.
383  changed |= Is.close_all_interactions();
384 
385  // Delete removed vertices.
386  Is.destroy_bin();
387 
388  return changed;
389  });
390 }
