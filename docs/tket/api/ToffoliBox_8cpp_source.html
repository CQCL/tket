29 ToffoliBox::ToffoliBox(
30  const state_perm_t &permutation, const OpType &rotation_axis)
31  : Box(OpType::ToffoliBox),
32  permutation_(permutation),
33  rotation_axis_(rotation_axis) {
34  // we need to check every element has the same size, and every bitstrings
35  // appears twice
36  if (permutation.size() == 0) {
37  throw std::invalid_argument(
38  "The permutation argument passed to ToffoliBox is empty.");
39  }
40  if (rotation_axis != OpType::Rx && rotation_axis != OpType::Ry) {
41  throw std::invalid_argument(
42  "The rotation_axis argument passed to ToffoliBox must be Rx or Ry.");
43  }
44  // verify the permutation is valid
45  auto it = permutation.begin();
46  if (it->first.size() > 32) {
47  throw std::invalid_argument(
48  "ToffoliBox only supports permutation up to 32 bits.");
49  }
50  unsigned n_qubits = (unsigned)it->first.size();
51  std::set<std::vector<bool>> rhs_states;
52  std::set<std::vector<bool>> lhs_states;
53  for (; it != permutation.end(); ++it) {
54  if (it->first.size() != n_qubits || it->second.size() != n_qubits) {
55  throw std::invalid_argument(
56  "The permutation argument passed to ToffoliBox contains bitstrings "
57  "with different sizes.");
58  }
59  lhs_states.insert(it->first);
60  rhs_states.insert(it->second);
61  }
62  if (lhs_states != rhs_states) {
63  throw std::invalid_argument(
64  "The permutation argument passed to ToffoliBox is not complete because "
65  "some states aren't mapped.");
66  }
67 }

98 static std::vector<std::vector<bool>> rearrange_along_col(
99  const std::vector<bool> &prefix, unsigned n_qubits, unsigned col_idx,
100  state_perm_t &perm) {
101  unsigned n_right_columns = n_qubits - prefix.size() - 1;
102  // Given a group of rows identified by a shared prefix and col_idx in {0,1}
103  // we rearrange these rows such that the bits(called postfix) after col_idx
104  // in their indices span the entire n_right_columns-bit space
105  // the rearrangement is done by swapping along col_idx
106 
107  // We only need to do the matching for the rows starting with prefix+0
108  // the solution will also take care of the rows starting with prefix+1
109 
110  // Construct the bipartite graph with 2*2^n_right_columns vertices
111  // that connects the postfix in the row entires and the postfix in the row
112  // indices. First half of the vertices represents the row entries the second
113  // half represents the row indices
114  cube_graph_t g(1ULL << (n_right_columns + 1));
115  for (unsigned long long postfix_dec = 0;
116  postfix_dec < (1ULL << n_right_columns); postfix_dec++) {
117  std::vector<bool> postfix = dec_to_bin(postfix_dec, n_right_columns);
118  // row0 = prefix+0+postfix
119  // row1 = prefix+1+postfix
120  std::vector<bool> row0(prefix);
121  std::vector<bool> row1(prefix);
122  row0.push_back(0);
123  row0.insert(row0.end(), postfix.begin(), postfix.end());
124  row1.push_back(1);
125  row1.insert(row1.end(), postfix.begin(), postfix.end());
126  // the postfix of row0's index
127  std::vector<bool> idx0_postfix(
128  perm[row0].begin() + prefix.size() + 1, perm[row0].end());
129  // the postfix of row1's index
130  std::vector<bool> idx1_postfix(
131  perm[row1].begin() + prefix.size() + 1, perm[row1].end());
132  // row0 can stay where it is
133  boost::add_edge(
134  postfix_dec, bin_to_dec(idx0_postfix) + (1ULL << n_right_columns), g);
135  if (idx0_postfix != idx1_postfix) {
136  // row0 can also move to row1's index by a swap
137  boost::add_edge(
138  postfix_dec, bin_to_dec(idx1_postfix) + (1ULL << n_right_columns), g);
139  }
140  }
141  // find a matching
142  std::vector<perm_vert_t> match(1ULL << (n_right_columns + 1));
143  // O(|V|*|E|*alpha(|V|, |E|)), alpha is a slow growing function
144  // that never exceeds 4 for any realistic input.
145  // TODO: use a specialised algo for bipartite matching
146  // e.g. Hopcroft-Karp for O(|E| sqrt(|V|))
147  edmonds_maximum_cardinality_matching(g, &match[0]);
148  std::vector<std::vector<bool>> swap_pairs;
149  for (unsigned long long postfix_dec = 0;
150  postfix_dec < (1ULL << n_right_columns); postfix_dec++) {
151  std::vector<bool> postfix = dec_to_bin(postfix_dec, n_right_columns);
152  std::vector<bool> row0(prefix);
153  row0.push_back(0);
154  row0.insert(row0.end(), postfix.begin(), postfix.end());
155  std::vector<bool> mapped_idx = dec_to_bin(
156  match[postfix_dec] - (1ULL << n_right_columns), n_right_columns);
157  if (!std::equal(
158  perm[row0].begin() + prefix.size() + 1, perm[row0].end(),
159  mapped_idx.begin())) {
160  // row0 matched to row1's index
161  row0.erase(row0.begin() + col_idx);
162  swap_pairs.push_back(row0);
163  }
164  }
165  return swap_pairs;
166 }

168 static void swap_rows(
169  const std::vector<bool> &pair, unsigned col_idx, state_perm_t &perm,
170  ctrl_op_map_t &op_map, std::vector<Complex> &phases,
171  const Op_ptr &zflip_op) {
172  // swap a pair of rows along col_idx
173  // assume two rows only differ at q
174  // update op_map, phases, and permutation
175  std::vector<bool> row0(pair);
176  std::vector<bool> row1(pair);
177  row0.insert(row0.begin() + col_idx, 0);
178  row1.insert(row1.begin() + col_idx, 1);
179  std::swap(perm.at(row0), perm.at(row1));
180  // phases is a vector, convert rows to decimal to track phases
181  unsigned row0_dec = bin_to_dec(row0);
182  unsigned row1_dec = bin_to_dec(row1);
183  // we currently only support Rx(pi) and Ry(pi) for permuting states
184  // we might introduce Rx(-pi) or Ry(-pi) in the future. Also, dynamically
185  // choosing between these 4 might give possible phase cancellation
186  // opportunities
187  if (zflip_op->get_type() == OpType::Rx &&
188  equiv_val(zflip_op->get_params()[0], 1., 4)) {
189  // check the Rx angle is (1 mod 4)
190  phases[row0_dec] *= -i_;
191  phases[row1_dec] *= -i_;
192  } else if (
193  zflip_op->get_type() == OpType::Ry &&
194  equiv_val(zflip_op->get_params()[0], 1., 4)) {
195  // check the Ry angle is (1 mod 4)
196  phases[row1_dec] *= -1;
197  } else {
198  // shouldn't happen since we assume the zflip_op must satisfy one of the
199  // above conditions. Still throw an error in case something goes wrong.
200  throw std::logic_error(
201  "Attempt to perform state permutation in ToffoliBox with unsupported "
202  "rotations.");
203  }
204  std::swap(phases[row0_dec], phases[row1_dec]);
205  op_map.insert({pair, zflip_op});
206 }

208 static std::vector<unsigned> get_multiplexor_args(
209  unsigned n_qubits, unsigned col_idx) {
210  std::vector<unsigned> multplx_args(n_qubits);
211  std::iota(std::begin(multplx_args), std::end(multplx_args), 0);
212  multplx_args.erase(multplx_args.begin() + col_idx);
213  multplx_args.push_back(col_idx);
214  return multplx_args;
215 }

226 static Circuit permute(
227  state_perm_t &perm, unsigned n_qubits, const Op_ptr &zflip_op) {
228  // Consider the permutation map as a boolean matrix with n columns and 2^n
229  // rows. Row i contains current location of the coefficient that needs to be
230  // permutated to the state |i>. We want to sort the rows such that the value
231  // of each row matches the its index.
232  //
233  // The algorithm has two steps:
234  // Step 1. We traverse columns to the right, finish at (n-2)th column.
235  // At jth column, we first group the rows by their first j+1 entries,
236  // so each group is identified by a bitstring [b_0,b_1,...,b_j]. We then
237  // rearrange the rows such that for each group, the last n-j-1 bits of
238  // its row indices span the entire (n-j-1)-bit space. Such rearrangement
239  // can be accomplished with one multiplexor targeting q[j].
240  //
241  // Step 2. We traverse all columns from right to left.
242  // At jth column j, for any row R whose jth bit doesn't match the jth bit of
243  // its row index, swap it with row R' where R and R' only differ at the jth
244  // bit. All the swaps can be done with one multiplexor targeting q[j].
245 
246  // In the implementation, we use a map from rows to rows indices to represent
247  // the matrix defined in the algorithm.
248 
249  Circuit circ(n_qubits);
250  // special case
251  if (n_qubits == 1) {
252  if (perm.begin()->first != perm.begin()->second) {
253  circ.add_op<unsigned>(OpType::X, {0});
254  }
255  return circ;
256  }
257  // track relative phases caused by using pi rotations
258  std::vector<Complex> phases(1ULL << n_qubits, 1.);
259  // step 1
260  for (unsigned col_idx = 0; col_idx < n_qubits - 1; col_idx++) {
261  ctrl_op_map_t op_map;
262  for (unsigned long long prefix = 0; prefix < (1ULL << col_idx); prefix++) {
263  std::vector<bool> prefix_bin;
264  if (col_idx != 0) {
265  // if col_idx == 0, the prefix is empty, otherwise convert the decimal
266  // prefix to its binary representation
267  prefix_bin = dec_to_bin(prefix, col_idx);
268  }
269  std::vector<std::vector<bool>> swap_pairs =
270  rearrange_along_col(prefix_bin, n_qubits, col_idx, perm);
271  for (const std::vector<bool> &pair : swap_pairs) {
272  // swap_rows mutates the perm (current permutation), phases (phases
273  // accumulated by using SU2 gates) and also updates the op_map to
274  // indicate which pairs of rows to swap
275  swap_rows(pair, col_idx, perm, op_map, phases, zflip_op);
276  }
277  }
278  if (!op_map.empty()) {
279  std::vector<unsigned> multplx_args =
280  get_multiplexor_args(n_qubits, col_idx);
281  circ.add_box(MultiplexedRotationBox(op_map), multplx_args);
282  }
283  }
284  // step 2
285  for (unsigned col_idx = n_qubits; col_idx-- > 0;) {
286  ctrl_op_map_t op_map;
287  for (unsigned long long row = 0; row < (1ULL << (n_qubits - 1)); row++) {
288  std::vector<bool> pair = dec_to_bin(row, n_qubits - 1);
289  pair.insert(pair.begin() + col_idx, 0);
290  if (perm[pair][col_idx] != 0) {
291  pair.erase(pair.begin() + col_idx);
292  swap_rows(pair, col_idx, perm, op_map, phases, zflip_op);
293  }
294  }
295  if (!op_map.empty()) {
296  std::vector<unsigned> multplx_args =
297  get_multiplexor_args(n_qubits, col_idx);
298  circ.add_box(MultiplexedRotationBox(op_map), multplx_args);
299  }
300  }
301  // correct the phases with a diagonal operator
302  Eigen::VectorXcd corrections(1ULL << n_qubits);
303  for (unsigned i = 0; i < phases.size(); i++) {
304  corrections[i] = 1. / phases[i];
305  }
306  DiagonalBox diag(corrections);
307  circ.add_box(diag, circ.all_qubits());
308  return circ;
309 }

311 void ToffoliBox::generate_circuit() const {
312  unsigned n_qubits = (unsigned)permutation_.begin()->first.size();
313  state_perm_t perm(permutation_);
314  // fill the permutation with identities
315  if (perm.size() != (1ULL << n_qubits)) {
316  for (unsigned long long i = 0; i < (1ULL << n_qubits); i++) {
317  std::vector<bool> state = dec_to_bin(i, n_qubits);
318  if (perm.find(state) == perm.end()) {
319  perm.insert({state, state});
320  }
321  }
322  }
323  circ_ = std::make_shared<Circuit>(
324  permute(perm, n_qubits, get_op_ptr(rotation_axis_, 1)));
325 }
