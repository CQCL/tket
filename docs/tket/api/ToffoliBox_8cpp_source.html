28static unsigned get_perm_size(const state_perm_t &permutation) {
29 if (permutation.size() == 0) {
30 throw std::invalid_argument(
31 "The permutation argument passed to ToffoliBox is empty.");
32 }
33 auto it = permutation.begin();
34 unsigned n = it->first.size();
35 // verify the permutation is valid
36 if (n > 32) {
37 throw std::invalid_argument(
38 "ToffoliBox only supports permutation up to 32 bits.");
39 }
40 // we need to check every element has the same size, and every bitstrings
41 // appears twice
42 std::set<std::vector<bool>> rhs_states;
43 std::set<std::vector<bool>> lhs_states;
44 for (; it != permutation.end(); ++it) {
45 if (it->first.size() != n || it->second.size() != n) {
46 throw std::invalid_argument(
47 "The permutation argument passed to ToffoliBox contains bitstrings "
48 "with different sizes.");
49 }
50 lhs_states.insert(it->first);
51 rhs_states.insert(it->second);
52 }
53 if (lhs_states != rhs_states) {
54 throw std::invalid_argument(
55 "The permutation argument passed to ToffoliBox is not complete because "
56 "some states aren't mapped.");
57 }
58 return n;
59}

63 const OpType &rotation_axis)

84std::optional<Eigen::MatrixXcd> ToffoliBox::get_box_unitary() const {
85 Eigen::MatrixXcd U = Eigen::MatrixXcd::Identity(pow2n_, pow2n_);
86 for (const auto &pair : permutation_) {
87 const unsigned i = bin_to_dec(pair.second);
88 const unsigned j = bin_to_dec(pair.first);
89 U(i, i) = 0.;
90 U(i, j) = 1.;
91 }
92 return U;
93}

120static std::vector<std::vector<bool>> rearrange_along_col(
121 const std::vector<bool> &prefix, unsigned n_qubits, unsigned col_idx,
122 state_perm_t &perm) {
123 unsigned n_right_columns = n_qubits - prefix.size() - 1;
124 // Given a group of rows identified by a shared prefix and col_idx in {0,1}
125 // we rearrange these rows such that the bits(called postfix) after col_idx
126 // in their indices span the entire n_right_columns-bit space
127 // the rearrangement is done by swapping along col_idx
128
129 // We only need to do the matching for the rows starting with prefix+0
130 // the solution will also take care of the rows starting with prefix+1
131
132 // Construct the bipartite graph with 2*2^n_right_columns vertices
133 // that connects the postfix in the row entires and the postfix in the row
134 // indices. First half of the vertices represents the row entries the second
135 // half represents the row indices
136 cube_graph_t g(1ULL << (n_right_columns + 1));
137 for (unsigned long long postfix_dec = 0;
138 postfix_dec < (1ULL << n_right_columns); postfix_dec++) {
139 std::vector<bool> postfix = dec_to_bin(postfix_dec, n_right_columns);
140 // row0 = prefix+0+postfix
141 // row1 = prefix+1+postfix
142 std::vector<bool> row0(prefix);
143 std::vector<bool> row1(prefix);
144 row0.push_back(0);
145 row0.insert(row0.end(), postfix.begin(), postfix.end());
146 row1.push_back(1);
147 row1.insert(row1.end(), postfix.begin(), postfix.end());
148 // the postfix of row0's index
149 std::vector<bool> idx0_postfix(
150 perm[row0].begin() + prefix.size() + 1, perm[row0].end());
151 // the postfix of row1's index
152 std::vector<bool> idx1_postfix(
153 perm[row1].begin() + prefix.size() + 1, perm[row1].end());
154 // row0 can stay where it is
155 boost::add_edge(
156 postfix_dec, bin_to_dec(idx0_postfix) + (1ULL << n_right_columns), g);
157 if (idx0_postfix != idx1_postfix) {
158 // row0 can also move to row1's index by a swap
159 boost::add_edge(
160 postfix_dec, bin_to_dec(idx1_postfix) + (1ULL << n_right_columns), g);
161 }
162 }
163 // find a matching
164 std::vector<perm_vert_t> match(1ULL << (n_right_columns + 1));
165 // O(|V|*|E|*alpha(|V|, |E|)), alpha is a slow growing function
166 // that never exceeds 4 for any realistic input.
167 // TODO: use a specialised algo for bipartite matching
168 // e.g. Hopcroft-Karp for O(|E| sqrt(|V|))
169 edmonds_maximum_cardinality_matching(g, &match[0]);
170 std::vector<std::vector<bool>> swap_pairs;
171 for (unsigned long long postfix_dec = 0;
172 postfix_dec < (1ULL << n_right_columns); postfix_dec++) {
173 std::vector<bool> postfix = dec_to_bin(postfix_dec, n_right_columns);
174 std::vector<bool> row0(prefix);
175 row0.push_back(0);
176 row0.insert(row0.end(), postfix.begin(), postfix.end());
177 std::vector<bool> mapped_idx = dec_to_bin(
178 match[postfix_dec] - (1ULL << n_right_columns), n_right_columns);
179 if (!std::equal(
180 perm[row0].begin() + prefix.size() + 1, perm[row0].end(),
181 mapped_idx.begin())) {
182 // row0 matched to row1's index
183 row0.erase(row0.begin() + col_idx);
184 swap_pairs.push_back(row0);
185 }
186 }
187 return swap_pairs;
188}

190static void swap_rows(
191 const std::vector<bool> &pair, unsigned col_idx, state_perm_t &perm,
192 ctrl_op_map_t &op_map, std::vector<Complex> &phases,
193 const Op_ptr &zflip_op) {
194 // swap a pair of rows along col_idx
195 // assume two rows only differ at q
196 // update op_map, phases, and permutation
197 std::vector<bool> row0(pair);
198 std::vector<bool> row1(pair);
199 row0.insert(row0.begin() + col_idx, 0);
200 row1.insert(row1.begin() + col_idx, 1);
201 std::swap(perm.at(row0), perm.at(row1));
202 // phases is a vector, convert rows to decimal to track phases
203 unsigned row0_dec = bin_to_dec(row0);
204 unsigned row1_dec = bin_to_dec(row1);
205 // we currently only support Rx(pi) and Ry(pi) for permuting states
206 // we might introduce Rx(-pi) or Ry(-pi) in the future. Also, dynamically
207 // choosing between these 4 might give possible phase cancellation
208 // opportunities
209 if (zflip_op->get_type() == OpType::Rx &&
210 equiv_val(zflip_op->get_params()[0], 1., 4)) {
211 // check the Rx angle is (1 mod 4)
212 phases[row0_dec] *= -i_;
213 phases[row1_dec] *= -i_;
214 } else if (
215 zflip_op->get_type() == OpType::Ry &&
216 equiv_val(zflip_op->get_params()[0], 1., 4)) {
217 // check the Ry angle is (1 mod 4)
218 phases[row1_dec] *= -1;
219 } else {
220 // shouldn't happen since we assume the zflip_op must satisfy one of the
221 // above conditions. Still throw an error in case something goes wrong.
222 throw std::logic_error(
223 "Attempt to perform state permutation in ToffoliBox with unsupported "
224 "rotations.");
225 }
226 std::swap(phases[row0_dec], phases[row1_dec]);
227 op_map.insert({pair, zflip_op});
228}

230static std::vector<unsigned> get_multiplexor_args(
231 unsigned n_qubits, unsigned col_idx) {
232 std::vector<unsigned> multplx_args(n_qubits);
233 std::iota(std::begin(multplx_args), std::end(multplx_args), 0);
234 multplx_args.erase(multplx_args.begin() + col_idx);
235 multplx_args.push_back(col_idx);
236 return multplx_args;
237}

248static Circuit permute(
249 state_perm_t &perm, unsigned n_qubits, const Op_ptr &zflip_op) {
250 // Consider the permutation map as a boolean matrix with n columns and 2^n
251 // rows. Row i contains current location of the coefficient that needs to be
252 // permutated to the state |i>. We want to sort the rows such that the value
253 // of each row matches the its index.
254 //
255 // The algorithm has two steps:
256 // Step 1. We traverse columns to the right, finish at (n-2)th column.
257 // At jth column, we first group the rows by their first j+1 entries,
258 // so each group is identified by a bitstring [b_0,b_1,...,b_j]. We then
259 // rearrange the rows such that for each group, the last n-j-1 bits of
260 // its row indices span the entire (n-j-1)-bit space. Such rearrangement
261 // can be accomplished with one multiplexor targeting q[j].
262 //
263 // Step 2. We traverse all columns from right to left.
264 // At jth column j, for any row R whose jth bit doesn't match the jth bit of
265 // its row index, swap it with row R' where R and R' only differ at the jth
266 // bit. All the swaps can be done with one multiplexor targeting q[j].
267
268 // In the implementation, we use a map from rows to rows indices to represent
269 // the matrix defined in the algorithm.
270
271 Circuit circ(n_qubits);
272 // special case
273 if (n_qubits == 1) {
274 if (perm.begin()->first != perm.begin()->second) {
275 circ.add_op<unsigned>(OpType::X, {0});
276 }
277 return circ;
278 }
279 // track relative phases caused by using pi rotations
280 std::vector<Complex> phases(1ULL << n_qubits, 1.);
281 // step 1
282 for (unsigned col_idx = 0; col_idx < n_qubits - 1; col_idx++) {
283 ctrl_op_map_t op_map;
284 for (unsigned long long prefix = 0; prefix < (1ULL << col_idx); prefix++) {
285 std::vector<bool> prefix_bin;
286 if (col_idx != 0) {
287 // if col_idx == 0, the prefix is empty, otherwise convert the decimal
288 // prefix to its binary representation
289 prefix_bin = dec_to_bin(prefix, col_idx);
290 }
291 std::vector<std::vector<bool>> swap_pairs =
292 rearrange_along_col(prefix_bin, n_qubits, col_idx, perm);
293 for (const std::vector<bool> &pair : swap_pairs) {
294 // swap_rows mutates the perm (current permutation), phases (phases
295 // accumulated by using SU2 gates) and also updates the op_map to
296 // indicate which pairs of rows to swap
297 swap_rows(pair, col_idx, perm, op_map, phases, zflip_op);
298 }
299 }
300 if (!op_map.empty()) {
301 std::vector<unsigned> multplx_args =
302 get_multiplexor_args(n_qubits, col_idx);
303 circ.add_box(MultiplexedRotationBox(op_map), multplx_args);
304 }
305 }
306 // step 2
307 for (unsigned col_idx = n_qubits; col_idx-- > 0;) {
308 ctrl_op_map_t op_map;
309 for (unsigned long long row = 0; row < (1ULL << (n_qubits - 1)); row++) {
310 std::vector<bool> pair = dec_to_bin(row, n_qubits - 1);
311 pair.insert(pair.begin() + col_idx, 0);
312 if (perm[pair][col_idx] != 0) {
313 pair.erase(pair.begin() + col_idx);
314 swap_rows(pair, col_idx, perm, op_map, phases, zflip_op);
315 }
316 }
317 if (!op_map.empty()) {
318 std::vector<unsigned> multplx_args =
319 get_multiplexor_args(n_qubits, col_idx);
320 circ.add_box(MultiplexedRotationBox(op_map), multplx_args);
321 }
322 }
323 // correct the phases with a diagonal operator
324 Eigen::VectorXcd corrections(1ULL << n_qubits);
325 for (unsigned i = 0; i < phases.size(); i++) {
326 corrections[i] = 1. / phases[i];
327 }
328 DiagonalBox diag(corrections);
329 circ.add_box(diag, circ.all_qubits());
330 return circ;
331}

336void ToffoliBox::generate_circuit() const {
337 state_perm_t perm(permutation_);
338 if (this->strat_ == ToffoliBoxSynthStrat::Cycle) {
339 circ_ =
340 std::make_shared<Circuit>(gen_circuit_using_toffoli_gates(perm, n_));
341 return;
342 }
343 // fill the permutation with identities
344 if (perm.size() != pow2n_) {
345 for (unsigned long long i = 0; i < pow2n_; i++) {
346 std::vector<bool> state = dec_to_bin(i, n_);
347 if (perm.find(state) == perm.end()) {
348 perm.insert({state, state});
349 }
350 }
351 }
352 circ_ = std::make_shared<Circuit>(
353 permute(perm, n_, get_op_ptr(rotation_axis_, 1)));
354}

356// check every key k in a,
357// either 1. a[k] == b[k] or 2. k is not in b and a[k]==k
358static bool oneway_perm_compare(const state_perm_t &a, const state_perm_t &b) {
359 for (auto ita = a.begin(); ita != a.end(); ita++) {
360 auto b_entry = b.find(ita->first);
361 if (b_entry == b.end() && ita->first != ita->second) {
362 return false;
363 }
364 if (b_entry != b.end() && b_entry->second != ita->second) {
365 return false;
366 }
367 }
368 return true;
369}
370static bool perm_compare(const state_perm_t &a, const state_perm_t &b) {
371 return oneway_perm_compare(a, b) && oneway_perm_compare(b, a);
372}

374bool ToffoliBox::is_equal(const Op &op_other) const {
375 const ToffoliBox &other = dynamic_cast<const ToffoliBox &>(op_other);
376 if (id_ == other.get_id()) return true;
377 return strat_ == other.strat_ && rotation_axis_ == other.rotation_axis_ &&
378 perm_compare(permutation_, other.permutation_);
379}

392static unsigned get_hamming_distance(
393 const std::vector<bool> &a, const std::vector<bool> &b) {
394 if (a.size() != b.size()) {
395 throw std::invalid_argument("Bitstrings must have identical size.");
396 }
397 unsigned counter = 0;
398 for (unsigned i = 0; i < a.size(); i++) {
399 if (a[i] != b[i]) {
400 ++counter;
401 }
402 }
403 return counter;
404}

406static cycle_transposition_t cycle_to_transposition(cycle_permutation_t cycle) {
419 cycle_transposition_t best_transposition;
420 unsigned best_hamming_distance = 0;
421
422 for (unsigned i = 0; i < cycle.size(); i++) {
423 unsigned accumulated_hamming_distance = 0;
424 cycle_transposition_t transposition;
425 for (unsigned j = 1; j < cycle.size(); j++) {
426 transposition.push_back({cycle[0], cycle[j], cycle[0]});
427 accumulated_hamming_distance += get_hamming_distance(cycle[0], cycle[j]);
428 }
429 if (best_transposition.empty() ||
430 accumulated_hamming_distance < best_hamming_distance) {
431 best_transposition = transposition;
432 best_hamming_distance = accumulated_hamming_distance;
433 }
434 std::rotate(cycle.begin(), cycle.begin() + 1, cycle.end());
435 }
436 return best_transposition;
437}

439static std::vector<cycle_transposition_t> get_transpositions(
440 const std::set<cycle_permutation_t> &cycles) {
441 std::vector<cycle_transposition_t> transpositions;
442 for (const cycle_permutation_t &cycle : cycles) {
443 // each cycle is costed via the Hamming distance to reduce the number of
444 // operations
445 transpositions.push_back(cycle_to_transposition(cycle));
446 }
447 return transpositions;
448}

450static cycle_transposition_t merge_cycles(
451 std::vector<cycle_transposition_t> &cycle_transpositions) {
452 cycle_transposition_t return_transposition;
453 for (unsigned k = 0; k < cycle_transpositions.size(); k++) {
454 cycle_transposition_t cycle = cycle_transpositions[k];
455 unsigned i = 0, j = 1;
456 while (j < cycle.size()) {
457 transposition_t transposition_i = cycle[i];
458 transposition_t transposition_j = cycle[j];
459
460 std::vector<bool> transposition_j_first = transposition_j.first;
461 std::vector<bool> transposition_i_last = transposition_i.last;
462 std::vector<bool> transposition_i_first = transposition_i.first;
463
464 TKET_ASSERT(transposition_i_last == transposition_j.first);
465 std::vector<bool> i_middle = transposition_i.middle;
466 std::vector<bool> j_middle = transposition_j.middle;
467
468 TKET_ASSERT(i_middle.size() == transposition_i.last.size());
469 TKET_ASSERT(j_middle.size() == transposition_i.last.size());
470 // if a transposition has already been reduced, still need to make sure we
471 // uncompute it
472 if (transposition_i_first != transposition_i_last) {
473 unsigned middle_last_distance =
474 get_hamming_distance(i_middle, transposition_i_last);
475 unsigned middle_first_distance =
476 get_hamming_distance(i_middle, transposition_i_first);
477 // this => the reduced transposition is on a good gray code between the
478 // new "first" and target
479 if (middle_first_distance < middle_last_distance &&
480 middle_first_distance > 1) {
481 transposition_i_last = transposition_i_first;
482 std::vector<bool> starting_point = transposition_i_last;
483 for (unsigned k = 0; k < i_middle.size(); k++) {
484 if (i_middle[k] == j_middle[k] &&
485 get_hamming_distance(starting_point, i_middle) > 1) {
486 starting_point[k] = i_middle[k];
487 }
488 }
489
490 cycle_transpositions[k][i].last = starting_point;
491 cycle_transpositions[k][j].first = starting_point;
492 }
493 } else { // else in this case just find any good transposition
494 std::vector<bool> starting_point = transposition_i_last;
495 for (unsigned k = 0; k < i_middle.size(); k++) {
496 if (i_middle[k] == j_middle[k] &&
497 get_hamming_distance(starting_point, i_middle) > 1) {
498 starting_point[k] = i_middle[k];
499 }
500 }
501
502 cycle[i].last = starting_point;
503 cycle[j].first = starting_point;
504 }
505
506 ++i;
507 ++j;
508 }
509
510 return_transposition.insert(
511 return_transposition.end(), cycle.begin(), cycle.end());
512 }
513
514 return return_transposition;
515}

517static gray_code_t transposition_to_gray_code(
518 const transposition_t &transposition) {
519 unsigned first_middle_hamming_distance =
520 get_hamming_distance(transposition.first, transposition.middle);
521 unsigned middle_last_hamming_distance =
522 get_hamming_distance(transposition.middle, transposition.last);
523 gray_code_t all_gray_code_entries;
524 // => that some optimisation is done to middle_last, so must go via
525 // transposition.last bitstring to allow proper cancellation
526 // At some point transposition.first == transposition.last
527 // If transposition.last != transposition.first & the hamming distance
528 // between transposition.last and transposition.middle is smaller than
529 // transposition.first and transposition.middle,
530 // then we need to make sure that the gray code between transposition.first
531 // and transposition.middle goes via transposition.last, such that the
532 // eventual gray code between transposition.middle and transposition.last
533 // uncomputes this we can assume that the remaining
534 // transposition.last->transposition.first gray code would be cancelled out if
535 // added
536 std::vector<bool> initial = transposition.first;
537 if (middle_last_hamming_distance < first_middle_hamming_distance) {
538 // get bitstrings for first -> last
539 for (unsigned i = 0; i < transposition.first.size(); i++) {
540 if (transposition.first[i] != transposition.last[i]) {
541 initial[i] = !initial[i];
542 all_gray_code_entries.push_back({initial, i});
543 }
544 }
545 }
546 // with the right middle bitstring now guaranteed, go from this bitstring to
547 // middle
548 std::vector<bool> bitstring = initial;
549 for (unsigned i = 0; i < transposition.first.size(); i++) {
550 if (initial[i] != transposition.middle[i]) {
551 bitstring[i] = !bitstring[i];
552 all_gray_code_entries.push_back({bitstring, i});
553 }
554 }
555 // now do the last->middle in reverse to guarantee right
556 // gray code path is taken
557 initial = transposition.last;
558 gray_code_t reverse_gray_code_entries;
559 // as before, implies some optimisation has been completed on
560 // first one
561 // thus make sure that gray code goes via transposition.first
562 // to make applied permutation right
563 if (first_middle_hamming_distance < middle_last_hamming_distance) {
564 // get bitstrings for first -> last
565 for (unsigned i = 0; i < transposition.first.size(); i++) {
566 if (initial[i] != transposition.first[i]) {
567 initial[i] = !initial[i];
568 reverse_gray_code_entries.push_back({initial, i});
569 }
570 }
571 }
572 // and then go from bitstring to middle
573 bitstring = initial;
574 for (unsigned i = 0; i < transposition.middle.size(); i++) {
575 if (transposition.middle[i] != initial[i]) {
576 bitstring[i] = !bitstring[i];
577 reverse_gray_code_entries.push_back({bitstring, i});
578 }
579 }
580 // don't want to add transformation for reaching final -> so pop_back
581 if (!reverse_gray_code_entries.empty()) {
582 reverse_gray_code_entries.pop_back();
583 }
584 all_gray_code_entries.insert(
585 all_gray_code_entries.end(), reverse_gray_code_entries.rbegin(),
586 reverse_gray_code_entries.rend());
587 return all_gray_code_entries;
588}

590Circuit get_bitstring_circuit(
591 const std::vector<bool> &bitstring, const unsigned &target,
592 const unsigned &n_qubits) {
593 // flip qubits that need to be state 0
594 Circuit x_circuit(n_qubits);
595 std::vector<unsigned> cnx_args;
596 for (unsigned i = 0; i < n_qubits; i++) {
597 if (i != target) {
598 if (!bitstring[i]) {
599 x_circuit.add_op<unsigned>(OpType::X, {i});
600 }
601 cnx_args.push_back(i);
602 }
603 }
604 cnx_args.push_back(target);
605 TKET_ASSERT(cnx_args.size() == n_qubits);
606
607 Circuit return_circuit(n_qubits);
608 return_circuit.append(x_circuit);
609 return_circuit.add_op<unsigned>(OpType::CnX, cnx_args);
610 return_circuit.append(x_circuit);
611 return return_circuit;
612}

614static Circuit gen_circuit_using_toffoli_gates(
615 state_perm_t &perm, unsigned n_qubits) {
616 // Convert passed permutation to cycles
617 std::set<cycle_permutation_t> cycles;
618 while (!perm.empty()) {
619 auto it = perm.begin();
620 cycle_permutation_t cycle = {it->first};
621 it = perm.find(it->second);
622 while (it->first != cycle[0]) {
623 cycle.push_back(it->first);
624 it = perm.find(it->second);
625 }
626 if (cycle.size() > 1) {
627 cycles.insert(cycle);
628 }
629 for (const std::vector<bool> &bitstring : cycle) {
630 perm.erase(bitstring);
631 }
632 }
633
634 // This decomposition is as described on page 191, section 4.5.2 "Single
635 // qubit and CNOT gates are universal" of Nielsen & Chuang
636 std::vector<cycle_transposition_t> cycle_transpositions =
637 get_transpositions(cycles);
638
639 // optionally, order the transpositions and cycles to allow gate
640 // cancellation
641 cycle_transposition_t ordered_transpositions;
642 ordered_transpositions = merge_cycles(cycle_transpositions);
643
644 if (ordered_transpositions.empty()) {
645 return Circuit(n_qubits);
646 }
647
648 // Now we have ordered transpositions, produced front->middle and
649 // middle->back gray codes for each transposition and add to circuit
650 Circuit circ(n_qubits);
651 for (const transposition_t &transposition : ordered_transpositions) {
652 TKET_ASSERT(transposition.first.size() == n_qubits);
653 TKET_ASSERT(transposition.middle.size() == n_qubits);
654 TKET_ASSERT(transposition.last.size() == n_qubits);
655 gray_code_t all_gray_code_entries =
656 transposition_to_gray_code(transposition);
657 for (const std::pair<std::vector<bool>, unsigned> &entry :
658 all_gray_code_entries) {
659 circ.append(get_bitstring_circuit(entry.first, entry.second, n_qubits));
660 }
661 }
662 return circ;
663}
