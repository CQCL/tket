36/* Helpers `run_bfs` and `run_dfs` return instances of TreeSearchBase
37 * subclasses. These provide a host of useful member functions to obtain the
38 * relevant results from the tree search.
39 *
40 * Note that run() must have been called before
41 * using the member functions to retrieve results.
42 * This is done automatically by the utility functions in `TreeSearch.hpp`
43 *
44 * First template argument is Graph type, second is the boost property map type
45 * that should be used as a vertex index.
46 */
47template <typename Graph, typename BoostPMap>
48class TreeSearchBase {
49 protected:
50 using vertex = utils::vertex<Graph>;
51 using parent_vec = std::vector<vertex>;
52 using color_vec = std::vector<boost::default_color_type>;
53 using index_pmap_t = BoostPMap;
54 using dist_pmap_t =
55 boost::iterator_property_map<typename dist_vec::iterator, index_pmap_t>;
56 using parent_pmap_t =
57 boost::iterator_property_map<typename parent_vec::iterator, index_pmap_t>;
58 using color_pmap_t =
59 boost::iterator_property_map<typename color_vec::iterator, index_pmap_t>;
60 using visitor_t = std::pair<
61 boost::distance_recorder<dist_pmap_t, boost::on_tree_edge>,
62 boost::predecessor_recorder<parent_pmap_t, boost::on_tree_edge>>;
63
64 public:
65 /* Constructor: search through Graph g with root v, using index map pmap */
66 TreeSearchBase(vertex v, Graph&& g, BoostPMap pmap)
67 : root(v),
68 to_index(pmap),
69 graph(g),
70 dists(boost::num_vertices(g)),
71 parents(boost::num_vertices(g)),
72 color_map(boost::num_vertices(g)),
73 visitor{
74 boost::record_distances(
75 boost::make_iterator_property_map(dists.begin(), to_index),
76 boost::on_tree_edge()),
77 boost::record_predecessors(
78 boost::make_iterator_property_map(parents.begin(), to_index),
79 boost::on_tree_edge())} {
80 // for every vertex, assign self as parent
81 for (auto v : boost::make_iterator_range(boost::vertices(graph))) {
82 parents[to_index[v]] = v;
83 }
84 }
85
86 /* Get results from tree search */
87 /* vector of vertex predecessor in search */
88 const parent_vec& get_parents() const& { return parents; }
89 const parent_vec&& get_parents() const&& { return std::move(parents); }
90 /* vector of distances from the root */
91 const dist_vec& get_dists() const& { return dists; }
92 const dist_vec&& get_dists() const&& { return std::move(dists); }
93 std::size_t get_dist(const vertex& v) const { return dists[to_index[v]]; }
94
95 /* Rebase tree search from some new root */
96 void change_root(vertex v) {
97 // naive implementation
98 // This might be made more efficient in the subclasses
99 vertex old_root = root;
100 root = v;
101 if (root != old_root) {
102 std::fill(dists.begin(), dists.end(), 0);
103 std::iota(parents.begin(), parents.end(), 0);
104 run();
105 }
106 }
107
108 /* Vector from target to root (following reverse edges) */
109 parent_vec path_to_root(vertex target) const {
110 std::vector<vertex> path;
111
112 vertex current = target;
113 path.push_back(target);
114 while (current != root) {
115 vertex parent = parents[to_index[current]];
116 if (parent == current) {
117 // graph is disconnected and there is no path to root
118 return {};
119 }
120 current = parent;
121 path.push_back(current);
122 }
123 return path;
124 }
125
126 /* Vector from root to target along graph edges */
127 parent_vec path_from_root(vertex target) const {
128 parent_vec out = path_to_root(target);
129 std::reverse(out.begin(), out.end());
130 return out;
131 }
132
133 /* Depth of tree search */
134 std::size_t max_depth() const {
135 auto it = std::max_element(dists.cbegin(), dists.cend());
136 if (it == dists.cend()) {
137 throw std::invalid_argument(
138 "TreeSearch::max_depth: There is no entry in distance "
139 "vector");
140 }
141 return *it;
142 }
143
144 /* Vertex of max depth
145 * if more than one exist, returns first found */
146 vertex max_depth_vertex() const {
147 const std::size_t target_depth = max_depth();
148 for (vertex v : boost::make_iterator_range(boost::vertices(this->graph))) {
149 if (get_dist(v) == target_depth) {
150 return v;
151 }
152 }
153 throw std::runtime_error("max_depth_vertex: No vertex has maximal depth");
154 }
155
156 /* Path from root to deepest vertex */
157 parent_vec longest_path() const { return path_from_root(max_depth_vertex()); }
158
159 /* Run tree search */
160 virtual void run() = 0;
161
162 protected:
163 template <typename Visitor>
164 auto get_default_named_params(Visitor& vis) {
165 return boost::visitor(vis).color_map(
166 boost::make_iterator_property_map(color_map.begin(), to_index));
167 }
168
169 vertex root; // root of tree search
170 index_pmap_t to_index; // vertex index pmap
171 Graph graph; // graph to be searched
172 dist_vec dists; // distance from root
173 parent_vec parents; // parent towards root
174 color_vec color_map; // color used by tree search algorithm
175 visitor_t visitor; // visitor passed to boost tree search alg
176};

178/* A vertex index map, mapping boost vertex descriptors to unsigned
179 * is required for tree searches (to index the parents and dists vectors).
180 *
181 * The TreeSearch class accepts a std::map-like PMap argument to define such a
182 * mapping. This is converted to a boost property map using
183 * boost::associative_property_map. If none is provided, the default
184 * boost::vertex_index is used (by default, this is available on boost graphs
185 * using the vecS underlying EdgeOutList type
186 */
187
188/* Note on the implementation:
189 * The third template argument (a bool) should always be left to its
190 * default value. It is used to flag whether the Graph type considered
191 * has an implicit integer vertex index.
192 * This flag is used in the template specialisation below to change
193 * the template definition when `HasImplicitIndex == True`,
194 * i.e. the first definition of `TreeSearch` is for the case `HasImplicitIndex
195 * == False` while the template partial specialisation below is for the case
196 * `HasImplicitIndex == True`.
197 */
198template <
199 typename Graph, typename PMap = void,
200 bool HasImplicitIndex = utils::detail::has_integral_index<Graph>>
201class TreeSearch
202 : public TreeSearchBase<Graph, boost::associative_property_map<PMap>> {
203 private:
204 using Base = TreeSearchBase<Graph, boost::associative_property_map<PMap>>;
205 using vertex = typename Base::vertex;
206
207 public:
208 TreeSearch(vertex v, Graph&& g, PMap& pmap)
209 : Base(v, std::forward<Graph>(g), boost::make_assoc_property_map(pmap)) {}
210};

212template <typename Graph>
213class TreeSearch<Graph, void, true>
214 : public TreeSearchBase<
215 Graph,
216 decltype(boost::get(boost::vertex_index, std::declval<Graph>()))> {
217 private:
218 using PMap = decltype(boost::get(boost::vertex_index, std::declval<Graph>()));
219 using Base = TreeSearchBase<Graph, PMap>;
220 using vertex = typename Base::vertex;
221
222 public:
223 TreeSearch(vertex v, Graph&& g)
224 : Base(v, std::forward<Graph>(g), boost::get(boost::vertex_index, g)) {}
225};

229/* Return type of `run_bfs`.
230 *
231 * For each vertex, computes the BFS distance to the root
232 * as well as the next node on the path towards root (parent).
233 * These are stored in the base class TreeSearchBase and can
234 * be retrieved through one of the member functions defined therein
235 */
236template <typename... Args>
237class BFS : public detail::TreeSearch<Args...> {
238 private:
239 using Base = detail::TreeSearch<Args...>;
240
241 public:
242 // inherit constructors
243 using Base::Base;
244
245 using parent_vec = typename Base::parent_vec;
246
247 /* Run bfs */
248 void run() override {
249 auto vis = boost::make_bfs_visitor(this->visitor);
250 boost::breadth_first_search(
251 this->graph, this->root, this->get_default_named_params(vis));
252 }
253};

255/* Return type of `run_dfs`.
256 *
257 * For each vertex, computes the DFS distance to the root
258 * as well as the next node on the path towards root (parent).
259 * These are stored in the base class TreeSearchBase and can
260 * be retrieved through one of the member functions defined therein
261 */
262template <typename... Args>
263class DFS : public detail::TreeSearch<Args...> {
264 private:
265 using Base = detail::TreeSearch<Args...>;
266
267 public:
268 // inherit constructors
269 using Base::Base;
270
271 using parent_vec = typename Base::parent_vec;
272
273 /* Run dfs */
274 void run() override {
275 auto vis = boost::make_dfs_visitor(this->visitor);
276 boost::depth_first_search(
277 this->graph,
278 this->get_default_named_params(vis).root_vertex(this->root));
279 }
