68class UnitID {
69 public:
70 UnitID() : data_(std::make_shared<UnitData>()) {}
71
73 std::string repr() const;
74
76 std::string reg_name() const { return data_->name_; }
77
79 unsigned reg_dim() const { return data_->index_.size(); }
80
82 std::vector<unsigned> index() const { return data_->index_; }
83
85 UnitType type() const { return data_->type_; }
86
88 register_info_t reg_info() const { return {type(), reg_dim()}; }
89
90 bool operator<(const UnitID &other) const {
91 int n = data_->name_.compare(other.data_->name_);
92 if (n > 0) return false;
93 if (n < 0) return true;
94 return data_->index_ < other.data_->index_;
95 }
96 bool operator>(const UnitID &other) const { return other < *this; }
97 bool operator==(const UnitID &other) const {
98 return (this->data_->name_ == other.data_->name_) &&
99 (this->data_->index_ == other.data_->index_);
100 }
101 bool operator!=(const UnitID &other) const { return !(*this == other); }
102
103 friend std::size_t hash_value(UnitID const &unitid) {
104 std::size_t seed = 0;
105 boost::hash_combine(seed, unitid.data_->name_);
106 boost::hash_combine(seed, unitid.data_->index_);
107 boost::hash_combine(seed, unitid.data_->type_);
108 return seed;
109 }
110
111 protected:
112 UnitID(
113 const std::string &name, const std::vector<unsigned> &index,
114 UnitType type)
115 : data_(std::make_shared<UnitData>(name, index, type)) {}
116
117 private:
118 struct UnitData {
119 std::string name_;
120 std::vector<unsigned> index_;
121 UnitType type_;
122
123 UnitData() : name_(), index_(), type_(UnitType::Qubit) {}
124 UnitData(
125 const std::string &name, const std::vector<unsigned> &index,
126 UnitType type)
127 : name_(name), index_(index), type_(type) {
128 static const std::string id_regex_str = "[a-z][A-Za-z0-9_]*";
129 static const std::regex id_regex(id_regex_str);
130 if (!name.empty() && !std::regex_match(name, id_regex)) {
131 std::stringstream msg;
132 msg << "UnitID name '" << name << "' does not match '" << id_regex_str
133 << "', as required for QASM conversion.";
134 tket_log()->warn(msg.str());
135 }
136 }
137 };
138 std::shared_ptr<UnitData> data_;
139};

328template <typename UnitA, typename UnitB>
329static bool update_map(unit_bimap_t &m, const std::map<UnitA, UnitB> &um) {
330 unit_map_t new_m;
331 bool changed = false;
332 for (const std::pair<const UnitA, UnitB> &pair : um) {
333 const auto &it = m.right.find(pair.first);
334 if (it == m.right.end()) {
335 continue;
336 }
337 new_m.insert({it->second, pair.second});
338 changed |= (m.right.erase(pair.first) > 0);
339 }
340 for (const std::pair<const UnitID, UnitID> &pair : new_m) {
341 changed |= m.left.insert(pair).second;
342 }
343 return changed;
344}

360template <typename UnitA, typename UnitB>
361bool update_maps(
362 std::shared_ptr<unit_bimaps_t> maps,
363 const std::map<UnitA, UnitB> &um_initial,
364 const std::map<UnitA, UnitB> &um_final) {
365 if (!maps) return false;
366 // Can only work for Unit classes
367 static_assert(std::is_base_of<UnitID, UnitA>::value);
368 static_assert(std::is_base_of<UnitID, UnitB>::value);
369 // Unit types must be related, so cannot rename e.g. Bits to Qubits
370 static_assert(
371 std::is_base_of<UnitA, UnitB>::value ||
372 std::is_base_of<UnitB, UnitA>::value);
373
374 bool changed = false;
375 changed |= update_map(maps->initial, um_initial);
376 changed |= update_map(maps->final, um_final);
377 return changed;
378}
