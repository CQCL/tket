69class UnitID {
70 public:
71 UnitID() : data_(std::make_shared<UnitData>()) {}
72
74 std::string repr() const;
75
77 std::string reg_name() const { return data_->name_; }
78
80 unsigned reg_dim() const { return data_->index_.size(); }
81
83 std::vector<unsigned> index() const { return data_->index_; }
84
86 UnitType type() const { return data_->type_; }
87
89 register_info_t reg_info() const { return {type(), reg_dim()}; }
90
91 bool operator<(const UnitID &other) const {
92 int n = data_->name_.compare(other.data_->name_);
93 if (n > 0) return false;
94 if (n < 0) return true;
95 return data_->index_ < other.data_->index_;
96 }
97 bool operator>(const UnitID &other) const { return other < *this; }
98 bool operator==(const UnitID &other) const {
99 return (this->data_->name_ == other.data_->name_) &&
100 (this->data_->index_ == other.data_->index_);
101 }
102 bool operator!=(const UnitID &other) const { return !(*this == other); }
103
104 friend std::size_t hash_value(UnitID const &unitid) {
105 std::size_t seed = 0;
106 boost::hash_combine(seed, unitid.data_->name_);
107 boost::hash_combine(seed, unitid.data_->index_);
108 boost::hash_combine(seed, unitid.data_->type_);
109 return seed;
110 }
111
112 protected:
113 UnitID(
114 const std::string &name, const std::vector<unsigned> &index,
115 UnitType type)
116 : data_(std::make_shared<UnitData>(name, index, type)) {}
117
118 private:
119 struct UnitData {
120 std::string name_;
121 std::vector<unsigned> index_;
122 UnitType type_;
123
124 UnitData() : name_(), index_(), type_(UnitType::Qubit) {}
125 UnitData(
126 const std::string &name, const std::vector<unsigned> &index,
127 UnitType type)
128 : name_(name), index_(index), type_(type) {
129 static const std::string id_regex_str = "[a-z][A-Za-z0-9_]*";
130 static const std::regex id_regex(id_regex_str);
131 if (!name.empty() && !std::regex_match(name, id_regex)) {
132 std::stringstream msg;
133 msg << "UnitID name '" << name << "' does not match '" << id_regex_str
134 << "', as required for QASM conversion.";
135 tket_log()->warn(msg.str());
136 }
137 }
138 };
139 std::shared_ptr<UnitData> data_;
140};

375template <typename UnitA, typename UnitB>
376static bool update_map(unit_bimap_t &m, const std::map<UnitA, UnitB> &um) {
377 unit_map_t new_m;
378 bool changed = false;
379 for (const std::pair<const UnitA, UnitB> &pair : um) {
380 const auto &it = m.right.find(pair.first);
381 if (it == m.right.end()) {
382 continue;
383 }
384 new_m.insert({it->second, pair.second});
385 changed |= (m.right.erase(pair.first) > 0);
386 }
387 for (const std::pair<const UnitID, UnitID> &pair : new_m) {
388 changed |= m.left.insert(pair).second;
389 }
390 return changed;
391}

407template <typename UnitA, typename UnitB>
408bool update_maps(
409 std::shared_ptr<unit_bimaps_t> maps,
410 const std::map<UnitA, UnitB> &um_initial,
411 const std::map<UnitA, UnitB> &um_final) {
412 if (!maps) return false;
413 // Can only work for Unit classes
414 static_assert(std::is_base_of<UnitID, UnitA>::value);
415 static_assert(std::is_base_of<UnitID, UnitB>::value);
416 // Unit types must be related, so cannot rename e.g. Bits to Qubits
417 static_assert(
418 std::is_base_of<UnitA, UnitB>::value ||
419 std::is_base_of<UnitB, UnitA>::value);
420
421 bool changed = false;
422 changed |= update_map(maps->initial, um_initial);
423 changed |= update_map(maps->final, um_final);
424 return changed;
425}
