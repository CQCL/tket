327 template <typename UnitA, typename UnitB>
328 static bool update_map(unit_bimap_t &m, const std::map<UnitA, UnitB> &um) {
329  unit_map_t new_m;
330  bool changed = false;
331  for (const std::pair<const UnitA, UnitB> &pair : um) {
332  const auto &it = m.right.find(pair.first);
333  if (it == m.right.end()) {
334  continue;
335  }
336  new_m.insert({it->second, pair.second});
337  changed |= (m.right.erase(pair.first) > 0);
338  }
339  for (const std::pair<const UnitID, UnitID> &pair : new_m) {
340  changed |= m.left.insert(pair).second;
341  }
342  return changed;
343 }

359 template <typename UnitA, typename UnitB>
360 bool update_maps(
361  std::shared_ptr<unit_bimaps_t> maps,
362  const std::map<UnitA, UnitB> &um_initial,
363  const std::map<UnitA, UnitB> &um_final) {
364  if (!maps) return false;
365  // Can only work for Unit classes
366  static_assert(std::is_base_of<UnitID, UnitA>::value);
367  static_assert(std::is_base_of<UnitID, UnitB>::value);
368  // Unit types must be related, so cannot rename e.g. Bits to Qubits
369  static_assert(
370  std::is_base_of<UnitA, UnitB>::value ||
371  std::is_base_of<UnitB, UnitA>::value);
372 
373  bool changed = false;
374  changed |= update_map(maps->initial, um_initial);
375  changed |= update_map(maps->final, um_final);
376  return changed;
377 }
