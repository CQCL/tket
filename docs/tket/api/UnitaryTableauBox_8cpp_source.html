22UnitaryTableauBox::UnitaryTableauBox(const UnitaryTableau& tab)
23 : Box(OpType::UnitaryTableauBox), tab_(tab) {
24 std::set<Qubit> tab_qbs = tab_.get_qubits();
25 std::set<Qubit> qbs;
26 for (unsigned i = 0; i < tab_qbs.size(); ++i) {
27 qbs.insert(Qubit(i));
28 }
29 if (tab_qbs != qbs)
30 throw std::invalid_argument(
31 "UnitaryTableauBox requires tableau qubits to have default, "
32 "consecutive indexing");
33}
