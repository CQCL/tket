21 UnitaryTableauBox::UnitaryTableauBox(const UnitaryTableau& tab)
22  : Box(OpType::UnitaryTableauBox), tab_(tab) {
23  std::set<Qubit> tab_qbs = tab_.get_qubits();
24  std::set<Qubit> qbs;
25  for (unsigned i = 0; i < tab_qbs.size(); ++i) {
26  qbs.insert(Qubit(i));
27  }
28  if (tab_qbs != qbs)
29  throw std::invalid_argument(
30  "UnitaryTableauBox requires tableau qubits to have default, "
31  "consecutive indexing");
32 }
