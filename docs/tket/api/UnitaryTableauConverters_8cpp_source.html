21 UnitaryTableau circuit_to_unitary_tableau(const Circuit& circ) {
22  UnitaryTableau tab(circ.all_qubits());
23  for (const Command& com : circ) {
24  auto args = com.get_args();
25  qubit_vector_t qbs = {args.begin(), args.end()};
26  tab.apply_gate_at_end(com.get_op_ptr()->get_type(), qbs);
27  }
28  return tab;
29 }

31 Circuit unitary_tableau_to_circuit(const UnitaryTableau& tab) {
32  SymplecticTableau tabl(tab.tab_);
33  unsigned size = tabl.get_n_qubits();
34  Circuit c(size);
35  /*
36  * Aaronson-Gottesman: Improved Simulation of Stabilizer Circuits, Theorem 8
37  * Any unitary stabilizer circuit has an equivalent circuit in canonical form
38  * (H-C-P-C-P-C-H-P-C-P-C).
39  * Produces a circuit realising the tableau, but consumes the tableau in the
40  * process.
41  */
42 
43  /*
44  * Step 1: Use Hadamards (in our case, Vs) to make C (z rows of xmat_) have
45  * full rank
46  */
47  MatrixXb echelon = tabl.xmat_.block(size, 0, size, size);
48  std::map<unsigned, unsigned> leading_val_to_col;
49  for (unsigned i = 0; i < size; i++) {
50  for (unsigned j = 0; j < size; j++) {
51  if (echelon(j, i)) {
52  if (leading_val_to_col.find(j) == leading_val_to_col.end()) {
53  leading_val_to_col.insert({j, i});
54  break;
55  } else {
56  unsigned l = leading_val_to_col.at(j);
57  for (unsigned k = 0; k < size; k++) {
58  echelon(k, i) ^= echelon(k, l);
59  }
60  }
61  }
62  }
63  if (leading_val_to_col.size() > i)
64  continue; // Independent of previous cols
65  c.add_op<unsigned>(OpType::V, {i});
66  tabl.apply_V(i);
67  tabl.apply_V(i);
68  tabl.apply_V(i);
69  echelon.col(i) = tabl.zmat_.block(size, i, size, 1);
70  for (unsigned j = 0; j < size; j++) {
71  if (echelon(j, i)) {
72  if (leading_val_to_col.find(j) == leading_val_to_col.end()) {
73  leading_val_to_col.insert({j, i});
74  break;
75  } else {
76  unsigned l = leading_val_to_col.at(j);
77  for (unsigned k = 0; k < size; k++) {
78  echelon(k, i) ^= echelon(k, l);
79  }
80  }
81  }
82  }
83  if (leading_val_to_col.size() == i)
84  throw std::invalid_argument("Stabilisers are not mutually independent");
85  }
86 
87  /*
88  * Step 2: Use CXs to perform Gaussian elimination on C (zpauli_x), producing
89  * / A B \
90  * \ I D /
91  */
92  MatrixXb to_reduce = tabl.xmat_.block(size, 0, size, size);
93  for (const std::pair<unsigned, unsigned>& qbs :
94  gaussian_elimination_col_ops(to_reduce)) {
95  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
96  tabl.apply_CX(qbs.first, qbs.second);
97  }
98 
99  /*
100  * Step 3: Commutativity of the stabilizer implies that ID^T is symmetric,
101  * therefore D is symmetric, and we can apply phase (S) gates to add a
102  * diagonal matrix to D and use Lemma 7 to convert D to the form D = MM^T
103  * for some invertible M.
104  */
105  std::pair<MatrixXb, MatrixXb> zp_z_llt =
106  binary_LLT_decomposition(tabl.zmat_.block(size, 0, size, size));
107  for (unsigned i = 0; i < size; i++) {
108  if (zp_z_llt.second(i, i)) {
109  c.add_op<unsigned>(OpType::S, {i});
110  tabl.apply_S(i);
111  tabl.apply_S(i);
112  tabl.apply_S(i);
113  }
114  }
115 
116  /*
117  * Step 4: Use CXs to produce
118  * / A B \
119  * \ M M /
120  * Note that when we map I to IM, we also map D to D(M^T)^{-1} = M.
121  */
122  std::vector<std::pair<unsigned, unsigned>> m_to_i =
123  gaussian_elimination_col_ops(zp_z_llt.first);
124  for (std::vector<std::pair<unsigned, unsigned>>::reverse_iterator it =
125  m_to_i.rbegin();
126  it != m_to_i.rend(); it++) {
127  c.add_op<unsigned>(OpType::CX, {it->first, it->second});
128  tabl.apply_CX(it->first, it->second);
129  }
130 
131  /*
132  * Step 5: Apply phases to all n qubits to obtain
133  * / A B \
134  * \ M 0 /
135  * Since M is full rank, there exists some subset S of qubits such that
136  * applying two phases in succession (Z) to every a \in S will preserve
137  * the tableau, but set r_{n+1} = ... = r_{2n} = 0 (zpauli_phase = 0^n).
138  * Apply two phases (Z) to every a \in S. DELAYED UNTIL END
139  */
140  for (unsigned i = 0; i < size; i++) {
141  c.add_op<unsigned>(OpType::S, {i});
142  tabl.apply_S(i);
143  tabl.apply_S(i);
144  tabl.apply_S(i);
145  }
146 
147  /*
148  * Step 6: Use CXs to perform Gaussian elimination on M, producing
149  * / A B \
150  * \ I 0 /
151  * By commutativity relations, IB^T = A0^T + I, therefore B = I.
152  */
153  to_reduce = tabl.xmat_.block(size, 0, size, size);
154  for (const std::pair<unsigned, unsigned>& qbs :
155  gaussian_elimination_col_ops(to_reduce)) {
156  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
157  tabl.apply_CX(qbs.first, qbs.second);
158  }
159 
160  /*
161  * Step 7: Use Hadamards to produce
162  * / I A \
163  * \ 0 I /
164  */
165  for (unsigned i = 0; i < size; i++) {
166  c.add_op<unsigned>(OpType::H, {i});
167  tabl.apply_S(i);
168  tabl.apply_V(i);
169  tabl.apply_S(i);
170  }
171 
172  /*
173  * Step 8: Now commutativity of the destabilizer implies that A is symmetric,
174  * therefore we can again use phase (S) gates and Lemma 7 to make A = NN^T for
175  * some invertible N.
176  */
177  std::pair<MatrixXb, MatrixXb> xp_z_llt =
178  binary_LLT_decomposition(tabl.zmat_.block(0, 0, size, size));
179  for (unsigned i = 0; i < size; i++) {
180  if (xp_z_llt.second(i, i)) {
181  c.add_op<unsigned>(OpType::S, {i});
182  tabl.apply_S(i);
183  tabl.apply_S(i);
184  tabl.apply_S(i);
185  }
186  }
187 
188  /*
189  * Step 9: Use CXs to produce
190  * / N N \
191  * \ 0 C /
192  */
193  std::vector<std::pair<unsigned, unsigned>> n_to_i =
194  gaussian_elimination_col_ops(xp_z_llt.first);
195  for (std::vector<std::pair<unsigned, unsigned>>::reverse_iterator it =
196  n_to_i.rbegin();
197  it != n_to_i.rend(); it++) {
198  c.add_op<unsigned>(OpType::CX, {it->first, it->second});
199  tabl.apply_CX(it->first, it->second);
200  }
201 
202  /*
203  * Step 10: Use phases (S) to produce
204  * / N 0 \
205  * \ 0 C /
206  * then by commutativity relations NC^T = I. Next apply two phases (Z) each to
207  * some subset of qubits in order to preserve the above tableau, but set
208  * r_1 = ... = r_n = 0 (xpauli_phase = 0^n). DELAYED UNTIL END
209  */
210  for (unsigned i = 0; i < size; i++) {
211  c.add_op<unsigned>(OpType::S, {i});
212  tabl.apply_S(i);
213  tabl.apply_S(i);
214  tabl.apply_S(i);
215  }
216 
217  /*
218  * Step 11: Use CXs to produce
219  * / I 0 \
220  * \ 0 I /
221  */
222  for (const std::pair<unsigned, unsigned>& qbs :
223  gaussian_elimination_col_ops(tabl.xmat_.block(0, 0, size, size))) {
224  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
225  tabl.apply_CX(qbs.first, qbs.second);
226  }
227 
228  /*
229  * DELAYED STEPS: Set all phases to 0 by applying Z or X gates
230  */
231  for (unsigned i = 0; i < size; i++) {
232  if (tabl.phase_(i)) {
233  c.add_op<unsigned>(OpType::Z, {i});
234  tabl.apply_S(i);
235  tabl.apply_S(i);
236  }
237  if (tabl.phase_(i + size)) {
238  c.add_op<unsigned>(OpType::X, {i});
239  tabl.apply_V(i);
240  tabl.apply_V(i);
241  }
242  }
243 
244  /*
245  * Rename qubits
246  */
247  unit_map_t rename_map;
248  for (boost::bimap<Qubit, unsigned>::const_iterator iter = tab.qubits_.begin(),
249  iend = tab.qubits_.end();
250  iter != iend; ++iter) {
251  rename_map.insert({Qubit(iter->right), iter->left});
252  }
253  c.rename_units(rename_map);
254 
255  return c.transpose();
256 }
