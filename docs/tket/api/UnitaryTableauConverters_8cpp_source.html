21 UnitaryTableau circuit_to_unitary_tableau(const Circuit& circ) {
22  UnitaryTableau tab(circ.all_qubits());
23  for (const Command& com : circ) {
24  auto args = com.get_args();
25  qubit_vector_t qbs = {args.begin(), args.end()};
26  tab.apply_gate_at_end(com.get_op_ptr()->get_type(), qbs);
27  }
28  return tab;
29 }

31 Circuit unitary_tableau_to_circuit(const UnitaryTableau& tab) {
32  SymplecticTableau tabl(tab.tab_);
33  unsigned size = tabl.get_n_qubits();
34  Circuit c(size);
35  /*
36  * Aaronson-Gottesman: Improved Simulation of Stabilizer Circuits, Theorem 8
37  * Any unitary stabilizer circuit has an equivalent circuit in canonical form
38  * (H-C-P-C-P-C-H-P-C-P-C).
39  * Produces a circuit realising the tableau, but consumes the tableau in the
40  * process.
41  */
42 
43  /*
44  * Step 1: Use Hadamards (in our case, Vs) to make C (z rows of xmat_) have
45  * full rank
46  */
47  MatrixXb echelon = tabl.xmat.block(size, 0, size, size);
48  std::map<unsigned, unsigned> leading_val_to_col;
49  for (unsigned i = 0; i < size; i++) {
50  for (unsigned j = 0; j < size; j++) {
51  if (echelon(j, i)) {
52  if (leading_val_to_col.find(j) == leading_val_to_col.end()) {
53  leading_val_to_col.insert({j, i});
54  break;
55  } else {
56  unsigned l = leading_val_to_col.at(j);
57  for (unsigned k = 0; k < size; k++) {
58  echelon(k, i) ^= echelon(k, l);
59  }
60  }
61  }
62  }
63  if (leading_val_to_col.size() > i)
64  continue; // Independent of previous cols
65  c.add_op<unsigned>(OpType::V, {i});
66  tabl.apply_V(i);
67  tabl.apply_X(i);
68  echelon.col(i) = tabl.zmat.block(size, i, size, 1);
69  for (unsigned j = 0; j < size; j++) {
70  if (echelon(j, i)) {
71  if (leading_val_to_col.find(j) == leading_val_to_col.end()) {
72  leading_val_to_col.insert({j, i});
73  break;
74  } else {
75  unsigned l = leading_val_to_col.at(j);
76  for (unsigned k = 0; k < size; k++) {
77  echelon(k, i) ^= echelon(k, l);
78  }
79  }
80  }
81  }
82  if (leading_val_to_col.size() == i)
83  throw std::invalid_argument("Stabilisers are not mutually independent");
84  }
85 
86  /*
87  * Step 2: Use CXs to perform Gaussian elimination on C (zpauli_x), producing
88  * / A B \
89  * \ I D /
90  */
91  MatrixXb to_reduce = tabl.xmat.block(size, 0, size, size);
92  for (const std::pair<unsigned, unsigned>& qbs :
93  gaussian_elimination_col_ops(to_reduce)) {
94  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
95  tabl.apply_CX(qbs.first, qbs.second);
96  }
97 
98  /*
99  * Step 3: Commutativity of the stabilizer implies that ID^T is symmetric,
100  * therefore D is symmetric, and we can apply phase (S) gates to add a
101  * diagonal matrix to D and use Lemma 7 to convert D to the form D = MM^T
102  * for some invertible M.
103  */
104  std::pair<MatrixXb, MatrixXb> zp_z_llt =
105  binary_LLT_decomposition(tabl.zmat.block(size, 0, size, size));
106  for (unsigned i = 0; i < size; i++) {
107  if (zp_z_llt.second(i, i)) {
108  c.add_op<unsigned>(OpType::S, {i});
109  tabl.apply_S(i);
110  tabl.apply_Z(i);
111  }
112  }
113 
114  /*
115  * Step 4: Use CXs to produce
116  * / A B \
117  * \ M M /
118  * Note that when we map I to IM, we also map D to D(M^T)^{-1} = M.
119  */
120  std::vector<std::pair<unsigned, unsigned>> m_to_i =
121  gaussian_elimination_col_ops(zp_z_llt.first);
122  for (std::vector<std::pair<unsigned, unsigned>>::reverse_iterator it =
123  m_to_i.rbegin();
124  it != m_to_i.rend(); it++) {
125  c.add_op<unsigned>(OpType::CX, {it->first, it->second});
126  tabl.apply_CX(it->first, it->second);
127  }
128 
129  /*
130  * Step 5: Apply phases to all n qubits to obtain
131  * / A B \
132  * \ M 0 /
133  * Since M is full rank, there exists some subset S of qubits such that
134  * applying two phases in succession (Z) to every a \in S will preserve
135  * the tableau, but set r_{n+1} = ... = r_{2n} = 0 (zpauli_phase = 0^n).
136  * Apply two phases (Z) to every a \in S. DELAYED UNTIL END
137  */
138  for (unsigned i = 0; i < size; i++) {
139  c.add_op<unsigned>(OpType::S, {i});
140  tabl.apply_S(i);
141  tabl.apply_Z(i);
142  }
143 
144  /*
145  * Step 6: Use CXs to perform Gaussian elimination on M, producing
146  * / A B \
147  * \ I 0 /
148  * By commutativity relations, IB^T = A0^T + I, therefore B = I.
149  */
150  to_reduce = tabl.xmat.block(size, 0, size, size);
151  for (const std::pair<unsigned, unsigned>& qbs :
152  gaussian_elimination_col_ops(to_reduce)) {
153  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
154  tabl.apply_CX(qbs.first, qbs.second);
155  }
156 
157  /*
158  * Step 7: Use Hadamards to produce
159  * / I A \
160  * \ 0 I /
161  */
162  for (unsigned i = 0; i < size; i++) {
163  c.add_op<unsigned>(OpType::H, {i});
164  tabl.apply_H(i);
165  }
166 
167  /*
168  * Step 8: Now commutativity of the destabilizer implies that A is symmetric,
169  * therefore we can again use phase (S) gates and Lemma 7 to make A = NN^T for
170  * some invertible N.
171  */
172  std::pair<MatrixXb, MatrixXb> xp_z_llt =
173  binary_LLT_decomposition(tabl.zmat.block(0, 0, size, size));
174  for (unsigned i = 0; i < size; i++) {
175  if (xp_z_llt.second(i, i)) {
176  c.add_op<unsigned>(OpType::S, {i});
177  tabl.apply_S(i);
178  tabl.apply_Z(i);
179  }
180  }
181 
182  /*
183  * Step 9: Use CXs to produce
184  * / N N \
185  * \ 0 C /
186  */
187  std::vector<std::pair<unsigned, unsigned>> n_to_i =
188  gaussian_elimination_col_ops(xp_z_llt.first);
189  for (std::vector<std::pair<unsigned, unsigned>>::reverse_iterator it =
190  n_to_i.rbegin();
191  it != n_to_i.rend(); it++) {
192  c.add_op<unsigned>(OpType::CX, {it->first, it->second});
193  tabl.apply_CX(it->first, it->second);
194  }
195 
196  /*
197  * Step 10: Use phases (S) to produce
198  * / N 0 \
199  * \ 0 C /
200  * then by commutativity relations NC^T = I. Next apply two phases (Z) each to
201  * some subset of qubits in order to preserve the above tableau, but set
202  * r_1 = ... = r_n = 0 (xpauli_phase = 0^n). DELAYED UNTIL END
203  */
204  for (unsigned i = 0; i < size; i++) {
205  c.add_op<unsigned>(OpType::S, {i});
206  tabl.apply_S(i);
207  tabl.apply_Z(i);
208  }
209 
210  /*
211  * Step 11: Use CXs to produce
212  * / I 0 \
213  * \ 0 I /
214  */
215  for (const std::pair<unsigned, unsigned>& qbs :
216  gaussian_elimination_col_ops(tabl.xmat.block(0, 0, size, size))) {
217  c.add_op<unsigned>(OpType::CX, {qbs.first, qbs.second});
218  tabl.apply_CX(qbs.first, qbs.second);
219  }
220 
221  /*
222  * DELAYED STEPS: Set all phases to 0 by applying Z or X gates
223  */
224  for (unsigned i = 0; i < size; i++) {
225  if (tabl.phase(i)) {
226  c.add_op<unsigned>(OpType::Z, {i});
227  tabl.apply_Z(i);
228  }
229  if (tabl.phase(i + size)) {
230  c.add_op<unsigned>(OpType::X, {i});
231  tabl.apply_X(i);
232  ;
233  }
234  }
235 
236  /*
237  * Rename qubits
238  */
239  unit_map_t rename_map;
240  for (boost::bimap<Qubit, unsigned>::const_iterator iter = tab.qubits_.begin(),
241  iend = tab.qubits_.end();
242  iter != iend; ++iter) {
243  rename_map.insert({Qubit(iter->right), iter->left});
244  }
245  c.rename_units(rename_map);
246 
247  return c.transpose();
248 }
