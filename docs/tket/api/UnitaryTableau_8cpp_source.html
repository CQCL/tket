24 UnitaryTableau::UnitaryTableau(unsigned n) : tab_({}) {
25  MatrixXb xmat(2 * n, n);
26  xmat << MatrixXb::Identity(n, n), MatrixXb::Zero(n, n);
27  MatrixXb zmat(2 * n, n);
28  zmat << MatrixXb::Zero(n, n), MatrixXb::Identity(n, n);
29  tab_ = SymplecticTableau(xmat, zmat, VectorXb::Zero(2 * n));
30  qubits_ = boost::bimap<Qubit, unsigned>();
31  for (unsigned i = 0; i < n; ++i) {
32  qubits_.insert({Qubit(i), i});
33  }
34 }

44 UnitaryTableau::UnitaryTableau(
45  const MatrixXb& xx, const MatrixXb& xz, const VectorXb& xph,
46  const MatrixXb& zx, const MatrixXb& zz, const VectorXb& zph)
47  : tab_({}) {
48  unsigned n_qubits = xx.rows();
49  if ((xx.cols() != n_qubits) || (xz.rows() != n_qubits) ||
50  (xz.cols() != n_qubits) || (xph.size() != n_qubits) ||
51  (zx.rows() != n_qubits) || (zx.cols() != n_qubits) ||
52  (zz.rows() != n_qubits) || (zz.cols() != n_qubits) ||
53  (zph.size() != n_qubits))
54  throw std::invalid_argument(
55  "Unitary tableau requires equally-sized square matrices and vectors");
56  MatrixXb xmat(2 * n_qubits, n_qubits);
57  xmat << xx, zx;
58  MatrixXb zmat(2 * n_qubits, n_qubits);
59  zmat << xz, zz;
60  VectorXb phase(2 * n_qubits);
61  phase << xph, zph;
62  tab_ = SymplecticTableau(xmat, zmat, phase);
63  MatrixXb expected_anticommutes(2 * n_qubits, 2 * n_qubits);
64  expected_anticommutes << MatrixXb::Zero(n_qubits, n_qubits),
65  MatrixXb::Identity(n_qubits, n_qubits),
66  MatrixXb::Identity(n_qubits, n_qubits),
67  MatrixXb::Zero(n_qubits, n_qubits);
68  if (tab_.anticommuting_rows() != expected_anticommutes)
69  throw std::invalid_argument(
70  "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
71  if (tab_.rank() != 2 * n_qubits)
72  throw std::invalid_argument(
73  "Rows of UnitaryTableau are not linearly independent");
74  qubits_ = boost::bimap<Qubit, unsigned>();
75  for (unsigned i = 0; i < n_qubits; ++i) {
76  qubits_.insert({Qubit(i), i});
77  }
78 }

80 QubitPauliTensor UnitaryTableau::get_xrow(const Qubit& qb) const {
81  unsigned uq = qubits_.left.at(qb);
82  PauliStabiliser stab = tab_.get_pauli(uq);
83  std::list<Qubit> qbs;
84  for (unsigned i = 0; i < qubits_.size(); ++i) {
85  qbs.push_back(qubits_.right.at(i));
86  }
87  std::list<Pauli> string = {stab.string.begin(), stab.string.end()};
88  Complex coeff = 1.;
89  if (!stab.coeff) coeff *= -1.;
90  return QubitPauliTensor(QubitPauliString(qbs, string), coeff);
91 }

93 QubitPauliTensor UnitaryTableau::get_zrow(const Qubit& qb) const {
94  unsigned uqb = qubits_.left.at(qb);
95  PauliStabiliser stab = tab_.get_pauli(uqb + qubits_.size());
96  std::list<Qubit> qbs;
97  for (unsigned i = 0; i < qubits_.size(); ++i) {
98  qbs.push_back(qubits_.right.at(i));
99  }

101  Complex coeff = 1.;
102  if (!stab.coeff) coeff *= -1.;
103  return QubitPauliTensor(QubitPauliString(qbs, string), coeff);
104 }

106 QubitPauliTensor UnitaryTableau::get_row_product(
107  const QubitPauliTensor& qpt) const {
108  QubitPauliTensor result(qpt.coeff);
109  for (const std::pair<const Qubit, Pauli>& p : qpt.string.map) {
110  auto qks_it = qubits_.left.find(p.first);
111  if (qks_it == qubits_.left.end()) {
112  // Acts as identity on p.first
113  result = result * QubitPauliTensor(p.first, p.second);
114  } else {
115  switch (p.second) {
116  case Pauli::I: {
117  break;
118  }
119  case Pauli::X: {
120  result = result * get_xrow(p.first);
121  break;
122  }
123  case Pauli::Y: {
124  // Y = iXZ
125  result = result * get_xrow(p.first);
126  result = result * get_zrow(p.first);
127  result.coeff *= i_;
128  break;
129  }
130  case Pauli::Z: {
131  result = result * get_zrow(p.first);
132  break;
133  }
134  }
135  }
136  }
137  return result;
138 }

185 void UnitaryTableau::apply_gate_at_front(
186  OpType type, const qubit_vector_t& qbs) {
187  switch (type) {
188  case OpType::Z: {
189  apply_S_at_front(qbs.at(0));
190  apply_S_at_front(qbs.at(0));
191  break;
192  }
193  case OpType::X: {
194  apply_V_at_front(qbs.at(0));
195  apply_V_at_front(qbs.at(0));
196  break;
197  }
198  case OpType::Y: {
199  apply_S_at_front(qbs.at(0));
200  apply_S_at_front(qbs.at(0));
201  apply_V_at_front(qbs.at(0));
202  apply_V_at_front(qbs.at(0));
203  break;
204  }
205  case OpType::S: {
206  apply_S_at_front(qbs.at(0));
207  break;
208  }
209  case OpType::Sdg: {
210  apply_S_at_front(qbs.at(0));
211  apply_S_at_front(qbs.at(0));
212  apply_S_at_front(qbs.at(0));
213  break;
214  }
215  case OpType::V: {
216  apply_V_at_front(qbs.at(0));
217  break;
218  }
219  case OpType::Vdg: {
220  apply_V_at_front(qbs.at(0));
221  apply_V_at_front(qbs.at(0));
222  apply_V_at_front(qbs.at(0));
223  break;
224  }
225  case OpType::H: {
226  apply_S_at_front(qbs.at(0));
227  apply_V_at_front(qbs.at(0));
228  apply_S_at_front(qbs.at(0));
229  break;
230  }
231  case OpType::CX: {
232  apply_CX_at_front(qbs.at(0), qbs.at(1));
233  break;
234  }
235  case OpType::CY: {
236  apply_S_at_front(qbs.at(1));
237  apply_CX_at_front(qbs.at(0), qbs.at(1));
238  apply_S_at_front(qbs.at(1));
239  apply_S_at_front(qbs.at(1));
240  apply_S_at_front(qbs.at(1));
241  break;
242  }
243  case OpType::CZ: {
244  apply_S_at_front(qbs.at(1));
245  apply_V_at_front(qbs.at(1));
246  apply_S_at_front(qbs.at(1));
247  apply_CX_at_front(qbs.at(0), qbs.at(1));
248  apply_S_at_front(qbs.at(1));
249  apply_V_at_front(qbs.at(1));
250  apply_S_at_front(qbs.at(1));
251  break;
252  }
253  case OpType::SWAP: {
254  apply_CX_at_front(qbs.at(0), qbs.at(1));
255  apply_CX_at_front(qbs.at(1), qbs.at(0));
256  apply_CX_at_front(qbs.at(0), qbs.at(1));
257  break;
258  }
259  case OpType::BRIDGE: {
260  apply_CX_at_front(qbs.at(0), qbs.at(2));
261  break;
262  }
263  case OpType::noop: {
264  break;
265  }
266  default: {
267  throw BadOpType(
268  "Cannot be applied to a UnitaryTableau: not a Clifford gate", type);
269  }
270  }

281 void UnitaryTableau::apply_pauli_at_front(
282  const QubitPauliTensor& pauli, unsigned half_pis) {
283  half_pis = half_pis % 4;
284  if (half_pis == 0) return; // Identity
285  if (half_pis == 2) { // Degenerates to product of PI rotations
286  for (const std::pair<const Qubit, Pauli>& term : pauli.string.map) {
287  switch (term.second) {
288  case Pauli::I: {
289  break;
290  }
291  case Pauli::X: {
292  apply_gate_at_front(OpType::X, {term.first});
293  break;
294  }
295  case Pauli::Y: {
296  apply_gate_at_front(OpType::Y, {term.first});
297  break;
298  }
299  case Pauli::Z: {
300  apply_gate_at_front(OpType::Z, {term.first});
301  break;
302  }
303  }
304  }
305  return;
306  }
307 
308  // From here, half_pis == 1 or 3
309  // They act the same except for a phase flip on the product term
310  MatrixXb product_x = MatrixXb::Zero(1, qubits_.size());
311  MatrixXb product_z = MatrixXb::Zero(1, qubits_.size());
312  MatrixXb::RowXpr px = product_x.row(0);
313  MatrixXb::RowXpr pz = product_z.row(0);
314  if (pauli.coeff != 1. && pauli.coeff != -1.)
315  throw std::domain_error(
316  "Can only apply Pauli gadgets with real unit coefficients to "
317  "UnitaryTableaux");
318  bool phase = (pauli.coeff == -1.) ^ (half_pis == 1);
319 
320  // Collect the product term
321  for (const std::pair<const Qubit, Pauli>& term : pauli.string.map) {
322  unsigned uqb = qubits_.left.at(term.first);
323  switch (term.second) {
324  case Pauli::I: {
325  break;
326  }
327  case Pauli::X: {
328  tab_.row_mult(
329  tab_.xmat_.row(uqb), tab_.zmat_.row(uqb), tab_.phase_(uqb), px, pz,
330  phase, 1., px, pz, phase);
331  break;
332  }
333  case Pauli::Y: {
334  tab_.row_mult(
335  tab_.xmat_.row(uqb), tab_.zmat_.row(uqb), tab_.phase_(uqb), px, pz,
336  phase, 1., px, pz, phase);
337  tab_.row_mult(
338  tab_.xmat_.row(uqb + qubits_.size()),
339  tab_.zmat_.row(uqb + qubits_.size()),
340  tab_.phase_(uqb + qubits_.size()), px, pz, phase, 1., px, pz,
341  phase);
342  break;
343  }
344  case Pauli::Z: {
345  tab_.row_mult(
346  tab_.xmat_.row(uqb + qubits_.size()),
347  tab_.zmat_.row(uqb + qubits_.size()),
348  tab_.phase_(uqb + qubits_.size()), px, pz, phase, 1., px, pz,
349  phase);
350  break;
351  }
352  }
353  }
354 
355  // Apply the product term on the anti-commuting rows
356  for (const std::pair<const Qubit, Pauli>& term : pauli.string.map) {
357  unsigned uqb = qubits_.left.at(term.first);
358  MatrixXb::RowXpr xx = tab_.xmat_.row(uqb);
359  MatrixXb::RowXpr xz = tab_.zmat_.row(uqb);
360  MatrixXb::RowXpr zx = tab_.xmat_.row(uqb + qubits_.size());
361  MatrixXb::RowXpr zz = tab_.zmat_.row(uqb + qubits_.size());
362  switch (term.second) {
363  case Pauli::I: {
364  break;
365  }
366  case Pauli::X: {
367  tab_.row_mult(
368  px, pz, phase, zx, zz, tab_.phase_(uqb + qubits_.size()), -i_, zx,
369  zz, tab_.phase_(uqb + qubits_.size()));
370  break;
371  }
372  case Pauli::Y: {
373  tab_.row_mult(
374  px, pz, phase, zx, zz, tab_.phase_(uqb + qubits_.size()), -i_, zx,
375  zz, tab_.phase_(uqb + qubits_.size()));
376  tab_.row_mult(
377  px, pz, phase, xx, xz, tab_.phase_(uqb), -i_, xx, xz,
378  tab_.phase_(uqb));
379  break;
380  }
381  case Pauli::Z: {
382  tab_.row_mult(
383  px, pz, phase, xx, xz, tab_.phase_(uqb), -i_, xx, xz,
384  tab_.phase_(uqb));
385  break;
386  }
387  }
388  }
389 }

405 UnitaryTableau UnitaryTableau::compose(
406  const UnitaryTableau& first, const UnitaryTableau& second) {
407  std::set<Qubit> qbs = first.get_qubits();
408  for (const Qubit& q : second.get_qubits()) {
409  qbs.insert(q);
410  }
411  UnitaryTableau result = UnitaryTableau({});
412  const unsigned nqb = qbs.size();
413 
414  std::vector<QubitPauliTensor> rows;
415 
416  unsigned qir = 0;
417  for (const Qubit& qi : qbs) {
418  auto qif_it = first.qubits_.left.find(qi);
419  if (qif_it == first.qubits_.left.end()) {
420  // First acts as identity on qi, so just take effect of second
421  rows.push_back(second.get_xrow(qi));
422  } else {
423  // Sum rows of second according to entries of first
424  QubitPauliTensor fxrow = first.get_xrow(qi);
425  QubitPauliTensor rxrow = second.get_row_product(fxrow);
426  rows.push_back(rxrow);
427  }
428 
429  result.qubits_.insert({qi, qir});
430  ++qir;
431  }
432 
433  // Do the same for the Z rows
434  for (const Qubit& qi : qbs) {
435  auto qif_it = first.qubits_.left.find(qi);
436  if (qif_it == first.qubits_.left.end()) {
437  // First acts as identity on qi, so just take effect of second
438  rows.push_back(second.get_zrow(qi));
439  } else {
440  // Sum rows of second according to entries of first
441  QubitPauliTensor fzrow = first.get_zrow(qi);
442  QubitPauliTensor rzrow = second.get_row_product(fzrow);
443  rows.push_back(rzrow);
444  }
445  }
446 
447  // Combine row lists and convert to PauliStabilisers
448  PauliStabiliserList all_rows;
449  for (const QubitPauliTensor& row : rows) {
450  TKET_ASSERT(row.coeff == 1. || row.coeff == -1.);
451  std::vector<Pauli> ps(nqb, Pauli::I);
452  for (const std::pair<const Qubit, Pauli>& p : row.string.map) {
453  unsigned q = result.qubits_.left.at(p.first);
454  ps[q] = p.second;
455  }
456  all_rows.push_back(PauliStabiliser(ps, row.coeff == 1.));
457  }
458 
459  result.tab_ = SymplecticTableau(all_rows);
460 
461  return result;
462 }

464 static const std::map<
465  std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>&
466 invert_cell_map() {
467  static const auto inv_map = []() {
468  const BoolPauli I = {false, false};
469  const BoolPauli X = {true, false};
470  const BoolPauli Y = {true, true};
471  const BoolPauli Z = {false, true};
472  return std::map<
473  std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>{
474  {{I, I}, {I, I}}, {{I, X}, {I, X}}, {{I, Y}, {X, X}}, {{I, Z}, {X, I}},
475  {{X, I}, {I, Z}}, {{X, X}, {I, Y}}, {{X, Y}, {X, Y}}, {{X, Z}, {X, Z}},
476  {{Y, I}, {Z, Z}}, {{Y, X}, {Z, Y}}, {{Y, Y}, {Y, Y}}, {{Y, Z}, {Y, Z}},
477  {{Z, I}, {Z, I}}, {{Z, X}, {Z, X}}, {{Z, Y}, {Y, X}}, {{Z, Z}, {Y, I}},
478  };
479  }();
480  return inv_map;
481 }

483 UnitaryTableau UnitaryTableau::dagger() const {
484  // This method is following Craig Gidney's tableau inversion method
485  // https://algassert.com/post/2002
486  unsigned nqb = qubits_.size();
487  MatrixXb dxx = MatrixXb::Zero(nqb, nqb);
488  MatrixXb dxz = MatrixXb::Zero(nqb, nqb);
489  VectorXb dxph = VectorXb::Zero(nqb);
490  MatrixXb dzx = MatrixXb::Zero(nqb, nqb);
491  MatrixXb dzz = MatrixXb::Zero(nqb, nqb);
492  VectorXb dzph = VectorXb::Zero(nqb);
493  for (unsigned i = 0; i < nqb; ++i) {
494  for (unsigned j = 0; j < nqb; ++j) {
495  // Take effect of some input on some output and invert
496  auto inv_cell = invert_cell_map().at(
497  {BoolPauli{tab_.xmat_(i, j), tab_.zmat_(i, j)},
498  BoolPauli{tab_.xmat_(i + nqb, j), tab_.zmat_(i + nqb, j)}});
499  // Transpose tableau and fill in cell
500  dxx(j, i) = inv_cell.first.x;
501  dxz(j, i) = inv_cell.first.z;
502  dzx(j, i) = inv_cell.second.x;
503  dzz(j, i) = inv_cell.second.z;
504  }
505  }
506 
507  UnitaryTableau dag(dxx, dxz, dxph, dzx, dzz, dzph);
508  dag.qubits_ = qubits_;
509 
510  // Correct phases
511  for (unsigned i = 0; i < nqb; ++i) {
512  QubitPauliTensor xr = dag.get_xrow(qubits_.right.at(i));
513  dag.tab_.phase_(i) = get_row_product(xr).coeff == -1.;
514  QubitPauliTensor zr = dag.get_zrow(qubits_.right.at(i));
515  dag.tab_.phase_(i + nqb) = get_row_product(zr).coeff == -1.;
516  }
517 
518  return dag;
519 }

549 bool UnitaryTableau::operator==(const UnitaryTableau& other) const {
550  if (get_qubits() != other.get_qubits()) return false;
551 
552  unsigned nq = qubits_.size();
553 
554  for (unsigned i = 0; i < nq; ++i) {
555  Qubit qi = qubits_.right.at(i);
556  unsigned oi = other.qubits_.left.at(qi);
557  for (unsigned j = 0; j < nq; ++j) {
558  Qubit qj = qubits_.right.at(j);
559  unsigned oj = other.qubits_.left.at(qj);
560  if (tab_.xmat_(i, j) != other.tab_.xmat_(oi, oj)) return false;

581 void from_json(const nlohmann::json& j, UnitaryTableau& tab) {
582  j.at("tab").get_to(tab.tab_);
583  if (tab.tab_.get_n_rows() != 2 * tab.tab_.get_n_qubits())
584  throw std::invalid_argument(
585  "Size of tableau does not match requirements for UnitaryTableau.");
586  qubit_vector_t qbs = j.at("qubits").get<qubit_vector_t>();
587  unsigned nqbs = qbs.size();
588  if (nqbs != tab.tab_.get_n_qubits())
589  throw std::invalid_argument(
590  "Number of qubits in json UnitaryTableau does not match tableau size.");
591  MatrixXb expected_anticommutes(2 * nqbs, 2 * nqbs);
592  expected_anticommutes << MatrixXb::Zero(nqbs, nqbs),
593  MatrixXb::Identity(nqbs, nqbs), MatrixXb::Identity(nqbs, nqbs),
594  MatrixXb::Zero(nqbs, nqbs);
595  if (tab.tab_.anticommuting_rows() != expected_anticommutes)
596  throw std::invalid_argument(
597  "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
598  if (tab.tab_.rank() != 2 * nqbs)
599  throw std::invalid_argument(
600  "Rows of UnitaryTableau are not linearly independent");
601  tab.qubits_.clear();
602  for (unsigned i = 0; i < nqbs; ++i) {
603  tab.qubits_.insert({qbs.at(i), i});
604  }
