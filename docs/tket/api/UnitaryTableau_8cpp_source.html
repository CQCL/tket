25UnitaryTableau::UnitaryTableau(unsigned n) : tab_({}) {
26 MatrixXb xmat(2 * n, n);
27 xmat << MatrixXb::Identity(n, n), MatrixXb::Zero(n, n);
28 MatrixXb zmat(2 * n, n);
29 zmat << MatrixXb::Zero(n, n), MatrixXb::Identity(n, n);
30 tab_ = SymplecticTableau(xmat, zmat, VectorXb::Zero(2 * n));
31 qubits_ = boost::bimap<Qubit, unsigned>();
32 for (unsigned i = 0; i < n; ++i) {
33 qubits_.insert({Qubit(i), i});
34 }
35}

45UnitaryTableau::UnitaryTableau(
46 const MatrixXb& xx, const MatrixXb& xz, const VectorXb& xph,
47 const MatrixXb& zx, const MatrixXb& zz, const VectorXb& zph)
48 : tab_({}) {
49 unsigned n_qubits = xx.rows();
50 if ((xx.cols() != n_qubits) || (xz.rows() != n_qubits) ||
51 (xz.cols() != n_qubits) || (xph.size() != n_qubits) ||
52 (zx.rows() != n_qubits) || (zx.cols() != n_qubits) ||
53 (zz.rows() != n_qubits) || (zz.cols() != n_qubits) ||
54 (zph.size() != n_qubits))
55 throw std::invalid_argument(
56 "Unitary tableau requires equally-sized square matrices and vectors");
57 MatrixXb xmat(2 * n_qubits, n_qubits);
58 xmat << xx, zx;
59 MatrixXb zmat(2 * n_qubits, n_qubits);
60 zmat << xz, zz;
61 VectorXb phase(2 * n_qubits);
62 phase << xph, zph;
63 tab_ = SymplecticTableau(xmat, zmat, phase);
64 MatrixXb expected_anticommutes(2 * n_qubits, 2 * n_qubits);
65 expected_anticommutes << MatrixXb::Zero(n_qubits, n_qubits),
66 MatrixXb::Identity(n_qubits, n_qubits),
67 MatrixXb::Identity(n_qubits, n_qubits),
68 MatrixXb::Zero(n_qubits, n_qubits);
69 if (tab_.anticommuting_rows() != expected_anticommutes)
70 throw std::invalid_argument(
71 "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
72 if (tab_.rank() != 2 * n_qubits)
73 throw std::invalid_argument(
74 "Rows of UnitaryTableau are not linearly independent");
75 qubits_ = boost::bimap<Qubit, unsigned>();
76 for (unsigned i = 0; i < n_qubits; ++i) {
77 qubits_.insert({Qubit(i), i});
78 }
79}

83SpPauliStabiliser UnitaryTableau::get_xrow(const Qubit& qb) const {
84 unsigned uq = qubits_.left.at(qb);
85 PauliStabiliser stab = tab_.get_pauli(uq);
86 QubitPauliMap qpm;
87 for (unsigned i = 0; i < qubits_.size(); ++i) {
88 qpm.insert({qubits_.right.at(i), stab.get(i)});
89 }
90 return SpPauliStabiliser(qpm, stab.coeff);
91}

93SpPauliStabiliser UnitaryTableau::get_zrow(const Qubit& qb) const {
94 unsigned uqb = qubits_.left.at(qb);
95 PauliStabiliser stab = tab_.get_pauli(uqb + qubits_.size());
96 QubitPauliMap qpm;
97 for (unsigned i = 0; i < qubits_.size(); ++i) {
98 qpm.insert({qubits_.right.at(i), stab.get(i)});
99 }
100 return SpPauliStabiliser(qpm, stab.coeff);
101}

103SpPauliStabiliser UnitaryTableau::get_row_product(
104 const SpPauliStabiliser& qpt) const {
105 SpPauliStabiliser result({}, qpt.coeff);
106 for (const std::pair<const Qubit, Pauli>& p : qpt.string) {
107 auto qks_it = qubits_.left.find(p.first);
108 if (qks_it == qubits_.left.end()) {
109 // Acts as identity on p.first
110 result = result * SpPauliStabiliser(p.first, p.second);
111 } else {
112 switch (p.second) {
113 case Pauli::I: {
114 break;
115 }
116 case Pauli::X: {
117 result = result * get_xrow(p.first);
118 break;
119 }
120 case Pauli::Y: {
121 // Y = iXZ
122 result = result * get_xrow(p.first);
123 result = result * get_zrow(p.first);
124 result.coeff += 1;
125 break;
126 }
127 case Pauli::Z: {
128 result = result * get_zrow(p.first);
129 break;
130 }
131 }
132 }
133 }
134 return result;
135}

187void UnitaryTableau::apply_H_at_front(const Qubit& qb) {
188 unsigned uqb = qubits_.left.at(qb);
189 unsigned n_qubits = qubits_.size();
190 bool temp = tab_.phase(uqb);
191 tab_.phase(uqb) = tab_.phase(uqb + n_qubits);
192 tab_.phase(uqb + n_qubits) = temp;
193 tab_.xmat.row(uqb).swap(tab_.xmat.row(uqb + n_qubits));
194 tab_.zmat.row(uqb).swap(tab_.zmat.row(uqb + n_qubits));
195}

217void UnitaryTableau::apply_gate_at_front(
218 OpType type, const qubit_vector_t& qbs) {
219 switch (type) {
220 case OpType::Z: {
221 apply_Z_at_front(qbs.at(0));
222 break;
223 }
224 case OpType::X: {
225 apply_X_at_front(qbs.at(0));
226 break;
227 }
228 case OpType::Y: {
229 apply_Z_at_front(qbs.at(0));
230 apply_X_at_front(qbs.at(0));
231 break;
232 }
233 case OpType::S: {
234 apply_S_at_front(qbs.at(0));
235 break;
236 }
237 case OpType::Sdg: {
238 apply_S_at_front(qbs.at(0));
239 apply_Z_at_front(qbs.at(0));
240 break;
241 }
242 case OpType::V:
243 case OpType::SX: {
244 apply_V_at_front(qbs.at(0));
245 break;
246 }
247 case OpType::Vdg:
248 case OpType::SXdg: {
249 apply_V_at_front(qbs.at(0));
250 apply_X_at_front(qbs.at(0));
251 break;
252 }
253 case OpType::H: {
254 apply_H_at_front(qbs.at(0));
255 break;
256 }
257 case OpType::CX: {
258 apply_CX_at_front(qbs.at(0), qbs.at(1));
259 break;
260 }
261 case OpType::CY: {
262 apply_S_at_front(qbs.at(1));
263 apply_CX_at_front(qbs.at(0), qbs.at(1));
264 apply_S_at_front(qbs.at(1));
265 apply_Z_at_front(qbs.at(1));
266 break;
267 }
268 case OpType::CZ: {
269 apply_H_at_front(qbs.at(1));
270 apply_CX_at_front(qbs.at(0), qbs.at(1));
271 apply_H_at_front(qbs.at(1));
272 break;
273 }
274 case OpType::ZZMax: {
275 apply_H_at_front(qbs.at(1));
276 apply_S_at_front(qbs.at(0));
277 apply_V_at_front(qbs.at(1));
278 apply_CX_at_front(qbs.at(0), qbs.at(1));
279 apply_H_at_front(qbs.at(1));
280 break;
281 }
282 case OpType::ECR: {
283 apply_CX_at_front(qbs.at(0), qbs.at(1));
284 apply_X_at_front(qbs.at(0));
285 apply_S_at_front(qbs.at(0));
286 apply_V_at_front(qbs.at(1));
287 apply_X_at_front(qbs.at(1));
288 break;
289 }
290 case OpType::ISWAPMax: {
291 apply_V_at_front(qbs.at(0));
292 apply_V_at_front(qbs.at(1));
293 apply_CX_at_front(qbs.at(0), qbs.at(1));
294 apply_V_at_front(qbs.at(0));
295 apply_S_at_front(qbs.at(1));
296 apply_Z_at_front(qbs.at(1));
297 apply_CX_at_front(qbs.at(0), qbs.at(1));
298 apply_V_at_front(qbs.at(0));
299 apply_V_at_front(qbs.at(1));
300 break;
301 }
302 case OpType::SWAP: {
303 apply_CX_at_front(qbs.at(0), qbs.at(1));
304 apply_CX_at_front(qbs.at(1), qbs.at(0));
305 apply_CX_at_front(qbs.at(0), qbs.at(1));
306 break;
307 }
308 case OpType::BRIDGE: {
309 apply_CX_at_front(qbs.at(0), qbs.at(2));
310 break;
311 }
312 case OpType::noop:
313 case OpType::Phase: {
314 break;
315 }
316 default: {
317 throw BadOpType(
318 "Cannot be applied to a UnitaryTableau: not a Clifford gate", type);
319 }
320 }
321}

346UnitaryTableau UnitaryTableau::compose(
347 const UnitaryTableau& first, const UnitaryTableau& second) {
348 std::set<Qubit> qbs = first.get_qubits();
349 for (const Qubit& q : second.get_qubits()) {
350 qbs.insert(q);
351 }
352 UnitaryTableau result = UnitaryTableau({});
353 const unsigned nqb = qbs.size();
354
355 std::vector<SpPauliStabiliser> rows;
356
357 unsigned qir = 0;
358 for (const Qubit& qi : qbs) {
359 auto qif_it = first.qubits_.left.find(qi);
360 if (qif_it == first.qubits_.left.end()) {
361 // First acts as identity on qi, so just take effect of second
362 rows.push_back(second.get_xrow(qi));
363 } else {
364 // Sum rows of second according to entries of first
365 SpPauliStabiliser fxrow = first.get_xrow(qi);
366 SpPauliStabiliser rxrow = second.get_row_product(fxrow);
367 rows.push_back(rxrow);
368 }
369
370 result.qubits_.insert({qi, qir});
371 ++qir;
372 }
373
374 // Do the same for the Z rows
375 for (const Qubit& qi : qbs) {
376 auto qif_it = first.qubits_.left.find(qi);
377 if (qif_it == first.qubits_.left.end()) {
378 // First acts as identity on qi, so just take effect of second
379 rows.push_back(second.get_zrow(qi));
380 } else {
381 // Sum rows of second according to entries of first
382 SpPauliStabiliser fzrow = first.get_zrow(qi);
383 SpPauliStabiliser rzrow = second.get_row_product(fzrow);
384 rows.push_back(rzrow);
385 }
386 }
387
388 // Combine row lists and convert to PauliStabilisers
389 PauliStabiliserVec all_rows;
390 for (const SpPauliStabiliser& row : rows) {
391 std::vector<Pauli> ps(nqb, Pauli::I);
392 for (const std::pair<const Qubit, Pauli>& p : row.string) {
393 unsigned q = result.qubits_.left.at(p.first);
394 ps[q] = p.second;
395 }
396 all_rows.push_back(PauliStabiliser(ps, row.coeff));
397 }
398
399 result.tab_ = SymplecticTableau(all_rows);
400
401 return result;
402}

404static const std::map<
405 std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>&
406invert_cell_map() {
407 static const auto inv_map = []() {
408 const BoolPauli I = {false, false};
409 const BoolPauli X = {true, false};
410 const BoolPauli Y = {true, true};
411 const BoolPauli Z = {false, true};
412 return std::map<
413 std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>{
414 {{I, I}, {I, I}}, {{I, X}, {I, X}}, {{I, Y}, {X, X}}, {{I, Z}, {X, I}},
415 {{X, I}, {I, Z}}, {{X, X}, {I, Y}}, {{X, Y}, {X, Y}}, {{X, Z}, {X, Z}},
416 {{Y, I}, {Z, Z}}, {{Y, X}, {Z, Y}}, {{Y, Y}, {Y, Y}}, {{Y, Z}, {Y, Z}},
417 {{Z, I}, {Z, I}}, {{Z, X}, {Z, X}}, {{Z, Y}, {Y, X}}, {{Z, Z}, {Y, I}},
418 };
419 }();
420 return inv_map;
421}

423UnitaryTableau UnitaryTableau::dagger() const {
424 // This method is following Craig Gidney's tableau inversion method
425 // https://algassert.com/post/2002
426 unsigned nqb = qubits_.size();
427 MatrixXb dxx = MatrixXb::Zero(nqb, nqb);
428 MatrixXb dxz = MatrixXb::Zero(nqb, nqb);
429 VectorXb dxph = VectorXb::Zero(nqb);
430 MatrixXb dzx = MatrixXb::Zero(nqb, nqb);
431 MatrixXb dzz = MatrixXb::Zero(nqb, nqb);
432 VectorXb dzph = VectorXb::Zero(nqb);
433 for (unsigned i = 0; i < nqb; ++i) {
434 for (unsigned j = 0; j < nqb; ++j) {
435 // Take effect of some input on some output and invert
436 auto inv_cell = invert_cell_map().at(
437 {BoolPauli{tab_.xmat(i, j), tab_.zmat(i, j)},
438 BoolPauli{tab_.xmat(i + nqb, j), tab_.zmat(i + nqb, j)}});
439 // Transpose tableau and fill in cell
440 dxx(j, i) = inv_cell.first.x;
441 dxz(j, i) = inv_cell.first.z;
442 dzx(j, i) = inv_cell.second.x;
443 dzz(j, i) = inv_cell.second.z;
444 }
445 }
446
447 UnitaryTableau dag(dxx, dxz, dxph, dzx, dzz, dzph);
448 dag.qubits_ = qubits_;
449
450 // Correct phases
451 for (unsigned i = 0; i < nqb; ++i) {
452 SpPauliStabiliser xr = dag.get_xrow(qubits_.right.at(i));
453 dag.tab_.phase(i) = get_row_product(xr).is_real_negative();
454 SpPauliStabiliser zr = dag.get_zrow(qubits_.right.at(i));
455 dag.tab_.phase(i + nqb) = get_row_product(zr).is_real_negative();
456 }
457
458 return dag;
459}

484 << std::endl;
485 }
486 return os;

489bool UnitaryTableau::operator==(const UnitaryTableau& other) const {
490 if (get_qubits() != other.get_qubits()) return false;
491
492 unsigned nq = qubits_.size();
493
494 for (unsigned i = 0; i < nq; ++i) {
495 Qubit qi = qubits_.right.at(i);
496 unsigned oi = other.qubits_.left.at(qi);
497 for (unsigned j = 0; j < nq; ++j) {
498 Qubit qj = qubits_.right.at(j);
499 unsigned oj = other.qubits_.left.at(qj);
500 if (tab_.xmat(i, j) != other.tab_.xmat(oi, oj)) return false;
501 if (tab_.zmat(i, j) != other.tab_.zmat(oi, oj)) return false;
502 if (tab_.xmat(i + nq, j) != other.tab_.xmat(oi + nq, oj)) return false;
503 if (tab_.zmat(i + nq, j) != other.tab_.zmat(oi + nq, oj)) return false;
504 }
505 if (tab_.phase(i) != other.tab_.phase(oi)) return false;
506 if (tab_.phase(i + nq) != other.tab_.phase(oi + nq)) return false;
507 }
508
509 return true;
510}

521void from_json(const nlohmann::json& j, UnitaryTableau& tab) {
522 j.at("tab").get_to(tab.tab_);
523 if (tab.tab_.get_n_rows() != 2 * tab.tab_.get_n_qubits())
524 throw std::invalid_argument(
525 "Size of tableau does not match requirements for UnitaryTableau.");
526 qubit_vector_t qbs = j.at("qubits").get<qubit_vector_t>();
527 unsigned nqbs = qbs.size();
528 if (nqbs != tab.tab_.get_n_qubits())
529 throw std::invalid_argument(
530 "Number of qubits in json UnitaryTableau does not match tableau size.");
531 MatrixXb expected_anticommutes(2 * nqbs, 2 * nqbs);
532 expected_anticommutes << MatrixXb::Zero(nqbs, nqbs),
533 MatrixXb::Identity(nqbs, nqbs), MatrixXb::Identity(nqbs, nqbs),
534 MatrixXb::Zero(nqbs, nqbs);
535 if (tab.tab_.anticommuting_rows() != expected_anticommutes)
536 throw std::invalid_argument(
537 "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
538 if (tab.tab_.rank() != 2 * nqbs)
539 throw std::invalid_argument(
540 "Rows of UnitaryTableau are not linearly independent");
541 tab.qubits_.clear();
542 for (unsigned i = 0; i < nqbs; ++i) {
543 tab.qubits_.insert({qbs.at(i), i});
544 }
545}
