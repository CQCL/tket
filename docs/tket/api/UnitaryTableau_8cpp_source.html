26 UnitaryTableau::UnitaryTableau(unsigned n) : tab_({}) {
27  MatrixXb xmat(2 * n, n);
28  xmat << MatrixXb::Identity(n, n), MatrixXb::Zero(n, n);
29  MatrixXb zmat(2 * n, n);
30  zmat << MatrixXb::Zero(n, n), MatrixXb::Identity(n, n);
31  tab_ = SymplecticTableau(xmat, zmat, VectorXb::Zero(2 * n));
32  qubits_ = boost::bimap<Qubit, unsigned>();
33  for (unsigned i = 0; i < n; ++i) {
34  qubits_.insert({Qubit(i), i});
35  }
36 }

46 UnitaryTableau::UnitaryTableau(
47  const MatrixXb& xx, const MatrixXb& xz, const VectorXb& xph,
48  const MatrixXb& zx, const MatrixXb& zz, const VectorXb& zph)
49  : tab_({}) {
50  unsigned n_qubits = xx.rows();
51  if ((xx.cols() != n_qubits) || (xz.rows() != n_qubits) ||
52  (xz.cols() != n_qubits) || (xph.size() != n_qubits) ||
53  (zx.rows() != n_qubits) || (zx.cols() != n_qubits) ||
54  (zz.rows() != n_qubits) || (zz.cols() != n_qubits) ||
55  (zph.size() != n_qubits))
56  throw std::invalid_argument(
57  "Unitary tableau requires equally-sized square matrices and vectors");
58  MatrixXb xmat(2 * n_qubits, n_qubits);
59  xmat << xx, zx;
60  MatrixXb zmat(2 * n_qubits, n_qubits);
61  zmat << xz, zz;
62  VectorXb phase(2 * n_qubits);
63  phase << xph, zph;
64  tab_ = SymplecticTableau(xmat, zmat, phase);
65  MatrixXb expected_anticommutes(2 * n_qubits, 2 * n_qubits);
66  expected_anticommutes << MatrixXb::Zero(n_qubits, n_qubits),
67  MatrixXb::Identity(n_qubits, n_qubits),
68  MatrixXb::Identity(n_qubits, n_qubits),
69  MatrixXb::Zero(n_qubits, n_qubits);
70  if (tab_.anticommuting_rows() != expected_anticommutes)
71  throw std::invalid_argument(
72  "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
73  if (tab_.rank() != 2 * n_qubits)
74  throw std::invalid_argument(
75  "Rows of UnitaryTableau are not linearly independent");
76  qubits_ = boost::bimap<Qubit, unsigned>();
77  for (unsigned i = 0; i < n_qubits; ++i) {
78  qubits_.insert({Qubit(i), i});
79  }
80 }

84 SpPauliStabiliser UnitaryTableau::get_xrow(const Qubit& qb) const {
85  unsigned uq = qubits_.left.at(qb);
86  PauliStabiliser stab = tab_.get_pauli(uq);
87  QubitPauliMap qpm;
88  for (unsigned i = 0; i < qubits_.size(); ++i) {
89  qpm.insert({qubits_.right.at(i), stab.get(i)});
90  }
91  return SpPauliStabiliser(qpm, stab.coeff);
92 }

94 SpPauliStabiliser UnitaryTableau::get_zrow(const Qubit& qb) const {
95  unsigned uqb = qubits_.left.at(qb);
96  PauliStabiliser stab = tab_.get_pauli(uqb + qubits_.size());
97  QubitPauliMap qpm;
98  for (unsigned i = 0; i < qubits_.size(); ++i) {
99  qpm.insert({qubits_.right.at(i), stab.get(i)});

104 SpPauliStabiliser UnitaryTableau::get_row_product(
105  const SpPauliStabiliser& qpt) const {
106  SpPauliStabiliser result({}, qpt.coeff);
107  for (const std::pair<const Qubit, Pauli>& p : qpt.string) {
108  auto qks_it = qubits_.left.find(p.first);
109  if (qks_it == qubits_.left.end()) {
110  // Acts as identity on p.first
111  result = result * SpPauliStabiliser(p.first, p.second);
112  } else {
113  switch (p.second) {
114  case Pauli::I: {
115  break;
116  }
117  case Pauli::X: {
118  result = result * get_xrow(p.first);
119  break;
120  }
121  case Pauli::Y: {
122  // Y = iXZ
123  result = result * get_xrow(p.first);
124  result = result * get_zrow(p.first);
125  result.coeff += 1;
126  break;
127  }
128  case Pauli::Z: {
129  result = result * get_zrow(p.first);
130  break;
131  }
132  }
133  }
134  }
135  return result;
136 }

183 void UnitaryTableau::apply_gate_at_front(
184  OpType type, const qubit_vector_t& qbs) {
185  switch (type) {
186  case OpType::Z: {
187  apply_S_at_front(qbs.at(0));
188  apply_S_at_front(qbs.at(0));
189  break;
190  }
191  case OpType::X: {
192  apply_V_at_front(qbs.at(0));
193  apply_V_at_front(qbs.at(0));
194  break;
195  }
196  case OpType::Y: {
197  apply_S_at_front(qbs.at(0));
198  apply_S_at_front(qbs.at(0));
199  apply_V_at_front(qbs.at(0));
200  apply_V_at_front(qbs.at(0));
201  break;
202  }
203  case OpType::S: {
204  apply_S_at_front(qbs.at(0));
205  break;
206  }
207  case OpType::Sdg: {
208  apply_S_at_front(qbs.at(0));
209  apply_S_at_front(qbs.at(0));
210  apply_S_at_front(qbs.at(0));
211  break;
212  }
213  case OpType::V: {
214  apply_V_at_front(qbs.at(0));
215  break;
216  }
217  case OpType::Vdg: {
218  apply_V_at_front(qbs.at(0));
219  apply_V_at_front(qbs.at(0));
220  apply_V_at_front(qbs.at(0));
221  break;
222  }
223  case OpType::H: {
224  apply_S_at_front(qbs.at(0));
225  apply_V_at_front(qbs.at(0));
226  apply_S_at_front(qbs.at(0));
227  break;
228  }
229  case OpType::CX: {
230  apply_CX_at_front(qbs.at(0), qbs.at(1));
231  break;
232  }
233  case OpType::CY: {
234  apply_S_at_front(qbs.at(1));
235  apply_CX_at_front(qbs.at(0), qbs.at(1));
236  apply_S_at_front(qbs.at(1));
237  apply_S_at_front(qbs.at(1));
238  apply_S_at_front(qbs.at(1));
239  break;
240  }
241  case OpType::CZ: {
242  apply_S_at_front(qbs.at(1));
243  apply_V_at_front(qbs.at(1));
244  apply_S_at_front(qbs.at(1));
245  apply_CX_at_front(qbs.at(0), qbs.at(1));
246  apply_S_at_front(qbs.at(1));
247  apply_V_at_front(qbs.at(1));
248  apply_S_at_front(qbs.at(1));
249  break;
250  }
251  case OpType::SWAP: {
252  apply_CX_at_front(qbs.at(0), qbs.at(1));
253  apply_CX_at_front(qbs.at(1), qbs.at(0));
254  apply_CX_at_front(qbs.at(0), qbs.at(1));
255  break;
256  }
257  case OpType::BRIDGE: {
258  apply_CX_at_front(qbs.at(0), qbs.at(2));
259  break;
260  }
261  case OpType::noop:
262  case OpType::Phase: {
263  break;
264  }
265  default: {
266  throw BadOpType(
267  "Cannot be applied to a UnitaryTableau: not a Clifford gate", type);
268  }
269  }
270 }

295 UnitaryTableau UnitaryTableau::compose(
296  const UnitaryTableau& first, const UnitaryTableau& second) {
297  std::set<Qubit> qbs = first.get_qubits();
298  for (const Qubit& q : second.get_qubits()) {
299  qbs.insert(q);
300  }
301  UnitaryTableau result = UnitaryTableau({});
302  const unsigned nqb = qbs.size();
303 
304  std::vector<SpPauliStabiliser> rows;
305 
306  unsigned qir = 0;
307  for (const Qubit& qi : qbs) {
308  auto qif_it = first.qubits_.left.find(qi);
309  if (qif_it == first.qubits_.left.end()) {
310  // First acts as identity on qi, so just take effect of second
311  rows.push_back(second.get_xrow(qi));
312  } else {
313  // Sum rows of second according to entries of first
314  SpPauliStabiliser fxrow = first.get_xrow(qi);
315  SpPauliStabiliser rxrow = second.get_row_product(fxrow);
316  rows.push_back(rxrow);
317  }
318 
319  result.qubits_.insert({qi, qir});
320  ++qir;
321  }
322 
323  // Do the same for the Z rows
324  for (const Qubit& qi : qbs) {
325  auto qif_it = first.qubits_.left.find(qi);
326  if (qif_it == first.qubits_.left.end()) {
327  // First acts as identity on qi, so just take effect of second
328  rows.push_back(second.get_zrow(qi));
329  } else {
330  // Sum rows of second according to entries of first
331  SpPauliStabiliser fzrow = first.get_zrow(qi);
332  SpPauliStabiliser rzrow = second.get_row_product(fzrow);
333  rows.push_back(rzrow);
334  }
335  }
336 
337  // Combine row lists and convert to PauliStabilisers
338  PauliStabiliserVec all_rows;
339  for (const SpPauliStabiliser& row : rows) {
340  std::vector<Pauli> ps(nqb, Pauli::I);
341  for (const std::pair<const Qubit, Pauli>& p : row.string) {
342  unsigned q = result.qubits_.left.at(p.first);
343  ps[q] = p.second;
344  }
345  all_rows.push_back(PauliStabiliser(ps, row.coeff));
346  }
347 
348  result.tab_ = SymplecticTableau(all_rows);
349 
350  return result;
351 }

353 static const std::map<
354  std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>&
355 invert_cell_map() {
356  static const auto inv_map = []() {
357  const BoolPauli I = {false, false};
358  const BoolPauli X = {true, false};
359  const BoolPauli Y = {true, true};
360  const BoolPauli Z = {false, true};
361  return std::map<
362  std::pair<BoolPauli, BoolPauli>, std::pair<BoolPauli, BoolPauli>>{
363  {{I, I}, {I, I}}, {{I, X}, {I, X}}, {{I, Y}, {X, X}}, {{I, Z}, {X, I}},
364  {{X, I}, {I, Z}}, {{X, X}, {I, Y}}, {{X, Y}, {X, Y}}, {{X, Z}, {X, Z}},
365  {{Y, I}, {Z, Z}}, {{Y, X}, {Z, Y}}, {{Y, Y}, {Y, Y}}, {{Y, Z}, {Y, Z}},
366  {{Z, I}, {Z, I}}, {{Z, X}, {Z, X}}, {{Z, Y}, {Y, X}}, {{Z, Z}, {Y, I}},
367  };
368  }();
369  return inv_map;
370 }

372 UnitaryTableau UnitaryTableau::dagger() const {
373  // This method is following Craig Gidney's tableau inversion method
374  // https://algassert.com/post/2002
375  unsigned nqb = qubits_.size();
376  MatrixXb dxx = MatrixXb::Zero(nqb, nqb);
377  MatrixXb dxz = MatrixXb::Zero(nqb, nqb);
378  VectorXb dxph = VectorXb::Zero(nqb);
379  MatrixXb dzx = MatrixXb::Zero(nqb, nqb);
380  MatrixXb dzz = MatrixXb::Zero(nqb, nqb);
381  VectorXb dzph = VectorXb::Zero(nqb);
382  for (unsigned i = 0; i < nqb; ++i) {
383  for (unsigned j = 0; j < nqb; ++j) {
384  // Take effect of some input on some output and invert
385  auto inv_cell = invert_cell_map().at(
386  {BoolPauli{tab_.xmat_(i, j), tab_.zmat_(i, j)},
387  BoolPauli{tab_.xmat_(i + nqb, j), tab_.zmat_(i + nqb, j)}});
388  // Transpose tableau and fill in cell
389  dxx(j, i) = inv_cell.first.x;
390  dxz(j, i) = inv_cell.first.z;
391  dzx(j, i) = inv_cell.second.x;
392  dzz(j, i) = inv_cell.second.z;
393  }
394  }
395 
396  UnitaryTableau dag(dxx, dxz, dxph, dzx, dzz, dzph);
397  dag.qubits_ = qubits_;
398 
399  // Correct phases
400  for (unsigned i = 0; i < nqb; ++i) {
401  SpPauliStabiliser xr = dag.get_xrow(qubits_.right.at(i));
402  dag.tab_.phase_(i) = get_row_product(xr).is_real_negative();
403  SpPauliStabiliser zr = dag.get_zrow(qubits_.right.at(i));
404  dag.tab_.phase_(i + nqb) = get_row_product(zr).is_real_negative();
405  }
406 
407  return dag;
408 }

438 bool UnitaryTableau::operator==(const UnitaryTableau& other) const {
439  if (get_qubits() != other.get_qubits()) return false;
440 
441  unsigned nq = qubits_.size();
442 
443  for (unsigned i = 0; i < nq; ++i) {
444  Qubit qi = qubits_.right.at(i);
445  unsigned oi = other.qubits_.left.at(qi);
446  for (unsigned j = 0; j < nq; ++j) {
447  Qubit qj = qubits_.right.at(j);
448  unsigned oj = other.qubits_.left.at(qj);
449  if (tab_.xmat_(i, j) != other.tab_.xmat_(oi, oj)) return false;
450  if (tab_.zmat_(i, j) != other.tab_.zmat_(oi, oj)) return false;
451  if (tab_.xmat_(i + nq, j) != other.tab_.xmat_(oi + nq, oj)) return false;
452  if (tab_.zmat_(i + nq, j) != other.tab_.zmat_(oi + nq, oj)) return false;
453  }
454  if (tab_.phase_(i) != other.tab_.phase_(oi)) return false;
455  if (tab_.phase_(i + nq) != other.tab_.phase_(oi + nq)) return false;
456  }
457 
458  return true;
459 }

470 void from_json(const nlohmann::json& j, UnitaryTableau& tab) {
471  j.at("tab").get_to(tab.tab_);
472  if (tab.tab_.get_n_rows() != 2 * tab.tab_.get_n_qubits())
473  throw std::invalid_argument(
474  "Size of tableau does not match requirements for UnitaryTableau.");
475  qubit_vector_t qbs = j.at("qubits").get<qubit_vector_t>();
476  unsigned nqbs = qbs.size();
477  if (nqbs != tab.tab_.get_n_qubits())
478  throw std::invalid_argument(
479  "Number of qubits in json UnitaryTableau does not match tableau size.");
480  MatrixXb expected_anticommutes(2 * nqbs, 2 * nqbs);
481  expected_anticommutes << MatrixXb::Zero(nqbs, nqbs),
482  MatrixXb::Identity(nqbs, nqbs), MatrixXb::Identity(nqbs, nqbs),
483  MatrixXb::Zero(nqbs, nqbs);
484  if (tab.tab_.anticommuting_rows() != expected_anticommutes)
485  throw std::invalid_argument(
486  "Rows of tableau do not (anti-)commute as expected for UnitaryTableau");
487  if (tab.tab_.rank() != 2 * nqbs)
488  throw std::invalid_argument(
489  "Rows of UnitaryTableau are not linearly independent");
490  tab.qubits_.clear();
491  for (unsigned i = 0; i < nqbs; ++i) {
492  tab.qubits_.insert({qbs.at(i), i});
493  }
494 }
