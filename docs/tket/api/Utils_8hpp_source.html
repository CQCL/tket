47/* These utility functions are meant to offer a more consistent
48 * alternative to boost::remove_vertex, so that the vertex storage container
49 * can be changed without changes in the implementation
50 *
51 * boost::remove_vertex behaves differently depending on the vertex container
52 * type
53 * - for vecS, remove_vertex will shift vertex indices so that they remain in
54 * the continuous interval [0..n-1].
55 * - for other types, remove_vertex will leave the vertex descriptors unchanged
56 *
57 * Instead, these wrappers behave as follows
58 * - support for explicit indices: if vertex descriptors are numbers
59 * (i.e. VertexList == vecS) or if an explicit index map is provided, indices
60 * will be shifted as with the default boost vecS implementation.
61 * - support for external map: additionally, a map from vertex descriptors to
62 * arbitrary types can be provided. The map keys will then automatically be
63 * changed to mirror the vertex descriptors changes.
64 *
65 * As a result, implementations can abstract away the vertex descriptor
66 * implementation, and either rely on explicit indices throughout, or stick to
67 * vertex descriptors and have the necessary translation maps be updated
68 * automatically
69 */
70template <typename Graph>
71inline void remove_vertex(vertex<Graph> v, Graph& graph) {
72 detail::graph_utils_impl<Graph> impl(graph);
73 impl.remove_vertex(v);
74}
75template <typename Graph, typename PMap>
76inline void remove_vertex(vertex<Graph> v, Graph& graph, PMap& p) {
77 detail::graph_utils_impl<Graph, PMap> impl(graph, p);
78 impl.remove_vertex(v);
79}
80template <typename Graph, typename PMap, typename Map>
81inline void remove_vertex(vertex<Graph> v, Graph& graph, PMap& pmap, Map& map) {
82 detail::graph_utils_impl_with_map<Graph, Map, PMap> impl(graph, map, pmap);
83 impl.remove_vertex(v);
84}

86inline void remove_vertex_with_map(

131// simple wrapper around boost::copy_graph that serves
132// to transform a directed into an undirected graph
133template <typename GraphOut, typename GraphIn>
134GraphOut symmetrise(const GraphIn& g) {
135 static_assert(
136 !is_directed<GraphOut>,
137 "The GraphOut template type should be undirected. Use "
138 "boost::copy_graph otherwise");
139 GraphOut out;
140 boost::copy_graph(g, out);
141 return out;
142}
143
144template <typename GraphOut, typename GraphIn>
145GraphOut symmetrise(
146 const GraphIn& g, boost::bimap<vertex<GraphIn>, vertex<GraphOut>>& v_map) {
147 static_assert(
148 !is_directed<GraphOut>,
149 "The GraphOut template type should be undirected. Use "
150 "boost::copy_graph otherwise");
151 GraphOut out;
152 // when vertices are copied, do the default thing but keep track of the map
153 boost::detail::vertex_copier<GraphIn, GraphOut> v_copier =
154 boost::detail::make_vertex_copier(g, out);
155 boost::copy_graph(
156 g, out,
157 boost::vertex_copy(
158 [&v_map, &v_copier](vertex<GraphIn> old_v, vertex<GraphOut> new_v) {
159 v_copier(old_v, new_v);
160 v_map.insert({old_v, new_v});
161 }));
162 return out;
163}

165template <typename Graph>
166std::size_t max_degree(const Graph& g) {
167 auto degree = [&g](vertex<Graph> v) { return boost::degree(v, g); };
168 auto [from, to] = boost::vertices(g);
169 vertex<Graph> max_vertex =
170 *std::max_element(from, to, detail::lt_with_key<vertex<Graph>>(degree));
171 return degree(max_vertex);
172}
173
174template <typename Graph>
175std::size_t min_degree(const Graph& g) {
176 auto degree = [&g](vertex<Graph> v) { return boost::degree(v, g); };
177 auto [from, to] = boost::vertices(g);
178 vertex<Graph> min_vertex =
179 *std::min_element(from, to, detail::lt_with_key<vertex<Graph>>(degree));
180 return degree(min_vertex);
181}

183template <typename Graph>
184std::set<vertex<Graph>> max_degree_nodes(const Graph& g) {
185 const auto max = max_degree(g);
186 auto [from, to] = boost::vertices(g);
187 std::set<vertex<Graph>> out;
188 std::copy_if(
189 from, to, std::inserter(out, out.begin()),
190 [&g, max](vertex<Graph> v) { return boost::degree(v, g) == max; });
191 return out;
192}
193
194template <typename Graph>
195std::set<vertex<Graph>> min_degree_nodes(const Graph& g) {
196 const auto min = min_degree(g);
197 auto [from, to] = boost::vertices(g);
198 std::set<vertex<Graph>> out;
199 std::copy_if(
200 from, to, std::inserter(out, out.begin()),
201 [&g, min](vertex<Graph> v) { return boost::degree(v, g) == min; });
202 return out;
203}
