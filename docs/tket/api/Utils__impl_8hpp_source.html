68/* Compile-time boolean used in template specialisations
69 * and graph_utils_helper_with_map to infer whether the Graph type supports
70 * integer indexing
71 *
72 * has_integral_index<Graph, PMap> is true iff
73 * either the vertex_descriptor of Graph are integers (i.e. VertexList = vecS)
74 * or an explicit property map PMap is provided with integral values
75 *
76 * Template argument Graph is the boost type of the graph to be checked
77 * PMap is the optional property map to provide integral index values
78 * The third argument should always be left to default, used for type inference
79 * ("SFINAE")
80 */
81template <typename Graph, typename PMap = void, typename = void>
82struct has_integral_index_struct {
83 static const bool value = false;
84};
85template <typename Graph>
86struct has_integral_index_struct<
87 Graph, void, std::enable_if_t<std::is_integral_v<vertex<Graph>>>> {
88 static const bool value = true;
89};
90template <typename Graph, typename PMap>
91struct has_integral_index_struct<
92 Graph, PMap,
93 std::enable_if_t<std::is_integral_v<typename PMap::mapped_type>>> {
94 static const bool value = true;
95};
96template <typename Graph, typename PMap = void>
97constexpr bool has_integral_index =
98 has_integral_index_struct<Graph, PMap>::value;

106/* This class, along with their inherited classes and template specialisations
107 * below implement the graphs::utils helper functions, defined further below.
108 *
109 * CLASS HIERARCHY
110 * ---------------
111 * +---------------------+ +---------------------------+
112 * Base classes: | graph_utils_base | | graph_utils_base_with_ind |
113 * +---------------------+ +---------------------------+
114 * | | | |
115 * v v v v
116 * Container-specific +---------+ +---------+ +---------+ +---------+
117 * classes | _helper | | _helper | | _helper | | _helper |
118 * (pmap / indexable) | YES/NO | | NO/NO | | YES/YES | | NO/YES |
119 * +---------+ +---------+ +---------+ +---------+
120 * | | | | | | | |
121 * | | | | +-------+ | | | |
122 * Child class to be used | | | +->| _impl |<--+ | | |
123 * in utils | +------------>| |<--------------+ |
124 * | | +-------+ | |
125 * v v v v
126 * Container-specific +---------+ +---------+ +---------+ +---------+
127 * classes with added | _helper | | _helper | | _helper | | _helper |
128 * support for maps |_with_map| |_with_map| |_with_map| |_with_map|
129 * (pmap / indexable) | YES/NO | | NO/NO | | YES/YES | | NO/YES |
130 * +---------+ +---------+ +---------+ +---------+
131 * | | | |
132 * | | +---------+ | |
133 * Child class to be used | +->| _impl |<----+ |
134 * in utils with map support +------------>|_with_map|<---------------+
135 * +---------+
136 *
137 * graph_utils_base and graph_utils_base_with_ind implement the functionality
138 * that is shared among all vertex container types (i.e. most of the code logic)
139 *
140 * Impl must be a subclass of this and must implement
141 * - on_remove_vertex: event handler when vertex is removed
142 */
143template <typename Graph, typename Impl>
144class graph_utils_base {
145 public:
146 explicit graph_utils_base(Graph& g) : graph(g) {}
147
148 void remove_vertex(vertex<Graph> v) {
149 remove_vertex_handler(v);
150 boost::remove_vertex(v, graph);
151 }
152
153 void remove_edge(edge<Graph> e, bool remove_unused_vertices = false) {
154 auto [u, v] = get_vertex_pair(e);
155 boost::remove_edge(e, graph);
156
157 if (remove_unused_vertices) {
158 if (boost::degree(u, graph) == 0) {
159 remove_vertex(u);
160 }
161 if (boost::degree(v, graph) == 0) {
162 remove_vertex(v);
163 }
164 }
165 }
166
167 protected:
168 virtual void remove_vertex_handler(vertex<Graph> v) {
169 Impl* impl = static_cast<Impl*>(this);
170 impl->on_remove_vertex(v);
171 }
172
173 /* returns source and target in an order in which they can be
174 * safely deleted if necessary
175 */
176 std::pair<vertex<Graph>, vertex<Graph>> get_vertex_pair(edge<Graph> e) {
177 vertex<Graph> u = boost::source(e, graph);
178 vertex<Graph> v = boost::target(e, graph);
179
180 return in_delete_order(u, v);
181 }
182
183 virtual std::pair<vertex<Graph>, vertex<Graph>> in_delete_order(
184 vertex<Graph> u, vertex<Graph> v) {
185 return {u, v};
186 }
187
188 Graph& graph;
189};

191/* small adjustments to graph_utils_helper for integral vertex indices.
192 * Unlike the general case, here indices get shifted when vertices are deleted
193 * so that the vertex indices remain contiguous
194 *
195 * Impl must be a subclass of this and must implement
196 * - to_index: return index of vertex v
197 * - typename index_type: type of index returned by to_index
198 * - on_index_change: event handler when indices are shifted
199 * - on_remove_vertex: event handler when vertex is removed, after all
200 * indices were updated
201 */
202template <typename Graph, typename Impl>
203class graph_utils_base_with_ind : public graph_utils_base<Graph, Impl> {
204 private:
205 using Base = graph_utils_base<Graph, Impl>;
206
207 public:
208 explicit graph_utils_base_with_ind(Graph& g) : Base(g) {}
209
210 protected:
211 using Base::graph;
212 void remove_vertex_handler(vertex<Graph> v) override {
213 Impl* impl = static_cast<Impl*>(this);
214 using index_type = typename Impl::index_type;
215
216 for (vertex<Graph> u : boost::make_iterator_range(boost::vertices(graph))) {
217 if (impl->to_index(u) > impl->to_index(v)) {
218 index_type new_ind = impl->to_index(u) - 1;
219 impl->on_index_change(u, new_ind);
220 }
221 }
222 impl->on_remove_vertex(v);
223 }
224
225 std::pair<vertex<Graph>, vertex<Graph>> in_delete_order(
226 vertex<Graph> u, vertex<Graph> v) override {
227 Impl* impl = static_cast<Impl*>(this);
228
229 // return larger index first
230 if (impl->to_index(u) < impl->to_index(v)) {
231 return {v, u};
232 } else {
233 return {u, v};
234 }
235 }
236};

238/* The classes graph_utils_helper and graph_utils_helper_with_map provide
239 * the container-specific implementation bits.
240 * The Derived template type is the child class that inherits from this class
241 * -> see Curiously Recurring Template Pattern (CRTP)
242 * The Indexable boolean template is true iff Graph support integer vertex
243 * indexing (see has_integral_index). It should be left to default and infered
244 * by compiler
245 *
246 * We distinguish four cases for template specialisations, given by the four
247 * combinations of two variables:
248 *
249 * a) whether an explicit property map is provided or not
250 * b) whether the graph is indexable, i.e. either it has an intrinsic
251 * vertex_index property (case VertexList == vecS) or the property map has
252 * integral values
253 *
254 * Case 1: explicit property map, not indexable
255 */
256template <
257 typename Derived, typename Graph, typename PMap = void,
258 bool Indexable = has_integral_index<Graph, PMap>>
259class graph_utils_helper : public graph_utils_base<Graph, Derived> {
260 private:
261 using Base = graph_utils_base<Graph, Derived>;
262
263 public:
264 graph_utils_helper(Graph& g, PMap& p) : Base(g), pmap(p) {}
265
266 void on_remove_vertex(vertex<Graph> v) { pmap.erase(v); }
267
268 private:
269 PMap& pmap;
270};
271
272// Case 2: no explicit property, not indexable
273template <typename Derived, typename Graph>
274class graph_utils_helper<Derived, Graph, void, false>
275 : public graph_utils_base<Graph, Derived> {
276 private:
277 using Base = graph_utils_base<Graph, Derived>;
278
279 public:
280 explicit graph_utils_helper(Graph& g) : Base(g) {}
281 void on_remove_vertex(vertex<Graph>) {}
282};

284// Case 3: explicit property, indexable
285// in this case we must ensure that we update the property map
286// when vertices are removed and indices get shifted
287template <typename Derived, typename Graph, typename PMap>
288class graph_utils_helper<Derived, Graph, PMap, true>
289 : public graph_utils_base_with_ind<Graph, Derived> {
290 private:
291 using Base = graph_utils_base_with_ind<Graph, Derived>;
292
293 public:
294 using index_type = typename PMap::mapped_type;
295 graph_utils_helper(Graph& g, PMap& p) : Base(g), pmap(p), new_pmap(p) {
296 static_assert(
297 !std::is_integral_v<vertex<Graph>>,
298 "Ambiguous vertex index: you cannot provide an explicit "
299 "index property map if vertex descriptors are integrals.");
300 }
301
302 index_type to_index(vertex<Graph> v) { return pmap[v]; }
303
304 void on_index_change(vertex<Graph> v, index_type new_ind) {
305 new_pmap[v] = new_ind;
306 }
307 void on_remove_vertex(vertex<Graph> v) {
308 new_pmap.erase(v);
309 pmap = new_pmap;
310 }
311
312 private:
313 PMap& pmap;
314 PMap new_pmap;
315};

317// Case 4: no explicit property, indexable
318// boost automatically shifts indices, so there is not much to do
319template <typename Derived, typename Graph>
320class graph_utils_helper<Derived, Graph, void, true>
321 : public graph_utils_base_with_ind<Graph, Derived> {
322 private:
323 using Base = graph_utils_base_with_ind<Graph, Derived>;
324
325 public:
326 using index_type = vertex_index<Graph>;
327 explicit graph_utils_helper(Graph& g) : Base(g) {}
328
329 index_type to_index(vertex<Graph> v) { return v; }
330 void on_index_change(vertex<Graph>, index_type) {}
331 void on_remove_vertex(vertex<Graph>) {}
332};

358// Case 3: explicit property, indexable
359template <typename Derived, typename Graph, typename Map, typename PMap>
360class graph_utils_helper_with_map<Derived, Graph, Map, PMap, true>
361 : public graph_utils_helper<Derived, Graph, PMap> {
362 private:
363 using Base = graph_utils_helper<Derived, Graph, PMap>;
364
365 public:
366 using index_type = typename Base::index_type;
367 graph_utils_helper_with_map(Graph& g, Map& m, PMap& p)
368 : Base(g, p), map(m), new_map(m) {}
369
370 void on_index_change(vertex<Graph> v, index_type new_ind) {
371 if (new_map.count(new_ind)) {
372 new_map.erase(new_ind);
373 }
374 new_map.insert({new_ind, new_map[Base::to_index(v)]});
375 Base::on_index_change(v, new_ind);
376 }
377 void on_remove_vertex(vertex<Graph> v) {
378 auto max_ind = std::max_element(std::begin(new_map), std::end(new_map));
379 new_map.erase(max_ind);
380 map.clear();
381 for (auto& p : new_map) {
382 map.insert(p);
383 }
384
385 Base::on_remove_vertex(v);
386 }
387
388 private:
389 Map& map;
390 Map new_map;
391};

393// Case 4: no explicit property, indexable
394template <typename Derived, typename Graph, typename Map>
395class graph_utils_helper_with_map<Derived, Graph, Map, void, true>
396 : public graph_utils_helper<Derived, Graph, void> {
397 private:
398 using Base = graph_utils_helper<Derived, Graph, void>;
399 using StdMap = std::map<typename Map::key_type, typename Map::mapped_type>;
400
401 static typename StdMap::value_type transformer_it(
402 typename Map::value_type& elem) {
403 return std::make_pair(elem.first, elem.second);
404 }
405 auto map_begin_it(Map& m) {
406 return boost::make_transform_iterator(m.begin(), &transformer_it);
407 }
408 auto map_end_it(Map& m) {
409 return boost::make_transform_iterator(m.end(), &transformer_it);
410 }
411
412 public:
413 using index_type = typename Base::index_type;
414 graph_utils_helper_with_map(Graph& g, Map& m)
415 : Base(g), map(m), new_map(map_begin_it(m), map_end_it(m)) {}
416 void on_index_change(vertex<Graph> v, index_type new_ind) {
417 if (new_map.count(new_ind)) {
418 new_map.erase(new_ind);
419 }
420 new_map.insert({new_ind, new_map[Base::to_index(v)]});
421 Base::on_index_change(v, new_ind);
422 }
423 void on_remove_vertex(vertex<Graph> v) {
424 auto max_ind = std::max_element(std::begin(new_map), std::end(new_map));
425 new_map.erase(max_ind);
426 map.clear();
427 for (auto& p : new_map) {
428 map.insert(p);
429 }
430
431 Base::on_remove_vertex(v);
432 }
433
434 protected:
435 Map& map;
436 StdMap new_map;
437};
