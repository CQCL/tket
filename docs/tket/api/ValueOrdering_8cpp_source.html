26 /*
27 We are using "solution biased searching" as our heuristic. See the paper
28 "Sequential and Parallel Solution-Biased Search for Subgraph Algorithms":
29 
30 https://link.springer.com/chapter/10.1007/978-3-030-19212-9_2
31 
32 A free version available from a co-author's website:
33 
34 http://blairarchibald.co.uk/resources/publications/cpaior19.pdf
35 
36 The idea is, rather than always mapping into the target vertex with largest
37 possible degree, we should try lower degrees occasionally.
38 This is done simply by choosing a random vertex,
39 but letting the probability of choosing a vertex depend on the degree,
40 so that lower degrees are less likely.
41 */
42 
43 ValueOrdering::ValueOrdering() {
44  /* 2^{-5} = 1/32 ~ 0.03. A choice with probability <3% is probably
45  not very significant for the overall search.
46 
47  The probability of picking a vertex (the mass) is proportional to
48  2^degree.
49 
50  Therefore if a vertex has degree <= max degree - 5, then it has weighting
51  <= 1/32 ~ 3% of any vertex of maximum degree.
52 
53  Thus, including these vertices also would usually only make a difference
54  in a very small number of cases.
55 
56  As explained in the linked paper, the heuristic generally chooses
57  highest degree vertices first, but occasionally chooses
58  slightly lower degrees.
59 
60  This is based on a very rough estimate that the expected number of
61  solutions in the subtree is roughly C . 2^degree.
62 
63  However, this is all very rough and experimental (as the paper says);
64  a better estimate is more like A.B^degree for unknown B>1, but that's
65  still very rough.
66 
67  As the paper says, it generally gives better results than non-exponential
68  formulae but there's no deep underlying theory here.
69 
70  The selection "5" is also an arbitrary choice; we need a cutoff to avoid
71  wasting time on vanishingly unlikely choices, and 3% seems small enough
72  that it's a reasonable cutoff. (OK, we could have 1 vertex of degree 10
73  and 1000 vertices of degree 5. But even in that case, I think we'd rather
74  just try the degree 10 vertex first).
75  */
76  m_entries_for_high_degree_vertices.resize(5);
77  m_entries_for_high_degree_vertices.back().mass = 1;
78  for (unsigned index = m_entries_for_high_degree_vertices.size() - 1;
79  index > 0; --index) {
80  m_entries_for_high_degree_vertices[index - 1].mass =
81  2 * m_entries_for_high_degree_vertices[index].mass;
82  }
83 }

85 void ValueOrdering::fill_entries_for_high_degree_vertices(
86  const std::set<VertexWSM>& possible_values,
87  const NeighboursData& target_ndata) {
88  // Multipass algorithm; simple and efficient enough.
89  // Pass one: find the max degree.
90  size_t max_degree = 0;
91  for (VertexWSM tv : possible_values) {
92  max_degree = std::max(max_degree, target_ndata.get_degree(tv));
93  }
94  TKET_ASSERT(max_degree > 0);
95 
96  // Pass two: record all those vertices with large enough degree.
97  for (HighDegreeVerticesData& entry : m_entries_for_high_degree_vertices) {
98  entry.vertices.clear();
99  }
100  for (VertexWSM tv : possible_values) {
101  const auto degree = target_ndata.get_degree(tv);
102  if (degree + m_entries_for_high_degree_vertices.size() > max_degree) {
103  const auto index = max_degree - degree;
104  m_entries_for_high_degree_vertices[index].vertices.push_back(tv);
105  }
106  }
107  TKET_ASSERT(!m_entries_for_high_degree_vertices[0].vertices.empty());
108 }

110 VertexWSM ValueOrdering::get_random_choice_from_data(RNG& rng) const {
111  // We need probability proportional to the mass; so get the total mass.
112  // Consider each vertex as having k copies, k corresponding to the mass,
113  // list them all, and choose one uniformly.
114 
115  std::size_t mass_sum = 0;
116  for (const auto& entry : m_entries_for_high_degree_vertices) {
117  mass_sum += entry.vertices.size() * entry.mass;
118  }
119  TKET_ASSERT(mass_sum > 0);
120  const auto index = rng.get_size_t(mass_sum - 1);
121 
122  // Now, choose the vertex corresponding to this index.
123  std::size_t preceding_mass = 0;
124  for (const auto& entry : m_entries_for_high_degree_vertices) {
125  std::size_t mass_after_these_vertices =
126  preceding_mass + entry.vertices.size() * entry.mass;
127  if (mass_after_these_vertices < index) {
128  preceding_mass = mass_after_these_vertices;
129  continue;
130  }
131  for (auto vertex : entry.vertices) {
132  size_t mass_after_this_vertex = preceding_mass + entry.mass;
133  if (mass_after_this_vertex >= index) {
134  return vertex;
135  }
136  preceding_mass = mass_after_this_vertex;
137  }
138  }
139 
140  // It's an error if we reach here, although a "harmless" one:
141  // it just means that our calculation of the solution biased heuristic
142  // is wrong.
143  // So, if ever this is reached, and it's an "emergency"
144  // (no time to find the bug), just comment it out temporarily
145  // until the bug is properly fixed.
146  // The worst that could happen is reduced performance; it can't
147  // lead to incorrect results.
148  TKET_ASSERT(false);
149  return m_entries_for_high_degree_vertices.at(0).vertices.at(0);
150 }

152 VertexWSM ValueOrdering::get_target_value(
153  const std::set<VertexWSM>& possible_values,
154  const NeighboursData& target_ndata, RNG& rng) {
155  TKET_ASSERT(possible_values.size() >= 2);
