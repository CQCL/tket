26 VariableOrdering::Result VariableOrdering::get_variable(
27  DomainsAccessor& accessor, RNG& rng) {
28  m_pv_list.clear();
29  std::size_t min_domain_size;
30  set_maximum(min_domain_size);
31  Result result;
32  m_work_vector.clear();
33 
34  // NOTE: it might seem natural to give priority to PV adjacent to
35  // newly assigned vertices, but tests showed that this is actually
36  // a bad idea; it's faster and simpler just to consider all
37  // unassigned vertices equally.
38  for (VertexWSM pv : accessor.get_unassigned_pattern_vertices_superset()) {
39  const auto domain_size = accessor.get_domain(pv).size();
40  if (domain_size == 0) {
41  result.empty_domain = true;
42  return result;
43  }
44  if (domain_size == 1) {
45  continue;
46  }
47  // The point is, we're guaranteed to consider every
48  // unassigned PV (and maybe some assigned ones).
49  // When we move down the search tree, the set of unassigned
50  // vertices decreases; thus we can save time by only searching
51  // this subset of PV.
52  // When we move up and return to a node,
53  // as we reduce further some vertices become assigned, BUT that
54  // doesn't bother us; it just means we have a few extra PV
55  // to search through next time.
56  // This is quicker than trying to maintain a strictly accurate
57  // list of unassigned PV (which would need a std::set).
58  m_work_vector.push_back(pv);
59 
60  if (domain_size > min_domain_size) {
61  continue;
62  }
63  if (domain_size < min_domain_size) {
64  // Strictly better.
65  m_pv_list.clear();
66  min_domain_size = domain_size;
67  }
68  m_pv_list.emplace_back(pv);
69  }
70 
71  // NOTE: what we're doing is, building up the new vector
72  // (the pattern_vertices_superset) in m_work_vector.
73  // Then when we reach this point, we just write the data back
74  // into pattern_vertices_superset (by a vector swap, which is O(1) time).
75  // This is practically just as efficient as overwriting
76  // pattern_vertices_superset directly, but simpler because
77  // we need to copy and manipulate the old data to produce the new data,
78  // so we don't have to do it all within the SAME vector object
79  // (although we could do).
80  accessor.get_unassigned_pattern_vertices_superset_to_overwrite().swap(
81  m_work_vector);
82 
83  result.empty_domain = false;
84  if (m_pv_list.empty()) {
85  return result;
86  }
87  // We have an unassigned vertex!
88  const auto index = rng.get_size_t(m_pv_list.size() - 1);
89  result.variable_opt = m_pv_list[index];
90  return result;
91 }
