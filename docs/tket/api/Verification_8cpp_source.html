18 bool respects_connectivity_constraints(
19  const Circuit &circ, const Architecture &arch, bool directed,
20  bool bridge_allowed) {
21  unit_set_t qb_lookup;
22  for (const Qubit &qb : circ.all_qubits()) {
23  if (!arch.node_exists(Node(qb))) return false;
24  qb_lookup.insert(qb);
25  }
26  for (Circuit::CommandIterator it = circ.begin(); it != circ.end(); ++it) {
27  const Command &com = *it;
28  unit_vector_t qbs;
29  for (const UnitID &arg : com.get_args()) {
30  if (qb_lookup.find(arg) != qb_lookup.end()) qbs.push_back(arg);
31  }
32  Op_ptr op = com.get_op_ptr();
33  if (op->get_type() == OpType::Barrier) continue;
34  if (op->get_type() == OpType::Conditional) {
35  std::shared_ptr<const Conditional> cond_ptr =
36  std::dynamic_pointer_cast<const Conditional>(op);
37  op = cond_ptr->get_op();
38  }
39  if (op->get_type() == OpType::CircBox) {
40  std::shared_ptr<const Box> box_ptr =
41  std::dynamic_pointer_cast<const Box>(op);
42  Circuit box_circ = *box_ptr->to_circuit().get();
43  qubit_vector_t all_units = box_circ.all_qubits();
44  if (all_units.size() != qbs.size()) return false;
45  unit_map_t rename_map;
46  for (unsigned i = 0; i < all_units.size(); i++)
47  rename_map.insert({all_units[i], qbs[i]});
48  box_circ.rename_units(rename_map);
49  if (!respects_connectivity_constraints(
50  box_circ, arch, directed, bridge_allowed))
51  return false;
52  continue;
53  }
54  unsigned n_qbs = qbs.size();
55  switch (n_qbs) {
56  case 0:
57  case 1:
58  break;
59  case 2: {
60  if (arch.get_distance(Node(qbs[0]), Node(qbs[1])) != 1) {
61  return false;
62  }
63  if (directed) {
64  OpType ot = op->get_type();
65  if ((ot == OpType::CX || ot == OpType::ECR) &&
66  !arch.edge_exists(Node(qbs[0]), Node(qbs[1])))
67  return false;
68  }
69  break;
70  }
71  case 3: {
72  if (bridge_allowed) {
73  if (directed)
74  throw std::logic_error(
75  "BRIDGE ops are disallowed on a directed "
76  "architecture. They must be decomposed.");
77  if (op->get_type() == OpType::BRIDGE) {
78  if (arch.get_distance(Node(qbs[0]), Node(qbs[1])) != 1 ||
79  arch.get_distance(Node(qbs[1]), Node(qbs[2])) != 1) {
80  return false;
81  }
82  } else
83  return false;
84  } else
85  return false;
86  break;
87  }
88  default:
89  return false;
90  }
91  }
92  return true;
93 }
