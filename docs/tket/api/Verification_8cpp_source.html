18bool respects_connectivity_constraints(
19 const Circuit &circ, const Architecture &arch, bool directed,
20 bool bridge_allowed) {
21 unit_set_t qb_lookup;
22 for (const Qubit &qb : circ.all_qubits()) {
23 if (!arch.node_exists(Node(qb))) return false;
24 qb_lookup.insert(qb);
25 }
26 for (Circuit::CommandIterator it = circ.begin(); it != circ.end(); ++it) {
27 const Command &com = *it;
28 unit_vector_t qbs;
29 for (const UnitID &arg : com.get_args()) {
30 if (qb_lookup.find(arg) != qb_lookup.end()) qbs.push_back(arg);
31 }
32 Op_ptr op = com.get_op_ptr();
33 while (op->get_type() == OpType::Conditional) {
34 op = static_cast<const Conditional &>(*op).get_op();
35 }
36 if (op->get_type() == OpType::Barrier) continue;
37 if (op->get_type() == OpType::CircBox) {
38 std::shared_ptr<const Box> box_ptr =
39 std::dynamic_pointer_cast<const Box>(op);
40 Circuit box_circ = *box_ptr->to_circuit().get();
41 qubit_vector_t all_units = box_circ.all_qubits();
42 if (all_units.size() != qbs.size()) return false;
43 unit_map_t rename_map;
44 for (unsigned i = 0; i < all_units.size(); i++)
45 rename_map.insert({all_units[i], qbs[i]});
46 box_circ.rename_units(rename_map);
47 if (!respects_connectivity_constraints(
48 box_circ, arch, directed, bridge_allowed))
49 return false;
50 continue;
51 }
52 unsigned n_qbs = qbs.size();
53 switch (n_qbs) {
54 case 0:
55 case 1:
56 break;
57 case 2: {
58 if (arch.get_distance(Node(qbs[0]), Node(qbs[1])) != 1) {
59 return false;
60 }
61 if (directed) {
62 OpType ot = op->get_type();
63 if ((ot == OpType::CX || ot == OpType::ECR) &&
64 !arch.edge_exists(Node(qbs[0]), Node(qbs[1])))
65 return false;
66 }
67 break;
68 }
69 case 3: {
70 if (bridge_allowed) {
71 if (directed)
72 throw std::logic_error(
73 "BRIDGE ops are disallowed on a directed "
74 "architecture. They must be decomposed.");
75 if (op->get_type() == OpType::BRIDGE) {
76 if (arch.get_distance(Node(qbs[0]), Node(qbs[1])) != 1 ||
77 arch.get_distance(Node(qbs[1]), Node(qbs[2])) != 1) {
78 return false;
79 }
80 } else
81 return false;
82 } else
83 return false;
84 break;
85 }
86 default:
87 return false;
88 }
89 }
90 return true;
91}
