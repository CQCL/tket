26 VertexRelabelling::VertexRelabelling(GraphEdgeWeights edges_and_weights)
27  : new_edges_and_weights(std::move(edges_and_weights)) {
28  for (const std::pair<const EdgeWSM, WeightWSM>& entry :
29  new_edges_and_weights) {
30  const VertexWSM& v1 = entry.first.first;
31  const VertexWSM& v2 = entry.first.second;
32  if (v1 == v2) {
33  throw std::runtime_error(
34  std::string("Loop found in graph; not allowed: v=") +
35  std::to_string(v1));
36  }
37  old_to_new_vertex_labels[v1];
38  old_to_new_vertex_labels[v2];
39  const std::optional<WeightWSM> weight_opt =
40  get_optional_value(new_edges_and_weights, std::make_pair(v2, v1));
41  if (weight_opt) {
42  const WeightWSM other_weight = weight_opt.value();
43  if (entry.second != other_weight) {
44  std::stringstream ss;
45  ss << "Edge (" << v1 << "," << v2 << ") has weight " << entry.second
46  << ", differs from edge (" << v2 << "," << v1 << ") with weight "
47  << other_weight;
48  throw std::runtime_error(ss.str());
49  }
50  }
51  }
52  if (old_to_new_vertex_labels.empty()) {
53  throw std::runtime_error(
54  "VertexRelabelling WSM: input graph has no edges!");
55  }
56  number_of_vertices = old_to_new_vertex_labels.size();
57  TKET_ASSERT(number_of_vertices >= 2);
58  if (old_to_new_vertex_labels.cbegin()->first == 0 &&
59  old_to_new_vertex_labels.crbegin()->first == number_of_vertices - 1) {
60  // Already contiguous!
61  old_to_new_vertex_labels.clear();
62  return;
63  }
64  new_to_old_vertex_labels.reserve(number_of_vertices);
65  for (std::pair<const VertexWSM, unsigned>& entry : old_to_new_vertex_labels) {
66  const unsigned new_vertex = new_to_old_vertex_labels.size();
67  new_to_old_vertex_labels.push_back(entry.first);
68  entry.second = new_vertex;
69  }
70  TKET_ASSERT(new_to_old_vertex_labels.size() == number_of_vertices);
71  TKET_ASSERT(old_to_new_vertex_labels.at(new_to_old_vertex_labels[0]) == 0);
72 
73  // Another copy is unavoidable (the vertices aren't contiguous,
74  // we MUST relabel; but this means changing the KEYS of the map
75  // new_edges_and_weights)
76  const auto original_edges_and_weights = new_edges_and_weights;
77  new_edges_and_weights.clear();
78  for (const auto& entry : original_edges_and_weights) {
79  const VertexWSM& old_v1 = entry.first.first;
80  const VertexWSM& old_v2 = entry.first.second;
81 
82  new_edges_and_weights[get_edge(
83  old_to_new_vertex_labels.at(old_v1),
84  old_to_new_vertex_labels.at(old_v2))] = entry.second;
85  }
86 }
