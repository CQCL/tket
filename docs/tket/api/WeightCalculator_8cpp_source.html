24 std::optional<WeightCalculator::Result> WeightCalculator::operator()(
25  const NeighboursData& pattern_ndata, const NeighboursData& target_ndata,
26  const DomainsAccessor& accessor,
27  std::size_t number_of_processed_assignments,
28  WeightWSM max_scalar_product) const {
29  m_p_vertices_seen.clear();
30 
31  Result result;
32  result.scalar_product = accessor.get_scalar_product();
33  result.total_extra_p_edge_weights = 0;
34 
35  const std::vector<std::pair<VertexWSM, VertexWSM>>& assignments =
36  accessor.get_new_assignments();
37 
38  for (auto ii = number_of_processed_assignments; ii < assignments.size();
39  ++ii) {
40  const VertexWSM& pv = assignments[ii].first;
41  const VertexWSM& tv = assignments[ii].second;
42 
43  TKET_ASSERT(m_p_vertices_seen.insert(pv).second);
44 
45  // Look for assigned neighbours.
46  for (const std::pair<VertexWSM, WeightWSM>& entry :
47  pattern_ndata.get_neighbours_and_weights(pv)) {
48  const VertexWSM& other_pv = entry.first;
49  const std::set<VertexWSM>& other_domain = accessor.get_domain(other_pv);
50  switch (other_domain.size()) {
51  case 0:
52  return {};
53  case 1: {
54  // We have an assigned edge.
55  if (m_p_vertices_seen.count(other_pv) != 0) {
56  // We've already seen both vertices,
57  // so the edge must already have been added.
58  break;
59  }
60  const VertexWSM& other_tv = *other_domain.cbegin();
61  const auto t_edge_weight_opt =
62  target_ndata.get_edge_weight_opt(tv, other_tv);
63  if (!t_edge_weight_opt) {
64  return {};
65  }
66  result.scalar_product += entry.second * t_edge_weight_opt.value();
67  if (result.scalar_product > max_scalar_product) {
68  return {};
69  }
70  result.total_extra_p_edge_weights += entry.second;
71  break;
72  }
73  default:
74  break;
75  }
76  }
77  }
78  return result;
79 }
