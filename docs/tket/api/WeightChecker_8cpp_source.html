50 bool WeightChecker::check(
51  const DomainsAccessor& accessor, WeightWSM max_extra_scalar_product) {
52  std::size_t current_number_of_unassigned_vertices = 0;
53 
54  for (VertexWSM pv : accessor.get_unassigned_pattern_vertices_superset()) {
55  switch (accessor.get_domain(pv).size()) {
56  // TODO: make a test case for this; but it's fiddly.
57  // GCOVR_EXCL_START
58  case 0:
59  return false;
60  // GCOVR_EXCL_STOP
61  case 1:
62  break;
63  default:
64  ++current_number_of_unassigned_vertices;
65  }
66  }
67  const std::size_t current_number_of_assigned_vertices =
68  accessor.get_number_of_pattern_vertices() -
69  current_number_of_unassigned_vertices;
70 
71  const auto current_scalar_product = accessor.get_scalar_product();
72  const auto max_scalar_product =
73  current_scalar_product + max_extra_scalar_product;
74 
75  if (!m_manager.should_activate_detector(
76  current_scalar_product, max_scalar_product,
77  accessor.get_total_p_edge_weights(),
78  current_number_of_assigned_vertices,
79  current_number_of_unassigned_vertices)) {
80  return true;
81  }
82 
83  // Finally, we use the detector; check if it's initialised.
84  if (!m_detector_ptr) {
85  const auto used_tv = m_search_branch.get_used_target_vertices();
86  m_tv_data.initial_number_of_tv = used_tv.size();
87 
88  m_detector_ptr = std::make_unique<WeightNogoodDetector>(
89  m_pattern_neighbours_data, m_target_neighbours_data, used_tv,
90  m_impossible_target_vertices);
91  TKET_ASSERT(m_detector_ptr);
92  }
93  const auto extra_scalar_product_lower_bound_opt =
94  m_detector_ptr->get_extra_scalar_product_lower_bound(
95  accessor, max_extra_scalar_product);
96 
97  if (extra_scalar_product_lower_bound_opt) {
98  // We have a lower bound, but it's not big enough to force the
99  // scalar product beyond the limit; thus we haven't found a weight nogood,
100  // so this counts as a failure.
101  m_manager.register_lower_bound_failure(
102  current_scalar_product, max_scalar_product,
103  extra_scalar_product_lower_bound_opt.value());
104  return true;
105  }
106  // We HAVE found a nogood, which is a SUCCESS
107  // (since it's the whole purpose of the detector).
108  m_manager.register_success();
109  return false;
110 }
