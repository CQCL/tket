49 bool WeightNogoodDetectorManager::should_activate_detector(
50  WeightWSM current_weight, WeightWSM max_weight,
51  WeightWSM current_sum_of_p_edge_weights, std::size_t n_assigned_vertices,
52  std::size_t n_unassigned_vertices) {
53  const auto total_n_vertices = n_assigned_vertices + n_unassigned_vertices;
54 
55  // Note that vertex numbers are sensibly small, so won't overflow.
56  // Check state resetting.
57  // (More detail: as we move up and down the search tree, nearby search nodes
58  // are quite similar to each other, so our dynamic feedback strategy
59  // of, roughly, "do more checks if it's successful a lot; reduce
60  // the amount of checking if it's failing a lot" seems sensible.
61  // But, if conditions change a lot, treat it as a fresh search).
62  if (m_state.can_reset &&
63  (n_assigned_vertices <= 2 ||
64  1024 * n_assigned_vertices <=
65  m_parameters.drop_below_n_assigned_vertices_reset_pk *
66  total_n_vertices)) {
67  // The number of assigned vertices is very low,
68  // so we've backtracked far enough that we'll reset
69  // (treat it as a whole new search).
70  m_state = State();
71  } else {
72  if (!m_state.can_reset &&
73  (n_unassigned_vertices <= 2 ||
74  1024 * n_assigned_vertices >=
75  m_parameters.rise_above_n_assigned_vertices_allow_reset_pk *
76  total_n_vertices)) {
77  // We haven't reset for a while; we've now got enough vertices
78  // (i.e., the search has progressed far enough that we'll
79  // now allow reset once we backtrack far enough).
80  m_state.can_reset = true;
81  }
82  }
83  if (m_state.remaining_skips > 0) {
84  --m_state.remaining_skips;
85  return false;
86  }
87 
88  // Are there enough assignments (and remaining unassigned vertices -
89  // if you're very close to a solution, just do it,
90  // rather than estimating)?
91  if (n_assigned_vertices <= 2 || n_unassigned_vertices <= 2 ||
92  current_weight == 0 || m_total_p_edge_weights == 0) {
93  // Not enough assignments (or remaining unassigned vertices)
94  // for a useful estimate.
95  return false;
96  }
97 
98  // Note that we are careful to avoid int oveflow.
99 
100  if (current_weight < max_weight / 1024 ||
101  DyadicFraction(current_weight) <
102  DyadicFraction(max_weight)
103  .mult_n_over_k(m_state.min_weight_pk_to_activate)) {
104  // We haven't got enough weight to start estimating.
105  return false;
106  }
107 
108  // If we very crudely estimate the total weight after all
109  // assignments are made (simply based on the fraction of
110  // pattern graph assignments so far:
111  // either from number of vertices, or total p-weight),
112  // is it large enough for a detection to be worthwhile
113  // (because it has more chance of success, i.e. getting
114  // a large estimate)?
115  if (
116  // The fraction of pattern graph assignments
117  // based upon total p-weight.
118  DyadicFraction(current_weight).mult(m_total_p_edge_weights) <
119  DyadicFraction(max_weight)
120  .mult(current_sum_of_p_edge_weights)
121  .mult_n_over_k(m_state.final_weight_estimate_pk_to_activate) &&
122 
123  DyadicFraction(current_weight).mult(total_n_vertices) <
124  DyadicFraction(max_weight)
125  .mult(n_assigned_vertices)
126  .mult_n_over_k(m_state.final_weight_estimate_pk_to_activate)) {
127  return false;
128  }
129  return true;
130 }

132 void WeightNogoodDetectorManager::register_success() {
133  get_counter().good_count += 1;
134  m_state.final_weight_estimate_pk_to_activate *=
135  m_parameters.final_weight_estimate_pk_success_growth_pk;
136  m_state.final_weight_estimate_pk_to_activate /= 1024;
137 
138  m_state.min_weight_pk_to_activate *=
139  m_parameters.success_pk_to_activate_growth_factor_pk;
140  m_state.min_weight_pk_to_activate /= 1024;
141  clamp_state_values();
142  m_state.remaining_skips = 0;

145 void WeightNogoodDetectorManager::register_lower_bound_failure(
146  WeightWSM current_weight, WeightWSM max_weight,
147  WeightWSM extra_weight_lower_bound) {
148  if (current_weight + 2 * extra_weight_lower_bound < max_weight) {
149  // Bad failure.
150  get_counter().bad_count += 1;
151  m_state.final_weight_estimate_pk_to_activate *=
152  m_parameters.final_weight_estimate_pk_bad_failure_growth_pk;
153  m_state.min_weight_pk_to_activate *=
154  m_parameters.bad_failure_pk_to_activate_growth_factor_pk;
155  m_state.remaining_skips = m_parameters.bad_failure_skip;
156  } else {
157  get_counter().ok_count += 1;
158  // "OK" failure; the lower bound wasn't that far off the amount needed.
159  m_state.final_weight_estimate_pk_to_activate *=
160  m_parameters.final_weight_estimate_pk_ok_failure_growth_pk;
161  m_state.min_weight_pk_to_activate *=
162  m_parameters.ok_failure_pk_to_activate_growth_factor_pk;
163  m_state.remaining_skips = m_parameters.ok_failure_skip;
164  }
165  m_state.final_weight_estimate_pk_to_activate /= 1024;
166  m_state.min_weight_pk_to_activate /= 1024;
167  clamp_state_values();
168 }
