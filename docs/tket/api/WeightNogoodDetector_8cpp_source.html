41 std::optional<WeightWSM> WeightNogoodDetector::get_min_weight_for_tv(
42  VertexWSM tv) const {
43  if (m_valid_target_vertices.count(tv) == 0) {
44  return {};
45  }
46  const auto weight_opt = get_optional_value(m_minimum_t_weights_from_tv, tv);
47  if (weight_opt) {
48  return weight_opt;
49  }
50  // We must find the minimum weight, by looking at all neighbours.
51  WeightWSM min_weight;
52  set_maximum(min_weight);
53  const std::vector<std::pair<VertexWSM, WeightWSM>>& data =
54  m_target_neighbours_data.get_neighbours_and_weights(tv);
55  for (const std::pair<VertexWSM, WeightWSM>& entry : data) {
56  const VertexWSM& neighbour_tv = entry.first;
57  if (m_valid_target_vertices.count(neighbour_tv) == 0) {
58  continue;
59  }
60  min_weight = std::min(min_weight, entry.second);
61  }
62  if (is_maximum(min_weight)) {
63  // This TV has no valid neighbours; so clearly, no (nonisolated) PV
64  // could ever be assigned to it.
65  // Really, to do this properly, we should update
66  // target_neighbours_data dynamically,
67  // erasing invalid target vertices and updating neighbour lists.
68  return {};
69  }
70  m_minimum_t_weights_from_tv[tv] = min_weight;
71  return min_weight;
72 }

74 WeightWSM WeightNogoodDetector::get_t_weight_lower_bound(VertexWSM pv) const {
75  // Of course, std::lower_bound is unfortunate terminology;
76  // completely unrelated to our weight lower bounds!
77  const auto citer = std::lower_bound(
78  m_t_weight_lower_bounds_for_p_edges_containing_pv.cbegin(),
79  m_t_weight_lower_bounds_for_p_edges_containing_pv.cend(),
80  std::make_pair(pv, WeightWSM(0)));
81  TKET_ASSERT(
82  citer != m_t_weight_lower_bounds_for_p_edges_containing_pv.cend());
83  TKET_ASSERT(citer->first == pv);
84  return citer->second;
85 }

87 bool WeightNogoodDetector::fill_t_weight_lower_bounds_for_p_edges_containing_pv(
88  const DomainsAccessor& accessor) const {
89  // Every unassigned edge must connect to an unassigned p-vertex.
90  m_t_weight_lower_bounds_for_p_edges_containing_pv.clear();
91 
92  for (unsigned pv = 0; pv < accessor.get_number_of_pattern_vertices(); ++pv) {
93  WeightWSM weight;
94  set_maximum(weight);
95  const std::set<VertexWSM>& domain = accessor.get_domain(pv);
96  TKET_ASSERT(!domain.empty());
97  for (VertexWSM tv : domain) {
98  const auto weight_opt_for_tv = get_min_weight_for_tv(tv);
99  if (weight_opt_for_tv) {
100  weight = std::min(weight, weight_opt_for_tv.value());
101  } else {
102  m_invalid_target_vertices.insert(tv);
103  m_valid_target_vertices.erase(tv);
104  }
105  }
106  if (is_maximum(weight)) {
107  // A nogood found already!
108  return false;
109  }
110  m_t_weight_lower_bounds_for_p_edges_containing_pv.emplace_back(pv, weight);
111  }
112  return true;
113 }

115 std::optional<WeightWSM>
116 WeightNogoodDetector::get_extra_scalar_product_lower_bound(
117  const DomainsAccessor& accessor, WeightWSM max_extra_scalar_product) {
118  if (!fill_t_weight_lower_bounds_for_p_edges_containing_pv(accessor)) {
119  // A nogood!
120  return {};
121  }
122  WeightWSM weight_lower_bound = 0;
123 
124  // Now, for each unassigned p-vertex, look at its neighbours
125  // and deduce the p-edge weights.
126  // There's a problem: if BOTH p-edge endpoints are unassigned,
127  // the edge will be counted twice.
128  // To solve this: only add the data when pv1 < pv2.
129  for (VertexWSM pv1 : accessor.get_unassigned_pattern_vertices_superset()) {
130  if (accessor.get_domain(pv1).size() == 1) {
131  // It's assigned.
132  continue;
133  }
134  // Note: we don't care about the domain of pv1,
135  // we've already gone through it before (when we called
136  // fill_t_weight_lower_bounds_for_p_edges_containing_pv).
137 
138  // A lower bound for all t-edges
139  // which could possibly contain f(pv).
140  const auto minimum_t_weight = get_t_weight_lower_bound(pv1);
141 
142  const std::vector<std::pair<VertexWSM, WeightWSM>>&
143  p_neighbours_and_weights =
144  m_pattern_neighbours_data.get_neighbours_and_weights(pv1);
145 
146  for (const std::pair<VertexWSM, WeightWSM>& pv2_weight_pair :
147  p_neighbours_and_weights) {
148  const VertexWSM& pv2 = pv2_weight_pair.first;
149  const WeightWSM& p_weight = pv2_weight_pair.second;
150  WeightWSM t_weight_estimate = minimum_t_weight;
151  const std::set<VertexWSM>& domain2 = accessor.get_domain(pv2);
152  if (domain2.size() == 1) {
153  // This other p-vertex PV2 is assigned already.
154  // So let's check the other t-weight estimate, it may be better.
155  const VertexWSM& tv2 = *domain2.cbegin();
156 
157  // We already know that this edge contains pv1,
158  // so DEFINITELY has t-weight >= this current estimate.
159  // But we also know that it will be assigned to a target edge
160  // containing tv2, which has t_weight >= the other value.
161  const auto other_tv_weight_bound_opt = get_min_weight_for_tv(tv2);
162 
163  if (!other_tv_weight_bound_opt) {
164  // We're at a nogood!
165  // (This COULD actually happen. It means that TV2 is invalid,
166  // and in fact always was;
167  // but we didn't realise this at the time we made the assignment).
168  m_invalid_target_vertices.insert(tv2);
169  return {};
170  }
171 
172  // We take the MAX to get a valid LOWER bound
173  // as LARGE as possible.
174  t_weight_estimate =
175  std::max(t_weight_estimate, other_tv_weight_bound_opt.value());
176  } else {
177  // pv2 is ALSO unassigned. Beware of double counting!
178  if (pv1 > pv2) {
179  continue;
180  }
181  // We'll do BOTH pv1--pv2 and pv2--pv1 NOW.
182  // If pv2 produces a larger value, we can use that instead.
183  t_weight_estimate =
184  std::max(t_weight_estimate, get_t_weight_lower_bound(pv2));
185  }
186  weight_lower_bound += p_weight * t_weight_estimate;
187  if (weight_lower_bound > max_extra_scalar_product) {
188  return {};
189  }
190  }
191  }
192  return weight_lower_bound;
193 }
