85  // As soon as a TV is newly discovered to be invalid,
86  // meaning that no p-vertex could ever be assigned to it
87  // (quite a rare event), store it here.
88  //
89  // This of course is even stronger than a normal impossible
90  // assignment PV->TV.
91  //
92  // Moreover (in case the caller ever tries parallel searches, etc.)
93  // the TV was ALWAYS invalid, so can be erased from ALL data.
94  std::set<VertexWSM>& m_invalid_target_vertices;
95 
96  // There are many mutable data members.
97  // Even though this class is "logically" const,
98  // it's not "physically" const because of lazy evaluation,
99  // as well as work data to avoid memory reallocation.
100 
101  // KEY: a target vertex TV.
102  // VALUE: the smallest t-edge weight which can arise
103  // from an edge containing TV.
104  // Excludes target edges, if any, which can never arise
105  // (i.e. containing at least one vertex which no pattern vertex
106  // can ever map to).
107  //
108  // Mutable because it's lazily initialised.
109  mutable std::map<VertexWSM, WeightWSM> m_minimum_t_weights_from_tv;
110 
111  // Calculated only on first use, and cached;
112  // if non-null, the minimum edge weight of any target edge
113  // containing the target vertex tv.
114  std::optional<WeightWSM> get_min_weight_for_tv(VertexWSM tv) const;
115 
116  // This really is mutable, to avoid reallocation.
117  // In each pair: the FIRST is an UNASSIGNED pattern vertex pv.
118  // The second is a guaranteed lower bound for the t-weight which will be
119  // matched to any p-edge containing pv. (The p-edge therefore must necessarily
120  // be unassigned; but BEWARE: the other vertex could be assigned already).
121  //
122  // It just so happens, from the way it's created, that it's ALWAYS sorted!
123  // Thus, we can use a sorted vector instead of a map.
124  mutable std::vector<std::pair<VertexWSM, WeightWSM>>
125  m_t_weight_lower_bounds_for_p_edges_containing_pv;
126 
127  // Fills m_t_weight_lower_bounds_for_p_edges_containing_pv,
128  // relevant for this current search node only.
129  // Returns false if it found we're already at a nogood.
130  bool fill_t_weight_lower_bounds_for_p_edges_containing_pv(
131  const DomainsAccessor& accessor) const;
132 
133  // Simply looks up and returns the weight in
134  // m_t_weight_lower_bounds_for_p_edges_containing_pv.
135  // So, out of all t-edges which could possibly contain
136  // F(pv), for any valid mapping F from this current node,
137  // this is the lowest weight.
138  // We're GUARANTEED that it exists, because
139  // m_t_weight_lower_bounds_for_p_edges_containing_pv was already filled
140  // with EVERY pv, except when there were no valid t-edges,
141  // in which case it was a nogood and we should already have returned.
142  WeightWSM get_t_weight_lower_bound(VertexWSM pv) const;
