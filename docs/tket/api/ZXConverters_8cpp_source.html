35 // Add a swicth with the on-state controlled by the on_value.
36 // qtype indicates the quantum type of the switch.
37 std::pair<ZXVertPort, ZXVertPort> add_switch(
38  ZXDiagram& zxd, const bool& on_value, const QuantumType& qtype) {
39  zxd.multiply_scalar(std::sqrt(2.));
40  ZXVert triangle = zxd.add_vertex(ZXType::Triangle, qtype);
41  ZXVert x = zxd.add_vertex(ZXType::XSpider, 0, qtype);
42  zxd.add_wire(triangle, x, ZXWireType::Basic, qtype, 1);
43  if (on_value) {
44  // Turn on the switch if the input is 1
45  ZXVert negate = zxd.add_vertex(ZXType::XSpider, 1, qtype);
46  zxd.add_wire(triangle, negate, ZXWireType::Basic, qtype, 0);
47  return {{negate, std::nullopt}, {x, std::nullopt}};
48  }
49  return {{triangle, 0}, {x, std::nullopt}};
50 }

52 // --o--s---o--
53 // | |
54 // n n
55 // |--G-o-|
56 // |
57 // s
58 // |-discard
59 // s, n are switches, G is the conditional zx diagram specified by 'left' and
60 // 'right'. s is on when the input is 0, n is on when the input is 1.
61 // return the input and the output of the conditional zx along with a vector of
62 // control spiders
63 std::pair<std::pair<ZXVertPort, ZXVertPort>, ZXVertPortVec> add_conditional_zx(
64  ZXDiagram& zxd, const ZXVert& left, const ZXVert& right,
65  const QuantumType& qtype) {
66  ZXVert in = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
67  ZXVert out = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
68  ZXVert discard_ctr = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
69  ZXVert discard = zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
70  auto switch_s0 = add_switch(zxd, false, qtype);
71  auto switch_n0 = add_switch(zxd, true, qtype);
72  auto switch_n1 = add_switch(zxd, true, qtype);
73  auto switch_s1 = add_switch(zxd, false, qtype);
74  zxd.add_wire(in, switch_s0.second.first, ZXWireType::Basic, qtype);
75  zxd.add_wire(out, switch_s0.second.first, ZXWireType::Basic, qtype);
76  zxd.add_wire(in, switch_n0.second.first, ZXWireType::Basic, qtype);
77  zxd.add_wire(left, switch_n0.second.first, ZXWireType::Basic, qtype);
78  zxd.add_wire(right, discard_ctr, ZXWireType::Basic, qtype);
79  zxd.add_wire(discard_ctr, switch_n1.second.first, ZXWireType::Basic, qtype);
80  zxd.add_wire(out, switch_n1.second.first, ZXWireType::Basic, qtype);
81  zxd.add_wire(discard_ctr, switch_s1.second.first, ZXWireType::Basic, qtype);
82  zxd.add_wire(discard, switch_s1.second.first, ZXWireType::Basic, qtype);
83  return {
84  {{in, std::nullopt}, {out, std::nullopt}},
85  {switch_s0.first, switch_s1.first, switch_n0.first, switch_n1.first}};
86 }

88 // n-bit AND spider to control the switches
89 // https://arxiv.org/abs/1910.06818
90 std::pair<ZXVertPortVec, ZXVertPort> add_n_bit_and(
91  ZXDiagram& zxd, unsigned n, const QuantumType& qtype) {
92  TKET_ASSERT(n > 1);
93  ZXVert z_vert = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
94  // Add Triangle -1
95  ZXVert z_pi_0 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
96  ZXVert tri_1 = zxd.add_vertex(ZXType::Triangle, qtype);
97  ZXVert z_pi_1 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
98  zxd.add_wire(z_vert, z_pi_0, ZXWireType::Basic, qtype);
99  zxd.add_wire(tri_1, z_pi_0, ZXWireType::Basic, qtype, 0);
100  zxd.add_wire(tri_1, z_pi_1, ZXWireType::Basic, qtype, 1);
101  ZXVertPortVec inputs;
102  for (unsigned i = 0; i < n; i++) {
103  ZXVert tri_0 = zxd.add_vertex(ZXType::Triangle, qtype);
104  zxd.add_wire(tri_0, z_vert, ZXWireType::Basic, qtype, 1);
105  inputs.push_back({tri_0, 0});
106  }
107  return {inputs, {z_pi_1, std::nullopt}};
108 }

110 // Add converted circ into zxd. Set add_boundary to true
111 // if boundary spiders are to be added to the boundary.
112 BoundaryVertMap circuit_to_zx_recursive(
113  const Circuit& circ, ZXDiagram& zxd, bool add_boundary) {
114  std::map<TypedVertPort, ZXVertPort> vert_lookup;
115  std::map<VertPort, ZXVertPort> boolean_outport_lookup;
116  BoundaryVertMap bmap;
117 
118  // Convert each vertex to ZXDiagram, raise error if not supported
119  BGL_FORALL_VERTICES(vert, circ.dag, DAG) {
120  // We currently throw an error if the vertex is either classical or flow
121  Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
122  if (is_flowop_type(op->get_type()) || is_classical_type(op->get_type())) {
123  throw Unsupported(
124  "Cannot convert OpType: " + op->get_name() + " to a ZX node. \n");
125  }
126  switch (op->get_type()) {
127  case OpType::Input: {
128  ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Quantum);
129  if (add_boundary) zxd.add_boundary(zx_vert);
130  vert_lookup.insert(
131  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
132  bmap.insert({zx_vert, vert});
133  break;
134  }
135  case OpType::Output: {
136  ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Quantum);
137  if (add_boundary) zxd.add_boundary(zx_vert);
138  vert_lookup.insert(
139  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
140  bmap.insert({zx_vert, vert});
141  break;
142  }
143  case OpType::ClInput: {
144  ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Classical);
145  if (add_boundary) zxd.add_boundary(zx_vert);
146  vert_lookup.insert(
147  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
148  bmap.insert({zx_vert, vert});
149  break;
150  }
151  case OpType::ClOutput: {
152  ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Classical);
153  if (add_boundary) zxd.add_boundary(zx_vert);
154  vert_lookup.insert(
155  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
156  bmap.insert({zx_vert, vert});
157  break;
158  }
159  // Spiderless ops are handled during vertex wiring
160  case OpType::Barrier:
161  case OpType::noop:
162  case OpType::SWAP: {
163  continue;
164  }
165  case OpType::H: {
166  ZXVert zx_vert = zxd.add_vertex(ZXType::Hbox, QuantumType::Quantum);
167  vert_lookup.insert(
168  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
169  vert_lookup.insert(
170  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
171  zxd.multiply_scalar(0.5);
172  break;
173  }
174  case OpType::Rz: {
175  ZXVert zx_vert = zxd.add_vertex(
176  ZXType::ZSpider, op->get_params()[0], QuantumType::Quantum);
177  vert_lookup.insert(
178  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
179  vert_lookup.insert(
180  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
181  break;
182  }
183  case OpType::Rx: {
184  ZXVert zx_vert = zxd.add_vertex(
185  ZXType::XSpider, op->get_params()[0], QuantumType::Quantum);
186  vert_lookup.insert(
187  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
188  vert_lookup.insert(
189  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
190  break;
191  }
192  case OpType::X: {
193  ZXVert zx_vert =
194  zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Quantum);
195  vert_lookup.insert(
196  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
197  vert_lookup.insert(
198  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
199  break;
200  }
201  case OpType::Z: {
202  ZXVert zx_vert =
203  zxd.add_vertex(ZXType::ZSpider, 1, QuantumType::Quantum);
204  vert_lookup.insert(
205  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
206  vert_lookup.insert(
207  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
208  break;
209  }
210  case OpType::CX: {
211  ZXVert zx_x_vert =
212  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
213  ZXVert zx_z_vert =
214  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
215  zxd.add_wire(zx_x_vert, zx_z_vert);
216  vert_lookup.insert(
217  {{{vert, 0}, ZXPortType::In}, {zx_z_vert, std::nullopt}});
218  vert_lookup.insert(
219  {{{vert, 0}, ZXPortType::Out}, {zx_z_vert, std::nullopt}});
220  vert_lookup.insert(
221  {{{vert, 1}, ZXPortType::In}, {zx_x_vert, std::nullopt}});
222  vert_lookup.insert(
223  {{{vert, 1}, ZXPortType::Out}, {zx_x_vert, std::nullopt}});
224  zxd.multiply_scalar(2);
225  break;
226  }
227  case OpType::CZ: {
228  ZXVert zx_za_vert =
229  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
230  ZXVert zx_zb_vert =
231  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
232  zxd.add_wire(zx_za_vert, zx_zb_vert, ZXWireType::H);
233  vert_lookup.insert(
234  {{{vert, 0}, ZXPortType::In}, {zx_za_vert, std::nullopt}});
235  vert_lookup.insert(
236  {{{vert, 0}, ZXPortType::Out}, {zx_za_vert, std::nullopt}});
237  vert_lookup.insert(
238  {{{vert, 1}, ZXPortType::In}, {zx_zb_vert, std::nullopt}});
239  vert_lookup.insert(
240  {{{vert, 1}, ZXPortType::Out}, {zx_zb_vert, std::nullopt}});
241  break;
242  }
243  case OpType::Measure: {
244  // Add a decoherence node
245  ZXVert zx_measure_vert =
246  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
247  // Add a delete operator
248  ZXVert zx_delete_vert =
249  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
250  vert_lookup.insert(
251  {{{vert, 0}, ZXPortType::In}, {zx_measure_vert, std::nullopt}});
252  vert_lookup.insert(
253  {{{vert, 0}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
254  vert_lookup.insert(
255  {{{vert, 1}, ZXPortType::In}, {zx_delete_vert, std::nullopt}});
256  vert_lookup.insert(
257  {{{vert, 1}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
258  break;
259  }
260  case OpType::Reset: {
261  // Discard
262  ZXVert zx_discard_vert =
263  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
264  // Add a node to prepare |0>
265  ZXVert zx_reset_vert =
266  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
267  zxd.multiply_scalar(0.5);
268  vert_lookup.insert(
269  {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
270  vert_lookup.insert(
271  {{{vert, 0}, ZXPortType::Out}, {zx_reset_vert, std::nullopt}});
272  break;
273  }
274  case OpType::Collapse: {
275  ZXVert zx_vert =
276  zxd.add_vertex(ZXType::ZSpider, QuantumType::Classical);
277  vert_lookup.insert(
278  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
279  vert_lookup.insert(
280  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
281  break;
282  }
283  case OpType::Create: {
284  ZXVert zx_init_vert =
285  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
286  zxd.multiply_scalar(0.5);
287  vert_lookup.insert(
288  {{{vert, 0}, ZXPortType::Out}, {zx_init_vert, std::nullopt}});
289  break;
290  }
291  case OpType::Discard: {
292  ZXVert zx_discard_vert =
293  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
294  vert_lookup.insert(
295  {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
296  break;
297  }
298  case OpType::Conditional: {
299  // Stores the condition for each port index
300  std::vector<bool> port_conditions;
301 
302  op_signature_t parent_sig = op->get_signature();
303 
304  // Find the nested quantum op and its overall conditions
305  // Assume bool ports are always the first few ports
306  while (op->get_type() == OpType::Conditional) {
307  const Conditional& cond = static_cast<const Conditional&>(*op);
308  for (unsigned i = 0; i < cond.get_width(); i++) {
309  bool set = cond.get_value() & (1 << (cond.get_width() - i - 1));
310  port_conditions.push_back(set);
311  }
312  op = cond.get_op();
313  }
314  unsigned port_conditions_size =
315  static_cast<unsigned>(port_conditions.size());
316  // Convert the underlying op to zx.
317  // If the op is a quantum gate, construct a 1 gate circuit
318  // If the op is a box, obtain its circuit decomposition
319  op_signature_t inner_sig = op->get_signature();
320  Circuit replacement;
321  if (is_box_type(op->get_type())) {
322  const Box& b = static_cast<const Box&>(*op);
323  replacement = *b.to_circuit();
324  } else {
325  unit_vector_t args;
326  unsigned q_index = 0;
327  unsigned c_index = 0;
328  for (const EdgeType& etype : inner_sig) {
329  if (etype == EdgeType::Quantum) {
330  args.push_back(Qubit(q_index++));
331  } else if (etype == EdgeType::Classical) {
332  args.push_back(Bit(c_index++));
333  }
334  TKET_ASSERT(etype != EdgeType::Boolean);
335  }
336  replacement = Circuit(q_index, c_index);
337  replacement.add_op(op, args);
338  }
339  BoundaryVertMap box_bm =
340  circuit_to_zx_recursive(replacement, zxd, false);
341 
342  // The Z spider controlling all switches
343  ZXVert master_switch =
344  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
345 
346  // For each qubit/bit path in the inner op, convert it into a
347  // conditional path.
348  unsigned q_index = 0;
349  unsigned c_index = 0;
350  for (unsigned i = 0; i < inner_sig.size(); i++) {
351  QuantumType qtype;
352  UnitID reg;
353  if (inner_sig[i] == EdgeType::Quantum) {
354  qtype = QuantumType::Quantum;
355  reg = Qubit(q_index++);
356  } else {
357  qtype = QuantumType::Classical;
358  reg = Bit(c_index++);
359  }
360  std::pair<ZXVertPort, ZXVertPort> boundary;
361  ZXVertPortVec controls;
362  Vertex inp = replacement.get_in(reg);
363  ZXVert zx_inp = box_bm.right.at(inp);
364  Vertex outp = replacement.get_out(reg);
365  ZXVert zx_outp = box_bm.right.at(outp);
366  // Convert the path between zx_inp and zx_outp into a conditional
367  // path
368  std::tie(boundary, controls) =
369  add_conditional_zx(zxd, zx_inp, zx_outp, qtype);
370  // Connect each switch to the master switch
371  for (const ZXVertPort& ctr : controls) {
372  zxd.add_wire(
373  ctr.first, master_switch, ZXWireType::Basic,
374  zxd.get_qtype(ctr.first).value(), ctr.second);
375  }
376  // Update lookup
377  TKET_ASSERT(parent_sig[i + port_conditions_size] == inner_sig[i]);
378  // Since we assume that the conditions always use the first few
379  // ports, the gate path should use the i + port_conditions.size()
380  // port.
381  vert_lookup.insert(
382  {{{vert, i + port_conditions_size}, ZXPortType::In},
383  boundary.first});
384  vert_lookup.insert(
385  {{{vert, i + port_conditions_size}, ZXPortType::Out},
386  boundary.second});
387  }
388  // Use either a Z spider or a AND operator to connect the master
389  // switch and the boolean inputs
390  ZXVertPortVec and_inputs;
391  if (port_conditions_size == 1) {
392  and_inputs.push_back({master_switch, std::nullopt});
393  } else {
394  ZXVertPort and_output;
395  std::tie(and_inputs, and_output) =
396  add_n_bit_and(zxd, port_conditions_size, QuantumType::Classical);
397  zxd.add_wire(
398  and_output.first, master_switch, ZXWireType::Basic,
399  QuantumType::Classical, and_output.second);
400  }
401  // Connect inputs to the nodes obtained from above
402  for (unsigned i = 0; i < port_conditions_size; i++) {
403  // Each boolean edge shares a source port with other
404  // classical/boolean edges. Use the classical Z spider to explicitly
405  // implement this copy operation
406  Edge in_edge = circ.get_nth_in_edge(vert, i);
407  port_t source_port = circ.get_source_port(in_edge);
408  Vertex vert_s = circ.source(in_edge);
409  // During the wiring stage, each edge originated from {vert_s,
410  // p_s} needs go through copy_vert
411  if (boolean_outport_lookup.find({vert_s, source_port}) ==
412  boolean_outport_lookup.end()) {
413  ZXVert copy_vert =
414  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
415  boolean_outport_lookup.insert(
416  {{vert_s, source_port}, {copy_vert, std::nullopt}});
417  }
418  if (port_conditions[i]) {
419  vert_lookup.insert({{{vert, i}, ZXPortType::In}, and_inputs[i]});
420  } else {
421  ZXVert x_vert =
422  zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Classical);
423  zxd.add_wire(
424  and_inputs[i].first, x_vert, ZXWireType::Basic,
425  QuantumType::Classical, and_inputs[i].second);
426  vert_lookup.insert(
427  {{{vert, i}, ZXPortType::In}, {x_vert, std::nullopt}});
428  }
429  }
430  break;
431  }
432  default:
433  if (is_box_type(op->get_type())) {
434  EdgeVec b_in_holes =
435  circ.get_in_edges_of_type(vert, EdgeType::Boolean);
436  if (b_in_holes.size() > 0) {
437  throw Unsupported("Cannot convert box type: " + op->get_name());
438  }
439  const Box& b = static_cast<const Box&>(*op);
440  Circuit replacement = *b.to_circuit();
441  // First, add the converted box to diagram.
442  BoundaryVertMap box_bm =
443  circuit_to_zx_recursive(replacement, zxd, false);
444  // Then, map the vertports in the box boundary to zx nodes.
445  // Assume that a box can't have Boolean input edges, and all Boolean
446  // output edges share ports with Classical edges. Therefore we don't
447  // have to map Boolean vertports.
448  EdgeVec q_in_holes =
449  circ.get_in_edges_of_type(vert, EdgeType::Quantum);
450  EdgeVec q_out_holes =
451  circ.get_out_edges_of_type(vert, EdgeType::Quantum);
452  EdgeVec c_in_holes =
453  circ.get_in_edges_of_type(vert, EdgeType::Classical);
454  EdgeVec c_out_holes =
455  circ.get_out_edges_of_type(vert, EdgeType::Classical);
456 
457  for (unsigned i = 0; i < q_in_holes.size(); i++) {
458  port_t port = circ.get_target_port(q_in_holes[i]);
459  Vertex inp = replacement.get_in(Qubit(i));
460  vert_lookup.insert(
461  {{{vert, port}, ZXPortType::In},
462  {box_bm.right.at(inp), std::nullopt}});
463  }
464  for (unsigned i = 0; i < q_out_holes.size(); i++) {
465  port_t port = circ.get_source_port(q_out_holes[i]);
466  Vertex outp = replacement.get_out(Qubit(i));
467  vert_lookup.insert(
468  {{{vert, port}, ZXPortType::Out},
469  {box_bm.right.at(outp), std::nullopt}});
470  }
471  for (unsigned i = 0; i < c_in_holes.size(); i++) {
472  port_t port = circ.get_target_port(c_in_holes[i]);
473  Vertex inp = replacement.get_in(Bit(i));
474  vert_lookup.insert(
475  {{{vert, port}, ZXPortType::In},
476  {box_bm.right.at(inp), std::nullopt}});
477  }
478  for (unsigned i = 0; i < c_out_holes.size(); i++) {
479  port_t port = circ.get_source_port(c_out_holes[i]);
480  Vertex outp = replacement.get_out(Bit(i));
481  vert_lookup.insert(
482  {{{vert, port}, ZXPortType::Out},
483  {box_bm.right.at(outp), std::nullopt}});
484  }
485  } else {
486  throw Unsupported(
487  "Cannot convert gate type: " + op->get_name() +
488  " to a ZX node, try rebase the gates to use Rx, Rz, "
489  "X, Z, H, CZ "
490  "or CX. \n");
491  }
492  }
493  }

495  // Connect the ZX nodes
496  BGL_FORALL_EDGES(edge, circ.dag, DAG) {
497  Vertex v_s = circ.source(edge);
498  Vertex v_t = circ.target(edge);
499  port_t p_s = circ.get_source_port(edge);
500  port_t p_t = circ.get_target_port(edge);
501  // Handle Spiderless ops
502  if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_s))) {
503  // We only handle in-edges
504  continue;
505  }
506  if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t))) {
507  // Traverse the path to find the next non-spiderless op
508  Edge next_e = edge;
509  do {
510  if (circ.get_OpType_from_Vertex(v_t) == OpType::SWAP) {
511  next_e = circ.get_nth_out_edge(
512  v_t, (circ.get_target_port(next_e) + 1) % 2);
513  v_t = circ.target(next_e);
514  } else {
515  std::tie(v_t, next_e) = circ.get_next_pair(v_t, next_e);
516  }
517  } while (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t)));
518  p_t = circ.get_target_port(next_e);
519  }
520 
521  ZXVertPort zx_vp_s =
522  vert_lookup.at(TypedVertPort(VertPort(v_s, p_s), ZXPortType::Out));
523  ZXVertPort zx_vp_t =
524  vert_lookup.at(TypedVertPort(VertPort(v_t, p_t), ZXPortType::In));
525  if (circ.get_edgetype(edge) == EdgeType::Quantum) {
526  zxd.add_wire(
527  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic, QuantumType::Quantum,
528  zx_vp_s.second, zx_vp_t.second);
529  } else {
530  auto bool_it = boolean_outport_lookup.find(VertPort(v_s, p_s));
531  if (bool_it == boolean_outport_lookup.end()) {
532  zxd.add_wire(
533  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic,
534  QuantumType::Classical, zx_vp_s.second, zx_vp_t.second);
535  } else {
536  // If the source port is boolean, then connect the copy spider to the
537  // source vertex. All out-edges originated from the source port should
538  // be connected to the copy spider.
539  if (zxd.degree(bool_it->second.first) == 0) {
540  zxd.add_wire(
541  zx_vp_s.first, bool_it->second.first, ZXWireType::Basic,
542  QuantumType::Classical, zx_vp_s.second, bool_it->second.second);
543  }
544  zxd.add_wire(
545  bool_it->second.first, zx_vp_t.first, ZXWireType::Basic,
546  QuantumType::Classical, bool_it->second.second, zx_vp_t.second);
547  }
548  }
549  }
550  return bmap;
551 }

553 std::pair<ZXDiagram, BoundaryVertMap> circuit_to_zx(const Circuit& circ) {
554  ZXDiagram zxd;
555  BoundaryVertMap bmap = circuit_to_zx_recursive(circ, zxd, true);
556  // Remove internal boundary vertices produced by the recursion
557  ZXVertVec true_boundary = zxd.get_boundary();
558  ZXVertIterator vi, vi_end, next;
559  tie(vi, vi_end) = boost::vertices(*zxd.get_graph());
560  for (next = vi; vi != vi_end; vi = next) {
561  ++next;
562  if ((zxd.get_zxtype(*vi) == ZXType::Input ||
563  zxd.get_zxtype(*vi) == ZXType::Output) &&
564  std::find(true_boundary.begin(), true_boundary.end(), *vi) ==
565  true_boundary.end()) {
566  WireVec adj_wires = zxd.adj_wires(*vi);
567  TKET_ASSERT(adj_wires.size() == 2);
568  TKET_ASSERT(zxd.get_qtype(adj_wires[0]) == zxd.get_qtype(adj_wires[1]));
569  TKET_ASSERT(
570  zxd.get_wire_type(adj_wires[0]) == zxd.get_wire_type(adj_wires[1]));
571  ZXVertVec neighbours = zxd.neighbours(*vi);
572  zxd.add_wire(
573  neighbours[0], neighbours[1], zxd.get_wire_type(adj_wires[0]),
574  zxd.get_qtype(adj_wires[0]));
575  zxd.remove_vertex(*vi);
576  }
577  }
578  return {std::move(zxd), std::move(bmap)};
579 }

581 void clean_frontier(
582  ZXDiagram& diag, ZXVertVec& frontier, Circuit& circ,
583  std::map<ZXVert, unsigned>& qubit_map) {
584  std::set<ZXVert> frontier_lookup;
585  std::list<std::pair<ZXVert, ZXVert>> czs;
586  for (const ZXVert& f : frontier) {
587  frontier_lookup.insert(f);
588  for (const Wire& w : diag.adj_wires(f)) {
589  ZXVert n = diag.other_end(w, f);
590  if (diag.get_zxtype(n) == ZXType::Output) {
591  unsigned q = qubit_map.at(n);
592  if (diag.get_wire_type(w) == ZXWireType::H) {
593  diag.set_wire_type(w, ZXWireType::Basic);
594  circ.add_op<unsigned>(OpType::H, {q});
595  }
596  switch (diag.get_zxtype(f)) {
597  case ZXType::Input: {
598  break;
599  }
600  case ZXType::XY: {
601  const PhasedGen& f_gen = diag.get_vertex_ZXGen<PhasedGen>(f);
602  Expr ph = f_gen.get_param();
603  if (!equiv_0(ph)) {
604  circ.add_op<unsigned>(OpType::U1, -ph, {q});
605  }
606  diag.set_vertex_ZXGen_ptr(
607  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
608  break;
609  }
610  case ZXType::PX: {
611  const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
612  if (f_gen.get_param()) {
613  circ.add_op<unsigned>(OpType::Z, {q});
614  diag.set_vertex_ZXGen_ptr(
615  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
616  }
617  break;
618  }
619  case ZXType::PY: {
620  const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
621  circ.add_op<unsigned>(
622  f_gen.get_param() ? OpType::S : OpType::Sdg, {q});
623  diag.set_vertex_ZXGen_ptr(
624  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
625  break;
626  }
627  default:
628  throw ZXError(
629  "Error during extraction from ZX diagram: unexpected ZXType in "
630  "frontier");
631  }
632  } else if (frontier_lookup.find(n) != frontier_lookup.end()) {
633  czs.push_back({f, n});
634  diag.remove_wire(w);
635  }
636  }
637  }
638  for (const std::pair<ZXVert, ZXVert>& pair : czs) {
639  circ.add_op<unsigned>(
640  OpType::CZ, {qubit_map.at(pair.first), qubit_map.at(pair.second)});
641  }
642  ZXVertVec new_frontier;
643  for (const ZXVert& f : frontier) {
644  bool removed = false;
645  ZXVertVec ns = diag.neighbours(f);
646  if (ns.size() == 2) {
647  ZXType nt0 = diag.get_zxtype(ns.at(0));
648  ZXType nt1 = diag.get_zxtype(ns.at(1));
649  if ((nt0 == ZXType::Input && nt1 == ZXType::Output) ||
650  (nt0 == ZXType::Output && nt1 == ZXType::Input)) {
651  diag.add_wire(ns.at(0), ns.at(1));
652  diag.remove_vertex(f);
653  removed = true;
654  }
655  }
656  if (!removed && diag.get_zxtype(f) != ZXType::Input)
657  new_frontier.push_back(f);
658  }
659  frontier = new_frontier;
660 }

676 static void bipartite_complementation(
677  ZXDiagram& diag, const ZXVertSeqSet& sa, const ZXVertSeqSet& sb) {
678  for (const ZXVert& a : sa.get<TagSeq>()) {
679  for (const ZXVert& b : sb.get<TagSeq>()) {

689 void extend_if_input(
690  ZXDiagram& diag, const ZXVert& v, std::map<ZXVert, ZXVert>& input_qubits) {
691  std::map<ZXVert, ZXVert>::iterator found = input_qubits.find(v);
692  if (found != input_qubits.end()) {
693  ZXVert in = found->second;
694  ZXVert ext0 = diag.add_vertex(ZXType::XY);
695  ZXVert ext1 = diag.add_vertex(ZXType::XY);
696  diag.remove_wire(diag.adj_wires(in).at(0));
697  diag.add_wire(in, ext0);
698  diag.add_wire(ext0, ext1, ZXWireType::H);
699  diag.add_wire(ext1, v, ZXWireType::H);
700  input_qubits.erase(found);
701  input_qubits.insert({ext0, in});
702  }

705 bool remove_all_gadgets(
706  ZXDiagram& diag, const ZXVertVec& frontier,
707  std::map<ZXVert, ZXVert>& input_qubits) {
708  bool removed_gadget = false;
709  for (const ZXVert& f : frontier) {
710  ZXVertVec f_ns = diag.neighbours(f);
711  std::optional<ZXVert> found_output;
712  for (const ZXVert& n : f_ns) {
713  // Each frontier vertex is connected to a unique output, find it
714  if (diag.get_zxtype(n) == ZXType::Output) {
715  found_output = n;
716  break;
717  }
718  }
719  if (!found_output)
720  throw ZXError(
721  "Error during extraction from ZXDiagram: frontier vertex not "
722  "adjacent to an output");
723  ZXVert o = *found_output;
724  for (const ZXVert& n : f_ns) {
725  if (diag.get_zxtype(n) == ZXType::YZ) {
726  // Pivot
727  // Identify three subsets of neighbours
728  ZXVertSeqSet excl_f;
729  // Need to recalculate neighbours rather than use f_ns as we might have
730  // extended
731  extend_if_input(diag, f, input_qubits);
732  for (const ZXVert& n : diag.neighbours(f)) {
733  extend_if_input(diag, n, input_qubits);
734  excl_f.insert(n);
735  }
736  excl_f.get<TagKey>().erase(n);
737  excl_f.get<TagKey>().erase(o);
738  ZXVertSeqSet excl_n, joint;
739  auto& lookup_f = excl_f.get<TagKey>();
740  for (const ZXVert& nn : diag.neighbours(n)) {
741  extend_if_input(diag, nn, input_qubits);
742  if (lookup_f.find(nn) != lookup_f.end())
743  joint.insert(nn);
744  else
745  excl_n.insert(nn);
746  }
747  excl_n.get<TagKey>().erase(f);
748  for (const ZXVert& nn : joint.get<TagSeq>())
749  excl_f.get<TagKey>().erase(nn);
750  // The is_MBQC check in zx_to_circuit guarantees QuantumType::Quantum
751  bipartite_complementation(diag, joint, excl_n);
752  bipartite_complementation(diag, joint, excl_f);
753  bipartite_complementation(diag, excl_n, excl_f);
754  // In place of switching vertices f and n, we invert their
755  // connectivities
756  excl_n.insert(excl_f.begin(), excl_f.end());
757  bipartite_complementation(diag, {f}, excl_n);
758  bipartite_complementation(diag, {n}, excl_n);
759  Wire ow = *diag.wire_between(f, o);
760  diag.set_wire_type(
761  ow, (diag.get_wire_type(ow) == ZXWireType::Basic)
762  ? ZXWireType::H
763  : ZXWireType::Basic);
764  const PhasedGen& yz_gen = diag.get_vertex_ZXGen<PhasedGen>(n);
765  diag.set_vertex_ZXGen_ptr(
766  n, ZXGen::create_gen(
767  ZXType::XY, -yz_gen.get_param(), QuantumType::Quantum));
768  for (const ZXVert& nn : joint.get<TagSeq>()) {
769  ZXGen_ptr new_gen;
770  switch (diag.get_zxtype(nn)) {
771  case ZXType::XY: {
772  const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
773  new_gen = ZXGen::create_gen(
774  ZXType::XY, ph_gen.get_param() + 1., QuantumType::Quantum);
775  break;
776  }
777  case ZXType::PX:
778  case ZXType::PY: {
779  const CliffordGen& cl_gen =
780  diag.get_vertex_ZXGen<CliffordGen>(nn);
781  new_gen = ZXGen::create_gen(
782  cl_gen.get_type(), !cl_gen.get_param(), QuantumType::Quantum);
783  break;
784  }
785  case ZXType::XZ:
786  case ZXType::YZ: {
787  const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
788  new_gen = ZXGen::create_gen(
789  ph_gen.get_type(), -ph_gen.get_param(), QuantumType::Quantum);
790  break;
791  }
792  case ZXType::PZ: {
793  new_gen = diag.get_vertex_ZXGen_ptr(nn);
794  break;
795  }
796  default:
797  throw ZXError(
798  "Error during extraction from ZX diagram: unexpected ZXType "
799  "during local complementation");
800  }
801  diag.set_vertex_ZXGen_ptr(nn, new_gen);
802  }
803  removed_gadget = true;
804  break;
805  }
806  }
807  }
808  return removed_gadget;
809 }

811 Circuit zx_to_circuit(const ZXDiagram& d) {
812  ZXDiagram diag = d;
813  if (!diag.is_MBQC())
814  throw ZXError("Can only extract a circuit from a ZX diagram in MBQC form");
815  ZXVertVec ins = diag.get_boundary(ZXType::Input);
816  ZXVertVec outs = diag.get_boundary(ZXType::Output);
817  if (ins.size() != outs.size())
818  throw ZXError("Can only extract a circuit from a unitary ZX diagram");
819 
820  Circuit circ(ins.size());
821 
822  ZXVertVec frontier;
823  std::map<ZXVert, unsigned> qubit_map;
824  unsigned q = 0;
825  for (const ZXVert& o : outs) {
826  ZXVert f_i = diag.neighbours(o).at(0);
827  frontier.push_back(f_i);
828  qubit_map.insert({o, q});
829  qubit_map.insert({f_i, q});
830  ++q;
831  }
832  std::map<ZXVert, ZXVert> input_qubits;
833  for (const ZXVert& i : ins)
834  input_qubits.insert({diag.neighbours(i).at(0), i});
835 
836  clean_frontier(diag, frontier, circ, qubit_map);
837  while (!frontier.empty()) {
838  if (remove_all_gadgets(diag, frontier, input_qubits)) {
839  clean_frontier(diag, frontier, circ, qubit_map);
840  }
841  ZXVertSeqSet neighbours = neighbours_of_frontier(diag, frontier);
842  boost::bimap<ZXVert, unsigned> correctors, preserve, ys;
843  ZXVertVec to_solve;
844  for (const ZXVert& f : frontier) {
845  if (input_qubits.find(f) == input_qubits.end())
846  correctors.insert({f, (unsigned)correctors.size()});
847  }
848  for (const ZXVert& n : neighbours.get<TagSeq>()) {
849  ZXType n_type = diag.get_zxtype(n);
850  if (n_type == ZXType::XY || n_type == ZXType::PX ||
851  n_type == ZXType::PY) {
852  preserve.insert({n, (unsigned)preserve.size()});
853  to_solve.push_back(n);
854  }
855  }
856  std::map<ZXVert, ZXVertSeqSet> candidates =
857  Flow::gauss_solve_correctors(diag, correctors, preserve, to_solve, ys);
858 
859  if (candidates.empty())
860  throw ZXError(
861  "Error during extraction from ZX diagram: diagram does not have "
862  "gflow");
863 
864  unsigned min = UINT_MAX;
865  ZXVert best;
866  for (const std::pair<const ZXVert, ZXVertSeqSet>& p : candidates) {
867  if (p.second.size() < min) {
868  min = p.second.size();
869  best = p.first;
870  }
871  }
872  ZXVertSeqSet g_best = candidates.at(best);
873 
874  ZXVert f_to_isolate = g_best.get<TagSeq>().front();
875  unsigned f_q = qubit_map.at(f_to_isolate);
876  for (const ZXVert& f : g_best) {
877  if (f != f_to_isolate) {
878  circ.add_op<unsigned>(OpType::CX, {f_q, qubit_map.at(f)});
879  }
880  }
881  ZXVert out;
882  Wire w_out;
883  for (const Wire& w : diag.adj_wires(f_to_isolate)) {
884  ZXVert n = diag.other_end(w, f_to_isolate);
885  if (diag.get_zxtype(n) == ZXType::Output) {
886  out = n;
887  w_out = w;
888  break;
889  }
890  }
891  diag.add_wire(
892  best, out,
893  (diag.get_wire_type(w_out) == ZXWireType::Basic) ? ZXWireType::H
894  : ZXWireType::Basic);
895  diag.remove_vertex(f_to_isolate);
896  qubit_map.erase(qubit_map.find(f_to_isolate));
897  qubit_map.insert({best, f_q});
898  for (ZXVertVec::iterator it = frontier.begin(); it != frontier.end();
899  ++it) {
900  if (*it == f_to_isolate) {
901  *it = best;
902  break;
903  }
904  }
905 
906  clean_frontier(diag, frontier, circ, qubit_map);
907  }
908 
909  qubit_map_t qm;
910  for (unsigned i = 0; i < ins.size(); ++i) {
911  ZXVert in = ins.at(i);
912  ZXVert out = diag.neighbours(in).at(0);
913  if (diag.get_zxtype(out) != ZXType::Output)
914  throw ZXError(
915  "Error during extraction from ZX diagram: input not adjacent to "
916  "output after extracting");
917  qm.insert({Qubit(qubit_map.at(out)), Qubit(i)});
918  }
919  circ.permute_boundary_output(qm);
920 
921  // Reverse gates in circuit (all gates added are self-transpose)
922  return circ.transpose();
923 }
