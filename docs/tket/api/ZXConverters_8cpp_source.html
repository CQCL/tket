29bool is_spiderless_optype(const OpType& optype) {
30 return optype == OpType::Barrier || optype == OpType::SWAP ||
31 optype == OpType::noop;

34// Add a swicth with the on-state controlled by the on_value.
35// qtype indicates the quantum type of the switch.
36std::pair<ZXVertPort, ZXVertPort> add_switch(
37 ZXDiagram& zxd, const bool& on_value, const QuantumType& qtype) {
38 zxd.multiply_scalar(std::sqrt(2.));
39 ZXVert triangle = zxd.add_vertex(ZXType::Triangle, qtype);
40 ZXVert x = zxd.add_vertex(ZXType::XSpider, 0, qtype);
41 zxd.add_wire(triangle, x, ZXWireType::Basic, qtype, 1);
42 if (on_value) {
43 // Turn on the switch if the input is 1
44 ZXVert negate = zxd.add_vertex(ZXType::XSpider, 1, qtype);
45 zxd.add_wire(triangle, negate, ZXWireType::Basic, qtype, 0);
46 return {{negate, std::nullopt}, {x, std::nullopt}};
47 }
48 return {{triangle, 0}, {x, std::nullopt}};
49}

51// --o--s---o--
52// | |
53// n n
54// |--G-o-|
55// |
56// s
57// |-discard
58// s, n are switches, G is the conditional zx diagram specified by 'left' and
59// 'right'. s is on when the input is 0, n is on when the input is 1.
60// return the input and the output of the conditional zx along with a vector of
61// control spiders
62std::pair<std::pair<ZXVertPort, ZXVertPort>, ZXVertPortVec> add_conditional_zx(
63 ZXDiagram& zxd, const ZXVert& left, const ZXVert& right,
64 const QuantumType& qtype) {
65 ZXVert in = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
66 ZXVert out = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
67 ZXVert discard_ctr = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
68 ZXVert discard = zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
69 auto switch_s0 = add_switch(zxd, false, qtype);
70 auto switch_n0 = add_switch(zxd, true, qtype);
71 auto switch_n1 = add_switch(zxd, true, qtype);
72 auto switch_s1 = add_switch(zxd, false, qtype);
73 zxd.add_wire(in, switch_s0.second.first, ZXWireType::Basic, qtype);
74 zxd.add_wire(out, switch_s0.second.first, ZXWireType::Basic, qtype);
75 zxd.add_wire(in, switch_n0.second.first, ZXWireType::Basic, qtype);
76 zxd.add_wire(left, switch_n0.second.first, ZXWireType::Basic, qtype);
77 zxd.add_wire(right, discard_ctr, ZXWireType::Basic, qtype);
78 zxd.add_wire(discard_ctr, switch_n1.second.first, ZXWireType::Basic, qtype);
79 zxd.add_wire(out, switch_n1.second.first, ZXWireType::Basic, qtype);
80 zxd.add_wire(discard_ctr, switch_s1.second.first, ZXWireType::Basic, qtype);
81 zxd.add_wire(discard, switch_s1.second.first, ZXWireType::Basic, qtype);
82 return {
83 {{in, std::nullopt}, {out, std::nullopt}},
84 {switch_s0.first, switch_s1.first, switch_n0.first, switch_n1.first}};
85}

87// n-bit AND spider to control the switches
88// https://arxiv.org/abs/1910.06818
89std::pair<ZXVertPortVec, ZXVertPort> add_n_bit_and(
90 ZXDiagram& zxd, unsigned n, const QuantumType& qtype) {
91 TKET_ASSERT(n > 1);
92 ZXVert z_vert = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
93 // Add Triangle -1
94 ZXVert z_pi_0 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
95 ZXVert tri_1 = zxd.add_vertex(ZXType::Triangle, qtype);
96 ZXVert z_pi_1 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
97 zxd.add_wire(z_vert, z_pi_0, ZXWireType::Basic, qtype);
98 zxd.add_wire(tri_1, z_pi_0, ZXWireType::Basic, qtype, 0);
99 zxd.add_wire(tri_1, z_pi_1, ZXWireType::Basic, qtype, 1);
100 ZXVertPortVec inputs;
101 for (unsigned i = 0; i < n; i++) {
102 ZXVert tri_0 = zxd.add_vertex(ZXType::Triangle, qtype);
103 zxd.add_wire(tri_0, z_vert, ZXWireType::Basic, qtype, 1);
104 inputs.push_back({tri_0, 0});
105 }
106 return {inputs, {z_pi_1, std::nullopt}};
107}

109// Add converted circ into zxd. Set add_boundary to true
110// if boundary spiders are to be added to the boundary.
111BoundaryVertMap circuit_to_zx_recursive(
112 const Circuit& circ, ZXDiagram& zxd, bool add_boundary) {
113 std::map<TypedVertPort, ZXVertPort> vert_lookup;
114 std::map<VertPort, ZXVertPort> boolean_outport_lookup;
115 BoundaryVertMap bmap;
116
117 // Convert each vertex to ZXDiagram, raise error if not supported
118 BGL_FORALL_VERTICES(vert, circ.dag, DAG) {
119 // We currently throw an error if the vertex is either classical or flow
120 Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
121 if (is_flowop_type(op->get_type()) || is_classical_type(op->get_type())) {
122 throw Unsupported(
123 "Cannot convert OpType: " + op->get_name() + " to a ZX node. \n");
124 }
125 switch (op->get_type()) {
126 case OpType::Input: {
127 ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Quantum);
128 if (add_boundary) zxd.add_boundary(zx_vert);
129 vert_lookup.insert(
130 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
131 bmap.insert({zx_vert, vert});
132 break;
133 }
134 case OpType::Output: {
135 ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Quantum);
136 if (add_boundary) zxd.add_boundary(zx_vert);
137 vert_lookup.insert(
138 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
139 bmap.insert({zx_vert, vert});
140 break;
141 }
142 case OpType::ClInput: {
143 ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Classical);
144 if (add_boundary) zxd.add_boundary(zx_vert);
145 vert_lookup.insert(
146 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
147 bmap.insert({zx_vert, vert});
148 break;
149 }
150 case OpType::ClOutput: {
151 ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Classical);
152 if (add_boundary) zxd.add_boundary(zx_vert);
153 vert_lookup.insert(
154 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
155 bmap.insert({zx_vert, vert});
156 break;
157 }
158 // Spiderless ops are handled during vertex wiring
159 case OpType::Barrier:
160 case OpType::noop:
161 case OpType::SWAP:
162 case OpType::Phase: {
163 continue;
164 }
165 case OpType::H: {
166 ZXVert zx_vert = zxd.add_vertex(ZXType::Hbox, QuantumType::Quantum);
167 vert_lookup.insert(
168 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
169 vert_lookup.insert(
170 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
171 zxd.multiply_scalar(0.5);
172 break;
173 }
174 case OpType::Rz: {
175 ZXVert zx_vert = zxd.add_vertex(
176 ZXType::ZSpider, op->get_params()[0], QuantumType::Quantum);
177 vert_lookup.insert(
178 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
179 vert_lookup.insert(
180 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
181 break;
182 }
183 case OpType::Rx: {
184 ZXVert zx_vert = zxd.add_vertex(
185 ZXType::XSpider, op->get_params()[0], QuantumType::Quantum);
186 vert_lookup.insert(
187 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
188 vert_lookup.insert(
189 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
190 break;
191 }
192 case OpType::X: {
193 ZXVert zx_vert =
194 zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Quantum);
195 vert_lookup.insert(
196 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
197 vert_lookup.insert(
198 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
199 break;
200 }
201 case OpType::Z: {
202 ZXVert zx_vert =
203 zxd.add_vertex(ZXType::ZSpider, 1, QuantumType::Quantum);
204 vert_lookup.insert(
205 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
206 vert_lookup.insert(
207 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
208 break;
209 }
210 case OpType::CX: {
211 ZXVert zx_x_vert =
212 zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
213 ZXVert zx_z_vert =
214 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
215 zxd.add_wire(zx_x_vert, zx_z_vert);
216 vert_lookup.insert(
217 {{{vert, 0}, ZXPortType::In}, {zx_z_vert, std::nullopt}});
218 vert_lookup.insert(
219 {{{vert, 0}, ZXPortType::Out}, {zx_z_vert, std::nullopt}});
220 vert_lookup.insert(
221 {{{vert, 1}, ZXPortType::In}, {zx_x_vert, std::nullopt}});
222 vert_lookup.insert(
223 {{{vert, 1}, ZXPortType::Out}, {zx_x_vert, std::nullopt}});
224 zxd.multiply_scalar(2);
225 break;
226 }
227 case OpType::CZ: {
228 ZXVert zx_za_vert =
229 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
230 ZXVert zx_zb_vert =
231 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
232 zxd.add_wire(zx_za_vert, zx_zb_vert, ZXWireType::H);
233 vert_lookup.insert(
234 {{{vert, 0}, ZXPortType::In}, {zx_za_vert, std::nullopt}});
235 vert_lookup.insert(
236 {{{vert, 0}, ZXPortType::Out}, {zx_za_vert, std::nullopt}});
237 vert_lookup.insert(
238 {{{vert, 1}, ZXPortType::In}, {zx_zb_vert, std::nullopt}});
239 vert_lookup.insert(
240 {{{vert, 1}, ZXPortType::Out}, {zx_zb_vert, std::nullopt}});
241 break;
242 }
243 case OpType::Measure: {
244 // Add a decoherence node
245 ZXVert zx_measure_vert =
246 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
247 // Add a delete operator
248 ZXVert zx_delete_vert =
249 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
250 vert_lookup.insert(
251 {{{vert, 0}, ZXPortType::In}, {zx_measure_vert, std::nullopt}});
252 vert_lookup.insert(
253 {{{vert, 0}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
254 vert_lookup.insert(
255 {{{vert, 1}, ZXPortType::In}, {zx_delete_vert, std::nullopt}});
256 vert_lookup.insert(
257 {{{vert, 1}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
258 break;
259 }
260 case OpType::Reset: {
261 // Discard
262 ZXVert zx_discard_vert =
263 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
264 // Add a node to prepare |0>
265 ZXVert zx_reset_vert =
266 zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
267 zxd.multiply_scalar(0.5);
268 vert_lookup.insert(
269 {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
270 vert_lookup.insert(
271 {{{vert, 0}, ZXPortType::Out}, {zx_reset_vert, std::nullopt}});
272 break;
273 }
274 case OpType::Collapse: {
275 ZXVert zx_vert =
276 zxd.add_vertex(ZXType::ZSpider, QuantumType::Classical);
277 vert_lookup.insert(
278 {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
279 vert_lookup.insert(
280 {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
281 break;
282 }
283 case OpType::Create: {
284 ZXVert zx_init_vert =
285 zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
286 zxd.multiply_scalar(0.5);
287 vert_lookup.insert(
288 {{{vert, 0}, ZXPortType::Out}, {zx_init_vert, std::nullopt}});
289 break;
290 }
291 case OpType::Discard: {
292 ZXVert zx_discard_vert =
293 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
294 vert_lookup.insert(
295 {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
296 break;
297 }
298 case OpType::Conditional: {
299 // Stores the condition for each port index
300 std::vector<bool> port_conditions;
301
302 op_signature_t parent_sig = op->get_signature();
303
304 // Find the nested quantum op and its overall conditions
305 // Assume bool ports are always the first few ports
306 while (op->get_type() == OpType::Conditional) {
307 const Conditional& cond = static_cast<const Conditional&>(*op);
308 for (unsigned i = 0; i < cond.get_width(); i++) {
309 bool set = cond.get_value() & (1 << (cond.get_width() - i - 1));
310 port_conditions.push_back(set);
311 }
312 op = cond.get_op();
313 }
314 if (is_barrier_type(op->get_type())) {
315 throw Unsupported(
316 "Cannot convert conditional barrier operations to a ZX node. \n");
317 }
318
319 unsigned port_conditions_size =
320 static_cast<unsigned>(port_conditions.size());
321 // Convert the underlying op to zx.
322 // If the op is a quantum gate, construct a 1 gate circuit
323 // If the op is a box, obtain its circuit decomposition
324 op_signature_t inner_sig = op->get_signature();
325 Circuit replacement;
326 if (is_box_type(op->get_type())) {
327 const Box& b = static_cast<const Box&>(*op);
328 replacement = *b.to_circuit();
329 } else {
330 unit_vector_t args;
331 unsigned q_index = 0;
332 unsigned c_index = 0;
333 for (const EdgeType& etype : inner_sig) {
334 if (etype == EdgeType::Quantum) {
335 args.push_back(Qubit(q_index++));
336 } else if (etype == EdgeType::Classical) {
337 args.push_back(Bit(c_index++));
338 }
339 TKET_ASSERT(etype != EdgeType::Boolean);
340 }
341 replacement = Circuit(q_index, c_index);
342 replacement.add_op(op, args);
343 }
344 BoundaryVertMap box_bm =
345 circuit_to_zx_recursive(replacement, zxd, false);
346
347 // The Z spider controlling all switches
348 ZXVert master_switch =
349 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
350
351 // For each qubit/bit path in the inner op, convert it into a
352 // conditional path.
353 unsigned q_index = 0;
354 unsigned c_index = 0;
355 for (unsigned i = 0; i < inner_sig.size(); i++) {
356 QuantumType qtype;
357 UnitID reg;
358 if (inner_sig[i] == EdgeType::Quantum) {
359 qtype = QuantumType::Quantum;
360 reg = Qubit(q_index++);
361 } else {
362 qtype = QuantumType::Classical;
363 reg = Bit(c_index++);
364 }
365 std::pair<ZXVertPort, ZXVertPort> boundary;
366 ZXVertPortVec controls;
367 Vertex inp = replacement.get_in(reg);
368 ZXVert zx_inp = box_bm.right.at(inp);
369 Vertex outp = replacement.get_out(reg);
370 ZXVert zx_outp = box_bm.right.at(outp);
371 // Convert the path between zx_inp and zx_outp into a conditional
372 // path
373 std::tie(boundary, controls) =
374 add_conditional_zx(zxd, zx_inp, zx_outp, qtype);
375 // Connect each switch to the master switch
376 for (const ZXVertPort& ctr : controls) {
377 zxd.add_wire(
378 ctr.first, master_switch, ZXWireType::Basic,
379 zxd.get_qtype(ctr.first).value(), ctr.second);
380 }
381 // Update lookup
382 TKET_ASSERT(parent_sig[i + port_conditions_size] == inner_sig[i]);
383 // Since we assume that the conditions always use the first few
384 // ports, the gate path should use the i + port_conditions.size()
385 // port.
386 vert_lookup.insert(
387 {{{vert, i + port_conditions_size}, ZXPortType::In},
388 boundary.first});
389 vert_lookup.insert(
390 {{{vert, i + port_conditions_size}, ZXPortType::Out},
391 boundary.second});
392 }
393 // Use either a Z spider or a AND operator to connect the master
394 // switch and the boolean inputs
395 ZXVertPortVec and_inputs;
396 if (port_conditions_size == 1) {
397 and_inputs.push_back({master_switch, std::nullopt});
398 } else {
399 ZXVertPort and_output;
400 std::tie(and_inputs, and_output) =
401 add_n_bit_and(zxd, port_conditions_size, QuantumType::Classical);
402 zxd.add_wire(
403 and_output.first, master_switch, ZXWireType::Basic,
404 QuantumType::Classical, and_output.second);
405 }
406 // Connect inputs to the nodes obtained from above
407 for (unsigned i = 0; i < port_conditions_size; i++) {
408 // Each boolean edge shares a source port with other
409 // classical/boolean edges. Use the classical Z spider to explicitly
410 // implement this copy operation
411 Edge in_edge = circ.get_nth_in_edge(vert, i);
412 port_t source_port = circ.get_source_port(in_edge);
413 Vertex vert_s = circ.source(in_edge);
414 // During the wiring stage, each edge originated from {vert_s,
415 // p_s} needs go through copy_vert
416 if (boolean_outport_lookup.find({vert_s, source_port}) ==
417 boolean_outport_lookup.end()) {
418 ZXVert copy_vert =
419 zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
420 boolean_outport_lookup.insert(
421 {{vert_s, source_port}, {copy_vert, std::nullopt}});
422 }
423 if (port_conditions[i]) {
424 vert_lookup.insert({{{vert, i}, ZXPortType::In}, and_inputs[i]});
425 } else {
426 ZXVert x_vert =
427 zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Classical);
428 zxd.add_wire(
429 and_inputs[i].first, x_vert, ZXWireType::Basic,
430 QuantumType::Classical, and_inputs[i].second);
431 vert_lookup.insert(
432 {{{vert, i}, ZXPortType::In}, {x_vert, std::nullopt}});
433 }
434 }
435 break;
436 }
437 default:
438 if (is_box_type(op->get_type())) {
439 EdgeVec b_in_holes =
440 circ.get_in_edges_of_type(vert, EdgeType::Boolean);
441 if (b_in_holes.size() > 0) {
442 throw Unsupported("Cannot convert box type: " + op->get_name());
443 }
444 const Box& b = static_cast<const Box&>(*op);
445 Circuit replacement = *b.to_circuit();
446 // First, add the converted box to diagram.
447 BoundaryVertMap box_bm =
448 circuit_to_zx_recursive(replacement, zxd, false);
449 // Then, map the vertports in the box boundary to zx nodes.
450 // Assume that a box can't have Boolean input edges, and all Boolean
451 // output edges share ports with Classical edges. Therefore we don't
452 // have to map Boolean vertports.
453 EdgeVec q_in_holes =
454 circ.get_in_edges_of_type(vert, EdgeType::Quantum);
455 EdgeVec q_out_holes =
456 circ.get_out_edges_of_type(vert, EdgeType::Quantum);
457 EdgeVec c_in_holes =
458 circ.get_in_edges_of_type(vert, EdgeType::Classical);
459 EdgeVec c_out_holes =
460 circ.get_out_edges_of_type(vert, EdgeType::Classical);
461
462 for (unsigned i = 0; i < q_in_holes.size(); i++) {
463 port_t port = circ.get_target_port(q_in_holes[i]);
464 Vertex inp = replacement.get_in(Qubit(i));
465 vert_lookup.insert(
466 {{{vert, port}, ZXPortType::In},
467 {box_bm.right.at(inp), std::nullopt}});
468 }
469 for (unsigned i = 0; i < q_out_holes.size(); i++) {
470 port_t port = circ.get_source_port(q_out_holes[i]);
471 Vertex outp = replacement.get_out(Qubit(i));
472 vert_lookup.insert(
473 {{{vert, port}, ZXPortType::Out},
474 {box_bm.right.at(outp), std::nullopt}});
475 }
476 for (unsigned i = 0; i < c_in_holes.size(); i++) {
477 port_t port = circ.get_target_port(c_in_holes[i]);
478 Vertex inp = replacement.get_in(Bit(i));
479 vert_lookup.insert(
480 {{{vert, port}, ZXPortType::In},
481 {box_bm.right.at(inp), std::nullopt}});
482 }
483 for (unsigned i = 0; i < c_out_holes.size(); i++) {
484 port_t port = circ.get_source_port(c_out_holes[i]);
485 Vertex outp = replacement.get_out(Bit(i));
486 vert_lookup.insert(
487 {{{vert, port}, ZXPortType::Out},
488 {box_bm.right.at(outp), std::nullopt}});
489 }
490 } else {
491 throw Unsupported(
492 "Cannot convert gate type: " + op->get_name() +
493 " to a ZX node, try rebase the gates to use Rx, Rz, "
494 "X, Z, H, CZ "
495 "or CX. \n");
496 }
497 }
498 }

500 // Connect the ZX nodes
501 BGL_FORALL_EDGES(edge, circ.dag, DAG) {
502 Vertex v_s = circ.source(edge);
503 Vertex v_t = circ.target(edge);
504 port_t p_s = circ.get_source_port(edge);
505 port_t p_t = circ.get_target_port(edge);
506 // Handle Spiderless ops
507 if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_s))) {
508 // We only handle in-edges
509 continue;
510 }
511 if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t))) {
512 // Traverse the path to find the next non-spiderless op
513 Edge next_e = edge;
514 do {
515 if (circ.get_OpType_from_Vertex(v_t) == OpType::SWAP) {
516 next_e = circ.get_nth_out_edge(
517 v_t, (circ.get_target_port(next_e) + 1) % 2);
518 v_t = circ.target(next_e);
519 } else {
520 std::tie(v_t, next_e) = circ.get_next_pair(v_t, next_e);
521 }
522 } while (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t)));
523 p_t = circ.get_target_port(next_e);
524 }
525
526 ZXVertPort zx_vp_s =
527 vert_lookup.at(TypedVertPort(VertPort(v_s, p_s), ZXPortType::Out));
528 ZXVertPort zx_vp_t =
529 vert_lookup.at(TypedVertPort(VertPort(v_t, p_t), ZXPortType::In));
530 if (circ.get_edgetype(edge) == EdgeType::Quantum) {
531 zxd.add_wire(
532 zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic, QuantumType::Quantum,
533 zx_vp_s.second, zx_vp_t.second);
534 } else {
535 auto bool_it = boolean_outport_lookup.find(VertPort(v_s, p_s));
536 if (bool_it == boolean_outport_lookup.end()) {
537 zxd.add_wire(
538 zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic,
539 QuantumType::Classical, zx_vp_s.second, zx_vp_t.second);
540 } else {
541 // If the source port is boolean, then connect the copy spider to the
542 // source vertex. All out-edges originated from the source port should
543 // be connected to the copy spider.
544 if (zxd.degree(bool_it->second.first) == 0) {
545 zxd.add_wire(
546 zx_vp_s.first, bool_it->second.first, ZXWireType::Basic,
547 QuantumType::Classical, zx_vp_s.second, bool_it->second.second);
548 }
549 zxd.add_wire(
550 bool_it->second.first, zx_vp_t.first, ZXWireType::Basic,
551 QuantumType::Classical, bool_it->second.second, zx_vp_t.second);
552 }
553 }
554 }
555 return bmap;
556}

558std::pair<ZXDiagram, BoundaryVertMap> circuit_to_zx(const Circuit& circ) {
559 ZXDiagram zxd;
560 BoundaryVertMap bmap = circuit_to_zx_recursive(circ, zxd, true);
561 // Remove internal boundary vertices produced by the recursion
562 ZXVertVec true_boundary = zxd.get_boundary();
563 ZXVertIterator vi, vi_end, next;
564 tie(vi, vi_end) = boost::vertices(*zxd.get_graph());
565 for (next = vi; vi != vi_end; vi = next) {
566 ++next;
567 if ((zxd.get_zxtype(*vi) == ZXType::Input ||
568 zxd.get_zxtype(*vi) == ZXType::Output) &&
569 std::find(true_boundary.begin(), true_boundary.end(), *vi) ==
570 true_boundary.end()) {
571 WireVec adj_wires = zxd.adj_wires(*vi);
572 TKET_ASSERT(adj_wires.size() == 2);
573 TKET_ASSERT(zxd.get_qtype(adj_wires[0]) == zxd.get_qtype(adj_wires[1]));
574 TKET_ASSERT(
575 zxd.get_wire_type(adj_wires[0]) == zxd.get_wire_type(adj_wires[1]));
576 ZXVertVec neighbours = zxd.neighbours(*vi);
577 zxd.add_wire(
578 neighbours[0], neighbours[1], zxd.get_wire_type(adj_wires[0]),
579 zxd.get_qtype(adj_wires[0]));
580 zxd.remove_vertex(*vi);
581 }
582 }
583 return {std::move(zxd), std::move(bmap)};
584}

586void clean_frontier(
587 ZXDiagram& diag, ZXVertVec& frontier, Circuit& circ,
588 std::map<ZXVert, unsigned>& qubit_map,
589 std::map<ZXVert, ZXVert>& input_qubits) {
590 std::set<ZXVert> frontier_lookup;
591 std::list<std::pair<ZXVert, ZXVert>> czs;
592 for (const ZXVert& f : frontier) {
593 frontier_lookup.insert(f);
594 for (const Wire& w : diag.adj_wires(f)) {
595 ZXVert n = diag.other_end(w, f);
596 if (diag.get_zxtype(n) == ZXType::Output) {
597 unsigned q = qubit_map.at(n);
598 if (diag.get_wire_type(w) == ZXWireType::H) {
599 diag.set_wire_type(w, ZXWireType::Basic);
600 circ.add_op<unsigned>(OpType::H, {q});
601 }
602 switch (diag.get_zxtype(f)) {
603 case ZXType::Input: {
604 break;
605 }
606 case ZXType::XY: {
607 const PhasedGen& f_gen = diag.get_vertex_ZXGen<PhasedGen>(f);
608 Expr ph = f_gen.get_param();
609 if (!equiv_0(ph)) {
610 circ.add_op<unsigned>(OpType::U1, -ph, {q});
611 }
612 diag.set_vertex_ZXGen_ptr(
613 f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
614 break;
615 }
616 case ZXType::PX: {
617 const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
618 if (f_gen.get_param()) {
619 circ.add_op<unsigned>(OpType::Z, {q});
620 diag.set_vertex_ZXGen_ptr(
621 f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
622 }
623 break;
624 }
625 case ZXType::PY: {
626 const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
627 circ.add_op<unsigned>(
628 f_gen.get_param() ? OpType::S : OpType::Sdg, {q});
629 diag.set_vertex_ZXGen_ptr(
630 f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
631 break;
632 }
633 default:
634 throw ZXError(
635 "Error during extraction from ZX diagram: unexpected ZXType in "
636 "frontier");
637 }
638 } else if (frontier_lookup.find(n) != frontier_lookup.end()) {
639 czs.push_back({f, n});
640 diag.remove_wire(w);
641 }
642 }
643 }
644 for (const std::pair<ZXVert, ZXVert>& pair : czs) {
645 circ.add_op<unsigned>(
646 OpType::CZ, {qubit_map.at(pair.first), qubit_map.at(pair.second)});
647 }
648 ZXVertVec new_frontier;
649 for (const ZXVert& f : frontier) {
650 bool removed = false;
651 ZXVertVec ns = diag.neighbours(f);
652 if (ns.size() == 2) {
653 ZXType nt0 = diag.get_zxtype(ns.at(0));
654 ZXType nt1 = diag.get_zxtype(ns.at(1));
655 if ((nt0 == ZXType::Input && nt1 == ZXType::Output) ||
656 (nt0 == ZXType::Output && nt1 == ZXType::Input)) {
657 diag.add_wire(ns.at(0), ns.at(1));
658 diag.remove_vertex(f);
659 auto input_found = input_qubits.find(f);
660 if (input_found != input_qubits.end()) input_qubits.erase(f);
661 removed = true;
662 }
663 }
664 if (!removed && diag.get_zxtype(f) != ZXType::Input)
665 new_frontier.push_back(f);
666 }
667 frontier = new_frontier;
668}

670ZXVertSeqSet neighbours_of_frontier(
671 const ZXDiagram& diag, const ZXVertVec& frontier) {
672 ZXVertSeqSet n_set;
673 for (const ZXVert& f : frontier) {
674 for (const Wire& w : diag.adj_wires(f)) {
675 ZXVert n = diag.other_end(w, f);
676 ZXType n_type = diag.get_zxtype(n);
677 if (n_type == ZXType::Output || n_type == ZXType::Input) continue;
678 n_set.insert(n);
679 }
680 }

684static void bipartite_complementation(
685 ZXDiagram& diag, const ZXVertSeqSet& sa, const ZXVertSeqSet& sb) {
686 for (const ZXVert& a : sa.get<TagSeq>()) {
687 for (const ZXVert& b : sb.get<TagSeq>()) {
688 std::optional<Wire> wire = diag.wire_between(a, b);
689 if (wire)
690 diag.remove_wire(*wire);
691 else
692 diag.add_wire(a, b, ZXWireType::H);
693 }
694 }

697void extend_if_input(
698 ZXDiagram& diag, const ZXVert& v, std::map<ZXVert, ZXVert>& input_qubits) {
699 std::map<ZXVert, ZXVert>::iterator found = input_qubits.find(v);
700 if (found != input_qubits.end()) {
701 ZXVert in = found->second;
702 ZXVert ext0 = diag.add_vertex(ZXType::XY);
703 ZXVert ext1 = diag.add_vertex(ZXType::XY);
704 diag.remove_wire(diag.adj_wires(in).at(0));
705 diag.add_wire(in, ext0);
706 diag.add_wire(ext0, ext1, ZXWireType::H);
707 diag.add_wire(ext1, v, ZXWireType::H);
708 input_qubits.erase(found);
709 input_qubits.insert({ext0, in});
710 }

713bool remove_all_gadgets(
714 ZXDiagram& diag, const ZXVertVec& frontier,
715 std::map<ZXVert, ZXVert>& input_qubits) {
716 bool removed_gadget = false;
717 for (const ZXVert& f : frontier) {
718 ZXVertVec f_ns = diag.neighbours(f);
719 std::optional<ZXVert> found_output;
720 for (const ZXVert& n : f_ns) {
721 // Each frontier vertex is connected to a unique output, find it
722 if (diag.get_zxtype(n) == ZXType::Output) {
723 found_output = n;
724 break;
725 }
726 }
727 if (!found_output)
728 throw ZXError(
729 "Error during extraction from ZXDiagram: frontier vertex not "
730 "adjacent to an output");
731 ZXVert o = *found_output;
732 for (const ZXVert& n : f_ns) {
733 if (diag.get_zxtype(n) == ZXType::YZ) {
734 // Pivot
735 // Identify three subsets of neighbours
736 ZXVertSeqSet excl_f;
737 // Need to recalculate neighbours rather than use f_ns as we might have
738 // extended
739 extend_if_input(diag, f, input_qubits);
740 for (const ZXVert& n : diag.neighbours(f)) {
741 extend_if_input(diag, n, input_qubits);
742 excl_f.insert(n);
743 }
744 auto& lookup_f = excl_f.get<TagKey>();
745 lookup_f.erase(lookup_f.find(n));
746 lookup_f.erase(lookup_f.find(o));
747 ZXVertSeqSet excl_n, joint;
748 for (const ZXVert& nn : diag.neighbours(n)) {
749 extend_if_input(diag, nn, input_qubits);
750 if (lookup_f.find(nn) != lookup_f.end())
751 joint.insert(nn);
752 else
753 excl_n.insert(nn);
754 }
755 auto& lookup_n = excl_n.get<TagKey>();
756 lookup_n.erase(lookup_n.find(f));
757 for (const ZXVert& nn : joint.get<TagSeq>()) {
758 auto found = lookup_f.find(nn);
759 if (found != lookup_f.end()) lookup_f.erase(found);
760 }
761 // The is_MBQC check in zx_to_circuit guarantees QuantumType::Quantum
762 bipartite_complementation(diag, joint, excl_n);
763 bipartite_complementation(diag, joint, excl_f);
764 bipartite_complementation(diag, excl_n, excl_f);
765 // In place of switching vertices f and n, we invert their
766 // connectivities
767 for (const ZXVert& v : excl_f.get<TagSeq>()) excl_n.insert(v);
768 bipartite_complementation(diag, {f}, excl_n);
769 bipartite_complementation(diag, {n}, excl_n);
770 Wire ow = *diag.wire_between(f, o);
771 diag.set_wire_type(
772 ow, (diag.get_wire_type(ow) == ZXWireType::Basic)
773 ? ZXWireType::H
774 : ZXWireType::Basic);
775 const PhasedGen& yz_gen = diag.get_vertex_ZXGen<PhasedGen>(n);
776 diag.set_vertex_ZXGen_ptr(
777 n, ZXGen::create_gen(
778 ZXType::XY, -yz_gen.get_param(), QuantumType::Quantum));
779 for (const ZXVert& nn : joint.get<TagSeq>()) {
780 ZXGen_ptr new_gen;
781 switch (diag.get_zxtype(nn)) {
782 case ZXType::XY: {
783 const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
784 new_gen = ZXGen::create_gen(
785 ZXType::XY, ph_gen.get_param() + 1., QuantumType::Quantum);
786 break;
787 }
788 case ZXType::PX:
789 case ZXType::PY: {
790 const CliffordGen& cl_gen =
791 diag.get_vertex_ZXGen<CliffordGen>(nn);
792 new_gen = ZXGen::create_gen(
793 cl_gen.get_type(), !cl_gen.get_param(), QuantumType::Quantum);
794 break;
795 }
796 case ZXType::XZ:
797 case ZXType::YZ: {
798 const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
799 new_gen = ZXGen::create_gen(
800 ph_gen.get_type(), -ph_gen.get_param(), QuantumType::Quantum);
801 break;
802 }
803 case ZXType::PZ: {
804 new_gen = diag.get_vertex_ZXGen_ptr(nn);
805 break;
806 }
807 default:
808 throw ZXError(
809 "Error during extraction from ZX diagram: unexpected ZXType "
810 "during local complementation");
811 }
812 diag.set_vertex_ZXGen_ptr(nn, new_gen);
813 }
814 removed_gadget = true;
815 break;
816 }
817 }
818 }
819 return removed_gadget;
820}

822Circuit zx_to_circuit(const ZXDiagram& d) {
823 ZXDiagram diag = d;
824 if (!diag.is_MBQC())
825 throw ZXError("Can only extract a circuit from a ZX diagram in MBQC form");
826 ZXVertVec ins = diag.get_boundary(ZXType::Input);
827 ZXVertVec outs = diag.get_boundary(ZXType::Output);
828 if (ins.size() != outs.size())
829 throw ZXError("Can only extract a circuit from a unitary ZX diagram");
830
831 BGL_FORALL_VERTICES(v, *diag.get_graph(), ZXGraph) {
832 ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
833 if (vgen->get_type() == ZXType::PY) {
834 const CliffordGen& cgen = dynamic_cast<const CliffordGen&>(*vgen);
835 diag.set_vertex_ZXGen_ptr(
836 v, ZXGen::create_gen(
837 ZXType::XY, cgen.get_param() ? Expr(-0.5) : Expr(0.5),
838 *cgen.get_qtype()));
839 }
840 }
841
842 Circuit circ(ins.size());
843
844 ZXVertVec frontier;
845 std::map<ZXVert, unsigned> qubit_map;
846 unsigned q = 0;
847 for (const ZXVert& o : outs) {
848 ZXVert f_i = diag.neighbours(o).at(0);
849 frontier.push_back(f_i);
850 qubit_map.insert({o, q});
851 qubit_map.insert({f_i, q});
852 ++q;
853 }
854 std::map<ZXVert, ZXVert> input_qubits;
855 for (const ZXVert& i : ins)
856 input_qubits.insert({diag.neighbours(i).at(0), i});
857
858 clean_frontier(diag, frontier, circ, qubit_map, input_qubits);
859 while (!frontier.empty()) {
860 if (remove_all_gadgets(diag, frontier, input_qubits)) {
861 clean_frontier(diag, frontier, circ, qubit_map, input_qubits);
862 }
863 ZXVertSeqSet neighbours = neighbours_of_frontier(diag, frontier);
864 boost::bimap<ZXVert, unsigned> correctors, preserve, ys;
865 ZXVertVec to_solve;
866 for (const ZXVert& f : frontier) {
867 if (input_qubits.find(f) == input_qubits.end())
868 correctors.insert({f, (unsigned)correctors.size()});
869 }
870 for (const ZXVert& n : neighbours.get<TagSeq>()) {
871 ZXType n_type = diag.get_zxtype(n);
872 if (n_type == ZXType::XY || n_type == ZXType::PX) {
873 preserve.insert({n, (unsigned)preserve.size()});
874 to_solve.push_back(n);
875 }
876 }
877 std::map<ZXVert, ZXVertSeqSet> candidates =
878 Flow::gauss_solve_correctors(diag, correctors, preserve, to_solve, ys);
879
880 if (candidates.empty())
881 throw ZXError(
882 "Error during extraction from ZX diagram: diagram does not have "
883 "gflow");
884
885 unsigned min = UINT_MAX;
886 ZXVert best;
887 for (const ZXVert& v : to_solve) {
888 auto found = candidates.find(v);
889 if (found != candidates.end() && found->second.size() < min) {
890 min = found->second.size();
891 best = v;
892 }
893 }
894 ZXVertSeqSet g_best = candidates.at(best);
895
896 ZXVert f_to_isolate = g_best.get<TagSeq>().front();
897 unsigned f_q = qubit_map.at(f_to_isolate);
898 for (const ZXVert& f : g_best.get<TagSeq>()) {
899 if (f != f_to_isolate) {
900 circ.add_op<unsigned>(OpType::CX, {f_q, qubit_map.at(f)});
901 }
902 }
903 ZXVert out;
904 Wire w_out;
905 for (const Wire& w : diag.adj_wires(f_to_isolate)) {
906 ZXVert n = diag.other_end(w, f_to_isolate);
907 if (diag.get_zxtype(n) == ZXType::Output) {
908 out = n;
909 w_out = w;
910 break;
911 }
912 }
913 diag.add_wire(
914 best, out,
915 (diag.get_wire_type(w_out) == ZXWireType::Basic) ? ZXWireType::H
916 : ZXWireType::Basic);
917 diag.remove_vertex(f_to_isolate);
918 qubit_map.erase(qubit_map.find(f_to_isolate));
919 qubit_map.insert({best, f_q});
920 for (ZXVertVec::iterator it = frontier.begin(); it != frontier.end();
921 ++it) {
922 if (*it == f_to_isolate) {
923 *it = best;
924 break;
925 }
926 }
927
928 clean_frontier(diag, frontier, circ, qubit_map, input_qubits);
929 }
930
931 qubit_map_t qm;
932 for (unsigned i = 0; i < ins.size(); ++i) {
933 ZXVert in = ins.at(i);
934 ZXVert out = diag.neighbours(in).at(0);
935 if (diag.get_zxtype(out) != ZXType::Output)
936 throw ZXError(
937 "Error during extraction from ZX diagram: input not adjacent to "
938 "output after extracting");
939 qm.insert({Qubit(qubit_map.at(out)), Qubit(i)});
940 }
941 circ.permute_boundary_output(qm);
942
943 // Reverse gates in circuit (all gates added are self-transpose)
944 return circ.transpose();
945}
