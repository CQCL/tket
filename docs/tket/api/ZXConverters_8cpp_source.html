35 // Add a swicth with the on-state controlled by the on_value.
36 // qtype indicates the quantum type of the switch.
37 std::pair<ZXVertPort, ZXVertPort> add_switch(
38  ZXDiagram& zxd, const bool& on_value, const QuantumType& qtype) {
39  zxd.multiply_scalar(std::sqrt(2.));
40  ZXVert triangle = zxd.add_vertex(ZXType::Triangle, qtype);
41  ZXVert x = zxd.add_vertex(ZXType::XSpider, 0, qtype);
42  zxd.add_wire(triangle, x, ZXWireType::Basic, qtype, 1);
43  if (on_value) {
44  // Turn on the switch if the input is 1
45  ZXVert negate = zxd.add_vertex(ZXType::XSpider, 1, qtype);
46  zxd.add_wire(triangle, negate, ZXWireType::Basic, qtype, 0);
47  return {{negate, std::nullopt}, {x, std::nullopt}};
48  }
49  return {{triangle, 0}, {x, std::nullopt}};
50 }

52 // --o--s---o--
53 // | |
54 // n n
55 // |--G-o-|
56 // |
57 // s
58 // |-discard
59 // s, n are switches, G is the conditional zx diagram specified by 'left' and
60 // 'right'. s is on when the input is 0, n is on when the input is 1.
61 // return the input and the output of the conditional zx along with a vector of
62 // control spiders
63 std::pair<std::pair<ZXVertPort, ZXVertPort>, ZXVertPortVec> add_conditional_zx(
64  ZXDiagram& zxd, const ZXVert& left, const ZXVert& right,
65  const QuantumType& qtype) {
66  ZXVert in = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
67  ZXVert out = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
68  ZXVert discard_ctr = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
69  ZXVert discard = zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
70  auto switch_s0 = add_switch(zxd, false, qtype);
71  auto switch_n0 = add_switch(zxd, true, qtype);
72  auto switch_n1 = add_switch(zxd, true, qtype);
73  auto switch_s1 = add_switch(zxd, false, qtype);
74  zxd.add_wire(in, switch_s0.second.first, ZXWireType::Basic, qtype);
75  zxd.add_wire(out, switch_s0.second.first, ZXWireType::Basic, qtype);
76  zxd.add_wire(in, switch_n0.second.first, ZXWireType::Basic, qtype);
77  zxd.add_wire(left, switch_n0.second.first, ZXWireType::Basic, qtype);
78  zxd.add_wire(right, discard_ctr, ZXWireType::Basic, qtype);
79  zxd.add_wire(discard_ctr, switch_n1.second.first, ZXWireType::Basic, qtype);
80  zxd.add_wire(out, switch_n1.second.first, ZXWireType::Basic, qtype);
81  zxd.add_wire(discard_ctr, switch_s1.second.first, ZXWireType::Basic, qtype);
82  zxd.add_wire(discard, switch_s1.second.first, ZXWireType::Basic, qtype);
83  return {
84  {{in, std::nullopt}, {out, std::nullopt}},
85  {switch_s0.first, switch_s1.first, switch_n0.first, switch_n1.first}};
86 }

88 // n-bit AND spider to control the switches
89 // https://arxiv.org/abs/1910.06818
90 std::pair<ZXVertPortVec, ZXVertPort> add_n_bit_and(
91  ZXDiagram& zxd, unsigned n, const QuantumType& qtype) {
92  TKET_ASSERT(n > 1);
93  ZXVert z_vert = zxd.add_vertex(ZXType::ZSpider, 0, qtype);
94  // Add Triangle -1
95  ZXVert z_pi_0 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
96  ZXVert tri_1 = zxd.add_vertex(ZXType::Triangle, qtype);
97  ZXVert z_pi_1 = zxd.add_vertex(ZXType::ZSpider, 1, qtype);
98  zxd.add_wire(z_vert, z_pi_0, ZXWireType::Basic, qtype);
99  zxd.add_wire(tri_1, z_pi_0, ZXWireType::Basic, qtype, 0);
100  zxd.add_wire(tri_1, z_pi_1, ZXWireType::Basic, qtype, 1);
101  ZXVertPortVec inputs;
102  for (unsigned i = 0; i < n; i++) {
103  ZXVert tri_0 = zxd.add_vertex(ZXType::Triangle, qtype);
104  zxd.add_wire(tri_0, z_vert, ZXWireType::Basic, qtype, 1);
105  inputs.push_back({tri_0, 0});
106  }
107  return {inputs, {z_pi_1, std::nullopt}};
108 }

110 // Add converted circ into zxd. Set add_boundary to true
111 // if boundary spiders are to be added to the boundary.
112 BoundaryVertMap circuit_to_zx_recursive(
113  const Circuit& circ, ZXDiagram& zxd, bool add_boundary) {
114  std::map<TypedVertPort, ZXVertPort> vert_lookup;
115  std::map<VertPort, ZXVertPort> boolean_outport_lookup;
116  BoundaryVertMap bmap;
117 
118  // Convert each vertex to ZXDiagram, raise error if not supported
119  BGL_FORALL_VERTICES(vert, circ.dag, DAG) {
120  // We currently throw an error if the vertex is either classical or flow
121  Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
122  if (is_flowop_type(op->get_type()) || is_classical_type(op->get_type())) {
123  throw Unsupported(
124  "Cannot convert OpType: " + op->get_name() + " to a ZX node. \n");
125  }
126  switch (op->get_type()) {
127  case OpType::Input: {
128  ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Quantum);
129  if (add_boundary) zxd.add_boundary(zx_vert);
130  vert_lookup.insert(
131  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
132  bmap.insert({zx_vert, vert});
133  break;
134  }
135  case OpType::Output: {
136  ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Quantum);
137  if (add_boundary) zxd.add_boundary(zx_vert);
138  vert_lookup.insert(
139  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
140  bmap.insert({zx_vert, vert});
141  break;
142  }
143  case OpType::ClInput: {
144  ZXVert zx_vert = zxd.add_vertex(ZXType::Input, QuantumType::Classical);
145  if (add_boundary) zxd.add_boundary(zx_vert);
146  vert_lookup.insert(
147  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
148  bmap.insert({zx_vert, vert});
149  break;
150  }
151  case OpType::ClOutput: {
152  ZXVert zx_vert = zxd.add_vertex(ZXType::Output, QuantumType::Classical);
153  if (add_boundary) zxd.add_boundary(zx_vert);
154  vert_lookup.insert(
155  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
156  bmap.insert({zx_vert, vert});
157  break;
158  }
159  // Spiderless ops are handled during vertex wiring
160  case OpType::Barrier:
161  case OpType::noop:
162  case OpType::SWAP:
163  case OpType::Phase: {
164  continue;
165  }
166  case OpType::H: {
167  ZXVert zx_vert = zxd.add_vertex(ZXType::Hbox, QuantumType::Quantum);
168  vert_lookup.insert(
169  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
170  vert_lookup.insert(
171  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
172  zxd.multiply_scalar(0.5);
173  break;
174  }
175  case OpType::Rz: {
176  ZXVert zx_vert = zxd.add_vertex(
177  ZXType::ZSpider, op->get_params()[0], QuantumType::Quantum);
178  vert_lookup.insert(
179  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
180  vert_lookup.insert(
181  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
182  break;
183  }
184  case OpType::Rx: {
185  ZXVert zx_vert = zxd.add_vertex(
186  ZXType::XSpider, op->get_params()[0], QuantumType::Quantum);
187  vert_lookup.insert(
188  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
189  vert_lookup.insert(
190  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
191  break;
192  }
193  case OpType::X: {
194  ZXVert zx_vert =
195  zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Quantum);
196  vert_lookup.insert(
197  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
198  vert_lookup.insert(
199  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
200  break;
201  }
202  case OpType::Z: {
203  ZXVert zx_vert =
204  zxd.add_vertex(ZXType::ZSpider, 1, QuantumType::Quantum);
205  vert_lookup.insert(
206  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
207  vert_lookup.insert(
208  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
209  break;
210  }
211  case OpType::CX: {
212  ZXVert zx_x_vert =
213  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
214  ZXVert zx_z_vert =
215  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
216  zxd.add_wire(zx_x_vert, zx_z_vert);
217  vert_lookup.insert(
218  {{{vert, 0}, ZXPortType::In}, {zx_z_vert, std::nullopt}});
219  vert_lookup.insert(
220  {{{vert, 0}, ZXPortType::Out}, {zx_z_vert, std::nullopt}});
221  vert_lookup.insert(
222  {{{vert, 1}, ZXPortType::In}, {zx_x_vert, std::nullopt}});
223  vert_lookup.insert(
224  {{{vert, 1}, ZXPortType::Out}, {zx_x_vert, std::nullopt}});
225  zxd.multiply_scalar(2);
226  break;
227  }
228  case OpType::CZ: {
229  ZXVert zx_za_vert =
230  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
231  ZXVert zx_zb_vert =
232  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Quantum);
233  zxd.add_wire(zx_za_vert, zx_zb_vert, ZXWireType::H);
234  vert_lookup.insert(
235  {{{vert, 0}, ZXPortType::In}, {zx_za_vert, std::nullopt}});
236  vert_lookup.insert(
237  {{{vert, 0}, ZXPortType::Out}, {zx_za_vert, std::nullopt}});
238  vert_lookup.insert(
239  {{{vert, 1}, ZXPortType::In}, {zx_zb_vert, std::nullopt}});
240  vert_lookup.insert(
241  {{{vert, 1}, ZXPortType::Out}, {zx_zb_vert, std::nullopt}});
242  break;
243  }
244  case OpType::Measure: {
245  // Add a decoherence node
246  ZXVert zx_measure_vert =
247  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
248  // Add a delete operator
249  ZXVert zx_delete_vert =
250  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
251  vert_lookup.insert(
252  {{{vert, 0}, ZXPortType::In}, {zx_measure_vert, std::nullopt}});
253  vert_lookup.insert(
254  {{{vert, 0}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
255  vert_lookup.insert(
256  {{{vert, 1}, ZXPortType::In}, {zx_delete_vert, std::nullopt}});
257  vert_lookup.insert(
258  {{{vert, 1}, ZXPortType::Out}, {zx_measure_vert, std::nullopt}});
259  break;
260  }
261  case OpType::Reset: {
262  // Discard
263  ZXVert zx_discard_vert =
264  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
265  // Add a node to prepare |0>
266  ZXVert zx_reset_vert =
267  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
268  zxd.multiply_scalar(0.5);
269  vert_lookup.insert(
270  {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
271  vert_lookup.insert(
272  {{{vert, 0}, ZXPortType::Out}, {zx_reset_vert, std::nullopt}});
273  break;
274  }
275  case OpType::Collapse: {
276  ZXVert zx_vert =
277  zxd.add_vertex(ZXType::ZSpider, QuantumType::Classical);
278  vert_lookup.insert(
279  {{{vert, 0}, ZXPortType::In}, {zx_vert, std::nullopt}});
280  vert_lookup.insert(
281  {{{vert, 0}, ZXPortType::Out}, {zx_vert, std::nullopt}});
282  break;
283  }
284  case OpType::Create: {
285  ZXVert zx_init_vert =
286  zxd.add_vertex(ZXType::XSpider, 0, QuantumType::Quantum);
287  zxd.multiply_scalar(0.5);
288  vert_lookup.insert(
289  {{{vert, 0}, ZXPortType::Out}, {zx_init_vert, std::nullopt}});
290  break;
291  }
292  case OpType::Discard: {
293  ZXVert zx_discard_vert =
294  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
295  vert_lookup.insert(
296  {{{vert, 0}, ZXPortType::In}, {zx_discard_vert, std::nullopt}});
297  break;
298  }
299  case OpType::Conditional: {
300  // Stores the condition for each port index
301  std::vector<bool> port_conditions;
302 
303  op_signature_t parent_sig = op->get_signature();
304 
305  // Find the nested quantum op and its overall conditions
306  // Assume bool ports are always the first few ports
307  while (op->get_type() == OpType::Conditional) {
308  const Conditional& cond = static_cast<const Conditional&>(*op);
309  for (unsigned i = 0; i < cond.get_width(); i++) {
310  bool set = cond.get_value() & (1 << (cond.get_width() - i - 1));
311  port_conditions.push_back(set);
312  }
313  op = cond.get_op();
314  }
315  if (is_barrier_type(op->get_type())) {
316  throw Unsupported(
317  "Cannot convert conditional barrier operations to a ZX node. \n");
318  }
319 
320  unsigned port_conditions_size =
321  static_cast<unsigned>(port_conditions.size());
322  // Convert the underlying op to zx.
323  // If the op is a quantum gate, construct a 1 gate circuit
324  // If the op is a box, obtain its circuit decomposition
325  op_signature_t inner_sig = op->get_signature();
326  Circuit replacement;
327  if (is_box_type(op->get_type())) {
328  const Box& b = static_cast<const Box&>(*op);
329  replacement = *b.to_circuit();
330  } else {
331  unit_vector_t args;
332  unsigned q_index = 0;
333  unsigned c_index = 0;
334  for (const EdgeType& etype : inner_sig) {
335  if (etype == EdgeType::Quantum) {
336  args.push_back(Qubit(q_index++));
337  } else if (etype == EdgeType::Classical) {
338  args.push_back(Bit(c_index++));
339  }
340  TKET_ASSERT(etype != EdgeType::Boolean);
341  }
342  replacement = Circuit(q_index, c_index);
343  replacement.add_op(op, args);
344  }
345  BoundaryVertMap box_bm =
346  circuit_to_zx_recursive(replacement, zxd, false);
347 
348  // The Z spider controlling all switches
349  ZXVert master_switch =
350  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
351 
352  // For each qubit/bit path in the inner op, convert it into a
353  // conditional path.
354  unsigned q_index = 0;
355  unsigned c_index = 0;
356  for (unsigned i = 0; i < inner_sig.size(); i++) {
357  QuantumType qtype;
358  UnitID reg;
359  if (inner_sig[i] == EdgeType::Quantum) {
360  qtype = QuantumType::Quantum;
361  reg = Qubit(q_index++);
362  } else {
363  qtype = QuantumType::Classical;
364  reg = Bit(c_index++);
365  }
366  std::pair<ZXVertPort, ZXVertPort> boundary;
367  ZXVertPortVec controls;
368  Vertex inp = replacement.get_in(reg);
369  ZXVert zx_inp = box_bm.right.at(inp);
370  Vertex outp = replacement.get_out(reg);
371  ZXVert zx_outp = box_bm.right.at(outp);
372  // Convert the path between zx_inp and zx_outp into a conditional
373  // path
374  std::tie(boundary, controls) =
375  add_conditional_zx(zxd, zx_inp, zx_outp, qtype);
376  // Connect each switch to the master switch
377  for (const ZXVertPort& ctr : controls) {
378  zxd.add_wire(
379  ctr.first, master_switch, ZXWireType::Basic,
380  zxd.get_qtype(ctr.first).value(), ctr.second);
381  }
382  // Update lookup
383  TKET_ASSERT(parent_sig[i + port_conditions_size] == inner_sig[i]);
384  // Since we assume that the conditions always use the first few
385  // ports, the gate path should use the i + port_conditions.size()
386  // port.
387  vert_lookup.insert(
388  {{{vert, i + port_conditions_size}, ZXPortType::In},
389  boundary.first});
390  vert_lookup.insert(
391  {{{vert, i + port_conditions_size}, ZXPortType::Out},
392  boundary.second});
393  }
394  // Use either a Z spider or a AND operator to connect the master
395  // switch and the boolean inputs
396  ZXVertPortVec and_inputs;
397  if (port_conditions_size == 1) {
398  and_inputs.push_back({master_switch, std::nullopt});
399  } else {
400  ZXVertPort and_output;
401  std::tie(and_inputs, and_output) =
402  add_n_bit_and(zxd, port_conditions_size, QuantumType::Classical);
403  zxd.add_wire(
404  and_output.first, master_switch, ZXWireType::Basic,
405  QuantumType::Classical, and_output.second);
406  }
407  // Connect inputs to the nodes obtained from above
408  for (unsigned i = 0; i < port_conditions_size; i++) {
409  // Each boolean edge shares a source port with other
410  // classical/boolean edges. Use the classical Z spider to explicitly
411  // implement this copy operation
412  Edge in_edge = circ.get_nth_in_edge(vert, i);
413  port_t source_port = circ.get_source_port(in_edge);
414  Vertex vert_s = circ.source(in_edge);
415  // During the wiring stage, each edge originated from {vert_s,
416  // p_s} needs go through copy_vert
417  if (boolean_outport_lookup.find({vert_s, source_port}) ==
418  boolean_outport_lookup.end()) {
419  ZXVert copy_vert =
420  zxd.add_vertex(ZXType::ZSpider, 0, QuantumType::Classical);
421  boolean_outport_lookup.insert(
422  {{vert_s, source_port}, {copy_vert, std::nullopt}});
423  }
424  if (port_conditions[i]) {
425  vert_lookup.insert({{{vert, i}, ZXPortType::In}, and_inputs[i]});
426  } else {
427  ZXVert x_vert =
428  zxd.add_vertex(ZXType::XSpider, 1, QuantumType::Classical);
429  zxd.add_wire(
430  and_inputs[i].first, x_vert, ZXWireType::Basic,
431  QuantumType::Classical, and_inputs[i].second);
432  vert_lookup.insert(
433  {{{vert, i}, ZXPortType::In}, {x_vert, std::nullopt}});
434  }
435  }
436  break;
437  }
438  default:
439  if (is_box_type(op->get_type())) {
440  EdgeVec b_in_holes =
441  circ.get_in_edges_of_type(vert, EdgeType::Boolean);
442  if (b_in_holes.size() > 0) {
443  throw Unsupported("Cannot convert box type: " + op->get_name());
444  }
445  const Box& b = static_cast<const Box&>(*op);
446  Circuit replacement = *b.to_circuit();
447  // First, add the converted box to diagram.
448  BoundaryVertMap box_bm =
449  circuit_to_zx_recursive(replacement, zxd, false);
450  // Then, map the vertports in the box boundary to zx nodes.
451  // Assume that a box can't have Boolean input edges, and all Boolean
452  // output edges share ports with Classical edges. Therefore we don't
453  // have to map Boolean vertports.
454  EdgeVec q_in_holes =
455  circ.get_in_edges_of_type(vert, EdgeType::Quantum);
456  EdgeVec q_out_holes =
457  circ.get_out_edges_of_type(vert, EdgeType::Quantum);
458  EdgeVec c_in_holes =
459  circ.get_in_edges_of_type(vert, EdgeType::Classical);
460  EdgeVec c_out_holes =
461  circ.get_out_edges_of_type(vert, EdgeType::Classical);
462 
463  for (unsigned i = 0; i < q_in_holes.size(); i++) {
464  port_t port = circ.get_target_port(q_in_holes[i]);
465  Vertex inp = replacement.get_in(Qubit(i));
466  vert_lookup.insert(
467  {{{vert, port}, ZXPortType::In},
468  {box_bm.right.at(inp), std::nullopt}});
469  }
470  for (unsigned i = 0; i < q_out_holes.size(); i++) {
471  port_t port = circ.get_source_port(q_out_holes[i]);
472  Vertex outp = replacement.get_out(Qubit(i));
473  vert_lookup.insert(
474  {{{vert, port}, ZXPortType::Out},
475  {box_bm.right.at(outp), std::nullopt}});
476  }
477  for (unsigned i = 0; i < c_in_holes.size(); i++) {
478  port_t port = circ.get_target_port(c_in_holes[i]);
479  Vertex inp = replacement.get_in(Bit(i));
480  vert_lookup.insert(
481  {{{vert, port}, ZXPortType::In},
482  {box_bm.right.at(inp), std::nullopt}});
483  }
484  for (unsigned i = 0; i < c_out_holes.size(); i++) {
485  port_t port = circ.get_source_port(c_out_holes[i]);
486  Vertex outp = replacement.get_out(Bit(i));
487  vert_lookup.insert(
488  {{{vert, port}, ZXPortType::Out},
489  {box_bm.right.at(outp), std::nullopt}});
490  }
491  } else {
492  throw Unsupported(
493  "Cannot convert gate type: " + op->get_name() +
494  " to a ZX node, try rebase the gates to use Rx, Rz, "
495  "X, Z, H, CZ "
496  "or CX. \n");
497  }
498  }
499  }
500 
501  // Connect the ZX nodes
502  BGL_FORALL_EDGES(edge, circ.dag, DAG) {
503  Vertex v_s = circ.source(edge);
504  Vertex v_t = circ.target(edge);
505  port_t p_s = circ.get_source_port(edge);
506  port_t p_t = circ.get_target_port(edge);
507  // Handle Spiderless ops
508  if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_s))) {
509  // We only handle in-edges
510  continue;
511  }
512  if (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t))) {
513  // Traverse the path to find the next non-spiderless op
514  Edge next_e = edge;
515  do {
516  if (circ.get_OpType_from_Vertex(v_t) == OpType::SWAP) {
517  next_e = circ.get_nth_out_edge(
518  v_t, (circ.get_target_port(next_e) + 1) % 2);
519  v_t = circ.target(next_e);
520  } else {
521  std::tie(v_t, next_e) = circ.get_next_pair(v_t, next_e);
522  }
523  } while (is_spiderless_optype(circ.get_OpType_from_Vertex(v_t)));
524  p_t = circ.get_target_port(next_e);
525  }
526 
527  ZXVertPort zx_vp_s =
528  vert_lookup.at(TypedVertPort(VertPort(v_s, p_s), ZXPortType::Out));
529  ZXVertPort zx_vp_t =
530  vert_lookup.at(TypedVertPort(VertPort(v_t, p_t), ZXPortType::In));
531  if (circ.get_edgetype(edge) == EdgeType::Quantum) {
532  zxd.add_wire(
533  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic, QuantumType::Quantum,
534  zx_vp_s.second, zx_vp_t.second);
535  } else {
536  auto bool_it = boolean_outport_lookup.find(VertPort(v_s, p_s));
537  if (bool_it == boolean_outport_lookup.end()) {
538  zxd.add_wire(
539  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic,
540  QuantumType::Classical, zx_vp_s.second, zx_vp_t.second);
541  } else {
542  // If the source port is boolean, then connect the copy spider to the
543  // source vertex. All out-edges originated from the source port should
544  // be connected to the copy spider.
545  if (zxd.degree(bool_it->second.first) == 0) {
546  zxd.add_wire(
547  zx_vp_s.first, bool_it->second.first, ZXWireType::Basic,
548  QuantumType::Classical, zx_vp_s.second, bool_it->second.second);
549  }
550  zxd.add_wire(
551  bool_it->second.first, zx_vp_t.first, ZXWireType::Basic,
552  QuantumType::Classical, bool_it->second.second, zx_vp_t.second);
553  }
554  }
555  }
556  return bmap;
557 }

559 std::pair<ZXDiagram, BoundaryVertMap> circuit_to_zx(const Circuit& circ) {
560  ZXDiagram zxd;
561  BoundaryVertMap bmap = circuit_to_zx_recursive(circ, zxd, true);
562  // Remove internal boundary vertices produced by the recursion
563  ZXVertVec true_boundary = zxd.get_boundary();
564  ZXVertIterator vi, vi_end, next;
565  tie(vi, vi_end) = boost::vertices(*zxd.get_graph());
566  for (next = vi; vi != vi_end; vi = next) {
567  ++next;
568  if ((zxd.get_zxtype(*vi) == ZXType::Input ||
569  zxd.get_zxtype(*vi) == ZXType::Output) &&
570  std::find(true_boundary.begin(), true_boundary.end(), *vi) ==
571  true_boundary.end()) {
572  WireVec adj_wires = zxd.adj_wires(*vi);
573  TKET_ASSERT(adj_wires.size() == 2);
574  TKET_ASSERT(zxd.get_qtype(adj_wires[0]) == zxd.get_qtype(adj_wires[1]));
575  TKET_ASSERT(
576  zxd.get_wire_type(adj_wires[0]) == zxd.get_wire_type(adj_wires[1]));
577  ZXVertVec neighbours = zxd.neighbours(*vi);
578  zxd.add_wire(
579  neighbours[0], neighbours[1], zxd.get_wire_type(adj_wires[0]),
580  zxd.get_qtype(adj_wires[0]));
581  zxd.remove_vertex(*vi);
582  }
583  }
584  return {std::move(zxd), std::move(bmap)};
585 }

587 void clean_frontier(
588  ZXDiagram& diag, ZXVertVec& frontier, Circuit& circ,
589  std::map<ZXVert, unsigned>& qubit_map) {
590  std::set<ZXVert> frontier_lookup;
591  std::list<std::pair<ZXVert, ZXVert>> czs;
592  for (const ZXVert& f : frontier) {
593  frontier_lookup.insert(f);
594  for (const Wire& w : diag.adj_wires(f)) {
595  ZXVert n = diag.other_end(w, f);
596  if (diag.get_zxtype(n) == ZXType::Output) {
597  unsigned q = qubit_map.at(n);
598  if (diag.get_wire_type(w) == ZXWireType::H) {
599  diag.set_wire_type(w, ZXWireType::Basic);
600  circ.add_op<unsigned>(OpType::H, {q});
601  }
602  switch (diag.get_zxtype(f)) {
603  case ZXType::Input: {
604  break;
605  }
606  case ZXType::XY: {
607  const PhasedGen& f_gen = diag.get_vertex_ZXGen<PhasedGen>(f);
608  Expr ph = f_gen.get_param();
609  if (!equiv_0(ph)) {
610  circ.add_op<unsigned>(OpType::U1, -ph, {q});
611  }
612  diag.set_vertex_ZXGen_ptr(
613  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
614  break;
615  }
616  case ZXType::PX: {
617  const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
618  if (f_gen.get_param()) {
619  circ.add_op<unsigned>(OpType::Z, {q});
620  diag.set_vertex_ZXGen_ptr(
621  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
622  }
623  break;
624  }
625  case ZXType::PY: {
626  const CliffordGen& f_gen = diag.get_vertex_ZXGen<CliffordGen>(f);
627  circ.add_op<unsigned>(
628  f_gen.get_param() ? OpType::S : OpType::Sdg, {q});
629  diag.set_vertex_ZXGen_ptr(
630  f, ZXGen::create_gen(ZXType::PX, QuantumType::Quantum));
631  break;
632  }
633  default:
634  throw ZXError(
635  "Error during extraction from ZX diagram: unexpected ZXType in "
636  "frontier");
637  }
638  } else if (frontier_lookup.find(n) != frontier_lookup.end()) {
639  czs.push_back({f, n});
640  diag.remove_wire(w);
641  }
642  }
643  }
644  for (const std::pair<ZXVert, ZXVert>& pair : czs) {
645  circ.add_op<unsigned>(
646  OpType::CZ, {qubit_map.at(pair.first), qubit_map.at(pair.second)});
647  }
648  ZXVertVec new_frontier;
649  for (const ZXVert& f : frontier) {
650  bool removed = false;
651  ZXVertVec ns = diag.neighbours(f);
652  if (ns.size() == 2) {
653  ZXType nt0 = diag.get_zxtype(ns.at(0));
654  ZXType nt1 = diag.get_zxtype(ns.at(1));
655  if ((nt0 == ZXType::Input && nt1 == ZXType::Output) ||
656  (nt0 == ZXType::Output && nt1 == ZXType::Input)) {
657  diag.add_wire(ns.at(0), ns.at(1));
658  diag.remove_vertex(f);
659  removed = true;
660  }
661  }
662  if (!removed && diag.get_zxtype(f) != ZXType::Input)
663  new_frontier.push_back(f);
664  }
665  frontier = new_frontier;
666 }

682 static void bipartite_complementation(
683  ZXDiagram& diag, const ZXVertSeqSet& sa, const ZXVertSeqSet& sb) {
684  for (const ZXVert& a : sa.get<TagSeq>()) {
685  for (const ZXVert& b : sb.get<TagSeq>()) {

695 void extend_if_input(
696  ZXDiagram& diag, const ZXVert& v, std::map<ZXVert, ZXVert>& input_qubits) {
697  std::map<ZXVert, ZXVert>::iterator found = input_qubits.find(v);
698  if (found != input_qubits.end()) {
699  ZXVert in = found->second;
700  ZXVert ext0 = diag.add_vertex(ZXType::XY);
701  ZXVert ext1 = diag.add_vertex(ZXType::XY);
702  diag.remove_wire(diag.adj_wires(in).at(0));
703  diag.add_wire(in, ext0);
704  diag.add_wire(ext0, ext1, ZXWireType::H);
705  diag.add_wire(ext1, v, ZXWireType::H);
706  input_qubits.erase(found);
707  input_qubits.insert({ext0, in});
708  }

711 bool remove_all_gadgets(
712  ZXDiagram& diag, const ZXVertVec& frontier,
713  std::map<ZXVert, ZXVert>& input_qubits) {
714  bool removed_gadget = false;
715  for (const ZXVert& f : frontier) {
716  ZXVertVec f_ns = diag.neighbours(f);
717  std::optional<ZXVert> found_output;
718  for (const ZXVert& n : f_ns) {
719  // Each frontier vertex is connected to a unique output, find it
720  if (diag.get_zxtype(n) == ZXType::Output) {
721  found_output = n;
722  break;
723  }
724  }
725  if (!found_output)
726  throw ZXError(
727  "Error during extraction from ZXDiagram: frontier vertex not "
728  "adjacent to an output");
729  ZXVert o = *found_output;
730  for (const ZXVert& n : f_ns) {
731  if (diag.get_zxtype(n) == ZXType::YZ) {
732  // Pivot
733  // Identify three subsets of neighbours
734  ZXVertSeqSet excl_f;
735  // Need to recalculate neighbours rather than use f_ns as we might have
736  // extended
737  extend_if_input(diag, f, input_qubits);
738  for (const ZXVert& n : diag.neighbours(f)) {
739  extend_if_input(diag, n, input_qubits);
740  excl_f.insert(n);
741  }
742  auto& lookup_f = excl_f.get<TagKey>();
743  lookup_f.erase(lookup_f.find(n));
744  lookup_f.erase(lookup_f.find(o));
745  ZXVertSeqSet excl_n, joint;
746  for (const ZXVert& nn : diag.neighbours(n)) {
747  extend_if_input(diag, nn, input_qubits);
748  if (lookup_f.find(nn) != lookup_f.end())
749  joint.insert(nn);
750  else
751  excl_n.insert(nn);
752  }
753  auto& lookup_n = excl_n.get<TagKey>();
754  lookup_n.erase(lookup_n.find(f));
755  for (const ZXVert& nn : joint.get<TagSeq>()) {
756  auto found = lookup_f.find(nn);
757  if (found != lookup_f.end()) lookup_f.erase(found);
758  }
759  // The is_MBQC check in zx_to_circuit guarantees QuantumType::Quantum
760  bipartite_complementation(diag, joint, excl_n);
761  bipartite_complementation(diag, joint, excl_f);
762  bipartite_complementation(diag, excl_n, excl_f);
763  // In place of switching vertices f and n, we invert their
764  // connectivities
765  for (const ZXVert& v : excl_f.get<TagSeq>()) excl_n.insert(v);
766  bipartite_complementation(diag, {f}, excl_n);
767  bipartite_complementation(diag, {n}, excl_n);
768  Wire ow = *diag.wire_between(f, o);
769  diag.set_wire_type(
770  ow, (diag.get_wire_type(ow) == ZXWireType::Basic)
771  ? ZXWireType::H
772  : ZXWireType::Basic);
773  const PhasedGen& yz_gen = diag.get_vertex_ZXGen<PhasedGen>(n);
774  diag.set_vertex_ZXGen_ptr(
775  n, ZXGen::create_gen(
776  ZXType::XY, -yz_gen.get_param(), QuantumType::Quantum));
777  for (const ZXVert& nn : joint.get<TagSeq>()) {
778  ZXGen_ptr new_gen;
779  switch (diag.get_zxtype(nn)) {
780  case ZXType::XY: {
781  const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
782  new_gen = ZXGen::create_gen(
783  ZXType::XY, ph_gen.get_param() + 1., QuantumType::Quantum);
784  break;
785  }
786  case ZXType::PX:
787  case ZXType::PY: {
788  const CliffordGen& cl_gen =
789  diag.get_vertex_ZXGen<CliffordGen>(nn);
790  new_gen = ZXGen::create_gen(
791  cl_gen.get_type(), !cl_gen.get_param(), QuantumType::Quantum);
792  break;
793  }
794  case ZXType::XZ:
795  case ZXType::YZ: {
796  const PhasedGen& ph_gen = diag.get_vertex_ZXGen<PhasedGen>(nn);
797  new_gen = ZXGen::create_gen(
798  ph_gen.get_type(), -ph_gen.get_param(), QuantumType::Quantum);
799  break;
800  }
801  case ZXType::PZ: {
802  new_gen = diag.get_vertex_ZXGen_ptr(nn);
803  break;
804  }
805  default:
806  throw ZXError(
807  "Error during extraction from ZX diagram: unexpected ZXType "
808  "during local complementation");
809  }
810  diag.set_vertex_ZXGen_ptr(nn, new_gen);
811  }
812  removed_gadget = true;
813  break;
814  }
815  }
816  }
817  return removed_gadget;
818 }

820 Circuit zx_to_circuit(const ZXDiagram& d) {
821  ZXDiagram diag = d;
822  if (!diag.is_MBQC())
823  throw ZXError("Can only extract a circuit from a ZX diagram in MBQC form");
824  ZXVertVec ins = diag.get_boundary(ZXType::Input);
825  ZXVertVec outs = diag.get_boundary(ZXType::Output);
826  if (ins.size() != outs.size())
827  throw ZXError("Can only extract a circuit from a unitary ZX diagram");
828 
829  BGL_FORALL_VERTICES(v, *diag.get_graph(), ZXGraph) {
830  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
831  if (vgen->get_type() == ZXType::PY) {
832  const CliffordGen& cgen = dynamic_cast<const CliffordGen&>(*vgen);
833  diag.set_vertex_ZXGen_ptr(
834  v, ZXGen::create_gen(
835  ZXType::XY, cgen.get_param() ? Expr(-0.5) : Expr(0.5),
836  *cgen.get_qtype()));
837  }
838  }
839 
840  Circuit circ(ins.size());
841 
842  ZXVertVec frontier;
843  std::map<ZXVert, unsigned> qubit_map;
844  unsigned q = 0;
845  for (const ZXVert& o : outs) {
846  ZXVert f_i = diag.neighbours(o).at(0);
847  frontier.push_back(f_i);
848  qubit_map.insert({o, q});
849  qubit_map.insert({f_i, q});
850  ++q;
851  }
852  std::map<ZXVert, ZXVert> input_qubits;
853  for (const ZXVert& i : ins)
854  input_qubits.insert({diag.neighbours(i).at(0), i});
855 
856  clean_frontier(diag, frontier, circ, qubit_map);
857  while (!frontier.empty()) {
858  if (remove_all_gadgets(diag, frontier, input_qubits)) {
859  clean_frontier(diag, frontier, circ, qubit_map);
860  }
861  ZXVertSeqSet neighbours = neighbours_of_frontier(diag, frontier);
862  boost::bimap<ZXVert, unsigned> correctors, preserve, ys;
863  ZXVertVec to_solve;
864  for (const ZXVert& f : frontier) {
865  if (input_qubits.find(f) == input_qubits.end())
866  correctors.insert({f, (unsigned)correctors.size()});
867  }
868  for (const ZXVert& n : neighbours.get<TagSeq>()) {
869  ZXType n_type = diag.get_zxtype(n);
870  if (n_type == ZXType::XY || n_type == ZXType::PX) {
871  preserve.insert({n, (unsigned)preserve.size()});
872  to_solve.push_back(n);
873  }
874  }
875  std::map<ZXVert, ZXVertSeqSet> candidates =
876  Flow::gauss_solve_correctors(diag, correctors, preserve, to_solve, ys);
877 
878  if (candidates.empty())
879  throw ZXError(
880  "Error during extraction from ZX diagram: diagram does not have "
881  "gflow");
882 
883  unsigned min = UINT_MAX;
884  ZXVert best;
885  for (const ZXVert& v : to_solve) {
886  auto found = candidates.find(v);
887  if (found != candidates.end() && found->second.size() < min) {
888  min = found->second.size();
889  best = v;
890  }
891  }
892  ZXVertSeqSet g_best = candidates.at(best);
893 
894  ZXVert f_to_isolate = g_best.get<TagSeq>().front();
895  unsigned f_q = qubit_map.at(f_to_isolate);
896  for (const ZXVert& f : g_best.get<TagSeq>()) {
897  if (f != f_to_isolate) {
898  circ.add_op<unsigned>(OpType::CX, {f_q, qubit_map.at(f)});
899  }
900  }
901  ZXVert out;
902  Wire w_out;
903  for (const Wire& w : diag.adj_wires(f_to_isolate)) {
904  ZXVert n = diag.other_end(w, f_to_isolate);
905  if (diag.get_zxtype(n) == ZXType::Output) {
906  out = n;
907  w_out = w;
908  break;
909  }
910  }
911  diag.add_wire(
912  best, out,
913  (diag.get_wire_type(w_out) == ZXWireType::Basic) ? ZXWireType::H
914  : ZXWireType::Basic);
915  diag.remove_vertex(f_to_isolate);
916  qubit_map.erase(qubit_map.find(f_to_isolate));
917  qubit_map.insert({best, f_q});
918  for (ZXVertVec::iterator it = frontier.begin(); it != frontier.end();
919  ++it) {
920  if (*it == f_to_isolate) {
921  *it = best;
922  break;
923  }
924  }
925 
926  clean_frontier(diag, frontier, circ, qubit_map);
927  }
928 
929  qubit_map_t qm;
930  for (unsigned i = 0; i < ins.size(); ++i) {
931  ZXVert in = ins.at(i);
932  ZXVert out = diag.neighbours(in).at(0);
933  if (diag.get_zxtype(out) != ZXType::Output)
934  throw ZXError(
935  "Error during extraction from ZX diagram: input not adjacent to "
936  "output after extracting");
937  qm.insert({Qubit(qubit_map.at(out)), Qubit(i)});
938  }
939  circ.permute_boundary_output(qm);
940 
941  // Reverse gates in circuit (all gates added are self-transpose)
942  return circ.transpose();
943 }
