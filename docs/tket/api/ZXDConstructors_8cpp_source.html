82 std::pair<std::map<ZXVert, ZXVert>, std::map<Wire, Wire>> ZXDiagram::copy_graph(
83  const ZXDiagram& other, bool merge_boundaries) {
84  // The isomorphism from vertices of `other` to vertices of `this`
85  std::map<ZXVert, ZXVert> iso;
86  std::map<Wire, Wire> wiso;
87  BGL_FORALL_VERTICES(v, *other.graph, ZXGraph) {
88  ZXGen_ptr gen = other.get_vertex_ZXGen_ptr(v);
89  ZXVert new_v = this->add_vertex(gen);
90  iso.emplace(v, new_v);
91  }
92 
93  BGL_FORALL_EDGES(w, *other.graph, ZXGraph) {
94  WireProperties wp = other.get_wire_info(w);
95  ZXVert source = other.source(w);
96  ZXVert target = other.target(w);
97  Wire new_w = this->add_wire(iso.at(source), iso.at(target), wp);
98  wiso.emplace(w, new_w);
99  }
100 
101  if (merge_boundaries) {
102  for (const ZXVert& b : other.boundary) {
103  this->boundary.push_back(iso.at(b));
104  }
105  }
106 
107  this->multiply_scalar(other.get_scalar());
108 
109  return {iso, wiso};
110 }
