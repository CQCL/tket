38 ZXDiagram ZXDiagram::to_doubled_diagram() const {
39  ZXDiagram doubled;
40  // map from vertex
41  std::map<CPMVert, ZXVert> iso;
42 
43  BGL_FORALL_VERTICES(v, *this->graph, ZXGraph) {
44  ZXGen_ptr op = this->get_vertex_ZXGen_ptr(v);
45  std::optional<QuantumType> qtype = op->get_qtype();
46  if (op->get_type() == ZXType::ZXBox) {
47  const ZXBox& box = static_cast<const ZXBox&>(*op);
48  ZXGen_ptr new_op = std::make_shared<const ZXBox>(
49  box.get_diagram()->to_doubled_diagram());
50  ZXVert added = doubled.add_vertex(new_op);
51  iso.insert({{v, CPMDouble::Original}, added});
52  } else {
53  TKET_ASSERT(qtype.has_value());
54  if (*qtype == QuantumType::Quantum) {
55  ZXGen_ptr orig_op, conj_op;
56  switch (op->get_type()) {
57  case ZXType::Input:
58  case ZXType::Output:
59  case ZXType::Open: {
60  orig_op = std::make_shared<const BoundaryGen>(
61  op->get_type(), QuantumType::Classical);
62  conj_op = orig_op;
63  break;
64  }
65  case ZXType::ZSpider:
66  case ZXType::XSpider:
67  case ZXType::XY:
68  case ZXType::YZ: {
69  const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
70  orig_op = std::make_shared<const PhasedGen>(
71  op->get_type(), bg.get_param(), QuantumType::Classical);
72  conj_op = std::make_shared<const PhasedGen>(
73  op->get_type(), -bg.get_param(), QuantumType::Classical);
74  break;
75  }
76  case ZXType::XZ: {
77  const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
78  orig_op = std::make_shared<const PhasedGen>(
79  op->get_type(), bg.get_param(), QuantumType::Classical);
80  conj_op = orig_op;
81  break;
82  }
83  case ZXType::PX:
84  case ZXType::PZ: {
85  const CliffordGen& bg = static_cast<const CliffordGen&>(*op);
86  orig_op = std::make_shared<const CliffordGen>(
87  op->get_type(), bg.get_param(), QuantumType::Classical);
88  conj_op = orig_op;
89  break;
90  }
91  case ZXType::PY: {
92  const CliffordGen& bg = static_cast<const CliffordGen&>(*op);
93  orig_op = std::make_shared<const CliffordGen>(
94  op->get_type(), bg.get_param(), QuantumType::Classical);
95  conj_op = std::make_shared<const CliffordGen>(
96  op->get_type(), !bg.get_param(), QuantumType::Classical);
97  break;
98  }
99  case ZXType::Hbox: {
100  const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
101  orig_op = std::make_shared<const PhasedGen>(
102  op->get_type(), bg.get_param(), QuantumType::Classical);
103  conj_op = std::make_shared<const PhasedGen>(
104  op->get_type(), SymEngine::conjugate(bg.get_param()),
105  QuantumType::Classical);
106  break;
107  }
108  case ZXType::Triangle: {
109  orig_op = std::make_shared<const DirectedGen>(
110  ZXType::Triangle, QuantumType::Classical);
111  conj_op = orig_op;
112  break;
113  }
114  default:
115  throw ZXError("Unrecognised ZXType in to_doubled_diagram()");
116  }
117  ZXVert orig = doubled.add_vertex(orig_op);
118  ZXVert conj = doubled.add_vertex(conj_op);
119  iso.insert({{v, CPMDouble::Original}, orig});
120  iso.insert({{v, CPMDouble::Conjugated}, conj});
121  } else {
122  ZXVert added = doubled.add_vertex(op);
123  iso.insert({{v, CPMDouble::Original}, added});
124  }
125  }
126  }
127 
128  BGL_FORALL_EDGES(w, *this->graph, ZXGraph) {
129  WireProperties wp = this->get_wire_info(w);
130  ZXVert s = source(w);
131  ZXVert t = target(w);
132  std::optional<unsigned> new_s_port, new_t_port;
133  ZXGen_ptr sgen = get_vertex_ZXGen_ptr(s);
134  ZXGen_ptr tgen = get_vertex_ZXGen_ptr(t);
135  // Quantum ports on a ZXBox get mapped to two Classical ports
136  if (sgen->get_type() == ZXType::ZXBox) {
137  const ZXBox& box = static_cast<const ZXBox&>(*sgen);
138  std::vector<QuantumType> sig = box.get_signature();
139  unsigned p = 0;
140  for (unsigned i = 0; i < *wp.source_port; ++i) {
141  if (sig.at(i) == QuantumType::Quantum)
142  p += 2;
143  else
144  p += 1;
145  }
146  new_s_port = p;
147  }
148  // Other generators just get duplicated
149  else {
150  new_s_port = wp.source_port;
151  }
152  // Do the same for the target
153  if (tgen->get_type() == ZXType::ZXBox) {
154  const ZXBox& box = static_cast<const ZXBox&>(*tgen);
155  std::vector<QuantumType> sig = box.get_signature();
156  unsigned p = 0;
157  for (unsigned i = 0; i < *wp.target_port; ++i) {
158  if (sig.at(i) == QuantumType::Quantum)
159  p += 2;
160  else
161  p += 1;
162  }
163  new_t_port = p;
164  } else {
165  new_t_port = wp.target_port;
166  }
167  WireProperties new_wp{
168  wp.type, QuantumType::Classical, new_s_port, new_t_port};
169  ZXVert new_s = iso.at({s, CPMDouble::Original});
170  ZXVert new_t = iso.at({t, CPMDouble::Original});
171  doubled.add_wire(new_s, new_t, new_wp);
172  // Handle the second wire for expanding Quantum wires
173  if (wp.qtype == QuantumType::Quantum) {
174  // For ZXBox, the conjugated port is the next one
175  if (sgen->get_type() == ZXType::ZXBox) *new_wp.source_port += 1;
176  // For other generators, the conjugate is a new vertex
177  else if (sgen->get_qtype() == QuantumType::Quantum)
178  new_s = iso.at({s, CPMDouble::Conjugated});
179  // Do the same for the target
180  if (tgen->get_type() == ZXType::ZXBox)
181  *new_wp.target_port += 1;
182  else if (tgen->get_qtype() == QuantumType::Quantum)
183  new_t = iso.at({t, CPMDouble::Conjugated});
184  doubled.add_wire(new_s, new_t, new_wp);
185  }
186  }
187 
188  for (const ZXVert& b : boundary) {
189  doubled.boundary.push_back(iso.at({b, CPMDouble::Original}));
190  if (get_qtype(b) == QuantumType::Quantum)
191  doubled.boundary.push_back(iso.at({b, CPMDouble::Conjugated}));
192  }
193 
194  return doubled;
195 }

197 ZXDiagram ZXDiagram::to_quantum_embedding() const {
198  ZXDiagram embedding(*this);
199  for (ZXVert& b : embedding.boundary) {
200  if (embedding.get_qtype(b) == QuantumType::Classical) {
201  ZXVert new_b =
202  embedding.add_vertex(embedding.get_zxtype(b), QuantumType::Quantum);
203  ZXGen_ptr id = std::make_shared<const PhasedGen>(
204  ZXType::ZSpider, 0., QuantumType::Classical);
205  embedding.set_vertex_ZXGen_ptr(b, id);
206  embedding.add_wire(new_b, b, ZXWireType::Basic, QuantumType::Quantum);
207  b = new_b;
208  }
209  }
210 
211  return embedding;
212 }
