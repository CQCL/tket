22 bool ZXDiagram::is_graphlike() const {
23  BGL_FORALL_EDGES(w, *graph, ZXGraph) {
24  if (is_boundary_type(get_zxtype(source(w))) ||
25  is_boundary_type(get_zxtype(target(w)))) {
26  if (get_wire_type(w) != ZXWireType::Basic) return false;
27  } else {
28  if (get_wire_type(w) != ZXWireType::H) return false;

33  if (type != ZXType::ZSpider && !is_boundary_type(type)) return false;

38 bool ZXDiagram::is_MBQC() const {
39  BGL_FORALL_EDGES(w, *graph, ZXGraph) {
40  if (get_qtype(w) != QuantumType::Quantum) return false;
41  if (is_boundary_type(get_zxtype(source(w))) ||
42  is_boundary_type(get_zxtype(target(w)))) {
43  if (get_wire_type(w) != ZXWireType::Basic) return false;
44  } else {
45  if (get_wire_type(w) != ZXWireType::H) return false;
46  }
47  }
48  BGL_FORALL_VERTICES(v, *graph, ZXGraph) {
49  ZXType type = get_zxtype(v);
50  if (!is_MBQC_type(type) && type != ZXType::Input && type != ZXType::Output)
51  return false;
52  if (get_qtype(v) != QuantumType::Quantum) return false;
53  }
54  return true;
55 }
