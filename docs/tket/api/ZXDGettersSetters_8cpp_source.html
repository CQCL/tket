22bool WireProperties::operator==(const WireProperties& other) const {
23 return (this->type == other.type) && (this->qtype == other.qtype) &&

28ZXVertVec ZXDiagram::get_boundary(
29 std::optional<ZXType> type, std::optional<QuantumType> qtype) const {
30 if (!type && !qtype) return boundary;
31
32 ZXVertVec sub_boundary;
33 for (const ZXVert& b : boundary) {
34 if ((!type || get_zxtype(b) == *type) &&
35 (!qtype || *get_qtype(b) == *qtype))
36 sub_boundary.push_back(b);
37 }
38 return sub_boundary;
39}

80ZXVertVec ZXDiagram::neighbours(const ZXVert& v) const {
81 ZXVertSeqSet neis;
82 BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) neis.insert(target(w));
83 BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) neis.insert(source(w));
84 auto& ordered = neis.get<TagSeq>();
85 ZXVertVec neighbours(ordered.begin(), ordered.end());
86 return neighbours;
87}

89WireVec ZXDiagram::adj_wires(const ZXVert& v) const {
90 WireVec adj;
91 BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) adj.push_back(w);
92 BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) {
93 // Only add self loops once (ignore the in_edge wires for loops)
94 if (source(w) != v) adj.push_back(w);
95 }
96 return adj;
97}

99WireVec ZXDiagram::wires_between(const ZXVert& u, const ZXVert& v) const {
100 WireVec wires;
101 for (const Wire& w : adj_wires(u)) {
102 ZXVert other = other_end(w, u);
103 if (other == v) wires.push_back(w);
104 }
105 return wires;
106}

108std::optional<Wire> ZXDiagram::wire_between(
109 const ZXVert& va, const ZXVert& vb,
110 ZXDiagram::WireSearchOption directed) const {
111 const auto& [wire, exists] = boost::edge(va, vb, *graph);
112 if (exists)
113 return wire;
114 else if (directed == WireSearchOption::UNDIRECTED)
115 return wire_between(vb, va, WireSearchOption::DIRECTED);
116 else
117 return std::nullopt;
118}

120Wire ZXDiagram::wire_at_port(
121 const ZXVert& v, std::optional<unsigned> port) const {
122 Wire w_found;
123 unsigned n_found = 0;
124 BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) {
125 if (source_port(w) == port) {
126 w_found = w;
127 ++n_found;
128 }
129 }
130 BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) {
131 if (target_port(w) == port) {
132 w_found = w;
133 ++n_found;
134 }
135 }
136 if (n_found != 1)
137 throw ZXError(
138 "Expected only one wire at port, found " + std::to_string(n_found));
139 return w_found;
140}

190ZXVert ZXDiagram::other_end(const Wire& w, const ZXVert& u) const {
191 ZXVert s = source(w);
192 ZXVert t = target(w);
193 if (s == u) {
194 return t;
195 } else if (t == u) {
196 return s;
197 } else {
198 throw ZXError("In other_end(w, u), u is not adjacent to w.");
199 }
200}

231bool ZXDiagram::is_pauli_spider(const ZXVert& v) const {
232 ZXGen_ptr op = get_vertex_ZXGen_ptr(v);
233 if (!is_spider_type(op->get_type())) return false;
234 const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
235 std::optional<unsigned> pi2_mult = equiv_Clifford(bg.get_param());
236 return (pi2_mult && ((*pi2_mult % 2) == 0));
237}
238
239bool ZXDiagram::is_proper_clifford_spider(const ZXVert& v) const {
240 ZXGen_ptr op = get_vertex_ZXGen_ptr(v);
241 if (!is_spider_type(op->get_type())) return false;
242 const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
243 std::optional<unsigned> pi2_mult = equiv_Clifford(bg.get_param());
244 return (pi2_mult && ((*pi2_mult % 2) == 1));
245}

247static std::string graphviz_vertex_props(ZXGen_ptr op) {
248 std::stringstream ss;
249
250 // Tooltips (rollover text) contains the get_name information
251 ss << "tooltip=\"" + op->get_name() + "\" ";
252
253 // Classical nodes are drawn thinner
254 if (op->get_qtype() == QuantumType::Classical) ss << "penwidth=1 ";
255
256 // Modify node drawing properties based on this type
257 ZXType type = op->get_type();
258
259 switch (type) {
260 case ZXType::Input:
261 case ZXType::Output:
262 case ZXType::Open: {
263 ss << "style=\"filled, dashed\" fillcolor=\"white\" shape=circle "
264 "label=\""
265 << op->get_name() << "\"";
266 break;
267 }
268 case ZXType::ZSpider:
269 case ZXType::XSpider: {
270 const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
271 Expr p = bg.get_param();
272 std::string colour = (type == ZXType::ZSpider) ? "green" : "red";
273 ss << "fillcolor=\"" << colour << "\" shape=circle label=\"";
274 if (!equiv_0(p)) ss << p;
275 ss << "\"";
276 break;
277 }
278 case ZXType::Hbox: {
279 const PhasedGen& bg = static_cast<const PhasedGen&>(*op);
280 Expr p = bg.get_param();
281 std::optional<Complex> ev = eval_expr_c(p);
282 ss << "fillcolor=\"gold\" shape=square label=\"";
283 if (!ev || (std::abs(*ev + 1.) >= EPS)) ss << p;
284 ss << "\"";
285 break;
286 }
287 case ZXType::XY:
288 case ZXType::XZ:
289 case ZXType::YZ:
290 case ZXType::PX:
291 case ZXType::PY:
292 case ZXType::PZ: {
293 ss << "shape=circle width=0.1 fixedsize=shape label=\"" << op->get_name()
294 << "\\n\\n\"";
295 break;
296 }
297 case ZXType::Triangle: {
298 ss << "fillcolor=\"gold\" shape=triangle";
299 break;
300 }
301 case ZXType::ZXBox: {
302 ss << "shape=box3d penwidth=2 label=\"Box\"";
303 break;
304 }
305 default: {
306 throw ZXError(
307 "Trying to render vertex in a ZXDiagram with unknown "
308 "ZXType");
309 }
310 }
311 return ss.str();
312}

314static std::string graphviz_wire_props(const WireProperties& wp) {
315 std::stringstream ss;
316
317 // (default assumption is that qtype==Quantum)
318 if (wp.qtype == QuantumType::Classical) ss << "penwidth=1 ";
319
320 if (wp.type == ZXWireType::H) ss << "style=dashed color=\"blue\" ";
321
322 // port information:
323 // in graphviz, the 'head' of an edge refers to the `Target` end
324 // and similarly, the 'tail' of an edge refers to the `Source` end
325 if (wp.source_port) ss << " taillabel=\"" << *wp.source_port << "\"";
326 if (wp.target_port) ss << " headlabel=\"" << *wp.target_port << "\"";
327
328 return ss.str();
329}

331std::string ZXDiagram::to_graphviz_str(
332 const std::set<ZXVert>& highlights) const {
333 std::stringstream out;
334
335 // Construct ZXVert index map (used as vertex IDs by graphviz)
336 std::map<ZXVert, unsigned> idm;
337 unsigned x = 0;
338 BGL_FORALL_VERTICES(v, *graph, ZXGraph) { idm.insert({v, x++}); }
339
340 out << "graph G {\n";
341
348 out << "node [penwidth=3 style=filled]\n";
349 BGL_FORALL_VERTICES(v, *graph, ZXGraph) {
350 // Specifying node ID we are defining properties on
351 out << idm.at(v);
352
353 // Defining the properties bracket
354 out << " [";
355
356 // Define vertex properties based on ZXGen
357 out << graphviz_vertex_props(get_vertex_ZXGen_ptr(v));
358
359 // Additional visual properties on node:
360 // exterior labels for the node ID information
361 out << " xlabel=<<font color=\"grey\">" << idm.at(v) << "</font>>";
362
363 // Highlight the vertices in `highlights` by red border
364 auto loc = highlights.find(v);
365 if (loc != highlights.end()) out << " color=\"red3\"";
366
367 // Close off the properties list for the vertices
368 out << "];\n";
369 }
370
377 out << "edge [penwidth=3]\n";
378 BGL_FORALL_EDGES(w, *graph, ZXGraph) {
379 // Draw an edge between the Wire's two ends
380 ZXVert s = source(w);
381 ZXVert t = target(w);
382 out << idm.at(s) << " -- " << idm.at(t);
383
384 // Defining the properties bracket
385 out << " [";
386 out << graphviz_wire_props(get_wire_info(w));
387 out << "]\n";
388 }
389
395 out << "rankdir = LR;\n"
396 "input_rank [style=invisible];\n"
397 "output_rank [style=invisible];\n"
398 "input_rank -- output_rank [style=invis];\n";
399
400 out << "{ rank = same\n"
401 "input_rank";
402 for (const ZXVert& v : get_boundary(ZXType::Input))
403 out << " -- " << idm.at(v);
404 out << " [style=invis]; }\n";
405
406 out << "{ rank = same\n"
407 "output_rank";
408 for (const ZXVert& v : get_boundary(ZXType::Output))
409 out << " -- " << idm.at(v);
410 out << " [style=invis]; }\n";
411
412 out << "}\n";
413 return out.str();
414}
