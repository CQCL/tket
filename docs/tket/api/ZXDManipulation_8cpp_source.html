74bool ZXDiagram::remove_wire(
75 const ZXVert& va, const ZXVert& vb, const WireProperties& prop,
76 ZXDiagram::WireSearchOption directed) {
77 BGL_FORALL_OUTEDGES(va, w, *graph, ZXGraph) {
78 if ((target(w) == vb) && (get_wire_info(w) == prop)) {
79 remove_wire(w);
80 return true;
81 }
82 }
83
84 // Recheck with reverse wire direction (including ports on the two ends)
85 if (directed == WireSearchOption::UNDIRECTED) {
86 WireProperties rev_props = prop;
87 rev_props.source_port = prop.target_port;
88 rev_props.target_port = prop.source_port;
89 return remove_wire(vb, va, rev_props, WireSearchOption::DIRECTED);
90 }
91 return false;
92}

134static void check_valid_wire(
135 const std::optional<unsigned>& port, QuantumType qtype,
136 const std::optional<unsigned>& n_ports, std::vector<bool>& ports_found,
137 ZXGen_ptr op) {
138 if (port) {
139 if (!n_ports)
140 throw ZXError("Wire at a named port of an undirected vertex");
141 else if (ports_found.at(*port))
142 throw ZXError("Multiple wires on the same port of a vertex");
143 else
144 ports_found.at(*port) = true;
145 } else if (n_ports)
146 throw ZXError("Wire at an unnamed port of a directed vertex");
147 if (!op->valid_edge(port, qtype))
148 throw ZXError("QuantumType of wire is incompatible with the given port");
149}

151void ZXDiagram::check_validity() const {
152 std::set<ZXVert> boundary_lookup;
153 for (const ZXVert& b : boundary) {
154 if (!is_boundary_type(get_zxtype(b)))
155 throw ZXError("Non-boundary vertex type in boundary");
156 if (!boundary_lookup.insert(b).second)
157 throw ZXError("Vertex appears in boundary multiple times");
158 }
159 BGL_FORALL_VERTICES(v, *graph, ZXGraph) {
160 ZXGen_ptr op = get_vertex_ZXGen_ptr(v);
161 ZXType type = op->get_type();
162 if (is_boundary_type(type)) {
163 if (degree(v) != 1)
164 throw ZXError("Boundary vertex does not have degree 1");
165 if (boundary_lookup.find(v) == boundary_lookup.end())
166 throw ZXError("Vertex of boundary type is not in the boundary");
167 }
168 std::optional<unsigned> n_ports = std::nullopt;
169 if (is_directed_type(type)) {
170 const ZXDirected& dir = static_cast<const ZXDirected&>(*op);
171 n_ports = dir.n_ports();
172 }
173 std::vector<bool> ports_found(n_ports ? *n_ports : 0, false);
174 BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) {
175 check_valid_wire(source_port(w), get_qtype(w), n_ports, ports_found, op);
176 }
177 BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) {
178 check_valid_wire(target_port(w), get_qtype(w), n_ports, ports_found, op);
179 }
180 if (n_ports && !std::all_of(
181 ports_found.begin(), ports_found.end(),
182 [](const bool& b) { return b; }))
183 throw ZXError("Not all ports of a directed vertex have wires connected");
184 }
185}
