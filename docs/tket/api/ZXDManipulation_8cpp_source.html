68 bool ZXDiagram::remove_wire(
69  const ZXVert& va, const ZXVert& vb, const WireProperties& prop,
70  ZXDiagram::WireSearchOption directed) {
71  BGL_FORALL_OUTEDGES(va, w, *graph, ZXGraph) {
72  if ((target(w) == vb) && (get_wire_info(w) == prop)) {
73  remove_wire(w);
74  return true;
75  }
76  }
77 
78  // Recheck with reverse wire direction (including ports on the two ends)
79  if (directed == WireSearchOption::UNDIRECTED) {
80  WireProperties rev_props = prop;
81  rev_props.source_port = prop.target_port;
82  rev_props.target_port = prop.source_port;
83  return remove_wire(vb, va, rev_props, WireSearchOption::DIRECTED);
84  }
85  return false;
86 }

128 static void check_valid_wire(
129  const std::optional<unsigned>& port, QuantumType qtype,
130  const std::optional<unsigned>& n_ports, std::vector<bool>& ports_found,
131  ZXGen_ptr op) {
132  if (port) {
133  if (!n_ports)
134  throw ZXError("Wire at a named port of an undirected vertex");
135  else if (ports_found.at(*port))
136  throw ZXError("Multiple wires on the same port of a vertex");
137  else
138  ports_found.at(*port) = true;
139  } else if (n_ports)
140  throw ZXError("Wire at an unnamed port of a directed vertex");
141  if (!op->valid_edge(port, qtype))
142  throw ZXError("QuantumType of wire is incompatible with the given port");
143 }

145 void ZXDiagram::check_validity() const {
146  std::set<ZXVert> boundary_lookup;
147  for (const ZXVert& b : boundary) {
148  if (!is_boundary_type(get_zxtype(b)))
149  throw ZXError("Non-boundary vertex type in boundary");
150  if (!boundary_lookup.insert(b).second)
151  throw ZXError("Vertex appears in boundary multiple times");
152  }
153  BGL_FORALL_VERTICES(v, *graph, ZXGraph) {
154  ZXGen_ptr op = get_vertex_ZXGen_ptr(v);
155  ZXType type = op->get_type();
156  if (is_boundary_type(type)) {
157  if (degree(v) != 1)
158  throw ZXError("Boundary vertex does not have degree 1");
159  if (boundary_lookup.find(v) == boundary_lookup.end())
160  throw ZXError("Vertex of boundary type is not in the boundary");
161  }
162  std::optional<unsigned> n_ports = std::nullopt;
163  if (is_directed_type(type)) {
164  const ZXDirected& dir = static_cast<const ZXDirected&>(*op);
165  n_ports = dir.n_ports();
166  }
167  std::vector<bool> ports_found(n_ports ? *n_ports : 0, false);
168  BGL_FORALL_OUTEDGES(v, w, *graph, ZXGraph) {
169  check_valid_wire(source_port(w), get_qtype(w), n_ports, ports_found, op);
170  }
171  BGL_FORALL_INEDGES(v, w, *graph, ZXGraph) {
172  check_valid_wire(target_port(w), get_qtype(w), n_ports, ports_found, op);
173  }
174  if (n_ports && !std::all_of(
175  ports_found.begin(), ports_found.end(),
176  [](const bool& b) { return b; }))
177  throw ZXError("Not all ports of a directed vertex have wires connected");
178  }
179 }
