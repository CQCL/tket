28void ZXDiagram::Subdiagram::check_validity(const ZXDiagram& diag) const {
29 std::set<std::pair<Wire, WireEnd>> boundary_lookup;
30 for (const std::pair<Wire, WireEnd>& w : boundary_) {
31 auto inserted = boundary_lookup.insert(w);
32 if (!inserted.second)
33 throw ZXError(
34 "Malformed ZX Subdiagram: Wire appears multiple times in boundary");
35 if (verts_.find(diag.vertex_at_end(w.first, w.second)) == verts_.end())
36 throw ZXError(
37 "Malformed ZX Subdiagram: Vertex adjacent to boundary is not in "
38 "vertex set");
39 }
40 for (const ZXVert& v : verts_) {
41 if (is_boundary_type(diag.get_zxtype(v)))
42 throw ZXError("Malformed ZX Subdiagram: Contains a boundary vertex");
43 for (const Wire& w : diag.adj_wires(v)) {
44 ZXVert n = diag.other_end(w, v);
45 if (verts_.find(n) == verts_.end()) {
46 if (boundary_lookup.find({w, diag.end_of(w, v)}) ==
47 boundary_lookup.end())
48 throw ZXError("Malformed ZX Subdiagram: subdiagram is not closed");
49 } else {
50 if ((boundary_lookup.find({w, WireEnd::Source}) ==
51 boundary_lookup.end()) ^
52 (boundary_lookup.find({w, WireEnd::Target}) ==
53 boundary_lookup.end()))
54 throw ZXError(
55 "Malformed ZX Subdiagram: wire between two interior vertices "
56 "contains one boundary");
57 }
58 }
59 }
60}

62ZXDiagram ZXDiagram::Subdiagram::to_diagram(const ZXDiagram& orig) const {
63 ZXDiagram diag;
64 std::map<ZXVert, ZXVert> vert_iso;
65 std::map<std::pair<Wire, WireEnd>, ZXVert> bound_iso;
66 for (const std::pair<Wire, WireEnd>& bw : boundary_) {
67 ZXVert bv = diag.add_vertex(ZXType::Open, orig.get_qtype(bw.first));
68 diag.boundary.push_back(bv);
69 bound_iso.insert({bw, bv});
70 }
71 for (const ZXVert& ov : verts_) {
72 ZXVert v = diag.add_vertex(orig.get_vertex_ZXGen_ptr(ov));
73 vert_iso.insert({ov, v});
74 for (const Wire& w : orig.adj_wires(ov)) {
75 bool internal = true;
76 if (orig.source(w) == ov) {
77 auto found_bound = bound_iso.find({w, WireEnd::Source});
78 if (found_bound != bound_iso.end()) {
79 internal = false;
80 WireProperties wp = orig.get_wire_info(w);
81 diag.add_wire(
82 v, found_bound->second, ZXWireType::Basic, wp.qtype,
83 wp.source_port, std::nullopt);
84 }
85 }
86 if (orig.target(w) == ov) {
87 auto found_bound = bound_iso.find({w, WireEnd::Target});
88 if (found_bound != bound_iso.end()) {
89 internal = false;
90 WireProperties wp = orig.get_wire_info(w);
91 diag.add_wire(
92 found_bound->second, v, ZXWireType::Basic, wp.qtype, std::nullopt,
93 wp.target_port);
94 }
95 }
96 if (internal) {
97 ZXVert other = orig.other_end(w, ov);
98 auto found_other = vert_iso.find(other);
99 if (found_other != vert_iso.end()) {
100 WireProperties wp = orig.get_wire_info(w);
101 if (orig.source(w) == ov)
102 diag.add_wire(v, found_other->second, wp);
103 else
104 diag.add_wire(found_other->second, v, wp);
105 }
106 }
107 }
108 }
109 return diag;
110}

112void ZXDiagram::substitute(
113 const ZXDiagram& to_insert, const Subdiagram& to_replace) {
114 unsigned n_bounds = to_insert.boundary.size();
115 if (n_bounds != to_replace.boundary_.size())
116 throw ZXError(
117 "ZXDiagram substitution error: boundary size of replacement does not "
118 "fit size of subdiagram");
119
120 std::pair<std::map<ZXVert, ZXVert>, std::map<Wire, Wire>> iso =
121 copy_graph(to_insert, false);
122
123 std::map<Wire, ZXVert> loops;
124 for (unsigned i = 0; i < n_bounds; ++i) {
125 std::pair<Wire, WireEnd> w = to_replace.boundary_.at(i);
126 WireProperties wp = get_wire_info(w.first);
127 ZXVert new_b = iso.first.at(to_insert.boundary.at(i));
128 if (wp.qtype != get_qtype(new_b))
129 throw ZXError(
130 "ZXDiagram substitution error: QuantumType mismatch at a boundary");
131 ZXVert to_connect =
132 (w.second == WireEnd::Source) ? target(w.first) : source(w.first);
133 if (to_replace.verts_.find(to_connect) != to_replace.verts_.end()) {
134 // Connect two boundaries together
135 auto inserted = loops.insert({w.first, new_b});
136 if (!inserted.second) {
137 // Already seen the other end, so actually connect the loop
138 ZXVert other_b = inserted.first->second;
139 Wire wb = adj_wires(new_b).at(0);
140 ZXVert adj = other_end(wb, new_b);
141 std::optional<unsigned> adj_port =
142 (source(wb) == adj) ? source_port(wb) : target_port(wb);
143 if (adj == other_b) {
144 // Creates a wire-loop, resolve to a scalar
145 bool hadamard =
146 (wp.type == ZXWireType::H) ^ (get_wire_type(wb) == ZXWireType::H);
147 if (hadamard)
148 multiply_scalar(0.);
149 else if (wp.qtype == QuantumType::Quantum)
150 multiply_scalar(4.);
151 else
152 multiply_scalar(2.);
153 } else {
154 // Connect the adjacent vertices
155 Wire wob = adj_wires(other_b).at(0);
156 ZXVert other_adj = other_end(wob, other_b);
157 std::optional<unsigned> other_adj_port =
158 (source(wob) == other_adj) ? source_port(wob) : target_port(wob);
159 bool hadamard = (wp.type == ZXWireType::H) ^
160 (get_wire_type(wb) == ZXWireType::H) ^
161 (get_wire_type(wob) == ZXWireType::H);
162 add_wire(
163 adj, other_adj, hadamard ? ZXWireType::H : ZXWireType::Basic,
164 wp.qtype, adj_port, other_adj_port);
165 }
166 remove_vertex(new_b);
167 remove_vertex(other_b);
168 }
169 } else {
170 Wire wb = adj_wires(new_b).at(0);
171 if (get_wire_type(wb) == ZXWireType::H) {
172 if (wp.type == ZXWireType::Basic)
173 wp.type = ZXWireType::H;
174 else
175 wp.type = ZXWireType::Basic;
176 }
177 ZXVert adj = other_end(wb, new_b);
178 std::optional<unsigned> adj_port =
179 (source(wb) == adj) ? source_port(wb) : target_port(wb);
180 if (w.second == WireEnd::Source) {
181 wp.source_port = adj_port;
182 add_wire(adj, target(w.first), wp);
183 } else {
184 wp.target_port = adj_port;
185 add_wire(source(w.first), adj, wp);
186 }
187 remove_vertex(new_b);
188 }
189 }
190 for (const ZXVert& v : to_replace.verts_) {
191 remove_vertex(v);
192 }
193}
