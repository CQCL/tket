91 ZXGen_ptr ZXGen::create_gen(ZXType type, QuantumType qtype) {
92  ZXGen_ptr op;
93  switch (type) {
94  case ZXType::Input:
95  case ZXType::Output:
96  case ZXType::Open: {
97  op = std::make_shared<const BoundaryGen>(type, qtype);
98  break;
99  }
100  case ZXType::ZSpider:
101  case ZXType::XSpider:
102  case ZXType::XY:
103  case ZXType::XZ:
104  case ZXType::YZ: {
105  op = std::make_shared<const PhasedGen>(type, 0., qtype);
106  break;
107  }
108  case ZXType::Hbox: {
109  op = std::make_shared<const PhasedGen>(type, -1., qtype);
110  break;
111  }
112  case ZXType::PX:
113  case ZXType::PY:
114  case ZXType::PZ: {
115  op = std::make_shared<const CliffordGen>(type, false, qtype);
116  break;
117  }
118  case ZXType::Triangle: {
119  op = std::make_shared<const DirectedGen>(type, qtype);
120  break;
121  }
122  default:
123  throw ZXError("Cannot instantiate a ZXGen of the required type");
124  }
125  return op;
126 }

416  if (!port) return false;

442  ZXVertVec boundary = diag_->get_boundary();
443  std::vector<QuantumType> sig;
444  for (const ZXVert& b : boundary) {
