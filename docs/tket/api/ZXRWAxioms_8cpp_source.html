23 bool Rewrite::red_to_green_fun(ZXDiagram& diag) {
24  bool success = false;
25  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
26  if (diag.get_zxtype(v) != ZXType::XSpider) continue;
27  // Found a match
28  success = true;
29  // Apply hadamards all around the spider
30  for (Wire& w : diag.adj_wires(v)) {
31  (*diag.graph)[w].type = ((*diag.graph)[w].type == ZXWireType::H)
32  ? ZXWireType::Basic
33  : ZXWireType::H;
34  }
35  // Replace X spider with Z spider
36  const PhasedGen& x = diag.get_vertex_ZXGen<PhasedGen>(v);
37  ZXGen_ptr z = std::make_shared<const PhasedGen>(
38  ZXType::ZSpider, x.get_param(), *x.get_qtype());
39  diag.set_vertex_ZXGen_ptr(v, z);
40  }
41  return success;
42 }

46 bool Rewrite::spider_fusion_fun(ZXDiagram& diag) {
47  bool success = false;
48  std::set<ZXVert> bin;
49  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
50  if (bin.contains(v)) continue;
51  ZXType vtype = diag.get_zxtype(v);
52  if (!is_spider_type(vtype)) continue;
59  WireVec adj_vec = diag.adj_wires(v);
60  std::list<Wire> adj_list{adj_vec.begin(), adj_vec.end()};
61  while (!adj_list.empty()) {
62  Wire w = adj_list.front();
63  adj_list.pop_front();
64  ZXWireType wtype = diag.get_wire_type(w);
65  ZXVert u = diag.other_end(w, v);
66  if (bin.contains(u)) continue;
67  ZXType utype = diag.get_zxtype(u);
68  bool same_colour = vtype == utype;
69  if (!is_spider_type(utype) || u == v ||
70  (wtype == ZXWireType::Basic) != same_colour)
71  continue;
72  // The spiders `u` and `v` can be fused together
73  // We merge into `v` and remove `u` so that we can efficiently continue to
74  // search the neighbours
75  const PhasedGen& vspid = diag.get_vertex_ZXGen<PhasedGen>(v);
76  const PhasedGen& uspid = diag.get_vertex_ZXGen<PhasedGen>(u);
77  ZXGen_ptr new_spid = std::make_shared<const PhasedGen>(
78  vtype, vspid.get_param() + uspid.get_param(),
79  (vspid.get_qtype() == QuantumType::Classical ||
80  uspid.get_qtype() == QuantumType::Classical)
81  ? QuantumType::Classical
82  : QuantumType::Quantum);
83  diag.set_vertex_ZXGen_ptr(v, new_spid);
84  for (const Wire& uw : diag.adj_wires(u)) {
85  WireEnd u_end = diag.end_of(uw, u);
86  ZXVert other = diag.other_end(uw, u);
87  WireProperties uwp = diag.get_wire_info(uw);
88  // Wires may need flipping type to match colours
89  if (!same_colour)
90  uwp.type = (uwp.type == ZXWireType::Basic) ? ZXWireType::H
91  : ZXWireType::Basic;
96  if (other == v && uwp.type == ZXWireType::Basic) continue;
97  // Self loops on `u` needs to become self loops on `v`
98  if (other == u) other = v;
99  // Connect edge to `v` instead with the same properties.
100  Wire new_w;
101  if (u_end == WireEnd::Source)
102  new_w = diag.add_wire(v, other, uwp);
103  else
104  new_w = diag.add_wire(other, v, uwp);
105  // Iteratively fuse along new wire if possible
106  adj_list.push_back(new_w);
107  }
108  // Remove `u`
109  bin.insert(u);
110  success = true;
111  }
112  }
113  for (ZXVert u : bin) {
114  diag.remove_vertex(u);
115  }
116  return success;
117 }

121 bool Rewrite::self_loop_removal_fun(ZXDiagram& diag) {
122  bool success = false;
123  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
124  ZXType vtype = diag.get_zxtype(v);
125  if (!is_spider_type(vtype)) continue;
126  unsigned n_pis = 0;
127  QuantumType vqtype = *diag.get_qtype(v);
128  for (const Wire& w : diag.adj_wires(v)) {
129  if (diag.other_end(w, v) != v) continue;
130  // Found a self-loop
131  ZXWireType wtype = diag.get_wire_type(w);
132  QuantumType wqtype = diag.get_qtype(w);
141  if ((vqtype == QuantumType::Quantum ||
142  wqtype == QuantumType::Classical) &&
143  wtype == ZXWireType::H)
144  ++n_pis;
145  diag.remove_wire(w);
146  success = true;
147  }
148  if ((n_pis % 2) == 1) {
149  const PhasedGen& spid = diag.get_vertex_ZXGen<PhasedGen>(v);
150  ZXGen_ptr new_spid = std::make_shared<const PhasedGen>(
151  vtype, spid.get_param() + 1., vqtype);
152  diag.set_vertex_ZXGen_ptr(v, new_spid);
153  }
154  }
155  return success;
156 }

160 bool Rewrite::parallel_h_removal_fun(ZXDiagram& diag) {
161  bool success = false;
162  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
163  ZXType vtype = diag.get_zxtype(v);
164  if (!is_spider_type(vtype)) continue;
165  QuantumType vqtype = *diag.get_qtype(v);
166  std::map<ZXVert, Wire> h_wires;
167  for (const Wire& w : diag.adj_wires(v)) {
168  ZXWireType wtype = diag.get_wire_type(w);
169  ZXVert u = diag.other_end(w, v);
170  ZXType utype = diag.get_zxtype(u);
171  if (!is_spider_type(utype)) continue;
172  if ((wtype == ZXWireType::H) != (utype == vtype)) continue;
173  // This is (effectively) a Hadamard edge
174  QuantumType uqtype = *diag.get_qtype(u);
175  QuantumType wqtype = diag.get_qtype(w);
176  if (vqtype == QuantumType::Classical &&
177  uqtype == QuantumType::Classical && wqtype == QuantumType::Quantum) {
178  // Doubled wire forms a pair
179  diag.remove_wire(w);
180  success = true;
181  continue;
182  }
183  // Look for another wire to pair it with
184  auto added = h_wires.insert({u, w});
185  if (!added.second) {
186  // Already found the other of the pair, so remove both
187  Wire other_w = added.first->second;
188  h_wires.erase(added.first);
189  diag.remove_wire(w);
190  diag.remove_wire(other_w);
191  success = true;
192  }
193  }
194  }
195  return success;
196 }
