23 bool Rewrite::decompose_boxes_fun(ZXDiagram& diag) {
24  bool success = false;
25  std::list<ZXVert> to_decompose;
26  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
27  if (diag.get_zxtype(v) == ZXType::ZXBox) to_decompose.push_back(v);
28  }
29  for (const ZXVert& box : to_decompose) {
30  const ZXBox& zxb = diag.get_vertex_ZXGen<ZXBox>(box);
31  ZXDiagram inner(*zxb.get_diagram());
32  decompose_boxes_fun(inner);
33  std::vector<std::pair<Wire, WireEnd>> sub_boundary(zxb.n_ports());
34  for (const Wire& w : diag.adj_wires(box)) {
35  if (diag.source(w) == box)
36  sub_boundary.at(*diag.source_port(w)) = {w, WireEnd::Source};
37  if (diag.target(w) == box)
38  sub_boundary.at(*diag.target_port(w)) = {w, WireEnd::Target};
39  }
40  ZXDiagram::Subdiagram sub{sub_boundary, {box}};
41  diag.substitute(inner, sub);
42  success = true;
43  }
44  return success;
45 }

49 bool Rewrite::basic_wires_fun(ZXDiagram& diag) {
50  ZXGen_ptr qhad =
51  std::make_shared<const PhasedGen>(ZXType::Hbox, -1, QuantumType::Quantum);
52  ZXGen_ptr chad = std::make_shared<const PhasedGen>(
53  ZXType::Hbox, -1, QuantumType::Classical);
54  WireVec targets;
55  BGL_FORALL_EDGES(w, *diag.graph, ZXGraph) {
56  if (diag.get_wire_type(w) == ZXWireType::H) targets.push_back(w);
57  }
58  for (const Wire& w : targets) {
59  WireProperties wp = diag.get_wire_info(w);
60  ZXGen_ptr had = (wp.qtype == QuantumType::Quantum) ? qhad : chad;
61  ZXVert h = diag.add_vertex(had);
62  ZXVert s = diag.source(w);
63  ZXVert t = diag.target(w);
64  wp.type = ZXWireType::Basic;
65  WireProperties wp2 = wp;
66  wp.target_port = std::nullopt;
67  wp2.source_port = std::nullopt;
68  diag.add_wire(s, h, wp);
69  diag.add_wire(h, t, wp);
70  diag.remove_wire(w);
71  }
72  return !targets.empty();
73 }

77 bool Rewrite::rebase_to_zx_fun(ZXDiagram& diag) {
78  ZXVertVec verts;
79  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
80  ZXType t = diag.get_zxtype(v);
81  if (!is_boundary_type(t) && !is_spider_type(t)) verts.push_back(v);
82  }
83  for (const ZXVert& v : verts) {
84  switch (diag.get_zxtype(v)) {
85  case ZXType::Hbox: {
86  // Use a combination of the equations in doi:10.1145/3209108.3209128 and
87  // doi:10.4204/EPTCS.340.16 Iteratively decompose based on the phase, so
88  // cannot be applied to symbolic phases
89  const PhasedGen& vg = diag.get_vertex_ZXGen<PhasedGen>(v);
90  std::optional<Complex> opt_ph = eval_expr_c(vg.get_param());
91  if (!opt_ph)
92  throw ZXError(
93  "Hbox with symbolic phase cannot be decomposed into ZX "
94  "generators");
95  QuantumType qt = *vg.get_qtype();
96  unsigned deg = diag.degree(v);
97  WireVec v_ws = diag.adj_wires(v);
98  std::vector<std::pair<Wire, WireEnd>> v_bounds;
99 
100  ZXDiagram rep(0, 0, 0, 0);
101  double r = std::abs(*opt_ph - 1.);
102  double ph = std::arg(*opt_ph - 1.) / PI;
103  // Reduce r to the range [0, 2]
104  if (r < 0.) {
105  r *= -1.;
106  ph += 1.;
107  }
108  // Core is an algebraic Zbox surrounded by triangles
109  ZXVert zph = rep.add_vertex(ZXType::ZSpider, Expr(ph), qt);
110  // Not every will may have the same QuantumType in general
111  bool all_same_qt = true;
112  for (const Wire& w : diag.adj_wires(v)) {
113  QuantumType wqt = diag.get_qtype(w);
114  if (qt != wqt) all_same_qt = false;
115  if (diag.source(w) == v) {
116  v_bounds.push_back({w, WireEnd::Source});
117  ZXVert bound = rep.add_vertex(ZXType::Open, wqt);
118  rep.boundary.push_back(bound);
119  ZXVert tri = rep.add_vertex(ZXType::Triangle, wqt);
120  rep.add_wire(bound, tri, ZXWireType::Basic, wqt, std::nullopt, 0);
121  rep.add_wire(tri, zph, ZXWireType::Basic, wqt, 1, std::nullopt);
122  }
123  if (diag.target(w) == v) {
124  v_bounds.push_back({w, WireEnd::Target});
125  ZXVert bound = rep.add_vertex(ZXType::Open, wqt);
126  rep.boundary.push_back(bound);
127  ZXVert tri = rep.add_vertex(ZXType::Triangle, wqt);
128  rep.add_wire(bound, tri, ZXWireType::Basic, wqt, std::nullopt, 0);
129  rep.add_wire(tri, zph, ZXWireType::Basic, wqt, 1, std::nullopt);
130  }
131  }
132  // Special cases for degree 2
133  if (deg == 2 && all_same_qt && approx_eq(opt_ph->real(), -1.) &&
134  approx_0(opt_ph->imag())) {
135  WireVec ws = diag.adj_wires(v);
136  if (ws.size() == 1) {
137  // self-loop
138  diag.multiply_scalar(0.);
139  } else {
140  // Replace with a Hadamard edge
141  ZXVert s = diag.other_end(ws.at(0), v);
142  ZXVert t = diag.other_end(ws.at(1), v);
143  unsigned n_hs = 0;
144  if (diag.get_wire_type(ws.at(0)) == ZXWireType::H) ++n_hs;
145  if (diag.get_wire_type(ws.at(1)) == ZXWireType::H) ++n_hs;
146  diag.add_wire(
147  s, t, (n_hs == 1) ? ZXWireType::Basic : ZXWireType::H, qt);
148  }
149  diag.remove_vertex(v);
150  break;
151  }
152  // Using the algebraic fusion rule, can break off 2-boxes (0-phase
153  // ZSpider and a triangle)
154  for (; r > 2.; r -= 1) {
155  ZXVert tri = rep.add_vertex(ZXType::Triangle, qt);
156  ZXVert one = rep.add_vertex(ZXType::ZSpider, qt);
157  rep.add_wire(zph, tri, ZXWireType::Basic, qt, std::nullopt, 0);
158  rep.add_wire(tri, one, ZXWireType::Basic, qt, 1, std::nullopt);
159  }
160  // Identify alpha s.t. r = e^{i*pi*alpha} + e^{-i*pi*alpha} =
161  // 2*cos(alpha) and implement the algebraic Zbox
162  double alpha = std::acos(r / 2.);
163  ZXVert tri = rep.add_vertex(ZXType::Triangle, qt);
164  ZXVert negal = rep.add_vertex(ZXType::ZSpider, Expr(-alpha), qt);
165  ZXVert xmerge = rep.add_vertex(ZXType::XSpider, qt);
166  ZXVert al = rep.add_vertex(ZXType::ZSpider, Expr(alpha), qt);
167  rep.add_wire(zph, tri, ZXWireType::Basic, qt, std::nullopt, 0);
168  rep.add_wire(tri, negal, ZXWireType::Basic, qt, 1, std::nullopt);
169  rep.add_wire(zph, xmerge, ZXWireType::Basic, qt);
170  rep.add_wire(negal, xmerge, ZXWireType::Basic, qt);
171  rep.add_wire(xmerge, al, ZXWireType::Basic, qt);
172  rep.multiply_scalar(
173  (qt == QuantumType::Quantum) ? Expr(2.)
174  : Expr(SymEngine::sqrt(Expr(2.))));
175 
176  // Decompose triangles
177  rebase_to_zx_fun(rep);
178 
179  // Substitute
180  ZXDiagram::Subdiagram sub{v_bounds, {v}};
181  diag.substitute(rep, sub);
182  break;
183  }
184  case ZXType::XY: {
185  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
186  const PhasedGen& vg = static_cast<const PhasedGen&>(*vgen);
187  ZXGen_ptr new_gen = ZXGen::create_gen(
188  ZXType::ZSpider, -vg.get_param(), *vg.get_qtype());
189  diag.set_vertex_ZXGen_ptr(v, new_gen);
190  break;
191  }
192  case ZXType::XZ: {
193  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
194  const PhasedGen& vg = static_cast<const PhasedGen&>(*vgen);
195  ZXVert phase_v =
196  diag.add_vertex(ZXType::XSpider, vg.get_param(), *vg.get_qtype());
197  ZXGen_ptr new_gen =
198  ZXGen::create_gen(ZXType::ZSpider, Expr(0.5), *vg.get_qtype());
199  diag.set_vertex_ZXGen_ptr(v, new_gen);
200  diag.add_wire(v, phase_v, ZXWireType::Basic, *vg.get_qtype());
201  break;
202  }
203  case ZXType::YZ: {
204  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
205  const PhasedGen& vg = static_cast<const PhasedGen&>(*vgen);
206  ZXVert phase_v =
207  diag.add_vertex(ZXType::XSpider, vg.get_param(), *vg.get_qtype());
208  ZXGen_ptr new_gen = ZXGen::create_gen(ZXType::ZSpider, *vg.get_qtype());
209  diag.set_vertex_ZXGen_ptr(v, new_gen);
210  diag.add_wire(v, phase_v, ZXWireType::Basic, *vg.get_qtype());
211  break;
212  }
213  case ZXType::PX: {
214  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
215  const CliffordGen& vg = static_cast<const CliffordGen&>(*vgen);
216  ZXGen_ptr new_gen = ZXGen::create_gen(
217  ZXType::ZSpider, vg.get_param() ? Expr(1.) : Expr(0.),
218  *vg.get_qtype());
219  diag.set_vertex_ZXGen_ptr(v, new_gen);
220  break;
221  }
222  case ZXType::PY: {
223  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
224  const CliffordGen& vg = static_cast<const CliffordGen&>(*vgen);
225  ZXGen_ptr new_gen = ZXGen::create_gen(
226  ZXType::ZSpider, vg.get_param() ? Expr(0.5) : Expr(-0.5),
227  *vg.get_qtype());
228  diag.set_vertex_ZXGen_ptr(v, new_gen);
229  break;
230  }
231  case ZXType::PZ: {
232  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
233  const CliffordGen& vg = static_cast<const CliffordGen&>(*vgen);
234  ZXVert phase_v = diag.add_vertex(
235  ZXType::XSpider, vg.get_param() ? Expr(1.) : Expr(0.),
236  *vg.get_qtype());
237  ZXGen_ptr new_gen = ZXGen::create_gen(ZXType::ZSpider, *vg.get_qtype());
238  diag.set_vertex_ZXGen_ptr(v, new_gen);
239  diag.add_wire(v, phase_v, ZXWireType::Basic, *vg.get_qtype());
240  break;
241  }
242  case ZXType::Triangle: {
243  // Decomposition of the triangle given in Lemma 3.3, "Completeness of
244  // the ZX-calculus for Pure Qubit Clifford+T Quantum Mechanics", K. Feng
245  // Ng, Q. Wang, 2018
246  ZXDiagram tri(0, 0, 0, 0);
247  QuantumType qt = *diag.get_qtype(v);
248  ZXVert in = tri.add_vertex(ZXType::Input, qt);
249  tri.boundary.push_back(in);
250  ZXVert out = tri.add_vertex(ZXType::Output, qt);
251  tri.boundary.push_back(out);
252  tri.multiply_scalar(
253  (qt == QuantumType::Quantum) ? Expr(2.)
254  : Expr(SymEngine::sqrt(Expr(2.))));
255  ZXVert split = tri.add_vertex(ZXType::XSpider, qt);
256  ZXVert lrz = tri.add_vertex(ZXType::ZSpider, -0.25, qt);
257  ZXVert rrz = tri.add_vertex(ZXType::ZSpider, 0.25, qt);
258  ZXVert laxis = tri.add_vertex(ZXType::XSpider, qt);
259  ZXVert raxis = tri.add_vertex(ZXType::XSpider, qt);
260  ZXVert lph = tri.add_vertex(ZXType::ZSpider, -0.25, qt);
261  ZXVert rph = tri.add_vertex(ZXType::ZSpider, 0.25, qt);
262  ZXVert merge = tri.add_vertex(ZXType::ZSpider, qt);
263  // Flat of the triangle is the output/port 1
264  tri.add_wire(out, split, ZXWireType::Basic, qt);
265  tri.add_wire(split, lrz, ZXWireType::Basic, qt);
266  tri.add_wire(split, rrz, ZXWireType::Basic, qt);
267  tri.add_wire(lrz, laxis, ZXWireType::Basic, qt);
268  tri.add_wire(rrz, raxis, ZXWireType::Basic, qt);
269  tri.add_wire(laxis, lph, ZXWireType::Basic, qt);
270  tri.add_wire(raxis, rph, ZXWireType::Basic, qt);
271  tri.add_wire(laxis, merge, ZXWireType::Basic, qt);
272  tri.add_wire(raxis, merge, ZXWireType::Basic, qt);
273  // Point of the triangle is the input/port 0
274  tri.add_wire(merge, in, ZXWireType::Basic, qt);
275 
276  Wire w0 = diag.wire_at_port(v, 0);
277  Wire w1 = diag.wire_at_port(v, 1);
278  WireEnd we0, we1;
279  if (w0 == w1) {
280  if (diag.source_port(w0) == 0) {
281  we0 = WireEnd::Source;
282  we1 = WireEnd::Target;
283  } else {
284  we0 = WireEnd::Target;
285  we1 = WireEnd::Source;
286  }
287  } else {
288  we0 = diag.end_of(w0, v);
289  we1 = diag.end_of(w1, v);
290  }
291  ZXDiagram::Subdiagram sub{{{w0, we0}, {w1, we1}}, {v}};
292  diag.substitute(tri, sub);
293  break;
294  }
295  default:
296  break;
297  }
298  }
299  return !verts.empty();
300 }

304 bool Rewrite::rebase_to_mbqc_fun(ZXDiagram& diag) {
305  ZXVertVec verts;
306  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
307  ZXType t = diag.get_zxtype(v);
308  if (!is_boundary_type(t) && !is_MBQC_type(t)) verts.push_back(v);
309  }
310  for (const ZXVert& v : verts) {
311  switch (diag.get_zxtype(v)) {
312  case ZXType::ZSpider: {
313  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
314  const PhasedGen& vg = static_cast<const PhasedGen&>(*vgen);
315  std::optional<unsigned> pi2_mult = equiv_Clifford(vg.get_param());
316  ZXGen_ptr new_gen;
317  if (pi2_mult) {
318  if (*pi2_mult % 2 == 0)
319  new_gen = ZXGen::create_gen(
320  ZXType::PX, (*pi2_mult % 4 == 2), *vg.get_qtype());
321  else
322  new_gen = ZXGen::create_gen(
323  ZXType::PY, (*pi2_mult % 4 == 1), *vg.get_qtype());
324  } else
325  new_gen =
326  ZXGen::create_gen(ZXType::XY, -vg.get_param(), *vg.get_qtype());
327  diag.set_vertex_ZXGen_ptr(v, new_gen);
328  break;
329  }
330  case ZXType::XSpider: {
331  ZXGen_ptr vgen = diag.get_vertex_ZXGen_ptr(v);
332  const PhasedGen& vg = static_cast<const PhasedGen&>(*vgen);
333  std::optional<unsigned> pi2_mult = equiv_Clifford(vg.get_param());
334  ZXGen_ptr new_gen;
335  if (pi2_mult) {
336  if (*pi2_mult % 2 == 0)
337  new_gen = ZXGen::create_gen(
338  ZXType::PX, (*pi2_mult % 4 == 2), *vg.get_qtype());
339  else
340  new_gen = ZXGen::create_gen(
341  ZXType::PY, (*pi2_mult % 4 == 1), *vg.get_qtype());
342  } else
343  new_gen =
344  ZXGen::create_gen(ZXType::XY, -vg.get_param(), *vg.get_qtype());
345  diag.set_vertex_ZXGen_ptr(v, new_gen);
346  for (const Wire& w : diag.adj_wires(v)) {
347  ZXWireType wt = diag.get_wire_type(w);
348  diag.set_wire_type(
349  w, (wt == ZXWireType::Basic) ? ZXWireType::H : ZXWireType::Basic);
350  }
351  break;
352  }
353  case ZXType::Hbox: {
354  WireVec v_ws = diag.adj_wires(v);
355  std::vector<std::pair<Wire, WireEnd>> v_bounds;
356  for (const Wire& w : v_ws) {
357  if (diag.source(w) == v) v_bounds.push_back({w, WireEnd::Source});
358  if (diag.target(w) == v) v_bounds.push_back({w, WireEnd::Target});
359  }
360  ZXDiagram::Subdiagram sub{v_bounds, {v}};
361  ZXDiagram h = sub.to_diagram(diag);
362  rebase_to_zx_fun(h);
363  rebase_to_mbqc_fun(h);
364  diag.substitute(h, sub);
365  break;
366  }
367  case ZXType::Triangle: {
368  Wire w0 = diag.wire_at_port(v, 0);
369  Wire w1 = diag.wire_at_port(v, 1);
370  WireEnd we0, we1;
371  if (w0 == w1) {
372  if (diag.source_port(w0) == 0) {
373  we0 = WireEnd::Source;
374  we1 = WireEnd::Target;
375  } else {
376  we0 = WireEnd::Target;
377  we1 = WireEnd::Source;
378  }
379  } else {
380  we0 = diag.end_of(w0, v);
381  we1 = diag.end_of(w1, v);
382  }
383  ZXDiagram::Subdiagram sub{{{w0, we0}, {w1, we1}}, {v}};
384  ZXDiagram tri = sub.to_diagram(diag);
385  rebase_to_zx_fun(tri);
386  rebase_to_mbqc_fun(tri);
387  diag.substitute(tri, sub);
388  break;
389  }
390  default:
391  break;
392  }
393  }
394  return !verts.empty();
395 }
