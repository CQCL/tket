21 bool Rewrite::separate_boundaries_fun(ZXDiagram& diag) {
22  bool success = false;
23  for (const ZXVert& b : diag.get_boundary()) {
24  // Boundaries in valid diagrams have degree 1
25  Wire w = diag.adj_wires(b).at(0);
26  // Since degree is 1, o is distinct from b
27  ZXVert o = diag.other_end(w, b);
28  // Other end of the wire needs to either be a boundary type or connected to
29  // another boundary
30  if (!is_boundary_type(diag.get_zxtype(o))) {
31  bool o_shared = false;
32  for (const ZXVert& n : diag.neighbours(o)) {
33  if (n != b && is_boundary_type(diag.get_zxtype(n))) {
34  o_shared = true;
35  break;
36  }
37  }
38  if (!o_shared) continue;
39  }
40  // New wires will inherit `w`'s `qtype`
41  QuantumType wq = diag.get_qtype(w);
42  ZXGen_ptr id = std::make_shared<const PhasedGen>(ZXType::ZSpider, 0., wq);
43  ZXVert z_at_b = diag.add_vertex(id);
44  diag.add_wire(b, z_at_b, ZXWireType::Basic, wq);
45  ZXVert z_at_o = diag.add_vertex(id);
46  diag.add_wire(o, z_at_o, ZXWireType::Basic, wq);
47  if (diag.get_wire_type(w) == ZXWireType::Basic) {
48  // Basic requires two Hadamard edges
49  ZXVert middle = diag.add_vertex(id);
50  diag.add_wire(z_at_b, middle, ZXWireType::H, wq);
51  diag.add_wire(z_at_o, middle, ZXWireType::H, wq);
52  } else {
53  // Hadamard edge
54  diag.add_wire(z_at_b, z_at_o, ZXWireType::H, wq);
55  }
56  diag.remove_wire(w);
57  success = true;
58  }
59  return success;
60 }

66 bool Rewrite::io_extension_fun(ZXDiagram& diag) {
67  bool success = false;
68  for (const ZXVert& b : diag.get_boundary()) {
69  // Boundaries in valid diagrams have degree 1
70  Wire w = diag.adj_wires(b).at(0);
71  WireProperties wp = diag.get_wire_info(w);
72  if (wp.type == ZXWireType::Basic) continue;
73  // Extend by an identity spider
74  ZXVert u = diag.other_end(w, b);
75  ZXVert z = diag.add_vertex(ZXType::ZSpider, 0., wp.qtype);
76  // `u` might be directed, so preserve ports by preserving direction
77  if (diag.end_of(w, u) == WireEnd::Source)
78  diag.add_wire(u, z, wp);
79  else
80  diag.add_wire(z, u, wp);
81  diag.add_wire(b, z, ZXWireType::Basic, wp.qtype);
82  diag.remove_wire(w);
83  success = true;
84  }
85  return success;
86 }
