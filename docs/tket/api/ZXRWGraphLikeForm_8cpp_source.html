21bool Rewrite::separate_boundaries_fun(ZXDiagram& diag) {
22 bool success = false;
23 for (const ZXVert& b : diag.get_boundary()) {
24 // Boundaries in valid diagrams have degree 1
25 Wire w = diag.adj_wires(b).at(0);
26 // Since degree is 1, o is distinct from b
27 ZXVert o = diag.other_end(w, b);
28 // Other end of the wire needs to either be a boundary type or connected to
29 // another boundary
30 if (!is_boundary_type(diag.get_zxtype(o))) {
31 bool o_shared = false;
32 for (const ZXVert& n : diag.neighbours(o)) {
33 if (n != b && is_boundary_type(diag.get_zxtype(n))) {
34 o_shared = true;
35 break;
36 }
37 }
38 if (!o_shared) continue;
39 }
40 // New wires will inherit `w`'s `qtype`
41 QuantumType wq = diag.get_qtype(w);
42 ZXGen_ptr id = std::make_shared<const PhasedGen>(ZXType::ZSpider, 0., wq);
43 ZXVert z_at_b = diag.add_vertex(id);
44 diag.add_wire(b, z_at_b, ZXWireType::Basic, wq);
45 ZXVert z_at_o;
46 if (diag.get_zxtype(o) == ZXType::ZSpider) {
47 z_at_o = o;
48 } else {
49 z_at_o = diag.add_vertex(id);
50 diag.add_wire(o, z_at_o, ZXWireType::Basic, wq);
51 }
52 if (diag.get_wire_type(w) == ZXWireType::Basic) {
53 // Basic requires two Hadamard edges
54 ZXVert middle = diag.add_vertex(id);
55 diag.add_wire(z_at_b, middle, ZXWireType::H, wq);
56 diag.add_wire(z_at_o, middle, ZXWireType::H, wq);
57 } else {
58 // Hadamard edge
59 diag.add_wire(z_at_b, z_at_o, ZXWireType::H, wq);
60 }
61 diag.remove_wire(w);
62 success = true;
63 }
64 return success;
65}

67Rewrite Rewrite::separate_boundaries() {
68 return Rewrite(separate_boundaries_fun);
69}

71bool Rewrite::io_extension_fun(ZXDiagram& diag) {
72 bool success = false;
73 for (const ZXVert& b : diag.get_boundary()) {
74 // Boundaries in valid diagrams have degree 1
75 Wire w = diag.adj_wires(b).at(0);
76 WireProperties wp = diag.get_wire_info(w);
77 if (wp.type == ZXWireType::Basic) continue;
78 // Extend by an identity spider
79 ZXVert u = diag.other_end(w, b);
80 ZXVert z = diag.add_vertex(ZXType::ZSpider, 0., wp.qtype);
81 // `u` might be directed, so preserve ports by preserving direction
82 if (diag.end_of(w, u) == WireEnd::Source)
83 diag.add_wire(u, z, wp);
84 else
85 diag.add_wire(z, u, wp);
86 diag.add_wire(b, z, ZXWireType::Basic, wp.qtype);
87 diag.remove_wire(w);
88 success = true;
89 }
90 return success;
91}
