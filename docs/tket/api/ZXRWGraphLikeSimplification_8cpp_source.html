27 static bool can_complement_neighbourhood(
28  const ZXDiagram& diag, QuantumType vqtype, const ZXVertVec& neighbours) {
29  for (const ZXVert& n : neighbours) {
30  if (diag.get_zxtype(n) != ZXType::ZSpider ||
31  (vqtype == QuantumType::Classical &&
32  *diag.get_qtype(n) == QuantumType::Quantum))
33  return false;
34  }
35  return true;
36 }

38 bool Rewrite::remove_interior_cliffords_fun(ZXDiagram& diag) {
39  if (!diag.is_graphlike()) return false;
40  bool success = false;
41  ZXVertSeqSet candidates;
42  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) { candidates.insert(v); }
43  auto& view = candidates.get<TagSeq>();
44  while (!candidates.empty()) {
45  auto it = view.begin();
46  ZXVert v = *it;
47  view.erase(it);
48  if (!diag.is_proper_clifford_spider(v)) continue;
49  const PhasedGen& spid = diag.get_vertex_ZXGen<PhasedGen>(v);
50  QuantumType vqtype = *spid.get_qtype();
51  ZXVertVec neighbours = diag.neighbours(v);
52  if (!can_complement_neighbourhood(diag, vqtype, neighbours)) continue;
53  // Found an internal proper clifford spider on which we can perform local
54  // complementation
59  auto xi = neighbours.begin(), x_end = neighbours.end();
60  for (; xi != x_end; ++xi) {
61  for (auto yi = xi + 1; yi != x_end; ++yi) {
62  // Don't add a doubled edge between classicals to preserve graph-like
63  if (!(vqtype == QuantumType::Quantum &&
64  *diag.get_qtype(*xi) == QuantumType::Classical &&
65  *diag.get_qtype(*yi) == QuantumType::Classical)) {
66  std::optional<Wire> wire = diag.wire_between(*xi, *yi);
67  if (wire)
68  diag.remove_wire(*wire);
69  else
70  diag.add_wire(*xi, *yi, ZXWireType::H, vqtype);
71  }
72  }
73  const PhasedGen& xi_op = diag.get_vertex_ZXGen<PhasedGen>(*xi);
74  // If `v` is Quantum, Classical neighbours will pick up both the +theta
75  // and -theta phases, cancelling out
76  if (vqtype == QuantumType::Quantum &&
77  *xi_op.get_qtype() == QuantumType::Classical)
78  continue;
79  // Update phase information
80  ZXGen_ptr xi_new_op = std::make_shared<const PhasedGen>(
81  ZXType::ZSpider, xi_op.get_param() - spid.get_param(),
82  *xi_op.get_qtype());
83  diag.set_vertex_ZXGen_ptr(*xi, xi_new_op);
84  candidates.insert(
85  *xi); // Changing the phase could introduce a new proper Clifford
86  }
87  diag.remove_vertex(v);
88  success = true;
89  }
90  return success;
91 }

97 static void add_phase_to_vertices(
98  ZXDiagram& diag, const ZXVertSeqSet& verts, const Expr& phase) {
99  for (const ZXVert& v : verts.get<TagSeq>()) {
100  const PhasedGen& old_spid = diag.get_vertex_ZXGen<PhasedGen>(v);
101  ZXGen_ptr new_spid = std::make_shared<const PhasedGen>(
102  ZXType::ZSpider, old_spid.get_param() + phase, *old_spid.get_qtype());
103  diag.set_vertex_ZXGen_ptr(v, new_spid);
104  }
105 }

107 static void bipartite_complementation(
108  ZXDiagram& diag, const ZXVertSeqSet& sa, const ZXVertSeqSet& sb,
109  QuantumType qtype) {
110  for (const ZXVert& a : sa.get<TagSeq>()) {
111  for (const ZXVert& b : sb.get<TagSeq>()) {
112  // Don't add a doubled edge between classicals to preserve graph-like
113  if (!(qtype == QuantumType::Quantum &&
114  *diag.get_qtype(a) == QuantumType::Classical &&
115  *diag.get_qtype(b) == QuantumType::Classical)) {
116  std::optional<Wire> wire = diag.wire_between(a, b);
117  if (wire)
118  diag.remove_wire(*wire);
119  else
120  diag.add_wire(a, b, ZXWireType::H, qtype);
121  }
122  }
123  }

126 bool Rewrite::remove_interior_paulis_fun(ZXDiagram& diag) {
127  if (!diag.is_graphlike()) return false;
128  bool success = false;
129  ZXVertSeqSet candidates; // Need an indirect iterator as BGL_FORALL_VERTICES
130  // breaks when removing the current vertex
131  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) { candidates.insert(v); }
132  auto& view = candidates.get<TagSeq>();
133  while (!candidates.empty()) {
134  auto it = view.begin();
135  ZXVert v = *it;
136  view.erase(it);
137  // Check `v` is an interior Pauli
138  if (!diag.is_pauli_spider(v)) continue;
139  ZXVertVec v_ns = diag.neighbours(v);
140  QuantumType vqtype = *diag.get_qtype(v);
141  if (!can_complement_neighbourhood(diag, vqtype, v_ns)) continue;
142  // Look for an interior Pauli neighbour
143  bool pair_found = false;
144  ZXVert u;
145  ZXVertVec u_ns;
146  for (const ZXVert& n : v_ns) {
147  if (!diag.is_pauli_spider(n)) continue;
148  ZXVertVec n_ns = diag.neighbours(n);
149  QuantumType nqtype = *diag.get_qtype(n);
150  if (can_complement_neighbourhood(diag, nqtype, n_ns)) {
151  pair_found = true;
152  u = n;
153  u_ns = n_ns;
154  break;
155  }
156  }
157  if (!pair_found) continue;
158  // Found a valid pair
159  // Identify the three sets from the neighbourhoods of `u` and `v`
160  ZXVertSeqSet excl_v{v_ns.begin(), v_ns.end()};
161  excl_v.erase(excl_v.find(u));
162  ZXVertSeqSet excl_u, joint;
163  auto& lookup_v = excl_v.get<TagKey>();
164  for (const ZXVert& nu : u_ns) {
165  if (lookup_v.find(nu) != lookup_v.end()) {
166  joint.insert(nu);
167  excl_v.erase(excl_v.find(nu));
168  } else {
169  excl_u.insert(nu);
170  }
171  }
172  excl_u.erase(excl_u.find(v));
173  const PhasedGen& v_spid = diag.get_vertex_ZXGen<PhasedGen>(v);
174  const PhasedGen& u_spid = diag.get_vertex_ZXGen<PhasedGen>(u);
175 
176  add_phase_to_vertices(
177  diag, joint, v_spid.get_param() + u_spid.get_param() + 1.);
178  add_phase_to_vertices(diag, excl_u, v_spid.get_param());
179  add_phase_to_vertices(diag, excl_v, u_spid.get_param());
180 
181  // Because `can_complement_neighbourhood` checks all neighbours,
182  // v and u have the same QuantumType
183  bipartite_complementation(diag, joint, excl_u, vqtype);
184  bipartite_complementation(diag, joint, excl_v, vqtype);
185  bipartite_complementation(diag, excl_u, excl_v, vqtype);
186 
187  diag.remove_vertex(u);
188  diag.remove_vertex(v);
189  candidates.erase(candidates.find(u));
190  success = true;
191  }
192  return success;
193 }

199 bool Rewrite::gadgetise_interior_paulis_fun(ZXDiagram& diag) {
200  if (!diag.is_graphlike()) return false;
201  bool success = false;
202  ZXVertVec candidates; // Need an indirect iterator as BGL_FORALL_VERTICES
203  // breaks when removing the current vertex
204  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) { candidates.push_back(v); }
205  for (const ZXVert& v : candidates) {
206  // Check `v` is an interior Pauli
207  if (!diag.is_pauli_spider(v)) continue;
208  ZXVertVec v_ns = diag.neighbours(v);
209  QuantumType vqtype = *diag.get_qtype(v);
210  if (!can_complement_neighbourhood(diag, vqtype, v_ns)) continue;
211  // Check it isn't already the axis of a gadget
212  bool is_axis = false;
213  for (const ZXVert& n : v_ns) {
214  if (diag.degree(n) == 1) {
215  is_axis = true;
216  break;
217  }
218  }
219  if (is_axis) continue;
220  // Pick a neighbour for pivoting
221  bool pair_found = false;
222  ZXVert u;
223  ZXVertVec u_ns;
224  for (const ZXVert& n : v_ns) {
225  ZXVertVec n_ns = diag.neighbours(n);
226  QuantumType nqtype = *diag.get_qtype(n);
227  if (can_complement_neighbourhood(diag, nqtype, n_ns)) {
228  pair_found = true;
229  u = n;
230  u_ns = n_ns;
231  break;
232  }
233  }
234  if (!pair_found) continue;
235  // Found a valid pair
236  // Identify the three sets from the neighbourhoods of `u` and `v`
237  ZXVertSeqSet excl_v{v_ns.begin(), v_ns.end()};
238  excl_v.erase(excl_v.find(u));
239  ZXVertSeqSet excl_u, joint;
240  auto& lookup_v = excl_v.get<TagKey>();
241  for (const ZXVert& nu : u_ns) {
242  auto found_nu = lookup_v.find(nu);
243  if (found_nu != lookup_v.end()) {
244  joint.insert(nu);
245  lookup_v.erase(found_nu);
246  } else {
247  excl_u.insert(nu);
248  }
249  }
250  excl_u.erase(excl_u.find(v));
251  const PhasedGen& v_spid = diag.get_vertex_ZXGen<PhasedGen>(v);
252  const PhasedGen& u_spid = diag.get_vertex_ZXGen<PhasedGen>(u);
253 
254  add_phase_to_vertices(diag, joint, v_spid.get_param() + 1.);
255  add_phase_to_vertices(diag, excl_u, v_spid.get_param());
256  std::optional<unsigned> pi2_mult = equiv_Clifford(v_spid.get_param());
257  Expr new_phase = ((*pi2_mult % 4 == 0) ? 1. : -1.) * u_spid.get_param();
258  diag.set_vertex_ZXGen_ptr(
259  u, std::make_shared<PhasedGen>(ZXType::ZSpider, new_phase, vqtype));
260  diag.set_vertex_ZXGen_ptr(
261  v, std::make_shared<PhasedGen>(ZXType::ZSpider, 0., vqtype));
262 
263  // Because `can_complement_neighbourhood` checks all neighbours,
264  // v and u have the same QuantumType
265  bipartite_complementation(diag, joint, excl_u, vqtype);
266  bipartite_complementation(diag, joint, excl_v, vqtype);
267  bipartite_complementation(diag, excl_u, excl_v, vqtype);
268 
269  Wire uv = *diag.wire_between(u, v);
270  WireProperties uv_prop = diag.get_wire_info(uv);
271  boost::clear_vertex(u, *diag.graph);
272  diag.add_wire(u, v, uv_prop);
273 
274  // No need to erase from candidates since it should not be a Pauli vertex,
275  // otherwise remove_interior_paulis should have removed the pair
276  success = true;
277  }
278  return success;
279 }

285 bool Rewrite::extend_at_boundary_paulis_fun(ZXDiagram& diag) {
286  if (!diag.is_graphlike()) return false;
287  bool success = false;
288  for (const ZXVert& b : diag.get_boundary()) {
289  // Valid ZX graph requires boundaries to have a unique neighbour
290  Wire bw = diag.adj_wires(b).at(0);
291  ZXVert u = diag.other_end(bw, b);
292  if (!diag.is_pauli_spider(u)) continue;
293  bool has_internal_pauli = false;
294  for (const ZXVert& w : diag.neighbours(u)) {
295  if (!diag.is_pauli_spider(w)) continue;
296  bool interior = true;
297  for (const ZXVert& wn : diag.neighbours(w)) {
298  if (is_boundary_type(diag.get_zxtype(wn))) {
299  interior = false;
300  break;
301  }
302  }
303  if (interior) {
304  has_internal_pauli = true;
305  break;
306  }
307  }
308  if (!has_internal_pauli) continue;
309  // We would like to pivot about (`u`, `w`) but `u` is by a boundary, so we
310  // extend it
311  ZXGen_ptr u_op = diag.get_vertex_ZXGen_ptr(u);
312  QuantumType qtype = *u_op->get_qtype();
313  ZXGen_ptr id =
314  std::make_shared<const PhasedGen>(ZXType::ZSpider, 0., qtype);
315  ZXVert z1 = diag.add_vertex(id);
316  ZXVert z2 = diag.add_vertex(u_op);
317  diag.add_wire(u, z1, ZXWireType::H, qtype);
318  diag.add_wire(z1, z2, ZXWireType::H, qtype);
319  diag.add_wire(z2, b, ZXWireType::Basic, qtype);
320  diag.remove_wire(bw);
321  diag.set_vertex_ZXGen_ptr(u, id);
322  success = true;
323  }
324  return success;
325 }

331 bool Rewrite::merge_gadgets_fun(ZXDiagram& diag) {
332  std::map<std::set<ZXVert>, ZXVert> neighbour_lookup;
333  std::list<ZXVert> to_remove;
334  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
335  if (diag.degree(v) == 1 && diag.get_zxtype(v) == ZXType::ZSpider) {
336  ZXVert axis = diag.neighbours(v).front();
337  if (diag.get_zxtype(axis) != ZXType::ZSpider ||
338  !equiv_expr(diag.get_vertex_ZXGen<PhasedGen>(axis).get_param(), 0.))
339  continue;
340  std::set<ZXVert> neighbours;
341  for (const ZXVert& n : diag.neighbours(axis)) neighbours.insert(n);
342  neighbours.erase(neighbours.find(v));
343  auto inserted = neighbour_lookup.insert({neighbours, v});
344  if (!inserted.second) {
345  ZXVert other_gadget = inserted.first->second;
346  Expr other_param =
347  diag.get_vertex_ZXGen<PhasedGen>(other_gadget).get_param();
348  Expr this_param = diag.get_vertex_ZXGen<PhasedGen>(v).get_param();
349  diag.set_vertex_ZXGen_ptr(
350  other_gadget, std::make_shared<PhasedGen>(
351  ZXType::ZSpider, other_param + this_param));
352  to_remove.push_back(v);
353  to_remove.push_back(axis);
354  }
355  }
356  }
357  for (const ZXVert& v : to_remove) diag.remove_vertex(v);
358  return !to_remove.empty();
359 }
