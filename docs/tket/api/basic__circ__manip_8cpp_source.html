31 // if there are any blank wires in the circuit,
32 // this method removes them and removes the vertices
33 // from boundaries
34 void Circuit::remove_blank_wires() {
35  VertexList bin;
36  unit_vector_t unused_units;
37  const Op_ptr noop = get_op_ptr(OpType::noop);
38  for (const BoundaryElement& el : boundary.get<TagID>()) {
39  Vertex in = el.in_;
40  Vertex out = el.out_;
41  VertexVec succs = get_successors(in);
42  if (succs.size() == 1 && succs.front() == out) {
43  dag[in].op = noop;
44  bin.push_back(in);
45  dag[out].op = noop;
46  bin.push_back(out);
47  unused_units.push_back(el.id_);
48  }
49  }
50  for (const UnitID& u : unused_units) {
51  boundary.get<TagID>().erase(u);
52  }
53  remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
54 }

78 template <>
79 Vertex Circuit::add_op<unsigned>(
80  const Op_ptr& gate, const std::vector<unsigned>& args,
81  std::optional<std::string> opgroup) {
82  op_signature_t sig = gate->get_signature();
83 
84  if (sig.size() != args.size()) {
85  throw CircuitInvalidity(
86  std::to_string(args.size()) + " args provided, but " +
87  gate->get_name() + " requires " + std::to_string(sig.size()));
88  }
89 
90  OpType optype = gate->get_type();
91  unit_vector_t arg_ids;
92  for (unsigned i = 0; i < args.size(); ++i) {
93  switch (sig.at(i)) {
94  case EdgeType::Quantum: {
95  arg_ids.push_back(Qubit(args[i]));
96  break;
97  }
98  case EdgeType::Classical:
99  case EdgeType::Boolean: {
100  arg_ids.push_back(Bit(args[i]));
101  break;
102  }
103  default: {
104  TKET_ASSERT(!"add_op found invalid edge type in signature");
105  }
106  }
107  }
108  if (optype == OpType::CnRy && args.size() == 1) {
109  return add_op(get_op_ptr(OpType::Ry, gate->get_params()), arg_ids);
110  } else if (optype == OpType::CnX && args.size() == 1) {
111  return add_op(get_op_ptr(OpType::X), arg_ids);
112  } else if (optype == OpType::CnZ && args.size() == 1) {
113  return add_op(get_op_ptr(OpType::Z), arg_ids);
114  } else if (optype == OpType::CnY && args.size() == 1) {
115  return add_op(get_op_ptr(OpType::Y), arg_ids);
116  }
117  return add_op(gate, arg_ids, opgroup);
118 }

159 static void append_debug_bits(
160  Circuit& circ, unit_vector_t& args,
161  const std::vector<bool>& expected_readouts,
162  const std::optional<std::string>& postfix) {
163  unsigned n_temp_bits = expected_readouts.size();
164  unsigned n_one_bits =
165  std::accumulate(expected_readouts.begin(), expected_readouts.end(), 0);
166  unsigned n_zero_bits = n_temp_bits - n_one_bits;
167 
168  // Add a classical registers for this assertion
169  std::string zero_reg_name =
170  (postfix == std::nullopt)
171  ? c_debug_zero_prefix() + "_" + c_debug_default_name()
172  : c_debug_zero_prefix() + "_" + postfix.value();
173  std::string one_reg_name =
174  (postfix == std::nullopt)
175  ? c_debug_one_prefix() + "_" + c_debug_default_name()
176  : c_debug_one_prefix() + "_" + postfix.value();
177  std::string incremented_zero_reg_name =
178  circ.get_next_c_reg_name(zero_reg_name);
179  if (n_zero_bits > 0) {
180  circ.add_c_register(incremented_zero_reg_name, n_zero_bits);
181  }
182  std::string incremented_one_reg_name = circ.get_next_c_reg_name(one_reg_name);
183  if (n_one_bits > 0) {
184  circ.add_c_register(incremented_one_reg_name, n_one_bits);
185  }
186 
187  unsigned zero_reg_index = 0;
188  unsigned one_reg_index = 0;
189 
190  for (unsigned i = 0; i < n_temp_bits; i++) {
191  if (expected_readouts[i]) {
192  args.push_back(Bit(incremented_one_reg_name, one_reg_index++));
193  } else {
194  args.push_back(Bit(incremented_zero_reg_name, zero_reg_index++));
195  }
196  }
197 }

199 Vertex Circuit::add_assertion(
200  const ProjectorAssertionBox& assertion_box,
201  const std::vector<Qubit>& qubits, const std::optional<Qubit>& ancilla,
202  const std::optional<std::string>& name) {
203  auto circ_ptr = assertion_box.to_circuit();
204  unsigned log2_dim = log2(assertion_box.get_matrix().rows());
205 
206  if (circ_ptr->n_qubits() > log2_dim && ancilla == std::nullopt) {
207  throw CircuitInvalidity("This assertion requires an ancilla");
208  }
209 
210  if (qubits.size() != log2_dim) {
211  throw CircuitInvalidity(
212  std::to_string(qubits.size()) +
213  " target qubits provided, but the projector requires " +
214  std::to_string(log2_dim));
215  }
216 
217  unit_vector_t args;
218  args.insert(args.end(), qubits.begin(), qubits.end());
219  if (circ_ptr->n_qubits() > log2_dim) {
220  args.push_back(*ancilla);
221  }
222  append_debug_bits(*this, args, assertion_box.get_expected_readouts(), name);
223  return add_op<UnitID>(
224  std::make_shared<ProjectorAssertionBox>(assertion_box), args);
225 }

262 // given vertices and desired in ports for i2 and out ports for i1, adds
263 // edge between them
264 // there are no checks to ensure the vertex exists in the graph
265 Edge Circuit::add_edge(
266  const VertPort& source, const VertPort& target, const EdgeType& type) {
267  // add new edge
268  std::pair<Edge, bool> edge_pairy =
269  boost::add_edge(source.first, target.first, this->dag);
270 
271  TKET_ASSERT(edge_pairy.second); // Cannot create edge between vertices
272 
273  Edge new_E = edge_pairy.first;
274  dag[new_E].ports.first = source.second;
275  dag[new_E].ports.second = target.second;
276  dag[new_E].type = type;
277 
278  return new_E;
279 }

281 // given vertex, eradicates it from dag, returning edges used for rewiring
282 // methods
283 // there are no checks to ensure the vertex exists in the graph. This may cause
284 // a segfault if called on a vertex which doesn't exist any more. Take care with
285 // ordering of vertex removal! has flags for whether you want to a) rewire the
286 // hole left by the vertex and b) destroy vertex once it has been isolated As a
287 // sanity check, you cannot remove boundary vertices; converting them to noops
288 // beforehand is recommended
289 void Circuit::remove_vertex(
290  const Vertex& deadvert, GraphRewiring graph_rewiring,
291  VertexDeletion vertex_deletion) {
292  if (graph_rewiring == GraphRewiring::Yes) {
293  EdgeVec ins = get_in_edges(deadvert);
294  std::vector<EdgeVec> bundles = get_b_out_bundles(deadvert);
295  port_t p = 0;
296  for (const Edge& e : ins) {
297  EdgeType type = get_edgetype(e);
298  if (type != EdgeType::Boolean) {
299  Vertex pred_vert = source(e);
300  port_t pred_port = get_source_port(e);
301  Edge next_e = get_nth_out_edge(deadvert, p);
302  Vertex succ_vert = target(next_e);
303  port_t succ_port = get_target_port(next_e);
304  add_edge({pred_vert, pred_port}, {succ_vert, succ_port}, type);
305  if (type == EdgeType::Classical) {
306  for (const Edge& cr : bundles[p]) {
307  Vertex sv = target(cr);
308  port_t sp = get_target_port(cr);
309  add_edge({pred_vert, pred_port}, {sv, sp}, EdgeType::Boolean);
310  }
311  }
312  }
313  ++p;
314  }
315  }
316 
317  boost::clear_vertex(deadvert, this->dag);
318  if (vertex_deletion == VertexDeletion::Yes) {
319  // Cannot remove a boundary vertex
320  TKET_ASSERT(!detect_boundary_Op(deadvert));
321  boost::remove_vertex(deadvert, this->dag);
322  }
323 }

346 unit_map_t Circuit::flatten_registers() {
347  unit_map_t rename_map;
348  unsigned q_index = 0;
349  unsigned c_index = 0;
350  for (const BoundaryElement& el : boundary.get<TagID>()) {
351  if (el.type() == UnitType::Qubit) {
352  rename_map.insert({el.id_, Qubit(q_index++)});
353  } else {
354  rename_map.insert({el.id_, Bit(c_index++)});
355  }
356  }
357  try {
358  rename_units(rename_map);
359  } catch (const std::exception& e) {
360  std::stringstream ss;
361  ss << "Unable to flatten registers: " << e.what();
362  throw std::runtime_error(ss.str());
363  }
364  return rename_map;
365 }

368 void Circuit::add_blank_wires(unsigned n) {
369  TKET_ASSERT(default_regs_ok()); // Incompatible registers exist with the
370  // default names
371 
372  unsigned index = 0;
373  for (unsigned i = 0; i < n; i++) {
374  Vertex in = add_vertex(OpType::Input);
375  Vertex out = add_vertex(OpType::Output);
376  add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
377  bool searching = true;
378  while (searching) {
379  Qubit q_id(index);
380  boundary_t::iterator found = boundary.get<TagID>().find(q_id);
381  if (found == boundary.get<TagID>().end()) {
382  searching = false;
383  boundary.insert({q_id, in, out});
384  }
385  index++;
386  }
387  }

390 void Circuit::add_qubit(const Qubit& id, bool reject_dups) {
391  boundary_t::index<TagID>::type::iterator found =
392  boundary.get<TagID>().find(id);
393  if (found != boundary.get<TagID>().end()) {
394  if (reject_dups) {
395  throw CircuitInvalidity(
396  "A unit with ID \"" + id.repr() + "\" already exists");
397  } else if (found->type() == UnitType::Qubit) {
398  return;
399  }
400  }
401  opt_reg_info_t reg_info = get_reg_info(id.reg_name());
402  register_info_t correct_info = {UnitType::Qubit, id.reg_dim()};
403 
404  if (reg_info && !(reg_info.value() == correct_info)) {
405  throw CircuitInvalidity(
406  "Cannot add qubit with ID \"" + id.repr() +
407  "\" as register is not compatible");
408  }
409 
410  Vertex in = add_vertex(OpType::Input);
411  Vertex out = add_vertex(OpType::Output);
412  add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
413  boundary.insert({id, in, out});
414 }

508 // for wiring in a single vertex with multiple qubits
509 // there are no checks to ensure the vertex exists in the graph
510 void Circuit::rewire(
511  const Vertex& new_vert, const EdgeVec& preds, const op_signature_t& types) {
512  // multi qubit gate
513  EdgeList bin;
514  for (unsigned i = 0; i < preds.size(); ++i) {
515  EdgeType insert_type = types[i];
516  EdgeType replace_type = get_edgetype(preds[i]);
517  port_t port1 = get_source_port(preds[i]);
518  port_t port2 = get_target_port(preds[i]);
519  Vertex old_v1 = source(preds[i]);
520  Vertex old_v2 = target(preds[i]);
521 
522  if (insert_type == EdgeType::Boolean) {
523  // Cannot rewire; Boolean needs a classical value to read from
524  TKET_ASSERT(replace_type == EdgeType::Classical);
525 
526  add_edge({old_v1, port1}, {new_vert, i}, insert_type);
527  } else {
528  // Cannot rewire; type of edge
529  if (insert_type != replace_type) {
530  throw CircuitInvalidity(
531  "Operation can not be added, found invalid parameter type.");
532  }
533 
534  add_edge({old_v1, port1}, {new_vert, i}, insert_type);
535  add_edge({new_vert, i}, {old_v2, port2}, insert_type);
536  bin.push_back(preds[i]);
537  }
538  }
539  for (const Edge& e : bin) remove_edge(e);
540 }
