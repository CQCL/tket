31 // if there are any blank wires in the circuit,
32 // this method removes them and removes the vertices
33 // from boundaries
34 void Circuit::remove_blank_wires() {
35  VertexList bin;
36  unit_vector_t unused_units;
37  const Op_ptr noop = get_op_ptr(OpType::noop);
38  for (const BoundaryElement& el : boundary.get<TagID>()) {
39  Vertex in = el.in_;
40  Vertex out = el.out_;
41  VertexVec succs = get_successors(in);
42  if (succs.size() == 1 && succs.front() == out) {
43  dag[in].op = noop;
44  bin.push_back(in);
45  dag[out].op = noop;
46  bin.push_back(out);
47  unused_units.push_back(el.id_);
48  }
49  }
50  for (const UnitID& u : unused_units) {
51  boundary.get<TagID>().erase(u);
52  }
53  remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
54 }

78 template <>
79 Vertex Circuit::add_op<unsigned>(
80  const Op_ptr& gate, const std::vector<unsigned>& args,
81  std::optional<std::string> opgroup) {
82  op_signature_t sig = gate->get_signature();
83  if (sig.size() != args.size()) {
84  throw CircuitInvalidity(
85  std::to_string(args.size()) + " args provided, but " +
86  gate->get_name() + " requires " + std::to_string(sig.size()));
87  }
88  OpType optype = gate->get_type();
89  unit_vector_t arg_ids;
90  for (unsigned i = 0; i < args.size(); ++i) {
91  if (sig.at(i) == EdgeType::Quantum) {
92  arg_ids.push_back(Qubit(args[i]));
93  } else {
94  arg_ids.push_back(Bit(args[i]));
95  }
96  }
97  if (optype == OpType::CnRy && args.size() == 1) {
98  return add_op(get_op_ptr(OpType::Ry, gate->get_params()), arg_ids);
99  } else if (optype == OpType::CnX && args.size() == 1) {
100  return add_op(get_op_ptr(OpType::X), arg_ids);
101  } else if (optype == OpType::CnZ && args.size() == 1) {
102  return add_op(get_op_ptr(OpType::Z), arg_ids);
103  } else if (optype == OpType::CnY && args.size() == 1) {
104  return add_op(get_op_ptr(OpType::Y), arg_ids);
105  }
106  return add_op(gate, arg_ids, opgroup);
107 }

148 static void append_debug_bits(
149  Circuit& circ, unit_vector_t& args,
150  const std::vector<bool>& expected_readouts,
151  const std::optional<std::string>& postfix) {
152  unsigned n_temp_bits = expected_readouts.size();
153  unsigned n_one_bits =
154  std::accumulate(expected_readouts.begin(), expected_readouts.end(), 0);
155  unsigned n_zero_bits = n_temp_bits - n_one_bits;
156 
157  // Add a classical registers for this assertion
158  std::string zero_reg_name =
159  (postfix == std::nullopt)
160  ? c_debug_zero_prefix() + "_" + c_debug_default_name()
161  : c_debug_zero_prefix() + "_" + postfix.value();
162  std::string one_reg_name =
163  (postfix == std::nullopt)
164  ? c_debug_one_prefix() + "_" + c_debug_default_name()
165  : c_debug_one_prefix() + "_" + postfix.value();
166  std::string incremented_zero_reg_name =
167  circ.get_next_c_reg_name(zero_reg_name);
168  if (n_zero_bits > 0) {
169  circ.add_c_register(incremented_zero_reg_name, n_zero_bits);
170  }
171  std::string incremented_one_reg_name = circ.get_next_c_reg_name(one_reg_name);
172  if (n_one_bits > 0) {
173  circ.add_c_register(incremented_one_reg_name, n_one_bits);
174  }
175 
176  unsigned zero_reg_index = 0;
177  unsigned one_reg_index = 0;
178 
179  for (unsigned i = 0; i < n_temp_bits; i++) {
180  if (expected_readouts[i]) {
181  args.push_back(Bit(incremented_one_reg_name, one_reg_index++));
182  } else {
183  args.push_back(Bit(incremented_zero_reg_name, zero_reg_index++));
184  }
185  }
186 }

188 Vertex Circuit::add_assertion(
189  const ProjectorAssertionBox& assertion_box,
190  const std::vector<Qubit>& qubits, const std::optional<Qubit>& ancilla,
191  const std::optional<std::string>& name) {
192  auto circ_ptr = assertion_box.to_circuit();
193  unsigned log2_dim = log2(assertion_box.get_matrix().rows());
194  if (circ_ptr->n_qubits() > log2_dim && ancilla == std::nullopt) {
195  throw CircuitInvalidity("This assertion requires an ancilla");
196  }
197  if (qubits.size() != log2_dim) {
198  throw CircuitInvalidity(
199  std::to_string(qubits.size()) +
200  " target qubits provided, but the projector requires " +
201  std::to_string(log2_dim));
202  }
203 
204  unit_vector_t args;
205  args.insert(args.end(), qubits.begin(), qubits.end());
206  if (circ_ptr->n_qubits() > log2_dim) {
207  args.push_back(*ancilla);
208  }
209  append_debug_bits(*this, args, assertion_box.get_expected_readouts(), name);
210  return add_op<UnitID>(
211  std::make_shared<ProjectorAssertionBox>(assertion_box), args);
212 }

249 // given vertices and desired in ports for i2 and out ports for i1, adds
250 // edge between them
251 // there are no checks to ensure the vertex exists in the graph
252 Edge Circuit::add_edge(
253  const VertPort& source, const VertPort& target, const EdgeType& type) {
254  // add new edge
255  std::pair<Edge, bool> edge_pairy =
256  boost::add_edge(source.first, target.first, this->dag);
257 
258  if (edge_pairy.second == false) {
259  throw MissingVertex("Cannot create edge between vertices");
260  }
261  Edge new_E = edge_pairy.first;
262  dag[new_E].ports.first = source.second;
263  dag[new_E].ports.second = target.second;
264  dag[new_E].type = type;
265 
266  return new_E;
267 }

269 // given vertex, eradicates it from dag, returning edges used for rewiring
270 // methods
271 // there are no checks to ensure the vertex exists in the graph. This may cause
272 // a segfault if called on a vertex which doesn't exist any more. Take care with
273 // ordering of vertex removal! has flags for whether you want to a) rewire the
274 // hole left by the vertex and b) destroy vertex once it has been isolated As a
275 // sanity check, you cannot remove boundary vertices; converting them to noops
276 // beforehand is recommended
277 void Circuit::remove_vertex(
278  const Vertex& deadvert, GraphRewiring graph_rewiring,
279  VertexDeletion vertex_deletion) {
280  if (graph_rewiring == GraphRewiring::Yes) {
281  EdgeVec ins = get_in_edges(deadvert);
282  std::vector<EdgeVec> bundles = get_b_out_bundles(deadvert);
283  port_t p = 0;
284  for (const Edge& e : ins) {
285  EdgeType type = get_edgetype(e);
286  if (type != EdgeType::Boolean) {
287  Vertex pred_vert = source(e);
288  port_t pred_port = get_source_port(e);
289  Edge next_e = get_nth_out_edge(deadvert, p);
290  Vertex succ_vert = target(next_e);
291  port_t succ_port = get_target_port(next_e);
292  add_edge({pred_vert, pred_port}, {succ_vert, succ_port}, type);
293  if (type == EdgeType::Classical) {
294  for (const Edge& cr : bundles[p]) {
295  Vertex sv = target(cr);
296  port_t sp = get_target_port(cr);
297  add_edge({pred_vert, pred_port}, {sv, sp}, EdgeType::Boolean);
298  }
299  }
300  }
301  ++p;
302  }
303  }
304 
305  boost::clear_vertex(deadvert, this->dag);
306  if (vertex_deletion == VertexDeletion::Yes) {
307  if (detect_boundary_Op(deadvert))
308  throw CircuitInvalidity("Cannot remove a boundary vertex");
309  boost::remove_vertex(deadvert, this->dag);
310  }
311 }

334 unit_map_t Circuit::flatten_registers() {
335  unsigned q_index = 0;
336  unsigned c_index = 0;
337  boundary_t new_map;
338  unit_map_t qmap;
339  for (const BoundaryElement& el : boundary.get<TagID>()) {
340  BoundaryElement new_el = el;
341  if (el.type() == UnitType::Qubit) {
342  new_el.id_ = Qubit(q_default_reg(), q_index);
343  q_index++;
344  } else {
345  new_el.id_ = Bit(c_default_reg(), c_index);
346  c_index++;
347  }
348  qmap.insert({el.id_, new_el.id_});
349  new_map.insert(new_el);
350  }
351  boundary = new_map;
352  return qmap;
353 }

355 // this automatically updates the circuit boundaries
356 void Circuit::add_blank_wires(unsigned n) {
357  if (!default_regs_ok())
358  throw CircuitInvalidity(
359  "Incompatible registers exist with the default names");
360  unsigned index = 0;
361  for (unsigned i = 0; i < n; i++) {
362  Vertex in = add_vertex(OpType::Input);
363  Vertex out = add_vertex(OpType::Output);
364  add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
365  bool searching = true;
366  while (searching) {
367  Qubit q_id(index);
368  boundary_t::iterator found = boundary.get<TagID>().find(q_id);
369  if (found == boundary.get<TagID>().end()) {
370  searching = false;
371  boundary.insert({q_id, in, out});
372  }
373  index++;
374  }
375  }
376 }

378 void Circuit::add_qubit(const Qubit& id, bool reject_dups) {
379  boundary_t::index<TagID>::type::iterator found =
380  boundary.get<TagID>().find(id);
381  if (found != boundary.get<TagID>().end()) {
382  if (reject_dups) {
383  throw CircuitInvalidity(
384  "A unit with ID \"" + id.repr() + "\" already exists");
385  } else if (found->type() == UnitType::Qubit) {
386  return;
387  } else {
388  throw CircuitInvalidity(
389  "A bit with ID \"" + id.repr() + "\" already exists");
390  }
391  }
392  opt_reg_info_t reg_info = get_reg_info(id.reg_name());
393  register_info_t correct_info = {UnitType::Qubit, id.reg_dim()};
394  if (reg_info && !(reg_info.value() == correct_info))
395  throw CircuitInvalidity(
396  "Cannot add qubit with ID \"" + id.repr() +
397  "\" as register is not compatible");
398  Vertex in = add_vertex(OpType::Input);
399  Vertex out = add_vertex(OpType::Output);
400  add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
401  boundary.insert({id, in, out});
402 }

484 // for wiring in a single vertex with multiple qubits
485 // there are no checks to ensure the vertex exists in the graph
486 void Circuit::rewire(
487  const Vertex& new_vert, const EdgeVec& preds, const op_signature_t& types) {
488  // multi qubit gate
489  EdgeList bin;
490  for (unsigned i = 0; i < preds.size(); ++i) {
491  EdgeType insert_type = types[i];
492  EdgeType replace_type = get_edgetype(preds[i]);
493  port_t port1 = get_source_port(preds[i]);
494  port_t port2 = get_target_port(preds[i]);
495  Vertex old_v1 = source(preds[i]);
496  Vertex old_v2 = target(preds[i]);
497  if (insert_type == EdgeType::Boolean) {
498  if (replace_type != EdgeType::Classical) {
499  throw CircuitInvalidity(
500  "Cannot rewire; Boolean needs a classical value "
501  "to read from");
502  }
503  add_edge({old_v1, port1}, {new_vert, i}, insert_type);
504  } else {
505  if (insert_type != replace_type) {
506  throw CircuitInvalidity("Cannot rewire; changing type of edge");
507  }
508  add_edge({old_v1, port1}, {new_vert, i}, insert_type);
509  add_edge({new_vert, i}, {old_v2, port2}, insert_type);
510  bin.push_back(preds[i]);
511  }
512  }
513  for (const Edge& e : bin) remove_edge(e);
514 }
