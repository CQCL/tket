32void Circuit::remove_blank_wires(
33 bool keep_blank_classical_wires,
34 bool remove_classical_only_at_end_of_register) {
35 VertexList bin;
36 unit_vector_t unused_units;
37 const Op_ptr noop = get_op_ptr(OpType::noop);
38 std::set<std::string> bit_names;
39
40 for (const BoundaryElement& el : boundary.get<TagID>()) {
41 if ((!keep_blank_classical_wires &&
42 !remove_classical_only_at_end_of_register) ||
43 el.type() == UnitType::Qubit) {
44 Vertex in = el.in_;
45 Vertex out = el.out_;
46 VertexVec succs = get_successors(in);
47 if (succs.front() == out && succs.size() == 1) {
48 dag[in].op = noop;
49 bin.push_back(in);
50 dag[out].op = noop;
51 bin.push_back(out);
52 unused_units.push_back(el.id_);
53 }
54 } else if (
55 !keep_blank_classical_wires &&
56 remove_classical_only_at_end_of_register &&
57 el.type() == UnitType::Bit && el.id_.reg_dim() == 1) {
58 bit_names.insert(el.id_.reg_name());
59 }
60 }
61
62 if (keep_blank_classical_wires || remove_classical_only_at_end_of_register) {
63 for (auto bit_name : bit_names) {
64 for (unsigned reg_size = get_reg(bit_name).size(); reg_size > 0;
65 --reg_size) {
66 boundary_t::iterator unit_found =
67 boundary.get<TagID>().find(Bit(bit_name, reg_size - 1));
68 if (unit_found != boundary.get<TagID>().end()) {
69 Vertex in = unit_found->in_;
70 Vertex out = unit_found->out_;
71 VertexVec succs = get_successors(in);
72 if (succs.front() == out && succs.size() == 1) {
73 dag[in].op = noop;
74 bin.push_back(in);
75 dag[out].op = noop;
76 bin.push_back(out);
77 unused_units.push_back(unit_found->id_);
78 } else {
79 break;
80 }
81 }
82 }
83 }
84 }
85
86 for (const UnitID& u : unused_units) {
87 boundary.get<TagID>().erase(u);
88 }
89 remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
90}

92void Circuit::remove_noops() {
93 VertexSet bin;
94 BGL_FORALL_VERTICES(v, dag, DAG) {
95 Op_ptr op = get_Op_ptr_from_Vertex(v);
96 std::optional<double> phase;
97 if (op->get_desc().is_gate() && (phase = op->is_identity())) {
98 remove_vertex(
99 v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
100 add_phase(*phase);
101 bin.insert(v);
102 }
103 }
104 remove_vertices(
105 bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
106}

114template <>
115Vertex Circuit::add_op<unsigned>(
116 const Op_ptr& gate, const std::vector<unsigned>& args,
117 std::optional<std::string> opgroup) {
118 op_signature_t sig = gate->get_signature();
119
120 if (sig.size() != args.size()) {
121 throw CircuitInvalidity(
122 std::to_string(args.size()) + " args provided, but " +
123 gate->get_name() + " requires " + std::to_string(sig.size()));
124 }
125
126 unit_vector_t arg_ids;
127 for (unsigned i = 0; i < args.size(); ++i) {
128 switch (sig.at(i)) {
129 case EdgeType::Quantum: {
130 arg_ids.push_back(Qubit(args[i]));
131 break;
132 }
133 case EdgeType::Classical:
134 case EdgeType::Boolean: {
135 arg_ids.push_back(Bit(args[i]));
136 break;
137 }
138 default: {
139 TKET_ASSERT(!"add_op found invalid edge type in signature");
140 }
141 }
142 }
143 return add_op(gate, arg_ids, opgroup);
144}

146Vertex Circuit::add_barrier(
147 const std::vector<unsigned>& qubits, const std::vector<unsigned>& bits,
148 const std::string& _data) {
149 op_signature_t sig(qubits.size(), EdgeType::Quantum);
150 op_signature_t cl_sig(bits.size(), EdgeType::Classical);
151 sig.insert(sig.end(), cl_sig.begin(), cl_sig.end());
152 std::vector<unsigned> args = qubits;
153 args.insert(args.end(), bits.begin(), bits.end());
154 return add_op(std::make_shared<BarrierOp>(sig, _data), args);
155}

172 const std::vector<unsigned>& barrier_bits,
173 const std::vector<unsigned>& condition_bits, unsigned value,
174 const std::string& _data, std::optional<std::string> opgroup) {

184 (unsigned)condition_bits.size(), value),

189 const unit_vector_t& barrier_args, const bit_vector_t& condition_bits,
190 unsigned value, const std::string& _data,

192 op_signature_t sig;

204 return add_op(

206 std::make_shared<BarrierOp>(sig, _data),
207 (unsigned)condition_bits.size(), value),
208 args, opgroup);

211std::string Circuit::get_next_c_reg_name(const std::string& reg_name) {
212 if (!get_reg_info(reg_name)) {
213 return reg_name;
214 }
215 unsigned post_fix = 1;
216 while (true) {
217 std::string incremented_reg_name =
218 reg_name + "(" + std::to_string(post_fix) + ")";
219 if (!get_reg_info(incremented_reg_name)) {
220 return incremented_reg_name;
221 }
222 post_fix++;
223 }

226static void append_debug_bits(
227 Circuit& circ, unit_vector_t& args,
228 const std::vector<bool>& expected_readouts,
229 const std::optional<std::string>& postfix) {
230 unsigned n_temp_bits = expected_readouts.size();
231 unsigned n_one_bits =
232 std::accumulate(expected_readouts.begin(), expected_readouts.end(), 0);
233 unsigned n_zero_bits = n_temp_bits - n_one_bits;
234
235 // Add a classical registers for this assertion
236 std::string zero_reg_name =
237 (postfix == std::nullopt)
238 ? c_debug_zero_prefix() + "_" + c_debug_default_name()
239 : c_debug_zero_prefix() + "_" + postfix.value();
240 std::string one_reg_name =
241 (postfix == std::nullopt)
242 ? c_debug_one_prefix() + "_" + c_debug_default_name()
243 : c_debug_one_prefix() + "_" + postfix.value();
244 std::string incremented_zero_reg_name =
245 circ.get_next_c_reg_name(zero_reg_name);
246 if (n_zero_bits > 0) {
247 circ.add_c_register(incremented_zero_reg_name, n_zero_bits);
248 }
249 std::string incremented_one_reg_name = circ.get_next_c_reg_name(one_reg_name);
250 if (n_one_bits > 0) {
251 circ.add_c_register(incremented_one_reg_name, n_one_bits);
252 }
253
254 unsigned zero_reg_index = 0;
255 unsigned one_reg_index = 0;
256
257 for (unsigned i = 0; i < n_temp_bits; i++) {
258 if (expected_readouts[i]) {
259 args.push_back(Bit(incremented_one_reg_name, one_reg_index++));
260 } else {
261 args.push_back(Bit(incremented_zero_reg_name, zero_reg_index++));
262 }
263 }

266Vertex Circuit::add_assertion(
267 const ProjectorAssertionBox& assertion_box,
268 const std::vector<Qubit>& qubits, const std::optional<Qubit>& ancilla,
269 const std::optional<std::string>& name) {
270 auto circ_ptr = assertion_box.to_circuit();
271 unsigned log2_dim = log2(assertion_box.get_matrix().rows());
272
273 if (circ_ptr->n_qubits() > log2_dim && ancilla == std::nullopt) {
274 throw CircuitInvalidity("This assertion requires an ancilla");
275 }
276
277 if (qubits.size() != log2_dim) {
278 throw CircuitInvalidity(
279 std::to_string(qubits.size()) +
280 " target qubits provided, but the projector requires " +
281 std::to_string(log2_dim));
282 }
283
284 unit_vector_t args;
285 args.insert(args.end(), qubits.begin(), qubits.end());
286 if (circ_ptr->n_qubits() > log2_dim) {
287 args.push_back(*ancilla);
288 }
289 append_debug_bits(*this, args, assertion_box.get_expected_readouts(), name);
290 return add_op<UnitID>(
291 std::make_shared<ProjectorAssertionBox>(assertion_box), args);
292}

294Vertex Circuit::add_assertion(
295 const StabiliserAssertionBox& assertion_box,
296 const std::vector<Qubit>& qubits, const Qubit& ancilla,
297 const std::optional<std::string>& name) {

311 return add_op<UnitID>(
312 std::make_shared<StabiliserAssertionBox>(assertion_box), args);
313}

329// given vertices and desired in ports for i2 and out ports for i1, adds
330// edge between them
331// there are no checks to ensure the vertex exists in the graph
332Edge Circuit::add_edge(
333 const VertPort& source, const VertPort& target, const EdgeType& type) {
334 // add new edge
335 std::pair<Edge, bool> edge_pairy =
336 boost::add_edge(source.first, target.first, this->dag);
337
338 TKET_ASSERT(edge_pairy.second); // Cannot create edge between vertices
339
340 Edge new_E = edge_pairy.first;
341 dag[new_E].ports.first = source.second;
342 dag[new_E].ports.second = target.second;
343 dag[new_E].type = type;
344
345 return new_E;
346}

348// given vertex, eradicates it from dag, returning edges used for rewiring
349// methods
350// there are no checks to ensure the vertex exists in the graph. This may cause
351// a segfault if called on a vertex which doesn't exist any more. Take care with
352// ordering of vertex removal! has flags for whether you want to a) rewire the
353// hole left by the vertex and b) destroy vertex once it has been isolated As a
354// sanity check, you cannot remove boundary vertices; converting them to noops
355// beforehand is recommended
356void Circuit::remove_vertex(
357 const Vertex& deadvert, GraphRewiring graph_rewiring,
358 VertexDeletion vertex_deletion) {
359 if (graph_rewiring == GraphRewiring::Yes) {
360 EdgeVec ins = get_in_edges(deadvert);
361 std::vector<EdgeVec> bundles = get_b_out_bundles(deadvert);
362 port_t p = 0;
363 for (const Edge& e : ins) {
364 EdgeType type = get_edgetype(e);
365 if (type != EdgeType::Boolean) {
366 Vertex pred_vert = source(e);
367 port_t pred_port = get_source_port(e);
368 Edge next_e = get_nth_out_edge(deadvert, p);
369 Vertex succ_vert = target(next_e);
370 port_t succ_port = get_target_port(next_e);
371 add_edge({pred_vert, pred_port}, {succ_vert, succ_port}, type);
372 if (type == EdgeType::Classical) {
373 for (const Edge& cr : bundles[p]) {
374 Vertex sv = target(cr);
375 port_t sp = get_target_port(cr);
376 add_edge({pred_vert, pred_port}, {sv, sp}, EdgeType::Boolean);
377 }
378 }
379 }
380 ++p;
381 }
382 }
383
384 boost::clear_vertex(deadvert, this->dag);
385 if (vertex_deletion == VertexDeletion::Yes) {
386 // Cannot remove a boundary vertex
387 TKET_ASSERT(!detect_boundary_Op(deadvert));
388 boost::remove_vertex(deadvert, this->dag);
389 }
390}

413unit_map_t Circuit::flatten_registers() {
414 unit_map_t rename_map;
415 unsigned q_index = 0;
416 unsigned c_index = 0;
417 for (const BoundaryElement& el : boundary.get<TagID>()) {
418 if (el.type() == UnitType::Qubit) {
419 rename_map.insert({el.id_, Qubit(q_index++)});
420 } else if (el.type() == UnitType::Bit) {
421 rename_map.insert({el.id_, Bit(c_index++)});
422 }
423 }
424 try {
425 rename_units(rename_map);
426 } catch (const std::exception& e) {
427 std::stringstream ss;
428 ss << "Unable to flatten registers: " << e.what();
429 throw std::runtime_error(ss.str());
430 }
431 return rename_map;
432}

434// this automatically updates the circuit boundaries
435void Circuit::add_blank_wires(unsigned n) {
436 TKET_ASSERT(default_regs_ok()); // Incompatible registers exist with the
437 // default names
438
439 unsigned index = 0;
440 for (unsigned i = 0; i < n; i++) {
441 Vertex in = add_vertex(OpType::Input);
442 Vertex out = add_vertex(OpType::Output);
443 add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
444 bool searching = true;
445 while (searching) {
446 Qubit q_id(index);
447 boundary_t::iterator found = boundary.get<TagID>().find(q_id);
448 if (found == boundary.get<TagID>().end()) {
449 searching = false;
450 boundary.insert({q_id, in, out});
451 }
452 index++;
453 }
454 }

457void Circuit::add_qubit(const Qubit& id, bool reject_dups) {
458 boundary_t::index<TagID>::type::iterator found =
459 boundary.get<TagID>().find(id);
460 if (found != boundary.get<TagID>().end()) {
461 if (reject_dups) {
462 throw CircuitInvalidity(
463 "A unit with ID \"" + id.repr() + "\" already exists");
464 } else if (found->type() == UnitType::Qubit) {
465 return;
466 }
467 }
468 opt_reg_info_t reg_info = get_reg_info(id.reg_name());
469 register_info_t correct_info = {UnitType::Qubit, id.reg_dim()};
470
471 if (reg_info && !(reg_info.value() == correct_info)) {
472 throw CircuitInvalidity(
473 "Cannot add qubit with ID \"" + id.repr() +
474 "\" as register is not compatible");
475 }
476
477 Vertex in = add_vertex(OpType::Input);
478 Vertex out = add_vertex(OpType::Output);
479 add_edge({in, 0}, {out, 0}, EdgeType::Quantum);
480 boundary.insert({id, in, out});
481}

575// for wiring in a single vertex with multiple qubits
576// there are no checks to ensure the vertex exists in the graph
577void Circuit::rewire(
578 const Vertex& new_vert, const EdgeVec& preds, const op_signature_t& types) {
579 // multi qubit gate
580 EdgeList bin;
581 for (unsigned i = 0; i < preds.size(); ++i) {
582 EdgeType insert_type = types[i];
583 EdgeType replace_type = get_edgetype(preds[i]);
584 port_t port1 = get_source_port(preds[i]);
585 port_t port2 = get_target_port(preds[i]);
586 Vertex old_v1 = source(preds[i]);
587 Vertex old_v2 = target(preds[i]);
588
589 if (insert_type == EdgeType::Boolean) {
590 // Cannot rewire; Boolean needs a classical value to read from
591 TKET_ASSERT(replace_type == EdgeType::Classical);
592
593 add_edge({old_v1, port1}, {new_vert, i}, insert_type);
594 } else {
595 // Cannot rewire; type of edge
596 if (insert_type != replace_type) {
597 throw CircuitInvalidity(
598 "Operation can not be added, found invalid parameter type.");
599 }
600
601 add_edge({old_v1, port1}, {new_vert, i}, insert_type);
602 add_edge({new_vert, i}, {old_v2, port2}, insert_type);
603 bin.push_back(preds[i]);
604 }
605 }
606 for (const Edge& e : bin) remove_edge(e);
607}
