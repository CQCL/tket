33 void add_latex_for_command(LatexContext& context, const Command& command) {
34  std::map<UnitID, unsigned>& line_ids = context.line_ids;
35  std::vector<LineBufferInfo>& lines = context.lines;
36  unit_vector_t args = command.get_args();
37  const Op_ptr op = command.get_op_ptr();
38  switch (op->get_type()) {
39  case OpType::CnRy:
40  case OpType::CnX:
41  case OpType::CnY:
42  case OpType::CnZ: {
43  int target_index = line_ids.at(args.back());
44  args.pop_back();
45  for (const UnitID& u : args) {
46  int control_index = line_ids.at(u);
47  lines.at(control_index).buffer
48  << "\\ctrl{" << target_index - control_index << "} & ";
49  lines.at(control_index).depth++;
50  }
51  if (op->get_type() == OpType::CnRy) {
52  lines.at(target_index).buffer
53  << "\\gate{\\text{"
54  << get_op_ptr(OpType::Ry, op->get_params())->get_name(true)
55  << "}} & ";
56  } else if (op->get_type() == OpType::CnX) {
57  lines.at(target_index).buffer << "\\targ{} & ";
58  } else if (op->get_type() == OpType::CnY) {
59  lines.at(target_index).buffer << "\\gate{\\text{"
60  << get_op_ptr(OpType::Y)->get_name(true)
61  << "}} & ";
62  } else {
63  lines.at(target_index).buffer << "\\control{} & ";
64  }
65  lines.at(target_index).depth++;
66  break;
67  }
68  case OpType::CCX: {
69  int control0_index = line_ids.at(args.at(0));
70  int control1_index = line_ids.at(args.at(1));
71  int target_index = line_ids.at(args.at(2));
72  lines.at(control0_index).buffer
73  << "\\ctrl{" << target_index - control0_index << "} & ";
74  lines.at(control0_index).depth++;
75  lines.at(control1_index).buffer
76  << "\\ctrl{" << target_index - control1_index << "} & ";
77  lines.at(control1_index).depth++;
78  lines.at(target_index).buffer << "\\targ{} & ";
79  lines.at(target_index).depth++;
80  break;
81  }
82  case OpType::CSWAP: {
83  int control_index = line_ids.at(args.at(0));
84  int target0_index = line_ids.at(args.at(1));
85  int target1_index = line_ids.at(args.at(2));
86  lines.at(control_index).buffer << "\\ctrl{"
87  << target0_index - control_index << "} & ";
88  lines.at(control_index).depth++;
89  lines.at(target0_index).buffer << "\\swap{"
90  << target1_index - target0_index << "} & ";
91  lines.at(target0_index).depth++;
92  lines.at(target1_index).buffer << "\\targX{} & ";
93  lines.at(target1_index).depth++;
94  break;
95  }
96  case OpType::CH:
97  case OpType::CRz:
98  case OpType::CRx:
99  case OpType::CRy:
100  case OpType::CU1:
101  case OpType::CU3:
102  case OpType::CY: {
103  std::stringstream gate_name;
104  switch (op->get_type()) {
105  case OpType::CH: {
106  gate_name << get_op_ptr(OpType::H)->get_name(true);
107  break;
108  }
109  case OpType::CRz: {
110  gate_name << get_op_ptr(OpType::Rz, op->get_params())->get_name(true);
111  break;
112  }
113  case OpType::CRx: {
114  gate_name << get_op_ptr(OpType::Rx, op->get_params())->get_name(true);
115  break;
116  }
117  case OpType::CRy: {
118  gate_name << get_op_ptr(OpType::Ry, op->get_params())->get_name(true);
119  break;
120  }
121  case OpType::CU1: {
122  gate_name << get_op_ptr(OpType::U1, op->get_params())->get_name(true);
123  break;
124  }
125  case OpType::CU3: {
126  gate_name << get_op_ptr(OpType::U3, op->get_params())->get_name(true);
127  break;
128  }
129  case OpType::CY: {
130  gate_name << get_op_ptr(OpType::Y)->get_name(true);
131  break;
132  }
133  case OpType::CV: {
134  gate_name << get_op_ptr(OpType::V)->get_name(true);
135  break;
136  }
137  case OpType::CVdg: {
138  gate_name << get_op_ptr(OpType::Vdg)->get_name(true);
139  break;
140  }
141  case OpType::CSX: {
142  gate_name << get_op_ptr(OpType::SX)->get_name(true);
143  break;
144  }
145  case OpType::CSXdg: {
146  gate_name << get_op_ptr(OpType::SXdg)->get_name(true);
147  break;
148  }
149  default: {
150  }
151  }
152  int control_index = line_ids.at(args.at(0));
153  int target_index = line_ids.at(args.at(1));
154  lines.at(control_index).buffer << "\\ctrl{"
155  << target_index - control_index << "} & ";
156  lines.at(control_index).depth++;
157  lines.at(target_index).buffer << "\\gate{\\text{" << gate_name.str()
158  << "}} & ";
159  lines.at(target_index).depth++;
160  break;
161  }
162  case OpType::CX: {
163  int control_index = line_ids.at(args.at(0));
164  int target_index = line_ids.at(args.at(1));
165  lines.at(control_index).buffer << "\\ctrl{"
166  << target_index - control_index << "} & ";
167  lines.at(control_index).depth++;
168  lines.at(target_index).buffer << "\\targ{} & ";
169  lines.at(target_index).depth++;
170  break;
171  }
172  case OpType::CZ: {
173  int control_index = line_ids.at(args.at(0));
174  int target_index = line_ids.at(args.at(1));
175  lines.at(control_index).buffer << "\\ctrl{"
176  << target_index - control_index << "} & ";
177  lines.at(control_index).depth++;
178  lines.at(target_index).buffer << "\\control{} & ";
179  lines.at(target_index).depth++;
180  break;
181  }
182  case OpType::Measure: {
183  int qb_index = line_ids.at(args.at(0));
184  int cb_index = line_ids.at(args.at(1));
185  lines.at(qb_index).buffer << "\\meter{} \\vcw{" << cb_index - qb_index
186  << "} & ";
187  lines.at(qb_index).depth++;
188  lines.at(cb_index).buffer << "\\cw & ";
189  lines.at(cb_index).depth++;
190  break;
191  }
192  case OpType::Collapse: {
193  int qb_index = line_ids.at(args.at(0));
194  lines.at(qb_index).buffer << "\\meter{} & ";
195  lines.at(qb_index).depth++;
196  break;
197  }
198  case OpType::SWAP: {
199  int qb0_index = line_ids.at(args.at(0));
200  int qb1_index = line_ids.at(args.at(1));
201  lines.at(qb0_index).buffer << "\\swap{" << qb1_index - qb0_index
202  << "} & ";
203  lines.at(qb0_index).depth++;
204  lines.at(qb1_index).buffer << "\\targX{} & ";
205  lines.at(qb1_index).depth++;
206  break;
207  }
208  case OpType::QControlBox: {
209  const QControlBox& box = static_cast<const QControlBox&>(*op);
210  unsigned n_controls = box.get_n_controls();
211  int max_index = 0;
212  unit_vector_t target_args;
213  for (unsigned i = n_controls; i < args.size(); ++i) {
214  target_args.push_back(args[i]);
215  int index = line_ids.at(args[i]);
216  if (index > max_index) max_index = index;
217  }
218  add_latex_for_command(context, Command(box.get_op(), target_args));
219  for (unsigned i = 0; i < n_controls; ++i) {
220  int index = line_ids.at(args[i]);
221  lines.at(index).buffer << "\\ctrl{" << max_index - index << "} & ";
222  lines.at(index).depth++;
223  }
224  break;
225  }
226  case OpType::Conditional: {
227  const Conditional& box = static_cast<const Conditional&>(*op);
228  unsigned n_controls = box.get_width();
229  int max_index = 0;
230  unit_vector_t target_args;
231  for (unsigned i = n_controls; i < args.size(); ++i) {
232  target_args.push_back(args[i]);
233  int index = line_ids.at(args[i]);
234  if (index > max_index) max_index = index;
235  }
236  add_latex_for_command(context, Command(box.get_op(), target_args));
237  for (unsigned i = 0; i < n_controls; ++i) {
238  int index = line_ids.at(args[i]);
239  lines.at(index).buffer << "\\cwbend{" << max_index - index << "} & ";
240  lines.at(index).depth++;
241  }
242  break;
243  }
244  default: {
245  unsigned min_index = std::numeric_limits<unsigned>::max();
246  unsigned max_index = 0;
247  for (const UnitID& arg : args) {
248  unsigned index = line_ids.at(arg);
249  if (index < min_index) min_index = index;
250  if (index > max_index) max_index = index;
251  }
252  lines.at(min_index).buffer << "\\gate[" << (max_index + 1 - min_index)
253  << "]{\\text{" + op->get_name(true) + "}} & ";
254  lines.at(min_index).depth++;
255  for (const UnitID& arg : args) {
256  unsigned index = line_ids.at(arg);
257  if (index != min_index) {
258  lines.at(index).buffer << "& ";
259  lines.at(index).depth++;
260  }
261  }
262  }
263  }

266 std::string Circuit::to_latex_str() const {
267  // Initial header
268  std::stringstream buffer;
269  buffer << "\\documentclass[tikz]{standalone}\n";
270  buffer << "\\usetikzlibrary{quantikz}\n";
271  buffer << "\\begin{document}\n";
272  buffer << "\\begin{quantikz}\n";
273 
274  // Wire labels
275  LatexContext context;
276  std::map<UnitID, unsigned>& line_ids = context.line_ids;
277  std::vector<LineBufferInfo>& lines = context.lines;
278  for (const Qubit& qb : this->all_qubits()) {
279  unsigned n_lines = lines.size();
280  line_ids.insert({qb, n_lines});
281  LineBufferInfo& line = *lines.emplace(lines.end());
282  line.buffer << "\\lstick{" + qb.repr() + "} & ";
283  line.is_quantum = true;
284  }
285  for (const Bit& cb : this->all_bits()) {
286  unsigned n_lines = lines.size();
287  line_ids.insert({cb, n_lines});
288  LineBufferInfo& line = *lines.emplace(lines.end());
289  line.buffer << "\\lstick{" + cb.repr() + "} & ";
290  line.is_quantum = false;
291  }
292 
293  // Commands
294  for (const Command& com : this->get_commands()) {
295  std::set<unsigned> used_lines;
296  unsigned min_index = std::numeric_limits<unsigned>::max();
297  unsigned max_index = 0;
298  for (const UnitID& arg : com.get_args()) {
299  used_lines.insert(line_ids.at(arg));
300  }
301  for (unsigned index : used_lines) {
302  if (index < min_index) min_index = index;
303  if (index > max_index) max_index = index;
304  }
305 
306  unsigned max_depth = 0;
307  for (unsigned index = min_index; index <= max_index; index++) {
308  if (lines.at(index).depth > max_depth) max_depth = lines.at(index).depth;
309  }
310  for (unsigned index : used_lines) {
311  for (unsigned d = lines.at(index).depth; d < max_depth; d++) {
312  if (lines.at(index).is_quantum) {
313  lines.at(index).buffer << "\\qw & ";
314  } else {
315  lines.at(index).buffer << "\\cw & ";
316  }
317  }
318  lines.at(index).depth = max_depth;
319  }
320 
321  add_latex_for_command(context, com);
322 
323  for (unsigned index = min_index; index <= max_index; index++) {
324  for (unsigned d = lines.at(index).depth; d <= max_depth; d++) {
325  if (lines.at(index).is_quantum) {
326  lines.at(index).buffer << "\\qw & ";
327  } else {
328  lines.at(index).buffer << "\\cw & ";
329  }
330  }
331  lines.at(index).depth = max_depth + 1;
332  }
333  }
334 
335  // Fill out ends
336  unsigned max_depth = 0;
337  for (const LineBufferInfo& l : lines) {
338  if (l.depth > max_depth) max_depth = l.depth;
339  }
340  for (LineBufferInfo& l : lines) {
341  for (unsigned d = l.depth; d < max_depth; d++) {
342  if (l.is_quantum) {
343  l.buffer << "\\qw & ";
344  } else {
345  l.buffer << "\\cw & ";
346  }
347  }
348  }
349  for (LineBufferInfo& l : lines) {
350  if (l.is_quantum) {
351  l.buffer << "\\qw \\\\";
352  } else {
353  l.buffer << "\\cw \\\\";
354  }
355  }
356 
357  // Build up overall buffer
358  for (LineBufferInfo& l : lines) {
359  buffer << l.buffer.str() << "\n";
360  }
361  buffer << "\\end{quantikz}\n";
362  buffer << "\\end{document}";
363 
364  return buffer.str();
365 }
