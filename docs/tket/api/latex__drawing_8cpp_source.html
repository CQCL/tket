34void add_latex_for_command(LatexContext& context, const Command& command) {
35 std::map<UnitID, unsigned>& line_ids = context.line_ids;
36 std::vector<LineBufferInfo>& lines = context.lines;
37 unit_vector_t args = command.get_args();
38 const Op_ptr op = command.get_op_ptr();
39 switch (op->get_type()) {
40 case OpType::CnRz:
41 case OpType::CnRx:
42 case OpType::CnRy:
43 case OpType::CnX:
44 case OpType::CnY:
45 case OpType::CnZ: {
46 int target_index = line_ids.at(args.back());
47 args.pop_back();
48 for (const UnitID& u : args) {
49 int control_index = line_ids.at(u);
50 lines.at(control_index).buffer
51 << "\\ctrl{" << target_index - control_index << "} & ";
52 lines.at(control_index).depth++;
53 }
54 if (op->get_type() == OpType::CnRy) {
55 lines.at(target_index).buffer
56 << "\\gate{\\text{"
57 << get_op_ptr(OpType::Ry, op->get_params())->get_name(true)
58 << "}} & ";
59 } else if (op->get_type() == OpType::CnRx) {
60 lines.at(target_index).buffer
61 << "\\gate{\\text{"
62 << get_op_ptr(OpType::Rx, op->get_params())->get_name(true)
63 << "}} & ";
64 } else if (op->get_type() == OpType::CnRz) {
65 lines.at(target_index).buffer
66 << "\\gate{\\text{"
67 << get_op_ptr(OpType::Rz, op->get_params())->get_name(true)
68 << "}} & ";
69 } else if (op->get_type() == OpType::CnX) {
70 lines.at(target_index).buffer << "\\targ{} & ";
71 } else if (op->get_type() == OpType::CnY) {
72 lines.at(target_index).buffer << "\\gate{\\text{"
73 << get_op_ptr(OpType::Y)->get_name(true)
74 << "}} & ";
75 } else {
76 lines.at(target_index).buffer << "\\control{} & ";
77 }
78 lines.at(target_index).depth++;
79 break;
80 }
81 case OpType::CCX: {
82 int control0_index = line_ids.at(args.at(0));
83 int control1_index = line_ids.at(args.at(1));
84 int target_index = line_ids.at(args.at(2));
85 lines.at(control0_index).buffer
86 << "\\ctrl{" << target_index - control0_index << "} & ";
87 lines.at(control0_index).depth++;
88 lines.at(control1_index).buffer
89 << "\\ctrl{" << target_index - control1_index << "} & ";
90 lines.at(control1_index).depth++;
91 lines.at(target_index).buffer << "\\targ{} & ";
92 lines.at(target_index).depth++;
93 break;
94 }
95 case OpType::CSWAP: {
96 int control_index = line_ids.at(args.at(0));
97 int target0_index = line_ids.at(args.at(1));
98 int target1_index = line_ids.at(args.at(2));
99 lines.at(control_index).buffer << "\\ctrl{"
100 << target0_index - control_index << "} & ";
101 lines.at(control_index).depth++;
102 lines.at(target0_index).buffer << "\\swap{"
103 << target1_index - target0_index << "} & ";
104 lines.at(target0_index).depth++;
105 lines.at(target1_index).buffer << "\\targX{} & ";
106 lines.at(target1_index).depth++;
107 break;
108 }
109 case OpType::CH:
110 case OpType::CRz:
111 case OpType::CRx:
112 case OpType::CRy:
113 case OpType::CU1:
114 case OpType::CU3:
115 case OpType::CY: {
116 std::stringstream gate_name;
117 switch (op->get_type()) {
118 case OpType::CH: {
119 gate_name << get_op_ptr(OpType::H)->get_name(true);
120 break;
121 }
122 case OpType::CRz: {
123 gate_name << get_op_ptr(OpType::Rz, op->get_params())->get_name(true);
124 break;
125 }
126 case OpType::CRx: {
127 gate_name << get_op_ptr(OpType::Rx, op->get_params())->get_name(true);
128 break;
129 }
130 case OpType::CRy: {
131 gate_name << get_op_ptr(OpType::Ry, op->get_params())->get_name(true);
132 break;
133 }
134 case OpType::CU1: {
135 gate_name << get_op_ptr(OpType::U1, op->get_params())->get_name(true);
136 break;
137 }
138 case OpType::CU3: {
139 gate_name << get_op_ptr(OpType::U3, op->get_params())->get_name(true);
140 break;
141 }
142 case OpType::CY: {
143 gate_name << get_op_ptr(OpType::Y)->get_name(true);
144 break;
145 }
146 case OpType::CV: {
147 gate_name << get_op_ptr(OpType::V)->get_name(true);
148 break;
149 }
150 case OpType::CVdg: {
151 gate_name << get_op_ptr(OpType::Vdg)->get_name(true);
152 break;
153 }
154 case OpType::CSX: {
155 gate_name << get_op_ptr(OpType::SX)->get_name(true);
156 break;
157 }
158 case OpType::CSXdg: {
159 gate_name << get_op_ptr(OpType::SXdg)->get_name(true);
160 break;
161 }
162 case OpType::CS: {
163 gate_name << get_op_ptr(OpType::S)->get_name(true);
164 break;
165 }
166 case OpType::CSdg: {
167 gate_name << get_op_ptr(OpType::Sdg)->get_name(true);
168 break;
169 }
170 default: {
171 }
172 }
173 int control_index = line_ids.at(args.at(0));
174 int target_index = line_ids.at(args.at(1));
175 lines.at(control_index).buffer << "\\ctrl{"
176 << target_index - control_index << "} & ";
177 lines.at(control_index).depth++;
178 lines.at(target_index).buffer << "\\gate{\\text{" << gate_name.str()
179 << "}} & ";
180 lines.at(target_index).depth++;
181 break;
182 }
183 case OpType::CX: {
184 int control_index = line_ids.at(args.at(0));
185 int target_index = line_ids.at(args.at(1));
186 lines.at(control_index).buffer << "\\ctrl{"
187 << target_index - control_index << "} & ";
188 lines.at(control_index).depth++;
189 lines.at(target_index).buffer << "\\targ{} & ";
190 lines.at(target_index).depth++;
191 break;
192 }
193 case OpType::CZ: {
194 int control_index = line_ids.at(args.at(0));
195 int target_index = line_ids.at(args.at(1));
196 lines.at(control_index).buffer << "\\ctrl{"
197 << target_index - control_index << "} & ";
198 lines.at(control_index).depth++;
199 lines.at(target_index).buffer << "\\control{} & ";
200 lines.at(target_index).depth++;
201 break;
202 }
203 case OpType::Measure: {
204 int qb_index = line_ids.at(args.at(0));
205 int cb_index = line_ids.at(args.at(1));
206 lines.at(qb_index).buffer << "\\meter{} \\vcw{" << cb_index - qb_index
207 << "} & ";
208 lines.at(qb_index).depth++;
209 lines.at(cb_index).buffer << "\\cw & ";
210 lines.at(cb_index).depth++;
211 break;
212 }
213 case OpType::Collapse: {
214 int qb_index = line_ids.at(args.at(0));
215 lines.at(qb_index).buffer << "\\meter{} & ";
216 lines.at(qb_index).depth++;
217 break;
218 }
219 case OpType::SWAP: {
220 int qb0_index = line_ids.at(args.at(0));
221 int qb1_index = line_ids.at(args.at(1));
222 lines.at(qb0_index).buffer << "\\swap{" << qb1_index - qb0_index
223 << "} & ";
224 lines.at(qb0_index).depth++;
225 lines.at(qb1_index).buffer << "\\targX{} & ";
226 lines.at(qb1_index).depth++;
227 break;
228 }
229 case OpType::QControlBox: {
230 const QControlBox& box = static_cast<const QControlBox&>(*op);
231 unsigned n_controls = box.get_n_controls();
232 int max_index = 0;
233 unit_vector_t target_args;
234 for (unsigned i = n_controls; i < args.size(); ++i) {
235 target_args.push_back(args[i]);
236 int index = line_ids.at(args[i]);
237 if (index > max_index) max_index = index;
238 }
239 add_latex_for_command(context, Command(box.get_op(), target_args));
240 for (unsigned i = 0; i < n_controls; ++i) {
241 int index = line_ids.at(args[i]);
242 lines.at(index).buffer << "\\ctrl{" << max_index - index << "} & ";
243 lines.at(index).depth++;
244 }
245 break;
246 }
247 case OpType::Conditional: {
248 const Conditional& box = static_cast<const Conditional&>(*op);
249 unsigned n_controls = box.get_width();
250 int max_index = 0;
251 unit_vector_t target_args;
252 for (unsigned i = n_controls; i < args.size(); ++i) {
253 target_args.push_back(args[i]);
254 int index = line_ids.at(args[i]);
255 if (index > max_index) max_index = index;
256 }
257 add_latex_for_command(context, Command(box.get_op(), target_args));
258 for (unsigned i = 0; i < n_controls; ++i) {
259 int index = line_ids.at(args[i]);
260 lines.at(index).buffer << "\\cwbend{" << max_index - index << "} & ";
261 lines.at(index).depth++;
262 }
263 break;
264 }
265 default: {
266 unsigned min_index = std::numeric_limits<unsigned>::max();
267 unsigned max_index = 0;
268 for (const UnitID& arg : args) {
269 unsigned index = line_ids.at(arg);
270 if (index < min_index) min_index = index;
271 if (index > max_index) max_index = index;
272 }
273 lines.at(min_index).buffer << "\\gate[" << (max_index + 1 - min_index)
274 << "]{\\text{" + op->get_name(true) + "}} & ";
275 lines.at(min_index).depth++;
276 for (const UnitID& arg : args) {
277 unsigned index = line_ids.at(arg);
278 if (index != min_index) {
279 lines.at(index).buffer << "& ";
280 lines.at(index).depth++;
281 }
282 }
283 }
284 }

287std::string Circuit::to_latex_str() const {
288 // Initial header
289 std::stringstream buffer;
290 buffer << "\\documentclass[tikz]{standalone}\n";
291 buffer << "\\usetikzlibrary{quantikz}\n";
292 buffer << "\\begin{document}\n";
293 buffer << "\\begin{quantikz}\n";
294
295 // Wire labels
296 LatexContext context;
297 std::map<UnitID, unsigned>& line_ids = context.line_ids;
298 std::vector<LineBufferInfo>& lines = context.lines;
299 for (const Qubit& qb : this->all_qubits()) {
300 unsigned n_lines = lines.size();
301 line_ids.insert({qb, n_lines});
302 LineBufferInfo& line = *lines.emplace(lines.end());
303 line.buffer << "\\lstick{" +
304 boost::replace_all_copy(qb.repr(), "_", "\\_") + "} & ";
305 line.is_quantum = true;
306 }
307 for (const Bit& cb : this->all_bits()) {
308 unsigned n_lines = lines.size();
309 line_ids.insert({cb, n_lines});
310 LineBufferInfo& line = *lines.emplace(lines.end());
311 line.buffer << "\\lstick{" +
312 boost::replace_all_copy(cb.repr(), "_", "\\_") + "} & ";
313 line.is_quantum = false;
314 }
315
316 // Commands
317 for (const Command& com : this->get_commands()) {
318 std::set<unsigned> used_lines;
319 unsigned min_index = std::numeric_limits<unsigned>::max();
320 unsigned max_index = 0;
321 for (const UnitID& arg : com.get_args()) {
322 used_lines.insert(line_ids.at(arg));
323 }
324 for (unsigned index : used_lines) {
325 if (index < min_index) min_index = index;
326 if (index > max_index) max_index = index;
327 }
328
329 unsigned max_depth = 0;
330 for (unsigned index = min_index; index <= max_index; index++) {
331 if (lines.at(index).depth > max_depth) max_depth = lines.at(index).depth;
332 }
333 for (unsigned index : used_lines) {
334 for (unsigned d = lines.at(index).depth; d < max_depth; d++) {
335 if (lines.at(index).is_quantum) {
336 lines.at(index).buffer << "\\qw & ";
337 } else {
338 lines.at(index).buffer << "\\cw & ";
339 }
340 }
341 lines.at(index).depth = max_depth;
342 }
343
344 add_latex_for_command(context, com);
345
346 for (unsigned index = min_index; index <= max_index; index++) {
347 for (unsigned d = lines.at(index).depth; d <= max_depth; d++) {
348 if (lines.at(index).is_quantum) {
349 lines.at(index).buffer << "\\qw & ";
350 } else {
351 lines.at(index).buffer << "\\cw & ";
352 }
353 }
354 lines.at(index).depth = max_depth + 1;
355 }
356 }
357
358 // Fill out ends
359 unsigned max_depth = 0;
360 for (const LineBufferInfo& l : lines) {
361 if (l.depth > max_depth) max_depth = l.depth;
362 }
363 for (LineBufferInfo& l : lines) {
364 for (unsigned d = l.depth; d < max_depth; d++) {
365 if (l.is_quantum) {
366 l.buffer << "\\qw & ";
367 } else {
368 l.buffer << "\\cw & ";
369 }
370 }
371 }
372 for (LineBufferInfo& l : lines) {
373 if (l.is_quantum) {
374 l.buffer << "\\qw \\\\";
375 } else {
376 l.buffer << "\\cw \\\\";
377 }
378 }
379
380 // Build up overall buffer
381 for (LineBufferInfo& l : lines) {
382 buffer << l.buffer.str() << "\n";
383 }
384 buffer << "\\end{quantikz}\n";
385 buffer << "\\end{document}";
386
387 return buffer.str();
388}
