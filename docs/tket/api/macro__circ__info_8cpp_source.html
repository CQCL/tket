30bool Circuit::is_simple() const {
31 if (!default_regs_ok()) return false;
32 for (const BoundaryElement& el : boundary.get<TagID>()) {
33 std::string reg = el.id_.reg_name();
34 if (!(reg == q_default_reg() || reg == c_default_reg() ||
35 reg == w_default_reg()))
36 return false;
37 }
38 return true;
39}

41bool Circuit::default_regs_ok() const {
42 opt_reg_info_t q_info = get_reg_info(q_default_reg());
43 register_info_t correct_q_info = {UnitType::Qubit, 1};
44 if (q_info && q_info.value() != correct_q_info) return false;
45 opt_reg_info_t c_info = get_reg_info(c_default_reg());
46 register_info_t correct_c_info = {UnitType::Bit, 1};
47 if (c_info && c_info.value() != correct_c_info) return false;
48 opt_reg_info_t w_info = get_reg_info(w_default_reg());
49 register_info_t correct_w_info = {UnitType::WasmState, 1};
50 if (w_info && w_info.value() != correct_w_info) return false;
51 return true;
52}

60unsigned Circuit::count_gates(
61 const OpType& op_type, const bool include_conditional) const {
62 unsigned counter = 0;
63 BGL_FORALL_VERTICES(v, dag, DAG) {
64 Op_ptr op = get_Op_ptr_from_Vertex(v);
65 while (include_conditional && op->get_type() == OpType::Conditional) {
66 op = static_cast<const Conditional&>(*op).get_op();
67 }
68 if (op->get_type() == op_type) ++counter;
69 }
70 return counter;
71}

92unsigned Circuit::count_n_qubit_gates(unsigned size) const {
93 unsigned counter = 0;
94 if (size == 0) return counter;
95 BGL_FORALL_VERTICES(v, dag, DAG) {
96 if (n_in_edges_of_type(v, EdgeType::Quantum) == size) {
97 const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
98 switch (op_ptr->get_type()) {
99 case OpType::Input:
100 case OpType::Create:
101 case OpType::Output:
102 case OpType::Discard:
103 case OpType::Reset:
104 case OpType::Measure:
105 case OpType::Barrier: {
106 break;
107 }
108 default: {
109 counter++;
110 }
111 }
112 }
113 }
114 return counter;
115}

117Circuit Circuit::subcircuit(const Subcircuit& sc) const {
118 Circuit sub;
119 // A map from vertices in *this to vertices in sub. There may be some vertices
120 // in *this corresponding to multiple distinct boundary vertices in sub, in
121 // which case an arbitrary one is given here and we use in_boundary_map or
122 // out_boundary_map to distinguish
123 vertex_map_t vmap;
124 VertexVec ins;
125 VertexVec outs;
126 std::map<Edge, Vertex> in_boundary_map;
127 std::map<Edge, Vertex> out_boundary_map;
128 unsigned n_qbs = 0;
129 unsigned n_bits = 0;
130 unsigned n_wasm = 0;
131 for (unsigned i = 0; i < sc.in_hole.size(); ++i) {
132 Edge in_edge = sc.in_hole.at(i);
133 std::optional<Edge> out_edge = sc.out_hole.at(i);
134 switch (get_edgetype(in_edge)) {
135 case EdgeType::Quantum: {
136 TKET_ASSERT(out_edge.has_value());
137 Vertex in = sub.add_vertex(OpType::Input);
138 Vertex out = sub.add_vertex(OpType::Output);
139 sub.boundary.insert({Qubit(n_qbs), in, out});
140 ++n_qbs;
141 vmap[source(in_edge)] = in;
142 in_boundary_map.insert({in_edge, in});
143 vmap[target(*out_edge)] = out;
144 out_boundary_map.insert({*out_edge, out});
145 break;
146 }
147 case EdgeType::Classical: {
148 TKET_ASSERT(out_edge.has_value());
149 Vertex in = sub.add_vertex(OpType::ClInput);
150 Vertex out = sub.add_vertex(OpType::ClOutput);
151 sub.boundary.insert({Bit(n_bits), in, out});
152 ++n_bits;
153 vmap[source(in_edge)] = in;
154 in_boundary_map.insert({in_edge, in});
155 vmap[target(*out_edge)] = out;
156 out_boundary_map.insert({*out_edge, out});
157 break;
158 }
159 case EdgeType::Boolean: {
160 TKET_ASSERT(!out_edge.has_value());
161 Vertex in = sub.add_vertex(OpType::ClInput);
162 Vertex out = sub.add_vertex(OpType::ClOutput);
163 sub.boundary.insert({Bit(n_bits), in, out});
164 ++n_bits;
165 sub.add_edge({in, 0}, {out, 0}, EdgeType::Classical);
166 vmap[source(in_edge)] = in;
167 in_boundary_map.insert({in_edge, in});
168 break;
169 }
170 case EdgeType::WASM: {
171 TKET_ASSERT(out_edge.has_value());
172 Vertex in = sub.add_vertex(OpType::WASMInput);
173 Vertex out = sub.add_vertex(OpType::WASMOutput);
174 sub.boundary.insert({WasmState(n_wasm), in, out});
175 sub.wasmwire.push_back(WasmState(n_wasm));
176 ++sub._number_of_wasm_wires;
177 ++n_wasm;
178 vmap[source(in_edge)] = in;
179 in_boundary_map.insert({in_edge, in});
180 vmap[target(*out_edge)] = out;
181 out_boundary_map.insert({*out_edge, out});
182 break;
183 }
184 default: {
185 TKET_ASSERT(false);
186 }
187 }
188 }
189 for (const Vertex& v : sc.verts) {
190 Vertex added = sub.add_vertex(get_Op_ptr_from_Vertex(v));
191 vmap[v] = added;
192 }
193 // Add edges into internal vertices
194 for (const Vertex& v : sc.verts) {
195 // iterate through original circuit as order of the set varies between Mac
196 // and Docker
197 BGL_FORALL_INEDGES(v, e, dag, DAG) {
198 Vertex source = this->source(e);
199 Vertex sub_source = vmap.at(source);
200 port_t in_port = get_source_port(e);
201 OpType type = sub.get_OpType_from_Vertex(sub_source);
202 if (is_initial_q_type(type) || type == OpType::ClInput ||
203 type == OpType::WASMInput) {
204 // Multiple inputs might be mapped to the same source
205 // so need to distinguish them.
206 sub_source = in_boundary_map.at(e);
207 in_port = 0;
208 }
209 sub.add_edge(
210 {sub_source, in_port}, {vmap.at(v), get_target_port(e)},
211 get_edgetype(e));
212 }
213 }
214 // Add edges into outputs
215 for (const std::optional<Edge>& e : sc.out_hole) {
216 if (e.has_value()) {
217 Vertex out = out_boundary_map.at(*e);
218 Vertex sub_source = vmap.at(source(*e));
219 port_t in_port = get_source_port(*e);
220 std::map<Edge, Vertex>::iterator found = in_boundary_map.find(*e);
221 if (found != in_boundary_map.end()) {
222 sub_source = found->second;
223 in_port = 0;
224 }
225 sub.add_edge({sub_source, in_port}, {out, 0}, get_edgetype(*e));
226 }
227 }
228 return sub;
229}

239// returns qubit path via vertices & inhabited port in vertices
240// used to construct a routing grid
241QPathDetailed Circuit::unit_path(const UnitID& unit) const {
242 Vertex current_v = get_in(unit);
243
244 QPathDetailed path = {{current_v, 0}};
245 Edge betweenEdge = get_nth_out_edge(current_v, 0);
246 current_v = target(betweenEdge);
247
248 while (detect_final_Op(current_v) == false) {
249 if (n_out_edges(current_v) == 0) {
250 throw CircuitInvalidity("A path ends before reaching an output vertex.");
251 }
252 port_t n = get_target_port(betweenEdge);
253 VertPort v_and_port = {current_v, n};
254 path.push_back(v_and_port);
255 betweenEdge = get_nth_out_edge(current_v, n);
256 current_v = target(betweenEdge);
257 }
258 path.push_back({current_v, 0});
259 return path;
260}

272std::map<UnitID, QPathDetailed> Circuit::all_unit_paths() const {
273 std::map<UnitID, QPathDetailed> new_list_of_paths;
274 for (const Qubit& q : all_qubits()) {
275 new_list_of_paths.insert({q, unit_path(q)});
276 }
277 for (const Bit& b : all_bits()) {
278 new_list_of_paths.insert({b, unit_path(b)});
279 }
280 for (const WasmState& w : wasmwire) {
281 new_list_of_paths.insert({w, unit_path(w)});
282 }
283 return new_list_of_paths;

296/*
297 Permute output boundary of circuit according to qubit map
298 Assumes all circuit Qubits are mapped
299*/
300void Circuit::permute_boundary_output(const qubit_map_t& qm) {
301 std::map<UnitID, BoundaryElement> new_entries;
302
303 for (const auto& qb_pair : qm) {
304 boundary_t::iterator input = boundary_elem(*this, qb_pair.first);
305 boundary_t::iterator output = boundary_elem(*this, qb_pair.second);
306 new_entries.insert({output->id_, {output->id_, output->in_, input->out_}});
307 }
308
309 for (const auto& pair : new_entries) {
310 boundary.erase(boundary.get<TagID>().find(pair.first));
311 }
312
313 for (const auto& pair : new_entries) {
314 boundary.insert(pair.second);
315 }
316}

318qubit_map_t Circuit::implicit_qubit_permutation() const {
319 qubit_map_t perm;
320 for (const QPathDetailed& path : all_qubit_paths()) {
321 Qubit in_q(get_id_from_in(path.front().first));
322 Qubit out_q(get_id_from_out(path.back().first));
323 perm.insert({in_q, out_q});
324 }
325 return perm;
326}

328bool Circuit::has_implicit_wireswaps() const {
329 qubit_map_t perm = implicit_qubit_permutation();
330 for (const std::pair<const Qubit, Qubit>& pair : perm) {
331 if (pair.first != pair.second) return true;
332 }
333 return false;

360Edge Circuit::skip_irrelevant_edges(Edge current) const {
361 Vertex try_next_v = target(current);
362 while (n_out_edges_of_type(try_next_v, EdgeType::Quantum) == 1) {
363 std::tie(try_next_v, current) = get_next_pair(try_next_v, current);
364 }
365 return current;
366}

368SliceVec Circuit::get_reverse_slices() const {
369 vertex_map_t mapping;
370 vertex_map_t rev_mapping;
371 Circuit rev;
372 for (const BoundaryElement& el : boundary.get<TagID>()) {
373 Vertex new_in, new_out;
374 if (el.type() == UnitType::Qubit) {
375 new_in = rev.add_vertex(OpType::Input);
376 new_out = rev.add_vertex(OpType::Output);
377 } else {
378 new_in = rev.add_vertex(OpType::ClInput);
379 new_out = rev.add_vertex(OpType::ClOutput);
380 }
381 mapping[el.in_] = new_out;
382 rev_mapping[new_out] = el.in_;
383 mapping[el.out_] = new_in;
384 rev_mapping[new_in] = el.out_;
385 rev.boundary.insert({el.id_, new_in, new_out});
386 }
387 BGL_FORALL_VERTICES(v, dag, DAG) {
388 const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
389 switch (op_ptr->get_type()) {
390 case OpType::Input:
391 case OpType::Create:
392 case OpType::Output:
393 case OpType::ClInput:
394 case OpType::ClOutput:
395 case OpType::Discard:
396 case OpType::WASMInput:
397 case OpType::WASMOutput: {
398 break;
399 }
400 default: {
401 Vertex v0 = rev.add_vertex(op_ptr);
402 mapping[v] = v0;
403 rev_mapping[v0] = v;
404 break;
405 }
406 }
407 }
408 BGL_FORALL_EDGES(e, dag, DAG) {
409 Vertex s = source(e);
410 port_t sp = get_source_port(e);
411 Vertex t = target(e);
412 port_t tp = get_target_port(e);
413 EdgeType type = dag[e].type;
414 if (type == EdgeType::Boolean) {
415 // Move Boolean to read from bit wire just before next write
416 Edge bit_wire = get_nth_out_edge(s, sp);
417 Vertex next_on_bit = target(bit_wire);
418 port_t next_p = get_target_port(bit_wire);
419 rev.add_edge({mapping[next_on_bit], next_p}, {mapping[t], tp}, type);
420 } else {
421 rev.add_edge({mapping[t], tp}, {mapping[s], sp}, type);
422 }
423 }
424 SliceVec slices_of_rev = rev.get_slices();
425 SliceVec rev_slices;
426 for (const Slice& s : slices_of_rev) {
427 Slice sl;
428 for (const Vertex& v : s) {
429 sl.push_back(rev_mapping[v]);
430 }
431 rev_slices.push_back(sl);
432 }
433 return rev_slices;
434}

436unsigned Circuit::depth() const {
437 unsigned count = 0;
438 std::function<bool(Op_ptr)> skip_func = [&](Op_ptr op) {
439 return (op->get_type() == OpType::Barrier);
440 };
441 SliceIterator slice_iter(*this, skip_func);
442 if (!(*slice_iter).empty()) count++;
443 while (!slice_iter.finished()) {
444 slice_iter.cut_ = this->next_cut(
445 slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
446 if (!(*slice_iter).empty()) count++;
447 }
448 return count;
449}

508 std::map<Vertex, unsigned> map;

538std::map<Edge, UnitID> Circuit::edge_unit_map() const {
539 std::map<Edge, UnitID> map;
540 for (const std::pair<const UnitID, QPathDetailed>& path : all_unit_paths()) {
541 QPathDetailed::const_iterator it = path.second.begin();
542 for (++it; it != path.second.end(); ++it) {
543 map.insert({get_nth_in_edge(it->first, it->second), path.first});
544 }
545 }
546 // Boolean edges are non-linear so are not added by the traces, but we can
547 // still link them to bits
548 BGL_FORALL_EDGES(e, dag, DAG) {
549 if (get_edgetype(e) == EdgeType::Boolean) {
550 Edge linear_edge = get_linear_edge(e);
551 map.insert({e, map.at(linear_edge)});
552 }
553 }
554 return map;
555}

586Circuit::CommandIterator& Circuit::CommandIterator::operator++() {
587 if (*this == circ_->end()) return *this;
588 if (current_index_ == (*current_slice_iterator_).size() - 1) {
589 if (current_slice_iterator_.finished()) {
590 *this = circ_->end();
591 return *this;
592 }
593 ++current_slice_iterator_;
594 current_index_ = 0;
595 } else {
596 ++current_index_;
597 }
598 if (current_index_ == (*current_slice_iterator_).size()) {
599 TKET_ASSERT(!"slice is empty");
600 }
601 current_vertex_ = (*current_slice_iterator_)[current_index_];
602 current_command_ = circ_->command_from_vertex(
603 current_vertex_, current_slice_iterator_.get_u_frontier(),
604 current_slice_iterator_.get_prev_b_frontier());
605 return *this;
606}

608unit_vector_t Circuit::args_from_frontier(
609 const Vertex& vert, std::shared_ptr<const unit_frontier_t> u_frontier,
610 std::shared_ptr<const b_frontier_t> prev_b_frontier) const {
611 EdgeVec ins = get_in_edges(vert);
612 unit_vector_t args;
613 for (port_t p = 0; p < ins.size(); ++p) {
614 switch (get_edgetype(ins[p])) {
615 case EdgeType::WASM: {
616 Edge out = get_next_edge(vert, ins[p]);
617 bool found = false;
618 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
619 if (pair.second == out) {
620 args.push_back(pair.first);
621 found = true;
622 break;
623 }
624 }
625 TKET_ASSERT(found); // Vertex edges not found in frontier.
626 break;
627 }
628 case EdgeType::Boolean: {
629 bool found = false;
630 for (const std::pair<Bit, EdgeVec>& pair :
631 prev_b_frontier->get<TagKey>()) {
632 for (const Edge& edge : pair.second) {
633 if (edge == ins[p]) {
634 args.push_back(pair.first);
635 found = true;
636 break;
637 }
638 }
639 if (found) {
640 break;
641 }
642 }
643 TKET_ASSERT(found); // Vertex edges not found in Boolean frontier.
644 break;
645 }
646 case EdgeType::Classical:
647 case EdgeType::Quantum: {
648 Edge out = get_next_edge(vert, ins[p]);
649 bool found = false;
650 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
651 if (pair.second == out) {
652 args.push_back(pair.first);
653 found = true;
654 break;
655 }
656 }
657 if (!found)
658 throw CircuitInvalidity(
659 "Vertex edges not found in frontier. Edge: " +
660 get_Op_ptr_from_Vertex(source(out))->get_name() + " -> " +

663 }

666 }
667 }
668 }
669 return args;
670}
