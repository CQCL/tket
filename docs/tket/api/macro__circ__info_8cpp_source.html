30bool Circuit::is_simple() const {
31 if (!default_regs_ok()) return false;
32 for (const BoundaryElement& el : boundary.get<TagID>()) {
33 std::string reg = el.id_.reg_name();
34 if (!(reg == q_default_reg() || reg == c_default_reg() ||
35 reg == w_default_reg() || reg == r_default_reg()))
36 return false;
37 }
38 return true;
39}

41bool Circuit::default_regs_ok() const {
42 opt_reg_info_t q_info = get_reg_info(q_default_reg());
43 register_info_t correct_q_info = {UnitType::Qubit, 1};
44 if (q_info && q_info.value() != correct_q_info) return false;
45 opt_reg_info_t c_info = get_reg_info(c_default_reg());
46 register_info_t correct_c_info = {UnitType::Bit, 1};
47 if (c_info && c_info.value() != correct_c_info) return false;
48 opt_reg_info_t w_info = get_reg_info(w_default_reg());
49 register_info_t correct_w_info = {UnitType::WasmState, 1};
50 if (w_info && w_info.value() != correct_w_info) return false;
51 opt_reg_info_t r_info = get_reg_info(r_default_reg());
52 register_info_t correct_r_info = {UnitType::RngState, 1};
53 if (r_info && r_info.value() != correct_r_info) return false;
54 return true;
55}

63unsigned Circuit::count_gates(
64 const OpType& op_type, const bool include_conditional) const {
65 unsigned counter = 0;
66 BGL_FORALL_VERTICES(v, dag, DAG) {
67 Op_ptr op = get_Op_ptr_from_Vertex(v);
68 while (include_conditional && op->get_type() == OpType::Conditional) {
69 op = static_cast<const Conditional&>(*op).get_op();
70 }
71 if (op->get_type() == op_type) ++counter;
72 }
73 return counter;
74}

95unsigned Circuit::count_n_qubit_gates(unsigned size) const {
96 unsigned counter = 0;
97 if (size == 0) return counter;
98 BGL_FORALL_VERTICES(v, dag, DAG) {
99 if (n_in_edges_of_type(v, EdgeType::Quantum) == size) {
100 const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
101 switch (op_ptr->get_type()) {
102 case OpType::Input:
103 case OpType::Create:
104 case OpType::Output:
105 case OpType::Discard:
106 case OpType::Reset:
107 case OpType::Measure:
108 case OpType::Barrier: {
109 break;
110 }
111 default: {
112 counter++;
113 }
114 }
115 }
116 }
117 return counter;
118}

120Circuit Circuit::subcircuit(const Subcircuit& sc) const {
121 Circuit sub;
122 // A map from vertices in *this to vertices in sub. There may be some vertices
123 // in *this corresponding to multiple distinct boundary vertices in sub, in
124 // which case an arbitrary one is given here and we use in_boundary_map or
125 // out_boundary_map to distinguish
126 vertex_map_t vmap;
127 VertexVec ins;
128 VertexVec outs;
129 std::map<Edge, Vertex> in_boundary_map;
130 std::map<Edge, Vertex> out_boundary_map;
131 unsigned n_qbs = 0;
132 unsigned n_bits = 0;
133 unsigned n_wasm = 0;
134 unsigned n_rng = 0;
135 for (unsigned i = 0; i < sc.in_hole.size(); ++i) {
136 Edge in_edge = sc.in_hole.at(i);
137 std::optional<Edge> out_edge = sc.out_hole.at(i);
138 switch (get_edgetype(in_edge)) {
139 case EdgeType::Quantum: {
140 TKET_ASSERT(out_edge.has_value());
141 Vertex in = sub.add_vertex(OpType::Input);
142 Vertex out = sub.add_vertex(OpType::Output);
143 sub.boundary.insert({Qubit(n_qbs), in, out});
144 ++n_qbs;
145 vmap[source(in_edge)] = in;
146 in_boundary_map.insert({in_edge, in});
147 vmap[target(*out_edge)] = out;
148 out_boundary_map.insert({*out_edge, out});
149 break;
150 }
151 case EdgeType::Classical: {
152 TKET_ASSERT(out_edge.has_value());
153 Vertex in = sub.add_vertex(OpType::ClInput);
154 Vertex out = sub.add_vertex(OpType::ClOutput);
155 sub.boundary.insert({Bit(n_bits), in, out});
156 ++n_bits;
157 vmap[source(in_edge)] = in;
158 in_boundary_map.insert({in_edge, in});
159 vmap[target(*out_edge)] = out;
160 out_boundary_map.insert({*out_edge, out});
161 break;
162 }
163 case EdgeType::Boolean: {
164 TKET_ASSERT(!out_edge.has_value());
165 Vertex in = sub.add_vertex(OpType::ClInput);
166 Vertex out = sub.add_vertex(OpType::ClOutput);
167 sub.boundary.insert({Bit(n_bits), in, out});
168 ++n_bits;
169 sub.add_edge({in, 0}, {out, 0}, EdgeType::Classical);
170 vmap[source(in_edge)] = in;
171 in_boundary_map.insert({in_edge, in});
172 break;
173 }
174 case EdgeType::WASM: {
175 TKET_ASSERT(out_edge.has_value());
176 Vertex in = sub.add_vertex(OpType::WASMInput);
177 Vertex out = sub.add_vertex(OpType::WASMOutput);
178 sub.boundary.insert({WasmState(n_wasm), in, out});
179 sub.wasmwire.push_back(WasmState(n_wasm));
180 ++sub._number_of_wasm_wires;
181 ++n_wasm;
182 vmap[source(in_edge)] = in;
183 in_boundary_map.insert({in_edge, in});
184 vmap[target(*out_edge)] = out;
185 out_boundary_map.insert({*out_edge, out});
186 break;
187 }
188 case EdgeType::RNG: {
189 TKET_ASSERT(out_edge.has_value());
190 Vertex in = sub.add_vertex(OpType::RNGInput);
191 Vertex out = sub.add_vertex(OpType::RNGOutput);
192 sub.boundary.insert({RngState(n_rng), in, out});
193 sub.rngwire.push_back(RngState(n_rng));
194 ++sub._number_of_rng_wires;
195 ++n_rng;
196 vmap[source(in_edge)] = in;
197 in_boundary_map.insert({in_edge, in});
198 vmap[target(*out_edge)] = out;
199 out_boundary_map.insert({*out_edge, out});
200 break;
201 }
202 default: {
203 TKET_ASSERT(false);
204 }
205 }
206 }
207 for (const Vertex& v : sc.verts) {
208 Vertex added = sub.add_vertex(get_Op_ptr_from_Vertex(v));
209 vmap[v] = added;
210 }
211 // Add edges into internal vertices
212 for (const Vertex& v : sc.verts) {
213 // iterate through original circuit as order of the set varies between Mac
214 // and Docker
215 BGL_FORALL_INEDGES(v, e, dag, DAG) {
216 Vertex source = this->source(e);
217 Vertex sub_source = vmap.at(source);
218 port_t in_port = get_source_port(e);
219 OpType type = sub.get_OpType_from_Vertex(sub_source);
220 if (is_initial_q_type(type) || type == OpType::ClInput ||
221 type == OpType::WASMInput || type == OpType::RNGInput) {
222 // Multiple inputs might be mapped to the same source
223 // so need to distinguish them.
224 sub_source = in_boundary_map.at(e);
225 in_port = 0;
226 }
227 sub.add_edge(
228 {sub_source, in_port}, {vmap.at(v), get_target_port(e)},
229 get_edgetype(e));
230 }
231 }
232 // Add edges into outputs
233 for (const std::optional<Edge>& e : sc.out_hole) {
234 if (e.has_value()) {
235 Vertex out = out_boundary_map.at(*e);
236 Vertex sub_source = vmap.at(source(*e));
237 port_t in_port = get_source_port(*e);
238 std::map<Edge, Vertex>::iterator found = in_boundary_map.find(*e);
239 if (found != in_boundary_map.end()) {
240 sub_source = found->second;
241 in_port = 0;
242 }
243 sub.add_edge({sub_source, in_port}, {out, 0}, get_edgetype(*e));
244 }
245 }
246 return sub;
247}

257// returns qubit path via vertices & inhabited port in vertices
258// used to construct a routing grid
259QPathDetailed Circuit::unit_path(const UnitID& unit) const {
260 Vertex current_v = get_in(unit);
261
262 QPathDetailed path = {{current_v, 0}};
263 Edge betweenEdge = get_nth_out_edge(current_v, 0);
264 current_v = target(betweenEdge);
265
266 while (detect_final_Op(current_v) == false) {
267 if (n_out_edges(current_v) == 0) {
268 throw CircuitInvalidity("A path ends before reaching an output vertex.");
269 }
270 port_t n = get_target_port(betweenEdge);
271 VertPort v_and_port = {current_v, n};
272 path.push_back(v_and_port);
273 betweenEdge = get_nth_out_edge(current_v, n);
274 current_v = target(betweenEdge);
275 }
276 path.push_back({current_v, 0});
277 return path;
278}

290std::map<UnitID, QPathDetailed> Circuit::all_unit_paths() const {
291 std::map<UnitID, QPathDetailed> new_list_of_paths;
292 for (const Qubit& q : all_qubits()) {
293 new_list_of_paths.insert({q, unit_path(q)});
294 }
295 for (const Bit& b : all_bits()) {
296 new_list_of_paths.insert({b, unit_path(b)});
297 }
298 for (const WasmState& w : wasmwire) {
299 new_list_of_paths.insert({w, unit_path(w)});
300 }
301 for (const RngState& r : rngwire) {
302 new_list_of_paths.insert({r, unit_path(r)});
303 }
304 return new_list_of_paths;
305}

317/*
318 Permute output boundary of circuit according to qubit map
319 Assumes all circuit Qubits are mapped
320*/
321void Circuit::permute_boundary_output(const qubit_map_t& qm) {
322 std::map<UnitID, BoundaryElement> new_entries;
323
324 for (const auto& qb_pair : qm) {
325 boundary_t::iterator input = boundary_elem(*this, qb_pair.first);
326 boundary_t::iterator output = boundary_elem(*this, qb_pair.second);
327 new_entries.insert({output->id_, {output->id_, output->in_, input->out_}});
328 }
329
330 for (const auto& pair : new_entries) {
331 boundary.erase(boundary.get<TagID>().find(pair.first));
332 }
333
334 for (const auto& pair : new_entries) {
335 boundary.insert(pair.second);
336 }
337}

339qubit_map_t Circuit::implicit_qubit_permutation() const {
340 qubit_map_t perm;
341 for (const QPathDetailed& path : all_qubit_paths()) {
342 Qubit in_q(get_id_from_in(path.front().first));
343 Qubit out_q(get_id_from_out(path.back().first));
344 perm.insert({in_q, out_q});
345 }
346 return perm;
347}

349bool Circuit::has_implicit_wireswaps() const {
350 qubit_map_t perm = implicit_qubit_permutation();
351 for (const std::pair<const Qubit, Qubit>& pair : perm) {
352 if (pair.first != pair.second) return true;
353 }
354 return false;

381Edge Circuit::skip_irrelevant_edges(Edge current) const {
382 Vertex try_next_v = target(current);
383 while (n_out_edges_of_type(try_next_v, EdgeType::Quantum) == 1) {

389SliceVec Circuit::get_reverse_slices() const {
390 vertex_map_t mapping;
391 vertex_map_t rev_mapping;
392 Circuit rev;
393 for (const BoundaryElement& el : boundary.get<TagID>()) {
394 Vertex new_in, new_out;
395 if (el.type() == UnitType::Qubit) {
396 new_in = rev.add_vertex(OpType::Input);
397 new_out = rev.add_vertex(OpType::Output);
398 } else {
399 new_in = rev.add_vertex(OpType::ClInput);
400 new_out = rev.add_vertex(OpType::ClOutput);
401 }
402 mapping[el.in_] = new_out;
403 rev_mapping[new_out] = el.in_;
404 mapping[el.out_] = new_in;
405 rev_mapping[new_in] = el.out_;
406 rev.boundary.insert({el.id_, new_in, new_out});
407 }
408 BGL_FORALL_VERTICES(v, dag, DAG) {
409 const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
410 switch (op_ptr->get_type()) {
411 case OpType::Input:
412 case OpType::Create:
413 case OpType::Output:
414 case OpType::ClInput:
415 case OpType::ClOutput:
416 case OpType::Discard:
417 case OpType::WASMInput:
418 case OpType::WASMOutput:
419 case OpType::RNGInput:
420 case OpType::RNGOutput: {
421 break;
422 }
423 default: {
424 Vertex v0 = rev.add_vertex(op_ptr);
425 mapping[v] = v0;
426 rev_mapping[v0] = v;
427 break;
428 }
429 }
430 }
431 BGL_FORALL_EDGES(e, dag, DAG) {
432 Vertex s = source(e);
433 port_t sp = get_source_port(e);
434 Vertex t = target(e);
435 port_t tp = get_target_port(e);
436 EdgeType type = dag[e].type;
437 if (type == EdgeType::Boolean) {
438 // Move Boolean to read from bit wire just before next write
439 Edge bit_wire = get_nth_out_edge(s, sp);
440 Vertex next_on_bit = target(bit_wire);
441 port_t next_p = get_target_port(bit_wire);
442 rev.add_edge({mapping[next_on_bit], next_p}, {mapping[t], tp}, type);
443 } else {
444 rev.add_edge({mapping[t], tp}, {mapping[s], sp}, type);
445 }
446 }
447 SliceVec slices_of_rev = rev.get_slices();
448 SliceVec rev_slices;
449 for (const Slice& s : slices_of_rev) {
450 Slice sl;
451 for (const Vertex& v : s) {
452 sl.push_back(rev_mapping[v]);
453 }
454 rev_slices.push_back(sl);
455 }
456 return rev_slices;
457}

459unsigned Circuit::depth() const {
460 unsigned count = 0;
461 std::function<bool(Op_ptr)> skip_func = [&](Op_ptr op) {
462 return (op->get_type() == OpType::Barrier);
463 };
464 SliceIterator slice_iter(*this, skip_func);
465 if (!(*slice_iter).empty()) count++;
466 while (!slice_iter.finished()) {
467 slice_iter.cut_ = this->next_cut(
468 slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
469 if (!(*slice_iter).empty()) count++;
470 }
471 return count;
472}

530std::map<Vertex, unsigned> Circuit::vertex_depth_map() const {
531 std::map<Vertex, unsigned> map;

558 return map;

561std::map<Edge, UnitID> Circuit::edge_unit_map() const {
562 std::map<Edge, UnitID> map;
563 for (const std::pair<const UnitID, QPathDetailed>& path : all_unit_paths()) {
564 QPathDetailed::const_iterator it = path.second.begin();
565 for (++it; it != path.second.end(); ++it) {
566 map.insert({get_nth_in_edge(it->first, it->second), path.first});
567 }
568 }
569 // Boolean edges are non-linear so are not added by the traces, but we can
570 // still link them to bits
571 BGL_FORALL_EDGES(e, dag, DAG) {
572 if (get_edgetype(e) == EdgeType::Boolean) {
573 Edge linear_edge = get_linear_edge(e);
574 map.insert({e, map.at(linear_edge)});
575 }
576 }
577 return map;
578}

609Circuit::CommandIterator& Circuit::CommandIterator::operator++() {
610 if (*this == circ_->end()) return *this;
611 if (current_index_ == (*current_slice_iterator_).size() - 1) {
612 if (current_slice_iterator_.finished()) {
613 *this = circ_->end();
614 return *this;
615 }
616 ++current_slice_iterator_;
617 current_index_ = 0;
618 } else {
619 ++current_index_;
620 }
621 if (current_index_ == (*current_slice_iterator_).size()) {
622 TKET_ASSERT(!"slice is empty");
623 }
624 current_vertex_ = (*current_slice_iterator_)[current_index_];
625 current_command_ = circ_->command_from_vertex(
626 current_vertex_, current_slice_iterator_.get_u_frontier(),
627 current_slice_iterator_.get_prev_b_frontier());
628 return *this;
629}

631unit_vector_t Circuit::args_from_frontier(
632 const Vertex& vert, std::shared_ptr<const unit_frontier_t> u_frontier,
633 std::shared_ptr<const b_frontier_t> prev_b_frontier) const {
634 EdgeVec ins = get_in_edges(vert);
635 unit_vector_t args;
636 for (port_t p = 0; p < ins.size(); ++p) {
637 if (get_edgetype(ins[p]) == EdgeType::Boolean) {
638 bool found = false;
639 for (const std::pair<Bit, EdgeVec>& pair :
640 prev_b_frontier->get<TagKey>()) {
641 for (const Edge& edge : pair.second) {
642 if (edge == ins[p]) {
643 args.push_back(pair.first);
644 found = true;
645 break;
646 }
647 }
648 if (found) {
649 break;
650 }
651 }
652 TKET_ASSERT(found); // Vertex edges not found in Boolean frontier.
653 } else {
654 Edge out = get_next_edge(vert, ins[p]);
655 bool found = false;
656 for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
657 if (pair.second == out) {
658 args.push_back(pair.first);
659 found = true;
660 break;
661 }
662 }
663 if (!found)
664 throw CircuitInvalidity(
665 "Vertex edges not found in frontier. Edge: " +
666 get_Op_ptr_from_Vertex(source(out))->get_name() + " -> " +
667 get_Op_ptr_from_Vertex(target(out))->get_name());
668 }
669 }
670 return args;
671}

675 std::shared_ptr<const b_frontier_t> prev_b_frontier) const {
