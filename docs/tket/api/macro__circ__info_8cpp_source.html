30 bool Circuit::is_simple() const {
31  if (!default_regs_ok()) return false;
32  for (const BoundaryElement& el : boundary.get<TagID>()) {
33  std::string reg = el.id_.reg_name();
34  if (!(reg == q_default_reg() || reg == c_default_reg())) return false;

39 bool Circuit::default_regs_ok() const {
40  opt_reg_info_t q_info = get_reg_info(q_default_reg());
41  register_info_t correct_q_info = {UnitType::Qubit, 1};
42  if (q_info && q_info.value() != correct_q_info) return false;
43  opt_reg_info_t c_info = get_reg_info(c_default_reg());
44  register_info_t correct_c_info = {UnitType::Bit, 1};
45  if (c_info && c_info.value() != correct_c_info) return false;
46  return true;
47 }

75 std::list<Command> Circuit::get_commands_of_type(OpType op_type) const {
76  std::list<Command> coms;
77  std::function<bool(Op_ptr)> skip_func = [=](Op_ptr op) {
78  return (op->get_type() != op_type);
79  };
80  Circuit::SliceIterator slice_iter(*this, skip_func);
81  for (const Vertex& v : *slice_iter) {
82  coms.push_back(command_from_vertex(
83  v, slice_iter.get_u_frontier(), slice_iter.get_prev_b_frontier()));
84  }
85  while (!slice_iter.finished()) {
86  slice_iter.cut_ = next_cut(
87  slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
88  for (const Vertex& v : *slice_iter) {
89  coms.push_back(command_from_vertex(
90  v, slice_iter.get_u_frontier(), slice_iter.get_prev_b_frontier()));
91  }
92  }
93  return coms;
94 }
95 unsigned Circuit::count_n_qubit_gates(unsigned size) const {
96  unsigned counter = 0;
97  if (size == 0) return counter;
98  BGL_FORALL_VERTICES(v, dag, DAG) {
99  if (n_in_edges_of_type(v, EdgeType::Quantum) == size) {
100  const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
101  switch (op_ptr->get_type()) {
102  case OpType::Input:
103  case OpType::Create:
104  case OpType::Output:
105  case OpType::Discard:
106  case OpType::Reset:
107  case OpType::Measure:
108  case OpType::Barrier: {
109  break;
110  }
111  default: {
112  counter++;
113  }
114  }
115  }
116  }
117  return counter;
118 }

120 Circuit Circuit::subcircuit(const Subcircuit& sc) const {
121  Circuit sub;
122  vertex_map_t vmap;
123  VertexVec q_ins;
124  VertexVec q_outs;
125  VertexVec c_ins;
126  VertexVec c_outs;
127  std::map<Edge, Vertex> in_boundary_map;
128  std::map<Edge, Vertex> out_boundary_map;
129  for (const Edge& e : sc.q_in_hole) {
130  Vertex added = sub.add_vertex(OpType::Input);
131  vmap[source(e)] = added;
132  q_ins.push_back(added);
133  in_boundary_map.insert({e, added});
134  }
135  for (const Edge& e : sc.q_out_hole) {
136  Vertex added = sub.add_vertex(OpType::Output);
137  vmap[target(e)] = added;
138  q_outs.push_back(added);
139  out_boundary_map.insert({e, added});
140  }
141  for (const Edge& e : sc.c_in_hole) {
142  Vertex added = sub.add_vertex(OpType::ClInput);
143  vmap[source(e)] = added;
144  c_ins.push_back(added);
145  in_boundary_map.insert({e, added});
146  }
147  for (const Edge& e : sc.c_out_hole) {
148  Vertex added = sub.add_vertex(OpType::ClOutput);
149  vmap[target(e)] = added;
150  c_outs.push_back(added);
151  out_boundary_map.insert({e, added});
152  }
153  for (unsigned i = 0; i < q_ins.size(); i++) {
154  sub.boundary.insert({Qubit(i), q_ins[i], q_outs[i]});
155  }
156  for (unsigned i = 0; i < c_ins.size(); i++) {
157  sub.boundary.insert({Bit(i), c_ins[i], c_outs[i]});
158  }
159  for (const Vertex& v : sc.verts) {
160  Vertex added = sub.add_vertex(get_Op_ptr_from_Vertex(v));
161  vmap[v] = added;
162  }
163  for (const Vertex& v : sc.verts) {
164  // iterate through original circuit as order of the set varies between Mac
165  // and Docker
166  BGL_FORALL_INEDGES(v, e, dag, DAG) {
167  Vertex source = this->source(e);
168  Vertex sub_source = vmap.at(source);
169  port_t in_port = get_source_port(e);
170  OpType type = sub.get_OpType_from_Vertex(sub_source);
171  if (is_initial_q_type(type) || type == OpType::ClInput) {
172  Edge boundary_edge = get_linear_edge(e);
173  // Multiple inputs might be mapped to the same source
174  // so need to distinguish them.
175  sub_source = in_boundary_map.at(boundary_edge);
176  in_port = 0;
177  }
178  sub.add_edge(
179  {sub_source, in_port}, {vmap.at(v), get_target_port(e)}, dag[e].type);
180  }
181  }
182  for (const Edge& e : sc.q_out_hole) {
183  // Multiple outputs might be mapped to the same target
184  // so need to distinguish them.
185  Vertex out = out_boundary_map[e];
186  Vertex sub_source = vmap.at(source(e));
187  port_t in_port = get_source_port(e);
188  std::map<Edge, Vertex>::iterator found = in_boundary_map.find(e);
189  if (found != in_boundary_map.end()) {
190  sub_source = found->second;
191  in_port = 0;
192  }
193  sub.add_edge({sub_source, in_port}, {out, 0}, EdgeType::Quantum);
194  }
195  for (const Edge& e : sc.c_out_hole) {
196  Vertex out = out_boundary_map[e];
197  Vertex sub_source = vmap.at(source(e));
198  port_t in_port = get_source_port(e);
199  std::map<Edge, Vertex>::iterator found = in_boundary_map.find(e);
200  if (found != in_boundary_map.end()) {
201  sub_source = found->second;
202  in_port = 0;
203  }
204  sub.add_edge({sub_source, in_port}, {out, 0}, EdgeType::Classical);
205  }
206  return sub;
207 }

209 // returns qubit path via vertices & inhabited port in vertices
210 // used to construct a routing grid
211 QPathDetailed Circuit::unit_path(const UnitID& unit) const {
212  Vertex current_v = get_in(unit);
213 
214  QPathDetailed path = {{current_v, 0}};
215  Edge betweenEdge = get_nth_out_edge(current_v, 0);
216  current_v = target(betweenEdge);
217 
218  while (detect_final_Op(current_v) == false) {
219  if (n_out_edges(current_v) == 0) {
220  throw CircuitInvalidity("A path ends before reaching an output vertex.");
221  }
222  port_t n = get_target_port(betweenEdge);
223  VertPort v_and_port = {current_v, n};
224  path.push_back(v_and_port);
225  betweenEdge = get_nth_out_edge(current_v, n);
226  current_v = target(betweenEdge);
227  }
228  path.push_back({current_v, 0});
229  return path;
230 }

263 /*
264  Permute output boundary of circuit according to qubit map
265  Assumes all circuit Qubits are mapped
266 */
267 void Circuit::permute_boundary_output(const qubit_map_t& qm) {
268  std::map<UnitID, BoundaryElement> new_entries;
269 
270  for (const auto& qb_pair : qm) {
271  boundary_t::iterator input = boundary_elem(*this, qb_pair.first);
272  boundary_t::iterator output = boundary_elem(*this, qb_pair.second);
273  new_entries.insert({output->id_, {output->id_, output->in_, input->out_}});
274  }
275 
276  for (const auto& pair : new_entries) {
277  boundary.erase(boundary.get<TagID>().find(pair.first));
278  }
279 
280  for (const auto& pair : new_entries) {
281  boundary.insert(pair.second);
282  }
283 }

335 static std::shared_ptr<unit_frontier_t> get_next_u_frontier(
336  const Circuit& circ, std::shared_ptr<const unit_frontier_t> u_frontier,
337  const VertexSet& next_slice_lookup) {
338  std::shared_ptr<unit_frontier_t> next_frontier =
339  std::make_shared<unit_frontier_t>();
340  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
341  Vertex next_v = circ.target(pair.second);
342  if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) {
343  next_frontier->insert(pair);
344  } else {
345  next_frontier->insert(
346  {pair.first, circ.get_next_edge(next_v, pair.second)});
347  }
348  }
349  return next_frontier;
350 }

352 static std::shared_ptr<b_frontier_t> get_next_b_frontier(
353  const Circuit& circ, std::shared_ptr<const b_frontier_t> b_frontier,
354  std::shared_ptr<const unit_frontier_t> u_frontier,
355  const VertexSet& next_slice_lookup) {
356  std::shared_ptr<b_frontier_t> next_b_frontier =
357  std::make_shared<b_frontier_t>();
358  // Copy any remaining edges
359  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
360  EdgeVec remaining;
361  for (const Edge& e : pair.second) {
362  Vertex targ = circ.target(e);
363  if (next_slice_lookup.find(targ) == next_slice_lookup.end()) {
364  remaining.push_back(e);
365  }
366  }
367  if (!remaining.empty()) {
368  next_b_frontier->insert({pair.first, remaining});
369  }
370  }
371  // Add any new bits introduced in this slice
372  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
373  if (circ.get_edgetype(pair.second) == EdgeType::Quantum) continue;
374  Vertex next_v = circ.target(pair.second);
375  if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) continue;
376  if (next_b_frontier->get<TagKey>().find(Bit(pair.first)) !=
377  next_b_frontier->end()) {
378  throw CircuitInvalidity("RAW hazard created in slicing");
379  }
380  port_t p = circ.get_target_port(pair.second);
381  EdgeVec reads = circ.get_nth_b_out_bundle(next_v, p);
382  if (!reads.empty()) next_b_frontier->insert({Bit(pair.first), reads});
383  }
384  return next_b_frontier;
385 }

387 CutFrontier Circuit::next_cut(
388  std::shared_ptr<const unit_frontier_t> u_frontier,
389  std::shared_ptr<const b_frontier_t> b_frontier) const {
390  auto next_slice = std::make_shared<Slice>();
391  VertexSet next_slice_lookup;
392  VertexSet bad_vertices;
393  std::list<Edge> all_edges;
394  EdgeSet edge_lookup;
395  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
396  if (pair.first.type() == UnitType::Bit) {
397  Vertex targ = target(pair.second);
398  b_frontier_t::const_iterator found =
399  b_frontier->get<TagKey>().find(Bit(pair.first));
400  if (found != b_frontier->get<TagKey>().end()) {
401  bool still_live = false;
402  for (const Edge& e : found->second) {
403  if (target(e) != targ) {
404  still_live = true;
405  break;
406  }
407  }
408  if (still_live) continue;
409  }
410  }
411  all_edges.push_back(pair.second);
412  edge_lookup.insert(pair.second);
413  }
414  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
415  for (const Edge& e : pair.second) {
416  all_edges.push_back(e);
417  edge_lookup.insert(e);
418  }
419  }
420  // find the next slice first
421  for (const Edge& e : all_edges) {
422  Vertex try_v = target(e);
423  if (detect_final_Op(try_v)) continue;
424  if (next_slice_lookup.find(try_v) != next_slice_lookup.end())
425  continue; // already going to be in next slice
426  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
427  if (!good_vertex) continue;
428  EdgeVec ins = get_in_edges(try_v);
429  for (const Edge& in : ins) {
430  if (edge_lookup.find(in) == edge_lookup.end()) {
431  good_vertex = false;
432  bad_vertices.insert(try_v);
433  break;
434  }
435  }
436  if (good_vertex) {
437  next_slice_lookup.insert(try_v);
438  next_slice->push_back(try_v);
439  }
440  }
441 
442  return {
443  next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
444  get_next_b_frontier(*this, b_frontier, u_frontier, next_slice_lookup)};
445 }

447 CutFrontier Circuit::next_cut(
448  std::shared_ptr<const unit_frontier_t> u_frontier,
449  std::shared_ptr<const b_frontier_t> b_frontier,
450  const std::function<bool(Op_ptr)>& skip_func) const {
451  VertexSet bad_vertices;
452  std::list<Edge> all_edges;
453  EdgeSet edge_lookup;
454  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
455  Edge e = pair.second;
456  all_edges.push_back(e);
457  edge_lookup.insert(e);
458  }
459  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
460  for (const Edge& edge : pair.second) {
461  Edge e = edge;
462  all_edges.push_back(e);
463  edge_lookup.insert(e);
464  }
465  }
466  // advance through skippable
467  bool can_skip;
468  do {
469  can_skip = false;
470  VertexSet skip_slice_lookup;
471  for (const Edge& e : all_edges) {
472  Vertex try_v = target(e);
473  if (detect_final_Op(try_v) || (!skip_func(get_Op_ptr_from_Vertex(try_v))))
474  continue;
475  if (skip_slice_lookup.find(try_v) != skip_slice_lookup.end()) continue;
476  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
477  if (!good_vertex) continue;
478  const EdgeVec ins = get_in_edges(try_v);
479  for (const Edge& in : ins) {
480  if (edge_lookup.find(in) == edge_lookup.end()) {
481  good_vertex = false;
482  break;
483  }
484  }
485  if (!good_vertex) {
486  bad_vertices.insert(try_v);
487  continue;
488  }
489  skip_slice_lookup.insert(try_v);
490  }
491  if (!skip_slice_lookup.empty()) {
492  b_frontier =
493  get_next_b_frontier(*this, b_frontier, u_frontier, skip_slice_lookup);
494  u_frontier = get_next_u_frontier(*this, u_frontier, skip_slice_lookup);
495  bad_vertices = {};
496  all_edges = {};
497  edge_lookup = {};
498 
499  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
500  Edge e = pair.second;
501  all_edges.push_back(e);
502  edge_lookup.insert(e);
503  }
504  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
505  for (const Edge& edge : pair.second) {
506  Edge e = edge;
507  all_edges.push_back(e);
508  edge_lookup.insert(e);
509  }
510  }
511  can_skip = true;
512  }
513  } while (can_skip);
514 
515  // find the next slice first
516  auto next_slice = std::make_shared<Slice>();
517  VertexSet next_slice_lookup;
518 
519  for (const Edge& e : all_edges) {
520  Vertex try_v = target(e);
521  if (detect_final_Op(try_v)) continue;
522  if (next_slice_lookup.find(try_v) != next_slice_lookup.end())
523  continue; // already going to be in next slice
524  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
525  if (!good_vertex) continue;
526  const EdgeVec ins = get_in_edges(try_v);
527  for (const Edge& in : ins) {
528  if (edge_lookup.find(in) == edge_lookup.end()) {
529  good_vertex = false;
530  bad_vertices.insert(try_v);
531  break;
532  }
533  }
534  if (good_vertex) {
535  next_slice_lookup.insert(try_v);
536  next_slice->push_back(try_v);
537  }
538  }
539 
540  return {
541  next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
542  get_next_b_frontier(*this, b_frontier, u_frontier, next_slice_lookup)};
543 }

545 CutFrontier Circuit::next_q_cut(
546  std::shared_ptr<const unit_frontier_t> u_frontier) const {
547  auto next_slice = std::make_shared<Slice>();
548  VertexSet next_slice_lookup;
549  VertexSet bad_vertices;
550  EdgeSet edge_lookup;
551  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
552  edge_lookup.insert(pair.second);
553  }
554 
555  // find the next slice first
556  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
557  Vertex try_v = target(pair.second);
558  if (detect_final_Op(try_v)) continue;
559  if (next_slice_lookup.contains(try_v))
560  continue; // already going to be in next slice
561  bool good_vertex = !bad_vertices.contains(try_v);
562  if (!good_vertex) continue;
563  EdgeVec ins = get_in_edges(try_v);
564  for (const Edge& in : ins) {
565  if (!edge_lookup.contains(in) && get_edgetype(in) == EdgeType::Quantum) {
566  good_vertex = false;
567  bad_vertices.insert(try_v);
568  break;
569  }
570  }
571  if (good_vertex) {
572  next_slice_lookup.insert(try_v);
573  next_slice->push_back(try_v);
574  }
575  }
576 
577  return {
578  next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
579  std::make_shared<b_frontier_t>()};
580 }

582 SliceVec Circuit::get_reverse_slices() const {
583  vertex_map_t mapping;
584  vertex_map_t rev_mapping;
585  Circuit rev;
586  for (const BoundaryElement& el : boundary.get<TagID>()) {
587  Vertex new_in, new_out;
588  if (el.type() == UnitType::Qubit) {
589  new_in = rev.add_vertex(OpType::Input);
590  new_out = rev.add_vertex(OpType::Output);
591  } else {
592  new_in = rev.add_vertex(OpType::ClInput);
593  new_out = rev.add_vertex(OpType::ClOutput);
594  }
595  mapping[el.in_] = new_out;
596  rev_mapping[new_out] = el.in_;
597  mapping[el.out_] = new_in;
598  rev_mapping[new_in] = el.out_;
599  rev.boundary.insert({el.id_, new_in, new_out});
600  }
601  BGL_FORALL_VERTICES(v, dag, DAG) {
602  const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
603  switch (op_ptr->get_type()) {
604  case OpType::Input:
605  case OpType::Create:
606  case OpType::Output:
607  case OpType::Discard:
608  case OpType::ClInput:
609  case OpType::ClOutput: {
610  break;
611  }
612  default: {
613  Vertex v0 = rev.add_vertex(op_ptr);
614  mapping[v] = v0;
615  rev_mapping[v0] = v;
616  break;
617  }
618  }
619  }
620  BGL_FORALL_EDGES(e, dag, DAG) {
621  Vertex s = source(e);
622  port_t sp = get_source_port(e);
623  Vertex t = target(e);
624  port_t tp = get_target_port(e);
625  EdgeType type = dag[e].type;
626  if (type == EdgeType::Boolean) {
627  // Move Boolean to read from bit wire just before next write
628  Edge bit_wire = get_nth_out_edge(s, sp);
629  Vertex next_on_bit = target(bit_wire);
630  port_t next_p = get_target_port(bit_wire);
631  rev.add_edge({mapping[next_on_bit], next_p}, {mapping[t], tp}, type);
632  } else {
633  rev.add_edge({mapping[t], tp}, {mapping[s], sp}, type);
634  }
635  }
636  SliceVec slices_of_rev = rev.get_slices();
637  SliceVec rev_slices;
638  for (const Slice& s : slices_of_rev) {
639  Slice sl;
640  for (const Vertex& v : s) {
641  sl.push_back(rev_mapping[v]);
642  }
643  rev_slices.push_back(sl);
644  }
645  return rev_slices;
646 }

747 Circuit::SliceIterator::SliceIterator(const Circuit& circ)
748  : cut_(), circ_(&circ) {
749  cut_.init();
750  for (const Qubit& q : circ.all_qubits()) {
751  Vertex in = circ.get_in(q);
752  cut_.slice->push_back(in);
753  cut_.u_frontier->insert({q, circ.get_nth_out_edge(in, 0)});
754  }

756  Vertex in = circ.get_in(b);
757  cut_.slice->push_back(in);

760  }

762  cut_ = circ.next_cut(cut_.u_frontier, cut_.b_frontier);
763 
764  // Add all vertices that have no Quantum or Classical edges (e.g. Phase) and
765  // no Boolean inputs:
766  VertexSet loners;
767  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
768  if (circ.n_in_edges(v) == 0 &&
769  circ.n_out_edges_of_type(v, EdgeType::Quantum) == 0 &&
770  circ.n_out_edges_of_type(v, EdgeType::Classical) == 0) {
771  loners.insert(v);
772  }
773  }
774  cut_.slice->insert(cut_.slice->end(), loners.begin(), loners.end());
775 }

876 unit_vector_t Circuit::args_from_frontier(
877  const Vertex& vert, std::shared_ptr<const unit_frontier_t> u_frontier,
878  std::shared_ptr<const b_frontier_t> prev_b_frontier) const {
879  EdgeVec ins = get_in_edges(vert);
880  unit_vector_t args;
881  for (port_t p = 0; p < ins.size(); ++p) {
882  if (get_edgetype(ins[p]) == EdgeType::Boolean) {
883  bool found = false;
884  for (const std::pair<Bit, EdgeVec>& pair :
885  prev_b_frontier->get<TagKey>()) {
886  for (const Edge& edge : pair.second) {
887  if (edge == ins[p]) {
888  args.push_back(pair.first);
889  found = true;
890  break;
891  }
892  }
893  if (found) break;
894  }
895  if (!found)
896  throw CircuitInvalidity(
897  "Vertex edges not found in CRead frontier. Edge: " +
898  get_Op_ptr_from_Vertex(source(ins[p]))->get_name() + " -> " +
899  get_Op_ptr_from_Vertex(target(ins[p]))->get_name());
900  } else {
901  Edge out = get_next_edge(vert, ins[p]);
902  bool found = false;
903  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
904  if (pair.second == out) {
905  args.push_back(pair.first);
906  found = true;
907  break;
908  }
909  }
910  if (!found)
911  throw CircuitInvalidity(
912  "Vertex edges not found in frontier. Edge: " +
913  get_Op_ptr_from_Vertex(source(out))->get_name() + " -> " +
914  get_Op_ptr_from_Vertex(target(out))->get_name());
915  }
916  }
917  return args;
918 }
