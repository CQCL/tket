30 bool Circuit::is_simple() const {
31  if (!default_regs_ok()) return false;
32  for (const BoundaryElement& el : boundary.get<TagID>()) {
33  std::string reg = el.id_.reg_name();
34  if (!(reg == q_default_reg() || reg == c_default_reg())) return false;

39 bool Circuit::default_regs_ok() const {
40  opt_reg_info_t q_info = get_reg_info(q_default_reg());
41  register_info_t correct_q_info = {UnitType::Qubit, 1};
42  if (q_info && q_info.value() != correct_q_info) return false;
43  opt_reg_info_t c_info = get_reg_info(c_default_reg());
44  register_info_t correct_c_info = {UnitType::Bit, 1};
45  if (c_info && c_info.value() != correct_c_info) return false;
46  return true;

75 std::list<Command> Circuit::get_commands_of_type(OpType op_type) const {
76  std::list<Command> coms;
77  std::function<bool(Op_ptr)> skip_func = [=](Op_ptr op) {
78  return (op->get_type() != op_type);
79  };
80  Circuit::SliceIterator slice_iter(*this, skip_func);
81  for (const Vertex& v : *slice_iter) {
82  coms.push_back(command_from_vertex(
83  v, slice_iter.get_u_frontier(), slice_iter.get_prev_b_frontier()));
84  }
85  while (!slice_iter.finished()) {
86  slice_iter.cut_ = next_cut(
87  slice_iter.cut_.u_frontier, slice_iter.cut_.b_frontier, skip_func);
88  for (const Vertex& v : *slice_iter) {
89  coms.push_back(command_from_vertex(
90  v, slice_iter.get_u_frontier(), slice_iter.get_prev_b_frontier()));
91  }
92  }
93  return coms;
94 }
95 unsigned Circuit::count_n_qubit_gates(unsigned size) const {
96  unsigned counter = 0;
97  if (size == 0) return counter;
98  BGL_FORALL_VERTICES(v, dag, DAG) {
99  if (n_in_edges_of_type(v, EdgeType::Quantum) == size) {
100  const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
101  switch (op_ptr->get_type()) {
102  case OpType::Input:
103  case OpType::Create:
104  case OpType::Output:
105  case OpType::Discard:
106  case OpType::Reset:
107  case OpType::Measure:
108  case OpType::Barrier: {
109  break;
110  }
111  default: {
112  counter++;
113  }
114  }
115  }
116  }
117  return counter;
118 }

120 Circuit Circuit::subcircuit(const Subcircuit& sc) const {
121  Circuit sub;
122  vertex_map_t vmap;
123  VertexVec q_ins;
124  VertexVec q_outs;
125  VertexVec c_ins;
126  VertexVec c_outs;
127  std::map<Edge, Vertex> in_boundary_map;
128  std::map<Edge, Vertex> out_boundary_map;
129  for (const Edge& e : sc.q_in_hole) {
130  Vertex added = sub.add_vertex(OpType::Input);
131  vmap[source(e)] = added;
132  q_ins.push_back(added);
133  in_boundary_map.insert({e, added});
134  }
135  for (const Edge& e : sc.q_out_hole) {
136  Vertex added = sub.add_vertex(OpType::Output);
137  vmap[target(e)] = added;
138  q_outs.push_back(added);
139  out_boundary_map.insert({e, added});
140  }
141  for (const Edge& e : sc.c_in_hole) {
142  Vertex added = sub.add_vertex(OpType::ClInput);
143  vmap[source(e)] = added;
144  c_ins.push_back(added);
145  in_boundary_map.insert({e, added});
146  }
147  for (const Edge& e : sc.c_out_hole) {
148  Vertex added = sub.add_vertex(OpType::ClOutput);
149  vmap[target(e)] = added;
150  c_outs.push_back(added);
151  out_boundary_map.insert({e, added});
152  }
153  for (unsigned i = 0; i < q_ins.size(); i++) {
154  sub.boundary.insert({Qubit(i), q_ins[i], q_outs[i]});
155  }
156  for (unsigned i = 0; i < c_ins.size(); i++) {
157  sub.boundary.insert({Bit(i), c_ins[i], c_outs[i]});
158  }
159  for (const Vertex& v : sc.verts) {
160  Vertex added = sub.add_vertex(get_Op_ptr_from_Vertex(v));
161  vmap[v] = added;
162  }
163  for (const Vertex& v : sc.verts) {
164  // iterate through original circuit as order of the set varies between Mac
165  // and Docker
166  BGL_FORALL_INEDGES(v, e, dag, DAG) {
167  Vertex source = this->source(e);
168  Vertex sub_source = vmap.at(source);
169  port_t in_port = get_source_port(e);
170  OpType type = sub.get_OpType_from_Vertex(sub_source);
171  if (is_initial_q_type(type) || type == OpType::ClInput) {
172  Edge boundary_edge = get_linear_edge(e);
173  // Multiple inputs might be mapped to the same source
174  // so need to distinguish them.
175  sub_source = in_boundary_map.at(boundary_edge);
176  in_port = 0;
177  }
178  sub.add_edge(
179  {sub_source, in_port}, {vmap.at(v), get_target_port(e)}, dag[e].type);
180  }
181  }
182  for (const Edge& e : sc.q_out_hole) {
183  // Multiple outputs might be mapped to the same target
184  // so need to distinguish them.
185  Vertex out = out_boundary_map[e];
186  Vertex sub_source = vmap.at(source(e));
187  port_t in_port = get_source_port(e);
188  std::map<Edge, Vertex>::iterator found = in_boundary_map.find(e);
189  if (found != in_boundary_map.end()) {
190  sub_source = found->second;
191  in_port = 0;
192  }
193  sub.add_edge({sub_source, in_port}, {out, 0}, EdgeType::Quantum);
194  }
195  for (const Edge& e : sc.c_out_hole) {
196  Vertex out = out_boundary_map[e];
197  Vertex sub_source = vmap.at(source(e));
198  port_t in_port = get_source_port(e);
199  std::map<Edge, Vertex>::iterator found = in_boundary_map.find(e);
200  if (found != in_boundary_map.end()) {
201  sub_source = found->second;
202  in_port = 0;
203  }
204  sub.add_edge({sub_source, in_port}, {out, 0}, EdgeType::Classical);
205  }
206  return sub;
207 }

209 // returns qubit path via vertices & inhabited port in vertices
210 // used to construct a routing grid
211 QPathDetailed Circuit::unit_path(const UnitID& unit) const {
212  Vertex current_v = get_in(unit);
213 
214  QPathDetailed path = {{current_v, 0}};
215  Edge betweenEdge = get_nth_out_edge(current_v, 0);
216  current_v = target(betweenEdge);
217 
218  while (detect_final_Op(current_v) == false) {
219  if (n_out_edges(current_v) == 0) {
220  throw CircuitInvalidity("A path ends before reaching an output vertex.");
221  }
222  port_t n = get_target_port(betweenEdge);
223  VertPort v_and_port = {current_v, n};
224  path.push_back(v_and_port);
225  betweenEdge = get_nth_out_edge(current_v, n);
226  current_v = target(betweenEdge);
227  }
228  path.push_back({current_v, 0});
229  return path;
230 }

263 /*
264  Permute output boundary of circuit according to qubit map
265  Assumes all circuit Qubits are mapped
266 */
267 void Circuit::permute_boundary_output(const qubit_map_t& qm) {
268  std::map<UnitID, BoundaryElement> new_entries;
269 
270  for (const auto& qb_pair : qm) {
271  boundary_t::iterator input = boundary_elem(*this, qb_pair.first);
272  boundary_t::iterator output = boundary_elem(*this, qb_pair.second);
273  new_entries.insert({output->id_, {output->id_, output->in_, input->out_}});
274  }
275 
276  for (const auto& pair : new_entries) {
277  boundary.erase(boundary.get<TagID>().find(pair.first));
278  }
279 
280  for (const auto& pair : new_entries) {
281  boundary.insert(pair.second);
282  }
283 }

335 static std::shared_ptr<unit_frontier_t> get_next_u_frontier(
336  const Circuit& circ, std::shared_ptr<const unit_frontier_t> u_frontier,
337  const VertexSet& next_slice_lookup) {
338  std::shared_ptr<unit_frontier_t> next_frontier =
339  std::make_shared<unit_frontier_t>();
340  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
341  Vertex next_v = circ.target(pair.second);
342  if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) {
343  next_frontier->insert(pair);
344  } else {
345  next_frontier->insert(
346  {pair.first, circ.get_next_edge(next_v, pair.second)});
347  }
348  }
349  return next_frontier;
350 }

352 static std::shared_ptr<b_frontier_t> get_next_b_frontier(
353  const Circuit& circ, std::shared_ptr<const b_frontier_t> b_frontier,
354  std::shared_ptr<const unit_frontier_t> u_frontier,
355  const VertexSet& next_slice_lookup) {
356  std::shared_ptr<b_frontier_t> next_b_frontier =
357  std::make_shared<b_frontier_t>();
358  // Copy any remaining edges
359  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
360  EdgeVec remaining;
361  for (const Edge& e : pair.second) {
362  Vertex targ = circ.target(e);
363  if (next_slice_lookup.find(targ) == next_slice_lookup.end()) {
364  remaining.push_back(e);
365  }
366  }
367  if (!remaining.empty()) {
368  next_b_frontier->insert({pair.first, remaining});
369  }
370  }
371  // Add any new bits introduced in this slice
372  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
373  switch (circ.get_edgetype(pair.second)) {
374  case EdgeType::Quantum:
375  case EdgeType::WASM: {
376  break;
377  }
378  case EdgeType::Classical: {
379  Vertex next_v = circ.target(pair.second);
380  if (next_slice_lookup.find(next_v) == next_slice_lookup.end()) continue;
381  if (next_b_frontier->get<TagKey>().find(Bit(pair.first)) !=
382  next_b_frontier->end()) {
383  TKET_ASSERT(!"RAW hazard created in slicing");
384  }
385  port_t p = circ.get_target_port(pair.second);
386  EdgeVec reads = circ.get_nth_b_out_bundle(next_v, p);
387  if (!reads.empty()) next_b_frontier->insert({Bit(pair.first), reads});
388  break;
389  }
390  case EdgeType::Boolean: {
391  throw CircuitInvalidity("Boolen edge not allowed in b_frontier_t");
392  }
393  default: {
394  TKET_ASSERT(!"get_next_b_frontier found invalid edge type in sig");
395  }
396  }
397  }
398  return next_b_frontier;
399 }

401 CutFrontier Circuit::next_cut(
402  std::shared_ptr<const unit_frontier_t> u_frontier,
403  std::shared_ptr<const b_frontier_t> b_frontier) const {
404  auto next_slice = std::make_shared<Slice>();
405  VertexSet next_slice_lookup;
406  VertexSet bad_vertices;
407  std::list<Edge> all_edges;
408  EdgeSet edge_lookup;
409  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
410  if (pair.first.type() == UnitType::Bit) {
411  Vertex targ = target(pair.second);
412  b_frontier_t::const_iterator found =
413  b_frontier->get<TagKey>().find(Bit(pair.first));
414  if (found != b_frontier->get<TagKey>().end()) {
415  bool still_live = false;
416  for (const Edge& e : found->second) {
417  if (target(e) != targ) {
418  still_live = true;
419  break;
420  }
421  }
422  if (still_live) continue;
423  }
424  }
425  all_edges.push_back(pair.second);
426  edge_lookup.insert(pair.second);
427  }
428 
429  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
430  for (const Edge& e : pair.second) {
431  all_edges.push_back(e);
432  edge_lookup.insert(e);
433  }
434  }
435  // find the next slice first
436  for (const Edge& e : all_edges) {
437  Vertex try_v = target(e);
438  if (detect_final_Op(try_v)) continue;
439  if (next_slice_lookup.find(try_v) != next_slice_lookup.end())
440  continue; // already going to be in next slice
441  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
442  if (!good_vertex) continue;
443  EdgeVec ins = get_in_edges(try_v);
444  for (const Edge& in : ins) {
445  if (edge_lookup.find(in) == edge_lookup.end()) {
446  good_vertex = false;
447  bad_vertices.insert(try_v);
448  break;
449  }
450  }
451  if (good_vertex) {
452  next_slice_lookup.insert(try_v);
453  next_slice->push_back(try_v);
454  }
455  }
456 
457  return {
458  next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
459  get_next_b_frontier(*this, b_frontier, u_frontier, next_slice_lookup)};
460 }

462 CutFrontier Circuit::next_cut(
463  std::shared_ptr<const unit_frontier_t> u_frontier,
464  std::shared_ptr<const b_frontier_t> b_frontier,
465  const std::function<bool(Op_ptr)>& skip_func) const {
466  VertexSet bad_vertices;
467  std::list<Edge> all_edges;
468  EdgeSet edge_lookup;
469  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
470  Edge e = pair.second;
471  all_edges.push_back(e);
472  edge_lookup.insert(e);
473  }
474  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
475  for (const Edge& edge : pair.second) {
476  Edge e = edge;
477  all_edges.push_back(e);
478  edge_lookup.insert(e);
479  }
480  }
481  // advance through skippable
482  bool can_skip;
483  do {
484  can_skip = false;
485  VertexSet skip_slice_lookup;
486  for (const Edge& e : all_edges) {
487  Vertex try_v = target(e);
488  if (detect_final_Op(try_v) || (!skip_func(get_Op_ptr_from_Vertex(try_v))))
489  continue;
490  if (skip_slice_lookup.find(try_v) != skip_slice_lookup.end()) continue;
491  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
492  if (!good_vertex) continue;
493  const EdgeVec ins = get_in_edges(try_v);
494  for (const Edge& in : ins) {
495  if (edge_lookup.find(in) == edge_lookup.end()) {
496  good_vertex = false;
497  break;
498  }
499  }
500  if (!good_vertex) {
501  bad_vertices.insert(try_v);
502  continue;
503  }
504  skip_slice_lookup.insert(try_v);
505  }
506  if (!skip_slice_lookup.empty()) {
507  b_frontier =
508  get_next_b_frontier(*this, b_frontier, u_frontier, skip_slice_lookup);
509  u_frontier = get_next_u_frontier(*this, u_frontier, skip_slice_lookup);
510  bad_vertices = {};
511  all_edges = {};
512  edge_lookup = {};
513 
514  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
515  Edge e = pair.second;
516  all_edges.push_back(e);
517  edge_lookup.insert(e);
518  }
519  for (const std::pair<Bit, EdgeVec>& pair : b_frontier->get<TagKey>()) {
520  for (const Edge& edge : pair.second) {
521  Edge e = edge;
522  all_edges.push_back(e);
523  edge_lookup.insert(e);
524  }
525  }
526  can_skip = true;
527  }
528  } while (can_skip);
529 
530  // find the next slice first
531  auto next_slice = std::make_shared<Slice>();
532  VertexSet next_slice_lookup;
533 
534  for (const Edge& e : all_edges) {
535  Vertex try_v = target(e);
536  if (detect_final_Op(try_v)) continue;
537  if (next_slice_lookup.find(try_v) != next_slice_lookup.end())
538  continue; // already going to be in next slice
539  bool good_vertex = bad_vertices.find(try_v) == bad_vertices.end();
540  if (!good_vertex) continue;
541  const EdgeVec ins = get_in_edges(try_v);
542  for (const Edge& in : ins) {
543  if (edge_lookup.find(in) == edge_lookup.end()) {
544  good_vertex = false;
545  bad_vertices.insert(try_v);
546  break;
547  }
548  }
549  if (good_vertex) {
550  next_slice_lookup.insert(try_v);
551  next_slice->push_back(try_v);
552  }
553  }
554 
555  return {
556  next_slice, get_next_u_frontier(*this, u_frontier, next_slice_lookup),
557  get_next_b_frontier(*this, b_frontier, u_frontier, next_slice_lookup)};
558 }

560 CutFrontier Circuit::next_q_cut(
561  std::shared_ptr<const unit_frontier_t> u_frontier) const {
562  auto next_slice = std::make_shared<Slice>();
563  VertexSet next_slice_lookup;
564  VertexSet bad_vertices;
565  EdgeSet edge_lookup;
566  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
567  edge_lookup.insert(pair.second);
568  }
569 
570  // find the next slice first
571  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
572  Vertex try_v = target(pair.second);
573  if (detect_final_Op(try_v)) continue;
574  if (next_slice_lookup.contains(try_v))
575  continue; // already going to be in next slice
576  bool good_vertex = !bad_vertices.contains(try_v);
577  if (!good_vertex) continue;
578  EdgeVec ins = get_in_edges(try_v);
579  for (const Edge& in : ins) {
580  if (!edge_lookup.contains(in) && (get_edgetype(in) == EdgeType::Quantum ||
581  get_edgetype(in) == EdgeType::WASM)) {
582  good_vertex = false;
583  bad_vertices.insert(try_v);
584  break;
585  }
586  }
587  if (good_vertex) {
588  next_slice_lookup.insert(try_v);
589  next_slice->push_back(try_v);
590  }
591  }
592 
593  return {

598 SliceVec Circuit::get_reverse_slices() const {
599  vertex_map_t mapping;
600  vertex_map_t rev_mapping;
601  Circuit rev;
602  for (const BoundaryElement& el : boundary.get<TagID>()) {
603  Vertex new_in, new_out;
604  if (el.type() == UnitType::Qubit) {
605  new_in = rev.add_vertex(OpType::Input);
606  new_out = rev.add_vertex(OpType::Output);
607  } else {
608  new_in = rev.add_vertex(OpType::ClInput);
609  new_out = rev.add_vertex(OpType::ClOutput);
610  }
611  mapping[el.in_] = new_out;
612  rev_mapping[new_out] = el.in_;
613  mapping[el.out_] = new_in;
614  rev_mapping[new_in] = el.out_;
615  rev.boundary.insert({el.id_, new_in, new_out});
616  }
617  BGL_FORALL_VERTICES(v, dag, DAG) {
618  const Op_ptr op_ptr = get_Op_ptr_from_Vertex(v);
619  switch (op_ptr->get_type()) {
620  case OpType::Input:
621  case OpType::Create:
622  case OpType::Output:
623  case OpType::ClInput:
624  case OpType::ClOutput:
625  case OpType::Discard:
626  case OpType::WASMInput:
627  case OpType::WASMOutput: {
628  break;
629  }
630  default: {
631  Vertex v0 = rev.add_vertex(op_ptr);
632  mapping[v] = v0;
633  rev_mapping[v0] = v;
634  break;
635  }
636  }
637  }
638  BGL_FORALL_EDGES(e, dag, DAG) {
639  Vertex s = source(e);
640  port_t sp = get_source_port(e);
641  Vertex t = target(e);
642  port_t tp = get_target_port(e);
643  EdgeType type = dag[e].type;
644  if (type == EdgeType::Boolean) {
645  // Move Boolean to read from bit wire just before next write
646  Edge bit_wire = get_nth_out_edge(s, sp);
647  Vertex next_on_bit = target(bit_wire);
648  port_t next_p = get_target_port(bit_wire);
649  rev.add_edge({mapping[next_on_bit], next_p}, {mapping[t], tp}, type);
650  } else {
651  rev.add_edge({mapping[t], tp}, {mapping[s], sp}, type);
652  }
653  }
654  SliceVec slices_of_rev = rev.get_slices();
655  SliceVec rev_slices;
656  for (const Slice& s : slices_of_rev) {
657  Slice sl;
658  for (const Vertex& v : s) {
659  sl.push_back(rev_mapping[v]);
660  }
661  rev_slices.push_back(sl);
662  }
663  return rev_slices;
664 }

934 unit_vector_t Circuit::args_from_frontier(
935  const Vertex& vert, std::shared_ptr<const unit_frontier_t> u_frontier,
936  std::shared_ptr<const b_frontier_t> prev_b_frontier) const {
937  EdgeVec ins = get_in_edges(vert);
938  unit_vector_t args;
939  for (port_t p = 0; p < ins.size(); ++p) {
940  switch (get_edgetype(ins[p])) {
941  case EdgeType::WASM: {
942  Edge out = get_next_edge(vert, ins[p]);
943  bool found = false;
944  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
945  if (pair.second == out) {
946  args.push_back(pair.first);
947  found = true;
948  break;
949  }
950  }
951  TKET_ASSERT(found); // Vertex edges not found in frontier.
952  break;
953  }
954  case EdgeType::Boolean: {
955  bool found = false;
956  for (const std::pair<Bit, EdgeVec>& pair :
957  prev_b_frontier->get<TagKey>()) {
958  for (const Edge& edge : pair.second) {
959  if (edge == ins[p]) {
960  args.push_back(pair.first);
961  found = true;
962  break;
963  }
964  }
965  if (found) {
966  break;
967  }
968  }
969  TKET_ASSERT(found); // Vertex edges not found in Boolean frontier.
970  break;
971  }
972  case EdgeType::Classical:
973  case EdgeType::Quantum: {
974  Edge out = get_next_edge(vert, ins[p]);
975  bool found = false;
976  for (const std::pair<UnitID, Edge>& pair : u_frontier->get<TagKey>()) {
977  if (pair.second == out) {
978  args.push_back(pair.first);
979  found = true;
980  break;
981  }
982  }
983  if (!found)
984  throw CircuitInvalidity(
985  "Vertex edges not found in frontier. Edge: " +
986  get_Op_ptr_from_Vertex(source(out))->get_name() + " -> " +
987  get_Op_ptr_from_Vertex(target(out))->get_name());
988  break;
989  }
990  default: {
991  TKET_ASSERT(!"args_from_frontier found invalid edge type in signature");
992  }
993  }
994  }
995  return args;
996 }
