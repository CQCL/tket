31 vertex_map_t Circuit::copy_graph(
32  const Circuit& c2, BoundaryMerge boundary_merge,
33  OpGroupTransfer opgroup_transfer) {
34  switch (opgroup_transfer) {
35  case OpGroupTransfer::Preserve:
36  // Fail if any collisions.
37  for (const auto& opgroupsig : c2.opgroupsigs) {
38  if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
39  throw CircuitInvalidity("Name collision in inserted circuit");
40  }
41  }
42  // Add inserted opgroups to circuit.
43  opgroupsigs.insert(c2.opgroupsigs.begin(), c2.opgroupsigs.end());
44  break;
45  case OpGroupTransfer::Disallow:
46  // Fail if any opgroups.
47  if (!c2.opgroupsigs.empty()) {
48  throw CircuitInvalidity("Named op groups in inserted circuit");
49  }
50  break;
51  case OpGroupTransfer::Merge:
52  // Fail if any mismatched signatures
53  for (const auto& opgroupsig : c2.opgroupsigs) {
54  if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
55  if (opgroupsigs[opgroupsig.first] != opgroupsig.second) {
56  throw CircuitInvalidity(
57  "Name signature mismatch in inserted circuit");
58  }
59  }
60  }
61  // Add inserted opgroups to circuit.
62  opgroupsigs.insert(c2.opgroupsigs.begin(), c2.opgroupsigs.end());
63  break;
64  default:
65  TKET_ASSERT(opgroup_transfer == OpGroupTransfer::Remove);
66  // Ignore inserted opgroups
67  break;
68  }
69 
70  vertex_map_t isomap;
71  if (&c2 == this) {
72  throw Unsupported(
73  "Circuit Cannot currently copy itself using this method. Use * "
74  "instead\n");
75  }
76  BGL_FORALL_VERTICES(v, c2.dag, DAG) {
77  Vertex v0 = boost::add_vertex(this->dag);
78  this->dag[v0].op = c2.get_Op_ptr_from_Vertex(v);
79  if (opgroup_transfer == OpGroupTransfer::Preserve ||
80  opgroup_transfer == OpGroupTransfer::Merge) {
81  this->dag[v0].opgroup = c2.get_opgroup_from_Vertex(v);
82  }
83  isomap.insert({v, v0});
84  }
85  BGL_FORALL_VERTICES(v, c2.dag, DAG) {
86  EdgeVec edges = c2.get_in_edges(v);
87  Vertex target_v = isomap.find(v)->second;
88  for (EdgeVec::iterator e1 = edges.begin(); e1 != edges.end(); ++e1) {
89  Vertex old_source_v = c2.source(*e1);
90  Vertex source_v = isomap.find(old_source_v)->second;
91  add_edge(
92  {source_v, get_source_port(*e1)}, {target_v, get_target_port(*e1)},
93  c2.dag[*e1].type);
94  }
95  }
96 
97  if (boundary_merge == BoundaryMerge::Yes) {
98  for (const BoundaryElement& el : c2.boundary.get<TagID>()) {
99  std::string reg_name = el.id_.reg_name();
100  register_info_t reg_type = el.reg_info();
101  opt_reg_info_t reg_found = get_reg_info(reg_name);
102  if (reg_found) {
103  if (reg_found.value() != reg_type) {
104  throw Unsupported(
105  "Cannot merge circuits with different types for "
106  "register with name: " +
107  reg_name);
108  }
109  boundary_t::iterator unit_found = boundary.get<TagID>().find(el.id_);
110  if (unit_found != boundary.get<TagID>().end()) {
111  throw Unsupported(
112  "Cannot merge circuits as both contain unit: " + el.id_.repr());
113  }
114  }
115  Vertex new_in = isomap[el.in_];
116  Vertex new_out = isomap[el.out_];
117  boundary.insert({el.id_, new_in, new_out});
118  }
119  }
120  return isomap;
121 }

135 // qm is from the units on the second (appended) circuit to the units on the
136 // first (this) circuit
137 void Circuit::append_with_map(const Circuit& c2, const unit_map_t& qm) {
138  Circuit copy = c2;
139  copy.rename_units(qm);
140 
141  // Check what we need to do at the joins:
142  // Output --- Input ==> -------------
143  // Output --- Create ==> --- Reset ---
144  // Discard --- Input ==> [not allowed]
145  // Discard --- Create ==> --- Reset ---
146  qubit_vector_t qbs = copy.all_qubits();
147  std::set<Qubit> qbs_set(qbs.begin(), qbs.end());
148  std::set<Qubit> reset_qbs;
149  for (auto qb : all_qubits()) {
150  if (qbs_set.find(qb) != qbs_set.end()) {
151  if (copy.is_created(qb)) {
152  reset_qbs.insert(qb);
153  } else if (is_discarded(qb)) {
154  throw CircuitInvalidity("Cannot append input qubit to discarded qubit");
155  }
156  }
157  }
158 
159  // Copy c2 into c1 but do not merge boundaries
160  vertex_map_t vm = copy_graph(copy, BoundaryMerge::No);
161  const Op_ptr noop = get_op_ptr(OpType::noop);
162 
163  // Connect each matching qubit and bit, merging remainder
164  for (const BoundaryElement& el : copy.boundary.get<TagID>()) {
165  std::string reg_name = el.reg_name();
166  register_info_t reg_type = el.reg_info();
167  opt_reg_info_t reg_found = get_reg_info(reg_name);
168  if (reg_found) {
169  if (reg_found.value() != reg_type) {
170  throw Unsupported(
171  "Cannot append circuits with different types for "
172  "register with name: " +
173  reg_name);
174  }
175  boundary_t::iterator unit_found = boundary.get<TagID>().find(el.id_);
176  if (unit_found != boundary.get<TagID>().end()) {
177  Vertex out = unit_found->out_;
178  Vertex in = vm[el.in_];
179  // Update map
180  BoundaryElement new_elem = *unit_found;
181  new_elem.out_ = vm[el.out_];
182  boundary.replace(unit_found, new_elem);
183  // Tie together
184  if (reg_type.first == UnitType::Qubit)
185  add_edge({out, 0}, {in, 0}, EdgeType::Quantum);
186  else
187  add_edge({out, 0}, {in, 0}, EdgeType::Classical);
188  dag[out].op = noop;
189  dag[in].op = noop;
190  remove_vertex(out, GraphRewiring::Yes, VertexDeletion::Yes);
191  if (el.type() != UnitType::Qubit ||
192  reset_qbs.find(Qubit(el.id_)) == reset_qbs.end()) {
193  remove_vertex(in, GraphRewiring::Yes, VertexDeletion::Yes);
194  } else {
195  dag[in].op = std::make_shared<const Gate>(OpType::Reset);
196  }
197  } else {
198  Vertex new_in = vm[el.in_];
199  Vertex new_out = vm[el.out_];
200  boundary.insert({el.id_, new_in, new_out});
201  }
202  } else {
203  Vertex new_in = vm[el.in_];
204  Vertex new_out = vm[el.out_];
205  boundary.insert({el.id_, new_in, new_out});
206  }
207  }
208  add_phase(c2.get_phase());
209 }

211 void Circuit::append_qubits(
212  const Circuit& c2, const std::vector<unsigned>& qubits,
213  const std::vector<unsigned>& bits) {
214  unit_map_t qm;
215  for (unsigned i = 0; i < qubits.size(); i++) {
216  qm.insert({Qubit(i), Qubit(qubits[i])});
217  }
218  for (unsigned i = 0; i < bits.size(); i++) {
219  qm.insert({Bit(i), Bit(bits[i])});
220  }
221  append_with_map(c2, qm);
222 }

233 // Important substitute method. Requires knowledge of the boundary to insert
234 // into, and the vertices inside which are to be removed when substitution is
235 // performed. Gives the option to isolate the removed vertices but not delete
236 // them.
237 void Circuit::substitute(
238  const Circuit& to_insert, const Subcircuit& to_replace,
239  VertexDeletion vertex_deletion, OpGroupTransfer opgroup_transfer) {
240  if (!to_insert.is_simple()) throw SimpleOnly();
241  if (to_insert.n_qubits() != to_replace.q_in_hole.size() ||
242  to_insert.n_bits() != to_replace.c_in_hole.size())
243  throw CircuitInvalidity("Subcircuit boundary mismatch to hole");
244 
245  vertex_map_t vm = copy_graph(to_insert, BoundaryMerge::No, opgroup_transfer);
246  VertexList bin;
247  EdgeSet ebin; // Needs to be a set since subcircuit to replace could be
248  // trivial, essentially rewiring on a cut
249  std::map<Edge, Vertex> c_out_map;
250 
251  const Op_ptr noop = get_op_ptr(OpType::noop);
252  for (unsigned i = 0; i < to_replace.q_in_hole.size(); i++) {
253  Edge edge = to_replace.q_in_hole[i];
254  Vertex pred_v = source(edge);
255  port_t port1 = get_source_port(edge);
256  ebin.insert(edge);
257  Vertex inp = vm[to_insert.get_in(Qubit(i))];
258  add_edge({pred_v, port1}, {inp, 0}, EdgeType::Quantum);
259  dag[inp].op = noop;
260  bin.push_back(inp);
261  }
262  for (unsigned i = 0; i < to_replace.q_out_hole.size(); i++) {
263  Edge edge = to_replace.q_out_hole[i];
264  Vertex succ_v = target(edge);
265  port_t port2 = get_target_port(edge);
266  ebin.insert(edge);
267  Vertex outp = vm[to_insert.get_out(Qubit(i))];
268  add_edge({outp, 0}, {succ_v, port2}, EdgeType::Quantum);
269  dag[outp].op = noop;
270  bin.push_back(outp);
271  }
272  for (unsigned i = 0; i < to_replace.c_in_hole.size(); i++) {
273  Edge edge = to_replace.c_in_hole[i];
274  Vertex pred_v = source(edge);
275  port_t port1 = get_source_port(edge);
276  ebin.insert(edge);
277  Vertex inp = vm[to_insert.get_in(Bit(i))];
278  add_edge({pred_v, port1}, {inp, 0}, EdgeType::Classical);
279  dag[inp].op = noop;
280  bin.push_back(inp);
281  }
282  for (unsigned i = 0; i < to_replace.c_out_hole.size(); i++) {
283  Edge edge = to_replace.c_out_hole[i];
284  Vertex succ_v = target(edge);
285  port_t port2 = get_target_port(edge);
286  ebin.insert(edge);
287  Vertex outp = vm[to_insert.get_out(Bit(i))];
288  add_edge({outp, 0}, {succ_v, port2}, EdgeType::Classical);
289  dag[outp].op = noop;
290  bin.push_back(outp);
291  c_out_map.insert({edge, outp});
292  }
293  for (const Edge& e : to_replace.b_future) {
294  Edge c_out = get_nth_out_edge(source(e), get_source_port(e));
295  Vertex outp = c_out_map[c_out];
296  add_edge({outp, 0}, {target(e), get_target_port(e)}, EdgeType::Boolean);
297  ebin.insert(e);
298  }
299  for (const Edge& e : ebin) {
300  remove_edge(e);
301  }
302  // automatically rewire these canned vertices
303  remove_vertices(bin, GraphRewiring::Yes, VertexDeletion::Yes);
304  remove_vertices(to_replace.verts, GraphRewiring::No, vertex_deletion);
305  add_phase(to_insert.get_phase());
306 }

321 void Circuit::substitute_conditional(
322  Circuit to_insert, const Vertex& to_replace, VertexDeletion vertex_deletion,
323  OpGroupTransfer opgroup_transfer) {
324  Op_ptr op = get_Op_ptr_from_Vertex(to_replace);
325  if (op->get_type() != OpType::Conditional)
326  throw CircuitInvalidity(
327  "substitute_conditional called with an unconditional gate");
328  Subcircuit sub = {
329  get_in_edges_of_type(to_replace, EdgeType::Quantum),
330  get_out_edges_of_type(to_replace, EdgeType::Quantum),
331  get_in_edges_of_type(to_replace, EdgeType::Classical),
332  get_out_edges_of_type(to_replace, EdgeType::Classical),
333  get_out_edges_of_type(to_replace, EdgeType::Boolean),
334  {to_replace}};
335  const Conditional& cond = static_cast<const Conditional&>(*op);
336  unsigned width = cond.get_width();
337  // Conditions are first few args, so increase index of all others
338  bit_map_t rename_map;
339  for (unsigned i = 0; i < to_insert.n_bits(); ++i) {
340  rename_map[Bit(i)] = Bit(i + width);
341  }
342  to_insert.rename_units(rename_map);
343  // Extend subcircuit hole with space for condition bits
344  bit_vector_t cond_bits(width);
345  EdgeVec cond_sources;
346  for (unsigned i = 0; i < width; ++i) {
347  cond_bits[i] = Bit(i);
348  Edge read_in = get_nth_in_edge(to_replace, i);
349  Edge source_write =
350  get_nth_out_edge(source(read_in), get_source_port(read_in));
351  cond_sources.push_back(source_write);
352  }
353  sub.c_in_hole.insert(
354  sub.c_in_hole.begin(), cond_sources.begin(), cond_sources.end());
355  sub.c_out_hole.insert(
356  sub.c_out_hole.begin(), cond_sources.begin(), cond_sources.end());
357  to_insert = to_insert.conditional_circuit(cond_bits, cond.get_value());
358  substitute(to_insert, sub, vertex_deletion, opgroup_transfer);
359 }

370 void Circuit::replace_SWAPs() {
371  VertexList bin;
372  BGL_FORALL_VERTICES(v, dag, DAG) {
373  if (get_Op_ptr_from_Vertex(v)->get_type() == OpType::SWAP) {
374  Vertex swap = v;
375  EdgeVec outs = get_all_out_edges(v);
376  Edge out1 = outs[0];
377  dag[out1].ports.first = 1;
378  Edge out2 = outs[1];
379  dag[out2].ports.first = 0;
380  remove_vertex(swap, GraphRewiring::Yes, VertexDeletion::No);
381  bin.push_back(swap);
382  }
383  }
384  remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
385 }

387 void Circuit::replace_implicit_wire_swap(
388  const Qubit first, const Qubit second, bool using_cx) {
389  Vertex last_v;
390  if (using_cx) {
391  add_op<UnitID>(OpType::CX, {first, second});
392  add_op<UnitID>(OpType::CX, {second, first});
393  last_v = add_op<UnitID>(OpType::CX, {first, second});
394  } else {
395  last_v = add_op<UnitID>(OpType::SWAP, {first, second});
396  }
397  EdgeVec outs = get_all_out_edges(last_v);
398  Edge out1 = outs[0];
399  dag[out1].ports.first = 1;
400  Edge out2 = outs[1];
401  dag[out2].ports.first = 0;
402 }

404 void Circuit::replace_all_implicit_wire_swaps() {
405  qubit_map_t perm = implicit_qubit_permutation();
406  std::set<Qubit> fixed_qubits;
407  // iterate permutation cycles and add swap for every adjacent elements
408  for (const std::pair<const Qubit, Qubit>& pair : perm) {
409  if (fixed_qubits.find(pair.first) != fixed_qubits.end()) {
410  // skip if visited
411  continue;
412  }
413  // start traverse a cycle
414  const Qubit head = pair.first;
415  Qubit current = pair.first;
416  Qubit next = pair.second;
417  while (true) {
418  if (next == head) {
419  // break if reaches the end of the cycle
420  fixed_qubits.insert(current);
421  break;
422  }
423  replace_implicit_wire_swap(current, next, false);
424  fixed_qubits.insert(current);
425  auto it = perm.find(next);
426  TKET_ASSERT(it != perm.end());
427  current = it->first;
428  next = it->second;
429  }
430  }
431 }

433 // helper functions for the dagger and transpose
434 void Circuit::_handle_boundaries(Circuit& circ, vertex_map_t& vmap) const {
435  // Handle boundaries
436  for (const BoundaryElement& el : this->boundary.get<TagID>()) {
437  Vertex new_in;
438  Vertex new_out;
439  if (el.id_.type() == UnitType::Bit) {
440  tket_log()->warn(
441  "The circuit contains classical data for which the dagger/transpose "
442  "might not be defined.");
443  new_in = circ.add_vertex(OpType::ClInput);
444  new_out = circ.add_vertex(OpType::ClOutput);
445  } else {
446  new_in = circ.add_vertex(OpType::Input);
447  new_out = circ.add_vertex(OpType::Output);
448  }
449  Vertex old_in = el.in_;
450  Vertex old_out = el.out_;
451  vmap[old_in] = new_out;
452  vmap[old_out] = new_in;
453  circ.boundary.insert({el.id_, new_in, new_out});
454  }
455 }

457 void Circuit::_handle_interior(
458  Circuit& circ, vertex_map_t& vmap, V_iterator& vi, V_iterator& vend,
459  ReverseType reverse_op) const {
460  // Handle interior
461  for (std::tie(vi, vend) = boost::vertices(this->dag); vi != vend; vi++) {
462  const Op_ptr op = get_Op_ptr_from_Vertex(*vi);
463  OpDesc desc = op->get_desc();
464  if (is_boundary_q_type(desc.type())) {
465  continue;
466  } else if ((desc.is_gate() || desc.is_box()) && !desc.is_oneway()) {
467  Op_ptr op_type_ptr;
468  switch (reverse_op) {
469  case ReverseType::dagger: {
470  op_type_ptr = op->dagger();
471  break;
472  }
473  case ReverseType::transpose: {
474  op_type_ptr = op->transpose();
475  break;
476  }
477  default: {
478  throw std::logic_error(
479  "Error in the definition of the dagger or transpose.");
480  }
481  }
482  Vertex v = circ.add_vertex(op_type_ptr);
483  vmap[*vi] = v;
484  } else {
485  throw CircuitInvalidity(
486  "Cannot dagger or transpose op: " + op->get_name());
487  }
488  }
489 }

491 void Circuit::_handle_edges(
492  Circuit& circ, vertex_map_t& vmap, E_iterator& ei, E_iterator& eend) const {
493  for (std::tie(ei, eend) = boost::edges(this->dag); ei != eend; ei++) {
494  Vertex s = source(*ei);
495  port_t sp = get_source_port(*ei);
496  Vertex t = target(*ei);
497  port_t tp = get_target_port(*ei);
498  circ.add_edge({vmap[t], tp}, {vmap[s], sp}, get_edgetype(*ei));
499  }
500 }

502 // returns Hermitian conjugate of circuit, ie its inverse
503 Circuit Circuit::dagger() const {
504  Circuit c;
505  vertex_map_t vmap;
506  _handle_boundaries(c, vmap);
507 
508  V_iterator vi, vend;
509  ReverseType dagger = ReverseType::dagger;
510  _handle_interior(c, vmap, vi, vend, dagger);
511 
512  E_iterator ei, eend;
513  _handle_edges(c, vmap, ei, eend);
514 
515  c.add_phase(-get_phase());
516  return c;
517 }

519 // returns transpose of circuit

536 bool Circuit::substitute_all(const Circuit& to_insert, const Op_ptr op) {
537  if (!to_insert.is_simple()) throw SimpleOnly();
538  if (op->n_qubits() != to_insert.n_qubits())
539  throw CircuitInvalidity(
540  "Cannot substitute all on mismatching arity between Vertex "
541  "and inserted Circuit");
542  VertexVec to_replace;
543  VertexVec conditional_to_replace;
544  BGL_FORALL_VERTICES(v, dag, DAG) {
545  Op_ptr v_op = get_Op_ptr_from_Vertex(v);
546  if (*v_op == *op)
547  to_replace.push_back(v);
548  else if (v_op->get_type() == OpType::Conditional) {
549  const Conditional& cond = static_cast<const Conditional&>(*v_op);
550  if (*cond.get_op() == *op) conditional_to_replace.push_back(v);
551  }
552  }
553  for (const Vertex& v : to_replace) {
554  substitute(to_insert, v, VertexDeletion::Yes);
555  }
556  for (const Vertex& v : conditional_to_replace) {
557  substitute_conditional(to_insert, v, VertexDeletion::Yes);
558  }
559  return !(to_replace.empty() && conditional_to_replace.empty());
560 }

562 bool Circuit::substitute_named(
563  const Circuit& to_insert, const std::string opname) {
564  if (!to_insert.is_simple()) throw SimpleOnly();
565 
566  // Check that no op group names are in common
567  for (const auto& opgroupsig : to_insert.opgroupsigs) {
568  if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
569  throw CircuitInvalidity("Name collision in replacement circuit");
570  }
571  }
572 
573  // Do nothing if opname not present
574  if (opgroupsigs.find(opname) == opgroupsigs.end()) {
575  return false;
576  }
577 
578  // Check signatures match
579  op_signature_t sig = opgroupsigs[opname];
580  unsigned sig_n_q = std::count(sig.begin(), sig.end(), EdgeType::Quantum);
581  unsigned sig_n_c = std::count(sig.begin(), sig.end(), EdgeType::Classical);
582  unsigned sig_n_b = std::count(sig.begin(), sig.end(), EdgeType::Boolean);
583  if (to_insert.n_qubits() != sig_n_q || to_insert.n_bits() != sig_n_c ||
584  sig_n_b != 0) {
585  throw CircuitInvalidity("Signature mismatch");
586  }

589  BGL_FORALL_VERTICES(v, dag, DAG) {
590  std::optional<std::string> v_opgroup = get_opgroup_from_Vertex(v);
591  if (v_opgroup && v_opgroup.value() == opname) {
592  to_replace.push_back(v);
593  }
594  }

637 Circuit Circuit::conditional_circuit(
638  const bit_vector_t& bits, unsigned value) const {
639  if (has_implicit_wireswaps()) {
640  throw CircuitInvalidity("Cannot add conditions to an implicit wireswap");
641  }
642  Circuit cond_circ(all_qubits(), all_bits());
643  for (const Bit& b : bits) {
644  if (contains_unit(b)) {
645  Vertex in = get_in(b);
646  Vertex out = get_out(b);
647  if (get_successors_of_type(in, EdgeType::Classical).front() != out) {
648  throw CircuitInvalidity(
649  "Cannot add condition. Circuit has non-trivial "
650  "actions on bit " +
651  b.repr());
652  }
653  } else {
654  cond_circ.add_bit(b);
655  }
656  }
657  unsigned width = bits.size();
658  for (const Command& com : *this) {
659  const Op_ptr op = com.get_op_ptr();
660  Op_ptr cond_op = std::make_shared<Conditional>(op, width, value);
661  unit_vector_t args = com.get_args();
662  args.insert(args.begin(), bits.begin(), bits.end());
663  cond_circ.add_op(cond_op, args);
664  }
665  // Replace global phase with conditional phase:
666  Expr alpha = get_phase();
667  if (!equiv_0(alpha)) {
668  Op_ptr op = get_op_ptr(OpType::Phase, {alpha});
669  Op_ptr cond_op = std::make_shared<Conditional>(op, width, value);
670  cond_circ.add_op(cond_op, bits);
671  }
672  return cond_circ;
673 }

675 bool Circuit::substitute_box_vertex(
676  Vertex& vert, VertexDeletion vertex_deletion) {
677  Op_ptr op = get_Op_ptr_from_Vertex(vert);
678  bool conditional = op->get_type() == OpType::Conditional;
679  if (conditional) {
680  const Conditional& cond = static_cast<const Conditional&>(*op);
681  op = cond.get_op();
682  }
683  if (!op->get_desc().is_box()) return false;
684  if (op->get_type() == OpType::ClassicalExpBox) return false;
685  const Box& b = static_cast<const Box&>(*op);
686  Circuit replacement = *b.to_circuit();

688  substitute_conditional(
689  replacement, vert, vertex_deletion, OpGroupTransfer::Merge);
690  } else {
691  substitute(replacement, vert, vertex_deletion, OpGroupTransfer::Merge);
692  }
693  return true;
694 }

714 std::map<Bit, bool> Circuit::classical_eval(
715  const std::map<Bit, bool>& values) const {
716  std::map<Bit, bool> v(values);
717  for (CommandIterator it = begin(); it != end(); ++it) {
718  Op_ptr op = it->get_op_ptr();
719  OpType optype = op->get_type();
720  if (!is_classical_type(optype)) {
721  throw CircuitInvalidity("Non-classical operation");
722  }
723  std::shared_ptr<const ClassicalEvalOp> cop =
724  std::dynamic_pointer_cast<const ClassicalEvalOp>(op);
725  unit_vector_t args = it->get_args();
726  unsigned n_args = args.size();
727  switch (optype) {
728  case OpType::ClassicalTransform: {
729  std::vector<bool> input(n_args);
730  for (unsigned i = 0; i < n_args; i++) {
731  input[i] = v[Bit(args[i])];
732  }
733  std::vector<bool> output = cop->eval(input);
734  TKET_ASSERT(output.size() == n_args);
735  for (unsigned i = 0; i < n_args; i++) {
736  v[Bit(args[i])] = output[i];
737  }
738  break;
739  }
740  case OpType::SetBits: {
741  std::vector<bool> output = cop->eval({});
742  TKET_ASSERT(output.size() == n_args);
743  for (unsigned i = 0; i < n_args; i++) {
744  v[Bit(args[i])] = output[i];
745  }
746  break;
747  }
748  default:
749  throw CircuitInvalidity("Unexpected operation in circuit");
750  }
751  }
752  return v;
753 }
