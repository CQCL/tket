33vertex_map_t Circuit::copy_graph(
34 const Circuit& c2, BoundaryMerge boundary_merge,
35 OpGroupTransfer opgroup_transfer) {
36 switch (opgroup_transfer) {
37 case OpGroupTransfer::Preserve:
38 // Fail if any collisions.
39 for (const auto& opgroupsig : c2.opgroupsigs) {
40 if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
41 throw CircuitInvalidity("Name collision in inserted circuit");
42 }
43 }
44 // Add inserted opgroups to circuit.
45 opgroupsigs.insert(c2.opgroupsigs.begin(), c2.opgroupsigs.end());
46 break;
47 case OpGroupTransfer::Disallow:
48 // Fail if any opgroups.
49 if (!c2.opgroupsigs.empty()) {
50 throw CircuitInvalidity("Named op groups in inserted circuit");
51 }
52 break;
53 case OpGroupTransfer::Merge:
54 // Fail if any mismatched signatures
55 for (const auto& opgroupsig : c2.opgroupsigs) {
56 if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
57 if (opgroupsigs[opgroupsig.first] != opgroupsig.second) {
58 throw CircuitInvalidity(
59 "Name signature mismatch in inserted circuit");
60 }
61 }
62 }
63 // Add inserted opgroups to circuit.
64 opgroupsigs.insert(c2.opgroupsigs.begin(), c2.opgroupsigs.end());
65 break;
66 default:
67 TKET_ASSERT(opgroup_transfer == OpGroupTransfer::Remove);
68 // Ignore inserted opgroups
69 break;
70 }
71
72 vertex_map_t isomap;
73 if (&c2 == this) {
74 throw Unsupported(
75 "Circuit Cannot currently copy itself using this method. Use * "
76 "instead\n");
77 }
78 BGL_FORALL_VERTICES(v, c2.dag, DAG) {
79 Vertex v0 = boost::add_vertex(this->dag);
80 this->dag[v0].op = c2.get_Op_ptr_from_Vertex(v);
81 if (opgroup_transfer == OpGroupTransfer::Preserve ||
82 opgroup_transfer == OpGroupTransfer::Merge) {
83 this->dag[v0].opgroup = c2.get_opgroup_from_Vertex(v);
84 }
85 isomap.insert({v, v0});
86 }
87 BGL_FORALL_VERTICES(v, c2.dag, DAG) {
88 EdgeVec edges = c2.get_in_edges(v);
89 Vertex target_v = isomap.find(v)->second;
90 for (EdgeVec::iterator e1 = edges.begin(); e1 != edges.end(); ++e1) {
91 Vertex old_source_v = c2.source(*e1);
92 Vertex source_v = isomap.find(old_source_v)->second;
93 add_edge(
94 {source_v, get_source_port(*e1)}, {target_v, get_target_port(*e1)},
95 c2.dag[*e1].type);
96 }
97 }
98
99 if (boundary_merge == BoundaryMerge::Yes) {
100 for (const BoundaryElement& el : c2.boundary.get<TagID>()) {
101 std::string reg_name = el.id_.reg_name();
102 register_info_t reg_type = el.reg_info();
103 opt_reg_info_t reg_found = get_reg_info(reg_name);
104 if (reg_found) {
105 if (reg_found.value() != reg_type) {
106 throw Unsupported(
107 "Cannot merge circuits with different types for "
108 "register with name: " +
109 reg_name);
110 }
111 boundary_t::iterator unit_found = boundary.get<TagID>().find(el.id_);
112 if (unit_found != boundary.get<TagID>().end()) {
113 throw Unsupported(
114 "Cannot merge circuits as both contain unit: " + el.id_.repr());
115 }
116 }
117 Vertex new_in = isomap[el.in_];
118 Vertex new_out = isomap[el.out_];
119 boundary.insert({el.id_, new_in, new_out});
120 }
121 }
122 return isomap;
123}

125// given two circuits, adds second circuit to first circuit object in parallel
126Circuit operator*(const Circuit& c1, const Circuit& c2) {
127 // preliminary method to add circuit objects together
128 Circuit new_circ;
129 new_circ.copy_graph(c1);
130 new_circ.copy_graph(c2);
131 new_circ.add_phase(c1.get_phase() + c2.get_phase());
132 return new_circ;
133}

137// qm is from the units on the second (appended) circuit to the units on the
138// first (this) circuit
139void Circuit::append_with_map(const Circuit& c2, const unit_map_t& qm) {
140 Circuit copy = c2;
141 copy.rename_units(qm);
142
143 if ((_number_of_wasm_wires > 0) && (copy._number_of_wasm_wires > 0)) {
144 if (copy.get_wasm_file_uid() != get_wasm_file_uid()) {
145 throw Unsupported(
146 "Cannot append circuits with different wasm uids: " +
147 get_wasm_file_uid().emplace("(none)") + " and " +
148 copy.get_wasm_file_uid().emplace("(none)"));
149 }
150 }
151
152 copy.add_wasm_register(_number_of_wasm_wires);
153 add_wasm_register(copy._number_of_wasm_wires);
154 copy.add_rng_register(_number_of_rng_wires);
155 add_rng_register(copy._number_of_rng_wires);
156
157 // Check what we need to do at the joins:
158 // Output --- Input ==> -------------
159 // Output --- Create ==> --- Reset ---
160 // Discard --- Input ==> [not allowed]
161 // Discard --- Create ==> --- Reset ---
162 qubit_vector_t qbs = copy.all_qubits();
163 std::set<Qubit> qbs_set(qbs.begin(), qbs.end());
164 std::set<Qubit> reset_qbs;
165 for (const auto& qb : all_qubits()) {
166 if (qbs_set.find(qb) != qbs_set.end()) {
167 if (copy.is_created(qb)) {
168 reset_qbs.insert(qb);
169 } else if (is_discarded(qb)) {
170 throw CircuitInvalidity("Cannot append input qubit to discarded qubit");
171 }
172 }
173 }
174
175 // Copy c2 into c1 but do not merge boundaries
176 vertex_map_t vm = copy_graph(copy, BoundaryMerge::No);
177 const Op_ptr noop = get_op_ptr(OpType::noop);
178
179 // Connect each matching qubit and bit, merging remainder
180 for (const BoundaryElement& el : copy.boundary.get<TagID>()) {
181 std::string reg_name = el.reg_name();
182 register_info_t reg_type = el.reg_info();
183 opt_reg_info_t reg_found = get_reg_info(reg_name);
184 if (reg_found) {
185 if (reg_found.value() != reg_type) {
186 throw Unsupported(
187 "Cannot append circuits with different types for "
188 "register with name: " +
189 reg_name);
190 }
191 boundary_t::iterator unit_found = boundary.get<TagID>().find(el.id_);
192 if (unit_found != boundary.get<TagID>().end()) {
193 Vertex out = unit_found->out_;
194 Vertex in = vm[el.in_];
195 // Update map
196 BoundaryElement new_elem = *unit_found;
197 new_elem.out_ = vm[el.out_];
198 boundary.replace(unit_found, new_elem);
199 // Tie together
200 if (reg_type.first == UnitType::Qubit)
201 add_edge({out, 0}, {in, 0}, EdgeType::Quantum);
202 else
203 add_edge({out, 0}, {in, 0}, EdgeType::Classical);
204 dag[out].op = noop;
205 dag[in].op = noop;
206 remove_vertex(out, GraphRewiring::Yes, VertexDeletion::Yes);
207 if (el.type() != UnitType::Qubit ||
208 reset_qbs.find(Qubit(el.id_)) == reset_qbs.end()) {
209 remove_vertex(in, GraphRewiring::Yes, VertexDeletion::Yes);
210 } else {
211 dag[in].op = std::make_shared<const Gate>(OpType::Reset);
212 }
213 } else {
214 Vertex new_in = vm[el.in_];
215 Vertex new_out = vm[el.out_];
216 boundary.insert({el.id_, new_in, new_out});
217 }
218 } else {
219 Vertex new_in = vm[el.in_];
220 Vertex new_out = vm[el.out_];
221 boundary.insert({el.id_, new_in, new_out});
222 }
223 }
224 add_phase(c2.get_phase());
225}

227void Circuit::append_qubits(
228 const Circuit& c2, const std::vector<unsigned>& qubits,
229 const std::vector<unsigned>& bits) {
230 unit_map_t qm;
231 for (unsigned i = 0; i < qubits.size(); i++) {
232 qm.insert({Qubit(i), Qubit(qubits[i])});
233 }
234 for (unsigned i = 0; i < bits.size(); i++) {
235 qm.insert({Bit(i), Bit(bits[i])});
236 }
237 append_with_map(c2, qm);
238}

249// Important substitute method. Requires knowledge of the boundary to insert
250// into, and the vertices inside which are to be removed when substitution is
251// performed. Gives the option to isolate the removed vertices but not delete
252// them.
253void Circuit::substitute(
254 const Circuit& to_insert, const Subcircuit& to_replace,
255 VertexDeletion vertex_deletion, OpGroupTransfer opgroup_transfer) {
256 if (!to_insert.is_simple()) throw SimpleOnly();
257 if (to_insert.n_qubits() + to_insert.n_bits() +
258 to_insert._number_of_wasm_wires + to_insert._number_of_rng_wires !=
259 to_replace.in_hole.size())
260 throw CircuitInvalidity("Subcircuit boundary mismatch to hole");
261
262 vertex_map_t vm = copy_graph(to_insert, BoundaryMerge::No, opgroup_transfer);
263 VertexList bin;
264 EdgeSet ebin; // Needs to be a set since subcircuit to replace could be
265 // trivial, essentially rewiring on a cut
266 std::map<Edge, Vertex> c_out_map;
267
268 std::set<Qubit> reset_qbs;
269 for (const auto& qb : to_insert.all_qubits()) {
270 if (to_insert.is_created(qb)) {
271 reset_qbs.insert(qb);
272 } else if (to_insert.is_discarded(qb)) {
273 throw CircuitInvalidity("Cannot substitute discarded qubit");
274 }
275 }
276
277 const Op_ptr noop = get_op_ptr(OpType::noop);
278 const Op_ptr reset = get_op_ptr(OpType::Reset);
279 unsigned qubit_id = 0;
280 unsigned bit_id = 0;
281 unsigned wasm_id = 0;
282 unsigned rng_id = 0;
283 for (unsigned i = 0; i < to_replace.in_hole.size(); ++i) {
284 Edge in_edge = to_replace.in_hole.at(i);
285 std::optional<Edge> out_edge = to_replace.out_hole.at(i);
286 Vertex in_pred = source(in_edge);
287 port_t in_port = get_source_port(in_edge);
288 ebin.insert(in_edge);
289 switch (get_edgetype(in_edge)) {
290 case EdgeType::Quantum: {
291 TKET_ASSERT(out_edge.has_value());
292 TKET_ASSERT(get_edgetype(*out_edge) == EdgeType::Quantum);
293 Vertex inp = vm[to_insert.get_in(Qubit(qubit_id))];
294 add_edge({in_pred, in_port}, {inp, 0}, EdgeType::Quantum);
295 if (reset_qbs.contains(Qubit(qubit_id))) {
296 set_vertex_Op_ptr(inp, reset);
297 } else {
298 set_vertex_Op_ptr(inp, noop);
299 bin.push_back(inp);
300 }
301 Vertex out_succ = target(*out_edge);
302 port_t out_port = get_target_port(*out_edge);
303 ebin.insert(*out_edge);
304 Vertex outp = vm[to_insert.get_out(Qubit(qubit_id))];
305 add_edge({outp, 0}, {out_succ, out_port}, EdgeType::Quantum);
306 set_vertex_Op_ptr(outp, noop);
307 bin.push_back(outp);
308 ++qubit_id;
309 break;
310 }
311 case EdgeType::Classical: {
312 TKET_ASSERT(out_edge.has_value());
313 TKET_ASSERT(get_edgetype(*out_edge) == EdgeType::Classical);
314 Vertex inp = vm[to_insert.get_in(Bit(bit_id))];
315 add_edge({in_pred, in_port}, {inp, 0}, EdgeType::Classical);
316 set_vertex_Op_ptr(inp, noop);
317 bin.push_back(inp);
318 Vertex out_succ = target(*out_edge);
319 port_t out_port = get_target_port(*out_edge);
320 ebin.insert(*out_edge);
321 Vertex outp = vm[to_insert.get_out(Bit(bit_id))];
322 add_edge({outp, 0}, {out_succ, out_port}, EdgeType::Classical);
323 set_vertex_Op_ptr(outp, noop);
324 bin.push_back(outp);
325 c_out_map.insert({*out_edge, outp});
326 ++bit_id;
327 break;
328 }
329 case EdgeType::Boolean: {
330 TKET_ASSERT(!out_edge.has_value());
331 Vertex inp = vm[to_insert.get_in(Bit(bit_id))];
332 Vertex outp = vm[to_insert.get_out(Bit(bit_id))];
333 if (to_insert.get_successors_of_type(inp, EdgeType::Classical).at(0) !=
334 outp)
335 throw CircuitInvalidity(
336 "Subcircuit replacement writes to a Bit from a read-only input "
337 "to the hole");
338 for (const Edge& new_edge :
339 get_out_edges_of_type(inp, EdgeType::Boolean)) {
340 add_edge(
341 {in_pred, in_port}, {target(new_edge), get_target_port(new_edge)},
342 EdgeType::Boolean);
343 }
344 set_vertex_Op_ptr(inp, noop);
345 set_vertex_Op_ptr(outp, noop);
346 bin.push_back(inp);
347 bin.push_back(outp);
348 ++bit_id;
349 break;
350 }
351 case EdgeType::WASM: {
352 TKET_ASSERT(out_edge.has_value());
353 TKET_ASSERT(get_edgetype(*out_edge) == EdgeType::WASM);
354 Vertex inp = vm[to_insert.get_in(WasmState(wasm_id))];
355 add_edge({in_pred, in_port}, {inp, 0}, EdgeType::WASM);
356 set_vertex_Op_ptr(inp, noop);
357 bin.push_back(inp);
358 Vertex out_succ = target(*out_edge);
359 port_t out_port = get_target_port(*out_edge);
360 ebin.insert(*out_edge);
361 Vertex outp = vm[to_insert.get_out(WasmState(wasm_id))];
362 add_edge({outp, 0}, {out_succ, out_port}, EdgeType::WASM);
363 set_vertex_Op_ptr(outp, noop);
364 bin.push_back(outp);
365 ++wasm_id;
366 break;
367 }
368 case EdgeType::RNG: {
369 TKET_ASSERT(out_edge.has_value());
370 TKET_ASSERT(get_edgetype(*out_edge) == EdgeType::RNG);
371 Vertex inp = vm[to_insert.get_in(RngState(rng_id))];
372 add_edge({in_pred, in_port}, {inp, 0}, EdgeType::RNG);
373 set_vertex_Op_ptr(inp, noop);
374 bin.push_back(inp);
375 Vertex out_succ = target(*out_edge);
376 port_t out_port = get_target_port(*out_edge);
377 ebin.insert(*out_edge);
378 Vertex outp = vm[to_insert.get_out(RngState(rng_id))];
379 add_edge({outp, 0}, {out_succ, out_port}, EdgeType::RNG);
380 set_vertex_Op_ptr(outp, noop);
381 bin.push_back(outp);
382 ++rng_id;
383 break;
384 }
385 }
386 }
387 for (const Edge& e : to_replace.b_future) {
388 Edge c_out = get_nth_out_edge(source(e), get_source_port(e));
389 Vertex outp = c_out_map.at(c_out);
390 add_edge({outp, 0}, {target(e), get_target_port(e)}, EdgeType::Boolean);
391 ebin.insert(e);
392 }
393 for (const Edge& e : ebin) {
394 remove_edge(e);
395 }
396 // automatically rewire these canned vertices
397 remove_vertices(bin, GraphRewiring::Yes, VertexDeletion::Yes);
398 remove_vertices(to_replace.verts, GraphRewiring::No, vertex_deletion);
399 add_phase(to_insert.get_phase());
400}

409// Helper function for substitute_conditional which recursively unpacks a
410// Conditional until we reach something that isn't a Conditional; we then wrap
411// base_circ with each layer of the conditional working back up
412Circuit recursive_conditional_circuit(
413 const Op_ptr& op, const Circuit& base_circ) {
414 if (op->get_type() != OpType::Conditional) return base_circ;
415 const Conditional& cond = static_cast<const Conditional&>(*op);
416 Op_ptr inner_op = cond.get_op();
417 Circuit inner_circ = recursive_conditional_circuit(inner_op, base_circ);
418 unsigned width = cond.get_width();
419 bit_map_t rename_map;
420 for (unsigned i = 0; i < inner_circ.n_bits(); ++i)
421 rename_map[Bit(i)] = Bit(i + width);
422 inner_circ.rename_units(rename_map);
423 bit_vector_t cond_bits(width);
424 for (unsigned i = 0; i < width; ++i) cond_bits[i] = Bit(i);
425 return inner_circ.conditional_circuit(cond_bits, cond.get_value());
426}

428void Circuit::substitute_conditional(
429 Circuit to_insert, const Vertex& to_replace, VertexDeletion vertex_deletion,
430 OpGroupTransfer opgroup_transfer) {
431 Op_ptr op = get_Op_ptr_from_Vertex(to_replace);
432 if (op->get_type() != OpType::Conditional)
433 throw CircuitInvalidity(
434 "substitute_conditional called with an unconditional gate");
435 Subcircuit sub = singleton_subcircuit(to_replace);
436 to_insert = recursive_conditional_circuit(op, to_insert);
437 substitute(to_insert, sub, vertex_deletion, opgroup_transfer);
438}

449bool Circuit::replace_SWAPs(bool replace_tk2_equivalents) {
450 VertexList bin;
451 bool changed = false;
452 double total_phase = 0.;
453 BGL_FORALL_VERTICES(v, dag, DAG) {
454 Op_ptr op = get_Op_ptr_from_Vertex(v);
455 OpType type = op->get_type();
456 if (type != OpType::SWAP &&
457 (type != OpType::TK2 || !replace_tk2_equivalents)) {
458 continue;
459 }
460 if (type == OpType::TK2) {
461 std::vector<Expr> params = op->get_params();
462 std::optional<double> phase =
463 is_TK2_SWAP(params[0], params[1], params[2]);
464 if (phase == std::nullopt) {
465 continue;
466 }
467 total_phase += *phase;
468 }
469 Vertex swap = v;
470 EdgeVec outs = get_all_out_edges(v);
471 Edge out1 = outs[0];
472 dag[out1].ports.first = 1;
473 Edge out2 = outs[1];
474 dag[out2].ports.first = 0;
475 remove_vertex(swap, GraphRewiring::Yes, VertexDeletion::No);
476 bin.push_back(swap);
477 changed = true;
478 }
479 remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
480 add_phase(total_phase);
481 return changed;
482}

484void Circuit::replace_implicit_wire_swap(
485 const Qubit first, const Qubit second, bool using_cx) {
486 Vertex last_v;
487 if (using_cx) {
488 add_op<UnitID>(OpType::CX, {first, second});
489 add_op<UnitID>(OpType::CX, {second, first});
490 last_v = add_op<UnitID>(OpType::CX, {first, second});
491 } else {
492 last_v = add_op<UnitID>(OpType::SWAP, {first, second});
493 }
494 EdgeVec outs = get_all_out_edges(last_v);
495 Edge out1 = outs[0];
496 dag[out1].ports.first = 1;
497 Edge out2 = outs[1];
498 dag[out2].ports.first = 0;
499}

501void Circuit::replace_all_implicit_wire_swaps() {
502 qubit_map_t perm = implicit_qubit_permutation();
503 std::set<Qubit> fixed_qubits;
504 // iterate permutation cycles and add swap for every adjacent elements
505 for (const std::pair<const Qubit, Qubit>& pair : perm) {
506 if (fixed_qubits.find(pair.first) != fixed_qubits.end()) {
507 // skip if visited
508 continue;
509 }
510 // start traverse a cycle
511 const Qubit head = pair.first;
512 Qubit current = pair.first;
513 Qubit next = pair.second;
514 while (true) {
515 if (next == head) {
516 // break if reaches the end of the cycle
517 fixed_qubits.insert(current);
518 break;
519 }
520 replace_implicit_wire_swap(current, next, false);
521 fixed_qubits.insert(current);
522 auto it = perm.find(next);
523 TKET_ASSERT(it != perm.end());
524 current = it->first;
525 next = it->second;
526 }
527 }

530// helper functions for the dagger and transpose
531void Circuit::_handle_boundaries(Circuit& circ, vertex_map_t& vmap) const {
532 // Handle boundaries
533 for (const BoundaryElement& el : this->boundary.get<TagID>()) {
534 Vertex new_in;
535 Vertex new_out;
536 if (el.id_.type() == UnitType::Bit) {
537 tket_log()->warn(
538 "The circuit contains classical data for which the dagger/transpose "
539 "might not be defined.");
540 new_in = circ.add_vertex(OpType::ClInput);
541 new_out = circ.add_vertex(OpType::ClOutput);
542 } else {
543 new_in = circ.add_vertex(OpType::Input);
544 new_out = circ.add_vertex(OpType::Output);
545 }
546 Vertex old_in = el.in_;
547 Vertex old_out = el.out_;
548 vmap[old_in] = new_out;
549 vmap[old_out] = new_in;
550 circ.boundary.insert({el.id_, new_in, new_out});
551 }
552}

554void Circuit::_handle_interior(
555 Circuit& circ, vertex_map_t& vmap, V_iterator& vi, V_iterator& vend,
556 ReverseType reverse_op) const {
557 // Handle interior
558 for (std::tie(vi, vend) = boost::vertices(this->dag); vi != vend; vi++) {
559 const Op_ptr op = get_Op_ptr_from_Vertex(*vi);
560 OpDesc desc = op->get_desc();
561 if (is_boundary_q_type(desc.type())) {
562 continue;
563 } else if ((desc.is_gate() || desc.is_box()) && !desc.is_oneway()) {
564 Op_ptr op_type_ptr;
565 switch (reverse_op) {
566 case ReverseType::dagger: {
567 op_type_ptr = op->dagger();
568 break;
569 }
570 case ReverseType::transpose: {
571 op_type_ptr = op->transpose();
572 break;
573 }
574 default: {
575 throw std::logic_error(
576 "Error in the definition of the dagger or transpose.");
577 }
578 }
579 Vertex v = circ.add_vertex(op_type_ptr);
580 vmap[*vi] = v;
581 } else if (desc.is_barrier()) {
582 Vertex v = circ.add_vertex(op);
583 vmap[*vi] = v;
584 } else {
585 throw CircuitInvalidity(
586 "Cannot dagger or transpose op: " + op->get_name());
587 }
588 }

591void Circuit::_handle_edges(
592 Circuit& circ, vertex_map_t& vmap, E_iterator& ei, E_iterator& eend) const {
593 for (std::tie(ei, eend) = boost::edges(this->dag); ei != eend; ei++) {
594 Vertex s = source(*ei);
595 port_t sp = get_source_port(*ei);
596 Vertex t = target(*ei);
597 port_t tp = get_target_port(*ei);

636bool Circuit::substitute_all(const Circuit& to_insert, const Op_ptr op) {
637 if (!to_insert.is_simple()) throw SimpleOnly();
638 if (op->n_qubits() != to_insert.n_qubits())
639 throw CircuitInvalidity(
640 "Cannot substitute all on mismatching arity between Vertex "
641 "and inserted Circuit");
642 VertexVec to_replace;
643 VertexVec conditional_to_replace;
644 BGL_FORALL_VERTICES(v, dag, DAG) {
645 Op_ptr v_op = get_Op_ptr_from_Vertex(v);
646 if (*v_op == *op)
647 to_replace.push_back(v);
648 else if (v_op->get_type() == OpType::Conditional) {
649 while (v_op->get_type() == OpType::Conditional) {
650 v_op = static_cast<const Conditional&>(*v_op).get_op();
651 }
652 if (*v_op == *op) conditional_to_replace.push_back(v);
653 }
654 }
655 for (const Vertex& v : to_replace) {
656 substitute(to_insert, v, VertexDeletion::Yes);
657 }
658 for (const Vertex& v : conditional_to_replace) {
659 substitute_conditional(to_insert, v, VertexDeletion::Yes);
660 }
661 return !(to_replace.empty() && conditional_to_replace.empty());
662}

664bool Circuit::substitute_named(
665 const Circuit& to_insert, const std::string opname) {
666 if (!to_insert.is_simple()) throw SimpleOnly();
667
668 // Check that no op group names are in common
669 for (const auto& opgroupsig : to_insert.opgroupsigs) {
670 if (opgroupsigs.find(opgroupsig.first) != opgroupsigs.end()) {
671 throw CircuitInvalidity("Name collision in replacement circuit");
672 }
673 }
674
675 // Do nothing if opname not present
676 if (opgroupsigs.find(opname) == opgroupsigs.end()) {
677 return false;
678 }
679
680 // Check signatures match
681 op_signature_t sig = opgroupsigs[opname];
682 unsigned sig_n_q = std::count(sig.begin(), sig.end(), EdgeType::Quantum);
683 unsigned sig_n_c = std::count(sig.begin(), sig.end(), EdgeType::Classical);
684 unsigned sig_n_b = std::count(sig.begin(), sig.end(), EdgeType::Boolean);
685 if (to_insert.n_qubits() != sig_n_q || to_insert.n_bits() != sig_n_c ||
686 sig_n_b != 0) {
687 throw CircuitInvalidity("Signature mismatch");
688 }
689
690 VertexVec to_replace;
691 BGL_FORALL_VERTICES(v, dag, DAG) {
692 std::optional<std::string> v_opgroup = get_opgroup_from_Vertex(v);
693 if (v_opgroup && v_opgroup.value() == opname) {
694 to_replace.push_back(v);
695 }
696 }
697
698 for (const Vertex& v : to_replace) {
699 substitute(to_insert, v, VertexDeletion::Yes, OpGroupTransfer::Merge);
700 }
701
702 return !to_replace.empty();
703}

705bool Circuit::substitute_named(Op_ptr to_insert, const std::string opname) {
706 // Do nothing if opname not present
707 if (opgroupsigs.find(opname) == opgroupsigs.end()) {
708 return false;
709 }
710
711 // Check signatures match
712 op_signature_t sig = opgroupsigs[opname];
713 if (to_insert->get_signature() != sig) {
714 throw CircuitInvalidity("Signature mismatch");
715 }
716
717 VertexVec to_replace;
718 BGL_FORALL_VERTICES(v, dag, DAG) {
719 std::optional<std::string> v_opgroup = get_opgroup_from_Vertex(v);
720 if (v_opgroup && v_opgroup.value() == opname) {
721 to_replace.push_back(v);
722 }
723 }
724
725 unsigned sig_n_q = std::count(sig.begin(), sig.end(), EdgeType::Quantum);
726 unsigned sig_n_c = std::count(sig.begin(), sig.end(), EdgeType::Classical);
727 Circuit c(sig_n_q, sig_n_c);
728 unit_vector_t args(sig_n_q + sig_n_c);
729 for (unsigned i = 0; i < sig_n_q; i++) args[i] = Qubit(i);
730 for (unsigned i = 0; i < sig_n_c; i++) args[sig_n_q + i] = Bit(i);
731 c.add_op(to_insert, args, opname);
732 for (const Vertex& v : to_replace) {
733 substitute(c, v, VertexDeletion::Yes, OpGroupTransfer::Merge);
734 }
735
736 return !to_replace.empty();
737}

739Circuit Circuit::conditional_circuit(
740 const bit_vector_t& bits, unsigned value) const {
741 if (has_implicit_wireswaps()) {
742 throw CircuitInvalidity("Cannot add conditions to an implicit wireswap");
743 }
744 Circuit cond_circ(all_qubits(), all_bits());
745 for (const Bit& b : bits) {
746 if (contains_unit(b)) {
747 Vertex in = get_in(b);
748 Vertex out = get_out(b);
749 if (get_successors_of_type(in, EdgeType::Classical).front() != out) {
750 throw CircuitInvalidity(
751 "Cannot add condition. Circuit has non-trivial "
752 "actions on bit " +
753 b.repr());
754 }
755 } else {
756 cond_circ.add_bit(b);
757 }
758 }
759 unsigned width = bits.size();
760 for (const Command& com : *this) {
761 const Op_ptr op = com.get_op_ptr();
762 Op_ptr cond_op = std::make_shared<Conditional>(op, width, value);
763 unit_vector_t args = com.get_args();
764 args.insert(args.begin(), bits.begin(), bits.end());
765 cond_circ.add_op(cond_op, args);
766 }
767 // Replace global phase with conditional phase:
768 Expr alpha = get_phase();
769 if (!equiv_0(alpha)) {
770 Op_ptr op = get_op_ptr(OpType::Phase, {alpha});
771 Op_ptr cond_op = std::make_shared<Conditional>(op, width, value);
772 cond_circ.add_op(cond_op, bits);
773 }
774 return cond_circ;
775}

777bool Circuit::substitute_box_vertex(
778 Vertex& vert, VertexDeletion vertex_deletion,
779 const std::unordered_set<OpType>& excluded_types,
780 const std::unordered_set<std::string>& excluded_opgroups) {
781 Op_ptr op = get_Op_ptr_from_Vertex(vert);
782 bool conditional = false;
783 while (op->get_type() == OpType::Conditional) {
784 op = static_cast<const Conditional&>(*op).get_op();
785 conditional = true;
786 }
787 if (!op->get_desc().is_box()) return false;
788 const Box& b = static_cast<const Box&>(*op);
789 Circuit replacement = *b.to_circuit();
790 replacement.decompose_boxes_recursively(excluded_types, excluded_opgroups);
791 replacement.flatten_registers();
792 if (conditional) {
793 substitute_conditional(
794 replacement, vert, vertex_deletion, OpGroupTransfer::Merge);
795 } else {
796 substitute(replacement, vert, vertex_deletion, OpGroupTransfer::Merge);
797 }
798 return true;
799}
800
801bool Circuit::decompose_boxes_recursively(
802 const std::unordered_set<OpType>& excluded_types,
803 const std::unordered_set<std::string>& excluded_opgroups,
804 const std::optional<std::unordered_set<OpType>>& included_types,
805 const std::optional<std::unordered_set<std::string>>& included_opgroups) {
806 bool success = false;
807 VertexList bin;
808 BGL_FORALL_VERTICES(v, dag, DAG) {
809 OpType ot = get_OpType_from_Vertex(v);
810 if (excluded_types.contains(ot)) continue;
811 if (included_types && !included_types->contains(ot)) continue;
812 std::optional<std::string> v_opgroup = get_opgroup_from_Vertex(v);
813 if (v_opgroup && excluded_opgroups.contains(v_opgroup.value())) continue;
814 if (included_opgroups &&
815 (!v_opgroup || !included_opgroups->contains(v_opgroup.value())))
816 continue;
817 if (substitute_box_vertex(
818 v, VertexDeletion::No, excluded_types, excluded_opgroups)) {
819 bin.push_back(v);
820 success = true;
821 }
822 }
823 remove_vertices(bin, GraphRewiring::No, VertexDeletion::Yes);
824 return success;
825}

827std::map<Bit, bool> Circuit::classical_eval(
828 const std::map<Bit, bool>& values) const {
829 std::map<Bit, bool> v(values);
830 for (CommandIterator it = begin(); it != end(); ++it) {
831 Op_ptr op = it->get_op_ptr();
832 OpType optype = op->get_type();
833 if (!is_classical_type(optype)) {
834 throw CircuitInvalidity("Non-classical operation");
835 }
836 std::shared_ptr<const ClassicalEvalOp> cop =
837 std::dynamic_pointer_cast<const ClassicalEvalOp>(op);
838 unit_vector_t args = it->get_args();
839 unsigned n_args = args.size();
840 switch (optype) {
841 case OpType::ClassicalTransform: {
842 std::vector<bool> input(n_args);
843 for (unsigned i = 0; i < n_args; i++) {
844 input[i] = v[Bit(args[i])];
845 }
846 std::vector<bool> output = cop->eval(input);
847 TKET_ASSERT(output.size() == n_args);
848 for (unsigned i = 0; i < n_args; i++) {
849 v[Bit(args[i])] = output[i];
850 }
851 break;
852 }
853 case OpType::SetBits: {
854 std::vector<bool> output = cop->eval({});
855 TKET_ASSERT(output.size() == n_args);
856 for (unsigned i = 0; i < n_args; i++) {
857 v[Bit(args[i])] = output[i];
858 }
859 break;
860 }
861 default:
862 throw CircuitInvalidity("Unexpected operation in circuit");
863 }
864 }
865 return v;
866}
