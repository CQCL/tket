190 std::map<Bit, unsigned> Circuit::bit_readout() const {
191  std::map<Bit, unsigned> res;
192 
193  // Order bits to generate indices
194  bit_vector_t all_bs = all_bits();
195  std::sort(all_bs.begin(), all_bs.end());

215 std::map<Qubit, Bit> Circuit::qubit_to_bit_map() const {
216  std::map<Qubit, Bit> res;
217  for (auto [it, end] = boundary.get<TagType>().equal_range(UnitType::Qubit);
218  it != end; it++) {
219  Vertex q_out = it->out_;
220  Edge last_gate_out_edge = get_nth_in_edge(q_out, 0);
221  Vertex last_gate = source(last_gate_out_edge);
222  // ignore barriers
223  while (get_OpType_from_Vertex(last_gate) == OpType::Barrier) {
224  std::tie(last_gate, last_gate_out_edge) =
225  get_prev_pair(last_gate, last_gate_out_edge);
226  }
227  if (get_OpType_from_Vertex(last_gate) == OpType::Measure) {
228  Edge possible_c_out_in_edge = get_nth_out_edge(last_gate, 1);
229  Vertex possible_c_out = target(possible_c_out_in_edge);
230  // ignore barriers
231  while (get_OpType_from_Vertex(possible_c_out) == OpType::Barrier) {
232  std::tie(possible_c_out, possible_c_out_in_edge) =
233  get_next_pair(possible_c_out, possible_c_out_in_edge);
234  }
235  if (get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
236  Bit b(get_id_from_out(possible_c_out));
237  res.insert({Qubit(it->id_), b});
238  }
239  }
240  }
241 
242  return res;
243 }

329 VertexVec Circuit::get_successors(const Vertex &vert) const {
330  EdgeVec outs = get_all_out_edges(vert);
331  VertexVec children;
332  std::unordered_set<Vertex> lookup;
333  for (const Edge &e : outs) {
334  Vertex succ = target(e);
335  if (lookup.find(succ) == lookup.end()) {
336  children.push_back(succ);
337  lookup.insert(succ);
338  }
339  }
340  return children;
341 }

420 EdgeVec Circuit::get_in_edges(const Vertex &vert) const {
421  unsigned n = n_in_edges(vert);
422  EdgeVec inedges(n);
423  std::vector<bool> port_found(n, false);
424  BGL_FORALL_INEDGES(vert, e, dag, DAG) {
425  port_t port = get_target_port(e);
426  if (port >= n) {
427  inedges.resize(port + 1);
428  port_found.resize(port + 1, false);
429  } else if (port_found[port]) {
430  throw CircuitInvalidity("Vertex has multiple inputs on the same port");
431  }
432  port_found[port] = true;
433  inedges[port] = e;
434  }
435  for (unsigned i = 0; i < n; i++)
436  if (!port_found[i]) {
437  throw CircuitInvalidity("Input ports on Vertex are non-contiguous");
438  }
439  return inedges;
440 }

453 std::vector<std::optional<Edge>> Circuit::get_linear_out_edges(
454  const Vertex &vert) const {
455  unsigned n = n_ports(vert);
456  std::vector<std::optional<Edge>> outedges(n);
457  BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
458  if (get_edgetype(e) == EdgeType::Boolean) continue;
459  port_t port = get_source_port(e);
460  if (port >= n) {
461  throw CircuitInvalidity("Vertex has an output on an unexpected port");
462  }
463  if (outedges[port]) {
464  throw CircuitInvalidity(
465  "Vertex has multiple linear outputs on the same port");
466  }
467  outedges[port] = e;
468  }
469  return outedges;
470 }

481 EdgeVec Circuit::get_all_out_edges(const Vertex &vert) const {
482  std::vector<std::optional<Edge>> lin_outs = get_linear_out_edges(vert);
483  std::vector<EdgeVec> b_bundles = get_b_out_bundles(vert);
484  EdgeVec outs;
485  for (unsigned i = 0; i < lin_outs.size(); ++i) {
486  std::optional<Edge> &l_out = lin_outs[i];
487  if (l_out) {
488  outs.push_back(*l_out);
489  outs.insert(outs.end(), b_bundles[i].begin(), b_bundles[i].end());
490  }
491  }
492  return outs;
