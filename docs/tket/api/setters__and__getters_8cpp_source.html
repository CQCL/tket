167 std::map<Bit, unsigned> Circuit::bit_readout() const {
168  std::map<Bit, unsigned> res;
169 
170  // Order bits to generate indices
171  bit_vector_t all_bs = all_bits();
172  std::sort(all_bs.begin(), all_bs.end());
173  unsigned i = 0;
174  for (const Bit &b : all_bs) {
175  res.insert({b, i});
176  i++;
177  }
178 
179  return res;
180 }

192 std::map<Qubit, Bit> Circuit::qubit_to_bit_map() const {
193  std::map<Qubit, Bit> res;
194  for (auto [it, end] = boundary.get<TagType>().equal_range(UnitType::Qubit);
195  it != end; it++) {
196  Vertex q_out = it->out_;
197  Edge last_gate_out_edge = get_nth_in_edge(q_out, 0);
198  Vertex last_gate = source(last_gate_out_edge);
199  // ignore barriers
200  while (get_OpType_from_Vertex(last_gate) == OpType::Barrier) {
201  std::tie(last_gate, last_gate_out_edge) =
202  get_prev_pair(last_gate, last_gate_out_edge);
203  }
204  if (get_OpType_from_Vertex(last_gate) == OpType::Measure) {
205  Edge possible_c_out_in_edge = get_nth_out_edge(last_gate, 1);
206  Vertex possible_c_out = target(possible_c_out_in_edge);
207  // ignore barriers
208  while (get_OpType_from_Vertex(possible_c_out) == OpType::Barrier) {
209  std::tie(possible_c_out, possible_c_out_in_edge) =
210  get_next_pair(possible_c_out, possible_c_out_in_edge);
211  }
212  if (get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
213  Bit b(get_id_from_out(possible_c_out));
214  res.insert({Qubit(it->id_), b});
215  }
216  }
217  }
218 
219  return res;
220 }

303 // given a vertex, returns a set of all its successor vertices
304 // this set can be empty, and no warnings are given if it is
305 // there are no checks to ensure the vertex exists in the graph
306 VertexVec Circuit::get_successors(const Vertex &vert) const {
307  EdgeVec outs = get_all_out_edges(vert);
308  VertexVec children;
309  std::unordered_set<Vertex> lookup;
310  for (const Edge &e : outs) {
311  Vertex succ = target(e);
312  if (lookup.find(succ) == lookup.end()) {
313  children.push_back(succ);
314  lookup.insert(succ);
315  }
316  }
317  return children;
318 }

397 EdgeVec Circuit::get_in_edges(const Vertex &vert) const {
398  unsigned n = n_in_edges(vert);
399  EdgeVec inedges(n);
400  std::vector<bool> port_found(n, false);
401  BGL_FORALL_INEDGES(vert, e, dag, DAG) {
402  port_t port = get_target_port(e);
403  if (port >= n) {
404  inedges.resize(port + 1);
405  port_found.resize(port + 1, false);
406  } else if (port_found[port]) {
407  throw CircuitInvalidity("Vertex has multiple inputs on the same port");
408  }
409  port_found[port] = true;
410  inedges[port] = e;
411  }
412  for (unsigned i = 0; i < n; i++)
413  if (!port_found[i]) {
414  throw CircuitInvalidity("Input ports on Vertex are non-contiguous");
415  }
416  return inedges;
417 }

430 std::vector<std::optional<Edge>> Circuit::get_linear_out_edges(
431  const Vertex &vert) const {
432  unsigned n = n_ports(vert);
433  std::vector<std::optional<Edge>> outedges(n);
434  BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
435  if (get_edgetype(e) == EdgeType::Boolean) continue;
436  port_t port = get_source_port(e);
437  if (port >= n) {
438  throw CircuitInvalidity("Vertex has an output on an unexpected port");
439  }
440  if (outedges[port]) {
441  throw CircuitInvalidity(
442  "Vertex has multiple linear outputs on the same port");
443  }
444  outedges[port] = e;
445  }
446  return outedges;
447 }

458 EdgeVec Circuit::get_all_out_edges(const Vertex &vert) const {
459  std::vector<std::optional<Edge>> lin_outs = get_linear_out_edges(vert);
460  std::vector<EdgeVec> b_bundles = get_b_out_bundles(vert);
461  EdgeVec outs;
462  for (unsigned i = 0; i < lin_outs.size(); ++i) {
463  std::optional<Edge> &l_out = lin_outs[i];
464  if (l_out) {
465  outs.push_back(*l_out);
466  outs.insert(outs.end(), b_bundles[i].begin(), b_bundles[i].end());
467  }
468  }
469  return outs;
470 }

472 EdgeVec Circuit::get_out_edges_of_type(
473  const Vertex &vert, EdgeType type) const {
474  if (type == EdgeType::Boolean) {
475  std::vector<EdgeVec> bundles = get_b_out_bundles(vert);
476  EdgeVec outs;
477  for (const EdgeVec &b : bundles) {
478  outs.insert(outs.end(), b.begin(), b.end());
479  }
480  return outs;
481  } else {
482  std::vector<std::optional<Edge>> outs = get_linear_out_edges(vert);
483  EdgeVec matching;
484  for (const std::optional<Edge> &e : outs) {
485  if (e && get_edgetype(*e) == type) {
486  matching.push_back(*e);
487  }
488  }
489  return matching;
490  }

704 unsigned Circuit::qubit_index(
705  const Vertex &vert, PortType port_type, port_t port) const {
706  const EdgeVec edges = (port_type == PortType::Source)
707  ? get_out_edges_of_type(vert, EdgeType::Quantum)
708  : get_in_edges_of_type(vert, EdgeType::Quantum);
709  unsigned n_edges = edges.size();
710  for (unsigned i = 0; i < n_edges; i++) {
711  const Edge &e = edges[i];
712  port_t p = (port_type == PortType::Source) ? get_source_port(e)
713  : get_target_port(e);
714  if (p == port) return i;
715  }
716  throw std::domain_error("Invalid port for vertex");
717 }
