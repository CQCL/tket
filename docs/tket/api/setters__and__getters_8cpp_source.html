55// copy assignment. Moves boundary pointers.
56Circuit &Circuit::operator=(const Circuit &other) // (1)
57{
58 dag = DAG();
59 boundary = boundary_t();
60 copy_graph(other);
61 phase = other.get_phase();
62 name = other.name;
63 add_wasm_register(other._number_of_wasm_wires);

65 return *this;

68// Move assignment
69Circuit &Circuit::operator=(Circuit &&other) {
70 if (this == &other) {
71 return *this;
72 }
73
74 boundary = std::move(other.boundary);
75 dag.m_vertices = std::move(other.dag.m_vertices);
76 dag.m_edges = std::move(other.dag.m_edges);
77 _number_of_wasm_wires = other._number_of_wasm_wires;
78 name = std::move(other.name);
79 phase = other.get_phase();
80 opgroupsigs = std::move(other.opgroupsigs);
81 return *this;
82}

210std::map<Bit, unsigned> Circuit::bit_readout() const {
211 std::map<Bit, unsigned> res;
212
213 // Order bits to generate indices
214 bit_vector_t all_bs = all_bits();
215 std::sort(all_bs.begin(), all_bs.end());
216 unsigned i = 0;
217 for (const Bit &b : all_bs) {
218 res.insert({b, i});
219 i++;
220 }
221
222 return res;
223}

225std::map<Qubit, unsigned> Circuit::qubit_readout() const {
226 std::map<Qubit, Bit> q2b_res = qubit_to_bit_map();
227 std::map<Qubit, unsigned> res;
228 std::map<Bit, unsigned> bmap = bit_readout();
229 for (auto it = q2b_res.begin(); it != q2b_res.end(); it++) {
230 res.insert({it->first, bmap.at(it->second)});
231 }
232 return res;
233}

235std::map<Qubit, Bit> Circuit::qubit_to_bit_map() const {
236 std::map<Qubit, Bit> res;
237 for (auto [it, end] = boundary.get<TagType>().equal_range(UnitType::Qubit);
238 it != end; it++) {
239 Vertex q_out = it->out_;
240 Edge last_gate_out_edge = get_nth_in_edge(q_out, 0);
241 Vertex last_gate = source(last_gate_out_edge);
242 // ignore barriers
243 while (get_OpType_from_Vertex(last_gate) == OpType::Barrier) {
244 std::tie(last_gate, last_gate_out_edge) =
245 get_prev_pair(last_gate, last_gate_out_edge);
246 }
247 if (get_OpType_from_Vertex(last_gate) == OpType::Measure) {
248 Edge possible_c_out_in_edge = get_nth_out_edge(last_gate, 1);
249 Vertex possible_c_out = target(possible_c_out_in_edge);
250 // ignore barriers
251 while (get_OpType_from_Vertex(possible_c_out) == OpType::Barrier) {
252 std::tie(possible_c_out, possible_c_out_in_edge) =
253 get_next_pair(possible_c_out, possible_c_out_in_edge);
254 }
255 if (get_OpType_from_Vertex(possible_c_out) == OpType::ClOutput) {
256 Bit b(get_id_from_out(possible_c_out));
257 res.insert({Qubit(it->id_), b});
258 }
259 }
260 }
261
262 return res;
263}

310std::optional<std::string> Circuit::get_wasm_file_uid() const {
311 std::optional<std::string> result = std::nullopt;
312 BGL_FORALL_VERTICES(v, dag, DAG) {
313 Op_ptr op = get_Op_ptr_from_Vertex(v);
314 while (op->get_type() == OpType::Conditional) {
315 op = static_cast<const Conditional &>(*op).get_op();
316 }
317 if (op->get_type() == OpType::WASM) {
318 result = static_cast<const WASMOp &>(*op).get_wasm_file_uid();
319 return result;
320 } else if (op->get_type() == OpType::CircBox) {
321 result =
322 static_cast<const CircBox &>(*op).to_circuit()->get_wasm_file_uid();
323 if (result != std::nullopt) return result;
324 }
325 }
326 return result;
327}

365// given a vertex, returns a set of all its successor vertices
366// this set can be empty, and no warnings are given if it is
367// there are no checks to ensure the vertex exists in the graph
368VertexVec Circuit::get_successors(const Vertex &vert) const {
369 EdgeVec outs = get_all_out_edges(vert);
370 VertexVec children;
371 std::unordered_set<Vertex> lookup;
372 for (const Edge &e : outs) {
373 Vertex succ = target(e);
374 if (lookup.find(succ) == lookup.end()) {
375 children.push_back(succ);
376 lookup.insert(succ);
377 }
378 }
379 return children;
380}

459EdgeVec Circuit::get_in_edges(const Vertex &vert) const {
460 unsigned n = n_in_edges(vert);
461 EdgeVec inedges(n);
462 std::vector<bool> port_found(n, false);
463 BGL_FORALL_INEDGES(vert, e, dag, DAG) {
464 port_t port = get_target_port(e);
465 if (port >= n) {
466 inedges.resize(port + 1);
467 port_found.resize(port + 1, false);
468 } else if (port_found[port]) {
469 throw CircuitInvalidity("Vertex has multiple inputs on the same port");
470 }
471 port_found[port] = true;
472 inedges[port] = e;
473 }
474 for (unsigned i = 0; i < n; i++)
475 if (!port_found[i]) {
476 throw CircuitInvalidity("Input ports on Vertex are non-contiguous");
477 }
478 return inedges;
479}

492std::vector<std::optional<Edge>> Circuit::get_linear_out_edges(
493 const Vertex &vert) const {
494 unsigned n = n_ports(vert);
495 std::vector<std::optional<Edge>> outedges(n);
496 BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
497 if (get_edgetype(e) == EdgeType::Boolean) continue;
498 port_t port = get_source_port(e);
499 if (port >= n) {
500 throw CircuitInvalidity("Vertex has an output on an unexpected port");
501 }
502 if (outedges[port]) {
503 throw CircuitInvalidity(
504 "Vertex has multiple linear outputs on the same port");
505 }
506 outedges[port] = e;
507 }
508 return outedges;
509}

520EdgeVec Circuit::get_all_out_edges(const Vertex &vert) const {
521 std::vector<std::optional<Edge>> lin_outs = get_linear_out_edges(vert);
522 std::vector<EdgeVec> b_bundles = get_b_out_bundles(vert);
523 EdgeVec outs;
524 for (unsigned i = 0; i < lin_outs.size(); ++i) {
525 std::optional<Edge> &l_out = lin_outs[i];
526 if (l_out) {
527 outs.push_back(*l_out);
528 outs.insert(outs.end(), b_bundles[i].begin(), b_bundles[i].end());
529 }
530 }
531 return outs;
532}

534EdgeVec Circuit::get_out_edges_of_type(
535 const Vertex &vert, EdgeType type) const {
536 if (type == EdgeType::Boolean) {
537 std::vector<EdgeVec> bundles = get_b_out_bundles(vert);
538 EdgeVec outs;
539 for (const EdgeVec &b : bundles) {
540 outs.insert(outs.end(), b.begin(), b.end());
541 }
542 return outs;
543 } else {
544 std::vector<std::optional<Edge>> outs = get_linear_out_edges(vert);
545 EdgeVec matching;
546 for (const std::optional<Edge> &e : outs) {
547 if (e && get_edgetype(*e) == type) {
548 matching.push_back(*e);
549 }
550 }
551 return matching;
552 }
553}

555std::vector<EdgeVec> Circuit::get_b_out_bundles(const Vertex &vert) const {
556 unsigned n = n_ports(vert);
557 std::vector<EdgeVec> bundles(n);
558 BGL_FORALL_OUTEDGES(vert, e, dag, DAG) {
559 if (get_edgetype(e) == EdgeType::Boolean) {
560 port_t port = get_source_port(e);
561 if (port > n) {
562 throw CircuitInvalidity("Vertex has an output on an unexpected port");
563 }
564 bundles.at(port).push_back(e);
565 }
566 }
567 return bundles;
568}

700// checks that the edge belongs to the vertex
701Edge Circuit::get_next_edge(const Vertex &vert, const Edge &in_edge) const {
702 if (target(in_edge) != vert)
703 throw CircuitInvalidity(
704 "Cannot get next edge: Edge is not an in edge to Vertex");
705 port_t order = get_target_port(in_edge);
706 return get_nth_out_edge(vert, order);
707}

709// there are no checks to ensure the vertex exists in the graph
710// checks that the edge belongs to the vertex
711Edge Circuit::get_last_edge(const Vertex &vert, const Edge &out_edge) const {
712 if (source(out_edge) != vert)
713 throw CircuitInvalidity(
714 "Cannot get last edge: Edge is not an out edge from Vertex");
715 port_t order = get_source_port(out_edge);
716 return get_nth_in_edge(vert, order);
717}

719// // given a vertex and corresponding in edge, returns next vertex
720// and edge
721// there are no checks to ensure the vertex exists in the graph
722std::pair<Vertex, Edge> Circuit::get_next_pair(
723 const Vertex &current_vertex, const Edge &inedge) const {
724 Edge new_edge = get_next_edge(current_vertex, inedge);
725 Vertex new_vert = target(new_edge);
726 if (new_vert == current_vertex) {
727 throw CircuitInvalidity("A qubit path is looping");
728 }
729 return {new_vert, new_edge};
730}

732// given a vertex and corresponding out edge, returns previous vertex
733// and edge pair
734// there are no checks to ensure the vertex exists in the graph
735std::pair<Vertex, Edge> Circuit::get_prev_pair(
736 const Vertex &current_vertex, const Edge &outedge) const {
737 Edge last_edge = get_last_edge(current_vertex, outedge);
738 Vertex last_vertex = source(last_edge);
739 if (last_vertex == current_vertex) {
740 throw CircuitInvalidity("A qubit path is looping");
741 }
742
743 return {last_vertex, last_edge};
744}
