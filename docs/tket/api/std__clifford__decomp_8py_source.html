18 def phase_diff(A, B):
19  """a s.t. e^{i pi a} A = B"""
20  U = B @ A.conjugate().transpose()
21  u00 = U[0, 0]
22  if not allclose(U, u00 * eye(len(A), dtype=complex)):
23  return None
24  return angle(u00) / pi

43 def std_cliff():
44  # Return dict d where d[(i,j,k)] gives the decomposition of Rz(i/2)Rx(j/2)Rz(k/2)
45  # as a sequence of operations Z? X? S? V? S? followed by a phase
46  etable = {}
47  for t0 in range(2):
48  for t1 in range(2):
49  for t2 in range(2):
50  for t3 in range(2):
51  for t4 in range(2):
52  m = I
53  if t0:
54  m = Z @ m
55  if t1:
56  m = X @ m
57  if t2:
58  m = S @ m
59  if t3:
60  m = V @ m
61  if t4:
62  m = S @ m
63  etable[(t0, t1, t2, t3, t4)] = m
64  d = {}
65  for i in range(4):
66  for j in range(4):
67  for k in range(4):
68  u = Rz(i / 2) @ Rx(j / 2) @ Rz(k / 2)
69  matches = []
70  for ts, m in etable.items():
71  p = phase_diff(m, u)
72  if p is not None:
73  matches.append((ts, p.round(3)))
74  if not matches:
75  raise RuntimeError(f"match not found for {i}, {j}, {k}")
76  # Return the match with fewest gates
77  best_i = min(
78  ((i, sum(x[0])) for i, x in enumerate(matches)), key=lambda y: y[1]
79  )[0]
80  d[(i, j, k)] = matches[best_i]
81  for i in range(4):
82  print("{")
83  for j in range(4):
84  print(" {")
85  for k in range(4):
86  ts, p = d[(i, j, k)]
87  print(" {" + ", ".join(f"{t}" for t in ts) + f", {p}" + "},")
88  print(" },")
89  print("},")
90  return d
