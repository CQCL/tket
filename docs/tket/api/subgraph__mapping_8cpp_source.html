31 std::vector<qubit_bimap_t> monomorphism_edge_break(
32  const Architecture& arc, const QubitGraph& q_graph, unsigned max_matches,
33  unsigned timeout) {
34  if (q_graph.n_nodes() > arc.n_nodes()) {
35  throw ArchitectureInvalidity(
36  "Interaction graph too large for architecture");
37  }
38 
39  Architecture::UndirectedConnGraph undirected_target =
40  arc.get_undirected_connectivity();
41  QubitGraph::UndirectedConnGraph undirected_pattern =
42  q_graph.get_undirected_connectivity();
43 
44  std::chrono::time_point<std::chrono::steady_clock> end_time =
45  std::chrono::steady_clock::now() + std::chrono::milliseconds(timeout);
46 
47  while (true) {
48  long search_timeout = std::chrono::duration_cast<std::chrono::milliseconds>(
49  (end_time - std::chrono::steady_clock::now()) / 2)
50  .count();
51  if (search_timeout <= 0) search_timeout = 1;
52 
53  std::vector<qubit_bimap_t> all_maps = get_unweighted_subgraph_monomorphisms(
54  undirected_pattern, undirected_target, max_matches, timeout);
55  std::sort(all_maps.begin(), all_maps.end());
56 
57  if (std::chrono::steady_clock::now() >= end_time) {
58  std::stringstream ss;
59  ss << "subgraph monomorphism reached " << timeout
60  << " millisecond timeout before reaching set max matches "
61  << max_matches << ", instead finding " << all_maps.size()
62  << " matches. "
63  "Please change PlacementConfig.timeout to allow more matches.";
64  tket_log()->warn(ss.str());
65  if (all_maps.empty()) {
66  throw std::runtime_error("No mappings found before timeout.");
67  }
68  return all_maps;
69  }
70  if (!all_maps.empty()) {
71  return all_maps;
72  }
73  const unsigned current_number_of_edges =
74  boost::num_edges(undirected_pattern);
75  // It MUST have found a solution, if no pattern edges!
76  TKET_ASSERT(current_number_of_edges > 0);
77  using edge_t = graphs::utils::edge<QubitGraph::UndirectedConnGraph>;
78  auto e_its = boost::edges(undirected_pattern);
79  auto max_e_it = boost::first_max_element(
80  e_its.first, e_its.second,
81  [&undirected_pattern](const edge_t& lhs, const edge_t& rhs) {
82  return undirected_pattern[lhs].weight <
83  undirected_pattern[rhs].weight;
84  });
85  graphs::utils::remove_edge(*max_e_it, undirected_pattern, true);
86  TKET_ASSERT(boost::num_edges(undirected_pattern) < current_number_of_edges);
87  }
88 }
