958 /**
959 * Applies the PauliTensor to a given statevector by matrix multiplication.
960 *
961 * Determines the number of qubits from the size of the statevector, and
962 * assumes default register qubits in ILO-BE format.
963 */
964 Eigen::VectorXcd dot_state(const Eigen::VectorXcd &state) const {
965 // allowing room for big states
966 unsigned long long n = state.size();
967 if (!(n && (!(n & (n - 1)))))
968 throw std::logic_error("Statevector size is not a power of two.");
969 unsigned n_qubits = 0;
970 while (n) {
971 n >>= 1;
972 ++n_qubits;
973 }
974 --n_qubits;
975 return (to_sparse_matrix(n_qubits) * state);
976 }

978 * Applies the PauliTensor to a given statevector by matrix multiplication.
979 *
980 * \p qubits dictates the order of Qubits in the state, assuming a Big Endian
981 * format. An exception is thrown if the size of the state does not match up
982 * with the number of Qubits given.
983 */
984 Eigen::VectorXcd dot_state(
985 const Eigen::VectorXcd &state, const qubit_vector_t &qubits) const {
986 if (state.size() != 1 << qubits.size())
987 throw std::logic_error(
988 "Size of statevector does not match number of qubits passed to "
989 "dot_state");
990 return (to_sparse_matrix(qubits) * state);
991 }
